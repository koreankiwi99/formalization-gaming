<lean>
axiom Person : Type
axiom Taylor : Person

-- Predicates
axiom EatsSalads : Person → Prop
axiom VeryConscious : Person → Prop
axiom GrewUpHealthConscious : Person → Prop
axiom FulfillNutrition : Person → Prop
axiom DisregardWellBeing : Person → Prop
axiom VisitsGymDaily : Person → Prop

-- Axioms
axiom A1 : ∀ p, EatsSalads p → VeryConscious p
axiom A2 : ∀ p, GrewUpHealthConscious p → EatsSalads p
axiom A3 : ∀ p, FulfillNutrition p → GrewUpHealthConscious p
axiom A4 : ∀ p, DisregardWellBeing p → ¬ VeryConscious p
axiom A5 : ∀ p, VisitsGymDaily p → FulfillNutrition p

-- Taylor's exclusive disjunction
axiom Taylor_disjunction : 
  (GrewUpHealthConscious Taylor ∧ DisregardWellBeing Taylor) ∨ 
  (¬ GrewUpHealthConscious Taylor ∧ ¬ DisregardWellBeing Taylor)

-- Proof that Taylor cannot visit gym daily
theorem taylor_not_gym : ¬ VisitsGymDaily Taylor := 
  fun h : VisitsGymDaily Taylor =>
  have F1 : FulfillNutrition Taylor := A5 Taylor h
  have F2 : GrewUpHealthConscious Taylor := A3 Taylor F1
  have F3 : EatsSalads Taylor := A2 Taylor F2
  have F4 : VeryConscious Taylor := A1 Taylor F3
  show False from 
    Or.elim Taylor_disjunction
      (fun case1 : GrewUpHealthConscious Taylor ∧ DisregardWellBeing Taylor =>
        have dw : DisregardWellBeing Taylor := case1.right
        show False from A4 Taylor dw F4)
      (fun case2 : ¬ GrewUpHealthConscious Taylor ∧ ¬ DisregardWellBeing Taylor =>
        show False from case2.left F2)
</lean>

ANSWER: False
