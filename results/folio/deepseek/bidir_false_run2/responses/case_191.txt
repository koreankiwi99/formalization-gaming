axiom obj : Type
axiom Yuri : obj
axiom P : obj → Prop  -- Professional basketball player
axiom A : obj → Prop  -- American national
axiom J : obj → Prop  -- Can do jump shots
axiom L : obj → Prop  -- Leaps straight into air
axiom M : obj → Prop  -- Activates leg muscles

-- Context axioms
axiom T1 : ∃ x, P x ∧ ¬A x
axiom T2 : ∀ x, P x → J x
axiom T3 : ∀ x, J x → L x
axiom T4 : ∀ x, L x → M x
axiom T5 : ¬M Yuri

-- Prove Yuri is not a professional basketball player
theorem not_P_Yuri : ¬P Yuri := by
  intro hP
  have hJ : J Yuri := T2 Yuri hP
  have hL : L Yuri := T3 Yuri hJ
  have hM : M Yuri := T4 Yuri hL
  exact T5 hM

-- Statement: (¬ (P Yuri ∧ A Yuri)) → P Yuri
-- Prove this statement is false by showing premise true and conclusion false
theorem false_statement : ¬((¬ (P Yuri ∧ A Yuri)) → P Yuri) := by
  intro h
  have h_premise : ¬(P Yuri ∧ A Yuri) := fun ⟨hP, _⟩ => not_P_Yuri hP
  have h_conc : P Yuri := h h_premise
  exact not_P_Yuri h_conc
