<lean>
axiom obj : Type
axiom Ted : obj
axiom BrownSwissCattle : obj → Prop
axiom Cow : obj → Prop
axiom Pet : obj → Prop
axiom DomesticatedAnimal : obj → Prop
axiom Alligator : obj → Prop

axiom all_brown_are_cows : ∀ x : obj, BrownSwissCattle x → Cow x
axiom some_pets_are_brown : ∃ x : obj, Pet x ∧ BrownSwissCattle x
axiom all_cows_domesticated : ∀ x : obj, Cow x → DomesticatedAnimal x
axiom alligators_not_domesticated : ∀ x : obj, Alligator x → ¬DomesticatedAnimal x
axiom ted_alligator : Alligator Ted

theorem ted_implication : BrownSwissCattle Ted → ¬Pet Ted :=
  fun h : BrownSwissCattle Ted =>
    have ted_cow : Cow Ted := all_brown_are_cows Ted h
    have ted_domesticated : DomesticatedAnimal Ted := all_cows_domesticated Ted ted_cow
    have ted_not_domesticated : ¬DomesticatedAnimal Ted := alligators_not_domesticated Ted ted_alligator
    False.elim (ted_not_domesticated ted_domesticated)
</lean>

ANSWER: True
