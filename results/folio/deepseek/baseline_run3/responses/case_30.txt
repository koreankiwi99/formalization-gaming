-- Everyone at the mixer is a Grand Slam champion or an Oscar-nominated actor.
axiom A1 : ∀ p : Person, AtMixer p → GrandSlamChampion p ∨ OscarNominatedActor p

-- Every Grand Slam champion at the mixer is a professional tennis player.
axiom A2 : ∀ p : Person, AtMixer p → GrandSlamChampion p → ProfessionalTennisPlayer p

-- All Oscar-nominated actors at the mixer are celebrities.
axiom A3 : ∀ p : Person, AtMixer p → OscarNominatedActor p → Celebrity p

-- All professional tennis players at the mixer are athletes.
axiom A4 : ∀ p : Person, AtMixer p → ProfessionalTennisPlayer p → Athlete p

-- If a person at the mixer is a celebrity, then they are well paid.
axiom A5 : ∀ p : Person, AtMixer p → Celebrity p → WellPaid p

-- If a person at the mixer is an athlete, then they are famous.
axiom A6 : ∀ p : Person, AtMixer p → Athlete p → Famous p

-- All well-paid people at the mixer live in tax havens.
axiom A7 : ∀ p : Person, AtMixer p → WellPaid p → LivesInTaxHaven p

-- Djokovic is at the mixer
axiom A8 : AtMixer djokovic

-- If Djokovic is a famous athlete, then Djokovic is well-paid.
axiom A9 : Famous djokovic ∧ Athlete djokovic → WellPaid djokovic

-- We cannot prove GrandSlamChampion djokovic nor its negation
theorem djokovic_gsc : GrandSlamChampion djokovic := by
  have h := A1 djokovic A8
  cases h with
  | inl gsc => exact gsc
  | inr ona => sorry  -- Cannot prove GSC from the actor branch

theorem not_djokovic_gsc : ¬ GrandSlamChampion djokovic := by
  intro gsc
  have tennis := A2 djokovic A8 gsc
  have athlete := A4 djokovic A8 tennis
  have famous := A6 djokovic A8 athlete
  have well_paid := A9 ⟨famous, athlete⟩
  -- No contradiction reached; Djokovic could be an actor instead
  have h := A1 djokovic A8
  cases h with
  | inl gsc' => exact gsc'  -- Contradiction? No, we assumed ¬GSC but this branch is GSC -> contradiction
  | inr ona => 
      let celeb := A3 djokovic A8 ona
      let wp := A5 djokovic A8 celeb
      -- Both branches lead to WellPaid, but no conflict with ¬GSC
      sorry  -- Cannot derive contradiction
