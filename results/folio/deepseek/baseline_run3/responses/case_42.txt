<lean>
axiom obj : Type
axiom CocoGauff : obj
axiom HighlyRankedByWTA : obj → Prop
axiom MostActive : obj → Prop
axiom LostToIgaAtRG2022 : obj → Prop
axiom FemaleTennisPlayerAtRG2022 : obj → Prop
axiom MaleTennisPlayerAtRG2022 : obj → Prop
axiom TennisPlayerAtRG2022 : obj → Prop
axiom LostToRafaelNadal : obj → Prop

-- If someone is ranked highly by WTA, they are most active
axiom R1 : ∀ x, HighlyRankedByWTA x → MostActive x
-- Everyone who lost to Iga Świątek at RG2022 is highly ranked by WTA
axiom R2 : ∀ x, LostToIgaAtRG2022 x → HighlyRankedByWTA x
-- All female tennis players at RG2022 lost to Iga Świątek
axiom R3 : ∀ x, FemaleTennisPlayerAtRG2022 x → LostToIgaAtRG2022 x
-- All tennis players at RG2022 are female or male
axiom R4 : ∀ x, TennisPlayerAtRG2022 x → FemaleTennisPlayerAtRG2022 x ∨ MaleTennisPlayerAtRG2022 x
-- All male tennis players at RG2022 lost to Rafael Nadal
axiom R5 : ∀ x, MaleTennisPlayerAtRG2022 x → LostToRafaelNadal x
-- If Coco Gauff is highly ranked or lost to Nadal, she is not male at RG2022
axiom R6 : (HighlyRankedByWTA CocoGauff ∨ LostToRafaelNadal CocoGauff) → ¬ MaleTennisPlayerAtRG2022 CocoGauff
-- Coco Gauff is at Roland Garros 2022
axiom T1 : TennisPlayerAtRG2022 CocoGauff

theorem coco_most_active : MostActive CocoGauff := by
  -- CocoGauff is either female or male at RG2022
  have h_gender := R4 CocoGauff T1
  cases h_gender with
  | inl female =>
    -- If female, lost to Iga (R3)
    have lost_to_iga := R3 CocoGauff female
    -- Therefore highly ranked by WTA (R2)
    have highly_ranked := R2 CocoGauff lost_to_iga
    -- Therefore most active (R1)
    exact R1 CocoGauff highly_ranked
  | inr male =>
    -- If male, lost to Nadal (R5)
    have lost_to_nadal := R5 CocoGauff male
    -- Therefore by R6: not male (since LostToRafaelNadal is true)
    have not_male := R6 (Or.inr lost_to_nadal)
    -- Contradiction: male and not male
    contradiction
</lean>

ANSWER: True
