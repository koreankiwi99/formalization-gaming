Based on the reasoning, the statement about Peter is true. The Lean proof demonstrates that the first case (Peter spills food and is cleanly) leads to a contradiction, forcing the second case (Peter does not spill food and is not cleanly). From this, it follows that Peter is not a clumsy foodie and his family does not prioritize order, satisfying the "neither" part of the statement.

<lean>
axiom obj : Type
axiom Peter : obj
axiom Spills : obj → Prop
axiom NotablyTidy : obj → Prop
axiom Cleanly : obj → Prop
axiom ValuesOrder : obj → Prop
axiom FamilyPrioritizes : obj → Prop
axiom ClumsyFoodie : obj → Prop

-- Axioms from context
axiom A1 : ∀ x, Spills x → ¬ NotablyTidy x
axiom A2 : ∀ x, ClumsyFoodie x → Spills x
axiom A3 : ∀ x, Cleanly x → NotablyTidy x
axiom A4 : ∀ x, ValuesOrder x → Cleanly x
axiom A5 : ∀ x, FamilyPrioritizes x → ValuesOrder x
axiom Peter_case : (Spills Peter ∧ Cleanly Peter) ∨ (¬ Spills Peter ∧ ¬ Cleanly Peter)

theorem peter_property : 
    (ClumsyFoodie Peter ∧ FamilyPrioritizes Peter) ∨ 
    (¬ ClumsyFoodie Peter ∧ ¬ FamilyPrioritizes Peter) := by
  -- Case 1: Spills and Cleanly leads to contradiction
  have case1_contra : ¬ (Spills Peter ∧ Cleanly Peter) := by
    intro ⟨spill, clean⟩
    have not_tidy : ¬ NotablyTidy Peter := A1 Peter spill
    have tidy : NotablyTidy Peter := A3 Peter clean
    contradiction
  -- Therefore, Peter must be in the second case
  have case2 : ¬ Spills Peter ∧ ¬ Cleanly Peter := by
    cases Peter_case with
    | inl c1 => contradiction
    | inr c2 => exact c2
  -- Unpack case2
  let ⟨no_spill, not_clean⟩ := case2
  -- Prove ¬ClumsyFoodie Peter
  have not_clumsy : ¬ ClumsyFoodie Peter := by
    intro clumsy
    have spill : Spills Peter := A2 Peter clumsy
    contradiction
  -- Prove ¬FamilyPrioritizes Peter
  have not_family : ¬ FamilyPrioritizes Peter := by
    intro family
    have values : ValuesOrder Peter := A5 Peter family
    have clean : Cleanly Peter := A4 Peter values
    contradiction
  -- Therefore, the second disjunct holds
  exact Or.inr ⟨not_clumsy, not_family⟩
</lean>

ANSWER: True
