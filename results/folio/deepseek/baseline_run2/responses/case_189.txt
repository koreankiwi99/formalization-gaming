<lean>
axiom obj : Type
axiom yuri : obj
axiom BasketballPlayer : obj → Prop
axiom American : obj → Prop
axiom CanJumpShot : obj → Prop
axiom Leap : obj → Prop
axiom ActivateLegMuscles : obj → Prop

-- Some professional basketball players are not American nationals
axiom some_non_american : ∃ x, BasketballPlayer x ∧ ¬ American x

-- All professional basketball players can do jump shots
axiom basketball_jump : ∀ x, BasketballPlayer x → CanJumpShot x

-- If someone can do jump shots, they leap straight into the air
axiom jump_to_leap : ∀ x, CanJumpShot x → Leap x

-- If someone leaps straight into the air, they activate their leg muscles
axiom leap_to_activate : ∀ x, Leap x → ActivateLegMuscles x

-- Yuri does not activate his leg muscles
axiom yuri_inactive : ¬ ActivateLegMuscles yuri

-- We want to consider the statement: American yuri ∧ BasketballPlayer yuri

-- First, prove that no basketball player can avoid activating leg muscles
theorem basketball_activate : ∀ x, BasketballPlayer x → ActivateLegMuscles x := by
  intro x hB
  apply leap_to_activate
  apply jump_to_leap
  exact basketball_jump x hB

-- Now prove that Yuri is not a basketball player
theorem yuri_not_basketball : ¬ BasketballPlayer yuri := by
  intro h
  have := basketball_activate yuri h
  exact yuri_inactive this

-- The question is about Yuri being an American professional basketball player
theorem yuri_not_american_basketball : ¬ (American yuri ∧ BasketballPlayer yuri) := by
  intro h
  exact yuri_not_basketball h.right

-- Since we proved the negation of the statement, the original statement is False
</lean>
