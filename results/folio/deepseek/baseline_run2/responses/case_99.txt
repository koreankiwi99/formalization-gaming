<lean>
axiom Animal : Type
axiom Invertebrate : Animal → Prop
axiom Vertebrate : Animal → Prop
axiom Backbone : Animal → Prop
axiom ReproduceMF : Animal → Prop
axiom Bee : Animal → Prop
axiom QueenBee : Animal → Prop
axiom Harry : Animal

axiom A1 : ∀ a : Animal, Invertebrate a ∨ Vertebrate a
axiom A2 : ∀ a : Animal, Backbone a → ReproduceMF a
axiom A3 : ∀ a : Animal, Vertebrate a → Backbone a
axiom A4 : ∀ a : Animal, Bee a → ¬ ReproduceMF a
axiom A5 : ∀ a : Animal, QueenBee a → Bee a
axiom A6 : Bee Harry

theorem harry_invertebrate_or_queen : Invertebrate Harry ∨ QueenBee Harry := by
  have noMF : ¬ ReproduceMF Harry := A4 Harry A6
  have noBackbone : ¬ Backbone Harry := fun h => noMF (A2 Harry h)
  have noVertebrate : ¬ Vertebrate Harry := fun h => noBackbone (A3 Harry h)
  have disj := A1 Harry
  cases disj with
  | inl inv => exact Or.inl inv
  | inr vert => contradiction
</lean>

ANSWER: True
