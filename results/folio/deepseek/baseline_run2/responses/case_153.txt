axiom obj : Type
axiom Mixture : obj
axiom Hydrocarbon : obj → Prop
axiom OrganicCompound : obj → Prop
axiom Alkane : obj → Prop
axiom ChemicalCompound : obj → Prop
axiom ContainsCarbon : obj → Prop
axiom ContainsOnlyOneElement : obj → Prop

-- Axioms from the context
axiom A1 : ∀ x, Hydrocarbon x → OrganicCompound x
axiom A2 : ∀ x, Alkane x → Hydrocarbon x
axiom A3 : ∀ x, OrganicCompound x → ChemicalCompound x
axiom A4 : ∀ x, OrganicCompound x → ContainsCarbon x
axiom A5 : ∀ x, ChemicalCompound x → ¬ ContainsOnlyOneElement x
axiom A6 : (ChemicalCompound Mixture ∧ ContainsOnlyOneElement Mixture) ∨ (¬ ChemicalCompound Mixture ∧ ¬ ContainsOnlyOneElement Mixture)

-- Prove negation: ¬ (Alkane Mixture ∧ ContainsCarbon Mixture)
theorem mixture_not_alkane_with_carbon : ¬ (Alkane Mixture ∧ ContainsCarbon Mixture) :=
  fun h => 
  let ⟨alkane_mix, _⟩ := h
  have H1 : Hydrocarbon Mixture := A2 Mixture alkane_mix
  have H2 : OrganicCompound Mixture := A1 Mixture H1
  have H3 : ChemicalCompound Mixture := A3 Mixture H2
  have H4 : ¬ ContainsOnlyOneElement Mixture := A5 Mixture H3
  have H5 : (ChemicalCompound Mixture ∧ ContainsOnlyOneElement Mixture) ∨ (¬ ChemicalCompound Mixture ∧ ¬ ContainsOnlyOneElement Mixture) := A6
  Or.elim H5
    (fun c1 => False.elim (H4 c1.right))
    (fun c2 => False.elim (c2.left H3))
