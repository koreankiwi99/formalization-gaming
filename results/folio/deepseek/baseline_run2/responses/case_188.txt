The reasoning chain shows that if Yuri were a professional basketball player, he would activate his leg muscles (via jump shots and leaping), but since he does not activate them, he cannot be a basketball player. Thus, he cannot be an American professional basketball player either.

<lean>
axiom obj : Type
axiom Yuri : obj
axiom BballPlayer : obj → Prop
axiom American : obj → Prop
axiom JumpShots : obj → Prop
axiom Leap : obj → Prop
axiom Activate : obj → Prop

axiom some_not_american : ∃ x, BballPlayer x ∧ ¬ American x
axiom all_bball_jump : ∀ x, BballPlayer x → JumpShots x
axiom jump_to_leap : ∀ x, JumpShots x → Leap x
axiom leap_to_activate : ∀ x, Leap x → Activate x
axiom yuri_not_activate : ¬ Activate Yuri

theorem yuri_not_bball : ¬ BballPlayer Yuri := by
  intro hB
  have hJ : JumpShots Yuri := all_bball_jump Yuri hB
  have hL : Leap Yuri := jump_to_leap Yuri hJ
  have hA : Activate Yuri := leap_to_activate Yuri hL
  exact yuri_not_activate hA

theorem result : ¬ (BballPlayer Yuri ∧ American Yuri) := by
  intro ⟨hB, _⟩
  exact yuri_not_bball hB
