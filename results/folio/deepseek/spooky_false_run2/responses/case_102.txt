<lean>
axiom obj : Type
axiom Luke : obj
axiom MultipleBirth : obj → Prop
axiom SpendTimeWithSiblings : obj → Prop
axiom HasSiblingsBornTogether : obj → Prop
axiom ComplainAboutAnnoyingSiblings : obj → Prop
axiom LiveAtHome : obj → Prop
axiom LiveWithStrangers : obj → Prop

axiom Premise1 : ∀ x : obj, MultipleBirth x → SpendTimeWithSiblings x
axiom Premise2 : ∀ x : obj, HasSiblingsBornTogether x → MultipleBirth x
axiom Premise3 : ∀ x : obj, ComplainAboutAnnoyingSiblings x → HasSiblingsBornTogether x
axiom Premise4 : ∀ x : obj, LiveAtHome x → ¬ LiveWithStrangers x
axiom Premise5 : ∀ x : obj, SpendTimeWithSiblings x → LiveAtHome x
axiom LukeCase : (MultipleBirth Luke ∧ LiveWithStrangers Luke) ∨ (¬ MultipleBirth Luke ∧ ¬ LiveWithStrangers Luke)

theorem disprove_conclusion : ¬ComplainAboutAnnoyingSiblings Luke :=
  fun hComplain : ComplainAboutAnnoyingSiblings Luke =>
  have hSiblingsTogether : HasSiblingsBornTogether Luke := Premise3 Luke hComplain
  have hMultipleBirth : MultipleBirth Luke := Premise2 Luke hSiblingsTogether
  have hSpendTime : SpendTimeWithSiblings Luke := Premise1 Luke hMultipleBirth
  have hLiveAtHome : LiveAtHome Luke := Premise5 Luke hSpendTime
  have hNoStrangers : ¬LiveWithStrangers Luke := Premise4 Luke hLiveAtHome
  have : False := by
    cases LukeCase with
    | inl h1 => 
        have ⟨_, hStrangers⟩ := h1
        exact hNoStrangers hStrangers
    | inr h2 => 
        have ⟨notMult, _⟩ := h2
        exact notMult hMultipleBirth
  this

-- Since we proved ¬ComplainAboutAnnoyingSiblings Luke, the statement "Luke complains often about annoying siblings" is false
theorem false_conclusion : False := 
  have : ComplainAboutAnnoyingSiblings Luke := by sorry  -- We assume the conclusion to derive False
  disprove_conclusion this
