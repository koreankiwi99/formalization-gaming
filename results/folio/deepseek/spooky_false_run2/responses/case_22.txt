<lean>
axiom obj : Type
axiom StrangerThings : obj
axiom BlackMirror : obj
axiom Popular : obj → Prop
axiom BingeWatches : obj → Prop
axiom Downloads : obj → Prop
axiom SharesWithLisa : obj → Prop
axiom NetflixShow : obj → Prop

-- Premises
axiom P1 : NetflixShow StrangerThings
axiom P2 : Popular StrangerThings
axiom P3 : ∀ x : obj, NetflixShow x ∧ Popular x → BingeWatches x
axiom P4 : ∀ x : obj, NetflixShow x → (BingeWatches x ↔ Downloads x)
axiom P5 : ¬ Downloads BlackMirror
axiom P6 : NetflixShow BlackMirror
axiom P7 : ∀ x : obj, NetflixShow x → (BingeWatches x → SharesWithLisa x)

-- Prove that the statement "Black Mirror is popular" is false
theorem black_mirror_not_popular : ¬ Popular BlackMirror := by
  intro h
  -- Since BlackMirror is a Netflix show (P6) and assumed popular, by P3 it should be binge-watched
  have H_binge : BingeWatches BlackMirror := P3 BlackMirror ⟨P6, h⟩
  -- Using the biconditional from P4
  have H_bw_download : BingeWatches BlackMirror ↔ Downloads BlackMirror := P4 BlackMirror P6
  rw [H_bw_download] at H_binge
  exact P5 H_binge

-- Therefore, the statement "Black Mirror is popular" is false
</lean>

ANSWER: False
