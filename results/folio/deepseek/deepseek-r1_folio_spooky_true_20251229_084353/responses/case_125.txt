<lean>
axiom obj : Type
axiom Rock : obj
axiom PeterPet : obj → Prop
axiom Mammal : obj → Prop
axiom Animal : obj → Prop
axiom Monkey : obj → Prop
axiom Bird : obj → Prop
axiom Fly : obj → Prop
axiom Breathe : obj → Prop
axiom Wings : obj → Prop

-- Premises:
axiom T1 : ∀ x : obj, PeterPet x → Mammal x → Animal x
axiom T2 : ∀ x : obj, Monkey x → Mammal x
axiom T3 : ∀ x : obj, PeterPet x → Monkey x ∨ Bird x
axiom T4 : ∀ x : obj, PeterPet x → Bird x → Fly x
axiom T5 : ∀ x : obj, Animal x → PeterPet x → Breathe x
axiom T6 : ∀ x : obj, PeterPet x → Fly x → Wings x
axiom T7 : PeterPet Rock
axiom T8 : Fly Rock ∨ Bird Rock ∨ ¬ Breathe Rock

theorem rock_wings : Wings Rock := by
  -- Use disjunction from T8
  cases T8 with
  | inl fly_rock => 
    -- Case 1: Rock can fly
    exact T6 Rock T7 fly_rock
  | inr rest => 
    cases rest with
    | inl bird_rock => 
      -- Case 2: Rock is a bird
      have fly_rock := T4 Rock T7 bird_rock
      exact T6 Rock T7 fly_rock
    | inr not_breathe => 
      -- Case 3: Rock cannot breathe
      -- Show Rock must be a bird (not monkey)
      have pet_type := T3 Rock T7
      cases pet_type with
      | inl monkey_rock => 
        -- If Rock is monkey, derive contradiction
        have mammal_rock := T2 Rock monkey_rock
        have animal_rock := T1 Rock T7 mammal_rock
        have breathe_rock := T5 Rock animal_rock T7
        contradiction
      | inr bird_rock => 
        -- Therefore Rock is a bird
        have fly_rock := T4 Rock T7 bird_rock
        exact T6 Rock T7 fly_rock
</lean>

ANSWER: True
