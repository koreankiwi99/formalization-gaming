axiom P : Prop                             -- George has access to a program
axiom W : Prop                             -- George wants to compose
axiom C : Prop                             -- George can compose
axiom Compose : Prop                       -- George will compose

axiom p1 : L → W                           -- If George likes music, he wants to compose  
axiom p2 : P → C                           -- If George has access to a program, he can compose
axiom p3 : C → W → Compose                 -- If George can compose and wants to compose, he will compose

theorem conclusion : L → P → Compose :=    -- If George likes music and has access to a program...
  fun hL hP => p3 (p2 hP) (p1 hL)          -- Proof by chaining premises: P → C → W → Compose
