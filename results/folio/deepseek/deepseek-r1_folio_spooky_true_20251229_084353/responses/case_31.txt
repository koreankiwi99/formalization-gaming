<lean>
axiom obj : Type
axiom Djokovic : obj
axiom AtMixer : obj → Prop
axiom GrandSlamChampion : obj → Prop
axiom OscarNominatedActor : obj → Prop
axiom ProfessionalTennisPlayer : obj → Prop
axiom Celebrity : obj → Prop
axiom Athlete : obj → Prop
axiom WellPaid : obj → Prop
axiom Famous : obj → Prop
axiom LivesInTaxHaven : obj → Prop

-- Premises:
axiom P1 : ∀ x, AtMixer x → GrandSlamChampion x ∨ OscarNominatedActor x
axiom P2 : ∀ x, AtMixer x → GrandSlamChampion x → ProfessionalTennisPlayer x
axiom P3 : ∀ x, AtMixer x → OscarNominatedActor x → Celebrity x
axiom P4 : ∀ x, AtMixer x → ProfessionalTennisPlayer x → Athlete x
axiom P5 : ∀ x, AtMixer x → Celebrity x → WellPaid x
axiom P6 : ∀ x, AtMixer x → Athlete x → Famous x
axiom P7 : ∀ x, AtMixer x → WellPaid x → LivesInTaxHaven x
axiom P8 : AtMixer Djokovic
axiom P9 : (Famous Djokovic ∧ Athlete Djokovic) → WellPaid Djokovic

theorem djokovic_tax_haven : LivesInTaxHaven Djokovic := by
  -- Djokovic is at the mixer (P8), so by P1 he is either GrandSlamChampion or OscarNominatedActor
  have h_cases : GrandSlamChampion Djokovic ∨ OscarNominatedActor Djokovic := P1 Djokovic P8
  cases h_cases with
  | inl gsc => 
      -- Case 1: Djokovic is a Grand Slam champion
      have tennis : ProfessionalTennisPlayer Djokovic := P2 Djokovic P8 gsc
      have athlete : Athlete Djokovic := P4 Djokovic P8 tennis
      have famous : Famous Djokovic := P6 Djokovic P8 athlete
      have well_paid : WellPaid Djokovic := P9 ⟨famous, athlete⟩
      exact P7 Djokovic P8 well_paid
  | inr ona => 
      -- Case 2: Djokovic is an Oscar-nominated actor
      have celebrity : Celebrity Djokovic := P3 Djokovic P8 ona
      have well_paid : WellPaid Djokovic := P5 Djokovic P8 celebrity
      exact P7 Djokovic P8 well_paid
</lean>

ANSWER: True
