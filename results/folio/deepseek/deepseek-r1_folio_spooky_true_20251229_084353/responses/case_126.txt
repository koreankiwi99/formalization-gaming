-- Attempt to prove ¬ Wings Rock leads to contradiction. Instead, we can prove Wings Rock.
theorem wings_rock : Wings Rock := by
  -- Rock is Peter's pet (A7)
  -- By A3, Rock is either monkey or bird
  cases A3 Rock A7 with
  | inl monkey_rock => 
      -- Rock is monkey -> mammal (A2)
      have mammal_rock : Mammal Rock := A2 Rock monkey_rock
      -- Mammal pet -> animal (A1)
      have animal_rock : Animal Rock := A1 Rock A7 mammal_rock
      -- Animal pet -> breathes (A5)
      have breathe_rock : Breathe Rock := A5 Rock animal_rock A7
      -- Now use A8: three cases
      cases A8 with
      | inl fly_rock => 
          -- Can fly -> has wings (A6)
          exact A6 Rock A7 fly_rock
      | inr rest => 
          cases rest with
          | inl bird_rock => 
              -- Bird pet -> can fly (A4) -> has wings (A6)
              have fly_rock : Fly Rock := A4 Rock A7 bird_rock
              exact A6 Rock A7 fly_rock
          | inr not_breathe => 
              -- Contradiction with breathe_rock
              contradiction
  | inr bird_rock => 
      -- Bird pet -> can fly (A4) -> has wings (A6)
      have fly_rock : Fly Rock := A4 Rock A7 bird_rock
      exact A6 Rock A7 fly_rock
