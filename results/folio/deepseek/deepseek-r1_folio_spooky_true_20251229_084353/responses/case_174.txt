<lean>
axiom obj : Type
axiom Taylor : obj
-- Predicates
axiom EatsSalads : obj → Prop                       -- Eats salads regularly
axiom HealthConscious : obj → Prop                  -- Very conscious about health
axiom GrewUpHealthConscious : obj → Prop            -- Grew up in health-conscious home
axiom FulfillsNutrition : obj → Prop                -- Fulfills daily nutritional intakes
axiom DisregardsWellBeing : obj → Prop              -- Disregards physical well-being
axiom VisitsGymDaily : obj → Prop                   -- Visits gym at least once daily

-- Premises
axiom A1 : ∀ x : obj, EatsSalads x → HealthConscious x
axiom A2 : ∀ x : obj, GrewUpHealthConscious x → EatsSalads x
axiom A3 : ∀ x : obj, FulfillsNutrition x → GrewUpHealthConscious x
axiom A4 : ∀ x : obj, DisregardsWellBeing x → ¬ HealthConscious x
axiom A5 : ∀ x : obj, VisitsGymDaily x → FulfillsNutrition x
axiom A6 : (GrewUpHealthConscious Taylor ∧ DisregardsWellBeing Taylor) ∨ 
           (¬ GrewUpHealthConscious Taylor ∧ ¬ DisregardsWellBeing Taylor)

-- Show that Taylor neither grew up in health-conscious home nor visits gym daily
theorem Taylor_case : ¬ GrewUpHealthConscious Taylor ∧ ¬ VisitsGymDaily Taylor := by
  -- Case analysis on Taylor's situation (A6)
  cases A6 with
  | inl h_and => 
      -- Case 1: Taylor has health-conscious home AND disregards well-being
      -- This leads to contradiction because:
      --   h_health: HealthConscious Taylor (from chain: GrewUp → EatsSalads → HealthConscious)
      --   h_not_health: ¬ HealthConscious Taylor (from DisregardsWellBeing → ¬ HealthConscious)
      have h_health : HealthConscious Taylor := by
        have h_salad := A2 Taylor h_and.left
        exact A1 Taylor h_salad
      have h_not_health : ¬ HealthConscious Taylor := 
        A4 Taylor h_and.right
      contradiction
  | inr h_neither =>
      -- Case 2: Taylor has neither health-conscious home NOR disregards well-being
      -- We specifically have ¬GrewUpHealthConscious Taylor from h_neither
      have H0 : ¬ GrewUpHealthConscious Taylor := h_neither.left
      
      -- Now prove ¬VisitsGymDaily Taylor by contradiction
      have H1 : ¬ VisitsGymDaily Taylor := by 
        intro h_gym
        -- Gym visit implies full nutrition (A5) which implies health-conscious home (A3)
        have h_nutrition := A5 Taylor h_gym
        have h_home := A3 Taylor h_nutrition
        contradiction  -- h_home contradicts H0
      -- Combine both parts
      exact ⟨H0, H1⟩
</lean>

ANSWER: True
