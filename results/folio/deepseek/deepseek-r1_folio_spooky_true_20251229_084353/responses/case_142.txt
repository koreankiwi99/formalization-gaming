<lean>
axiom obj : Type
axiom HeckCattle : obj → Prop
axiom ArtificiallySelected : obj → Prop
axiom BredBack : obj → Prop

-- Breeding back is a form of artificial selection
axiom A1 : ∀ x, BredBack x → ArtificiallySelected x

-- Heck cattle were bred back (interpreted existentially: at least one Heck cattle underwent breeding back)
axiom A2 : ∃ x, HeckCattle x ∧ BredBack x

theorem some_heck_selected : ∃ x, HeckCattle x ∧ ArtificiallySelected x := by
  obtain ⟨w, hwHC, hwBB⟩ := A2  -- Unpack existential: w is an object that is both Heck cattle and bred back
  have hwAS : ArtificiallySelected w := A1 w hwBB  -- Apply selection property to w
  exact ⟨w, hwHC, hwAS⟩  -- Construct required existential

</lean>

ANSWER: True
