<lean>
axiom Person : Type
axiom Hannah : Person
axiom WorksJob : Person → Prop
axiom NeedsEarn : Person → Prop
axiom OrderTakeout : Person → Prop
axiom EnjoyDining : Person → Prop
axiom NotPicky : Person → Prop
axiom SpendsTime : Person → Prop

-- Premises:
axiom P1 : ∀ x : Person, WorksJob x → NeedsEarn x
axiom P2 : ∀ x : Person, OrderTakeout x → WorksJob x
axiom P3 : ∀ x : Person, OrderTakeout x ∨ EnjoyDining x
axiom P4 : ∀ x : Person, EnjoyDining x → NotPicky x
axiom P5 : ∀ x : Person, EnjoyDining x → SpendsTime x
axiom P6 : WorksJob Hannah
axiom P7 : NeedsEarn Hannah → (NotPicky Hannah ∧ ¬NeedsEarn Hannah)

-- Proof for the statement: Hannah is not picky and spends time
theorem hannah_goal : NotPicky Hannah ∧ SpendsTime Hannah :=
  have Hneed : NeedsEarn Hannah := P1 Hannah P6  -- By P1 and P6
  have Hcontra : NotPicky Hannah ∧ ¬NeedsEarn Hannah := P7 Hneed  -- By P7
  have Hfalse : False := Hcontra.2 Hneed         -- Contradiction (Hneed and ¬Hneed)
  False.elim Hfalse                              -- Ex falso: derive goal
</lean>

ANSWER: True
