<lean>
axiom obj : Type
axiom Marvin : obj
axiom Alien : obj → Prop
axiom Extraterrestrial : obj → Prop
axiom FromMars : obj → Prop
axiom Human : obj → Prop
axiom HighlyIntelligent : obj → Prop
axiom FromEarth : obj → Prop

-- Context axioms
axiom T1 : ∀ x : obj, Alien x → Extraterrestrial x
axiom T2 : ∀ x : obj, FromMars x → Alien x
axiom T3 : ∀ x : obj, Extraterrestrial x → ¬ Human x
axiom T4 : ∀ x : obj, (HighlyIntelligent x ∧ FromEarth x) → Human x
axiom T5 : HighlyIntelligent Marvin
axiom T6 : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬ FromEarth Marvin ∧ ¬ FromMars Marvin)
axiom T7 : ¬ FromEarth Marvin → Extraterrestrial Marvin

-- Prove: Marvin is neither human nor from Mars
theorem goal : ¬ Human Marvin ∧ ¬ FromMars Marvin := by
  cases T6 with
  | inl h =>  -- Case 1: From Earth and Mars
    have ⟨earth, mars⟩ := h
    have humanMarvin : Human Marvin := T4 Marvin ⟨T5, earth⟩
    have alienMarvin : Alien Marvin := T2 Marvin mars
    have etMarvin : Extraterrestrial Marvin := T1 Marvin alienMarvin
    have notHuman : ¬ Human Marvin := T3 Marvin etMarvin
    contradiction
  | inr h =>  -- Case 2: From neither
    have ⟨notEarth, notMars⟩ := h
    have etMarvin : Extraterrestrial Marvin := T7 notEarth
    have notHuman : ¬ Human Marvin := T3 Marvin etMarvin
    exact ⟨notHuman, notMars⟩
</lean>

ANSWER: True
