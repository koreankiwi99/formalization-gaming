<lean>
axiom obj : Type
axiom Event : Type
axiom Beijing : obj
axiom Hosted : obj → Event → Prop
axiom isSummerOlympics : Event → Prop
axiom isWinterOlympics : Event → Prop

-- Axioms from context
axiom specific_summer : Event
axiom hosted_specific_summer : Hosted Beijing specific_summer
axiom is_specific_summer : isSummerOlympics specific_summer

axiom has_hosted_both_types : 
  (∃ s, isSummerOlympics s ∧ Hosted Beijing s) ∧ 
  (∃ w, isWinterOlympics w ∧ Hosted Beijing w)

-- Key components for the proof
theorem winter_exists : ∃ w, isWinterOlympics w ∧ Hosted Beijing w := 
  has_hosted_both_types.right

theorem statement :
  Hosted Beijing specific_summer ∧ (∃ w, isWinterOlympics w ∧ Hosted Beijing w) :=
  by 
    apply And.intro hosted_specific_summer 
    exact winter_exists

-- Since the statement is proven directly
</lean>

ANSWER: True
