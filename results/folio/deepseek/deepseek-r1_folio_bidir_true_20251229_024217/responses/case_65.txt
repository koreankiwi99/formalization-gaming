<lean>
axiom Entity : Type
axiom Yale : Entity
axiom Private : Entity → Prop
axiom IvyLeague : Entity → Prop
axiom ResearchUniversity : Entity → Prop
axiom MovedToNewHaven : Entity → Prop

-- Yale University is a private Ivy League research university
axiom A1 : Private Yale
axiom A2 : IvyLeague Yale
axiom A3 : ResearchUniversity Yale

-- Yale University moved to New Haven
axiom A4 : MovedToNewHaven Yale

-- Prove: There exists a private Ivy League research university that moved to New Haven
theorem exists_private_ivy_research_moved : ∃ x, Private x ∧ IvyLeague x ∧ ResearchUniversity x ∧ MovedToNewHaven x :=
by
  exists Yale
  constructor
  . exact A1
  . constructor
    . exact A2
    . constructor
      . exact A3
      . exact A4
</lean>

ANSWER: True
