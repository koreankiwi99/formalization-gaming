axiom obj : Type
axiom DiamondMine : obj
axiom RoderickStrong : obj
axiom ProfessionalWrestlingStable : obj → Prop
axiom Leads : obj → obj → Prop

axiom stableDiamondMine : ProfessionalWrestlingStable DiamondMine

theorem strong_leads_stable : ∃ s, ProfessionalWrestlingStable s ∧ Leads RoderickStrong s :=
