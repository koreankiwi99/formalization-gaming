<lean>
axiom Applicant : Type
axiom Tom : Applicant
axiom ProvesEconomicHardship : Applicant → Prop
axiom GetsFinancialAid : Applicant → Prop
axiom SingleParentFamily : Applicant → Prop
axiom HasFewResources : Applicant → Prop

axiom aid_rule : ∀ x : Applicant, ProvesEconomicHardship x → GetsFinancialAid x
axiom hardship_rule : ∀ x : Applicant, (SingleParentFamily x ∨ HasFewResources x) → ProvesEconomicHardship x
axiom tom_family : SingleParentFamily Tom
axiom tom_resources : HasFewResources Tom

theorem ets_aids_Tom : GetsFinancialAid Tom := by
  apply aid_rule Tom
  apply hardship_rule Tom
  left
  exact tom_family
</lean>

ANSWER: True
