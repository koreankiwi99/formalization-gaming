<lean>
axiom Animal : Type
axiom Rabbit : Animal → Prop
axiom Turtle : Animal → Prop
axiom Squirrel : Animal → Prop
axiom SpottedNearCampus : Animal → Prop
axiom Cute : Animal → Prop
axiom Skittish : Animal → Prop
axiom Calm : Animal → Prop

axiom Rockie : Animal

-- Premises:
axiom all_spotted_rabbits_cute : ∀ a, Rabbit a ∧ SpottedNearCampus a → Cute a
axiom some_spotted_turtles : ∃ a, Turtle a ∧ SpottedNearCampus a
axiom only_rabbits_squirrels_spotted : ∀ a, SpottedNearCampus a → Rabbit a ∨ Squirrel a
axiom skittish_not_calm : ∀ a, Skittish a → ¬ Calm a
axiom spotted_squirrels_skittish : ∀ a, Squirrel a ∧ SpottedNearCampus a → Skittish a
axiom rockie_spotted_calm : SpottedNearCampus Rockie ∧ Calm Rockie

theorem rockie_turtle_or_cute : Turtle Rockie ∨ Cute Rockie := by
  have R_spotted := rockie_spotted_calm.1
  have R_calm := rockie_spotted_calm.2
  have species := only_rabbits_squirrels_spotted Rockie R_spotted
  cases species with
  | inl R_rabbit => 
    apply Or.inr
    exact all_spotted_rabbits_cute Rockie ⟨R_rabbit, R_spotted⟩
  | inr R_squirrel =>
    have R_skittish := spotted_squirrels_skittish Rockie ⟨R_squirrel, R_spotted⟩
    have not_calm := skittish_not_calm Rockie R_skittish
    contradiction
</lean>

ANSWER: True
