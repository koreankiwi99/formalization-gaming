<lean>
axiom Person : Type
axiom Djokovic : Person
axiom GrandSlamChampion : Person → Prop
axiom OscarNominatedActor : Person → Prop
axiom ProfessionalTennisPlayer : Person → Prop
axiom Celebrity : Person → Prop
axiom Athlete : Person → Prop
axiom WellPaid : Person → Prop
axiom Famous : Person → Prop
axiom LivesInTaxHaven : Person → Prop

-- Context axioms
axiom A1 : ∀ p : Person, GrandSlamChampion p ∨ OscarNominatedActor p
axiom A2 : ∀ p : Person, GrandSlamChampion p → ProfessionalTennisPlayer p
axiom A3 : ∀ p : Person, OscarNominatedActor p → Celebrity p
axiom A4 : ∀ p : Person, ProfessionalTennisPlayer p → Athlete p
axiom A5 : ∀ p : Person, Celebrity p → WellPaid p
axiom A6 : ∀ p : Person, Athlete p → Famous p
axiom A7 : ∀ p : Person, WellPaid p → LivesInTaxHaven p
axiom A8 : (Athlete Djokovic ∧ Famous Djokovic) → WellPaid Djokovic

theorem djokovic_tax_haven : LivesInTaxHaven Djokovic := by
  -- Djokovic is either GrandSlamChampion or OscarNominatedActor (A1)
  have cases := A1 Djokovic
  cases cases with
  | inl gs =>
    -- Case: GrandSlamChampion Djokovic
    have tennis := A2 Djokovic gs
    have athlete := A4 Djokovic tennis
    have famous := A6 Djokovic athlete
    have famous_athlete := And.intro athlete famous
    have well_paid := A8 famous_athlete
    exact A7 Djokovic well_paid
  | inr oscar =>
    -- Case: OscarNominatedActor Djokovic
    have celebrity := A3 Djokovic oscar
    have well_paid := A5 Djokovic celebrity
    exact A7 Djokovic well_paid

</lean>

ANSWER: True
