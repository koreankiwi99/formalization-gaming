axiom obj : Type
axiom Square : obj → Prop
axiom FourSided : obj → Prop
axiom Shape : obj → Prop
axiom T1 : ∀ x : obj, Square x → FourSided x
axiom T2 : ∀ x : obj, FourSided x → Shape x
theorem AllSquaresAreShapes : ∀ x : obj, Square x → Shape x :=

2. Assuming `h : Square x`
3. Applying `T1` to get `FourSided x`
4. Applying `T2` to get `Shape x`
