<lean>
axiom obj : Type
axiom SymphonyNo9 : obj
axiom Beethoven : obj
axiom ViennaMusicSociety : obj

axiom MusicPiece : obj → Prop
axiom Premiered : obj → obj → Prop
axiom Orchestra : obj → Prop

axiom T1 : MusicPiece SymphonyNo9
axiom T2 : Premiered ViennaMusicSociety SymphonyNo9
axiom T3 : Orchestra ViennaMusicSociety

theorem exists_orchestra_premiered : ∃ (x : obj) (y : obj), Orchestra x ∧ MusicPiece y ∧ Premiered x y :=
  ⟨ViennaMusicSociety, SymphonyNo9, T3, T1, T2⟩
</lean>

ANSWER: True
