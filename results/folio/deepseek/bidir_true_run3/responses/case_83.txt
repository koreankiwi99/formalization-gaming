<lean>
axiom obj : Type
axiom HamdenPlazaSubway : obj
axiom ListedInYelpRecs : obj → Prop
axiom ManyNegativeReviews : obj → Prop
axiom RatingGreaterThanFour : obj → Prop
axiom ProvidesTakeOut : obj → Prop
axiom PopularAmongLocals : obj → Prop

axiom A1 : ∀ x : obj, ListedInYelpRecs x → ¬ ManyNegativeReviews x
axiom A2 : ∀ x : obj, RatingGreaterThanFour x → ListedInYelpRecs x
axiom A3 : ∃ x : obj, ¬ ProvidesTakeOut x ∧ ManyNegativeReviews x
axiom A4 : ∀ x : obj, PopularAmongLocals x → RatingGreaterThanFour x
axiom A5 : RatingGreaterThanFour HamdenPlazaSubway ∨ PopularAmongLocals HamdenPlazaSubway

theorem target : 
  (ProvidesTakeOut HamdenPlazaSubway ∧ ManyNegativeReviews HamdenPlazaSubway) → 
  (RatingGreaterThanFour HamdenPlazaSubway ∧ ¬ ProvidesTakeOut HamdenPlazaSubway) :=
fun h => 
  let ⟨p, m⟩ := h
  have notListed : ¬ ListedInYelpRecs HamdenPlazaSubway := 
    fun listed => A1 HamdenPlazaSubway listed m
  have notR : ¬ RatingGreaterThanFour HamdenPlazaSubway := 
    fun r => notListed (A2 HamdenPlazaSubway r)
  have L : PopularAmongLocals HamdenPlazaSubway := 
    match A5 with
    | Or.inl r => False.elim (notR r)
    | Or.inr l => l
  have R : RatingGreaterThanFour HamdenPlazaSubway := A4 HamdenPlazaSubway L
  False.elim (notR R)
</lean>

ANSWER: True
