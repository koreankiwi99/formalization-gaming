<lean>
axiom entity : Type
axiom DiamondMine : entity
axiom RoderickStrong : entity
axiom ProfessionalWrestlingStable : entity → Prop
axiom Leads : entity → entity → Prop

axiom stable_formation : ProfessionalWrestlingStable DiamondMine
axiom leadership : Leads RoderickStrong DiamondMine

theorem strong_leads_stable : ∃ s, ProfessionalWrestlingStable s ∧ Leads RoderickStrong s :=
  by
    apply Exists.intro DiamondMine
    constructor
    · exact stable_formation
    · exact leadership
</lean>

ANSWER: True
