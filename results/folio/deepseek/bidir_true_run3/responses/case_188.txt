<lean>
axiom obj : Type
axiom BasketballPlayer : obj → Prop
axiom American : obj → Prop
axiom JumpShots : obj → Prop
axiom Leap : obj → Prop
axiom ActivateLegMuscles : obj → Prop
axiom Yuri : obj

-- Some professional basketball players are not American nationals
axiom T1 : ∃ x, BasketballPlayer x ∧ ¬ American x

-- All professional basketball players can do jump shots
axiom T2 : ∀ x, BasketballPlayer x → JumpShots x

-- If someone can do jump shots, they leap straight into the air
axiom T3 : ∀ x, JumpShots x → Leap x

-- If someone leaps straight into the air, they activate their leg muscles
axiom T4 : ∀ x, Leap x → ActivateLegMuscles x

-- Yuri does not activate his leg muscles
axiom T5 : ¬ ActivateLegMuscles Yuri

-- Prove Yuri is not an American professional basketball player
theorem yuri_not_american_player : ¬ (BasketballPlayer Yuri ∧ American Yuri) :=
  fun ⟨hBP, hAm⟩ => 
  have hJS : JumpShots Yuri := T2 Yuri hBP
  have hLeap : Leap Yuri := T3 Yuri hJS
  have hAct : ActivateLegMuscles Yuri := T4 Yuri hLeap
  T5 hAct
</lean>

ANSWER: True
