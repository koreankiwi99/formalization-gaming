axiom WeddingPlans : obj → Prop

-- All people who make wedding plans are engaged
axiom A1 : ∀ x : obj, WeddingPlans x → Engaged x

-- All who invite others make wedding plans
axiom A2 : ∀ x : obj, InviteOthers x → WeddingPlans x

-- Well-attended wedding implies invited others
axiom A3 : ∀ x : obj, WellAttendedWedding x → InviteOthers x

-- Some with well-attended weddings have larger families
axiom A4 : ∃ x : obj, WellAttendedWedding x ∧ LargerFamily x

-- If John is engaged, then he doesn't invite friends and doesn't make wedding plans
axiom A5 : Engaged John → ¬ InviteFriends John ∧ ¬ WeddingPlans John

-- If John has larger family, then he has well-attended wedding or invites friends
axiom A6 : LargerFamily John → WellAttendedWedding John ∨ InviteFriends John

-- Attempt to prove ¬LargerFamily John
theorem not_larger_family : ¬ LargerFamily John := by
  intro hLarger  -- Assume John has larger family
  have hDisj := A6 hLarger  -- Now hDisj: WellAttendedWedding John ∨ InviteFriends John
  cases hDisj with
  | inl hWedding =>  -- Case 1: WellAttendedWedding John
    have hInvite := A3 John hWedding  -- InviteOthers John
    have hPlans := A2 John hInvite    -- WeddingPlans John
    have hEngaged := A1 John hPlans   -- Engaged John
    have hNot := A5 hEngaged          -- ¬InviteFriends John ∧ ¬WeddingPlans John
    cases hNot with
    | intro hNoFriends hNoPlans =>
      exact hNoPlans hPlans          -- Contradiction: ¬WeddingPlans vs WeddingPlans
  | inr hFriends =>  -- Case 2: InviteFriends John
    -- No contradiction found in this branch
    -- We cannot derive a contradiction from InviteFriends John alone
    admit  -- Proof fails here
