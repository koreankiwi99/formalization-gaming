<lean>
axiom obj : Type
axiom Tom : obj
axiom Applicant : obj → Prop
axiom EconomicHardship : obj → Prop
axiom FinancialAid : obj → Prop
axiom SingleParentFamily : obj → Prop
axiom FewResources : obj → Prop

axiom rule1 : ∀ x : obj, Applicant x ∧ EconomicHardship x → FinancialAid x
axiom rule2 : ∀ x : obj, (SingleParentFamily x ∨ FewResources x) → EconomicHardship x
axiom fact1 : SingleParentFamily Tom
axiom fact2 : FewResources Tom
axiom fact3 : Applicant Tom

theorem ets_aid_tom : FinancialAid Tom := by
  have h_hardship : EconomicHardship Tom := by
    apply rule2 Tom
    left  -- Using left because SingleParentFamily Tom is true (fact1). Alternatively, we could use right for FewResources Tom.
    exact fact1
  apply rule1 Tom
  constructor
  · exact fact3
  · exact h_hardship
</lean>

ANSWER: True
