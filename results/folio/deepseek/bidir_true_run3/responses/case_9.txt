The reasoning is based on the six types being exhaustive for wild turkeys. Since Tom is a wild turkey and excluded from the first five types, he must be the remaining type (Ocellated).

<lean>
axiom Turkey : Type
axiom Tom : Turkey
axiom Eastern: Turkey → Prop
axiom Osceola: Turkey → Prop
axiom Goulds: Turkey → Prop
axiom Merriam: Turkey → Prop
axiom RioGrande: Turkey → Prop
axiom Ocellated: Turkey → Prop

axiom types_exhaustive : ∀ t : Turkey, 
    Eastern t ∨ Osceola t ∨ Goulds t ∨ Merriam t ∨ RioGrande t ∨ Ocellated t

axiom not_eastern : ¬ Eastern Tom
axiom not_osceola : ¬ Osceola Tom
axiom not_goulds : ¬ Goulds Tom
axiom not_merriam_rio : ¬ Merriam Tom ∧ ¬ RioGrande Tom

theorem tom_ocellated : Ocellated Tom := by
  have H := types_exhaustive Tom
  rcases H with e | o | g | m | r | oc
  · exact (not_eastern e).elim
  · exact (not_osceola o).elim
  · exact (not_goulds g).elim
  · exact (not_merriam_rio.left m).elim
  · exact (not_merriam_rio.right r).elim
  · exact oc
</lean>

ANSWER: True
