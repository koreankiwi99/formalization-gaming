<lean>
axiom Season : Type
axiom Spring : Season
axiom Summer : Season
axiom Fall : Season
axiom Winter : Season
axiom season_eq : ∀ s : Season, s = Spring ∨ s = Summer ∨ s = Fall ∨ s = Winter

axiom Student : Type
axiom Emma : Student
axiom Mia : Student
axiom James : Student

axiom FavoriteSeason : Student → Season → Prop
axiom WantsLongVacation : Student → Prop

-- All students who want long vacation have summer as favorite season
axiom vacation_summer : ∀ s : Student, WantsLongVacation s → FavoriteSeason s Summer

-- Emma's favorite season is summer
axiom emma_summer : FavoriteSeason Emma Summer

-- Mia's favorite season ≠ Emma's
axiom mia_not_emma : ∀ ms : Season, FavoriteSeason Mia ms → ms ≠ Summer

-- James wants long vacation
axiom james_vacation : WantsLongVacation James

-- Prove James's favorite season is summer
theorem james_summer : FavoriteSeason James Summer := by
  exact vacation_summer James james_vacation
</lean>

ANSWER: True
