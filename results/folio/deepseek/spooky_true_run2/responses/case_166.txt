- We have an axiom: `German HeinrichSchmidt` (from the context).
- The theorem shows `German HeinrichSchmidt ∨ Russian HeinrichSchmidt` holds because the left disjunct is true.

axiom obj : Type
axiom HeinrichSchmidt : obj

-- Theorem: Heinrich Schmidt was German or Russian (or both)
theorem schmidt_nationality : German HeinrichSchmidt ∨ Russian HeinrichSchmidt :=
