axiom U : Type
axiom Marvin : U
axiom Alien : U → Prop
axiom Extraterrestrial : U → Prop
axiom FromMars : U → Prop
axiom Human : U → Prop

axiom all_aliens_are_extraterrestrials : ∀ x, Alien x → Extraterrestrial x

axiom no_extraterrestrials_are_human : ∀ x, Extraterrestrial x → ¬ Human x

axiom marvin_highly_intelligent : HighlyIntelligent Marvin
axiom marvin_earth_mars : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬ FromEarth Marvin ∧ ¬ FromMars Marvin)
axiom not_earth_then_et : ¬ FromEarth Marvin → Extraterrestrial Marvin

theorem disprove_statement : ¬(¬ FromMars Marvin → Human Marvin) := by
  intro impl
  have h1 : (FromEarth Marvin ∧ FromMars Marvin) → False := by
    intro cs
    have ⟨h_earth, h_mars⟩ := cs
    have alien : Alien Marvin := from_mars_are_aliens Marvin h_mars
    have et : Extraterrestrial Marvin := all_aliens_are_extraterrestrials Marvin alien
    have not_human : ¬Human Marvin := no_extraterrestrials_are_human Marvin et
    have human : Human Marvin := intelligent_earthlings_are_human Marvin ⟨marvin_highly_intelligent, h_earth⟩
    contradiction
  cases marvin_earth_mars
  case inl h => contradiction
  case inr h => 
    have ⟨not_earth, not_mars⟩ := h
    have et : Extraterrestrial Marvin := not_earth_then_et not_earth
    have not_human : ¬Human Marvin := no_extraterrestrials_are_human Marvin et
    have : Human Marvin := impl not_mars
    contradiction

The proof establishes that:
1. The premise about Marvin's origin (either both Earth & Mars or neither) contains a contradiction in the first case (Earth & Mars) because it forces Marvin to be both human (via Earth intelligence) and non-human (via extraterrestrial).
2. This eliminates the first case, leaving only "neither Earth nor Mars".
3. In this case, Marvin is extraterrestrial (by premise) and thus non-human.
4. With ¬FromMars Marvin true and Human Marvin false, the implication (¬FromMars → Human) must be false.

ANSWER: False
