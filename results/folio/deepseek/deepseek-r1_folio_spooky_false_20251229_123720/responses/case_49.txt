<lean>
axiom obj : Type
axiom s : obj  -- Represents "1984"
axiom Digital : obj → Prop
axiom Analog : obj → Prop
axiom PrintedText : obj → Prop
axiom StreamingService : obj → Prop
axiom HardcoverBook : obj → Prop

-- Premises
axiom P₁ : ∀ x, Digital x → ¬ Analog x
axiom P₂ : ∀ x, PrintedText x → Analog x
axiom P₃ : ∀ x, StreamingService x → Digital x
axiom P₄ : ∀ x, HardcoverBook x → PrintedText x
axiom P₅ : StreamingService s → HardcoverBook s

-- Given statement
axiom H : StreamingService s

theorem contradiction : False :=
  have h₁ : HardcoverBook s := P₅ H
  have h₂ : PrintedText s := P₄ s h₁
  have h₃ : Analog s := P₂ s h₂
  have h₄ : Digital s := P₃ s H
  have h₅ : ¬ Analog s := P₁ s h₄
  h₅ h₃
</lean>

ANSWER: False
