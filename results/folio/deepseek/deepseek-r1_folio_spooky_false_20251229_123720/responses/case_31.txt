axiom Athlete : obj → Prop
axiom Celebrity : obj → Prop
axiom WellPaid : obj → Prop
axiom Famous : obj → Prop
axiom LivesInTaxHaven : obj → Prop

axiom P1 : ∀ x, AtMixer x → GrandSlam x ∨ OscarNominated x
axiom P2 : ∀ x, AtMixer x → GrandSlam x → TennisPlayer x
axiom P3 : ∀ x, AtMixer x → OscarNominated x → Celebrity x
axiom P4 : ∀ x, AtMixer x → TennisPlayer x → Athlete x
axiom P5 : ∀ x, AtMixer x → Celebrity x → WellPaid x
axiom P6 : ∀ x, AtMixer x → Athlete x → Famous x
axiom P7 : ∀ x, AtMixer x → WellPaid x → LivesInTaxHaven x

-- Attempt to prove ¬LivesInTaxHaven Djokovic leads to contradiction
theorem disprove_conclusion : ¬LivesInTaxHaven Djokovic → False := by
  intro h_not_tax_haven
  have h_mixer : AtMixer Djokovic := P8.1
  have h_disj : GrandSlam Djokovic ∨ OscarNominated Djokovic := P1 Djokovic h_mixer
  cases h_disj with
  | inl gs =>
    have tennis : TennisPlayer Djokovic := P2 Djokovic h_mixer gs
    have athlete : Athlete Djokovic := P4 Djokovic h_mixer tennis
    have famous : Famous Djokovic := P6 Djokovic h_mixer athlete
    have well_paid : WellPaid Djokovic := P8.2 ⟨famous, athlete⟩
    have tax_haven : LivesInTaxHaven Djokovic := P7 Djokovic h_mixer well_paid
    exact h_not_tax_haven tax_haven
  | inr on =>
    have celebrity : Celebrity Djokovic := P3 Djokovic h_mixer on
    have well_paid : WellPaid Djokovic := P5 Djokovic h_mixer celebrity
    have tax_haven : LivesInTaxHaven Djokovic := P7 Djokovic h_mixer well_paid
    exact h_not_tax_haven tax_haven

-- Since we proved ¬LivesInTaxHaven Djokovic → False, we have LivesInTaxHaven Djokovic
