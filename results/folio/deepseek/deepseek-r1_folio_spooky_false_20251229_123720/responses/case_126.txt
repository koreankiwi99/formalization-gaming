<lean>
axiom obj : Type
axiom Rock : obj
axiom PeterPet : obj → Prop
axiom Mammal : obj → Prop
axiom Animal : obj → Prop
axiom Monkey : obj → Prop
axiom Bird : obj → Prop
axiom CanFly : obj → Prop
axiom CanBreathe : obj → Prop
axiom HasWings : obj → Prop

axiom P1 : ∀ x : obj, PeterPet x ∧ Mammal x → Animal x
axiom P2 : ∀ x : obj, Monkey x → Mammal x
axiom P3 : ∀ x : obj, PeterPet x → Monkey x ∨ Bird x
axiom P4 : ∀ x : obj, PeterPet x ∧ Bird x → CanFly x
axiom P5 : ∀ x : obj, Animal x ∧ PeterPet x → CanBreathe x
axiom P6 : ∀ x : obj, PeterPet x ∧ CanFly x → HasWings x
axiom P7 : PeterPet Rock
axiom P8 : CanFly Rock ∨ Bird Rock ∨ ¬CanBreathe Rock

theorem rock_has_wings : HasWings Rock := by
  apply P6 Rock
  constructor
  · exact P7
  · -- Prove CanFly Rock using P8
    cases P8 with
    | inl h => exact h
    | inr h =>
      cases h with
      | inl birdRock => 
        exact P4 Rock ⟨P7, birdRock⟩
      | inr notBreathe => 
        have petRock := P7
        have typeRock := P3 Rock petRock
        cases typeRock with
        | inl monkeyRock =>
          have mammalRock := P2 Rock monkeyRock
          have animalRock := P1 Rock ⟨petRock, mammalRock⟩
          have breatheRock := P5 Rock ⟨animalRock, petRock⟩
          contradiction
        | inr birdRock =>
          exact P4 Rock ⟨P7, birdRock⟩
</lean>

ANSWER: False
