<lean>
axiom Student : Type

axiom WorksInLibrary : Student → Prop
axiom FromCS : Student → Prop
axiom HasUniPartTimeJob : Student → Prop
axiom TakingDatabase : Student → Prop
axiom TakingClassWithDavid : Student → Prop
axiom WorksInLab : Student → Prop

axiom James : Student

-- None of the students who currently work in the library are from the computer science department.
axiom A1 : ∀ (s : Student), WorksInLibrary s → ¬FromCS s

-- All of the students with part-time jobs offered by the university currently work in the library.
axiom A2 : ∀ (s : Student), HasUniPartTimeJob s → WorksInLibrary s

-- All of the students who are taking the database course are from the computer science department.
axiom A3 : ∀ (s : Student), TakingDatabase s → FromCS s

-- All of the students taking a class with Professor David are taking the database course.
axiom A4 : ∀ (s : Student), TakingClassWithDavid s → TakingDatabase s

-- All students working in the lab are taking a class with Professor David.
axiom A5 : ∀ (s : Student), WorksInLab s → TakingClassWithDavid s

-- James is a student working in the lab.
axiom A6 : WorksInLab James

-- James doesn't work in the lab or have a part-time job offered by the university.
axiom A7 : ¬WorksInLab James ∨ ¬HasUniPartTimeJob James

-- Prove that James does NOT have a part-time job
theorem James_no_job : ¬HasUniPartTimeJob James :=
  by
    intro hJob
    -- James has Uni job → Works in library (by A2)
    have hLib : WorksInLibrary James := A2 James hJob
    -- Works in library → not from CS (by A1)
    have hNotCS : ¬FromCS James := A1 James hLib
    -- James works in lab (by A6)
    have hLab : WorksInLab James := A6
    -- Works in lab → takes class with David (by A5)
    have hDavid : TakingClassWithDavid James := A5 James hLab
    -- Takes class with David → takes database (by A4)
    have hDB : TakingDatabase James := A4 James hDavid
    -- Takes database → from CS (by A3)
    have hCS : FromCS James := A3 James hDB
    -- Contradiction: FromCS and ¬FromCS
    exact hNotCS hCS

-- The question is: "James has a part-time job offered by the university."
-- We proved ¬HasUniPartTimeJob James, so the statement is false.
</lean>

ANSWER: False
