1. Djokovic is at the mixer (A8)

   - Oscar-nominated actors → celebrities (A3)

axiom GSC : Person → Prop

axiom LivesInTaxHaven : Person → Prop

theorem djokovic_tax_haven : LivesInTaxHaven Djokovic := by

  cases A1 Djokovic hMixer with
