<lean>
axiom Person : Type
axiom Taylor : Person

axiom EatsSaladsRegularly : Person → Prop
axiom HealthConscious : Person → Prop
axiom GrewUpHealthConsciousHome : Person → Prop
axiom FulfillsNutrition : Person → Prop
axiom DisregardsPhysicalWellBeing : Person → Prop
axiom VisitsGymDaily : Person → Prop

-- Axioms from context
axiom A1 : ∀ x : Person, EatsSaladsRegularly x → HealthConscious x
axiom A2 : ∀ x : Person, GrewUpHealthConsciousHome x → EatsSaladsRegularly x
axiom A3 : ∀ x : Person, FulfillsNutrition x → GrewUpHealthConsciousHome x
axiom A4 : ∀ x : Person, DisregardsPhysicalWellBeing x → ¬ HealthConscious x
axiom A5 : ∀ x : Person, VisitsGymDaily x → FulfillsNutrition x
axiom A6 : (GrewUpHealthConsciousHome Taylor ∧ DisregardsPhysicalWellBeing Taylor) ∨ 
           (¬ GrewUpHealthConsciousHome Taylor ∧ ¬ DisregardsPhysicalWellBeing Taylor)

theorem taylor_case : ¬ GrewUpHealthConsciousHome Taylor ∧ ¬ VisitsGymDaily Taylor := by
  -- First, eliminate the disjunction in A6
  have h : ¬ (GrewUpHealthConsciousHome Taylor ∧ DisregardsPhysicalWellBeing Taylor) := by
    intro ⟨hGrew, hDisregard⟩
    have salads := A2 Taylor hGrew
    have healthCons := A1 Taylor salads
    exact A4 Taylor hDisregard healthCons
  have finalCase : ¬ GrewUpHealthConsciousHome Taylor ∧ ¬ DisregardsPhysicalWellBeing Taylor := by
    cases A6 with
    | inl h1 => exact absurd h1 h
    | inr h2 => exact h2
  apply And.intro finalCase.1
  intro hGym
  -- Derive contradiction from gym visit
  have fulfills := A5 Taylor hGym
  have grewUp := A3 Taylor fulfills
  exact finalCase.1 grewUp
</lean>

ANSWER: True
