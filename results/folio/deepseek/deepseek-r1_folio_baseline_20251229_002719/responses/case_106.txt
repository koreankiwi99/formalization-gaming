axiom GREApplicant : Person → Prop
axiom ProvesEconomicHardship : Person → Prop
axiom GetsFinancialAid : Person → Prop
axiom LivesInSingleParentFamily : Person → Prop

axiom ets_aid : ∀ (p : Person), GREApplicant p → ProvesEconomicHardship p → GetsFinancialAid p
axiom hardship_criteria : ∀ (p : Person), LivesInSingleParentFamily p ∨ HasFewResources p → ProvesEconomicHardship p

axiom tom_few_resources : HasFewResources Tom
axiom tom_applicant : GREApplicant Tom

-- Tom proves economic hardship (from at least one of the criteria)

  hardship_criteria Tom (Or.inl tom_single_parent)  -- Or.inr tom_few_resources would also work

  ets_aid Tom tom_applicant tom_hardship

-- The claim: "No GRE applicant has financial aid" (i.e., ∀ p, GREApplicant p → ¬GetsFinancialAid p)
