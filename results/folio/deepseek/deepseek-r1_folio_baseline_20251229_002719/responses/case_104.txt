<lean>
axiom Person : Type
axiom Tom : Person
axiom ApplyingGRE : Person → Prop
axiom FinancialAid : Person → Prop
axiom ProveEconomicHardship : Person → Prop
axiom LivingInSingleParentFamily : Person → Prop
axiom HasFewResources : Person → Prop

axiom rule1 : ∀ p : Person, ApplyingGRE p ∧ ProveEconomicHardship p → FinancialAid p
axiom rule2 : ∀ p : Person, LivingInSingleParentFamily p ∨ HasFewResources p → ProveEconomicHardship p

axiom tom_single_parent : LivingInSingleParentFamily Tom
axiom tom_few_resources : HasFewResources Tom
axiom tom_applying : ApplyingGRE Tom

theorem tom_gets_aid : FinancialAid Tom := by
  apply rule1 Tom
  constructor
  . exact tom_applying
  . apply rule2 Tom
    left  -- Using left disjunct since Tom has a single-parent family (also could use right with few resources)
    exact tom_single_parent
</lean>

ANSWER: True
