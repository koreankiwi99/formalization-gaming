<lean>
axiom obj : Type
axiom Rock : obj
axiom Pet : obj → Prop
axiom Mammal : obj → Prop
axiom Animal : obj → Prop
axiom Monkey : obj → Prop
axiom Bird : obj → Prop
axiom CanFly : obj → Prop
axiom CanBreathe : obj → Prop
axiom HasWings : obj → Prop

axiom T1 : ∀x : obj, Pet x ∧ Mammal x → Animal x
axiom T2 : ∀x : obj, Monkey x → Mammal x
axiom T3 : ∀x : obj, Pet x → Monkey x ∨ Bird x
axiom T4 : ∀x : obj, Pet x ∧ Bird x → CanFly x
axiom T5 : ∀x : obj, Animal x ∧ Pet x → CanBreathe x
axiom T6 : ∀x : obj, Pet x ∧ CanFly x → HasWings x
axiom T7 : Pet Rock
axiom T8 : CanFly Rock ∨ Bird Rock ∨ ¬ CanBreathe Rock

theorem rock_has_wings : HasWings Rock := by
  have pet_rock: Pet Rock := T7
  apply Or.elim T8
  . intro h_fly  -- Case 1: Rock can fly
    exact T6 Rock ⟨T7, h_fly⟩
  . intro h_disj -- Case 2: Either Bird(Rock) or ¬CanBreathe(Rock)
    apply Or.elim h_disj
    . intro h_bird  -- Case 2a: Rock is a bird
      have fly_rock : CanFly Rock := T4 Rock ⟨T7, h_bird⟩
      exact T6 Rock ⟨T7, fly_rock⟩
    . intro h_no_breath  -- Case 2b: Rock cannot breathe
      have contra : ¬ Animal Rock := by
        intro h_animal
        have breath : CanBreathe Rock := T5 Rock ⟨h_animal, T7⟩
        contradiction
      have not_mammal : ¬ Mammal Rock := by
        intro h_mammal
        have h_animal : Animal Rock := T1 Rock ⟨T7, h_mammal⟩
        contradiction
      have animal_type : Monkey Rock ∨ Bird Rock := T3 Rock T7
      apply Or.elim animal_type
      . intro h_monkey
        have mammal_rock : Mammal Rock := T2 Rock h_monkey
        contradiction
      . intro h_bird
        have fly_rock : CanFly Rock := T4 Rock ⟨T7, h_bird⟩
        exact T6 Rock ⟨T7, fly_rock⟩
</lean>

ANSWER: True
