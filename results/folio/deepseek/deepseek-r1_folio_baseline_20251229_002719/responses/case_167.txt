<lean>
axiom person : Type
axiom HeinrichSchmidt : person
axiom GermanPolitician : person → Prop
axiom InPrussianStateParliament : person → Prop
axiom InNaziReichstag : person → Prop

axiom fact1 : GermanPolitician HeinrichSchmidt
axiom fact2 : InPrussianStateParliament HeinrichSchmidt
axiom fact3 : InNaziReichstag HeinrichSchmidt

theorem some_german_in_both : ∃ x, GermanPolitician x ∧ InPrussianStateParliament x ∧ InNaziReichstag x :=
  by
    apply Exists.intro HeinrichSchmidt
    constructor
    · exact fact1
    · constructor
      · exact fact2
      · exact fact3
</lean>

ANSWER: True
