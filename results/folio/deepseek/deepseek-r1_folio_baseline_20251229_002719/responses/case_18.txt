<lean>
axiom Animal : Type
axiom Rockie : Animal

axiom Spotted : Animal → Prop
axiom Cute : Animal → Prop
axiom Rabbit : Animal → Prop
axiom Turtle : Animal → Prop
axiom Squirrel : Animal → Prop
axiom Skittish : Animal → Prop
axiom Calm : Animal → Prop

axiom all_spotted_rabbits_cute : ∀ x, Rabbit x ∧ Spotted x → Cute x
axiom some_turtle_spotted : ∃ x, Turtle x ∧ Spotted x
axiom only_rabbits_squirrels_spotted : ∀ x, Spotted x → Rabbit x ∨ Squirrel x
axiom skittish_not_calm : ∀ x, Skittish x → ¬ Calm x
axiom spotted_squirrels_skittish : ∀ x, Squirrel x ∧ Spotted x → Skittish x
axiom rockie_facts : Spotted Rockie ∧ Calm Rockie

theorem rockie_turtle_or_cute : Turtle Rockie ∨ Cute Rockie := by
  have spotted_rockie := rockie_facts.left
  have calm_rockie := rockie_facts.right
  have rockie_type := only_rabbits_squirrels_spotted Rockie spotted_rockie
  cases rockie_type with
  | inl rabbit_rockie =>
      apply Or.inr
      exact all_spotted_rabbits_cute Rockie ⟨rabbit_rockie, spotted_rockie⟩
  | inr squirrel_rockie =>
      have skittish_rockie := spotted_squirrels_skittish Rockie ⟨squirrel_rockie, spotted_rockie⟩
      have not_calm_rockie := skittish_not_calm Rockie skittish_rockie
      contradiction
</lean>

ANSWER: True
