The premises lead to a proof that Rock has wings, contradicting the statement "Rock does not have wings."

axiom Rock : obj
axiom Pet : obj → Prop

axiom Animal : obj → Prop

axiom Bird : obj → Prop
axiom Fly : obj → Prop

axiom A1 : ∀ x, Pet x → Mammal x → Animal x
axiom A2 : ∀ x, Monkey x → Mammal x
axiom A3 : ∀ x, Pet x → Monkey x ∨ Bird x
axiom A4 : ∀ x, Pet x → Bird x → Fly x
axiom A5 : ∀ x, Animal x → Pet x → Breathe x
axiom A6 : ∀ x, Pet x → Fly x → Wings x
axiom A7 : Pet Rock
axiom A8 : Fly Rock ∨ Bird Rock ∨ ¬ Breathe Rock

theorem wings_Rock : Wings Rock := by
  cases A8 with
  | inl h_fly => exact A6 Rock A7 h_fly
  | inr h_rest =>
    cases h_rest with
    | inl h_bird =>
      have h_fly : Fly Rock := A4 Rock A7 h_bird
      exact A6 Rock A7 h_fly
    | inr h_no_breath =>
      have h_animal : Monkey Rock ∨ Bird Rock := A3 Rock A7
      cases h_animal with
      | inl h_monkey =>
        have h_mammal : Mammal Rock := A2 Rock h_monkey
        have h_animal : Animal Rock := A1 Rock A7 h_mammal
        have h_breath : Breathe Rock := A5 Rock h_animal A7
        contradiction
      | inr h_bird =>
        have h_fly : Fly Rock := A4 Rock A7 h_bird
        exact A6 Rock A7 h_fly

theorem neg_conclusion_false : ¬(¬ Wings Rock) := fun h => h wings_Rock
