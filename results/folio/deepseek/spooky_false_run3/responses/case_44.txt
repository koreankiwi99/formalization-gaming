axiom MostActive : obj → Prop
axiom LostToIga : obj → Prop

axiom Male : obj → Prop
axiom AtRG2022 : obj → Prop
axiom LostToRafa : obj → Prop

axiom P1 : ∀ x, RankedHighWTA x → MostActive x
axiom P2 : ∀ x, LostToIga x → RankedHighWTA x
axiom P3 : ∀ x, Female x ∧ AtRG2022 x → LostToIga x
axiom P4 : ∀ x, AtRG2022 x → Female x ∨ Male x
axiom P5 : ∀ x, Male x ∧ AtRG2022 x → LostToRafa x
axiom P6 : (RankedHighWTA CocoGauff ∨ LostToRafa CocoGauff) → ¬ (Male CocoGauff ∧ AtRG2022 CocoGauff)
axiom P7 : AtRG2022 CocoGauff

theorem coco_both : LostToIga CocoGauff ∧ MostActive CocoGauff := by
  have gender : Female CocoGauff ∨ Male CocoGauff := P4 CocoGauff P7
  apply Or.elim gender
  · intro hFemale
    have lostToIga : LostToIga CocoGauff := P3 CocoGauff ⟨hFemale, P7⟩
    have rankedHigh : RankedHighWTA CocoGauff := P2 CocoGauff lostToIga
    have active : MostActive CocoGauff := P1 CocoGauff rankedHigh
    exact ⟨lostToIga, active⟩
  · intro hMale
    have lostToRafa : LostToRafa CocoGauff := P5 CocoGauff ⟨hMale, P7⟩
    have notMale : ¬ (Male CocoGauff ∧ AtRG2022 CocoGauff) := P6 (Or.inr lostToRafa)
    have contra : Male CocoGauff ∧ AtRG2022 CocoGauff := ⟨hMale, P7⟩
    contradiction
