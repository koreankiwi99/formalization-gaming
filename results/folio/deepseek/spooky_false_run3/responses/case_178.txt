<lean>
axiom obj : Type
axiom Horse : obj → Prop
axiom Human : obj → Prop
axiom HasHooves : obj → Prop

axiom all_horses_have_hooves : ∀ x : obj, Horse x → HasHooves x
axiom no_humans_have_hooves : ∀ x : obj, Human x → ¬HasHooves x

theorem disprove_some_humans_are_horses : ¬∃ x : obj, Human x ∧ Horse x := by
  intro h
  rcases h with ⟨x, hHuman, hHorse⟩
  have hooves : HasHooves x := all_horses_have_hooves x hHorse
  have no_hooves : ¬HasHooves x := no_humans_have_hooves x hHuman
  exact no_hooves hooves

-- Since we've proven the negation of the statement, the original statement is false
</lean>

ANSWER: False
