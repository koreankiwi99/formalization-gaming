<lean>
axiom obj : Type
axiom a1984 : obj
axiom AnalogMedia : obj → Prop
axiom DigitalMedia : obj → Prop
axiom PrintedText : obj → Prop
axiom StreamingService : obj → Prop
axiom HardcoverBook : obj → Prop

axiom P1 : ∀ x : obj, DigitalMedia x → ¬ AnalogMedia x
axiom P2 : ∀ x : obj, PrintedText x → AnalogMedia x
axiom P3 : ∀ x : obj, StreamingService x → DigitalMedia x
axiom P4 : ∀ x : obj, HardcoverBook x → PrintedText x
axiom P5 : StreamingService a1984 → HardcoverBook a1984

theorem not_streaming : ¬ StreamingService a1984 :=
  fun h : StreamingService a1984 =>
  have hb : HardcoverBook a1984 := P5 h
  have pt : PrintedText a1984 := P4 a1984 hb
  have analog : AnalogMedia a1984 := P2 a1984 pt
  have digital : DigitalMedia a1984 := P3 a1984 h
  have not_analog : ¬ AnalogMedia a1984 := P1 a1984 digital
  not_analog analog
</lean>

ANSWER: False
