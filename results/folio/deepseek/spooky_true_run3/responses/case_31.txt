<lean>
axiom Person : Type
axiom Djokovic : Person

-- Predicates
axiom IsAtMixer : Person → Prop
axiom IsGrandSlamChamp : Person → Prop
axiom IsOscarNominatedActor : Person → Prop
axiom IsProTennisPlayer : Person → Prop
axiom IsCelebrity : Person → Prop
axiom IsAthlete : Person → Prop
axiom IsWellPaid : Person → Prop
axiom IsFamous : Person → Prop
axiom LivesInTaxHaven : Person → Prop

-- Premises
axiom A1 : ∀ p : Person, IsAtMixer p → IsGrandSlamChamp p ∨ IsOscarNominatedActor p
axiom A2 : ∀ p : Person, IsAtMixer p → IsGrandSlamChamp p → IsProTennisPlayer p
axiom A3 : ∀ p : Person, IsAtMixer p → IsOscarNominatedActor p → IsCelebrity p
axiom A4 : ∀ p : Person, IsAtMixer p → IsProTennisPlayer p → IsAthlete p
axiom A5 : ∀ p : Person, IsAtMixer p → IsCelebrity p → IsWellPaid p
axiom A6 : ∀ p : Person, IsAtMixer p → IsAthlete p → IsFamous p
axiom A7 : ∀ p : Person, IsAtMixer p → IsWellPaid p → LivesInTaxHaven p
axiom A8 : IsAtMixer Djokovic
axiom A9 : (IsFamous Djokovic ∧ IsAthlete Djokovic) → IsWellPaid Djokovic

-- We need to prove: LivesInTaxHaven Djokovic
theorem djokovic_tax_haven : LivesInTaxHaven Djokovic := by
  -- Djokovic is at the mixer (A8) so by A1 he is either GS champ or Oscar-nominated
  have h_main := A1 Djokovic A8
  cases h_main with
  | inl gs_champ => 
      -- Case 1: Djokovic is a Grand Slam champion
      have tennis : IsProTennisPlayer Djokovic := A2 Djokovic A8 gs_champ
      have athlete : IsAthlete Djokovic := A4 Djokovic A8 tennis
      have famous : IsFamous Djokovic := A6 Djokovic A8 athlete
      have well_paid : IsWellPaid Djokovic := A9 ⟨famous, athlete⟩
      exact A7 Djokovic A8 well_paid
  | inr oscar => 
      -- Case 2: Djokovic is an Oscar-nominated actor
      have celeb : IsCelebrity Djokovic := A3 Djokovic A8 oscar
      have well_paid : IsWellPaid Djokovic := A5 Djokovic A8 celeb
      exact A7 Djokovic A8 well_paid

</lean>

ANSWER: True
