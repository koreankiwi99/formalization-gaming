3. **Conclusion**:
   - Rockie is cute, so "Rockie is a turtle or cute" is true.

axiom IsTurtle : Animal → Prop

axiom P2 : ∃ a : Animal, IsTurtle a ∧ Spotted a

theorem rockie_turtle_or_cute : IsTurtle Rockie ∨ Cute Rockie := by
