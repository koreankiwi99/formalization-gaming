<lean>
axiom obj : Type
axiom Marvin : obj
axiom Alien : obj → Prop
axiom Extraterrestrial : obj → Prop
axiom FromMars : obj → Prop
axiom Human : obj → Prop
axiom HighlyIntelligent : obj → Prop
axiom FromEarth : obj → Prop

-- Premises:
axiom P1 : ∀ x : obj, Alien x → Extraterrestrial x
axiom P2 : ∀ x : obj, FromMars x → Alien x
axiom P3 : ∀ x : obj, Extraterrestrial x → ¬ Human x
axiom P4 : ∀ x : obj, HighlyIntelligent x → FromEarth x → Human x
axiom P5 : HighlyIntelligent Marvin
axiom P6 : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬ FromEarth Marvin ∧ ¬ FromMars Marvin)
axiom P7 : ¬ FromEarth Marvin → Extraterrestrial Marvin

-- Goal: Marvin is neither human nor from Mars
theorem marvin_neither : ¬ Human Marvin ∧ ¬ FromMars Marvin := by
  cases P6 with
  | inl h1 =>  -- Case: FromEarth Marvin and FromMars Marvin
    have h_earth := h1.left
    have h_mars := h1.right
    have h_human : Human Marvin := P4 Marvin P5 h_earth
    have h_alien : Alien Marvin := P2 Marvin h_mars
    have h_et : Extraterrestrial Marvin := P1 Marvin h_alien
    have h_not_human : ¬ Human Marvin := P3 Marvin h_et
    contradiction
  | inr h2 =>  -- Case: ¬FromEarth Marvin and ¬FromMars Marvin
    have h_not_earth := h2.left
    have h_not_mars := h2.right
    have h_et : Extraterrestrial Marvin := P7 h_not_earth
    have h_not_human : ¬ Human Marvin := P3 Marvin h_et
    exact ⟨h_not_human, h_not_mars⟩
</lean>

ANSWER: True
