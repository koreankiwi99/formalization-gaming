axiom obj : Type
axiom IsStock : obj → Prop
axiom GrowthStock : obj → Prop
axiom MatureStock : obj → Prop
axiom BoughtForRapidAppreciation : obj → Prop
axiom SuitableForRetirement : obj → Prop

-- Growth stocks are a subset of all stocks
axiom A1 : ∀ x, GrowthStock x → IsStock x
-- Mature stocks are a subset of all stocks
axiom A2 : ∀ x, MatureStock x → IsStock x
-- All growth stocks are bought for rapid appreciation
axiom A3 : ∀ x, GrowthStock x → BoughtForRapidAppreciation x
-- Bought for rapid appreciation implies not suitable for retirement
axiom A4 : ∀ x, BoughtForRapidAppreciation x → ¬ SuitableForRetirement x
-- Some (existing) stocks are growth stocks
axiom A5 : ∃ x, IsStock x ∧ GrowthStock x
-- All mature stocks are suitable for retirement
axiom A6 : ∀ x, MatureStock x → SuitableForRetirement x

axiom KO : obj
axiom A7 : MatureStock KO -- KO is a mature stock

-- We attempt to prove KO is NOT a stock (to show that "KO is a stock" is false)
theorem ko_not_stock : ¬IsStock KO :=
  by
    -- Since KO is a mature stock, by A2 we get IsStock KO
    have h : IsStock KO := A2 KO A7
    -- This contradicts the goal ¬IsStock KO, thus we must abort
    sorry -- Cannot complete proof

-- We show the negation cannot be proven
</lean>
