<lean>
axiom obj : Type
axiom Yuri : obj
axiom BasketballPlayer : obj → Prop
axiom American : obj → Prop
axiom JumpShots : obj → Prop
axiom Leap : obj → Prop
axiom ActivateMuscles : obj → Prop

-- Premises:
-- Some professional basketball players are not American nationals
axiom T1 : ∃ x, BasketballPlayer x ∧ ¬American x
-- All professional basketball players can do jump shots
axiom T2 : ∀ x, BasketballPlayer x → JumpShots x
-- If someone can do jump shots, they leap straight into air
axiom T3 : ∀ x, JumpShots x → Leap x
-- If someone leaps straight into air, they activate leg muscles
axiom T4 : ∀ x, Leap x → ActivateMuscles x
-- Yuri does not activate leg muscles
axiom T5 : ¬ActivateMuscles Yuri

theorem disproof : ¬(BasketballPlayer Yuri ∧ American Yuri) :=
  fun ⟨hBP, _⟩ => 
    have jump : JumpShots Yuri := T2 Yuri hBP
    have leap : Leap Yuri := T3 Yuri jump
    have activate : ActivateMuscles Yuri := T4 Yuri leap
    T5 activate
</lean>

ANSWER: False
