<lean>
axiom obj : Type
axiom Horse : obj → Prop
axiom Human : obj → Prop
axiom Hooves : obj → Prop

axiom all_horses_have_hooves : ∀ x : obj, Horse x → Hooves x
axiom no_humans_have_hooves : ∀ x : obj, Human x → ¬ Hooves x

theorem humans_not_horses : ¬∃ x : obj, Human x ∧ Horse x := by
  intro ⟨x, hx_human, hx_horse⟩
  have hooves_x : Hooves x := all_horses_have_hooves x hx_horse
  exact no_humans_have_hooves x hx_human hooves_x
</lean>

ANSWER: False
