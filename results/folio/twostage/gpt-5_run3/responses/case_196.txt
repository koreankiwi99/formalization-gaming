    intro h
    have hdisj := P7 h
    cases hdisj with
    | inl hTerr =>
      have hlegs : HasLegs KiKi := P5 KiKi P6 hTerr
      exact h.right hlegs
    | inr hlegs =>
      exact h.right hlegs

    by_contra hNoJump

      intro hNoLegs
