<lean>
axiom obj : Type

axiom AtMixer : obj → Prop
axiom GrandSlamChampion : obj → Prop
axiom OscarNominatedActor : obj → Prop
axiom ProfessionalTennisPlayer : obj → Prop
axiom Celebrity : obj → Prop
axiom Athlete : obj → Prop
axiom WellPaid : obj → Prop
axiom Famous : obj → Prop
axiom LivesInTaxHaven : obj → Prop

axiom Djokovic : obj

-- Premises
axiom P1 : ∀ x : obj, AtMixer x → (GrandSlamChampion x ∨ OscarNominatedActor x)
axiom P2 : ∀ x : obj, AtMixer x → GrandSlamChampion x → ProfessionalTennisPlayer x
axiom P3 : ∀ x : obj, AtMixer x → OscarNominatedActor x → Celebrity x
axiom P4 : ∀ x : obj, AtMixer x → ProfessionalTennisPlayer x → Athlete x
axiom P5 : ∀ x : obj, AtMixer x → Celebrity x → WellPaid x
axiom P6 : ∀ x : obj, AtMixer x → Athlete x → Famous x
axiom P7 : ∀ x : obj, AtMixer x → WellPaid x → LivesInTaxHaven x
axiom P8 : AtMixer Djokovic
axiom P9 : (Famous Djokovic ∧ Athlete Djokovic) → WellPaid Djokovic

-- Conclusion
theorem goal : LivesInTaxHaven Djokovic :=
by
  have hdisj : GrandSlamChampion Djokovic ∨ OscarNominatedActor Djokovic := P1 Djokovic P8
  have hwp : WellPaid Djokovic :=
    Or.elim hdisj
      (fun hGS =>
        have hPTP : ProfessionalTennisPlayer Djokovic := P2 Djokovic P8 hGS
        have hAth : Athlete Djokovic := P4 Djokovic P8 hPTP
        have hFam : Famous Djokovic := P6 Djokovic P8 hAth
        show WellPaid Djokovic from P9 (And.intro hFam hAth)
      )
      (fun hOscar =>
        have hCel : Celebrity Djokovic := P3 Djokovic P8 hOscar
        show WellPaid Djokovic from P5 Djokovic P8 hCel
      )
  exact P7 Djokovic P8 hwp
</lean>
