axiom Aurochs : obj → Prop
axiom BredBack : obj → Prop

axiom ToBeBredBack : obj → Prop
axiom Resembles : obj → obj → Prop

axiom P1 : ∀ x : obj, BredBack x → ArtificiallySelected x
axiom P1b : ∀ x : obj, BredBack x → Domestic x

axiom P2 : ∃ x : obj, HeckCattle x ∧ BredBack x ∧ (∃ y : obj, Aurochs y ∧ Resembles x y)

-- Conclusion: Some Heck cattle are artificially selected.
theorem goal : ∃ x : obj, HeckCattle x ∧ ArtificiallySelected x :=
by
  obtain ⟨x, hx⟩ := P2
  have hHC : HeckCattle x := And.left hx
  have hBB : BredBack x := And.left (And.right hx)
  have hAS : ArtificiallySelected x := P1 x hBB
  exact ⟨x, And.intro hHC hAS⟩
