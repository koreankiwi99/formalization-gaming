--- Iteration 1 ---

<lean>
axiom obj : Type

-- Objects
axiom Rock : obj

-- Predicates
axiom Pet : obj → Prop            -- Peter's pet
axiom Mammal : obj → Prop
axiom Animal : obj → Prop
axiom Monkey : obj → Prop
axiom Bird : obj → Prop
axiom Fly : obj → Prop
axiom Breathe : obj → Prop
axiom HasWings : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, Pet x → Mammal x → Animal x
axiom P2 : ∀ x : obj, Monkey x → Mammal x
axiom P3 : ∀ x : obj, Pet x → (Monkey x ∨ Bird x)
axiom P4 : ∀ x : obj, Pet x → Bird x → Fly x
axiom P5 : ∀ x : obj, Pet x → Animal x → Breathe x
axiom P6 : ∀ x : obj, Pet x → Fly x → HasWings x
axiom P7 : Pet Rock
axiom P8 : Fly Rock ∨ Bird Rock ∨ ¬ Breathe Rock

-- Conclusion
theorem goal : HasWings Rock := by
  have hPet : Pet Rock := P7
  cases P8 with
  | inl hFly =>
      exact P6 Rock hPet hFly
  | inr h =>
      cases h with
      | inl hBird =>
          have hFly : Fly Rock := P4 Rock hPet hBird
          exact P6 Rock hPet hFly
      | inr hNotBreathe =>
          have hMB : Monkey Rock ∨ Bird Rock := P3 Rock hPet
          cases hMB with
          | inl hMonkey =>
              have hMammal : Mammal Rock := P2 Rock hMonkey
              have hAnimal : Animal Rock := P1 Rock hPet hMammal
              have hBreathe : Breathe Rock := P5 Rock hPet hAnimal
              have hFalse : False := hNotBreathe hBreathe
              exact False.elim hFalse
          | inr hBird2 =>
              have hFly : Fly Rock := P4 Rock hPet hBird2
              exact P6 Rock hPet hFly
</lean>
ANSWER: True
