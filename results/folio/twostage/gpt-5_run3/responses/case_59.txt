axiom InUS : obj → Prop

axiom Montana : obj

axiom Missoula : obj
axiom WhiteSulphurSprings : obj
axiom StPierre : obj
axiom Bristol : obj
axiom Texarkana : obj
axiom Texhoma : obj
axiom UnionCity : obj

-- Billings is a city in the state of Montana in U.S.
axiom A1 : City Billings
axiom A2 : State Montana
axiom A3 : In Billings Montana
axiom A4 : InUS Montana

-- The state of Montana includes the cities of Butte, Helena, and Missoula.
axiom A5 : City Butte

axiom A7 : City Missoula
axiom A8 : In Butte Montana
axiom A9 : In Helena Montana
axiom A10 : In Missoula Montana

-- White Sulphur Springs and Butte are cities in the same state in U.S.
axiom A11 : City WhiteSulphurSprings
axiom A12 : SameStateUS WhiteSulphurSprings Butte

-- The city of St Pierre is not in the state of Montana.
axiom A13 : City StPierre
axiom A14 : ¬ In StPierre Montana

-- Any city in Butte is not in St Pierre.
axiom A15 : ∀ x : obj, City x → In x Butte → ¬ In x StPierre

-- A city can only be in one state in U.S. except for Bristol, Texarkana, Texhoma and Union City.
axiom A16 : City Bristol
axiom A17 : City Texarkana
axiom A18 : City Texhoma
axiom A19 : City UnionCity
axiom A20 : ∀ c s1 s2 : obj,
  City c → State s1 → State s2 → InUS s1 → InUS s2 → In c s1 → In c s2 →
  c = Bristol ∨ c = Texarkana ∨ c = Texhoma ∨ c = UnionCity ∨ s1 = s2

-- Conclusion: Montana is home to the city of Missoula.
theorem goal : In Missoula Montana := sorry
</lean>
