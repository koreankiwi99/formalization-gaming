<lean>
axiom obj : Type

axiom DriedThaiChilies : obj

axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom FromBBM : obj → Prop

axiom P1 : ∀ x : obj, BakedSweet x → ¬ Spicy x
axiom P2 : ∀ x : obj, Cupcake x → BakedSweet x
axiom P3 : ∀ x : obj, MalaHotpot x → Spicy x
axiom P4 : ∀ x : obj, FromBBM x → Cupcake x
axiom P5 : Spicy DriedThaiChilies ∨ MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies

theorem goal : ¬ FromBBM DriedThaiChilies := by
  intro hFrom
  have hCup : Cupcake DriedThaiChilies := P4 DriedThaiChilies hFrom
  have hBaked : BakedSweet DriedThaiChilies := P2 DriedThaiChilies hCup
  have hNotSpicy : ¬ Spicy DriedThaiChilies := P1 DriedThaiChilies hBaked
  have h := P5
  cases h with
  | inl hSpicy =>
      exact hNotSpicy hSpicy
  | inr h' =>
      cases h' with
      | inl hMala =>
          have hs : Spicy DriedThaiChilies := P3 DriedThaiChilies hMala
          exact hNotSpicy hs
      | inr hNotBaked =>
          exact hNotBaked hBaked
</lean>
