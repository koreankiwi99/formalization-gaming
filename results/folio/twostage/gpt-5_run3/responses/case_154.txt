--- Iteration 1 ---

<lean>
axiom obj : Type
axiom mixture : obj

axiom Hydrocarbon : obj → Prop
axiom OrganicCompound : obj → Prop
axiom ChemicalCompound : obj → Prop
axiom ContainsCarbon : obj → Prop
axiom ContainsOnlyOneElement : obj → Prop
axiom Alkane : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, Hydrocarbon x → OrganicCompound x
axiom P2 : ∀ x : obj, Alkane x → Hydrocarbon x
axiom P3 : ∀ x : obj, OrganicCompound x → ChemicalCompound x
axiom P4 : ∀ x : obj, OrganicCompound x → ContainsCarbon x
axiom P5 : ∀ x : obj, ChemicalCompound x → ¬ ContainsOnlyOneElement x
axiom P6 : (ChemicalCompound mixture ∧ ContainsOnlyOneElement mixture) ∨ (¬ ChemicalCompound mixture ∧ ¬ ContainsOnlyOneElement mixture)

-- Conclusion to formalize
theorem goal :
  (ContainsOnlyOneElement mixture ∨ ContainsCarbon mixture) →
  (¬ ChemicalCompound mixture ∧ ¬ Alkane mixture) := by
  intro _
  have hNotChem : ¬ ChemicalCompound mixture := by
    cases P6 with
    | inl hBoth =>
        have hContra : ¬ ContainsOnlyOneElement mixture := P5 mixture hBoth.left
        exact False.elim (hContra hBoth.right)
    | inr hNeither =>
        exact hNeither.left
  have hNotAlkane : ¬ Alkane mixture := by
    intro hAlk
    have hHydro : Hydrocarbon mixture := P2 mixture hAlk
    have hOrg : OrganicCompound mixture := P1 mixture hHydro
    have hChem : ChemicalCompound mixture := P3 mixture hOrg
    exact hNotChem hChem
  exact And.intro hNotChem hNotAlkane
</lean>
ANSWER: True
