--- Iteration 1 ---

<lean>
axiom obj : Type

axiom Alien : obj → Prop
axiom Extraterrestrial : obj → Prop
axiom Human : obj → Prop
axiom FromEarth : obj → Prop
axiom FromMars : obj → Prop
axiom HighlyIntelligent : obj → Prop

axiom Marvin : obj

-- Premises
axiom P1 : ∀ x : obj, Alien x → Extraterrestrial x
axiom P2 : ∀ x : obj, FromMars x → Alien x
axiom P3 : ∀ x : obj, Extraterrestrial x → ¬ Human x
axiom P4 : ∀ x : obj, (HighlyIntelligent x ∧ FromEarth x) → Human x
axiom P5 : HighlyIntelligent Marvin
axiom P6 : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬ FromEarth Marvin ∧ ¬ FromMars Marvin)
axiom P7 : ¬ FromEarth Marvin → Extraterrestrial Marvin

-- Conclusion to formalize
theorem goal : ¬ Human Marvin ∧ ¬ FromMars Marvin := by
  cases P6 with
  | inl h =>
      have hE : FromEarth Marvin := And.left h
      have hM : FromMars Marvin := And.right h
      have hAlien : Alien Marvin := P2 Marvin hM
      have ex : Extraterrestrial Marvin := P1 Marvin hAlien
      have nh : ¬ Human Marvin := P3 Marvin ex
      have human : Human Marvin := P4 Marvin (And.intro P5 hE)
      have notMars : ¬ FromMars Marvin := by
        intro fm
        exact nh human
      exact And.intro nh notMars
  | inr h =>
      have nE : ¬ FromEarth Marvin := And.left h
      have nM : ¬ FromMars Marvin := And.right h
      have ex : Extraterrestrial Marvin := P7 nE
      have nh : ¬ Human Marvin := P3 Marvin ex
      exact And.intro nh nM
</lean>
ANSWER: True
