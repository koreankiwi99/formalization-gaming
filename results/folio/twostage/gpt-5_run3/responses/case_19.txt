--- Iteration 1 ---

<lean>
axiom obj : Type

axiom Rabbit : obj → Prop
axiom Turtle : obj → Prop
axiom Squirrel : obj → Prop
axiom SpottedNearCampus : obj → Prop
axiom Cute : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop

axiom Rockie : obj

-- Premises
axiom P1 : ∀ x : obj, Rabbit x → SpottedNearCampus x → Cute x
axiom P2 : ∃ x : obj, Turtle x ∧ SpottedNearCampus x
axiom P3 : ∀ x : obj, SpottedNearCampus x → (Rabbit x ∨ Squirrel x)
axiom P4 : ∀ x : obj, Skittish x → ¬ Calm x
axiom P5 : ∀ x : obj, Squirrel x → SpottedNearCampus x → Skittish x
axiom P6 : SpottedNearCampus Rockie ∧ Calm Rockie

-- Conclusion to formalize
theorem goal : ¬ (Turtle Rockie ∧ Squirrel Rockie) → (Cute Rockie ∨ Skittish Rockie) :=
by
  intro _
  have hspot : SpottedNearCampus Rockie := And.left P6
  have hRS : Rabbit Rockie ∨ Squirrel Rockie := P3 Rockie hspot
  cases hRS with
  | inl hrab =>
      exact Or.inl (P1 Rockie hrab hspot)
  | inr hsq =>
      exact Or.inr (P5 Rockie hsq hspot)
</lean>
ANSWER: True
