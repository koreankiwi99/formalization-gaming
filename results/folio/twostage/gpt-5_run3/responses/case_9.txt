axiom Eastern : obj → Prop

axiom RioGrande : obj → Prop
axiom Ocellated : obj → Prop

-- Premise: Every wild turkey is one of exactly these six types (exhaustive disjunction)
axiom SixTypes :
  ∀ x : obj, WildTurkey x → (Eastern x ∨ Osceola x ∨ Goulds x ∨ Merriams x ∨ RioGrande x ∨ Ocellated x)

-- Premises about Tom
axiom NotEasternTom : ¬ Eastern Tom

theorem goal : Ocellated Tom := by
  have h := SixTypes Tom TomIsWildTurkey
  cases h with
  | inl hEastern =>

      cases h1 with
      | inl hOsceola =>
          exact False.elim (NotOsceolaTom hOsceola)
      | inr h2 =>
          cases h2 with
          | inl hGoulds =>
              exact False.elim (NotGouldsTom hGoulds)
          | inr h3 =>
              cases h3 with
              | inl hMerriams =>
                  exact False.elim (NotMerriamsTom hMerriams)
              | inr h4 =>

                  | inl hRio =>
