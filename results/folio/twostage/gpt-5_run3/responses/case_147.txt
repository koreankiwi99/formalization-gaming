<lean>
axiom Person : Type
axiom Ekaterina : Person
axiom Vladimir : Person

axiom CanRegisterUS : Person → Prop
axiom CanParticipate2024USPresElection : Person → Prop
axiom USCitizen : Person → Prop
axiom TaiwaneseCitizen : Person → Prop
axiom RussianFederationOfficial : Person → Prop
axiom ManagerAtGazprom : Person → Prop

-- Premises
axiom P1 : ∀ x : Person, CanRegisterUS x → CanParticipate2024USPresElection x
axiom P2 : ∀ x : Person, USCitizen x → CanRegisterUS x
axiom P3 : ∀ x : Person, USCitizen x ∨ TaiwaneseCitizen x
axiom P4 : ∀ x : Person, RussianFederationOfficial x → ¬ TaiwaneseCitizen x
axiom P5 : ¬ TaiwaneseCitizen Vladimir ∧ ¬ ManagerAtGazprom Vladimir
axiom P6 : CanRegisterUS Ekaterina ∨ RussianFederationOfficial Ekaterina

-- Conclusion to formalize
theorem goal : CanParticipate2024USPresElection Ekaterina ∨ ManagerAtGazprom Vladimir :=
by
  cases P6 with
  | inl hreg =>
      exact Or.inl (P1 Ekaterina hreg)
  | inr hrfo =>
      have nt : ¬ TaiwaneseCitizen Ekaterina := P4 Ekaterina hrfo
      have dis : USCitizen Ekaterina ∨ TaiwaneseCitizen Ekaterina := P3 Ekaterina
      have hUSC : USCitizen Ekaterina :=
        by
          cases dis with
          | inl hu => exact hu
          | inr ht => exact False.elim (nt ht)
      have hreg : CanRegisterUS Ekaterina := P2 Ekaterina hUSC
      exact Or.inl (P1 Ekaterina hreg)
</lean>
