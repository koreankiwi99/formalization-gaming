<lean>
axiom obj : Type

axiom Peter : obj

axiom SpillsFoodOnClothing : obj → Prop
axiom NotablyTidy : obj → Prop
axiom ClumsyFoodie : obj → Prop
axiom GoesOutFrequentlyForNewFood : obj → Prop
axiom Cleanly : obj → Prop
axiom ValuesOrderAndSpotlessness : obj → Prop
axiom FamilyPrioritizesOrderAndSpotlessness : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, SpillsFoodOnClothing x → ¬ NotablyTidy x
axiom P2 : ∀ x : obj, (ClumsyFoodie x ∧ GoesOutFrequentlyForNewFood x) → SpillsFoodOnClothing x
axiom P3 : ∀ x : obj, Cleanly x → NotablyTidy x
axiom P4 : ∀ x : obj, ValuesOrderAndSpotlessness x → Cleanly x
axiom P5 : ∀ x : obj, FamilyPrioritizesOrderAndSpotlessness x → ValuesOrderAndSpotlessness x
axiom P6 : (SpillsFoodOnClothing Peter ∧ Cleanly Peter) ∨ (¬ SpillsFoodOnClothing Peter ∧ ¬ Cleanly Peter)

-- Conclusion
theorem goal :
  (ClumsyFoodie Peter ∧ GoesOutFrequentlyForNewFood Peter ∧ FamilyPrioritizesOrderAndSpotlessness Peter)
  ∨ (¬ (ClumsyFoodie Peter ∧ GoesOutFrequentlyForNewFood Peter) ∧ ¬ FamilyPrioritizesOrderAndSpotlessness Peter)
:= by
  cases P6 with
  | inl h =>
      have hS : SpillsFoodOnClothing Peter := And.left h
      have hC : Cleanly Peter := And.right h
      have hNT : NotablyTidy Peter := P3 Peter hC
      have hNNT : ¬ NotablyTidy Peter := P1 Peter hS
      have contra : False := hNNT hNT
      exact False.elim contra
  | inr h =>
      have hNotSpills : ¬ SpillsFoodOnClothing Peter := And.left h
      have hNotClean : ¬ Cleanly Peter := And.right h
      have hNotFamily : ¬ FamilyPrioritizesOrderAndSpotlessness Peter :=
        fun hFam =>
          have hVals : ValuesOrderAndSpotlessness Peter := P5 Peter hFam
          have hClean : Cleanly Peter := P4 Peter hVals
          show False from hNotClean hClean
      have hNotCG : ¬ (ClumsyFoodie Peter ∧ GoesOutFrequentlyForNewFood Peter) :=
        fun hCG =>
          have hSpills : SpillsFoodOnClothing Peter := P2 Peter hCG
          show False from hNotSpills hSpills
      exact Or.inr (And.intro hNotCG hNotFamily)
</lean>
