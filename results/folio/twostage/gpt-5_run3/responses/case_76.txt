<lean>
axiom obj : Type

axiom AtMarysSchool : obj → Prop
axiom WorksInStudentJobsOnCampus : obj → Prop
axiom NeedsToEarnMoneyToHelpPayForCollegeTuition : obj → Prop
axiom OrdersTakeoutFrequentlyInCollege : obj → Prop
axiom EnjoysDiningHallMealsAndRecipes : obj → Prop
axiom NotPickyEater : obj → Prop
axiom SpendsALotOfTheirTimeEatingAndCatchingUpWithFriendsInTheCampusDiningHalls : obj → Prop

axiom Hannah : obj

-- Premises
axiom P1 : ∀ x : obj, AtMarysSchool x → (WorksInStudentJobsOnCampus x → NeedsToEarnMoneyToHelpPayForCollegeTuition x)
axiom P2 : ∀ x : obj, AtMarysSchool x → (OrdersTakeoutFrequentlyInCollege x → WorksInStudentJobsOnCampus x)
axiom P3 : ∀ x : obj, AtMarysSchool x → (OrdersTakeoutFrequentlyInCollege x ∨ EnjoysDiningHallMealsAndRecipes x)
axiom P4 : ∀ x : obj, AtMarysSchool x → (EnjoysDiningHallMealsAndRecipes x → NotPickyEater x)
axiom P5 : ∀ x : obj, AtMarysSchool x → (EnjoysDiningHallMealsAndRecipes x → SpendsALotOfTheirTimeEatingAndCatchingUpWithFriendsInTheCampusDiningHalls x)
axiom P6 : AtMarysSchool Hannah
axiom P7 : WorksInStudentJobsOnCampus Hannah ∧ (NeedsToEarnMoneyToHelpPayForCollegeTuition Hannah → (NotPickyEater Hannah ∧ ¬ NeedsToEarnMoneyToHelpPayForCollegeTuition Hannah))

-- Conclusion
theorem goal : AtMarysSchool Hannah ∧ NotPickyEater Hannah ∧ SpendsALotOfTheirTimeEatingAndCatchingUpWithFriendsInTheCampusDiningHalls Hannah := by
  have hAS : AtMarysSchool Hannah := P6
  have hW : WorksInStudentJobsOnCampus Hannah := And.left P7
  have hNeeds : NeedsToEarnMoneyToHelpPayForCollegeTuition Hannah := (P1 Hannah hAS) hW
  have hNP_and_notN : NotPickyEater Hannah ∧ ¬ NeedsToEarnMoneyToHelpPayForCollegeTuition Hannah := (And.right P7) hNeeds
  have hNP : NotPickyEater Hannah := And.left hNP_and_notN
  have hNotNeeds : ¬ NeedsToEarnMoneyToHelpPayForCollegeTuition Hannah := And.right hNP_and_notN
  have hFalse : False := hNotNeeds hNeeds
  have hSpends : SpendsALotOfTheirTimeEatingAndCatchingUpWithFriendsInTheCampusDiningHalls Hannah := False.elim hFalse
  exact And.intro hAS (And.intro hNP hSpends)
</lean>
