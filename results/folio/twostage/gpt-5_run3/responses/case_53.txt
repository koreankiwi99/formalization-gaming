<lean>
axiom Lang : Type

axiom French : Lang
axiom Spanish : Lang
axiom German : Lang
axiom Basque : Lang

axiom Romance : Lang → Prop
axiom IndoEuropean : Lang → Prop
axiom Related : Lang → Lang → Prop
axiom SameFamily : Lang → Lang → Prop

-- All Romance languages are Indo-European languages.
axiom A1 : ∀ x : Lang, Romance x → IndoEuropean x

-- Romance languages are a language family (members share the same family).
axiom A2 : ∀ x y : Lang, Romance x → Romance y → SameFamily x y

-- All languages within a language family are related to each other.
axiom A3 : ∀ x y : Lang, SameFamily x y → Related x y

-- French and Spanish are both Romance languages.
axiom A4 : Romance French
axiom A5 : Romance Spanish

-- German is related to Spanish.
axiom A6 : Related German Spanish

-- Basque is not related to any other language.
axiom A7 : ∀ x : Lang, x ≠ Basque → (¬ Related Basque x ∧ ¬ Related x Basque)

-- Conclusion: French is an Indo-European language.
theorem goal : IndoEuropean French := sorry
</lean>
