-- Predicates and relations
axiom Man : obj → Prop
axiom InMClass : obj → Prop        -- membership in Michael's class
axiom Taller : obj → obj → Prop    -- Taller x y: x is taller than y
axiom CanBlock : obj → obj → Prop  -- CanBlock x y: x can block y's shooting
axiom JumpsWhenShooting : obj → Prop
axiom GreatShooter : obj → Prop

-- Premises
-- If a man is taller than another man, the taller one can block the other's shooting.
axiom P1 : ∀ x y : obj, Man x → Man y → Taller x y → CanBlock x y

-- Michael is a man who is taller than everyone else in his class.
axiom P2a : Man Michael
axiom P2b : ∀ x : obj, InMClass x → Taller Michael x

-- If person x is taller than person y, and person y is taller than person z, then x is taller than z.
axiom P3 : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom P4a : Man Peter
axiom P4b : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom P5 : ∀ p : obj, ¬ JumpsWhenShooting p → CanBlock Michael p

-- Michael cannot block Windy's shooting.
axiom P6 : ¬ CanBlock Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom P7 : ∀ x : obj, JumpsWhenShooting x → GreatShooter x

-- Conclusion to formalize: Peter is shorter than a man in Michael's class.
-- Interpreted as: there exists a man m in Michael's class such that m is taller than Peter.
theorem goal : ∃ m : obj, Man m ∧ InMClass m ∧ Taller m Peter := sorry
