--- Iteration 1 ---

<lean>
axiom obj : Type

axiom PicurisMountains : obj
axiom NewMexico : obj
axiom Texas : obj
axiom JuanDeOnate : obj
axiom HardingPegmatiteMine : obj

axiom MountainRange : obj → Prop
axiom Mine : obj → Prop
axiom In : obj → obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Visited : obj → obj → Prop
axiom Donated : obj → Prop

-- The Picuris Mountains are a mountain range in New Mexico or Texas.
axiom P1 : MountainRange PicurisMountains
axiom P2 : In PicurisMountains NewMexico ∨ In PicurisMountains Texas

-- Juan de Onate visited the Picuris Mountains.
axiom P3 : Visited JuanDeOnate PicurisMountains

-- The Harding Pegmatite Mine, located in the Picuris Mountains, was donated.
axiom P4 : Mine HardingPegmatiteMine
axiom P5 : LocatedIn HardingPegmatiteMine PicurisMountains
axiom P6 : Donated HardingPegmatiteMine

-- There are no mountain ranges in Texas that have mines that have been donated.
axiom P7 : ∀ r : obj, MountainRange r → In r Texas → ¬ ∃ m : obj, Mine m ∧ LocatedIn m r ∧ Donated m

-- Conclusion to formalize: Juan de Onate visited a mountain range in Texas.
theorem goal : ∃ r : obj, MountainRange r ∧ In r Texas ∧ Visited JuanDeOnate r := sorry
</lean>
