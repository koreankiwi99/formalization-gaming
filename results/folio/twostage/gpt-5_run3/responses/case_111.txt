by
  cases P2 with
  | inl hNM =>
      exact Exists.intro PicurisMountains (And.intro P1 (And.intro hNM P3))

      have hNot : ¬ ∃ m : obj, Mine m ∧ In m PicurisMountains ∧ Donated m :=
        P7 PicurisMountains (And.intro P1 hTX)
