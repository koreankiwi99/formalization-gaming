<lean>
axiom obj : Type

axiom Hannah : obj

axiom AtMarysSchool : obj → Prop
axiom WorksStudentJobsOnCampus : obj → Prop
axiom NeedsToEarnMoneyForTuition : obj → Prop
axiom OrdersTakeoutFrequentlyInCollege : obj → Prop
axiom EnjoysDiningHallMealsAndRecipes : obj → Prop
axiom PickyEater : obj → Prop
axiom SpendsLotsOfTimeEatingAndCatchingUpWithFriendsInCampusDiningHalls : obj → Prop

axiom A1 : ∀ x : obj, AtMarysSchool x ∧ WorksStudentJobsOnCampus x → NeedsToEarnMoneyForTuition x
axiom A2 : ∀ x : obj, AtMarysSchool x ∧ OrdersTakeoutFrequentlyInCollege x → WorksStudentJobsOnCampus x
axiom A3 : ∀ x : obj, AtMarysSchool x → (OrdersTakeoutFrequentlyInCollege x ∨ EnjoysDiningHallMealsAndRecipes x)
axiom A4 : ∀ x : obj, AtMarysSchool x ∧ EnjoysDiningHallMealsAndRecipes x → ¬ PickyEater x
axiom A5 : ∀ x : obj, AtMarysSchool x ∧ EnjoysDiningHallMealsAndRecipes x → SpendsLotsOfTimeEatingAndCatchingUpWithFriendsInCampusDiningHalls x
axiom A6 : AtMarysSchool Hannah
axiom A7 : WorksStudentJobsOnCampus Hannah ∧ (NeedsToEarnMoneyForTuition Hannah → (¬ PickyEater Hannah ∧ ¬ NeedsToEarnMoneyForTuition Hannah))

theorem goal :
  AtMarysSchool Hannah ∧
  (¬ PickyEater Hannah ∨ (PickyEater Hannah → SpendsLotsOfTimeEatingAndCatchingUpWithFriendsInCampusDiningHalls Hannah)) := by
  have hAt : AtMarysSchool Hannah := A6
  have hWorks : WorksStudentJobsOnCampus Hannah := And.left A7
  have hNeeds : NeedsToEarnMoneyForTuition Hannah := A1 Hannah (And.intro hAt hWorks)
  have hImp : NeedsToEarnMoneyForTuition Hannah → (¬ PickyEater Hannah ∧ ¬ NeedsToEarnMoneyForTuition Hannah) := And.right A7
  have hNotPicky : ¬ PickyEater Hannah := And.left (hImp hNeeds)
  exact And.intro hAt (Or.inl hNotPicky)
</lean>
