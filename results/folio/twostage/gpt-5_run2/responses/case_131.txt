<lean>
axiom obj : Type
axiom Mammal : obj → Prop
axiom HasTeeth : obj → Prop
axiom Platypus : obj → Prop
axiom Human : obj → Prop

axiom SomeMammalsHaveTeeth : ∃ x : obj, Mammal x ∧ HasTeeth x
axiom PlatypusesHaveNoTeeth : ∀ x : obj, Platypus x → ¬ HasTeeth x
axiom PlatypusesAreMammals : ∀ x : obj, Platypus x → Mammal x
axiom HumansHaveTeeth : ∀ x : obj, Human x → HasTeeth x

theorem goal : ∀ x : obj, Platypus x → (Mammal x ∧ ¬ HasTeeth x) :=
by
  intro x
  intro hx
  have hm : Mammal x := PlatypusesAreMammals x hx
  have hnt : ¬ HasTeeth x := PlatypusesHaveNoTeeth x hx
  exact And.intro hm hnt
</lean>
