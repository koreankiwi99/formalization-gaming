axiom P2a : InState Butte Montana
axiom P2b : InState Helena Montana
axiom P2c : InState Missoula Montana

-- White Sulphur Springs and Butte are cities in the same state in U.S.
axiom P3 : ∃ s : State, InState WhiteSulphurSprings s ∧ InState Butte s ∧ InCountry s UnitedStates

-- The city of St Pierre is not in the state of Montana.
axiom P4 : ¬ InState StPierre Montana

-- Any city in Butte is not in St Pierre.
axiom P5 : ∀ c : City, InCity c Butte → ¬ InCity c StPierre

-- A city can only be in one state in U.S. except for Bristol, Texarkana, Texhoma and Union City.
axiom P6 : ∀ c : City, ¬ Exceptional c → ∀ s1 s2 : State,
  InState c s1 → InState c s2 → InCountry s1 UnitedStates → InCountry s2 UnitedStates → s1 = s2

-- Conclusion to formalize: Butte and St Pierre are in the same state.
theorem goal : ∃ s : State, InState Butte s ∧ InState StPierre s := sorry
