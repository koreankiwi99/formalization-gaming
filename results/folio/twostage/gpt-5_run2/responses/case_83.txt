axiom A1 : ∀ x : obj, ListedInYelp x → ¬ ManyNegativeReviews x

axiom A2 : ∀ x : obj, RatingGT4 x → ListedInYelp x

-- Some restaurants that do not provide take-out service receive many negative reviews.
axiom A3 : ∃ x : obj, ¬ TakeOutService x ∧ ManyNegativeReviews x

-- All restaurants that are popular among local residents have ratings greater than four.
axiom A4 : ∀ x : obj, PopularLocal x → RatingGT4 x

-- The Hamden Plaza Subway store has a rating greater than four, or it is popular among local residents.
axiom A5 : RatingGT4 HamdenPlazaSubway ∨ PopularLocal HamdenPlazaSubway

-- Conclusion to formalize:
-- If the Hamden Plaza Subway store provides take-out service and receives many negative reviews,
-- then its rating is greater than 4 and it does not provide take-out service.
theorem goal :
  (TakeOutService HamdenPlazaSubway ∧ ManyNegativeReviews HamdenPlazaSubway) →
  (RatingGT4 HamdenPlazaSubway ∧ ¬ TakeOutService HamdenPlazaSubway) := by
  intro h
  have mnr : ManyNegativeReviews HamdenPlazaSubway := And.right h
  have r : RatingGT4 HamdenPlazaSubway :=
    Or.elim A5 (fun h1 => h1) (fun h2 => A4 _ h2)
  have listed : ListedInYelp HamdenPlazaSubway := A2 _ r
  have nmnr : ¬ ManyNegativeReviews HamdenPlazaSubway := A1 _ listed
  have f : False := nmnr mnr
  have notTake : ¬ TakeOutService HamdenPlazaSubway := fun _ => f
  exact And.intro r notTake
