--- Iteration 1 ---

<lean>
axiom obj : Type

axiom BrownSwissCattle : obj → Prop
axiom Cow : obj → Prop
axiom Pet : obj → Prop
axiom DomesticatedAnimal : obj → Prop
axiom Alligator : obj → Prop

axiom Ted : obj

-- Premises
axiom P1 : ∀ x : obj, BrownSwissCattle x → Cow x
axiom P2 : ∃ x : obj, Pet x ∧ BrownSwissCattle x
axiom P3 : ∀ x : obj, Cow x → DomesticatedAnimal x
axiom P4 : ∀ x : obj, Alligator x → ¬ DomesticatedAnimal x
axiom P5 : Alligator Ted

-- Conclusion to formalize
theorem goal : BrownSwissCattle Ted → ¬ Pet Ted := by
  intro hBSC
  have hCow : Cow Ted := P1 Ted hBSC
  have hDom : DomesticatedAnimal Ted := P3 Ted hCow
  have hNotDom : ¬ DomesticatedAnimal Ted := P4 Ted P5
  have hFalse : False := hNotDom hDom
  intro _
  exact hFalse
</lean>
ANSWER: True
