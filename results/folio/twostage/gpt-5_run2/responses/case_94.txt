axiom P4a : Owns Tom Fluffy
axiom P4b : Cat Fluffy
axiom P5 : ∀ a : Animal, Cat a → Pet a
axiom P6 : Managed OliveGarden
axiom P7 : MonthlyRent OliveGarden = Dollar2000
axiom P8 : More Dollar2000 Dollar1500
axiom P9 : ∀ b : Building, Managed b → AllowedMoveInWith Tom Fluffy b → NoMoreThan (SecurityDeposit b) Dollar1500 → WillRent Tom b
axiom P10 : ∀ p : Person, ∀ a : Animal, ∀ b : Building, Managed b → AllowsPets b → Pet a → AllowedMoveInWith p a b

-- Conclusion to formalize
theorem goal : SecurityDeposit OliveGarden = Dollar2000 ∨ More (SecurityDeposit OliveGarden) Dollar2000 := sorry

<lean>
axiom Person : Type
axiom Animal : Type
axiom Building : Type
axiom Money : Type

axiom Tom : Person
axiom Fluffy : Animal
axiom OliveGarden : Building
axiom Dollar2000 : Money
axiom Dollar1500 : Money

axiom Managed : Building → Prop
axiom AllowsPets : Building → Prop
axiom Pet : Animal → Prop
axiom Cat : Animal → Prop
axiom Owns : Person → Animal → Prop
axiom AllowedMoveInWith : Person → Animal → Building → Prop
axiom WillRent : Person → Building → Prop

axiom MonthlyRent : Building → Money
axiom SecurityDeposit : Building → Money

axiom More : Money → Money → Prop
axiom NoMoreThan : Money → Money → Prop
axiom DepositRequired : Building → Prop

-- Premises
axiom P1 : ∃ b : Building, Managed b ∧ AllowsPets b
axiom P2 : ∀ b : Building, Managed b → DepositRequired b
axiom P3 : ∀ b : Building, Managed b → (SecurityDeposit b = MonthlyRent b ∨ More (SecurityDeposit b) (MonthlyRent b))
axiom P4a : Owns Tom Fluffy
axiom P4b : Cat Fluffy
axiom P5 : ∀ a : Animal, Cat a → Pet a
axiom P6 : Managed OliveGarden
axiom P7 : MonthlyRent OliveGarden = Dollar2000
axiom P8 : More Dollar2000 Dollar1500
axiom P9 : ∀ b : Building, Managed b → AllowedMoveInWith Tom Fluffy b → NoMoreThan (SecurityDeposit b) Dollar1500 → WillRent Tom b
axiom P10 : ∀ p : Person, ∀ a : Animal, ∀ b : Building, Managed b → AllowsPets b → Pet a → AllowedMoveInWith p a b

-- Conclusion to formalize
theorem goal : SecurityDeposit OliveGarden = Dollar2000 ∨ More (SecurityDeposit OliveGarden) Dollar2000 :=
by
  have h := P3 OliveGarden P6
  cases h with
  | inl h1 =>
    left
    rw [P7] at h1
    exact h1
  | inr h2 =>
    right
    rw [P7] at h2
    exact h2
</lean>
ANSWER: True
