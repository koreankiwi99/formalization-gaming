<lean>
axiom obj : Type
axiom Peter : obj

axiom SpillFoodOnClothes : obj → Prop
axiom NotablyTidy : obj → Prop
axiom ClumsyFoodieExplorer : obj → Prop
axiom Cleanly : obj → Prop
axiom ValuesOrderSpotlessness : obj → Prop
axiom FamilyPrioritizesOrderSpotlessness : obj → Prop

axiom A1 : ∀ x : obj, SpillFoodOnClothes x → ¬ NotablyTidy x
axiom A2 : ∀ x : obj, ClumsyFoodieExplorer x → SpillFoodOnClothes x
axiom A3 : ∀ x : obj, Cleanly x → NotablyTidy x
axiom A4 : ∀ x : obj, ValuesOrderSpotlessness x → Cleanly x
axiom A5 : ∀ x : obj, FamilyPrioritizesOrderSpotlessness x → ValuesOrderSpotlessness x
axiom A6 : (SpillFoodOnClothes Peter ∧ Cleanly Peter) ∨ (¬ SpillFoodOnClothes Peter ∧ ¬ Cleanly Peter)

theorem goal :
  (ClumsyFoodieExplorer Peter ∧ FamilyPrioritizesOrderSpotlessness Peter) ∨
  (¬ ClumsyFoodieExplorer Peter ∧ ¬ FamilyPrioritizesOrderSpotlessness Peter) :=
by
  cases A6 with
  | inl h =>
      have s : SpillFoodOnClothes Peter := h.left
      have c : Cleanly Peter := h.right
      have nt : NotablyTidy Peter := A3 Peter c
      have nnt : ¬ NotablyTidy Peter := A1 Peter s
      have f : False := nnt nt
      exact Or.inr ⟨(fun _ => False.elim f), (fun _ => False.elim f)⟩
  | inr h =>
      have ns : ¬ SpillFoodOnClothes Peter := h.left
      have nc : ¬ Cleanly Peter := h.right
      have nCFE : ¬ ClumsyFoodieExplorer Peter :=
        fun hCFE => ns (A2 Peter hCFE)
      have nFP : ¬ FamilyPrioritizesOrderSpotlessness Peter :=
        fun hFP =>
          let v := A5 Peter hFP
          let cl := A4 Peter v
          nc cl
      exact Or.inr ⟨nCFE, nFP⟩
</lean>
