  ∀ x : obj, WildTurkey x →
    (EasternWildTurkey x ∨ OsceolaWildTurkey x ∨ GouldsWildTurkey x ∨

              cases h4 with
              | inl hMerriams =>
                  exact False.elim (NotMerriamsTom hMerriams)
              | inr h5 =>
