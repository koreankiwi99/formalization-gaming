<lean>
axiom person : Type

axiom Hannah : person

axiom AtMarysSchool : person → Prop
axiom WorksStudentJobsOnCampus : person → Prop
axiom NeedsToEarnMoneyForTuition : person → Prop
axiom OrdersTakeoutFrequentlyInCollege : person → Prop
axiom EnjoysDiningHallMealsAndRecipes : person → Prop
axiom PickyEater : person → Prop
axiom SpendsALotOfTimeEatingAndCatchingUpInCampusDiningHalls : person → Prop

-- Premises
axiom P1 : ∀ x : person, AtMarysSchool x → WorksStudentJobsOnCampus x → NeedsToEarnMoneyForTuition x
axiom P2 : ∀ x : person, AtMarysSchool x → OrdersTakeoutFrequentlyInCollege x → WorksStudentJobsOnCampus x
axiom P3 : ∀ x : person, AtMarysSchool x → (OrdersTakeoutFrequentlyInCollege x ∨ EnjoysDiningHallMealsAndRecipes x)
axiom P4 : ∀ x : person, AtMarysSchool x → EnjoysDiningHallMealsAndRecipes x → ¬ PickyEater x
axiom P5 : ∀ x : person, AtMarysSchool x → EnjoysDiningHallMealsAndRecipes x → SpendsALotOfTimeEatingAndCatchingUpInCampusDiningHalls x
axiom P6 : AtMarysSchool Hannah
axiom P7 : WorksStudentJobsOnCampus Hannah ∧ (NeedsToEarnMoneyForTuition Hannah → (¬ PickyEater Hannah ∧ ¬ NeedsToEarnMoneyForTuition Hannah))

-- Conclusion
theorem goal : AtMarysSchool Hannah ∧ ¬ PickyEater Hannah ∧ SpendsALotOfTimeEatingAndCatchingUpInCampusDiningHalls Hannah := by
  have hAt : AtMarysSchool Hannah := P6
  have hor : OrdersTakeoutFrequentlyInCollege Hannah ∨ EnjoysDiningHallMealsAndRecipes Hannah := P3 Hannah hAt
  have hNP : ¬ PickyEater Hannah := by
    cases hor with
    | inl horders =>
      have hworks : WorksStudentJobsOnCampus Hannah := P2 Hannah hAt horders
      have hneeds : NeedsToEarnMoneyForTuition Hannah := P1 Hannah hAt hworks
      have hpair : ¬ PickyEater Hannah ∧ ¬ NeedsToEarnMoneyForTuition Hannah := (And.right P7) hneeds
      have hcontr : False := (And.right hpair) hneeds
      exact False.elim hcontr
    | inr henjoys =>
      exact P4 Hannah hAt henjoys
  have hSp : SpendsALotOfTimeEatingAndCatchingUpInCampusDiningHalls Hannah := by
    cases hor with
    | inl horders =>
      have hworks : WorksStudentJobsOnCampus Hannah := P2 Hannah hAt horders
      have hneeds : NeedsToEarnMoneyForTuition Hannah := P1 Hannah hAt hworks
      have hpair : ¬ PickyEater Hannah ∧ ¬ NeedsToEarnMoneyForTuition Hannah := (And.right P7) hneeds
      have hcontr : False := (And.right hpair) hneeds
      have henjoys : EnjoysDiningHallMealsAndRecipes Hannah := False.elim hcontr
      exact P5 Hannah hAt henjoys
    | inr henjoys =>
      exact P5 Hannah hAt henjoys
  exact And.intro hAt (And.intro hNP hSp)
</lean>
