<lean>
axiom Person : Type
axiom Vladimir : Person
axiom Ekaterina : Person

axiom CanRegisterUS : Person → Prop
axiom CanParticipate2024USElection : Person → Prop
axiom USCitizenship : Person → Prop
axiom TaiwaneseCitizenship : Person → Prop
axiom RFOfficial : Person → Prop
axiom ManagerGazprom : Person → Prop

-- Premises
axiom P1 : ∀ x : Person, CanRegisterUS x → CanParticipate2024USElection x
axiom P2 : ∀ x : Person, USCitizenship x → CanRegisterUS x
axiom P3 : ∀ x : Person, USCitizenship x ∨ TaiwaneseCitizenship x
axiom P4 : ∀ x : Person, RFOfficial x → ¬ TaiwaneseCitizenship x
axiom P5a : ¬ TaiwaneseCitizenship Vladimir
axiom P5b : ¬ ManagerGazprom Vladimir
axiom P6 : CanRegisterUS Ekaterina ∨ RFOfficial Ekaterina

-- Conclusion to formalize
theorem goal : CanParticipate2024USElection Ekaterina ∨ ManagerGazprom Vladimir :=
by
  cases P6 with
  | inl hReg =>
      exact Or.inl (P1 Ekaterina hReg)
  | inr hRF =>
      have hNotTai : ¬ TaiwaneseCitizenship Ekaterina := P4 Ekaterina hRF
      have hUS : USCitizenship Ekaterina :=
        Or.elim (P3 Ekaterina)
          (fun hUS => hUS)
          (fun hTai => False.elim (hNotTai hTai))
      have hReg : CanRegisterUS Ekaterina := P2 Ekaterina hUS
      exact Or.inl (P1 Ekaterina hReg)
</lean>
