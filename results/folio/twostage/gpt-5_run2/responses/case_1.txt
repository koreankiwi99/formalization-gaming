--- Iteration 1 ---

<lean>
axiom obj : Type

axiom InClub : obj → Prop
axiom PerformsTalentShowsOften : obj → Prop
axiom AttendsSchoolEvents : obj → Prop
axiom VeryEngagedWithSchoolEvents : obj → Prop
axiom InactiveDisinterestedCommunityMember : obj → Prop
axiom ChaperonesHighSchoolDances : obj → Prop
axiom StudentWhoAttendsSchool : obj → Prop
axiom YoungChild : obj → Prop
axiom Teenager : obj → Prop
axiom WishesToFurtherAcademicCareersAndEducationalOpportunities : obj → Prop

axiom Bonnie : obj

-- Premises
axiom P1 : ∀ x : obj, InClub x ∧ PerformsTalentShowsOften x → AttendsSchoolEvents x ∧ VeryEngagedWithSchoolEvents x
axiom P2 : ∀ x : obj, InClub x → (PerformsTalentShowsOften x ∨ InactiveDisinterestedCommunityMember x)
axiom P3 : ∀ x : obj, InClub x ∧ ChaperonesHighSchoolDances x → ¬ StudentWhoAttendsSchool x
axiom P4 : ∀ x : obj, InClub x ∧ InactiveDisinterestedCommunityMember x → ChaperonesHighSchoolDances x
axiom P5 : ∀ x : obj, InClub x ∧ (YoungChild x ∨ Teenager x) ∧ WishesToFurtherAcademicCareersAndEducationalOpportunities x → StudentWhoAttendsSchool x

axiom BonnieInClub : InClub Bonnie
axiom BonnieEither :
  (AttendsSchoolEvents Bonnie ∧ VeryEngagedWithSchoolEvents Bonnie ∧ StudentWhoAttendsSchool Bonnie) ∨
  (¬ (AttendsSchoolEvents Bonnie ∧ VeryEngagedWithSchoolEvents Bonnie) ∧ ¬ StudentWhoAttendsSchool Bonnie)

-- Conclusion to formalize
theorem goal :
  (((YoungChild Bonnie ∨ Teenager Bonnie) ∧ WishesToFurtherAcademicCareersAndEducationalOpportunities Bonnie ∧ ChaperonesHighSchoolDances Bonnie)
    ∨ ¬ ((YoungChild Bonnie ∨ Teenager Bonnie) ∧ WishesToFurtherAcademicCareersAndEducationalOpportunities Bonnie))
  → (StudentWhoAttendsSchool Bonnie ∨ InactiveDisinterestedCommunityMember Bonnie) := by
  intro _
  have h2 : PerformsTalentShowsOften Bonnie ∨ InactiveDisinterestedCommunityMember Bonnie :=
    P2 Bonnie BonnieInClub
  cases h2 with
  | inl perf =>
      have hAttendEng : AttendsSchoolEvents Bonnie ∧ VeryEngagedWithSchoolEvents Bonnie :=
        P1 Bonnie (And.intro BonnieInClub perf)
      have student : StudentWhoAttendsSchool Bonnie := by
        cases BonnieEither with
        | inl triple =>
            exact triple.right.right
        | inr negpair =>
            have contra : False := by
              exact negpair.left hAttendEng
            exact False.elim contra
      exact Or.inl student
  | inr inactive =>
      exact Or.inr inactive
</lean>
