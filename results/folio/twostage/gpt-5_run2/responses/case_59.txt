STAGE 1: TRANSLATION

--- Iteration 1 ---

<lean>
axiom City : Type
axiom State : Type

-- Cities
axiom Billings : City
axiom Butte : City
axiom Helena : City
axiom Missoula : City
axiom WhiteSulphurSprings : City
axiom StPierre : City
axiom Bristol : City
axiom Texarkana : City
axiom Texhoma : City
axiom UnionCity : City

-- States
axiom Montana : State

-- Predicates/relations
axiom InState : City → State → Prop
axiom InUS : State → Prop
axiom InCity : City → City → Prop

-- Premises
-- Billings is a city in the state of Montana in U.S.
axiom A1 : InState Billings Montana
axiom A1b : InUS Montana

-- The state of Montana includes the cities of Butte, Helena, and Missoula.
axiom A2a : InState Butte Montana
axiom A2b : InState Helena Montana
axiom A2c : InState Missoula Montana

-- White Sulphur Springs and Butte are cities in the same state in U.S.
axiom A3 : ∃ s : State, InUS s ∧ InState WhiteSulphurSprings s ∧ InState Butte s

-- The city of St Pierre is not in the state of Montana.
axiom A4 : ¬ InState StPierre Montana

-- Any city in Butte is not in St Pierre.
axiom A5 : ∀ c : City, InCity c Butte → ¬ InCity c StPierre

-- A city can only be in one state in U.S. except for Bristol, Texarkana, Texhoma and Union City.
axiom A6 :
  ∀ c : City, ∀ s1 s2 : State,
    InState c s1 → InState c s2 → InUS s1 → InUS s2 →
    c = Bristol ∨ c = Texarkana ∨ c = Texhoma ∨ c = UnionCity ∨ s1 = s2

-- Conclusion: Montana is home to the city of Missoula.
theorem goal : InState Missoula Montana := sorry
</lean>
