--- Iteration 1 ---

<lean>
axiom obj : Type

axiom Animal : obj → Prop
axiom HeckCattle : obj → Prop
axiom Aurochs : obj → Prop
axiom Extinct : obj → Prop
axiom Resembles : obj → obj → Prop
axiom BredBack : obj → Prop
axiom ArtificiallySelected : obj → Prop
axiom ToBeBredBack : obj → Prop

-- Breeding back is a form of artificial selection
axiom P1 : ∀ x : obj, BredBack x → ArtificiallySelected x

-- Heck cattle were bred back in the 1920s to resemble the aurochs
axiom P2 : ∃ x : obj, HeckCattle x ∧ BredBack x ∧ ∃ y : obj, Aurochs y ∧ Resembles x y

-- Heck cattle are animals
axiom P3 : ∀ x : obj, HeckCattle x → Animal x

-- Aurochs are animals
axiom P4 : ∀ x : obj, Aurochs x → Animal x

-- Some animals to be bred back resemble extinct animals
axiom P5 : ∃ x : obj, Animal x ∧ ToBeBredBack x ∧ ∃ y : obj, Extinct y ∧ Animal y ∧ Resembles x y

theorem goal : ∃ x : obj, HeckCattle x ∧ ArtificiallySelected x := by
  cases P2 with
  | intro x hx =>
    have hxC : HeckCattle x := And.left hx
    have hx' : BredBack x ∧ ∃ y : obj, Aurochs y ∧ Resembles x y := And.right hx
    have hb : BredBack x := And.left hx'
    have ha : ArtificiallySelected x := P1 x hb
    exact Exists.intro x (And.intro hxC ha)
</lean>
ANSWER: True
