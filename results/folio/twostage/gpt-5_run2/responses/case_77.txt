<lean>
axiom obj : Type

axiom AtMarysSchool : obj → Prop
axiom WorksStudentJobsOnCampus : obj → Prop
axiom NeedsToEarnMoneyForTuition : obj → Prop
axiom OrdersTakeoutFrequentlyInCollege : obj → Prop
axiom EnjoysDiningHallMealsAndRecipes : obj → Prop
axiom PickyEater : obj → Prop
axiom SpendsLotOfTimeEatingAndCatchingUpInCampusDiningHalls : obj → Prop

-- Premises
axiom R1 : ∀ x : obj, AtMarysSchool x → WorksStudentJobsOnCampus x → NeedsToEarnMoneyForTuition x
axiom R2 : ∀ x : obj, AtMarysSchool x → OrdersTakeoutFrequentlyInCollege x → WorksStudentJobsOnCampus x
axiom R3 : ∀ x : obj, AtMarysSchool x → (OrdersTakeoutFrequentlyInCollege x ∨ EnjoysDiningHallMealsAndRecipes x)
axiom R4 : ∀ x : obj, AtMarysSchool x → EnjoysDiningHallMealsAndRecipes x → ¬ PickyEater x
axiom R5 : ∀ x : obj, AtMarysSchool x → EnjoysDiningHallMealsAndRecipes x → SpendsLotOfTimeEatingAndCatchingUpInCampusDiningHalls x

axiom Hannah : obj
axiom H_at : AtMarysSchool Hannah
axiom H_work_and_cond :
  WorksStudentJobsOnCampus Hannah ∧
  (NeedsToEarnMoneyForTuition Hannah → (¬ PickyEater Hannah ∧ ¬ NeedsToEarnMoneyForTuition Hannah))

-- Conclusion
theorem goal :
  AtMarysSchool Hannah ∧
  (¬ PickyEater Hannah ∨ (PickyEater Hannah → SpendsLotOfTimeEatingAndCatchingUpInCampusDiningHalls Hannah)) := by
  have H_work : WorksStudentJobsOnCampus Hannah := And.left H_work_and_cond
  have H_if : NeedsToEarnMoneyForTuition Hannah → (¬ PickyEater Hannah ∧ ¬ NeedsToEarnMoneyForTuition Hannah) := And.right H_work_and_cond
  have H_need : NeedsToEarnMoneyForTuition Hannah := R1 Hannah H_at H_work
  have H_notPicky_and_notNeed : ¬ PickyEater Hannah ∧ ¬ NeedsToEarnMoneyForTuition Hannah := H_if H_need
  have H_notPicky : ¬ PickyEater Hannah := And.left H_notPicky_and_notNeed
  exact And.intro H_at (Or.inl H_notPicky)
</lean>
