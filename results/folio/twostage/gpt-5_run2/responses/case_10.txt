<lean>
axiom obj : Type

axiom Tom : obj

axiom WildTurkey : obj → Prop
axiom Eastern : obj → Prop
axiom Osceola : obj → Prop
axiom Goulds : obj → Prop
axiom Merriams : obj → Prop
axiom RioGrande : obj → Prop
axiom Ocellated : obj → Prop

-- There are six types of wild turkeys (exhaustiveness)
axiom T_types_exhaustive :
  ∀ x : obj, WildTurkey x → (Eastern x ∨ Osceola x ∨ Goulds x ∨ Merriams x ∨ RioGrande x ∨ Ocellated x)

-- Tom is not an Eastern wild turkey.
axiom T_not_Eastern : ¬ Eastern Tom
-- Tom is not an Osceola wild turkey.
axiom T_not_Osceola : ¬ Osceola Tom
-- Tom is not a Gould's wild turkey.
axiom T_not_Goulds : ¬ Goulds Tom
-- Tom is neither a Merriam's wild turkey nor a Rio Grande wild turkey.
axiom T_not_Merriams : ¬ Merriams Tom
axiom T_not_RioGrande : ¬ RioGrande Tom
-- Tom is a wild turkey.
axiom T_wild : WildTurkey Tom

theorem goal : Eastern Tom := sorry
</lean>
