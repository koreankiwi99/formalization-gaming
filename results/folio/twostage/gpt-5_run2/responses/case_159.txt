<lean>
axiom obj : Type
axiom Student : obj → Prop
axiom WorkLibrary : obj → Prop
axiom WorkLab : obj → Prop
axiom PartTimeUni : obj → Prop
axiom TakesDatabase : obj → Prop
axiom ClassWithDavid : obj → Prop
axiom FromCSDept : obj → Prop
axiom James : obj

-- None of the students who currently work in the library are from the computer science department.
axiom A1 : ∀ x : obj, Student x ∧ WorkLibrary x → ¬ FromCSDept x

-- All of the students with part-time jobs offered by the university currently work in the library.
axiom A2 : ∀ x : obj, Student x ∧ PartTimeUni x → WorkLibrary x

-- All of the students who are taking the database course are from the computer science department.
axiom A3 : ∀ x : obj, Student x ∧ TakesDatabase x → FromCSDept x

-- All of the students taking a class with Professor David are taking the database course.
axiom A4 : ∀ x : obj, Student x ∧ ClassWithDavid x → TakesDatabase x

-- All students working in the lab are taking a class with Professor David.
axiom A5 : ∀ x : obj, Student x ∧ WorkLab x → ClassWithDavid x

-- James is a student working in the lab.
axiom A6_student : Student James
axiom A6_lab : WorkLab James

-- James doesn't work in the lab or have a part-time job offered by the university.
axiom A7_not_lab : ¬ WorkLab James
axiom A7_not_pt : ¬ PartTimeUni James

-- Conclusion to formalize: James takes the database course.
theorem goal : TakesDatabase James := sorry
</lean>
