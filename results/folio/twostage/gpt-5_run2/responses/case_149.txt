<lean>
axiom obj : Type

-- Entities
axiom NewVesselPress : obj
axiom NeapolitanChronicles : obj
axiom PalaceOfFlies : obj
axiom Italian : obj

-- Predicates/Relations
axiom PublishingHouse : obj → Prop
axiom SpecializesTranslatingForeignToEnglish : obj → Prop
axiom Book : obj → Prop
axiom PublishedBy : obj → obj → Prop
axiom English : obj → Prop
axiom TranslatedFrom : obj → obj → Prop

-- Premises
axiom P1 : PublishingHouse NewVesselPress
axiom P2 : SpecializesTranslatingForeignToEnglish NewVesselPress
axiom P3 : ∀ x : obj, PublishedBy x NewVesselPress → English x
axiom P4 : Book NeapolitanChronicles
axiom P5 : PublishedBy NeapolitanChronicles NewVesselPress
axiom P6 : TranslatedFrom NeapolitanChronicles Italian
axiom P7 : Book PalaceOfFlies
axiom P8 : PublishedBy PalaceOfFlies NewVesselPress

-- Conclusion to formalize: Neapolitan Chronicles is an English book.
theorem goal : English NeapolitanChronicles ∧ Book NeapolitanChronicles := by
  apply And.intro
  · exact P3 NeapolitanChronicles P5
  · exact P4
