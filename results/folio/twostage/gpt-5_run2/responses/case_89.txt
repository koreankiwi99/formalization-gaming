<lean>
axiom obj : Type

axiom Person : obj → Prop
axiom Book : obj → Prop
axiom ContainsKnowledge : obj → Prop
axiom Read : obj → obj → Prop
axiom GainsKnowledge : obj → Prop
axiom Smarter : obj → Prop

axiom Harry : obj
axiom Walden : obj

-- Premises
axiom P1 : ∀ b : obj, Book b → ContainsKnowledge b
axiom P2 : ∀ p b : obj, Person p → Book b → Read p b → GainsKnowledge p
axiom P3 : ∀ p : obj, Person p → GainsKnowledge p → Smarter p
axiom P4 : Person Harry
axiom P5 : Book Walden
axiom P6 : Read Harry Walden

-- Conclusion to formalize: A smarter person has gained knowledge (existential reading)
theorem goal : ∃ x : obj, Person x ∧ Smarter x ∧ GainsKnowledge x :=
by
  have gk : GainsKnowledge Harry := P2 Harry Walden P4 P5 P6
  have sm : Smarter Harry := P3 Harry P4 gk
  exact Exists.intro Harry (And.intro P4 (And.intro sm gk))
</lean>
