axiom VeryConsciousAboutHealthAndEatingHabits : obj → Prop

            (¬ GrewUpInHealthConsciousChildhoodHome Taylor ∧ ¬ DisregardsPhysicalWellBeing Taylor)

theorem goal : EatsSaladsRegularly Taylor := sorry
