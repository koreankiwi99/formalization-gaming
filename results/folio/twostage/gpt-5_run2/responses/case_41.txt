axiom P2 : ∀ x : obj, BrutalistBuilding x ∧ AdoresMax x → ¬ InterestingGeometry x

axiom P3 : ∀ x : obj, AdoresMax x → (ZahaStyle x ∨ KellyStyle x)

axiom P4 : ∀ x : obj, KellyStyle x ∧ AdoresMax x → Evocative x

axiom P5 : ∀ x : obj, KellyStyle x ∧ AdoresMax x → Dreamy x

axiom P6 : ∀ x : obj, DesignByMax x ∧ AdoresMax x ∧ InterestingGeometry x → (BrutalistBuilding x ∧ Evocative x)

-- Auxiliary axiom: Every design by Max is adored by Max.
axiom AdmDesign : ∀ x : obj, DesignByMax x → AdoresMax x

-- Conclusion: A design by Max is either evocative or dreamy.
theorem goal : ∀ x : obj, DesignByMax x → (Evocative x ∨ Dreamy x) := by
  intro x hx
  have had : AdoresMax x := AdmDesign x hx
  have hs : ZahaStyle x ∨ KellyStyle x := P3 x had
  cases hs with
  | inl hz =>
      have hg : InterestingGeometry x := P1 x ⟨hz, had⟩
      have hbe : BrutalistBuilding x ∧ Evocative x := P6 x ⟨hx, had, hg⟩
      exact Or.inl hbe.right
  | inr hk =>
      have hev : Evocative x := P4 x ⟨hk, had⟩
      exact Or.inl hev
