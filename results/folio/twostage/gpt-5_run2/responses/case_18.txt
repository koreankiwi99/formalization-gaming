--- Iteration 1 ---

<lean>
axiom obj : Type
axiom Rabbit : obj → Prop
axiom Turtle : obj → Prop
axiom Squirrel : obj → Prop
axiom SpottedNearCampus : obj → Prop
axiom Cute : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop

axiom Rockie : obj

-- Premises
axiom P1 : ∀ x : obj, Rabbit x → SpottedNearCampus x → Cute x
axiom P2 : ∃ x : obj, Turtle x ∧ SpottedNearCampus x
axiom P3 : ∀ x : obj, SpottedNearCampus x → Rabbit x ∨ Squirrel x
axiom P4 : ∀ x : obj, Skittish x → ¬ Calm x
axiom P5 : ∀ x : obj, Squirrel x → SpottedNearCampus x → Skittish x
axiom P6 : SpottedNearCampus Rockie ∧ Calm Rockie

-- Conclusion
theorem goal : Turtle Rockie ∨ Cute Rockie := by
  have spottedR : SpottedNearCampus Rockie := And.left P6
  have calmR : Calm Rockie := And.right P6
  have h : Rabbit Rockie ∨ Squirrel Rockie := P3 Rockie spottedR
  cases h with
  | inl hr =>
      have cuteR : Cute Rockie := P1 Rockie hr spottedR
      exact Or.inr cuteR
  | inr hs =>
      have skR : Skittish Rockie := P5 Rockie hs spottedR
      have notCalm : ¬ Calm Rockie := P4 Rockie skR
      have contra : False := notCalm calmR
      exact False.elim contra
</lean>
ANSWER: True
