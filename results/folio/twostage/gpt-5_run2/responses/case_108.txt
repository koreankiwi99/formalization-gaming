<lean>
axiom obj : Type

axiom Peter : obj

axiom Spills : obj → Prop
axiom Tidy : obj → Prop
axiom ClumsyFoodie : obj → Prop
axiom GoesOut : obj → Prop
axiom Cleanly : obj → Prop
axiom ValuesOrder : obj → Prop
axiom FamilyPrioritizesOrder : obj → Prop

axiom P1 : ∀ x : obj, Spills x → ¬ Tidy x
axiom P2 : ∀ x : obj, (ClumsyFoodie x ∧ GoesOut x) → Spills x
axiom P3 : ∀ x : obj, Cleanly x → Tidy x
axiom P4 : ∀ x : obj, ValuesOrder x → Cleanly x
axiom P5 : ∀ x : obj, FamilyPrioritizesOrder x → ValuesOrder x
axiom P6 : (Spills Peter ∧ Cleanly Peter) ∨ (¬ Spills Peter ∧ ¬ Cleanly Peter)

theorem goal : (ClumsyFoodie Peter ∧ GoesOut Peter) ∨ FamilyPrioritizesOrder Peter := sorry
