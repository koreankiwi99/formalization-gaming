--- Iteration 1 ---

<lean>
axiom obj : Type

axiom James : obj

axiom Student : obj → Prop
axiom WorksInLibrary : obj → Prop
axiom FromCSDept : obj → Prop
axiom HasPTUniJob : obj → Prop
axiom TakingDBCourse : obj → Prop
axiom TakingWithProfDavid : obj → Prop
axiom WorkingInLab : obj → Prop

-- None of the students who currently work in the library are from the computer science department.
axiom A1 : ∀ x : obj, (Student x ∧ WorksInLibrary x) → ¬ FromCSDept x

-- All of the students with part-time jobs offered by the university currently work in the library.
axiom A2 : ∀ x : obj, (Student x ∧ HasPTUniJob x) → WorksInLibrary x

-- All of the students who are taking the database course are from the computer science department.
axiom A3 : ∀ x : obj, (Student x ∧ TakingDBCourse x) → FromCSDept x

-- All of the students taking a class with Professor David are taking the database course.
axiom A4 : ∀ x : obj, (Student x ∧ TakingWithProfDavid x) → TakingDBCourse x

-- All students working in the lab are taking a class with Professor David.
axiom A5 : ∀ x : obj, (Student x ∧ WorkingInLab x) → TakingWithProfDavid x

-- James is a student working in the lab.
axiom A6 : Student James
axiom A7 : WorkingInLab James

-- James doesn't work in the lab or have a part-time job offered by the university.
axiom A8 : ¬ (WorkingInLab James ∨ HasPTUniJob James)

-- Conclusion: James has a part-time job offered by the university.
theorem goal : HasPTUniJob James := sorry
</lean>
