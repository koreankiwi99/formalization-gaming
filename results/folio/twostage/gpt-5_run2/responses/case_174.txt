--- Iteration 1 ---

<lean>
axiom obj : Type
axiom Taylor : obj

axiom EatSaladsRegularly : obj → Prop
axiom VeryConsciousHealthEating : obj → Prop
axiom GrewUpHealthConsciousHome : obj → Prop
axiom FulfillDailyNutrition : obj → Prop
axiom DisregardPhysicalWellbeing : obj → Prop
axiom VisitGymDaily : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, EatSaladsRegularly x → VeryConsciousHealthEating x
axiom A2 : ∀ x : obj, GrewUpHealthConsciousHome x → EatSaladsRegularly x
axiom A3 : ∀ x : obj, FulfillDailyNutrition x → GrewUpHealthConsciousHome x
axiom A4 : ∀ x : obj, DisregardPhysicalWellbeing x → ¬ VeryConsciousHealthEating x
axiom A5 : ∀ x : obj, VisitGymDaily x → FulfillDailyNutrition x
axiom A6 : (GrewUpHealthConsciousHome Taylor ∧ DisregardPhysicalWellbeing Taylor) ∨ (¬ GrewUpHealthConsciousHome Taylor ∧ ¬ DisregardPhysicalWellbeing Taylor)

-- Conclusion to formalize
theorem goal : ¬ GrewUpHealthConsciousHome Taylor ∧ ¬ VisitGymDaily Taylor :=
by
  cases A6 with
  | inl h =>
      have v : VeryConsciousHealthEating Taylor := A1 Taylor (A2 Taylor h.left)
      have nv : ¬ VeryConsciousHealthEating Taylor := A4 Taylor h.right
      exact False.elim (nv v)
  | inr h =>
      have hnotG : ¬ GrewUpHealthConsciousHome Taylor := h.left
      have hnotVisit : ¬ VisitGymDaily Taylor :=
        fun hv => hnotG (A3 Taylor (A5 Taylor hv))
      exact And.intro hnotG hnotVisit
</lean>
ANSWER: True
