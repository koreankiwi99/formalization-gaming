axiom obj : Type
axiom Jane : obj
axiom KiKi : obj

axiom Animal : obj → Prop
axiom Barks : obj → Prop
axiom Jumps : obj → Prop
axiom Legs : obj → Prop
axiom Terricolous : obj → Prop
axiom Likes : obj → obj → Prop

axiom P1 : ∀ x : obj, Animal x → Barks x → ¬ (Likes Jane x)
axiom P2 : ∀ x : obj, Animal x → Barks x  -- All dogs are animals that bark
axiom P3 : ∀ x : obj, Animal x → Jumps x → Likes Jane x
axiom P4 : ∀ x : obj, Legs x → Jumps x  -- If has legs then jumps
axiom P5 : ∀ x : obj, Terricolous x → Legs x  -- If terricolous then has legs
axiom P6 : Animal KiKi
axiom P7 : (¬ Jumps KiKi ∧ ¬ Legs KiKi) → (Terricolous KiKi ∨ Legs KiKi)

theorem goal : Terricolous KiKi := by
  have bark_KiKi : Barks KiKi := P2 KiKi P6
  have no_like : ¬ Likes Jane KiKi := P1 KiKi P6 bark_KiKi
  have no_jump : ¬ Jumps KiKi := by 
    intro jump
    exact no_like (P3 KiKi P6 jump)
  have no_legs : ¬ Legs KiKi := by 
    intro legs
    exact no_jump (P4 KiKi legs)
  have disj : Terricolous KiKi ∨ Legs KiKi := P7 ⟨no_jump, no_legs⟩
  cases disj with
  | inl t => exact t
  | inr legs => contradiction
</lean>
