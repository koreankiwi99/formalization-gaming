axiom Nation : Type
axiom Region : Type
axiom University : Type
axiom Year : Type
axiom OlympicType : Type
axiom OlympicSeason : Type

-- Constants
axiom Beijing : City
axiom PRC : Nation
axiom NorthernChina : Region
axiom summer : OlympicSeason
axiom winter : OlympicSeason
axiom olympics : OlympicType
axiom y2008 : Year

-- Predicates
axiom capital_of : City → Nation → Prop
axiom most_populous : Nation → Prop
axiom located_in : City → Region → Prop
axiom hosted : City → OlympicType → OlympicSeason → Year → Prop
axiom university : University → Prop
axiom university_in : University → City → Prop
axiom ranks_best : University → Prop

-- Premises
axiom capital_prc : capital_of Beijing PRC
axiom capital_most_populous : ∃ n : Nation, most_populous n ∧ capital_of Beijing n
axiom location : located_in Beijing NorthernChina
axiom hosted_2008_summer : hosted Beijing olympics summer y2008
axiom hosted_summer_winter : 
  (∃ yr : Year, hosted Beijing olympics summer yr) ∧ 
  (∃ yr : Year, hosted Beijing olympics winter yr)
axiom top_universities : 
  ∃ u : University, university_in u Beijing ∧ university u ∧ ranks_best u

-- Conclusion (to be proved)
theorem conclusion : 
  hosted Beijing olympics summer y2008 ∧ 
  (∃ yr : Year, hosted Beijing olympics winter yr) := 
⟨hosted_2008_summer, hosted_summer_winter.right⟩
</lean>

ANSWER: True
