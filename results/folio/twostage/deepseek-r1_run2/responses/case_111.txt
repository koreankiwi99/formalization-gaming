    MountainRange mr → InTexas mr → 
    ¬ ∃ (m : Obj), Mine m ∧ LocatedIn m mr ∧ Donated m

-- Conclusion: Onate visited a NM mountain range
theorem goal : ∃ (mr : Obj), 
    MountainRange mr ∧ InNewMexico mr ∧ VisitedByOnate mr := 
by
  -- We have no axiom for MountainRange PicurisMountains so 
  -- use classical case analysis (excluded middle)
  have h_range_or_not : MountainRange PicurisMountains ∨ ¬MountainRange PicurisMountains
    := by apply Classical.em
  cases h_range_or_not with
  | inl h_range =>  -- Assume it IS a mountain range
    have loc_split : InNewMexico PicurisMountains ∨ InTexas PicurisMountains 
      := range_location h_range
    cases loc_split with
    | inl nm =>  -- Located in New Mexico
      -- PicurisMountains directly satisfies all conditions
      exact ⟨PicurisMountains, ⟨h_range, ⟨nm, visited⟩⟩⟩ 
    | inr texas =>  -- Located in Texas 
      -- Show mine exists to contradict no_tx_donations
      have mine_exists : ∃ (m : Obj), Mine m ∧ LocatedIn m PicurisMountains ∧ Donated m
        := ⟨HardingPegmatiteMine, is_mine, mine_location, mine_donated⟩
      have no_mine : ¬ ∃ (m : Obj), Mine m ∧ LocatedIn m PicurisMountains ∧ Donated m
        := no_tx_donations PicurisMountains h_range texas
      contradiction  -- Thus impossible branch
  | inr not_range =>  -- Assume it IS NOT a mountain range
    -- We must show ∃mr, MountainRange mr ∧ ... 
    -- Given info: All properties about PicurisMountains which aren't mountains
    --             No other locations mentioned
    -- How to find a witness? Impossible by current axioms
    -- Since excluded middle gives disjunction, need magic 
    apply False.elim  -- Get rid of impossible case by assuming contradiction...
    apply not_range   -- ...but cannot generate contradiction without extra knowledge
    sorry             -- MUST REMAIN: Unfixable gap in axioms. See analysis.
