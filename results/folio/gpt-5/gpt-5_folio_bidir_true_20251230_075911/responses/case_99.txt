<lean>
axiom obj : Type
axiom Harry : obj

axiom Invertebrate : obj → Prop
axiom Vertebrate : obj → Prop
axiom Backbone : obj → Prop
axiom ReproduceMF : obj → Prop
axiom Bee : obj → Prop
axiom QueenBee : obj → Prop

-- Animals are either invertebrates or vertebrates.
axiom A1 : ∀ x : obj, Invertebrate x ∨ Vertebrate x
-- All animals with backbones reproduce by male-and-female mating.
axiom A2 : ∀ x : obj, Backbone x → ReproduceMF x
-- All vertebrate animals have a backbone.
axiom A3 : ∀ x : obj, Vertebrate x → Backbone x
-- All bees do not reproduce by male-and-female mating.
axiom A4 : ∀ x : obj, Bee x → ¬ ReproduceMF x
-- All queen bees are bees.
axiom A5 : ∀ x : obj, QueenBee x → Bee x
-- Harry is a bee.
axiom A6 : Bee Harry

theorem harry_invertebrate_or_queenbee : Invertebrate Harry ∨ QueenBee Harry := by
  have hNotMF : ¬ ReproduceMF Harry := A4 Harry A6
  have hNotBack : ¬ Backbone Harry := by
    intro hb
    apply hNotMF
    exact A2 Harry hb
  have hNotVert : ¬ Vertebrate Harry := by
    intro hv
    apply hNotBack
    exact A3 Harry hv
  have hdisj : Invertebrate Harry ∨ Vertebrate Harry := A1 Harry
  have hinv : Invertebrate Harry := by
    cases hdisj with
    | inl hinv => exact hinv
    | inr hvert => exact False.elim (hNotVert hvert)
  exact Or.inl hinv
</lean>

ANSWER: True
