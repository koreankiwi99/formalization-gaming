<lean>
axiom obj : Type

axiom AiltonSilva : obj
axiom Ailton : obj
axiom Braga : obj
axiom Nautico : obj
axiom Fluminense : obj

axiom FootballPlayer : obj → Prop
axiom LoanedOutTo : obj → obj → Prop
axiom FootballClub : obj → Prop
axiom PlaysFor : obj → obj → Prop
axiom BrazilianFootballer : obj → Prop

-- Ailton Silva is commonly known as Ailton (same person)
axiom Alias : AiltonSilva = Ailton

-- Ailton was loaned out to Braga and is a football player
axiom L_player : FootballPlayer Ailton
axiom L_loan : LoanedOutTo Ailton Braga

-- Ailton Silva plays for Náutico and is Brazilian
axiom T_brazil : BrazilianFootballer AiltonSilva
axiom T_plays : PlaysFor AiltonSilva Nautico

-- Clubs
axiom C_nautico : FootballClub Nautico
axiom C_braga : FootballClub Braga
axiom C_flu : FootballClub Fluminense

theorem ailton_silva_loaned_to_club : ∃ c : obj, FootballClub c ∧ LoanedOutTo AiltonSilva c :=
by
  have h : AiltonSilva = Ailton := Alias
  have t0 : LoanedOutTo Ailton Braga := L_loan
  have t1 : LoanedOutTo AiltonSilva Braga := by
    simpa [h.symm] using t0
  exact ⟨Braga, And.intro C_braga t1⟩
</lean>

ANSWER: True
