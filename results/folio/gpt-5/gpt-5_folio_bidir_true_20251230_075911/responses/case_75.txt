<lean>
axiom obj : Type
axiom Hannah : obj

axiom AtMS : obj → Prop
axiom W : obj → Prop
axiom N : obj → Prop
axiom O : obj → Prop
axiom E : obj → Prop
axiom P : obj → Prop
axiom S : obj → Prop

-- If people at Mary's school work in student jobs on campus, then they need to earn money
axiom A1 : ∀ x : obj, AtMS x → (W x → N x)
-- If people at Mary's school order takeout frequently, then they work in student jobs on campus
axiom A2 : ∀ x : obj, AtMS x → (O x → W x)
-- People at Mary's school order takeout frequently or enjoy the dining hall meals
axiom A3 : ∀ x : obj, AtMS x → (O x ∨ E x)
-- If people at Mary's school enjoy the dining hall meals, then they are not picky eaters
axiom A4 : ∀ x : obj, AtMS x → (E x → ¬ P x)
-- If people at Mary's school enjoy the dining hall meals, then they spend a lot of time in dining halls
axiom A5 : ∀ x : obj, AtMS x → (E x → S x)

-- Hannah is at Mary's school
axiom T1 : AtMS Hannah
-- Hannah works in student jobs on campus and if she needs money then she is neither picky nor needs money
axiom T2 : W Hannah ∧ (N Hannah → (¬ P Hannah ∧ ¬ N Hannah))

-- Statement to evaluate: Hannah needs to earn money to help pay for her college tuition.
theorem hannah_needs_money : N Hannah := by
  have hWN : W Hannah → N Hannah := A1 Hannah T1
  have hW : W Hannah := And.left T2
  exact hWN hW
</lean>

ANSWER: True
