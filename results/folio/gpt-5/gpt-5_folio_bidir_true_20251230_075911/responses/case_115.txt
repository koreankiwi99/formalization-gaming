axiom obj : Type

axiom Man : obj → Prop
axiom Taller : obj → obj → Prop
axiom CanBlock : obj → obj → Prop
axiom JumpsWhenShooting : obj → Prop
axiom GreatShooter : obj → Prop

axiom Michael : obj
axiom Peter : obj
axiom Windy : obj

axiom InMichaelsClass : obj → Prop

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom T1 : ∀ x y : obj, Man x → Man y → Taller x y → CanBlock x y

-- Michael is a man who is taller than everyone else in his class.
axiom T2a : Man Michael
axiom T2b : ∀ y : obj, InMichaelsClass y → Taller Michael y

-- If person x is taller than person y, and person y is taller than person z, then x is taller than z
axiom T3 : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom T4a : Man Peter
axiom T4b : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom T5 : ∀ p : obj, ¬ JumpsWhenShooting p → CanBlock Michael p

-- Michael cannot block Windy's shooting.
axiom T6 : ¬ CanBlock Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom T7 : ∀ p : obj, JumpsWhenShooting p → GreatShooter p

theorem Windy_is_great_shooter : GreatShooter Windy :=
by
  classical
  -- From T5 and T6, deduce that Windy must be able to jump when shooting
  have hjump : JumpsWhenShooting Windy := by
    by_contra hnojump
    exact T6 (T5 Windy hnojump)
  -- Then apply T7
  exact T7 Windy hjump
</lean>

ANSWER: True

theorem Windy_is_great_shooter : GreatShooter Windy :=
by
  classical
  have hem : JumpsWhenShooting Windy ∨ ¬ JumpsWhenShooting Windy := Classical.em (JumpsWhenShooting Windy)
  exact
    match hem with
    | Or.inl hjump => T7 Windy hjump
    | Or.inr hnojump =>
        have hb : CanBlock Michael Windy := T5 Windy hnojump
        have : False := T6 hb
        False.elim this
