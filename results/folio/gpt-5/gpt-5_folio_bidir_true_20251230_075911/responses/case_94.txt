<lean>
axiom obj : Type
axiom Money : Type

-- Entities and functions
axiom OliveGarden : obj
axiom ManagedBuilding : obj → Prop
axiom Rent : obj → Money
axiom Deposit : obj → Money
axiom More : Money → Money → Prop
axiom M2000 : Money

-- Given facts/rules
axiom OG_managed : ManagedBuilding OliveGarden
axiom OG_rent_2000 : Rent OliveGarden = M2000
axiom deposit_rule : ∀ b : obj, ManagedBuilding b → (Deposit b = Rent b ∨ More (Deposit b) (Rent b))

-- Target statement
theorem deposit_OG_2000_or_more : Deposit OliveGarden = M2000 ∨ More (Deposit OliveGarden) M2000 := by
  have h := deposit_rule OliveGarden OG_managed
  cases h with
  | inl h1 =>
      left
      simpa [OG_rent_2000] using h1
  | inr h2 =>
      right
      simpa [OG_rent_2000] using h2
</lean>

ANSWER: True
