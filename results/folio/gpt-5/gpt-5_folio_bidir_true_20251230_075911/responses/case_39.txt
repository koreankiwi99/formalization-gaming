axiom obj : Type

axiom DesignByMax : obj → Prop
axiom MaxAdores : obj → Prop
axiom ZahaHadidStyle : obj → Prop
axiom KellyWearstlerStyle : obj → Prop
axiom InterestingGeometries : obj → Prop
axiom BrutalistBuilding : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop

-- All of Zaha Hadid's design styles that Max adores have interesting geometries.
axiom A1 : ∀ x : obj, ZahaHadidStyle x ∧ MaxAdores x → InterestingGeometries x
-- No brutalist buildings that Max adores have interesting geometries.
axiom A2 : ∀ x : obj, BrutalistBuilding x ∧ MaxAdores x → ¬ InterestingGeometries x
-- Every style that Max adores is either Zaha Hadid's design style or Kelly Wearstler's design style.
axiom A3 : ∀ x : obj, MaxAdores x → (ZahaHadidStyle x ∨ KellyWearstlerStyle x)
-- All of Kelly Wearstler's design styles that Max adores are evocative.
axiom A4 : ∀ x : obj, KellyWearstlerStyle x ∧ MaxAdores x → Evocative x
-- All of Kelly Wearstler's design styles that Max adores are dreamy.
axiom A5 : ∀ x : obj, KellyWearstlerStyle x ∧ MaxAdores x → Dreamy x
-- If a design by Max that he adores has interesting geometries, then it is a brutalist building and evocative.
axiom A6 : ∀ x : obj, DesignByMax x ∧ MaxAdores x ∧ InterestingGeometries x → BrutalistBuilding x ∧ Evocative x

-- A derived consequence: There cannot be a design by Max that he adores which is a Zaha Hadid style.
theorem no_design_by_max_adored_ZH : ∀ x : obj, DesignByMax x ∧ MaxAdores x ∧ ZahaHadidStyle x → False := by
  intro x h
  have hDM : DesignByMax x := And.left h
  have hA_ZH : MaxAdores x ∧ ZahaHadidStyle x := And.right h
  have hA : MaxAdores x := And.left hA_ZH
  have hZH : ZahaHadidStyle x := And.right hA_ZH
  have hI : InterestingGeometries x := A1 x (And.intro hZH hA)
  have hBE : BrutalistBuilding x ∧ Evocative x := A6 x (And.intro hDM (And.intro hA hI))
  have hB : BrutalistBuilding x := And.left hBE
  have hNotI : ¬ InterestingGeometries x := A2 x (And.intro hB hA)
  exact hNotI hI
