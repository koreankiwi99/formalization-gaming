<lean>
axiom obj : Type
axiom Student : obj → Prop
axiom WorkLibrary : obj → Prop
axiom CS : obj → Prop
axiom TakingDatabase : obj → Prop
axiom TakingWithDavid : obj → Prop
axiom WorkLab : obj → Prop
axiom PartTimeUniJob : obj → Prop

axiom James : obj

-- Premises
axiom P1 : ∀ x : obj, Student x ∧ WorkLibrary x → ¬ CS x
axiom P2 : ∀ x : obj, Student x ∧ PartTimeUniJob x → WorkLibrary x
axiom P3 : ∀ x : obj, Student x ∧ TakingDatabase x → CS x
axiom P4 : ∀ x : obj, Student x ∧ TakingWithDavid x → TakingDatabase x
axiom P5 : ∀ x : obj, Student x ∧ WorkLab x → TakingWithDavid x
axiom P6a : Student James
axiom P6b : WorkLab James
axiom P7 : ¬ WorkLab James ∧ ¬ PartTimeUniJob James

theorem james_has_university_part_time_job : PartTimeUniJob James := by
  have hnotWL : ¬ WorkLab James := And.left P7
  have hWL : WorkLab James := P6b
  have hFalse : False := hnotWL hWL
  exact False.elim hFalse
</lean>

ANSWER: True
