<lean>
axiom obj : Type

-- Entities
axiom Badults : obj
axiom AndrewCollins : obj
axiom SDS : obj  -- represents the title "The Secret Dude Society"

-- Predicates
axiom Series : obj → Prop
axiom WorkingTitle : obj → obj → Prop  -- WorkingTitle series title
axiom ScriptEditor : obj → obj → Prop   -- ScriptEditor person series

-- Given facts
axiom T1 : Series Badults
axiom T2 : WorkingTitle Badults SDS
axiom T3 : ScriptEditor AndrewCollins Badults

-- Goal: Andrew Collins was the script editor for a series with the working title "The Secret Dude Society".
theorem stmt : ∃ s : obj, Series s ∧ WorkingTitle s SDS ∧ ScriptEditor AndrewCollins s := by
  refine Exists.intro Badults ?_
  exact And.intro T1 (And.intro T2 T3)
</lean>

ANSWER: True
