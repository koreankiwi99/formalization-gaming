<lean>
axiom obj : Type
axiom Tom : obj

axiom Eastern : obj → Prop
axiom Osceola : obj → Prop
axiom Gould : obj → Prop
axiom Merriam : obj → Prop
axiom RioGrande : obj → Prop
axiom Ocellated : obj → Prop
axiom WildTurkey : obj → Prop

-- Facts about Tom
axiom TomNotEastern : ¬ Eastern Tom
axiom TomNotOsceola : ¬ Osceola Tom
axiom TomNotGould : ¬ Gould Tom
axiom TomNotMerriam : ¬ Merriam Tom
axiom TomNotRioGrande : ¬ RioGrande Tom
axiom TomWild : WildTurkey Tom

-- Classification: every wild turkey is one of the six types
axiom WT_classification :
  ∀ x : obj, WildTurkey x →
    (Eastern x ∨ Osceola x ∨ Gould x ∨ Merriam x ∨ RioGrande x ∨ Ocellated x)

theorem tom_ocellated : Ocellated Tom :=
by
  have h := WT_classification Tom TomWild
  cases h with
  | inl hE => exact False.elim (TomNotEastern hE)
  | inr h1 =>
    cases h1 with
    | inl hOs => exact False.elim (TomNotOsceola hOs)
    | inr h2 =>
      cases h2 with
      | inl hGo => exact False.elim (TomNotGould hGo)
      | inr h3 =>
        cases h3 with
        | inl hMe => exact False.elim (TomNotMerriam hMe)
        | inr h4 =>
          cases h4 with
          | inl hRg => exact False.elim (TomNotRioGrande hRg)
          | inr hOc => exact hOc
</lean>

ANSWER: True
