<lean>
axiom obj : Type
axiom HeinrichSchmidt : obj

axiom GermanPolitician : obj → Prop
axiom MemberPrussianStateParliament : obj → Prop
axiom MemberNaziReichstag : obj → Prop

axiom T1 : GermanPolitician HeinrichSchmidt
axiom T2 : MemberPrussianStateParliament HeinrichSchmidt
axiom T3 : MemberNaziReichstag HeinrichSchmidt

theorem exists_german_politician_in_both :
  ∃ x : obj, GermanPolitician x ∧ MemberPrussianStateParliament x ∧ MemberNaziReichstag x :=
by
  apply Exists.intro HeinrichSchmidt
  exact And.intro T1 (And.intro T2 T3)
</lean>

ANSWER: True
