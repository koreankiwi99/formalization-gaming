<lean>
axiom Person : Type
axiom Tom : Person

axiom GREApplicant : Person → Prop
axiom EconomicHardship : Person → Prop
axiom SingleParentFamily : Person → Prop
axiom FewResources : Person → Prop
axiom ProvidesAid : Person → Prop

-- ETS provides financial aid to those GRE applicants who prove economic hardship
axiom R1 : ∀ x : Person, GREApplicant x ∧ EconomicHardship x → ProvidesAid x
-- Those living in single-parent families or having few resources can prove economic hardship
axiom R2 : ∀ x : Person, (SingleParentFamily x ∨ FewResources x) → EconomicHardship x

-- Tom's facts
axiom T1 : SingleParentFamily Tom
axiom T2 : FewResources Tom
axiom T3 : GREApplicant Tom

theorem ets_provides_aid_to_tom : ProvidesAid Tom := by
  have hEH : EconomicHardship Tom := R2 Tom (Or.inl T1)
  exact R1 Tom (And.intro T3 hEH)
</lean>

ANSWER: True
