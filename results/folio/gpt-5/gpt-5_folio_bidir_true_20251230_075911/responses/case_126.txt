<lean>
axiom obj : Type
axiom Rock : obj

axiom PeterPet : obj → Prop
axiom Mammal : obj → Prop
axiom Animal : obj → Prop
axiom Monkey : obj → Prop
axiom Bird : obj → Prop
axiom CanFly : obj → Prop
axiom CanBreathe : obj → Prop
axiom HasWings : obj → Prop

-- All of Peter's pets that are mammals are animals
axiom A1 : ∀ x : obj, PeterPet x → Mammal x → Animal x
-- All monkeys are mammals
axiom A2 : ∀ x : obj, Monkey x → Mammal x
-- Peter's pets are either monkeys or birds
axiom A3 : ∀ x : obj, PeterPet x → (Monkey x ∨ Bird x)
-- Peter's birds can fly
axiom A4 : ∀ x : obj, PeterPet x → Bird x → CanFly x
-- All animals that are Peter's pets can breathe
axiom A5 : ∀ x : obj, PeterPet x → Animal x → CanBreathe x
-- If Peter's pet can fly, then it has wings
axiom A6 : ∀ x : obj, PeterPet x → CanFly x → HasWings x
-- Rock is Peter's pet
axiom A7 : PeterPet Rock
-- Rock can fly, or Rock is a bird, or Rock cannot breathe
axiom A8 : CanFly Rock ∨ Bird Rock ∨ ¬ CanBreathe Rock

-- From the premises we can derive that Rock has wings
theorem wings_Rock : HasWings Rock :=
by
  apply Or.elim A8
  · intro hFly
    exact A6 Rock A7 hFly
  · intro h
    apply Or.elim h
    · intro hBird
      have hFly : CanFly Rock := A4 Rock A7 hBird
      exact A6 Rock A7 hFly
    · intro hNotBreath
      have hMB : Monkey Rock ∨ Bird Rock := A3 Rock A7
      apply Or.elim hMB
      · intro hMonk
        have hMam : Mammal Rock := A2 Rock hMonk
        have hAni : Animal Rock := A1 Rock A7 hMam
        have hBreath : CanBreathe Rock := A5 Rock A7 hAni
        have : False := hNotBreath hBreath
        exact False.elim this
      · intro hBird
        have hFly : CanFly Rock := A4 Rock A7 hBird
        exact A6 Rock A7 hFly

-- We cannot prove ¬HasWings Rock; in fact, we derived HasWings Rock.
