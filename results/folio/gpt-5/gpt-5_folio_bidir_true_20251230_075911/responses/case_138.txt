axiom Alg : Type
axiom Unsupervised : Alg → Prop
axiom Supervised : Alg → Prop
axiom Reinforcement : Alg → Prop
axiom RequiresLabeled : Alg → Prop
axiom ML : Alg → Prop

-- The specific algorithm used to train the SOTA text summarization model
axiom TrainAlg : Alg

-- Premises:
-- The SOTA model is trained with machine learning algorithms
axiom ML_train : ML TrainAlg
-- Unsupervised learning algorithms do not require labeled data
axiom NoLabelsForUnsup : ∀ a : Alg, Unsupervised a → ¬ RequiresLabeled a
-- Reinforcement learning is not used to train the SOTA model
axiom RL_not_used : ¬ Reinforcement TrainAlg
-- The ML algorithm for training text summarization models requires labeled data
axiom RequiresLabeled_train : RequiresLabeled TrainAlg
-- ML algorithms can be categorized as supervised, unsupervised, or reinforcement
axiom Categorization : ∀ a : Alg, ML a → (Supervised a ∨ Unsupervised a ∨ Reinforcement a)

-- From the premises we can show the training algorithm is not unsupervised
theorem not_unsupervised_trainalg : ¬ Unsupervised TrainAlg := by
  intro hUnsup
  have hNoLabels : ¬ RequiresLabeled TrainAlg := NoLabelsForUnsup TrainAlg hUnsup
  exact hNoLabels RequiresLabeled_train

-- In fact, we can even conclude it is supervised (optional)
theorem supervised_trainalg : Supervised TrainAlg := by
  have h := Categorization TrainAlg ML_train
  cases h with
  | inl hSup => exact hSup
  | inr h' =>
    cases h' with
    | inl hUnsup =>
      have : False := not_unsupervised_trainalg hUnsup
      exact False.elim this
    | inr hRL =>
      have : False := RL_not_used hRL
      exact False.elim this
