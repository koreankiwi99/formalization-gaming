<lean>
axiom obj : Type
axiom Coco : obj
axiom AtRG : obj → Prop
axiom Female : obj → Prop
axiom Male : obj → Prop
axiom LostToIga : obj → Prop
axiom LostToRafa : obj → Prop
axiom RankedHigh : obj → Prop
axiom MostActive : obj → Prop

axiom A1 : ∀ x : obj, RankedHigh x → MostActive x
axiom A2 : ∀ x : obj, LostToIga x → RankedHigh x
axiom A3 : ∀ x : obj, AtRG x ∧ Female x → LostToIga x
axiom A4 : ∀ x : obj, AtRG x → Female x ∨ Male x
axiom A5 : ∀ x : obj, AtRG x ∧ Male x → LostToRafa x
axiom A6 : (RankedHigh Coco ∨ LostToRafa Coco) → ¬ (Male Coco ∧ AtRG Coco)
axiom A7 : AtRG Coco

theorem coco_most_active : MostActive Coco := by
  have hAt : AtRG Coco := A7
  have hNotMale : ¬ Male Coco := by
    intro hMale
    have hAM : AtRG Coco ∧ Male Coco := And.intro hAt hMale
    have hRafa : LostToRafa Coco := A5 Coco hAM
    have hDisj : RankedHigh Coco ∨ LostToRafa Coco := Or.inr hRafa
    have hContra : ¬ (Male Coco ∧ AtRG Coco) := A6 hDisj
    exact hContra (And.intro hMale hAt)
  have hFemOrMale : Female Coco ∨ Male Coco := A4 Coco hAt
  have hFem : Female Coco := by
    cases hFemOrMale with
    | inl hf => exact hf
    | inr hm => exact False.elim (hNotMale hm)
  have hLostIga : LostToIga Coco := A3 Coco (And.intro hAt hFem)
  have hRank : RankedHigh Coco := A2 Coco hLostIga
  exact A1 Coco hRank
</lean>

ANSWER: True
