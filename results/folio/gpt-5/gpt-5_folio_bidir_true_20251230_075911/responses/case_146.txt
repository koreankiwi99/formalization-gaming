<lean>
axiom obj : Type

-- Individuals
axiom Vladimir : obj
axiom Ekaterina : obj

-- Predicates
axiom CanRegisterUS : obj → Prop
axiom Participate2024 : obj → Prop
axiom USCitizen : obj → Prop
axiom TaiwaneseCitizen : obj → Prop
axiom RussianOfficial : obj → Prop
axiom ManagerGazprom : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, CanRegisterUS x → Participate2024 x
axiom A2 : ∀ x : obj, USCitizen x → CanRegisterUS x
axiom A3 : ∀ x : obj, USCitizen x ∨ TaiwaneseCitizen x
axiom A4 : ∀ x : obj, RussianOfficial x → ¬ TaiwaneseCitizen x
axiom A5a : ¬ TaiwaneseCitizen Vladimir
axiom A5b : ¬ ManagerGazprom Vladimir
axiom A6 : CanRegisterUS Ekaterina ∨ RussianOfficial Ekaterina

-- Conclusion
theorem ekaterina_participates_or_manager : Participate2024 Ekaterina ∨ ManagerGazprom Ekaterina := by
  cases A6 with
  | inl hreg =>
      apply Or.inl
      exact A1 Ekaterina hreg
  | inr hOff =>
      have hnotTW : ¬ TaiwaneseCitizen Ekaterina := A4 Ekaterina hOff
      have hdisj : USCitizen Ekaterina ∨ TaiwaneseCitizen Ekaterina := A3 Ekaterina
      cases hdisj with
      | inl hUS =>
          apply Or.inl
          exact A1 Ekaterina (A2 Ekaterina hUS)
      | inr hTW =>
          have hFalse : False := hnotTW hTW
          apply Or.inl
          exact False.elim hFalse
</lean>

ANSWER: True
