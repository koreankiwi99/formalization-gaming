axiom BlackMirror : Show

axiom NetflixShow : Show → Prop
axiom Popular : Show → Prop
axiom BingeKaren : Show → Prop
axiom DownloadKaren : Show → Prop
axiom ShareWithLisaKaren : Show → Prop

-- Facts from the context

axiom ST_netflix : NetflixShow StrangerThings

-- If a Netflix show is popular, Karen will binge-watch it.
axiom binge_of_netflix_and_popular :
  ∀ s : Show, NetflixShow s → Popular s → BingeKaren s

-- If and only if Karen binge-watches a Netflix show, she will download it.
axiom binge_iff_download_on_netflix :
  ∀ s : Show, NetflixShow s → (BingeKaren s ↔ DownloadKaren s)

-- Karen does not download Black Mirror.
axiom not_download_bm : ¬ DownloadKaren BlackMirror

-- Black Mirror is a Netflix show.
axiom BM_netflix : NetflixShow BlackMirror

-- If Karen binge-watches a Netflix show, she will share it with Lisa.
axiom share_if_binge :
  ∀ s : Show, NetflixShow s → BingeKaren s → ShareWithLisaKaren s

-- From the premises we can derive that Black Mirror is not popular.
theorem not_popular_bm : ¬ Popular BlackMirror :=
by
  intro hpop
  have hnet : NetflixShow BlackMirror := BM_netflix
  have hbinge : BingeKaren BlackMirror := binge_of_netflix_and_popular BlackMirror hnet hpop
  have hiff : BingeKaren BlackMirror ↔ DownloadKaren BlackMirror :=
    binge_iff_download_on_netflix BlackMirror hnet
  have hdown : DownloadKaren BlackMirror := (Iff.mp hiff) hbinge
  exact not_download_bm hdown
