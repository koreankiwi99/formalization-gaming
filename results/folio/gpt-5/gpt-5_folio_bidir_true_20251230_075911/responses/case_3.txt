axiom obj : Type
axiom Employee : obj → Prop
axiom SchedulesMeeting : obj → Prop
axiom GoesToBuildingToday : obj → Prop
axiom LunchInBuilding : obj → Prop
axiom LunchAtHome : obj → Prop
axiom RemoteFromHome : obj → Prop
axiom InOtherCountries : obj → Prop
axiom Manager : obj → Prop
axiom James : obj

-- All employees who schedule a meeting with their customers will go to the company building today.
axiom A1 : ∀ x : obj, Employee x ∧ SchedulesMeeting x → GoesToBuildingToday x
-- Everyone who has lunch in the company building schedules meetings with their customers.
axiom A2 : ∀ x : obj, LunchInBuilding x → SchedulesMeeting x
-- Employees have lunch either in the company building or at home.
axiom A3 : ∀ x : obj, Employee x → (LunchInBuilding x ∨ LunchAtHome x)
-- If an employee has lunch at home, they are working remotely from home.
axiom A4 : ∀ x : obj, Employee x ∧ LunchAtHome x → RemoteFromHome x
-- All employees who are in other countries work remotely from home.
axiom A5 : ∀ x : obj, Employee x ∧ InOtherCountries x → RemoteFromHome x
-- No managers work remotely from home.
axiom A6 : ∀ x : obj, Manager x → ¬ RemoteFromHome x
-- James will appear in the company today iff he is a manager (appear = go to building).
axiom A7a : GoesToBuildingToday James → Manager James
axiom A7b : Manager James → GoesToBuildingToday James

-- From these axioms alone, we cannot derive LunchInBuilding James.
-- We are not given that James is an Employee, nor that he is a Manager or appears today.
-- Hence LunchInBuilding James is not provable from the premises.
