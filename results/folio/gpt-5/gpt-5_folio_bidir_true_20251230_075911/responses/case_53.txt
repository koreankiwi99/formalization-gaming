<lean>
axiom obj : Type
axiom French : obj
axiom Spanish : obj
axiom German : obj
axiom Basque : obj

axiom Family : Type
axiom InFamily : obj → Family → Prop
axiom Related : obj → obj → Prop

axiom Romance : Family
axiom IndoEuropean : Family

-- All Romance languages are Indo-European languages.
axiom A1 : ∀ x : obj, InFamily x Romance → InFamily x IndoEuropean
-- All languages within a language family are related to each other.
axiom A2 : ∀ f : Family, ∀ x y : obj, InFamily x f → InFamily y f → Related x y
-- French and Spanish are Romance languages.
axiom A3 : InFamily French Romance
axiom A4 : InFamily Spanish Romance
-- German is related to Spanish.
axiom A5 : Related German Spanish
-- Basque is not related to any other language.
axiom A6 : ∀ y : obj, y ≠ Basque → ¬ Related Basque y

-- Statement to prove: French is an Indo-European language.
theorem french_is_indo_european : InFamily French IndoEuropean :=
  A1 French A3
</lean>

ANSWER: True
