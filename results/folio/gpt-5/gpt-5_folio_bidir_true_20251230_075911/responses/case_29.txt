-- From the premises we can derive that if Marvin is not from Mars, then Marvin is not human.

        have hFalse : False := hNotMars hFM
        exact (fun _ => False.elim hFalse)
    | inr hNeither =>
        exact hNeither.left
  -- Then Marvin is an extraterrestrial

  -- No extraterrestrial is human
  exact P3 Marvin hExt
