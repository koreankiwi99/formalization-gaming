<lean>
axiom Person : Type
axiom Taylor : Person

axiom EatsSalads : Person → Prop
axiom Conscious : Person → Prop
axiom HealthHome : Person → Prop
axiom FulfillNutrition : Person → Prop
axiom DisregardWellbeing : Person → Prop
axiom GymDaily : Person → Prop

-- Premises
axiom A1 : ∀ x : Person, EatsSalads x → Conscious x
axiom A2 : ∀ x : Person, HealthHome x → EatsSalads x
axiom A3 : ∀ x : Person, FulfillNutrition x → HealthHome x
axiom A4 : ∀ x : Person, DisregardWellbeing x → ¬ Conscious x
axiom A5 : ∀ x : Person, GymDaily x → FulfillNutrition x

axiom TaylorCase :
  (HealthHome Taylor ∧ DisregardWellbeing Taylor) ∨ (¬ HealthHome Taylor ∧ ¬ DisregardWellbeing Taylor)

-- The combination HealthHome and DisregardWellbeing leads to a contradiction
theorem H_and_D_impossible : ¬ (HealthHome Taylor ∧ DisregardWellbeing Taylor) := by
  intro hhd
  have h : HealthHome Taylor := And.left hhd
  have d : DisregardWellbeing Taylor := And.right hhd
  have e : EatsSalads Taylor := A2 Taylor h
  have c : Conscious Taylor := A1 Taylor e
  have nc : ¬ Conscious Taylor := A4 Taylor d
  exact nc c

-- From the disjunction, derive ¬ HealthHome Taylor
theorem notH : ¬ HealthHome Taylor := by
  apply Or.elim TaylorCase
  · intro hhd
    intro h
    have d : DisregardWellbeing Taylor := And.right hhd
    have e : EatsSalads Taylor := A2 Taylor h
    have c : Conscious Taylor := A1 Taylor e
    have nc : ¬ Conscious Taylor := A4 Taylor d
    exact nc c
  · intro nhnd
    exact And.left nhnd

-- GymDaily implies HealthHome (via fulfilling nutrition)
theorem gym_implies_home : GymDaily Taylor → HealthHome Taylor := by
  intro g
  have f : FulfillNutrition Taylor := A5 Taylor g
  exact A3 Taylor f

-- Therefore, Taylor does not visit the gym daily
theorem notGym : ¬ GymDaily Taylor := by
  intro g
  have h : HealthHome Taylor := gym_implies_home g
  exact notH h

-- We cannot prove GymDaily Taylor; in fact, we proved its negation.
