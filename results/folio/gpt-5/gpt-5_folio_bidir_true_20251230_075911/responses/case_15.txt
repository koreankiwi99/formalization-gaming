<lean>
axiom obj : Type
axiom Square : obj → Prop
axiom FourSided : obj → Prop
axiom Shape : obj → Prop

-- Context axioms
axiom A1 : ∀ x : obj, Square x → FourSided x
axiom A2 : ∀ x : obj, FourSided x → Shape x

-- Goal: All squares are shapes.
theorem all_squares_are_shapes : ∀ x : obj, Square x → Shape x :=
fun x hx => A2 x (A1 x hx)
</lean>

ANSWER: True
