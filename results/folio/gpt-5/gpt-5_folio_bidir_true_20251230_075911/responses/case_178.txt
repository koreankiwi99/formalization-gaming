axiom obj : Type
axiom Human : obj → Prop
axiom Horse : obj → Prop
axiom Hooves : obj → Prop
axiom all_horses_have_hooves : ∀ x : obj, Horse x → Hooves x
axiom no_humans_have_hooves : ∀ x : obj, Human x → ¬ Hooves x

-- From the premises, we can derive that no human is a horse
theorem no_human_is_horse : ∀ x : obj, Human x → ¬ Horse x :=
  fun x hx => fun hhorse => (no_humans_have_hooves x hx) (all_horses_have_hooves x hhorse)

-- Therefore, the existential statement "Some humans are horses" cannot be proved
-- and is in fact false: there does not exist x with Human x ∧ Horse x.
theorem not_some_humans_are_horses : ¬ (∃ x : obj, Human x ∧ Horse x) :=
  by
    intro h
    cases h with
    | intro x hx =>
      have hx_hum : Human x := And.left hx
      have hx_horse : Horse x := And.right hx
      have : ¬ Horse x := no_human_is_horse x hx_hum
      exact this hx_horse
