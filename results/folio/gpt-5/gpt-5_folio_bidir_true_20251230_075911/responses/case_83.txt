<lean>
axiom obj : Type
axiom HamdenSubway : obj

axiom Listed : obj → Prop
axiom ManyNeg : obj → Prop
axiom RatingGT4 : obj → Prop
axiom Takeout : obj → Prop
axiom Popular : obj → Prop

-- If listed in Yelp’s recommendations, then not many negative reviews
axiom A1 : ∀ x : obj, Listed x → ¬ ManyNeg x
-- All with rating > 4 are listed
axiom A2 : ∀ x : obj, RatingGT4 x → Listed x
-- Some do not provide take-out and receive many negative reviews
axiom A3 : ∃ x : obj, ¬ Takeout x ∧ ManyNeg x
-- Popular among locals implies rating > 4
axiom A4 : ∀ x : obj, Popular x → RatingGT4 x
-- Hamden Plaza Subway has rating > 4 or is popular
axiom A5 : RatingGT4 HamdenSubway ∨ Popular HamdenSubway

theorem query :
  (Takeout HamdenSubway ∧ ManyNeg HamdenSubway) →
  (RatingGT4 HamdenSubway ∧ ¬ Takeout HamdenSubway) := by
  intro h
  have hNeg : ManyNeg HamdenSubway := And.right h
  have hR4orPop : RatingGT4 HamdenSubway ∨ Popular HamdenSubway := A5
  have hR4 : RatingGT4 HamdenSubway :=
    Or.elim hR4orPop
      (fun hR4 => hR4)
      (fun hPop => A4 HamdenSubway hPop)
  have hListed : Listed HamdenSubway := A2 HamdenSubway hR4
  have hNotNeg : ¬ ManyNeg HamdenSubway := A1 HamdenSubway hListed
  have hFalse : False := hNotNeg hNeg
  have hNotTakeout : ¬ Takeout HamdenSubway := fun _ => hFalse
  exact And.intro hR4 hNotTakeout
</lean>

ANSWER: True
