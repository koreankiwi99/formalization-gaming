<lean>
axiom obj : Type
axiom Zelda : obj
axiom Top10 : obj → Prop
axiom SoldMillion : obj → Prop
axiom JapaneseCompany : obj → Prop
axiom CreatedBy : obj → obj → Prop

-- A Japanese game company created the Legend of Zelda
axiom T1 : ∃ c : obj, JapaneseCompany c ∧ CreatedBy c Zelda

-- All games on the Top 10 list are made by Japanese game companies
axiom R1 : ∀ x : obj, Top10 x → ∃ c : obj, JapaneseCompany c ∧ CreatedBy c x

-- If a game sells more than one million copies, then it is included in the Top 10 list
axiom R2 : ∀ x : obj, SoldMillion x → Top10 x

-- The Legend of Zelda sold more than one million copies
axiom T2 : SoldMillion Zelda

theorem zelda_top10 : Top10 Zelda := by
  apply R2 Zelda
  exact T2
</lean>

ANSWER: True
