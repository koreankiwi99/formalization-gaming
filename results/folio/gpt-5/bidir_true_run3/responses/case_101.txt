axiom obj : Type
axiom Luke : obj

axiom MultipleBirth : obj → Prop
axiom PlaySibs : obj → Prop
axiom HaveSibsBornTogether : obj → Prop
axiom ComplainOften : obj → Prop
axiom LiveAtHome : obj → Prop
axiom LiveWithStrangers : obj → Prop

-- People who are born in a multiple birth with siblings spend a lot of time with their siblings.
axiom A1 : ∀ x : obj, MultipleBirth x → PlaySibs x
-- If people have siblings who were born together, then they were born in a multiple birth.
axiom A2 : ∀ x : obj, HaveSibsBornTogether x → MultipleBirth x
-- If people complain often about annoying siblings, then they have siblings who were born together.
axiom A3 : ∀ x : obj, ComplainOften x → HaveSibsBornTogether x
-- If people live at home, then they do not live with strangers.
axiom A4 : ∀ x : obj, LiveAtHome x → ¬ LiveWithStrangers x
-- If people spend a lot of time with their siblings, then they often live at home.
axiom A5 : ∀ x : obj, PlaySibs x → LiveAtHome x

-- Luke either is born in a multiple birth and lives with strangers, or is not born in a multiple birth and does not live with strangers.
axiom A6 : (MultipleBirth Luke ∧ LiveWithStrangers Luke) ∨ (¬ MultipleBirth Luke ∧ ¬ LiveWithStrangers Luke)

-- Derivable helper: being born in a multiple birth implies not living with strangers.
lemma M_implies_notS : ∀ x : obj, MultipleBirth x → ¬ LiveWithStrangers x := by
  intro x hM
  have hPlay : PlaySibs x := A1 x hM
  have hHome : LiveAtHome x := A5 x hPlay
  exact A4 x hHome

-- Hence the left disjunct in A6 is impossible.
lemma left_impossible : ¬ (MultipleBirth Luke ∧ LiveWithStrangers Luke) := by
  intro h
  have hM : MultipleBirth Luke := And.left h
  have hS : LiveWithStrangers Luke := And.right h
  have hNotS : ¬ LiveWithStrangers Luke := M_implies_notS Luke hM
  exact hNotS hS

-- Therefore the right disjunct must hold: Luke is not born in a multiple birth and does not live with strangers.
theorem luke_not_multiple_and_not_strangers : (¬ MultipleBirth Luke ∧ ¬ LiveWithStrangers Luke) := by
  have disj := A6
  apply Or.elim disj
  · intro hleft
    have : False := by
      exact left_impossible hleft
    exact False.elim this
  · intro hright
    exact hright
