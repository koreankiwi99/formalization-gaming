<lean>
axiom obj : Type

axiom SymphonyNo9 : obj
axiom ViennaMusicSociety : obj
axiom Beethoven : obj

axiom MusicPiece : obj → Prop
axiom Orchestra : obj → Prop
axiom Premiered : obj → obj → Prop
axiom Leads : obj → obj → Prop
axiom Composer : obj → Prop
axiom Conductor : obj → Prop
axiom Writes : obj → obj → Prop

-- Facts from the context
axiom T1 : MusicPiece SymphonyNo9
axiom T2 : ∀ x y, Composer x → MusicPiece y → Writes x y
axiom T3 : Writes Beethoven SymphonyNo9
axiom T4 : Premiered ViennaMusicSociety SymphonyNo9
axiom T5 : Orchestra ViennaMusicSociety
axiom T6 : Leads Beethoven ViennaMusicSociety
axiom R1 : ∀ c o, Leads c o → Orchestra o → Conductor c

-- Statement: Some orchestras premiered music pieces.
theorem some_orchestra_premiered_music_piece :
  ∃ o, Orchestra o ∧ ∃ m, MusicPiece m ∧ Premiered o m := by
  refine Exists.intro ViennaMusicSociety ?_
  refine And.intro T5 ?_
  refine Exists.intro SymphonyNo9 ?_
  exact And.intro T1 T4
</lean>

ANSWER: True
