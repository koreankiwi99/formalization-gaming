<lean>
axiom obj : Type
axiom Yuri : obj

axiom ProfessionalBasketballPlayer : obj → Prop
axiom AmericanNational : obj → Prop
axiom CanDoJumpShots : obj → Prop
axiom LeapStraightIntoTheAir : obj → Prop
axiom ActivateLegMuscles : obj → Prop

-- Some professional basketball players are not American nationals
axiom some_non_american_pro : ∃ x : obj, ProfessionalBasketballPlayer x ∧ ¬ AmericanNational x

-- All professional basketball players can do jump shots
axiom pro_can_jump : ∀ x : obj, ProfessionalBasketballPlayer x → CanDoJumpShots x

-- If someone can jump shots, they leap straight into the air
axiom jump_implies_leap : ∀ x : obj, CanDoJumpShots x → LeapStraightIntoTheAir x

-- If someone leaps straight into the air, they activate their leg muscles
axiom leap_implies_activate : ∀ x : obj, LeapStraightIntoTheAir x → ActivateLegMuscles x

-- Yuri does not activate his leg muscles
axiom yuri_not_activate : ¬ ActivateLegMuscles Yuri

theorem yuri_not_american_pro :
  ¬ (AmericanNational Yuri ∧ ProfessionalBasketballPlayer Yuri) :=
by
  intro h
  have hy_pro : ProfessionalBasketballPlayer Yuri := And.right h
  have hy_can : CanDoJumpShots Yuri := pro_can_jump Yuri hy_pro
  have hy_leap : LeapStraightIntoTheAir Yuri := jump_implies_leap Yuri hy_can
  have hy_act : ActivateLegMuscles Yuri := leap_implies_activate Yuri hy_leap
  exact yuri_not_activate hy_act
</lean>

ANSWER: True
