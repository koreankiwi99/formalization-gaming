<lean>
axiom Person : Type
axiom James : Person

axiom Student : Person → Prop
axiom WorksInLab : Person → Prop
axiom WorksInLibrary : Person → Prop
axiom FromCS : Person → Prop
axiom UniPartTimeJob : Person → Prop
axiom TakesDB : Person → Prop
axiom ClassWithDavid : Person → Prop

-- None of the students who currently work in the library are from the CS department.
axiom A1 : ∀ x : Person, WorksInLibrary x → ¬ FromCS x
-- All students with university part-time jobs currently work in the library.
axiom A2 : ∀ x : Person, UniPartTimeJob x → WorksInLibrary x
-- All students who are taking the database course are from the CS department.
axiom A3 : ∀ x : Person, TakesDB x → FromCS x
-- All students taking a class with Professor David are taking the database course.
axiom A4 : ∀ x : Person, ClassWithDavid x → TakesDB x
-- All students working in the lab are taking a class with Professor David.
axiom A5 : ∀ x : Person, WorksInLab x → ClassWithDavid x
-- James is a student working in the lab.
axiom A6 : Student James
axiom A7 : WorksInLab James
-- James doesn't work in the lab or have a part-time job offered by the university.
axiom A8 : ¬ WorksInLab James ∧ ¬ UniPartTimeJob James

theorem james_takes_db : TakesDB James := by
  have hDavid : ClassWithDavid James := A5 James A7
  exact A4 James hDavid
</lean>

ANSWER: True
