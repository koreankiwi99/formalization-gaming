<lean>
axiom obj : Type

axiom Picuris : obj
axiom Juan : obj

axiom MountainRange : obj → Prop
axiom InNewMexico : obj → Prop
axiom InTexas : obj → Prop
axiom Visited : obj → obj → Prop

axiom Mine : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop

-- Context axioms
axiom MR_Picuris : MountainRange Picuris
axiom NM_or_TX : InNewMexico Picuris ∨ InTexas Picuris
axiom Visited_Juan_Picuris : Visited Juan Picuris

axiom Harding : obj
axiom Mine_Harding : Mine Harding
axiom LocatedIn_Harding_Picuris : LocatedIn Harding Picuris
axiom Donated_Harding : Donated Harding

-- No Texas mountain ranges have a donated mine located in them
axiom No_TX_MR_with_donated_mine :
  ∀ x : obj, MountainRange x ∧ InTexas x → ¬ (∃ m : obj, Mine m ∧ LocatedIn m x ∧ Donated m)

theorem juan_visited_mr_in_nm :
  ∃ y : obj, MountainRange y ∧ InNewMexico y ∧ Visited Juan y :=
by
  -- There exists a donated mine located in the Picuris Mountains
  have exMinePicuris : ∃ m : obj, Mine m ∧ LocatedIn m Picuris ∧ Donated m :=
    Exists.intro Harding (And.intro Mine_Harding (And.intro LocatedIn_Harding_Picuris Donated_Harding))

  -- Hence, Picuris is not a Texas mountain range
  have notTexas : ¬ InTexas Picuris :=
    fun htx =>
      let hno := No_TX_MR_with_donated_mine Picuris (And.intro MR_Picuris htx)
      hno exMinePicuris

  -- From "in NM or in TX" and "not in TX", deduce "in NM"
  have inNM : InNewMexico Picuris :=
    Or.elim NM_or_TX (fun h => h) (fun htx => False.elim (notTexas htx))

  -- Conclude existence: Juan visited a mountain range in New Mexico (namely, Picuris)
  exact Exists.intro Picuris (And.intro MR_Picuris (And.intro inNM Visited_Juan_Picuris))
</lean>

ANSWER: True
