axiom obj : Type
axiom ZahaStyle : obj → Prop
axiom KellyStyle : obj → Prop
axiom Brutalist : obj → Prop
axiom Interesting : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop
axiom ByMax : obj → Prop
axiom Adores : obj → Prop
axiom Style : obj → Prop

-- All of Zaha Hadid's design styles that Max adores have interesting geometries.
axiom A1 : ∀ x : obj, ZahaStyle x ∧ Adores x → Interesting x

-- No brutalist buildings that Max adores have interesting geometries.
axiom A2 : ∀ x : obj, Brutalist x ∧ Adores x → ¬ Interesting x

-- Every style that Max adores is either Zaha Hadid's design style or Kelly Wearstler's design style.
axiom A3 : ∀ x : obj, Style x ∧ Adores x → (ZahaStyle x ∨ KellyStyle x)

-- All of Kelly Wearstler's design styles that Max adores are evocative.
axiom A4 : ∀ x : obj, KellyStyle x ∧ Adores x → Evocative x

-- All of Kelly Wearstler's design styles that Max adores are dreamy.
axiom A5 : ∀ x : obj, KellyStyle x ∧ Adores x → Dreamy x

-- If a design by Max that he adores has interesting geometries, then it is a brutalist building and evocative.
axiom A6 : ∀ x : obj, ByMax x ∧ Adores x ∧ Interesting x → Brutalist x ∧ Evocative x

-- From A2 and A6, we can derive that there is no design by Max that he adores which is interesting.
theorem no_interesting_adored_design_by_Max : ∀ x : obj, ByMax x ∧ Adores x ∧ Interesting x → False := by
  intro x h
  have hx_by : ByMax x := And.left h
  have hx_ad_int : Adores x ∧ Interesting x := And.right h
  have hx_ad : Adores x := And.left hx_ad_int
  have hx_int : Interesting x := And.right hx_ad_int
  have hx_brut_ev : Brutalist x ∧ Evocative x := A6 x h
  have hx_brut : Brutalist x := And.left hx_brut_ev
  have not_int : ¬ Interesting x := A2 x (And.intro hx_brut hx_ad)
  exact not_int hx_int

-- ∀ x, ByMax x → Evocative x ∨ Dreamy x

-- and the only rule (A6) that yields Evocative requires Interesting, which leads to contradiction with A2,
-- providing no way to conclude Evocative or Dreamy for arbitrary designs by Max.
-- Therefore, the target statement is not derivable from the premises.
