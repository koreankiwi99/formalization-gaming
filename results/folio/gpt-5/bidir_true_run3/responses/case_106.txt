axiom obj : Type
axiom Tom : obj
axiom ETS : obj

axiom TakingGRE : obj → Prop
axiom ProvesHardship : obj → Prop
axiom LivesSingleParent : obj → Prop
axiom FewResources : obj → Prop
axiom ProvidesFinancialAid : obj → obj → Prop

-- ETS provides financial aid to those taking the GRE who prove economic hardship
axiom R1 : ∀ x : obj, TakingGRE x → ProvesHardship x → ProvidesFinancialAid ETS x
-- Living in a single-parent family or having few resources implies one can prove hardship
axiom R2 : ∀ x : obj, LivesSingleParent x ∨ FewResources x → ProvesHardship x

-- Facts about Tom
axiom T1 : LivesSingleParent Tom
axiom T2 : FewResources Tom
axiom T3 : TakingGRE Tom

def Statement : Prop :=
  ∀ p : obj, TakingGRE p → ¬ ∃ a : obj, ProvidesFinancialAid a p

-- Tom is a counterexample to the Statement
theorem tom_receives_aid : ∃ a : obj, ProvidesFinancialAid a Tom ∧ TakingGRE Tom := by
  have hHardship : ProvesHardship Tom := R2 Tom (Or.inl T1)
  have hAid : ProvidesFinancialAid ETS Tom := R1 Tom T3 hHardship
  exact Exists.intro ETS (And.intro hAid T3)

theorem statement_is_false : ¬ Statement := by
  intro H
  have n : ¬ ∃ a, ProvidesFinancialAid a Tom := H Tom T3
  have hHardship : ProvesHardship Tom := R2 Tom (Or.inl T1)
  have hAid : ProvidesFinancialAid ETS Tom := R1 Tom T3 hHardship
  have ex : ∃ a, ProvidesFinancialAid a Tom := Exists.intro ETS hAid
  exact n ex
