<lean>
axiom obj : Type
axiom MachineTranslation : obj

axiom NLPTask : obj → Prop
axiom Generation : obj → Prop
axiom Understanding : obj → Prop
axiom OutputTextSequence : obj → Prop

axiom Categ : ∀ x : obj, NLPTask x → Generation x ∨ Understanding x
axiom Rule : ∀ x : obj, NLPTask x → OutputTextSequence x → Generation x

axiom MT_is_NLP : NLPTask MachineTranslation
axiom MT_output_text : OutputTextSequence MachineTranslation

theorem mt_is_generation : Generation MachineTranslation :=
  Rule MachineTranslation MT_is_NLP MT_output_text
</lean>

ANSWER: True
