<lean>
axiom obj : Type
axiom KO : obj

axiom Stock : obj → Prop
axiom Growth : obj → Prop
axiom BoughtRapid : obj → Prop
axiom Suitable : obj → Prop
axiom Mature : obj → Prop
axiom Volatile : obj → Prop

-- All growth stocks are bought to earn profits from rapid price appreciation.
axiom A1 : ∀ x : obj, Growth x → BoughtRapid x
-- If a stock is bought for rapid price appreciation, then it is not suitable for a retirement fund.
axiom A2 : ∀ x : obj, BoughtRapid x → ¬ Suitable x
-- Some stocks are growth stocks.
axiom A3 : ∃ x : obj, Stock x ∧ Growth x
-- All mature stocks are suitable for a retirement fund.
axiom A4 : ∀ x : obj, Mature x → Suitable x
-- KO is a mature stock.
axiom A5 : Mature KO

lemma suitable_KO : Suitable KO := A4 KO A5

lemma not_bought_KO : ¬ BoughtRapid KO := by
  intro h
  have ns : ¬ Suitable KO := A2 KO h
  exact ns suitable_KO

lemma not_growth_KO : ¬ Growth KO := by
  intro g
  have br : BoughtRapid KO := A1 KO g
  exact not_bought_KO br

theorem ko_statement :
  (Growth KO ∨ BoughtRapid KO) → (¬ Stock KO ∧ ¬ Volatile KO) := by
  intro h
  have f : False := by
    cases h with
    | inl g => exact not_growth_KO g
    | inr b => exact not_bought_KO b
  exact And.intro (False.elim f) (False.elim f)
</lean>

ANSWER: True
