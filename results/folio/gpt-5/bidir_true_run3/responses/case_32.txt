axiom Person : Type
axiom Mixer : Person → Prop
axiom GrandSlamChampion : Person → Prop
axiom OscarNominatedActor : Person → Prop
axiom ProTennisPlayer : Person → Prop
axiom Celebrity : Person → Prop
axiom Athlete : Person → Prop
axiom Famous : Person → Prop
axiom WellPaid : Person → Prop
axiom TaxHaven : Person → Prop
axiom Djokovic : Person

-- Everyone at the mixer is a Grand Slam champion or an Oscar-nominated actor.
axiom A1 : ∀ x : Person, Mixer x → (GrandSlamChampion x ∨ OscarNominatedActor x)
-- Every Grand Slam champion at the mixer is a professional tennis player.
axiom A2 : ∀ x : Person, Mixer x → (GrandSlamChampion x → ProTennisPlayer x)
-- All Oscar-nominated actors at the mixer are celebrities.
axiom A3 : ∀ x : Person, Mixer x → (OscarNominatedActor x → Celebrity x)
-- All professional tennis players at the mixer are athletes.
axiom A4 : ∀ x : Person, Mixer x → (ProTennisPlayer x → Athlete x)
-- If a person at the mixer is a celebrity, then they are well paid.
axiom A5 : ∀ x : Person, Mixer x → (Celebrity x → WellPaid x)
-- If a person at the mixer is an athlete, then they are famous.
axiom A6 : ∀ x : Person, Mixer x → (Athlete x → Famous x)
-- All well-paid people at the mixer live in tax havens.
axiom A7 : ∀ x : Person, Mixer x → (WellPaid x → TaxHaven x)
-- Djokovic is at the mixer
axiom A8 : Mixer Djokovic
-- If Djokovic is a famous athlete, then Djokovic is well-paid.
axiom A9 : (Famous Djokovic ∧ Athlete Djokovic) → WellPaid Djokovic

-- From the premises, we can derive that Djokovic lives in a tax haven.
theorem djokovic_tax_haven : TaxHaven Djokovic := by
  have m : Mixer Djokovic := A8
  have h : GrandSlamChampion Djokovic ∨ OscarNominatedActor Djokovic := A1 Djokovic m
  cases h with
  | inl g =>
      have t : ProTennisPlayer Djokovic := (A2 Djokovic m) g
      have a : Athlete Djokovic := (A4 Djokovic m) t
      have f : Famous Djokovic := (A6 Djokovic m) a
      have w : WellPaid Djokovic := A9 ⟨f, a⟩
      exact (A7 Djokovic m) w
  | inr o =>
      have c : Celebrity Djokovic := (A3 Djokovic m) o
      have w : WellPaid Djokovic := (A5 Djokovic m) c
      exact (A7 Djokovic m) w
