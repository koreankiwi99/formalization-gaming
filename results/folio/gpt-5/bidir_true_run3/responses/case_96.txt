<lean>
axiom obj : Type
axiom HWBC : obj  -- The Harvard Weekly Book Club

axiom BusinessOrg : obj → Prop
axiom Company : obj → Prop
axiom PrivateCompany : obj → Prop
axiom LegalEntity : obj → Prop
axiom CreatedUnderLaw : obj → Prop
axiom HasLegalObligations : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, BusinessOrg x → LegalEntity x
axiom A2 : ∀ x : obj, Company x → BusinessOrg x
axiom A3 : ∀ x : obj, PrivateCompany x → Company x
axiom A4 : ∀ x : obj, LegalEntity x → CreatedUnderLaw x
axiom A5 : ∀ x : obj, LegalEntity x → HasLegalObligations x
axiom A6 : CreatedUnderLaw HWBC → ¬ PrivateCompany HWBC

-- From the premises we can derive that HWBC is not a private company
theorem not_private_HWBC : ¬ PrivateCompany HWBC := by
  intro hpc
  have hco : Company HWBC := A3 HWBC hpc
  have hbo : BusinessOrg HWBC := A2 HWBC hco
  have hle : LegalEntity HWBC := A1 HWBC hbo
  have hcul : CreatedUnderLaw HWBC := A4 HWBC hle
  have hnpc : ¬ PrivateCompany HWBC := A6 hcul
  exact hnpc hpc

-- Cannot prove: HasLegalObligations HWBC ∧ PrivateCompany HWBC
</lean>
