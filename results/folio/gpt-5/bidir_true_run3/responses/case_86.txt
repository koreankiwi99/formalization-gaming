<lean>
axiom obj : Type

-- Entities
axiom Surprising : obj
axiom SirDigby : obj
axiom Nemesis : obj

-- Predicates/Relations
axiom SuperheroMovie : obj → Prop
axiom NamedAfter : obj → obj → Prop
axiom InMovie : obj → obj → Prop
axiom GoodGuy : obj → Prop
axiom Wins : obj → Prop
axiom Fights : obj → obj → Prop

-- Axioms from the context
axiom A1 : SuperheroMovie Surprising
axiom A2 : NamedAfter Surprising SirDigby
axiom R1 : ∀ m p, SuperheroMovie m → InMovie p m → GoodGuy p → Wins p
axiom R2 : ∀ m p, SuperheroMovie m → NamedAfter m p → GoodGuy p
axiom R3 : ∀ m p, SuperheroMovie m → NamedAfter m p → InMovie p m
axiom T1 : Fights SirDigby Nemesis
axiom R4 : ∀ a b, Fights a b → Wins a → ¬ Wins b

-- Goal: Sir Digby's nemesis does not win.
theorem nemesis_not_win : ¬ Wins Nemesis := by
  have hGood : GoodGuy SirDigby := R2 Surprising SirDigby A1 A2
  have hIn : InMovie SirDigby Surprising := R3 Surprising SirDigby A1 A2
  have hWin : Wins SirDigby := R1 Surprising SirDigby A1 hIn hGood
  exact R4 SirDigby Nemesis T1 hWin
</lean>

ANSWER: True
