<lean>
axiom obj : Type
axiom Hannah : obj

axiom AtSchool : obj → Prop
axiom WorksOnCampus : obj → Prop
axiom NeedsMoney : obj → Prop
axiom OrdersTakeoutFrequently : obj → Prop
axiom EnjoysDiningHall : obj → Prop
axiom NotPickyEater : obj → Prop
axiom SpendsTimeInDiningHalls : obj → Prop
axiom PickyEater : obj → Prop

-- If people at Mary's school work in student jobs on campus, then they need to earn money.
axiom R1 : ∀ x : obj, AtSchool x ∧ WorksOnCampus x → NeedsMoney x
-- If people at Mary's school order takeout frequently in college, then they work in student jobs on campus.
axiom R2 : ∀ x : obj, AtSchool x ∧ OrdersTakeoutFrequently x → WorksOnCampus x
-- People at Mary's school order takeout frequently in college or enjoy the dining hall meals and recipes.
axiom R3 : ∀ x : obj, AtSchool x → (OrdersTakeoutFrequently x ∨ EnjoysDiningHall x)
-- If people at Mary's school enjoy the dining hall meals and recipes, then they are not picky eaters.
axiom R4 : ∀ x : obj, AtSchool x ∧ EnjoysDiningHall x → NotPickyEater x
-- If people at Mary's school enjoy the dining hall meals and recipes, then they spend a lot of their time eating and catching up with friends in the campus dining halls.
axiom R5 : ∀ x : obj, AtSchool x ∧ EnjoysDiningHall x → SpendsTimeInDiningHalls x

-- Hannah is at Mary's school.
axiom T1 : AtSchool Hannah
-- Hannah works in student jobs on campus
-- and if she needs money, then she is neither picky nor needs money.
axiom T2a : WorksOnCampus Hannah
axiom T2b : NeedsMoney Hannah → (NotPickyEater Hannah ∧ ¬ NeedsMoney Hannah)

theorem hannah_statement :
  AtSchool Hannah ∧ (NotPickyEater Hannah ∨ (PickyEater Hannah → SpendsTimeInDiningHalls Hannah)) := by
  have hAt : AtSchool Hannah := T1
  have hWorks : WorksOnCampus Hannah := T2a
  have hNeeds : NeedsMoney Hannah := R1 Hannah ⟨hAt, hWorks⟩
  have hNP_and_notNeeds : NotPickyEater Hannah ∧ ¬ NeedsMoney Hannah := T2b hNeeds
  have hNP : NotPickyEater Hannah := And.left hNP_and_notNeeds
  exact And.intro hAt (Or.inl hNP)
</lean>

ANSWER: True
