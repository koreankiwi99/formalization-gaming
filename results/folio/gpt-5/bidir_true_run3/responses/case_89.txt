<lean>
axiom obj : Type
axiom Harry : obj
axiom Walden : obj

axiom Reads : obj → obj → Prop
axiom GainsKnowledge : obj → Prop
axiom Smarter : obj → Prop

-- When a person reads a book, that person gains knowledge.
axiom R1 : ∀ p b : obj, Reads p b → GainsKnowledge p
-- If a person gains knowledge, they become smarter.
axiom R2 : ∀ p : obj, GainsKnowledge p → Smarter p
-- Harry read the book “Walden”.
axiom T1 : Reads Harry Walden

-- Statement: A smarter person has gained knowledge. (∃ x, Smarter x ∧ GainsKnowledge x)
theorem smarter_person_has_gained : ∃ x : obj, Smarter x ∧ GainsKnowledge x :=
by
  have hk : GainsKnowledge Harry := R1 Harry Walden T1
  have hs : Smarter Harry := R2 Harry hk
  exact Exists.intro Harry (And.intro hs hk)
</lean>

ANSWER: True
