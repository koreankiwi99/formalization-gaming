axiom obj : Type

-- Entities
axiom Picuris : obj
axiom HardingMine : obj

-- Predicates
axiom InNewMexico : obj → Prop
axiom InTexas : obj → Prop
axiom LocatedInRange : obj → obj → Prop
axiom Donated : obj → Prop

-- Facts from the context
axiom Picuris_in_NM_or_TX : InNewMexico Picuris ∨ InTexas Picuris
axiom Harding_located_Picuris : LocatedInRange HardingMine Picuris
axiom Harding_donated : Donated HardingMine
axiom no_TX_range_has_donated_mine :
  ∀ r : obj, InTexas r → ¬ ∃ m : obj, LocatedInRange m r ∧ Donated m

-- From the premises we can derive that Picuris is not in Texas, hence is in New Mexico,
-- and therefore the Harding Mine is located in a mountain range in New Mexico.
theorem harding_in_nm_range : ∃ r : obj, InNewMexico r ∧ LocatedInRange HardingMine r := by
  -- First, show Picuris is not in Texas
  have h_not_tx : ¬ InTexas Picuris := by
    intro htx
    have h_ex : ∃ m : obj, LocatedInRange m Picuris ∧ Donated m := by
      refine Exists.intro HardingMine ?_
      exact And.intro Harding_located_Picuris Harding_donated
    have hnone := no_TX_range_has_donated_mine Picuris htx
    exact hnone h_ex
  -- Therefore, by the given disjunction, Picuris is in New Mexico
  have h_in_nm : InNewMexico Picuris := Or.resolve_right Picuris_in_NM_or_TX h_not_tx
  -- Conclude existence of an NM mountain range containing the Harding Mine
  refine Exists.intro Picuris ?_
  exact And.intro h_in_nm Harding_located_Picuris
