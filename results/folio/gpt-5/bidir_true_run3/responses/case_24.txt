<lean>
axiom obj : Type
axiom Beijing : obj

axiom Hosted2008SummerOlympics : obj → Prop
axiom HostedWinterOlympics : obj → Prop

-- From the context:
-- "Beijing hosted the 2008 Summer Olympics and 2008 Summer Paralympics Games."
axiom A1 : Hosted2008SummerOlympics Beijing
-- "Beijing has hosted the Summer and Winter Olympics and the Summer and Winter Paralympics."
axiom A2 : HostedWinterOlympics Beijing

theorem beijing_hosted_2008_and_winter :
  Hosted2008SummerOlympics Beijing ∧ HostedWinterOlympics Beijing := by
  apply And.intro
  · exact A1
  · exact A2
</lean>

ANSWER: True
