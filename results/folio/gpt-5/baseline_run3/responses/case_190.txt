axiom American : obj → Prop

axiom canjs_implies_leap : ∀ x : obj, CanJS x → Leap x
axiom leap_implies_activate : ∀ x : obj, Leap x → Activate x
axiom yuri_not_activate : ¬ Activate Yuri

-- Derived facts about Yuri
lemma yuri_not_leap : ¬ Leap Yuri := by
  intro hleap
  have hact : Activate Yuri := leap_implies_activate Yuri hleap
  exact yuri_not_activate hact

lemma yuri_not_pro : ¬ ProBB Yuri := by
  intro hpro
  have hjs : CanJS Yuri := pro_implies_canjs Yuri hpro
  have hleap : Leap Yuri := canjs_implies_leap Yuri hjs
  have hact : Activate Yuri := leap_implies_activate Yuri hleap
  exact yuri_not_activate hact

theorem query_false : ¬( (¬ Leap Yuri) → (American Yuri ∧ ProBB Yuri) ) := by
  intro himp
  have hconj : American Yuri ∧ ProBB Yuri := himp yuri_not_leap
  have hpro : ProBB Yuri := hconj.right
  exact yuri_not_pro hpro
