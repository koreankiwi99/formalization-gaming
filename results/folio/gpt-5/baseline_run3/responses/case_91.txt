<lean>
axiom obj : Type
axiom MachineTranslation : obj
axiom NLP : obj → Prop
axiom Generation : obj → Prop
axiom Understanding : obj → Prop
axiom TextOut : obj → Prop

-- Categorization is exclusive: every NLP task is either Generation or Understanding, but not both
axiom exclusive_category :
  ∀ x : obj, NLP x → (Generation x ∧ ¬Understanding x) ∨ (Understanding x ∧ ¬Generation x)

-- If an NLP task's output is a text sequence, then it is a generation task
axiom text_outputs_imply_generation :
  ∀ x : obj, NLP x ∧ TextOut x → Generation x

-- Facts about machine translation
axiom MT_is_NLP : NLP MachineTranslation
axiom MT_text_output : TextOut MachineTranslation

theorem mt_not_understanding : ¬Understanding MachineTranslation := by
  have h_nlp : NLP MachineTranslation := MT_is_NLP
  have h_text : TextOut MachineTranslation := MT_text_output
  have h_gen : Generation MachineTranslation :=
    text_outputs_imply_generation MachineTranslation (And.intro h_nlp h_text)
  have hx := exclusive_category MachineTranslation h_nlp
  cases hx with
  | inl h =>
    exact h.right
  | inr h =>
    have hnotgen := h.right
    have contra : False := hnotgen h_gen
    exact False.elim contra
</lean>

ANSWER: False
