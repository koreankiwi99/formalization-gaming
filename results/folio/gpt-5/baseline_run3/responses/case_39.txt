axiom obj : Type

-- Predicates over objects
axiom Zaha : obj → Prop
axiom Kelly : obj → Prop
axiom Adores : obj → Prop            -- Max adores x
axiom Interesting : obj → Prop
axiom Brutalist : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop
axiom DesignByMax : obj → Prop

-- Given information
axiom H1 : ∀ x : obj, Zaha x ∧ Adores x → Interesting x
axiom H2 : ∀ x : obj, Brutalist x ∧ Adores x → ¬ Interesting x
axiom H3 : ∀ x : obj, Adores x → Zaha x ∨ Kelly x
axiom H4 : ∀ x : obj, Kelly x ∧ Adores x → Evocative x
axiom H5 : ∀ x : obj, Kelly x ∧ Adores x → Dreamy x
axiom H6 : ∀ x : obj, DesignByMax x ∧ Adores x ∧ Interesting x → Brutalist x ∧ Evocative x

-- Useful consequence: there cannot exist a design by Max that he adores which is interesting
theorem no_adored_interesting_designbymax :
  ¬ (∃ x : obj, DesignByMax x ∧ Adores x ∧ Interesting x) :=
by
  intro h
  cases h with
  | intro x hx =>
    -- hx : DesignByMax x ∧ Adores x ∧ Interesting x
    have hb : Brutalist x := (H6 x hx).left
    have ha : Adores x := hx.right.left
    have hnot : ¬ Interesting x := H2 x ⟨hb, ha⟩
    exact hnot hx.right.right

-- Claim in question: ∃ x, DesignByMax x ∧ Brutalist x
-- From the given axioms, we cannot derive existence of such an x, nor can we refute it.
-- No axioms assert existence of any DesignByMax object or any Brutalist object,
