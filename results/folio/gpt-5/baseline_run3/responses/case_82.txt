axiom obj : Type

-- Entities and predicates
axiom DriedThaiChilies : obj
axiom Spicy : obj → Prop
axiom BakedSweet : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom FromBBM : obj → Prop

-- Given axioms from the context
axiom no_baked_sweets_spicy : ∀ x : obj, BakedSweet x → ¬ Spicy x
axiom cupcakes_are_baked_sweets : ∀ x : obj, Cupcake x → BakedSweet x
axiom mala_hotpots_spicy : ∀ x : obj, MalaHotpot x → Spicy x
axiom bbm_products_are_cupcakes : ∀ x : obj, FromBBM x → Cupcake x
axiom dtc_disj : Spicy DriedThaiChilies ∨ MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies

-- Derive that Dried Thai chilies are not baked sweets
theorem dtc_not_baked : ¬ BakedSweet DriedThaiChilies := by
  cases dtc_disj with
  | inl h1 =>
    cases h1 with
    | inl hSpicy =>
      intro hBaked
      have hNotSpicy := no_baked_sweets_spicy DriedThaiChilies hBaked
      exact hNotSpicy hSpicy
    | inr hMala =>
      intro hBaked
      have hSpicy := mala_hotpots_spicy DriedThaiChilies hMala
      have hNotSpicy := no_baked_sweets_spicy DriedThaiChilies hBaked
      exact hNotSpicy hSpicy
  | inr hNotBaked =>
    exact hNotBaked

-- Hence they are not cupcakes
theorem dtc_not_cupcake : ¬ Cupcake DriedThaiChilies := by
  intro hCup
  have hBaked := cupcakes_are_baked_sweets DriedThaiChilies hCup
  exact dtc_not_baked hBaked

-- Therefore, they cannot be both cupcakes and products of Baked by Melissa
theorem dtc_not_cupcake_and_bbm : ¬ (Cupcake DriedThaiChilies ∧ FromBBM DriedThaiChilies) := by
  intro h
  exact dtc_not_cupcake h.left
