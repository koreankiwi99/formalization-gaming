<lean>
axiom obj : Type
axiom HeinrichSchmidt : obj
axiom GermanPolitician : obj → Prop
axiom MemberPrussianStateParliament : obj → Prop
axiom MemberNaziReichstag : obj → Prop

axiom HS_german : GermanPolitician HeinrichSchmidt
axiom HS_prussian : MemberPrussianStateParliament HeinrichSchmidt
axiom HS_reichstag : MemberNaziReichstag HeinrichSchmidt

theorem some_german_politician_in_both :
  ∃ x : obj, GermanPolitician x ∧ MemberPrussianStateParliament x ∧ MemberNaziReichstag x := by
  apply Exists.intro HeinrichSchmidt
  apply And.intro
  · exact HS_german
  · apply And.intro
    · exact HS_prussian
    · exact HS_reichstag
</lean>

ANSWER: True
