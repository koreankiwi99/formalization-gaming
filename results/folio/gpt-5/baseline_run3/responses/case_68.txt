<lean>
axiom Series : Type
axiom Person : Type
axiom Title : Type

axiom Badults : Series
axiom AndrewCollins : Person
axiom SecretDudeSociety : Title

axiom ScriptEditor : Person → Series → Prop
axiom WorkingTitle : Series → Title → Prop

-- Context axioms
axiom WT : WorkingTitle Badults SecretDudeSociety
axiom SE : ScriptEditor AndrewCollins Badults

-- Claim: There exists a series with working title "The Secret Dude Society" for which Andrew Collins was the script editor.
theorem collins_editor_for_secretdude_series :
  ∃ s : Series, WorkingTitle s SecretDudeSociety ∧ ScriptEditor AndrewCollins s := by
  refine ⟨Badults, And.intro WT SE⟩
</lean>

ANSWER: True
