<lean>
axiom obj : Type

axiom At : obj → Prop
axiom Works : obj → Prop
axiom Needs : obj → Prop
axiom Orders : obj → Prop
axiom Enjoys : obj → Prop
axiom Picky : obj → Prop
axiom Spends : obj → Prop

axiom Hannah : obj

axiom A1 : ∀ x : obj, At x → Works x → Needs x
axiom A2 : ∀ x : obj, At x → Orders x → Works x
axiom A3 : ∀ x : obj, At x → (Orders x ∨ Enjoys x)
axiom A4 : ∀ x : obj, At x → Enjoys x → ¬ Picky x
axiom A5 : ∀ x : obj, At x → Enjoys x → Spends x

axiom A6 : At Hannah
axiom A7a : Works Hannah
axiom A7b : Needs Hannah → (¬ Picky Hannah ∧ ¬ Needs Hannah)

theorem hannah_statement : At Hannah ∧ ¬ Picky Hannah ∧ Spends Hannah := by
  have hAt : At Hannah := A6
  have hWork : Works Hannah := A7a
  have hNeeds : Needs Hannah := A1 Hannah hAt hWork
  have hnp : ¬ Picky Hannah ∧ ¬ Needs Hannah := A7b hNeeds
  have hNotPicky : ¬ Picky Hannah := And.left hnp
  have hNotNeeds : ¬ Needs Hannah := And.right hnp
  have hFalse : False := hNotNeeds hNeeds
  have hSpends : Spends Hannah := False.elim hFalse
  exact And.intro hAt (And.intro hNotPicky hSpends)
</lean>

ANSWER: True
