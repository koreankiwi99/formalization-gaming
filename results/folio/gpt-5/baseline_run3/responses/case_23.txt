axiom obj : Type

-- Shows
axiom ST : obj
axiom BM : obj

-- Predicates
axiom NetflixShow : obj → Prop
axiom Popular : obj → Prop
axiom BingeKaren : obj → Prop
axiom DownloadKaren : obj → Prop
axiom ShareWithLisa : obj → Prop

-- Facts from the context
axiom ST_netflix : NetflixShow ST
axiom ST_popular : Popular ST

-- If a Netflix show is popular, Karen will binge-watch it.
axiom R_binge_if_popular_netflix :
  ∀ x : obj, NetflixShow x → Popular x → BingeKaren x

-- If and only if Karen binge-watches a Netflix show, she will download it.
axiom R_binge_download_iff :
  ∀ x : obj, NetflixShow x → (BingeKaren x ↔ DownloadKaren x)

-- "Black Mirror" is a Netflix show.
axiom BM_netflix : NetflixShow BM

-- Karen does not download "Black Mirror."
axiom BM_not_download : ¬ DownloadKaren BM

-- If Karen binge-watches a Netflix show, she will share it with Lisa.
axiom R_share_if_binge :
  ∀ x : obj, NetflixShow x → BingeKaren x → ShareWithLisa x

-- Derivable: Karen does not binge-watch "Black Mirror."
theorem BM_not_binge : ¬ BingeKaren BM := by
  intro hbinge
  have hiff : BingeKaren BM ↔ DownloadKaren BM := R_binge_download_iff BM BM_netflix
  have hdownload : DownloadKaren BM := (Iff.mp hiff) hbinge
  exact BM_not_download hdownload

-- From the given information, we cannot prove ShareWithLisa BM or its negation.
-- Hence the truth of "Karen will share 'Black Mirror' with Lisa" is undecidable from the premises.
