axiom A1 : ∀ x : obj, DeadlyDisease x → LowSurvival x
-- Severe cancers are deadly diseases.
axiom A2 : ∀ x : obj, SevereCancer x → DeadlyDisease x
-- Bile duct cancer is a severe form cancer.
axiom A3 : ∀ x : obj, BileDuctCancer x → SevereCancer x
-- All Cholangiocarcinoma is bile duct cancer.
axiom A4 : ∀ x : obj, Cholangiocarcinoma x → BileDuctCancer x
-- Mild flu comes with a low survival rate.
axiom A5 : LowSurvival MildFlu
-- Colorectal cancer is not both a bile duct cancer and with a low survival rate.
axiom A6 : ¬(BileDuctCancer ColorectalCancer ∧ LowSurvival ColorectalCancer)

-- Useful derived lemma: bile duct cancer implies low survival (via severe → deadly → low survival)
theorem bdc_implies_low : ∀ x : obj, BileDuctCancer x → LowSurvival x := by
  intro x
  intro hbdc
  have hs : SevereCancer x := A3 x hbdc
  have hd : DeadlyDisease x := A2 x hs
  have hl : LowSurvival x := A1 x hd
  exact hl

-- From A6 and the above, we can conclude colorectal cancer is not a bile duct cancer.

  intro hbdc
  have hl : LowSurvival ColorectalCancer := bdc_implies_low ColorectalCancer hbdc
  exact A6 (And.intro hbdc hl)
