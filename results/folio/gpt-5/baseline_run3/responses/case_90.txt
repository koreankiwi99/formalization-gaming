<lean>
axiom Task : Type
axiom MT : Task
axiom NLPTask : Task → Prop
axiom LangGen : Task → Prop
axiom LangUnderstand : Task → Prop
axiom OutputText : Task → Prop

axiom cat_either : ∀ x : Task, NLPTask x → (LangGen x ∨ LangUnderstand x)
axiom textseq_gen : ∀ x : Task, NLPTask x → OutputText x → LangGen x
axiom mt_is_nlp : NLPTask MT
axiom mt_output_text : OutputText MT

theorem mt_is_generation : LangGen MT := by
  have hN : NLPTask MT := mt_is_nlp
  have hT : OutputText MT := mt_output_text
  exact textseq_gen MT hN hT
</lean>
