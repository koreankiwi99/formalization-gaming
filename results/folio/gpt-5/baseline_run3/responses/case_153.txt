axiom obj : Type
axiom Mix : obj
axiom Hydrocarbon : obj → Prop
axiom Organic : obj → Prop
axiom Chemical : obj → Prop
axiom ContainsCarbon : obj → Prop
axiom OnlyOneElement : obj → Prop
axiom Alkane : obj → Prop

axiom A1 : ∀ x : obj, Hydrocarbon x → Organic x
axiom A2 : ∀ x : obj, Alkane x → Hydrocarbon x
axiom A3 : ∀ x : obj, Organic x → Chemical x
axiom A4 : ∀ x : obj, Organic x → ContainsCarbon x
axiom A5 : ∀ x : obj, Chemical x → ¬ OnlyOneElement x
axiom A6 : (Chemical Mix ∧ OnlyOneElement Mix) ∨ (¬ Chemical Mix ∧ ¬ OnlyOneElement Mix)

-- From A6 and A5, the mixture cannot be a chemical compound.
theorem mix_not_chemical : ¬ Chemical Mix := by
  intro hchem
  have honeFalse : ¬ OnlyOneElement Mix := A5 Mix hchem
  cases A6 with
  | inl hboth =>
      exact honeFalse hboth.right
  | inr hneither =>
      exact hneither.left hchem

-- Alkanes imply chemical compounds via hydrocarbons and organic compounds.
theorem alkane_implies_chemical : ∀ x : obj, Alkane x → Chemical x := by
  intro x halk
  have hhydro : Hydrocarbon x := A2 x halk
  have horg : Organic x := A1 x hhydro
  exact A3 x horg

-- Hence the mixture is not an alkane.
theorem mix_not_alkane : ¬ Alkane Mix := by
  intro halk
  have hchem : Chemical Mix := alkane_implies_chemical Mix halk
  exact mix_not_chemical hchem

-- Therefore, the conjunction "Alkane Mix ∧ ContainsCarbon Mix" is false.
theorem mixture_not_alkane_and_contains : ¬ (Alkane Mix ∧ ContainsCarbon Mix) := by
  intro h
  exact mix_not_alkane h.left
