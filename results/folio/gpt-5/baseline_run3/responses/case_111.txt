<lean>
axiom obj : Type

-- Entities
axiom Picuris : obj
axiom Juan : obj
axiom Harding : obj
axiom NewMexico : obj
axiom Texas : obj

-- Predicates and relations
axiom MountainRange : obj → Prop
axiom Mine : obj → Prop
axiom Donated : obj → Prop
axiom Visited : obj → obj → Prop
axiom In : obj → obj → Prop
axiom LocatedIn : obj → obj → Prop

-- Facts from the context
axiom A1 : MountainRange Picuris
axiom A2 : In Picuris NewMexico ∨ In Picuris Texas
axiom A3 : Visited Juan Picuris
axiom A4 : Mine Harding
axiom A5 : LocatedIn Harding Picuris
axiom A6 : Donated Harding

-- Rule: No Texas mountain range has a mine located in it that was donated
axiom A7 : ∀ x : obj, MountainRange x → In x Texas → ∀ m : obj, Mine m → LocatedIn m x → ¬ Donated m

-- Claim: Juan de Onate visited a mountain range in New Mexico
theorem visited_nm : ∃ x : obj, MountainRange x ∧ In x NewMexico ∧ Visited Juan x := by
  have h_not_texas : ¬ In Picuris Texas := by
    intro h_in_tx
    have h_forall := A7 Picuris A1 h_in_tx
    have h_neg := h_forall Harding A4 A5
    exact h_neg A6
  have h_in_nm : In Picuris NewMexico := by
    cases A2 with
    | inl hnm => exact hnm
    | inr htx => exact False.elim (h_not_texas htx)
  exact Exists.intro Picuris (And.intro A1 (And.intro h_in_nm A3))
</lean>

ANSWER: True
