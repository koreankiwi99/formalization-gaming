<lean>
axiom obj : Type
axiom Tom : obj

axiom NiceToAnimals : obj → Prop
axiom MeanToAnimals : obj → Prop
axiom Grumpy : obj → Prop
axiom AnimalLover : obj → Prop
axiom PetOwner : obj → Prop

-- No one nice to animals is also mean to animals.
axiom A1 : ∀ x, NiceToAnimals x → ¬ MeanToAnimals x
-- Some grumpy people are mean to animals.
axiom A2 : ∃ x, Grumpy x ∧ MeanToAnimals x
-- All animal lovers are nice to animals.
axiom A3 : ∀ x, AnimalLover x → NiceToAnimals x
-- All pet owners love animals.
axiom A4 : ∀ x, PetOwner x → AnimalLover x
-- Tom is a pet owner.
axiom A5 : PetOwner Tom

-- From the axioms, Tom is an animal lover and thus not mean to animals.
theorem tom_animal_lover : AnimalLover Tom := A4 Tom A5
theorem tom_not_mean : ¬ MeanToAnimals Tom :=
  A1 Tom (A3 Tom tom_animal_lover)

-- The questioned statement:
-- If Tom is not both a grumpy person and mean to animals,
-- then Tom is neither a grumpy person nor an animal lover.
def Statement :=
  (¬ (Grumpy Tom ∧ MeanToAnimals Tom)) → (¬ Grumpy Tom ∧ ¬ AnimalLover Tom)

-- We show the statement is false by proving its negation.
theorem statement_false : ¬ Statement := by
  intro h
  -- antecedent is true because Tom is not mean to animals
  have hAnte : ¬ (Grumpy Tom ∧ MeanToAnimals Tom) := by
    intro hgm
    exact tom_not_mean hgm.right
  -- apply the implication to get the consequent
  have hCons : ¬ Grumpy Tom ∧ ¬ AnimalLover Tom := h hAnte
  -- but Tom is an animal lover, contradiction
  exact (And.right hCons) tom_animal_lover
</lean>

ANSWER: False
