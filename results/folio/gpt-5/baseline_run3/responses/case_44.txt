axiom Coco : obj

-- Predicates
axiom RankedHighlyWTA : obj → Prop
axiom ActiveMajor : obj → Prop
axiom LostIgaRG22 : obj → Prop
axiom FemaleRG22 : obj → Prop
axiom MaleRG22 : obj → Prop
axiom PlayerRG22 : obj → Prop
axiom LostNadalRG22 : obj → Prop

-- Axioms from the context
axiom R1 : ∀ x : obj, RankedHighlyWTA x → ActiveMajor x
axiom R2 : ∀ x : obj, LostIgaRG22 x → RankedHighlyWTA x
axiom R3 : ∀ x : obj, FemaleRG22 x → LostIgaRG22 x
axiom R4 : ∀ x : obj, PlayerRG22 x → FemaleRG22 x ∨ MaleRG22 x
axiom R5 : ∀ x : obj, MaleRG22 x → LostNadalRG22 x
axiom R6 : (RankedHighlyWTA Coco ∨ LostNadalRG22 Coco) → ¬ MaleRG22 Coco
axiom T1 : PlayerRG22 Coco

-- We prove that Coco both lost to Iga at RG22 and is one of the most active players
theorem coco_both : LostIgaRG22 Coco ∧ ActiveMajor Coco := by
  -- First, show Coco is not male at RG22
  have notMale : ¬ MaleRG22 Coco := by
    intro hM
    have hN : LostNadalRG22 Coco := R5 Coco hM
    have hdisj : RankedHighlyWTA Coco ∨ LostNadalRG22 Coco := Or.inr hN
    have hnm : ¬ MaleRG22 Coco := R6 hdisj
    exact hnm hM
  -- From being a player at RG22, Coco is female or male; but not male, so female
  have hfm_or_mm : FemaleRG22 Coco ∨ MaleRG22 Coco := R4 Coco T1
  have hF : FemaleRG22 Coco := by
    cases hfm_or_mm with
    | inl hf => exact hf
    | inr hm => exact False.elim (notMale hm)
  -- Therefore Coco lost to Iga
  have hLI : LostIgaRG22 Coco := R3 Coco hF
  -- Hence Coco is ranked highly, thus one of the most active players
  have hRH : RankedHighlyWTA Coco := R2 Coco hLI
  have hA : ActiveMajor Coco := R1 Coco hRH
  exact And.intro hLI hA
