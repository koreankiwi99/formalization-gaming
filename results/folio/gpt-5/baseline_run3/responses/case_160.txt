<lean>
axiom obj : Type
axiom George : obj

axiom LikesMusic : obj → Prop
axiom WantsCompose : obj → Prop
axiom CanCompose : obj → Prop
axiom AccessProgram : obj → Prop
axiom Compose : obj → Prop

-- If George likes music, he wants to compose.
axiom R1 : LikesMusic George → WantsCompose George
-- If George has access to a program, he can compose.
axiom R2 : AccessProgram George → CanCompose George
-- If George can compose and wants to compose, he will compose.
axiom R3 : CanCompose George ∧ WantsCompose George → Compose George

-- If George likes music and has access to a program, he will compose.
theorem george_will_compose :
  (LikesMusic George ∧ AccessProgram George) → Compose George := by
  intro h
  have hL : LikesMusic George := And.left h
  have hA : AccessProgram George := And.right h
  have hW : WantsCompose George := R1 hL
  have hC : CanCompose George := R2 hA
  exact R3 ⟨hC, hW⟩
</lean>

ANSWER: True
