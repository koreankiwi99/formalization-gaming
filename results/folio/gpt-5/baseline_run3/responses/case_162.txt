<lean>
axiom obj : Type
axiom Bird : obj → Prop
axiom Hawk : obj → Prop
axiom Swims : obj → Prop

-- A hawk never swims.
axiom hawk_never_swims : ∀ x : obj, Hawk x → ¬ Swims x
-- Some birds are hawks.
axiom some_bird_is_hawk : ∃ x : obj, Bird x ∧ Hawk x

-- The claim "All birds swim" is false.
theorem not_all_birds_swim : ¬ (∀ x : obj, Bird x → Swims x) :=
by
  intro h
  cases some_bird_is_hawk with
  | intro b hb_hh =>
    have hb : Bird b := And.left hb_hh
    have hh : Hawk b := And.right hb_hh
    have hs : Swims b := h b hb
    have hns : ¬ Swims b := hawk_never_swims b hh
    exact hns hs
</lean>

ANSWER: False
