<lean>
axiom obj : Type
axiom Zelda : obj
axiom MadeByJapaneseCompany : obj → Prop
axiom OnTop10 : obj → Prop
axiom SoldOverOneMillion : obj → Prop

-- Context axioms
axiom C1 : MadeByJapaneseCompany Zelda
axiom C2 : ∀ x : obj, OnTop10 x → MadeByJapaneseCompany x
axiom C3 : ∀ x : obj, SoldOverOneMillion x → OnTop10 x
axiom C4 : SoldOverOneMillion Zelda

theorem zelda_on_top10 : OnTop10 Zelda := by
  exact C3 Zelda C4
</lean>

ANSWER: True
