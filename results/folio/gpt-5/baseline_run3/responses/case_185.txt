axiom Podcast : obj → Prop
axiom Novel : obj → Prop
axiom Book : obj → Prop
axiom Person : obj → Prop
axiom WrittenBy : obj → obj → Prop
axiom NovelWriter : obj → Prop
axiom American : obj → Prop
axiom AmericanCity : obj → Prop
axiom BornIn : obj → obj → Prop
axiom CreatedBy : obj → obj → Prop
axiom Writer : obj → Prop

-- Entities
axiom DaniShapiro : obj
axiom FamilyHistory : obj
axiom FamilySecrets : obj
axiom Boston : obj

-- Axioms from context
-- A podcast is not a novel.
axiom A1 : ∀ x : obj, Podcast x → ¬ Novel x
-- If a person is born in an American city, the person is American.
axiom A2 : ∀ p c : obj, Person p → AmericanCity c → BornIn p c → American p
-- If a book is a novel and it is written by a person, then the person is a novel writer.
axiom A3 : ∀ b p : obj, Book b → Novel b → Person p → WrittenBy b p → NovelWriter p
-- Dani Shapiro is an American writer.
axiom A4a : American DaniShapiro
axiom A4b : Writer DaniShapiro
axiom A4c : Person DaniShapiro
-- Family History is written by Dani Shapiro.
axiom A5 : WrittenBy FamilyHistory DaniShapiro
-- Family History is a novel written in 2003.
axiom A6 : Novel FamilyHistory
-- Dani Shapiro created a podcast called Family Secrets.
axiom A7a : Podcast FamilySecrets
axiom A7b : CreatedBy FamilySecrets DaniShapiro
-- Boston is an American city.
axiom A8 : AmericanCity Boston

-- From "A podcast is not a novel" and "Family Secrets is a podcast"
-- we conclude that Family Secrets is not a novel.
theorem family_secrets_not_novel : ¬ Novel FamilySecrets := A1 FamilySecrets A7a
