axiom CanCompose : obj → Prop

-- If George has access to a program, he can compose.
axiom A2 : HasAccessProgram George → CanCompose George
-- If George can compose and wants to compose, he will compose.
axiom A3 : CanCompose George ∧ WantsToCompose George → WillCompose George

-- From not (will compose), we can derive not (can compose and wants to compose).
-- This does not imply not (can compose) alone.
lemma not_will_implies_not_can_and_wants :
  ¬ WillCompose George → ¬ (CanCompose George ∧ WantsToCompose George) := by
  intro hnot
  intro hboth
  have hw : WillCompose George := A3 hboth
  exact hnot hw
