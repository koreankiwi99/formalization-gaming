axiom Cat : obj → Prop
axiom Mammal : obj → Prop
axiom Pet : obj → Prop

-- All cats are mammals.
axiom allCatsMammals : ∀ x : obj, Cat x → Mammal x
-- Some pets are not mammals.
axiom somePetNotMammal : ∃ x : obj, Pet x ∧ ¬Mammal x

-- From the given information we can derive that there exists a pet that is not a cat.
theorem exists_pet_not_cat : ∃ x : obj, Pet x ∧ ¬Cat x :=
  match somePetNotMammal with
  | ⟨x, ⟨hxP, hxNotM⟩⟩ =>
      have hNotCat : ¬Cat x := fun hxCat => hxNotM (allCatsMammals x hxCat)
      ⟨x, ⟨hxP, hNotCat⟩⟩
