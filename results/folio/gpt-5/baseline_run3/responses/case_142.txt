<lean>
axiom obj : Type
axiom Animal : obj → Prop
axiom HeckCattle : obj → Prop
axiom Aurochs : obj → Prop
axiom Extinct : obj → Prop
axiom Resemble : obj → obj → Prop
axiom BredBack : obj → Prop
axiom ArtificialSelection : obj → Prop

-- Breeding back is a form of artificial selection
axiom R_bb_as : ∀ x : obj, BredBack x → ArtificialSelection x

-- Heck cattle are animals
axiom T_hc_anim : ∀ x : obj, HeckCattle x → Animal x

-- Aurochs are animals
axiom T_aur_anim : ∀ x : obj, Aurochs x → Animal x

-- Some animals to be bred back resemble extinct animals
axiom E_some_anim_bb_resemble_extinct :
  ∃ x : obj, Animal x ∧ BredBack x ∧ ∃ y : obj, Animal y ∧ Extinct y ∧ Resemble x y

-- Heck cattle were bred back (existential reading)
axiom E_some_hc_bredback : ∃ x : obj, HeckCattle x ∧ BredBack x

theorem some_hc_artificial : ∃ x : obj, HeckCattle x ∧ ArtificialSelection x := by
  apply Exists.elim E_some_hc_bredback
  intro x hx
  have hxhc : HeckCattle x := And.left hx
  have hxbb : BredBack x := And.right hx
  have hxa : ArtificialSelection x := R_bb_as x hxbb
  exact Exists.intro x (And.intro hxhc hxa)
</lean>

ANSWER: True
