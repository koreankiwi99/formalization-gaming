<lean>
axiom Show : Type
axiom Person : Type

axiom Karen : Person
axiom Lisa : Person

axiom StrangerThings : Show
axiom BlackMirror : Show

axiom Popular : Show → Prop
axiom NetflixShow : Show → Prop
axiom Binge : Person → Show → Prop
axiom Download : Person → Show → Prop
axiom Share : Person → Show → Person → Prop

-- Facts about shows
axiom ST_popular : Popular StrangerThings
axiom ST_netflix : NetflixShow StrangerThings

-- Rules
axiom R1 : ∀ s : Show, NetflixShow s → Popular s → Binge Karen s
axiom R2 : ∀ s : Show, NetflixShow s → (Download Karen s ↔ Binge Karen s)
axiom R3 : ∀ s : Show, NetflixShow s → Binge Karen s → Share Karen s Lisa

-- Facts about Black Mirror
axiom BM_not_download : ¬ Download Karen BlackMirror
axiom BM_netflix : NetflixShow BlackMirror

theorem karen_shares_stranger_things_with_lisa : Share Karen StrangerThings Lisa := by
  have hBinge : Binge Karen StrangerThings := R1 StrangerThings ST_netflix ST_popular
  exact R3 StrangerThings ST_netflix hBinge
</lean>

ANSWER: True
