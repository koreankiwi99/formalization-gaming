<lean>
axiom obj : Type
axiom Horse : obj → Prop
axiom Human : obj → Prop
axiom Hooves : obj → Prop

-- All horses have hooves.
axiom A1 : ∀ x : obj, Horse x → Hooves x
-- No humans have hooves.
axiom A2 : ∀ x : obj, Human x → ¬ Hooves x

theorem no_human_is_horse : ∀ x : obj, Human x → ¬ Horse x := by
  intro x
  intro hx
  intro hh
  have hhoof : Hooves x := A1 x hh
  have nhhoof : ¬ Hooves x := A2 x hx
  exact nhhoof hhoof

-- Therefore, it is false that some humans are horses.
theorem not_some_humans_are_horses : ¬ (∃ x : obj, Human x ∧ Horse x) := by
  intro h
  cases h with
  | intro x hx =>
    have nh : ¬ Horse x := no_human_is_horse x hx.left
    exact nh hx.right
</lean>

ANSWER: False
