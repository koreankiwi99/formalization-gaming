<lean>
axiom obj : Type
axiom John : obj

axiom MakePlans : obj → Prop
axiom Engaged : obj → Prop
axiom InviteOthers : obj → Prop
axiom WellAttended : obj → Prop
axiom LargeFamily : obj → Prop
axiom InviteFriends : obj → Prop

-- 1. All who make wedding plans are engaged.
axiom A1 : ∀ x : obj, MakePlans x → Engaged x
-- 2. All who invite others to come to their ceremony make wedding plans.
axiom A2 : ∀ x : obj, InviteOthers x → MakePlans x
-- 3. Anyone who has a well-attended wedding invites others.
axiom A3 : ∀ x : obj, WellAttended x → InviteOthers x
-- 4. Some people that have well-attended weddings have larger families.
axiom A4 : ∃ x : obj, WellAttended x ∧ LargeFamily x
-- 5. If John is engaged, then John does not invite friends and does not make wedding plans.
axiom A5 : Engaged John → (¬ InviteFriends John ∧ ¬ MakePlans John)
-- 6. If John has a larger family, then either John has a well-attended wedding or invites friends.
axiom A6 : LargeFamily John → (WellAttended John ∨ InviteFriends John)

-- Derived: well-attended implies engaged (for any person)
theorem wellAttended_implies_engaged : ∀ x : obj, WellAttended x → Engaged x := by
  intro x hW
  have hIO : InviteOthers x := A3 x hW
  have hMP : MakePlans x := A2 x hIO
  exact A1 x hMP

-- For John specifically, well-attended leads to a contradiction; hence not well-attended.
theorem not_wellAttended_John : ¬ WellAttended John := by
  intro hW
  have hIO : InviteOthers John := A3 John hW
  have hMP : MakePlans John := A2 John hIO
  have hEng : Engaged John := A1 John hMP
  have hNoMP : ¬ MakePlans John := And.right (A5 hEng)
  exact hNoMP hMP

-- From (6) and not_wellAttended_John, if John has a larger family then he invites friends.
theorem largeFamily_implies_inviteFriends_John : LargeFamily John → InviteFriends John := by
  intro hLF
  have hDisj : WellAttended John ∨ InviteFriends John := A6 hLF
  cases hDisj with
  | inl hW =>
      exact False.elim (not_wellAttended_John hW)
  | inr hF =>
      exact hF
