axiom obj : Type
axiom HeinrichSchmidt : obj
axiom Politician : obj → Prop
axiom MemberPrussianStateParliament : obj → Prop
axiom MemberNaziReichstag : obj → Prop

axiom HS_politician : Politician HeinrichSchmidt
axiom HS_prussian : MemberPrussianStateParliament HeinrichSchmidt
axiom HS_nazi : MemberNaziReichstag HeinrichSchmidt

-- Claim: No politicians are part of the Nazi Reichstag
-- Formalization: ∀ x, Politician x → ¬ MemberNaziReichstag x
-- We prove the negation by a counterexample (Heinrich Schmidt).
theorem not_no_politicians_in_nazi_reichstag :
  ¬ (∀ x : obj, Politician x → ¬ MemberNaziReichstag x) := by
  intro h
  have hneg : ¬ MemberNaziReichstag HeinrichSchmidt := h HeinrichSchmidt HS_politician
  exact hneg HS_nazi
