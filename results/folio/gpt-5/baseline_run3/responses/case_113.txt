<lean>
axiom obj : Type

-- Entities
axiom Picuris : obj
axiom HardingMine : obj
axiom JuanDeOnate : obj

-- Predicates
axiom MountainRange : obj → Prop
axiom InNewMexico : obj → Prop
axiom InTexas : obj → Prop
axiom Mine : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop
axiom Visited : obj → obj → Prop

-- Facts from the context
axiom T1 : MountainRange Picuris
axiom T2 : InNewMexico Picuris ∨ InTexas Picuris
axiom T3 : Visited JuanDeOnate Picuris
axiom T4 : Mine HardingMine
axiom T5 : LocatedIn HardingMine Picuris
axiom T6 : Donated HardingMine

-- No Texas mountain range has a mine that has been donated
axiom T7 : ∀ r : obj, MountainRange r → InTexas r → ¬ ∃ m : obj, Mine m ∧ LocatedIn m r ∧ Donated m

-- The asked statement is: ¬ (∃ r, MountainRange r ∧ InNewMexico r ∧ LocatedIn HardingMine r)
-- We will prove its negation: ∃ r, MountainRange r ∧ InNewMexico r ∧ LocatedIn HardingMine r
theorem harding_in_nm_range : ∃ r : obj, MountainRange r ∧ InNewMexico r ∧ LocatedIn HardingMine r := by
  -- First, show the Picuris Mountains are not in Texas
  have not_picuris_tx : ¬ InTexas Picuris := by
    intro htx
    -- But there exists a donated mine located in Picuris (namely, HardingMine)
    have hex : ∃ m : obj, Mine m ∧ LocatedIn m Picuris ∧ Donated m := by
      exact Exists.intro HardingMine (And.intro T4 (And.intro T5 T6))
    -- This contradicts T7 if Picuris were in Texas
    have hno := T7 Picuris T1 htx
    exact hno hex
  -- From "Picuris is in NM or TX" and "not in TX", conclude "in NM"
  have picuris_nm : InNewMexico Picuris :=
    Or.elim T2
      (fun hnm => hnm)
      (fun htx => False.elim (not_picuris_tx htx))
  -- Conclude the mine is located in a mountain range in New Mexico (Picuris)
  exact Exists.intro Picuris (And.intro T1 (And.intro picuris_nm T5))
</lean>

ANSWER: False
