axiom Person : Type
axiom Taylor : Person

axiom EatsSaladsRegularly : Person → Prop
axiom VeryConscious : Person → Prop
axiom GrewUpHCCH : Person → Prop
axiom FulfillDailyNutritionalIntakes : Person → Prop
axiom DisregardPhysicalWellBeing : Person → Prop
axiom VisitsGymDaily : Person → Prop

-- All people who eat salads regularly are very conscious about their health and eating habits.
axiom A1 : ∀ x : Person, EatsSaladsRegularly x → VeryConscious x
-- All people who grew up in health-conscious childhood homes eat salads regularly.
axiom A2 : ∀ x : Person, GrewUpHCCH x → EatsSaladsRegularly x
-- All people who fulfill their daily nutritional intakes grew up in health-conscious childhood homes.
axiom A3 : ∀ x : Person, FulfillDailyNutritionalIntakes x → GrewUpHCCH x
-- All people who disregard their physical well-being are not very conscious about their health and eating habits.
axiom A4 : ∀ x : Person, DisregardPhysicalWellBeing x → ¬ VeryConscious x
-- If people visit the gym at least once a day, then they always fulfill their daily nutritional intakes.
axiom A5 : ∀ x : Person, VisitsGymDaily x → FulfillDailyNutritionalIntakes x
-- Taylor either grew up in a HCCH and disregards her well-being, or did neither.
axiom A6 : (GrewUpHCCH Taylor ∧ DisregardPhysicalWellBeing Taylor) ∨ (¬GrewUpHCCH Taylor ∧ ¬DisregardPhysicalWellBeing Taylor)

lemma not_both_grew_and_disregard : ¬ (GrewUpHCCH Taylor ∧ DisregardPhysicalWellBeing Taylor) := by
  intro h
  have hG : GrewUpHCCH Taylor := h.left
  have hD : DisregardPhysicalWellBeing Taylor := h.right
  have hESR : EatsSaladsRegularly Taylor := A2 Taylor hG
  have hVC : VeryConscious Taylor := A1 Taylor hESR
  have hNotVC : ¬ VeryConscious Taylor := A4 Taylor hD
  exact hNotVC hVC

lemma taylor_neither : ¬GrewUpHCCH Taylor ∧ ¬DisregardPhysicalWellBeing Taylor := by
  have hx := A6
  have hnot := not_both_grew_and_disregard
  cases hx with
  | inl hGD =>
      exact False.elim (hnot hGD)
  | inr hNeither =>
      exact hNeither

theorem not_taylor_visits_gym_daily : ¬ VisitsGymDaily Taylor := by
  have hNotG : ¬ GrewUpHCCH Taylor := taylor_neither.left
  intro hVG
  have hFDNI : FulfillDailyNutritionalIntakes Taylor := A5 Taylor hVG
  have hG : GrewUpHCCH Taylor := A3 Taylor hFDNI
  exact hNotG hG
</lean>

ANSWER: False
