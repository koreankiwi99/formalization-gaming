<lean>
axiom obj : Type
axiom Event : Type

axiom Beijing : obj
axiom Summer2008 : Event
axiom WinterOlympics : Event

axiom Hosted : obj → Event → Prop

-- From: "Beijing hosted the 2008 Summer Olympics and 2008 Summer Paralympics Games."
axiom H1 : Hosted Beijing Summer2008

-- From: "Beijing has hosted the Summer and Winter Olympics and the Summer and Winter Paralympics."
axiom H2 : Hosted Beijing WinterOlympics

theorem beijing_hosted_both : Hosted Beijing Summer2008 ∧ Hosted Beijing WinterOlympics := by
  constructor
  · exact H1
  · exact H2
</lean>

ANSWER: True
