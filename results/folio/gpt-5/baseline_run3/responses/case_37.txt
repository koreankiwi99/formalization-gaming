<lean>
axiom obj : Type

-- Entities
axiom Symphony9 : obj
axiom ViennaMusicSociety : obj
axiom Beethoven : obj

-- Predicates and relations
axiom MusicPiece : obj → Prop
axiom Composer : obj → Prop
axiom Orchestra : obj → Prop
axiom Conductor : obj → Prop
axiom Writes : obj → obj → Prop
axiom Premiered : obj → obj → Prop
axiom Leads : obj → obj → Prop

-- Contextual facts
axiom S9_is_piece : MusicPiece Symphony9
axiom composers_write_pieces : ∀ x y : obj, Composer x → MusicPiece y → Writes x y
axiom beethoven_wrote_s9 : Writes Beethoven Symphony9
axiom vms_premiered_s9 : Premiered ViennaMusicSociety Symphony9
axiom vms_is_orchestra : Orchestra ViennaMusicSociety
axiom beethoven_leads_vms : Leads Beethoven ViennaMusicSociety
axiom leaders_of_orchestras_are_conductors : ∀ o p : obj, Orchestra o → Leads p o → Conductor p

-- Target: Some orchestras premiered music pieces.
theorem some_orchestra_premiered_piece :
  ∃ o : obj, ∃ m : obj, Orchestra o ∧ MusicPiece m ∧ Premiered o m := by
  refine Exists.intro ViennaMusicSociety ?_
  refine Exists.intro Symphony9 ?_
  exact And.intro vms_is_orchestra (And.intro S9_is_piece vms_premiered_s9)
</lean>

ANSWER: True
