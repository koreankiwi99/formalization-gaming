<lean>
axiom obj : Type
axiom USA : obj
axiom Tokyo : obj
axiom LastSOG : obj

-- Map a host city to the edition of the summer Olympic games held there
axiom GamesIn : obj → obj

-- WonMostMedals c e: country c won the most medals at event e
axiom WonMostMedals : obj → obj → Prop

-- The last summer Olympic games was in Tokyo
axiom H1 : GamesIn Tokyo = LastSOG
-- The United States won the most medals in Tokyo (i.e., at the Tokyo games)
axiom H2 : WonMostMedals USA (GamesIn Tokyo)

theorem us_won_most_last : WonMostMedals USA LastSOG := by
  simpa [H1] using H2
</lean>

ANSWER: True
