<lean>
axiom Money : Type
axiom c205 : Money
axiom c300 : Money
axiom lt : Money → Money → Prop
axiom GRE_cost : Money
axiom cost_is_205 : GRE_cost = c205
axiom less_205_300 : lt c205 c300

theorem gre_cost_below_300 : lt GRE_cost c300 := by
  simpa [cost_is_205] using less_205_300
</lean>
