<lean>
axiom obj : Type
axiom NineteenEightyFour : obj

axiom Digital : obj → Prop
axiom Analog : obj → Prop
axiom PrintedText : obj → Prop
axiom Streaming : obj → Prop
axiom Hardcover : obj → Prop

-- No digital media are analog.
axiom A1 : ∀ x : obj, Digital x → ¬ Analog x
-- Every printed text is analog media.
axiom A2 : ∀ x : obj, PrintedText x → Analog x
-- All streaming services are digital media.
axiom A3 : ∀ x : obj, Streaming x → Digital x
-- If an object is a hardcover book, then it is printed text.
axiom A4 : ∀ x : obj, Hardcover x → PrintedText x
-- If 1984 is a streaming service, then 1984 is a hardcover book.
axiom A5 : Streaming NineteenEightyFour → Hardcover NineteenEightyFour

theorem not_streaming_1984 : ¬ Streaming NineteenEightyFour :=
by
  intro hS
  have hD : Digital NineteenEightyFour := A3 NineteenEightyFour hS
  have hNotA : ¬ Analog NineteenEightyFour := A1 NineteenEightyFour hD
  have hHC : Hardcover NineteenEightyFour := A5 hS
  have hPT : PrintedText NineteenEightyFour := A4 NineteenEightyFour hHC
  have hA : Analog NineteenEightyFour := A2 NineteenEightyFour hPT
  exact hNotA hA
</lean>

ANSWER: True
