<lean>
axiom obj : Type
axiom Jane KiKi : obj

axiom Animal : obj → Prop
axiom Dog : obj → Prop
axiom Barks : obj → Prop
axiom Jumps : obj → Prop
axiom HasLegs : obj → Prop
axiom Terricolous : obj → Prop
axiom Likes : obj → obj → Prop

-- Jane does not like any animals that bark.
axiom j_not_like_barkers : ∀ x : obj, Animal x ∧ Barks x → ¬ Likes Jane x
-- All dogs are animals that bark.
axiom dogs_bark : ∀ x : obj, Dog x → Animal x ∧ Barks x
-- Jane likes all animals that jump.
axiom j_likes_jumpers : ∀ x : obj, Animal x ∧ Jumps x → Likes Jane x
-- If an animal has legs, then it jumps.
axiom legs_imply_jump : ∀ x : obj, Animal x ∧ HasLegs x → Jumps x
-- If an animal is terricolous, then it has legs.
axiom terricolous_implies_legs : ∀ x : obj, Animal x ∧ Terricolous x → HasLegs x
-- KiKi is an animal.
axiom kiki_animal : Animal KiKi
-- If KiKi neither jumps nor has legs, then KiKi is terricolous or has legs.
axiom kiki_special : (¬ Jumps KiKi ∧ ¬ HasLegs KiKi) → (Terricolous KiKi ∨ HasLegs KiKi)

-- Useful consequences of "all dogs are animals that bark"
theorem dog_implies_barks (x : obj) : Dog x → Barks x := by
  intro hd
  have h := dogs_bark x hd
  exact h.right

theorem dog_implies_animal (x : obj) : Dog x → Animal x := by
  intro hd
  have h := dogs_bark x hd
  exact h.left

-- From the special condition, we can show KiKi cannot simultaneously lack jumping and legs.
theorem kiki_not_both_no_jump_and_no_legs : ¬ (¬ Jumps KiKi ∧ ¬ HasLegs KiKi) := by
  intro h
  have hdisj := kiki_special h
  cases hdisj with
  | inl hterr =>
      have hlegs : HasLegs KiKi := terricolous_implies_legs KiKi (And.intro kiki_animal hterr)
      exact h.right hlegs
  | inr hlegs =>
      exact h.right hlegs

-- We cannot derive Dog KiKi (and thus cannot derive Dog KiKi ∧ Barks KiKi) from the given axioms,
-- nor can we derive its negation.
-- Hence the truth of "KiKi is a barking dog" is not determined by the premises.
</lean>
