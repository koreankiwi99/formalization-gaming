axiom ActivateLegs : obj → Prop           -- activate leg muscles

  exact h3

  have hAct : ActivateLegs Yuri := pro_implies_activate Yuri hPro
  exact Yuri_not_activate hAct

  intro hAnd
  exact yuri_not_pro hAnd.right
