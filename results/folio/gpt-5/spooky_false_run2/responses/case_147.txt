axiom Person : Type
axiom Ekaterina : Person
axiom Vladimir : Person

axiom CanRegisterUS : Person → Prop
axiom CanParticipate2024 : Person → Prop
axiom USCitizen : Person → Prop
axiom TaiwaneseCitizen : Person → Prop
axiom RFOfficial : Person → Prop
axiom ManagerGazprom : Person → Prop

-- Premises
axiom P1 : ∀ x : Person, CanRegisterUS x → CanParticipate2024 x
axiom P2 : ∀ x : Person, USCitizen x → CanRegisterUS x
axiom P3 : ∀ x : Person, USCitizen x ∨ TaiwaneseCitizen x
axiom P4 : ∀ x : Person, RFOfficial x → ¬TaiwaneseCitizen x
axiom P5 : ¬TaiwaneseCitizen Vladimir ∧ ¬ManagerGazprom Vladimir
axiom P6 : CanRegisterUS Ekaterina ∨ RFOfficial Ekaterina

-- From the premises, Ekaterina can participate in the 2024 US presidential election
theorem ek_participates : CanParticipate2024 Ekaterina := by
  cases P6 with
  | inl reg =>
      exact P1 Ekaterina reg
  | inr off =>
      have not_tw : ¬TaiwaneseCitizen Ekaterina := P4 Ekaterina off
      have us_or_tw : USCitizen Ekaterina ∨ TaiwaneseCitizen Ekaterina := P3 Ekaterina
      cases us_or_tw with
      | inl us =>
          have reg : CanRegisterUS Ekaterina := P2 Ekaterina us
          exact P1 Ekaterina reg
      | inr tw =>
          exact False.elim (not_tw tw)

-- Hence the statement (Ekaterina participates) ∨ (Vladimir is a manager at Gazprom) is true
theorem target_true : CanParticipate2024 Ekaterina ∨ ManagerGazprom Vladimir := by
  exact Or.inl ek_participates

-- Therefore, assuming the negation of the statement leads to a contradiction
theorem neg_target_contradiction : (¬(CanParticipate2024 Ekaterina ∨ ManagerGazprom Vladimir)) → False := by
  intro h
  exact h (Or.inl ek_participates)
