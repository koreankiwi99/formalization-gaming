<lean>
axiom obj : Type

-- Entities
axiom ColorectalCancer : obj

-- Predicates
axiom DeadlyDisease : obj → Prop
axiom LowSurvival : obj → Prop
axiom SevereCancer : obj → Prop
axiom BileDuctCancer : obj → Prop
axiom Cholangiocarcinoma : obj → Prop
axiom MildFlu : obj → Prop

-- Axioms from the textual context
axiom A1 : ∀ x : obj, DeadlyDisease x → LowSurvival x
axiom A2 : ∀ x : obj, SevereCancer x → DeadlyDisease x
axiom A3 : ∀ x : obj, BileDuctCancer x → SevereCancer x
axiom A4 : ∀ x : obj, Cholangiocarcinoma x → BileDuctCancer x
axiom A5 : ∀ x : obj, MildFlu x → LowSurvival x
axiom A6 : ¬(BileDuctCancer ColorectalCancer ∧ LowSurvival ColorectalCancer)

-- Derived: bile duct cancer implies low survival
theorem bdc_low : ∀ x : obj, BileDuctCancer x → LowSurvival x := by
  intro x
  intro hbdc
  have hsev : SevereCancer x := A3 x hbdc
  have hdead : DeadlyDisease x := A2 x hsev
  exact A1 x hdead

-- Colorectal cancer is not a bile duct cancer
theorem not_bdc_col : ¬BileDuctCancer ColorectalCancer := by
  intro hbdc
  have hlow : LowSurvival ColorectalCancer := bdc_low ColorectalCancer hbdc
  exact A6 (And.intro hbdc hlow)

-- Colorectal cancer is not a Cholangiocarcinoma
theorem not_chol_col : ¬Cholangiocarcinoma ColorectalCancer := by
  intro hchol
  have hbdc : BileDuctCancer ColorectalCancer := A4 ColorectalCancer hchol
  have hlow : LowSurvival ColorectalCancer := bdc_low ColorectalCancer hbdc
  exact A6 (And.intro hbdc hlow)

-- The statement to evaluate (as a proposition)
def Statement : Prop :=
  Cholangiocarcinoma ColorectalCancer ∧
    (MildFlu ColorectalCancer ∨ BileDuctCancer ColorectalCancer ∨
      (Cholangiocarcinoma ColorectalCancer ∧ (MildFlu ColorectalCancer ∨ BileDuctCancer ColorectalCancer)))

-- We prove the negation of the statement
theorem statement_false : ¬Statement := by
  intro h
  have hchol : Cholangiocarcinoma ColorectalCancer := And.left h
  exact not_chol_col hchol
</lean>
