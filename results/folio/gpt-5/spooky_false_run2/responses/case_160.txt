axiom P2 : HasProgram George → CanCompose George
axiom P3 : CanCompose George ∧ WantsCompose George → WillCompose George

theorem compose_if_likes_and_has_program :
  LikesMusic George ∧ HasProgram George → WillCompose George := by
  intro h
  have hm : LikesMusic George := And.left h
  have hp : HasProgram George := And.right h
  have w : WantsCompose George := P1 hm
  have c : CanCompose George := P2 hp
  have hwc : CanCompose George ∧ WantsCompose George := And.intro c w
  exact P3 hwc
