<lean>
axiom obj : Type
axiom Employee : obj → Prop
axiom Manager : obj → Prop
axiom SchedulesMeeting : obj → Prop
axiom AppearsCompanyToday : obj → Prop
axiom LunchInCompany : obj → Prop
axiom LunchAtHome : obj → Prop
axiom RemoteFromHome : obj → Prop
axiom InOtherCountries : obj → Prop
axiom James : obj

-- 1) Employees who schedule meetings go to the company building (appear in the company) today
axiom A1 : ∀ x : obj, Employee x ∧ SchedulesMeeting x → AppearsCompanyToday x
-- 2) Everyone who has lunch in the company building schedules meetings
axiom A2 : ∀ x : obj, LunchInCompany x → SchedulesMeeting x
-- 3) Employees have lunch either in the company building or at home
axiom A3 : ∀ x : obj, Employee x → LunchInCompany x ∨ LunchAtHome x
-- 4) If an employee has lunch at home, they are working remotely from home
axiom A4 : ∀ x : obj, Employee x ∧ LunchAtHome x → RemoteFromHome x
-- 5) All employees who are in other countries work remotely from home
axiom A5 : ∀ x : obj, Employee x ∧ InOtherCountries x → RemoteFromHome x
-- 6) No managers work remotely from home
axiom A6 : ∀ x : obj, Manager x → ¬RemoteFromHome x
-- 7) James will appear in the company today iff he is a manager
axiom A7 : AppearsCompanyToday James ↔ Manager James

-- Some derived implications (they do not yield ¬LunchInCompany James unconditionally)

theorem not_appears_imp_not_lunch_if_employee :
  Employee James → (¬AppearsCompanyToday James → ¬LunchInCompany James) := by
  intro hEmp hNotApp hLunch
  have hSched : SchedulesMeeting James := A2 James hLunch
  have hApp : AppearsCompanyToday James := A1 James ⟨hEmp, hSched⟩
  exact hNotApp hApp

theorem not_manager_imp_not_appears : ¬Manager James → ¬AppearsCompanyToday James := by
  intro hNotMan hApp
  have hMan : Manager James := (Iff.mp A7) hApp
  exact hNotMan hMan
