<lean>
axiom obj : Type
axiom Rabbit : obj → Prop
axiom Squirrel : obj → Prop
axiom Turtle : obj → Prop
axiom Cute : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop
axiom Near : obj → Prop

axiom Rockie : obj

-- All rabbits that can be spotted near the campus are cute.
axiom H1 : ∀ x : obj, Rabbit x ∧ Near x → Cute x
-- Some turtles can be spotted near the campus.
axiom H2 : ∃ x : obj, Turtle x ∧ Near x
-- The only animals that can be spotted near the campus are rabbits and squirrels.
axiom H3 : ∀ x : obj, Near x → (Rabbit x ∨ Squirrel x)
-- If something is skittish, then it is not calm.
axiom H4 : ∀ x : obj, Skittish x → ¬Calm x
-- All the squirrels that can be spotted near the campus are skittish.
axiom H5 : ∀ x : obj, Squirrel x ∧ Near x → Skittish x
-- Rockie can be spotted near the campus, and it is calm.
axiom H6 : Near Rockie ∧ Calm Rockie

-- From the data, we can show Rockie is not a squirrel.
theorem Rockie_not_squirrel : ¬Squirrel Rockie := by
  intro hSq
  have hN : Near Rockie := And.left H6
  have hSk : Skittish Rockie := H5 Rockie (And.intro hSq hN)
  have hNotCalm : ¬Calm Rockie := H4 Rockie hSk
  exact hNotCalm (And.right H6)

-- Therefore Rockie must be a rabbit (given only rabbits or squirrels are near).
theorem Rockie_is_rabbit : Rabbit Rockie := by
  have hN : Near Rockie := And.left H6
  have hOr : Rabbit Rockie ∨ Squirrel Rockie := H3 Rockie hN
  have hNSq : ¬Squirrel Rockie := Rockie_not_squirrel
  exact Or.elim hOr (fun hR => hR) (fun hSq => False.elim (hNSq hSq))

-- Note: From the given premises, we cannot derive ¬Turtle Rockie.
-- The information supplied does not exclude Rockie from also being a turtle.
