<lean>
axiom obj : Type
axiom Beijing : obj

axiom LocatedInNorthernChina : obj → Prop
axiom LocatedInSouthernChina : obj → Prop

-- From the context: Beijing is located in Northern China.
axiom Beijing_in_North : LocatedInNorthernChina Beijing

-- Background geographic exclusivity: if something is in Northern China, it is not in Southern China.
axiom north_excludes_south : ∀ x : obj, LocatedInNorthernChina x → ¬LocatedInSouthernChina x

-- Therefore, Beijing is not located in Southern China.
theorem beijing_not_southern : ¬LocatedInSouthernChina Beijing := by
  intro hSouth
  have hNorth : LocatedInNorthernChina Beijing := Beijing_in_North
  have hNotSouth : ¬LocatedInSouthernChina Beijing := north_excludes_south Beijing hNorth
  exact hNotSouth hSouth
</lean>

ANSWER: False
