axioms

  (A2 : ∀ x, InClub x → PerformOften x ∨ Inactive x)
  (A3 : ∀ x, InClub x ∧ Chaperone x → ¬ StudentAttends x)

axiom Bonnie : obj
axiom InClub_Bonnie : InClub Bonnie
axiom Bonnie_xnor : (AttendEngaged Bonnie ∧ StudentAttends Bonnie) ∨ (¬AttendEngaged Bonnie ∧ ¬StudentAttends Bonnie)

theorem negate_statement :
  ¬((Chaperone Bonnie ∨ (¬ Chaperone Bonnie → PerformOften Bonnie))
     → ((YoungTeen Bonnie ∧ WishFurther Bonnie) ∧ Inactive Bonnie)) := by
  -- Antecedent holds: Chaperone ∨ PerformOften
  have disj_CP : Chaperone Bonnie ∨ PerformOften Bonnie := by
    have h := A2 Bonnie InClub_Bonnie
    cases h with
    | inl hPerf =>
      exact Or.inr hPerf
    | inr hInact =>
      have hCh := A4 Bonnie (And.intro InClub_Bonnie hInact)
      exact Or.inl hCh
  -- Strengthen to the given antecedent shape
  have antecedent_true : Chaperone Bonnie ∨ (¬ Chaperone Bonnie → PerformOften Bonnie) := by
    cases disj_CP with
    | inl hC =>
      exact Or.inl hC
    | inr hP =>
      apply Or.inr
      intro _
      exact hP
  -- The consequent is impossible
  have not_conseq : ¬(((YoungTeen Bonnie ∧ WishFurther Bonnie)) ∧ Inactive Bonnie) := by
    intro h
    have hYW : YoungTeen Bonnie ∧ WishFurther Bonnie := And.left h
    have hInact : Inactive Bonnie := And.right h
    have hStud : StudentAttends Bonnie := by
      have : InClub Bonnie ∧ YoungTeen Bonnie ∧ WishFurther Bonnie :=
        And.intro InClub_Bonnie (And.intro hYW.left hYW.right)
      exact A5 Bonnie this
    have hCh : Chaperone Bonnie := A4 Bonnie (And.intro InClub_Bonnie hInact)
    have hNotStud : ¬ StudentAttends Bonnie := A3 Bonnie (And.intro InClub_Bonnie hCh)
    exact hNotStud hStud
  -- Negate the implication
  intro hImp
  have conseq := hImp antecedent_true
  exact not_conseq conseq
