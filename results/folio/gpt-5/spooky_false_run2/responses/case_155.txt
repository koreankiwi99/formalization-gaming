axiom Greek : obj → Prop
axiom Mortal : obj → Prop

-- Premises
axiom H1 : ∀ x : obj, Human x → Mortal x         -- All humans are mortal.
axiom H2 : ∀ x : obj, Greek x → Human x          -- All Greeks are humans.

-- Derivable: All Greeks are mortal.
theorem all_greeks_mortal : ∀ x : obj, Greek x → Mortal x :=
  fun x gx => H1 x (H2 x gx)

-- If there exists a Greek, then some Greek is mortal.
theorem some_greek_mortal_of_some_greek :
  (∃ x : obj, Greek x) → (∃ x : obj, Greek x ∧ Mortal x) := by
  intro h
  rcases h with ⟨x, gx⟩
  have mx : Mortal x := all_greeks_mortal x gx
  exact ⟨x, And.intro gx mx⟩
