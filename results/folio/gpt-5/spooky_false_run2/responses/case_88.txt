axiom Book : obj → Prop
axiom Reads : obj → obj → Prop
axiom GainsKnowledge : obj → Prop
axiom SmarterThanBefore : obj → Prop
axiom ContainsTonsKnowledge : obj → Prop

axiom Walden : obj

axiom Harry_is_person : Person Harry
axiom Walden_is_book : Book Walden

theorem harry_smarter : SmarterThanBefore Harry := by

  exact Knowledge_makes_smarter Harry hk
