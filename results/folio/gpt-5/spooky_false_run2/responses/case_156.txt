<lean>
axiom obj : Type
axiom James : obj

axiom Student : obj → Prop
axiom FromCS : obj → Prop
axiom WorksLibrary : obj → Prop
axiom PTJob : obj → Prop
axiom TakingDB : obj → Prop
axiom WithDavid : obj → Prop
axiom WorksLab : obj → Prop

-- Premises from the text
axiom P1 : ∀ x : obj, Student x → WorksLibrary x → ¬ FromCS x
axiom P2 : ∀ x : obj, Student x → PTJob x → WorksLibrary x
axiom P3 : ∀ x : obj, Student x → TakingDB x → FromCS x
axiom P4 : ∀ x : obj, Student x → WithDavid x → TakingDB x
axiom P5 : ∀ x : obj, Student x → WorksLab x → WithDavid x
axiom P6 : Student James
axiom P7 : WorksLab James
axiom P8 : ¬ (WorksLab James ∨ PTJob James)

-- Directly refute that James has a university part-time job from P8
theorem james_not_PT : ¬ PTJob James := by
  intro hPT
  exact P8 (Or.inr hPT)

-- An independent derivation (not using P8) that also yields ¬ PTJob James:
-- From working in the lab, James is CS; but having a PT job would imply
-- working in the library, which contradicts being CS.
theorem james_CS : FromCS James := by
  have hWD : WithDavid James := P5 James P6 P7
  have hDB : TakingDB James := P4 James P6 hWD
  exact P3 James P6 hDB

theorem james_not_PT_via_CS : ¬ PTJob James := by
  intro hPT
  have hLib : WorksLibrary James := P2 James P6 hPT
  have hNotCS : ¬ FromCS James := P1 James P6 hLib
  exact hNotCS james_CS
</lean>
