<lean>
axiom obj : Type

axiom Ailton : obj
axiom Braga : obj
axiom Nautico : obj
axiom Fluminense : obj

axiom FootballClub : obj → Prop
axiom LoanedOutTo : obj → obj → Prop

-- Premises
axiom P1 : LoanedOutTo Ailton Braga
axiom P2 : FootballClub Braga
axiom P3 : FootballClub Nautico
axiom P4 : FootballClub Fluminense

-- Statement: Ailton was not loaned out to a football club.
def Statement : Prop := ¬∃ c : obj, FootballClub c ∧ LoanedOutTo Ailton c

-- We prove the negation of the statement: there exists a football club c such that Ailton was loaned out to c.
theorem neg_Statement : ∃ c : obj, FootballClub c ∧ LoanedOutTo Ailton c := by
  refine Exists.intro Braga ?_
  exact And.intro P2 P1

-- Therefore, the statement is false.
theorem refute_Statement : Statement → False := by
  intro h
  exact h neg_Statement
</lean>

ANSWER: False
