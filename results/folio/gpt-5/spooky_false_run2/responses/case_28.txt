<lean>
axiom obj : Type

axiom Marvin : obj
axiom Alien : obj → Prop
axiom Extraterrestrial : obj → Prop
axiom Human : obj → Prop
axiom FromEarth : obj → Prop
axiom FromMars : obj → Prop
axiom HighlyIntelligent : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, Alien x → Extraterrestrial x
axiom A2 : ∀ x : obj, FromMars x → Alien x
axiom A3 : ∀ x : obj, Extraterrestrial x → ¬Human x
axiom A4 : ∀ x : obj, HighlyIntelligent x ∧ FromEarth x → Human x
axiom A5 : HighlyIntelligent Marvin
axiom A6 : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬FromEarth Marvin ∧ ¬FromMars Marvin)
axiom A7 : ¬FromEarth Marvin → Extraterrestrial Marvin

-- The statement (for reference): ¬Human Marvin ∧ ¬FromMars Marvin
-- We can actually prove the statement itself from the premises.
theorem marvin_neither : ¬Human Marvin ∧ ¬FromMars Marvin := by
  have hi : HighlyIntelligent Marvin := A5
  have disj : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬FromEarth Marvin ∧ ¬FromMars Marvin) := A6
  cases disj with
  | inl h =>
      have hFE : FromEarth Marvin := h.left
      have hFM : FromMars Marvin := h.right
      have hHuman : Human Marvin := A4 Marvin (And.intro hi hFE)
      have hAlien : Alien Marvin := A2 Marvin hFM
      have hExt : Extraterrestrial Marvin := A1 Marvin hAlien
      have hNotHuman : ¬Human Marvin := A3 Marvin hExt
      have contra : False := hNotHuman hHuman
      exact False.elim contra
  | inr h =>
      have hNFE : ¬FromEarth Marvin := h.left
      have hNFM : ¬FromMars Marvin := h.right
      have hExt : Extraterrestrial Marvin := A7 hNFE
      have hNotHuman : ¬Human Marvin := A3 Marvin hExt
      exact And.intro hNotHuman hNFM
