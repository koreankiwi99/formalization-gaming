axiom obj : Type
axiom InClub : obj → Prop
axiom PerformOften : obj → Prop
axiom AttendEngaged : obj → Prop
axiom InactiveDisinterested : obj → Prop
axiom Chaperone : obj → Prop
axiom Student : obj → Prop
axiom YoungChild : obj → Prop
axiom Teenager : obj → Prop
axiom WishesFurther : obj → Prop
axiom Bonnie : obj

-- Premises
axiom P1 : ∀ x : obj, InClub x ∧ PerformOften x → AttendEngaged x
axiom P2 : ∀ x : obj, InClub x → PerformOften x ∨ InactiveDisinterested x
axiom P3 : ∀ x : obj, InClub x ∧ Chaperone x → ¬ Student x
axiom P4 : ∀ x : obj, InClub x ∧ InactiveDisinterested x → Chaperone x
axiom P5 : ∀ x : obj, InClub x ∧ (YoungChild x ∨ Teenager x) ∧ WishesFurther x → Student x
axiom P6 : InClub Bonnie ∧ ((AttendEngaged Bonnie ∧ Student Bonnie) ∨ (¬AttendEngaged Bonnie ∧ ¬Student Bonnie))

-- The given statement holds from the premises (hence its negation is not provable).
theorem statement_true :
  ( ((YoungChild Bonnie ∨ Teenager Bonnie) ∧ WishesFurther Bonnie ∧ Chaperone Bonnie)
    ∨ ¬((YoungChild Bonnie ∨ Teenager Bonnie) ∧ WishesFurther Bonnie) )
  → (Student Bonnie ∨ InactiveDisinterested Bonnie) := by
  intro _h
  have hClub : InClub Bonnie := And.left P6
  have hPOI : PerformOften Bonnie ∨ InactiveDisinterested Bonnie := P2 Bonnie hClub
  cases hPOI with
  | inl hPerf =>
      have hAE : AttendEngaged Bonnie := P1 Bonnie (And.intro hClub hPerf)
      have hEq : (AttendEngaged Bonnie ∧ Student Bonnie) ∨ (¬AttendEngaged Bonnie ∧ ¬Student Bonnie) := And.right P6
      have hStu : Student Bonnie := by
        cases hEq with
        | inl h =>
            exact And.right h
        | inr h =>
            have : False := (And.left h) hAE
            exact False.elim this
      exact Or.inl hStu
  | inr hInact =>
      exact Or.inr hInact
