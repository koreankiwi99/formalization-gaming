<lean>
axiom obj : Type
axiom Harry : obj
axiom Invertebrate : obj → Prop
axiom Vertebrate : obj → Prop
axiom Backbone : obj → Prop
axiom Bee : obj → Prop
axiom QueenBee : obj → Prop
axiom ReproMF : obj → Prop

-- Premises (domain = animals)
axiom A1 : ∀ x : obj, Invertebrate x ∨ Vertebrate x
axiom A2 : ∀ x : obj, Backbone x → ReproMF x
axiom A3 : ∀ x : obj, Vertebrate x → Backbone x
axiom A4 : ∀ x : obj, Bee x → ¬ReproMF x
axiom A5 : ∀ x : obj, QueenBee x → Bee x
axiom A6 : Bee Harry

theorem not_repro_Harry : ¬ReproMF Harry := A4 Harry A6

theorem not_backbone_Harry : ¬Backbone Harry := by
  intro hb
  have hr : ReproMF Harry := A2 Harry hb
  exact not_repro_Harry hr

theorem not_vertebrate_Harry : ¬Vertebrate Harry := by
  intro hv
  have hb : Backbone Harry := A3 Harry hv
  exact not_backbone_Harry hb

theorem invertebrate_Harry : Invertebrate Harry := by
  have disj : Invertebrate Harry ∨ Vertebrate Harry := A1 Harry
  cases disj with
  | inl hinv => exact hinv
  | inr hvert =>
      have contra : False := not_vertebrate_Harry hvert
      exact False.elim contra

-- Negation of the statement to evaluate
theorem negate_statement :
  ¬(((Vertebrate Harry ∧ Backbone Harry) ∨ (¬Vertebrate Harry ∧ ¬Backbone Harry)) → (¬Invertebrate Harry ∧ ¬QueenBee Harry)) := by
  intro h
  -- antecedent holds (the "neither" disjunct)
  have hp : (Vertebrate Harry ∧ Backbone Harry) ∨ (¬Vertebrate Harry ∧ ¬Backbone Harry) := by
    apply Or.inr
    exact And.intro not_vertebrate_Harry not_backbone_Harry
  -- apply implication to get consequent
  have hq : (¬Invertebrate Harry ∧ ¬QueenBee Harry) := h hp
  -- contradict using that Harry is invertebrate
  exact hq.left invertebrate_Harry
</lean>
