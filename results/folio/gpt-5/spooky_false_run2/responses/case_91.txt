<lean>
axiom obj : Type
axiom MachineTranslation : obj
axiom NLP : obj → Prop
axiom Generation : obj → Prop
axiom Understanding : obj → Prop
axiom OutputTextSeq : obj → Prop

-- A natural language processing task is categorized exclusively as either generation or understanding
axiom exclusive_category :
  ∀ x : obj, NLP x → (Generation x ∧ ¬Understanding x) ∨ (Understanding x ∧ ¬Generation x)

-- If the output of an NLP task is a text sequence, then it is a language generation task
axiom gen_if_text : ∀ x : obj, NLP x → OutputTextSeq x → Generation x

-- Machine translation is an NLP task
axiom mt_is_nlp : NLP MachineTranslation

-- The output of machine translation is a text sequence
axiom mt_output_text : OutputTextSeq MachineTranslation

-- Prove the negation: Machine translation is not a language understanding task
theorem mt_not_understanding : ¬Understanding MachineTranslation := by
  have g : Generation MachineTranslation := gen_if_text MachineTranslation mt_is_nlp mt_output_text
  have d := exclusive_category MachineTranslation mt_is_nlp
  cases d with
  | inl h =>
      exact h.right
  | inr h =>
      have f : False := h.right g
      exact False.elim f
</lean>

ANSWER: False
