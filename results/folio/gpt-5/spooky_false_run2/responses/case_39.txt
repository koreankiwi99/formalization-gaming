<lean>
axiom obj : Type

-- Predicates
axiom DesignByMax : obj → Prop
axiom Adores : obj → Prop
axiom ZahaStyle : obj → Prop
axiom KellyStyle : obj → Prop
axiom Interesting : obj → Prop
axiom Brutalist : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop

-- Premises
-- All of Zaha Hadid's design styles that Max adores have interesting geometries.
axiom A1 : ∀ x : obj, ZahaStyle x ∧ Adores x → Interesting x
-- No brutalist buildings that Max adores have interesting geometries.
axiom A2 : ∀ x : obj, Brutalist x ∧ Adores x → ¬Interesting x
-- Every style that Max adores is either Zaha Hadid's design style or Kelly Wearstler's design style.
axiom A3 : ∀ x : obj, Adores x → ZahaStyle x ∨ KellyStyle x
-- All of Kelly Wearstler's design styles that Max adores are evocative.
axiom A4 : ∀ x : obj, KellyStyle x ∧ Adores x → Evocative x
-- All of Kelly Wearstler's design styles that Max adores are dreamy.
axiom A5 : ∀ x : obj, KellyStyle x ∧ Adores x → Dreamy x
-- If a design by Max that he adores has interesting geometries, then it is a brutalist building and evocative.
axiom A6 : ∀ x : obj, DesignByMax x ∧ Adores x ∧ Interesting x → Brutalist x ∧ Evocative x

-- Useful consequences

-- No Zaha-style design by Max that he adores can exist (it would be both interesting and not interesting).
theorem no_zaha_adored_design_by_max : ∀ x : obj, DesignByMax x ∧ Adores x ∧ ZahaStyle x → False := by
  intro x h
  have hxD : DesignByMax x := And.left h
  have hA_Z : Adores x ∧ ZahaStyle x := And.right h
  have hxA : Adores x := And.left hA_Z
  have hxZ : ZahaStyle x := And.right hA_Z
  have hxI : Interesting x := A1 x (And.intro hxZ hxA)
  have hD_A_I : DesignByMax x ∧ Adores x ∧ Interesting x := And.intro hxD (And.intro hxA hxI)
  have hB_E : Brutalist x ∧ Evocative x := A6 x hD_A_I
  have hxB : Brutalist x := And.left hB_E
  have notI : ¬Interesting x := A2 x (And.intro hxB hxA)
  exact notI hxI

-- Any design by Max that he adores must be KellyStyle (cannot be ZahaStyle by the previous lemma).
theorem adored_design_by_max_is_kelly : ∀ x : obj, DesignByMax x ∧ Adores x → KellyStyle x := by
  intro x hDA
  have hxD : DesignByMax x := And.left hDA
  have hxA : Adores x := And.right hDA
  have hz_or_hk : ZahaStyle x ∨ KellyStyle x := A3 x hxA
  cases hz_or_hk with
  | inl hz =>
      have contra : False := no_zaha_adored_design_by_max x (And.intro hxD (And.intro hxA hz))
      exact False.elim contra
  | inr hk =>
      exact hk

-- No design by Max that he adores can be interesting.
theorem no_interesting_adored_design_by_max : ∀ x : obj, DesignByMax x ∧ Adores x → ¬Interesting x := by
  intro x hDA
  intro hI
  have hB_E : Brutalist x ∧ Evocative x :=
    A6 x (And.intro (And.left hDA) (And.intro (And.right hDA) hI))
  have hB : Brutalist x := And.left hB_E
  have notI : ¬Interesting x := A2 x (And.intro hB (And.right hDA))
  exact notI hI
