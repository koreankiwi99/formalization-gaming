axiom obj : Type

-- Individuals
axiom Michael : obj
axiom Peter : obj
axiom Windy : obj

-- Predicates
axiom Man : obj → Prop
axiom Taller : obj → obj → Prop
axiom CanBlock : obj → obj → Prop
axiom JumpsWhenShooting : obj → Prop
axiom Shooter : obj → Prop
axiom GreatShooter : obj → Prop
axiom InMichaelClass : obj → Prop

-- Premises
-- If a man is taller than another man, the taller one can block the other's shooting.
axiom P1 : ∀ x y : obj, Man x → Man y → Taller x y → CanBlock x y

-- Michael is a man who is taller than everyone else in his class.
axiom P2a : Man Michael
axiom P2b : ∀ y : obj, InMichaelClass y → y ≠ Michael → Taller Michael y

-- Transitivity of taller
axiom P3 : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom P4a : Man Peter
axiom P4b : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom P5 : ∀ p : obj, ¬JumpsWhenShooting p → CanBlock Michael p

-- Michael cannot block Windy's shooting.
axiom P6 : ¬CanBlock Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom P7 : ∀ x : obj, Shooter x ∧ JumpsWhenShooting x → GreatShooter x

-- If someone can jump when shooting, then they are (indeed) a shooter.
axiom P8 : ∀ x : obj, JumpsWhenShooting x → Shooter x

-- From P5 and P6 we can infer that Windy jumps when shooting (classically).
theorem Windy_jumps : JumpsWhenShooting Windy :=
by
  have hnn : ¬¬JumpsWhenShooting Windy := by
    intro hNoJump
    have hblock : CanBlock Michael Windy := P5 Windy hNoJump
    exact P6 hblock
  open Classical
  by_contra hNoJump
  exact hnn hNoJump

-- Therefore Windy is a great shooter.
theorem Windy_is_great : GreatShooter Windy :=
by
  have hj : JumpsWhenShooting Windy := Windy_jumps
  have hs : Shooter Windy := P8 Windy hj
  exact P7 Windy ⟨hs, hj⟩
