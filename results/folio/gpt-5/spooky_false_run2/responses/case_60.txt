axiom A1 : ∀ x : obj, International x ∧ InUS x → (F1 x ∨ J1 x)

-- An international student in the US with an F1 visa needs to apply for CPT or OPT if the student wants to work in the US.
axiom A2 : ∀ x : obj, International x ∧ InUS x ∧ F1 x ∧ WantsWorkUS x → (NeedsApplyCPT x ∨ NeedsApplyOPT x)

-- Mike needs to apply for CPT if he wants to work in the US.

  intro h
  exact Or.inl (A4 h)

  have hW : WantsWorkUS Mike := And.right hF1W

  have pack : International Mike ∧ InUS Mike ∧ F1 Mike ∧ WantsWorkUS Mike :=
