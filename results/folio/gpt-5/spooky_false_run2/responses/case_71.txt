<lean>
axiom obj : Type
axiom KO : obj

axiom Stock : obj → Prop
axiom Growth : obj → Prop
axiom BoughtRapid : obj → Prop
axiom Suitable : obj → Prop
axiom Mature : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, Growth x → BoughtRapid x
axiom A2 : ∀ x : obj, BoughtRapid x → ¬Suitable x
axiom A3 : ∃ x : obj, Stock x ∧ Growth x
axiom A4 : ∀ x : obj, Mature x → Suitable x
axiom A5 : Mature KO

-- Goal: KO is not both a stock and a growth stock
theorem ko_not_stock_and_growth : ¬(Stock KO ∧ Growth KO) := by
  have hS : Suitable KO := A4 KO A5
  have hNotGrowth : ¬Growth KO := by
    intro hg
    have hNotS : ¬Suitable KO := A2 KO (A1 KO hg)
    exact hNotS hS
  intro hAnd
  exact hNotGrowth hAnd.right
</lean>

ANSWER: False
