axiom MultipleBirth : obj → Prop
axiom HaveSiblingsBornTogether : obj → Prop
axiom ComplainOften : obj → Prop
axiom SpendsTimeWithSiblings : obj → Prop
axiom LiveAtHome : obj → Prop
axiom LiveWithStrangers : obj → Prop
axiom Baby : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, MultipleBirth x → SpendsTimeWithSiblings x
axiom A2 : ∀ x : obj, HaveSiblingsBornTogether x → MultipleBirth x
axiom A3 : ∀ x : obj, ComplainOften x → HaveSiblingsBornTogether x
axiom A4 : ∀ x : obj, LiveAtHome x → ¬LiveWithStrangers x
axiom A5 : ∀ x : obj, SpendsTimeWithSiblings x → LiveAtHome x

-- Luke either is a baby born in a multiple birth and lives with strangers,
-- or is not a baby born in a multiple birth and does not live with strangers.
axiom A6 :
  (Baby Luke ∧ MultipleBirth Luke ∧ LiveWithStrangers Luke) ∨
  ((¬(Baby Luke ∧ MultipleBirth Luke)) ∧ ¬LiveWithStrangers Luke)

-- From being in a multiple birth, one cannot live with strangers (via spending time -> live at home -> not with strangers)
theorem multipleBirth_not_strangers : ∀ x : obj, MultipleBirth x → ¬LiveWithStrangers x := by
  intro x hx
  have h1 : SpendsTimeWithSiblings x := A1 x hx
  have h2 : LiveAtHome x := A5 x h1
  exact A4 x h2

-- The first disjunct in A6 is impossible
theorem luke_not_first_case : ¬(Baby Luke ∧ MultipleBirth Luke ∧ LiveWithStrangers Luke) := by
  intro h
  have hBM : Baby Luke ∧ MultipleBirth Luke := And.left h
  have hMB : MultipleBirth Luke := And.right hBM
  have hLW : LiveWithStrangers Luke := And.right h
  have hnLW : ¬LiveWithStrangers Luke := multipleBirth_not_strangers Luke hMB
  exact hnLW hLW

-- Hence the second disjunct must hold: Luke does not live with strangers
theorem luke_second_case : (¬(Baby Luke ∧ MultipleBirth Luke)) ∧ ¬LiveWithStrangers Luke := by
  have h := A6
  cases h with
  | inl h1 =>
      have contra : False := by
        have hBM : Baby Luke ∧ MultipleBirth Luke := And.left h1
        have hMB : MultipleBirth Luke := And.right hBM
        have hLW : LiveWithStrangers Luke := And.right h1
        have hnLW : ¬LiveWithStrangers Luke := multipleBirth_not_strangers Luke hMB
        exact hnLW hLW
      exact False.elim contra
  | inr h2 => exact h2

theorem luke_not_live_with_strangers : ¬LiveWithStrangers Luke := And.right luke_second_case
