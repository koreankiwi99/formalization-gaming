<lean>
axiom obj : Type

-- Predicates over objects/designs/styles
axiom MaxDesign : obj → Prop
axiom MaxAdores : obj → Prop
axiom Style : obj → Prop
axiom Zaha : obj → Prop
axiom Kelly : obj → Prop
axiom Brutalist : obj → Prop
axiom Interesting : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop

-- Premises translated
-- 1. All of Zaha Hadid's design styles that Max adores have interesting geometries.
axiom P1 : ∀ x : obj, Zaha x ∧ MaxAdores x → Interesting x

-- 2. No brutalist buildings that Max adores have interesting geometries.
axiom P2 : ∀ x : obj, Brutalist x ∧ MaxAdores x → ¬Interesting x

-- 3. Every style that Max adores is either Zaha Hadid's design style or Kelly Wearstler's design style.
axiom P3 : ∀ x : obj, MaxAdores x ∧ Style x → Zaha x ∨ Kelly x

-- 4. All of Kelly Wearstler's design styles that Max adores are evocative.
axiom P4 : ∀ x : obj, Kelly x ∧ MaxAdores x → Evocative x

-- 5. All of Kelly Wearstler's design styles that Max adores are dreamy.
axiom P5 : ∀ x : obj, Kelly x ∧ MaxAdores x → Dreamy x

-- 6. If a design by Max that he adores has interesting geometries,
--    then the design is a brutalist building and evocative.
axiom P6 : ∀ x : obj, MaxDesign x ∧ MaxAdores x ∧ Interesting x → Brutalist x ∧ Evocative x

-- A derived consequence: there are no interesting designs by Max that he adores.
theorem no_adored_interesting_design :
  ∀ x : obj, MaxDesign x → MaxAdores x → ¬Interesting x :=
by
  intro x hxDesign hxAdores
  intro hxInt
  have hBE : Brutalist x ∧ Evocative x :=
    P6 x (And.intro hxDesign (And.intro hxAdores hxInt))
  have hB : Brutalist x := And.left hBE
  have hNotInt : ¬Interesting x := P2 x (And.intro hB hxAdores)
  exact hNotInt hxInt

-- Combining with P1: no object can simultaneously be a Max design, adored by Max, and a Zaha style.
theorem no_maxdesign_adored_zaha :
  ∀ x : obj, MaxDesign x → MaxAdores x → Zaha x → False :=
by
  intro x hxDesign hxAdores hZaha
  have hxInt : Interesting x := P1 x (And.intro hZaha hxAdores)
  have hNotInt : ¬Interesting x := no_adored_interesting_design x hxDesign hxAdores
  exact hNotInt hxInt
