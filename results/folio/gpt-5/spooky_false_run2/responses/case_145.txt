axiom Participate2024 : obj → Prop
axiom USCitizen : obj → Prop

-- Premises
-- 1. Everyone who can register to vote in the US can participate in the 2024 US presidential election.
axiom P1 : ∀ x : obj, CanRegisterUS x → Participate2024 x
-- 2. If someone has US citizenship, then they can register to vote in the US.
axiom P2 : ∀ x : obj, USCitizen x → CanRegisterUS x
-- 3. A person either has US citizenship or Taiwanese citizenship.
axiom P3 : ∀ x : obj, USCitizen x ∨ TaiwaneseCitizen x
-- 4. No Russian Federation officials hold Taiwanese citizenship.
axiom P4 : ∀ x : obj, RussianOfficial x → ¬TaiwaneseCitizen x
-- 5. Vladimir neither holds Taiwanese citizenship nor is he a manager at Gazprom.
axiom P5a : ¬TaiwaneseCitizen Vladimir
axiom P5b : ¬GazpromManager Vladimir
-- 6. Ekaterina can register to vote in the US, or she is a Russian federation official.
axiom P6 : CanRegisterUS Ekaterina ∨ RussianOfficial Ekaterina

-- Some derivable facts (do not yield the target negation)
theorem vladimir_US_citizen : USCitizen Vladimir := by
  have h := P3 Vladimir
  cases h with
  | inl hUS => exact hUS
  | inr hTW =>
      have : False := P5a hTW
      exact False.elim this

theorem vladimir_can_register : CanRegisterUS Vladimir := by
  have hUS := vladimir_US_citizen
  exact P2 Vladimir hUS

theorem vladimir_participates_2024 : Participate2024 Vladimir := by
  have hReg := vladimir_can_register
  exact P1 Vladimir hReg
