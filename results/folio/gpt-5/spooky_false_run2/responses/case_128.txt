<lean>
axiom obj : Type

-- Individuals
axiom John : obj

-- Predicates
axiom MakePlans : obj → Prop
axiom Engaged : obj → Prop
axiom InviteOthers : obj → Prop
axiom WellAttended : obj → Prop
axiom LargerFamily : obj → Prop
axiom InviteFriends : obj → Prop

-- Premises
-- 1. All people who make wedding plans are engaged.
axiom A1 : ∀ x : obj, MakePlans x → Engaged x
-- 2. All people who invite others to their ceremony make wedding plans.
axiom A2 : ∀ x : obj, InviteOthers x → MakePlans x
-- 3. Anyone with a well-attended wedding invites others to their ceremony.
axiom A3 : ∀ x : obj, WellAttended x → InviteOthers x
-- 4. Some people that have well-attended weddings have larger families.
axiom A4 : ∃ x : obj, WellAttended x ∧ LargerFamily x
-- 5. If John is engaged, then John does not invite friends to their ceremony and makes wedding plans.
axiom A5 : Engaged John → (¬InviteFriends John ∧ MakePlans John)
-- 6. If John has a larger family, then either John has a well-attended wedding or invites friends to their ceremony.
axiom A6 : LargerFamily John → (WellAttended John ∨ InviteFriends John)

-- Derived lemma: Well-attended implies engaged (for any person).
theorem wellAttended_implies_engaged : ∀ x : obj, WellAttended x → Engaged x := by
  intro x
  intro hwa
  have hio : InviteOthers x := A3 x hwa
  have hmp : MakePlans x := A2 x hio
  exact A1 x hmp

-- Derived lemma: For John, larger family together with not inviting friends implies engaged.
theorem john_largeFamily_and_notInviteFriends_implies_engaged :
  LargerFamily John → ¬InviteFriends John → Engaged John := by
  intro hLF
  intro hNotIF
  have h := A6 hLF
  cases h with
  | inl hWA =>
      exact wellAttended_implies_engaged John hWA
  | inr hIF =>
      have : False := hNotIF hIF
      exact False.elim this
