axiom obj : Type
axiom Pet : obj → Prop
axiom Mammal : obj → Prop
axiom Animal : obj → Prop
axiom Monkey : obj → Prop
axiom Bird : obj → Prop
axiom CanFly : obj → Prop
axiom Breathe : obj → Prop
axiom Wings : obj → Prop

axiom T1 : ∀ x : obj, Pet x → Mammal x → Animal x
axiom T2 : ∀ x : obj, Monkey x → Mammal x
axiom T3 : ∀ x : obj, Pet x → (Monkey x ∨ Bird x)
axiom T4 : ∀ x : obj, Pet x → Bird x → CanFly x
axiom T5 : ∀ x : obj, Pet x → Animal x → Breathe x
axiom T6 : ∀ x : obj, Pet x → CanFly x → Wings x
axiom Rock : obj
axiom T7 : Pet Rock
axiom T8 : CanFly Rock ∨ Bird Rock ∨ ¬Breathe Rock

theorem rock_has_wings : Wings Rock := by
  have hPet : Pet Rock := T7
  cases T8 with
  | inl hFly =>
      exact T6 Rock hPet hFly
  | inr h =>
      cases h with
      | inl hBird =>
          have hFly2 : CanFly Rock := T4 Rock hPet hBird
          exact T6 Rock hPet hFly2
      | inr hNotBreathe =>
          cases T3 Rock hPet with
          | inl hMonkey =>
              have hMammal : Mammal Rock := T2 Rock hMonkey
              have hAnimal : Animal Rock := T1 Rock hPet hMammal
              have hB : Breathe Rock := T5 Rock hPet hAnimal
              have contra : False := hNotBreathe hB
              exact False.elim contra
          | inr hBird2 =>
              have hFly3 : CanFly Rock := T4 Rock hPet hBird2
              exact T6 Rock hPet hFly3
