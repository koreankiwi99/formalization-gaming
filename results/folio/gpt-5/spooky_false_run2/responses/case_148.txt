<lean>
axiom obj : Type

axiom Ekaterina : obj
axiom Vladimir : obj

axiom RegisterUS : obj → Prop
axiom Participate2024 : obj → Prop
axiom USCitizen : obj → Prop
axiom TWCitizen : obj → Prop
axiom RussianOfficial : obj → Prop
axiom GazpromManager : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, RegisterUS x → Participate2024 x
axiom P2 : ∀ x : obj, USCitizen x → RegisterUS x
axiom P3 : ∀ x : obj, USCitizen x ∨ TWCitizen x
axiom P4 : ∀ x : obj, RussianOfficial x → ¬ TWCitizen x
axiom P5 : ¬ TWCitizen Vladimir ∧ ¬ GazpromManager Vladimir
axiom P6 : RegisterUS Ekaterina ∨ RussianOfficial Ekaterina

theorem Ekaterina_can_register : RegisterUS Ekaterina := by
  have h := P6
  cases h with
  | inl hreg => exact hreg
  | inr hro =>
      have hnotTW : ¬ TWCitizen Ekaterina := P4 Ekaterina hro
      have hdisj := P3 Ekaterina
      cases hdisj with
      | inl hUS => exact P2 Ekaterina hUS
      | inr hTW =>
          have : False := hnotTW hTW
          exact False.elim this

theorem Vladimir_can_participate : Participate2024 Vladimir := by
  have hnotTW : ¬ TWCitizen Vladimir := And.left P5
  have hdisj := P3 Vladimir
  cases hdisj with
  | inl hUS =>
      have hreg : RegisterUS Vladimir := P2 Vladimir hUS
      exact P1 Vladimir hreg
  | inr hTW =>
      have : False := hnotTW hTW
      exact False.elim this

theorem conclusion_is_false : (¬ (RegisterUS Ekaterina ∧ Participate2024 Vladimir)) → False := by
  intro h
  have hreg := Ekaterina_can_register
  have hpart := Vladimir_can_participate
  exact h (And.intro hreg hpart)
</lean>
