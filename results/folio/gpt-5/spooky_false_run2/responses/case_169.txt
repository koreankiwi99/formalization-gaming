axiom obj : Type

axiom Deadly : obj → Prop
axiom LowSurvival : obj → Prop
axiom Severe : obj → Prop
axiom BileDuct : obj → Prop
axiom Cholangiocarcinoma : obj → Prop

axiom MildFlu : obj
axiom Colorectal : obj

-- Premises
axiom P1 : ∀ x : obj, Deadly x → LowSurvival x
axiom P2 : ∀ x : obj, Severe x → Deadly x
axiom P3 : ∀ x : obj, BileDuct x → Severe x
axiom P4 : ∀ x : obj, Cholangiocarcinoma x → BileDuct x
axiom P5 : LowSurvival MildFlu
axiom P6 : ¬(BileDuct Colorectal ∧ LowSurvival Colorectal)

-- Derived: Bile duct cancer implies low survival rate
theorem duct_low : ∀ x : obj, BileDuct x → LowSurvival x := by
  intro x hbd
  have hs : Severe x := P3 x hbd
  have hd : Deadly x := P2 x hs
  exact P1 x hd

-- From the constraint on colorectal cancer, we can conclude it is not a bile duct cancer
theorem not_bile_duct_colorectal : ¬BileDuct Colorectal := by
  intro hbd
  have hlow : LowSurvival Colorectal := duct_low Colorectal hbd
  exact P6 ⟨hbd, hlow⟩
