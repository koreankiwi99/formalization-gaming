<lean>
axiom obj : Type

axiom DTC : obj  -- Dried Thai chilies

axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop
axiom MalaHotpot : obj → Prop
axiom Cupcake : obj → Prop
axiom FromBBM : obj → Prop  -- products from Baked by Melissa

-- No baked sweets are spicy.
axiom R1 : ∀ x : obj, BakedSweet x → ¬ Spicy x
-- All cupcakes are baked sweets.
axiom R2 : ∀ x : obj, Cupcake x → BakedSweet x
-- All mala hotpots are spicy.
axiom R3 : ∀ x : obj, MalaHotpot x → Spicy x
-- All products from Baked by Melissa are cupcakes.
axiom R4 : ∀ x : obj, FromBBM x → Cupcake x
-- Dried Thai chilies are spicy or mala hotpots or not baked sweets.
axiom R5 : Spicy DTC ∨ (MalaHotpot DTC ∨ ¬ BakedSweet DTC)

theorem not_baked_DTC : ¬ BakedSweet DTC := by
  intro hB
  have hDisj : Spicy DTC ∨ (MalaHotpot DTC ∨ ¬ BakedSweet DTC) := R5
  cases hDisj with
  | inl hS =>
      have hNotS : ¬ Spicy DTC := R1 DTC hB
      exact hNotS hS
  | inr h2 =>
      cases h2 with
      | inl hM =>
          have hs : Spicy DTC := R3 DTC hM
          have hNotS : ¬ Spicy DTC := R1 DTC hB
          exact hNotS hs
      | inr hNB =>
          exact hNB hB

-- From the premises, the statement "Dried Thai chilies are not products of Baked by Melissa" is true.
theorem statement_true : ¬ FromBBM DTC := by
  intro hF
  have hC : Cupcake DTC := R4 DTC hF
  have hB : BakedSweet DTC := R2 DTC hC
  have hNB : ¬ BakedSweet DTC := not_baked_DTC
  exact hNB hB
