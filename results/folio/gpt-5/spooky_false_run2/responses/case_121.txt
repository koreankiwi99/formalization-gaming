axiom OneSeater : obj → Prop
axiom Tesla3 : obj → Prop
axiom SingleSeatElectric : obj → Prop

axiom A1 : ∀ x : obj, TransportMulti x → ¬ OneSeater x
axiom A2 : ∀ x : obj, Tesla3 x → TransportMulti x
axiom A3 : ∀ x : obj, SingleSeatElectric x → OneSeater x
axiom A4 : ∀ x : obj, SoloEV x → SingleSeatElectric x
axiom A5 : ¬ Tesla3 JohnCar → (¬ Tesla3 JohnCar ∨ SingleSeatElectric JohnCar)

theorem Tesla3_not_OneSeater : ∀ x : obj, Tesla3 x → ¬ OneSeater x :=
  fun x h => A1 x (A2 x h)

theorem no_both_Tesla3_and_Single : ¬ (Tesla3 JohnCar ∧ SingleSeatElectric JohnCar) := by
  intro h
  have hT : Tesla3 JohnCar := And.left h
  have hS : SingleSeatElectric JohnCar := And.right h
  have hOne : OneSeater JohnCar := A3 JohnCar hS
  have hNotOne : ¬ OneSeater JohnCar := A1 JohnCar (A2 JohnCar hT)
  exact hNotOne hOne

theorem SoloEV_implies_not_Tesla3 :
  SoloEV JohnCar → ¬ Tesla3 JohnCar := by
  intro hSolo hTes
  have hSingle : SingleSeatElectric JohnCar := A4 JohnCar hSolo
  have hOne : OneSeater JohnCar := A3 JohnCar hSingle
  have hNotOne : ¬ OneSeater JohnCar := A1 JohnCar (A2 JohnCar hTes)
  exact hNotOne hOne
