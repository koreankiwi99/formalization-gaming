axiom Participate2024 : obj → Prop

axiom TaiwaneseCitizen : obj → Prop
axiom OfficialRF : obj → Prop
axiom ManagerGazprom : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, CanRegisterUS x → Participate2024 x
axiom A2 : ∀ x : obj, USCitizen x → CanRegisterUS x
axiom A3 : ∀ x : obj, USCitizen x ∨ TaiwaneseCitizen x
axiom A4 : ∀ x : obj, OfficialRF x → ¬TaiwaneseCitizen x
axiom A5 : ¬TaiwaneseCitizen Vladimir ∧ ¬ManagerGazprom Vladimir
axiom A6 : CanRegisterUS Ekaterina ∨ OfficialRF Ekaterina

-- Some derived facts (do not yield ¬OfficialRF Vladimir)
theorem vladimir_not_taiwanese : ¬TaiwaneseCitizen Vladimir := by
  exact And.left A5

theorem vladimir_US_citizen : USCitizen Vladimir := by
  have h := A3 Vladimir
  cases h with
  | inl hUS => exact hUS
  | inr hTW =>
      have hnot : ¬TaiwaneseCitizen Vladimir := And.left A5
      exact False.elim (hnot hTW)

theorem vladimir_can_register : CanRegisterUS Vladimir := by
  have hUS : USCitizen Vladimir := vladimir_US_citizen
  exact A2 Vladimir hUS

theorem vladimir_can_participate : Participate2024 Vladimir := by
  have hreg : CanRegisterUS Vladimir := vladimir_can_register
  exact A1 Vladimir hreg
