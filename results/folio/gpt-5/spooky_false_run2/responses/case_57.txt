axiom obj : Type

axiom Butte : obj
axiom Helena : obj
axiom Missoula : obj
axiom Billings : obj
axiom Montana : obj
axiom WhiteSulphurSprings : obj
axiom StPierre : obj
axiom Bristol : obj
axiom Texarkana : obj
axiom Texhoma : obj
axiom UnionCity : obj

axiom In : obj → obj → Prop

-- Billings is in Montana
axiom Billings_in_Montana : In Billings Montana
-- The state of Montana includes the cities of Butte, Helena, and Missoula
axiom Butte_in_Montana : In Butte Montana
axiom Helena_in_Montana : In Helena Montana
axiom Missoula_in_Montana : In Missoula Montana
-- White Sulphur Springs and Butte are in the same state
axiom WSS_and_Butte_same_state : ∃ s : obj, In WhiteSulphurSprings s ∧ In Butte s
-- The city of St Pierre is not in the state of Montana
axiom StPierre_not_in_Montana : ¬ In StPierre Montana
-- Any city in Butte is not in St Pierre
axiom anti_overlap_Butte_StPierre : ∀ x : obj, In x Butte → ¬ In x StPierre
-- A city can only be in one state in U.S. except for Bristol, Texarkana, Texhoma and Union City
axiom unique_state_except :
  ∀ x s1 s2, In x s1 → In x s2 →
    (s1 = s2 ∨ x = Bristol ∨ x = Texarkana ∨ x = Texhoma ∨ x = UnionCity)

-- Distinctness of Butte from the exceptions
axiom Butte_ne_Bristol : Butte ≠ Bristol
axiom Butte_ne_Texarkana : Butte ≠ Texarkana
axiom Butte_ne_Texhoma : Butte ≠ Texhoma
axiom Butte_ne_UnionCity : Butte ≠ UnionCity

-- Target: Butte and St Pierre are NOT in the same state
theorem not_same_state_Butte_StPierre : ¬ (∃ s : obj, In Butte s ∧ In StPierre s) := by
  intro h
  rcases h with ⟨s, hButteS, hStPierreS⟩
  have hunique := unique_state_except Butte Montana s Butte_in_Montana hButteS
  have hs_eq_mont : s = Montana := by
    cases hunique with
    | inl h => exact h.symm
    | inr h =>
      cases h with
      | inl h1 => exact False.elim (Butte_ne_Bristol h1)
      | inr hrest =>
        cases hrest with
        | inl h2 => exact False.elim (Butte_ne_Texarkana h2)
        | inr hmore =>
          cases hmore with
          | inl h3 => exact False.elim (Butte_ne_Texhoma h3)
          | inr h4 => exact False.elim (Butte_ne_UnionCity h4)
  have : In StPierre Montana := by simpa [hs_eq_mont] using hStPierreS
  exact StPierre_not_in_Montana this
