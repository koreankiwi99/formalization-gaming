<lean>
axiom obj : Type

axiom At : obj → Prop
axiom StudentJob : obj → Prop
axiom NeedMoney : obj → Prop
axiom OrderTakeout : obj → Prop
axiom EnjoyDining : obj → Prop
axiom NotPicky : obj → Prop
axiom SpendDining : obj → Prop

axiom Hannah : obj

-- Premises
axiom P1 : ∀ x : obj, At x ∧ StudentJob x → NeedMoney x
axiom P2 : ∀ x : obj, At x ∧ OrderTakeout x → StudentJob x
axiom P3 : ∀ x : obj, At x → OrderTakeout x ∨ EnjoyDining x
axiom P4 : ∀ x : obj, At x ∧ EnjoyDining x → NotPicky x
axiom P5 : ∀ x : obj, At x ∧ EnjoyDining x → SpendDining x
axiom H_at : At Hannah
axiom H_work_and_cond : StudentJob Hannah ∧ (NeedMoney Hannah → (NotPicky Hannah ∧ ¬NeedMoney Hannah))

-- Target statement:
-- Hannah is at Mary's school and she either is not a picky eater or,
-- if she is, then she spends a lot of her time eating and catching up with friends in the campus dining halls.
def Statement : Prop :=
  At Hannah ∧ (NotPicky Hannah ∨ (¬NotPicky Hannah → SpendDining Hannah))

-- We prove the negation of the statement from the premises.
theorem neg_Statement : ¬Statement := by
  intro _
  -- Derive a contradiction about Hannah's need for money.
  have h_work : StudentJob Hannah := H_work_and_cond.left
  have h_need : NeedMoney Hannah := P1 Hannah ⟨H_at, h_work⟩
  have h_not_need : ¬NeedMoney Hannah := (H_work_and_cond.right h_need).right
  exact h_not_need h_need
</lean>

ANSWER: False
