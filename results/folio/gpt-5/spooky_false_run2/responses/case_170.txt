axiom P2 : ∀ x : obj, Severe x → Deadly x

axiom P4 : ∀ x : obj, Cholangio x → BileDuct x
axiom P5 : ∀ x : obj, MildFlu x → LowSurvival x

-- From the chain: BileDuct → Severe → Deadly → LowSurvival
theorem bileduct_implies_low : ∀ x : obj, BileDuct x → LowSurvival x := by
  intro x hx
  have hs : Severe x := P3 x hx
  have hd : Deadly x := P2 x hs
  exact P1 x hd

-- Antecedent implies contradiction (False)
theorem antecedent_implies_false :
  (BileDuct Colorectal ∨ Cholangio Colorectal) → False := by
  intro h
  have hb : BileDuct Colorectal := by
    cases h with
    | inl hb => exact hb
    | inr hc =>
      exact P4 Colorectal hc
  have hl : LowSurvival Colorectal := bileduct_implies_low Colorectal hb
  have hboth : BileDuct Colorectal ∧ LowSurvival Colorectal := And.intro hb hl
  exact P6 hboth
