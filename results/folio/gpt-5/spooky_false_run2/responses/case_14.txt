<lean>
axiom Game : Type
axiom Zelda : Game

axiom Top10 : Game → Prop
axiom SoldMillion : Game → Prop
axiom MadeByJapanese : Game → Prop

-- Textual context axioms
axiom Zelda_madeByJapanese : MadeByJapanese Zelda
axiom AllTop10MadeByJapanese : ∀ g : Game, Top10 g → MadeByJapanese g
axiom IfSoldMillionThenTop10 : ∀ g : Game, SoldMillion g → Top10 g
axiom Zelda_soldMillion : SoldMillion Zelda

-- Prove the negation of the statement ("Zelda is not on the Top 10 list") by showing Zelda is on the Top 10 list
theorem zelda_top10 : Top10 Zelda := by
  have hsm : SoldMillion Zelda := Zelda_soldMillion
  exact IfSoldMillionThenTop10 Zelda hsm

-- Optional: show the statement is false (i.e., ¬(¬Top10 Zelda))
theorem neg_statement_false : ¬(¬ Top10 Zelda) := by
  intro h
  exact h zelda_top10
