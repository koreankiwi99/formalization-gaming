axiom NineteenEightyFour : obj

axiom Digital : obj → Prop
axiom Analog : obj → Prop
axiom PrintedText : obj → Prop
axiom StreamingService : obj → Prop
axiom HardcoverBook : obj → Prop

-- No digital media are analog.

-- Every printed text is analog media.
axiom PA : ∀ x : obj, PrintedText x → Analog x
-- All streaming services are digital media.
axiom SD : ∀ x : obj, StreamingService x → Digital x
-- If an object is a hardcover book, then it is printed text.
axiom HP : ∀ x : obj, HardcoverBook x → PrintedText x
-- If 1984 is a streaming service, then 1984 is a hardcover book.
axiom S1984_H : StreamingService NineteenEightyFour → HardcoverBook NineteenEightyFour

  have hD : Digital NineteenEightyFour := SD NineteenEightyFour hS
  have hNotA : ¬Analog NineteenEightyFour := ND NineteenEightyFour hD
  have hH : HardcoverBook NineteenEightyFour := S1984_H hS
  have hP : PrintedText NineteenEightyFour := HP NineteenEightyFour hH
  have hA : Analog NineteenEightyFour := PA NineteenEightyFour hP
  exact hNotA hA
