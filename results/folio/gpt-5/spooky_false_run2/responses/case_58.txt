<lean>
axiom obj : Type

-- constants
axiom Billings : obj
axiom Butte : obj
axiom Helena : obj
axiom Missoula : obj
axiom WhiteSulphurSprings : obj
axiom StPierre : obj
axiom Bismarck : obj
axiom Montana : obj

-- predicates
axiom City : obj → Prop
axiom State : obj → Prop
axiom InState : obj → obj → Prop

-- same state relation
def SameState (x y : obj) : Prop := ∃ s : obj, State s ∧ InState x s ∧ InState y s

-- axioms from text
axiom City_Billings : City Billings
axiom City_Butte : City Butte
axiom City_Helena : City Helena
axiom City_Missoula : City Missoula
axiom City_WSS : City WhiteSulphurSprings
axiom City_StPierre : City StPierre
axiom City_Bismarck : City Bismarck

axiom State_Montana : State Montana

axiom Billings_in_Montana : InState Billings Montana
axiom Butte_in_Montana : InState Butte Montana
axiom Helena_in_Montana : InState Helena Montana
axiom Missoula_in_Montana : InState Missoula Montana

axiom WSS_same_as_Butte : SameState WhiteSulphurSprings Butte

axiom StPierre_not_in_Montana : ¬ InState StPierre Montana

-- Uniqueness of state for cities (ignoring the special cases not present here)
axiom unique_state :
  ∀ c s1 s2, City c → State s1 → State s2 → InState c s1 → InState c s2 → s1 = s2

-- "Any city in Butte is not in St Pierre"
axiom no_city_in_butte_is_in_stpierre :
  ∀ x, City x → SameState x Butte → ¬ SameState x StPierre

-- some derived lemmas we can prove
theorem SameState_Butte_Butte : SameState Butte Butte := by
  exact ⟨Montana, State_Montana, Butte_in_Montana, Butte_in_Montana⟩

theorem not_SameState_Butte_StPierre : ¬ SameState Butte StPierre := by
  have h := no_city_in_butte_is_in_stpierre Butte City_Butte SameState_Butte_Butte
  exact h

-- From WSS_same_as_Butte and uniqueness, WSS is in Montana
theorem WSS_in_Montana : InState WhiteSulphurSprings Montana := by
  rcases WSS_same_as_Butte with ⟨s, hs, hwss, hbutte⟩
  have := unique_state Butte s Montana City_Butte hs State_Montana hbutte Butte_in_Montana
  -- s = Montana
  cases this
  simpa
