axiom obj : Type

-- Entities and predicates
axiom Rock : obj
axiom Pet : obj → Prop
axiom Mammal : obj → Prop
axiom Animal : obj → Prop
axiom Monkey : obj → Prop
axiom Bird : obj → Prop
axiom Fly : obj → Prop
axiom Wings : obj → Prop
axiom Breathe : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, Pet x → Mammal x → Animal x                     -- Peter's pets that are mammals are animals
axiom A2 : ∀ x : obj, Monkey x → Mammal x                             -- All monkeys are mammals
axiom A3 : ∀ x : obj, Pet x → (Monkey x ∨ Bird x)                     -- Peter's pets are either monkeys or birds
axiom A4 : ∀ x : obj, Pet x → Bird x → Fly x                          -- Peter's birds can fly
axiom A5 : ∀ x : obj, Pet x → Animal x → Breathe x                    -- All animals that are Peter's pets can breathe
axiom A6 : ∀ x : obj, Pet x → Fly x → Wings x                         -- If Peter's pet can fly, then it has wings
axiom A7 : Pet Rock                                                   -- Rock is Peter's pet
axiom A8 : Fly Rock ∨ Bird Rock ∨ ¬Breathe Rock                       -- Rock can fly, or is a bird, or cannot breathe

-- Some derivable facts (do not yield ¬Monkey Rock)
theorem rock_breathes_if_monkey : Monkey Rock → Breathe Rock := by
  intro hm
  have mm : Mammal Rock := A2 Rock hm
  have an : Animal Rock := A1 Rock A7 mm
  exact A5 Rock A7 an

theorem rock_fly_or_bird_if_monkey : Monkey Rock → Fly Rock ∨ Bird Rock := by
  intro hm
  have br : Breathe Rock := rock_breathes_if_monkey hm
  have h := A8
  cases h with
  | inl hf => exact Or.inl hf
  | inr rest =>
    cases rest with
    | inl hb => exact Or.inr hb
    | inr nb => exact False.elim (nb br)

theorem rock_can_fly_if_monkey : Monkey Rock → Fly Rock := by
  intro hm
  have h := rock_fly_or_bird_if_monkey hm
  cases h with
  | inl hf => exact hf
  | inr hb => exact A4 Rock A7 hb
