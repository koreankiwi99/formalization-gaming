axiom Extraterrestrial : obj → Prop
axiom Human : obj → Prop
axiom FromEarth : obj → Prop
axiom FromMars : obj → Prop
axiom HighlyIntelligent : obj → Prop

axiom Marvin : obj

axiom A1 : ∀ x : obj, Alien x → Extraterrestrial x
axiom A2 : ∀ x : obj, FromMars x → Alien x
axiom A3 : ∀ x : obj, Extraterrestrial x → ¬Human x
axiom A4 : ∀ x : obj, HighlyIntelligent x ∧ FromEarth x → Human x
axiom A5 : HighlyIntelligent Marvin
axiom A6 : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬FromEarth Marvin ∧ ¬FromMars Marvin)
axiom A7 : ¬FromEarth Marvin → Extraterrestrial Marvin

theorem both_case_impossible : ¬(FromEarth Marvin ∧ FromMars Marvin) := by
  intro h
  have hFE : FromEarth Marvin := And.left h
  have hFM : FromMars Marvin := And.right h
  have hAlien : Alien Marvin := A2 Marvin hFM
  have hET : Extraterrestrial Marvin := A1 Marvin hAlien
  have hNotHuman : ¬Human Marvin := A3 Marvin hET
  have hHuman : Human Marvin := A4 Marvin (And.intro A5 hFE)
  exact hNotHuman hHuman

theorem neither_case : ¬FromEarth Marvin ∧ ¬FromMars Marvin := by
  have hdisj := A6
  exact Or.elim hdisj
    (fun hboth => False.elim (both_case_impossible hboth))
    (fun hne => hne)

theorem not_from_earth_marvin : ¬FromEarth Marvin := by
  exact And.left neither_case

theorem not_from_mars_marvin : ¬FromMars Marvin := by
  exact And.right neither_case

theorem extraterrestrial_marvin : Extraterrestrial Marvin := by
  exact A7 not_from_earth_marvin

theorem not_human_marvin : ¬Human Marvin := by
  exact A3 Marvin extraterrestrial_marvin
