axiom Calm : obj → Prop

axiom A3 : ∀ x : obj, NearCampus x → (Rabbit x ∨ Squirrel x)
axiom A4 : ∀ x : obj, Skittish x → ¬Calm x
axiom A5 : ∀ x : obj, (Squirrel x ∧ NearCampus x) → Skittish x
axiom A6 : NearCampus Rockie ∧ Calm Rockie

  have hCalm : Calm Rockie := And.right A6

  cases hDisj with
  | inl r =>
      exact A1 Rockie (And.intro r hNear)
  | inr s =>
      have sSk : Skittish Rockie := A5 Rockie (And.intro s hNear)
      have notCalm : ¬Calm Rockie := A4 Rockie sSk
      exact False.elim (notCalm hCalm)

theorem target : Turtle Rockie ∨ Cute Rockie := Or.inr rockie_cute
