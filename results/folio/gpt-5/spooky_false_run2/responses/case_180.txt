<lean>
axiom Person : Type
axiom Airport : Type

axiom Susan : Person
axiom John : Person
axiom LGA : Airport

axiom FliesTo : Person → Airport → Prop
axiom FliesFrom : Person → Airport → Prop

-- Premises
axiom P1 : FliesTo Susan LGA
axiom P2 : ∀ p : Person, ∀ a : Airport, FliesFrom p a → ¬ FliesTo p a
axiom P3 : FliesFrom John LGA

-- Negation of the statement: John does not fly to LGA
theorem john_not_flies_to_LGA : ¬ FliesTo John LGA := by
  exact P2 John LGA P3
</lean>
