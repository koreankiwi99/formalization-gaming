axiom Vladimir : obj
axiom Ekaterina : obj

axiom CanRegisterUS : obj → Prop
axiom CanParticipate2024Election : obj → Prop
axiom USCitizen : obj → Prop
axiom TaiwaneseCitizen : obj → Prop
axiom RussianOfficial : obj → Prop

-- Everyone who can register to vote in the US can participate in the 2024 US presidential election.
axiom A1 : ∀ x : obj, CanRegisterUS x → CanParticipate2024Election x
-- If someone has US citizenship, then they can register to vote in the US.
axiom A2 : ∀ x : obj, USCitizen x → CanRegisterUS x
-- A person either has US citizenship or Taiwanese citizenship.
axiom A3 : ∀ x : obj, USCitizen x ∨ TaiwaneseCitizen x
-- No Russian Federation officials hold Taiwanese citizenship.
axiom A4 : ∀ x : obj, RussianOfficial x → ¬TaiwaneseCitizen x 
-- Vladimir neither holds Taiwanese citizenship nor is he a manager at Gazprom.
axiom A5a : ¬TaiwaneseCitizen Vladimir
axiom A5b : ¬GazpromManager Vladimir
-- Ekaterina can register to vote in the US, or she is a Russian federation official.
axiom A6 : CanRegisterUS Ekaterina ∨ RussianOfficial Ekaterina

-- Derivable facts about Vladimir (unrelated to being a Russian official)
theorem vladimir_US : USCitizen Vladimir := by
  have h := A3 Vladimir
  cases h with
  | inl hu => exact hu
  | inr htw =>
      exact False.elim (A5a htw)

theorem vladimir_can_register : CanRegisterUS Vladimir :=
  A2 Vladimir vladimir_US

theorem vladimir_can_participate : CanParticipate2024Election Vladimir :=
  A1 Vladimir vladimir_can_register
