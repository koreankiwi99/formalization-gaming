<lean>
axiom obj : Type
axiom Bonnie : obj

axiom InClub : obj → Prop
axiom PerformOften : obj → Prop
axiom AttendEngagedEvents : obj → Prop
axiom InactiveDisinterested : obj → Prop
axiom ChaperonesDances : obj → Prop
axiom StudentAtSchool : obj → Prop
axiom YouthOrTeen : obj → Prop
axiom WishesToFurtherAcademics : obj → Prop

-- People in this club who perform in school talent shows often attend and are very engaged with school events.
axiom R1 : ∀ x : obj, InClub x ∧ PerformOften x → AttendEngagedEvents x
-- People in this club either perform in school talent shows often or are inactive and disinterested community members.
axiom R2 : ∀ x : obj, InClub x → PerformOften x ∨ InactiveDisinterested x
-- People in this club who chaperone high school dances are not students who attend the school.
axiom R3 : ∀ x : obj, InClub x ∧ ChaperonesDances x → ¬StudentAtSchool x
-- All people in this club who are inactive and disinterested members of their community chaperone high school dances.
axiom R4 : ∀ x : obj, InClub x ∧ InactiveDisinterested x → ChaperonesDances x
-- All young children and teenagers in this club who wish to further their academic careers and educational opportunities are students who attend the school. 
axiom R5 : ∀ x : obj, InClub x ∧ YouthOrTeen x ∧ WishesToFurtherAcademics x → StudentAtSchool x

-- Bonnie is in this club
axiom Bonnie_in : InClub Bonnie
-- Bonnie either (attends and is very engaged with school events and is a student) 
-- or (is not someone who attends and is very engaged with school events and is not a student).
axiom Bonnie_split :
  (AttendEngagedEvents Bonnie ∧ StudentAtSchool Bonnie) ∨
  (¬AttendEngagedEvents Bonnie ∧ ¬StudentAtSchool Bonnie)

-- A derived fact: if Bonnie performs often, then she is a student (but this does not lead to contradiction).
theorem Bonnie_perform_implies_student : PerformOften Bonnie → StudentAtSchool Bonnie :=
by
  intro bp
  have ae : AttendEngagedEvents Bonnie := R1 Bonnie (And.intro Bonnie_in bp)
  cases Bonnie_split with
  | inl h =>
      exact h.right
  | inr h =>
      have contra : False := h.left ae
      exact False.elim contra
