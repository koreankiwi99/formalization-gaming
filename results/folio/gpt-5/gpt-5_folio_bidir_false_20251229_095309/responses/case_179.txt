<lean>
axiom Person : Type
axiom Airport : Type

axiom Susan : Person
axiom John : Person
axiom LGA : Airport

axiom FliesTo : Person → Airport → Prop
axiom FliesFrom : Person → Airport → Prop

-- Premises
axiom T1 : FliesTo Susan LGA
axiom T2 : FliesFrom John LGA
axiom R1 : ∀ p : Person, ∀ a : Airport, ¬ (FliesTo p a ∧ FliesFrom p a)

-- Conclusion to refute: Susan flies from LGA
theorem susan_not_from_LGA : ¬ FliesFrom Susan LGA := by
  intro hfrom
  exact (R1 Susan LGA) (And.intro T1 hfrom)
</lean>

ANSWER: False
