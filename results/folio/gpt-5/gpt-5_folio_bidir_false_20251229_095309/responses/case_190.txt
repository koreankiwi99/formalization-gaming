<lean>
axiom obj : Type
axiom Yuri : obj

axiom ProBasketballPlayer : obj → Prop
axiom American : obj → Prop
axiom CanJumpShots : obj → Prop
axiom LeapsStraight : obj → Prop
axiom ActivatesLegMuscles : obj → Prop

-- Premises
axiom some_pro_not_american : ∃ x : obj, ProBasketballPlayer x ∧ ¬American x
axiom all_pro_can_jump : ∀ x : obj, ProBasketballPlayer x → CanJumpShots x
axiom jump_implies_leap : ∀ x : obj, CanJumpShots x → LeapsStraight x
axiom leap_implies_activate : ∀ x : obj, LeapsStraight x → ActivatesLegMuscles x
axiom yuri_not_activate : ¬ActivatesLegMuscles Yuri

theorem not_leap_yuri : ¬LeapsStraight Yuri := by
  intro h
  have act : ActivatesLegMuscles Yuri := leap_implies_activate Yuri h
  exact yuri_not_activate act

theorem pro_implies_activation : ∀ x : obj, ProBasketballPlayer x → ActivatesLegMuscles x := by
  intro x hx
  have hj : CanJumpShots x := all_pro_can_jump x hx
  have hl : LeapsStraight x := jump_implies_leap x hj
  exact leap_implies_activate x hl

theorem not_pro_yuri : ¬ProBasketballPlayer Yuri := by
  intro hy
  have ha : ActivatesLegMuscles Yuri := pro_implies_activation Yuri hy
  exact yuri_not_activate ha

theorem negate_statement :
  ¬(¬LeapsStraight Yuri → (American Yuri ∧ ProBasketballPlayer Yuri)) := by
  intro h
  have conj : American Yuri ∧ ProBasketballPlayer Yuri := h not_leap_yuri
  have hypro : ProBasketballPlayer Yuri := And.right conj
  exact not_pro_yuri hypro
