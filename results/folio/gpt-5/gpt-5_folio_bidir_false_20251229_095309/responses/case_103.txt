theorem luke_not_live_with_strangers : ¬LiveWithStrangers Luke := by
  have r : MultipleBirth Luke → ¬LiveWithStrangers Luke := by
    intro m
    exact A4 Luke (A5 Luke (A1 Luke m))
  cases A6 with
  | inl h =>
      -- h.left : MultipleBirth Luke, h.right : LiveWithStrangers Luke
      exact False.elim ((r h.left) h.right)
  | inr h =>
      exact h.right

      intro m0
      exact (r m0) h.right
  | inr h =>
      exact h.left
