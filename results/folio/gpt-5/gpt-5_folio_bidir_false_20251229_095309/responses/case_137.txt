axiom alg : Type
axiom MachineLearning : alg → Prop
axiom Supervised : alg → Prop
axiom Unsupervised : alg → Prop
axiom Reinforcement : alg → Prop
axiom RequiresLabeled : alg → Prop

-- Machine Learning algorithms can be categorized as supervised, unsupervised, or reinforcement
axiom categorize : ∀ a : alg, MachineLearning a → (Supervised a ∨ Unsupervised a ∨ Reinforcement a)
-- Unsupervised learning algorithms do not require labeled data
axiom unsup_no_labels : ∀ a : alg, Unsupervised a → ¬ RequiresLabeled a

-- The state-of-the-art text summarization model is trained with a machine learning algorithm
axiom SOTA_alg : alg
axiom SOTA_is_ML : MachineLearning SOTA_alg
-- Reinforcement learning is not used to train the SOTA model
axiom SOTA_not_RL : ¬ Reinforcement SOTA_alg
-- The ML algorithm for training text summarization models requires labeled data
axiom SOTA_req_labels : RequiresLabeled SOTA_alg

theorem not_unsupervised_SOTA : ¬ Unsupervised SOTA_alg := by
  intro h
  have hNoLabel : ¬ RequiresLabeled SOTA_alg := unsup_no_labels SOTA_alg h
  exact hNoLabel SOTA_req_labels

theorem supervised_used : Supervised SOTA_alg := by
  have hcat := categorize SOTA_alg SOTA_is_ML
  cases hcat with
  | inl hsup => exact hsup
  | inr hrest =>
    cases hrest with
    | inl hunsup =>
      exact False.elim (not_unsupervised_SOTA hunsup)
    | inr hrl =>
      exact False.elim (SOTA_not_RL hrl)
