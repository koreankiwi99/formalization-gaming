<lean>
axiom Person : Type
axiom Hannah : Person

axiom AtSchool : Person → Prop
axiom WorkOnCampus : Person → Prop
axiom NeedEarnMoney : Person → Prop
axiom OrderTakeoutFrequently : Person → Prop
axiom EnjoyDiningHall : Person → Prop
axiom SpendTimeDiningHalls : Person → Prop
axiom PickyEater : Person → Prop

-- Rules
axiom R1 : ∀ x : Person, AtSchool x → WorkOnCampus x → NeedEarnMoney x
axiom R2 : ∀ x : Person, AtSchool x → OrderTakeoutFrequently x → WorkOnCampus x
axiom R3 : ∀ x : Person, AtSchool x → (OrderTakeoutFrequently x ∨ EnjoyDiningHall x)
axiom R4 : ∀ x : Person, AtSchool x → EnjoyDiningHall x → ¬PickyEater x
axiom R5 : ∀ x : Person, AtSchool x → EnjoyDiningHall x → SpendTimeDiningHalls x

-- Facts about Hannah
axiom H_at : AtSchool Hannah
axiom H_work : WorkOnCampus Hannah
axiom H_cond : NeedEarnMoney Hannah → (¬PickyEater Hannah ∧ ¬NeedEarnMoney Hannah)

-- Target statement
def Statement : Prop :=
  AtSchool Hannah ∧ ¬PickyEater Hannah ∧ SpendTimeDiningHalls Hannah

-- Prove the negation of the statement
theorem neg_statement : ¬Statement := by
  intro _
  have hneed : NeedEarnMoney Hannah := R1 Hannah H_at H_work
  have hpair : ¬PickyEater Hannah ∧ ¬NeedEarnMoney Hannah := H_cond hneed
  have hnotneed : ¬NeedEarnMoney Hannah := And.right hpair
  exact hnotneed hneed
</lean>

ANSWER: False
