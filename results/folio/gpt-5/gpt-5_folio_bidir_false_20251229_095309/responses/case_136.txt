<lean>
axiom obj : Type

axiom Sucks : obj → Prop
axiom HouseholdAppliance : obj → Prop
axiom Vacuum : obj → Prop
axiom Vampire : obj → Prop
axiom Plunger : obj → Prop

-- Plungers suck
axiom R_plunger : ∀ x : obj, Plunger x → Sucks x
-- Vacuums suck
axiom R_vacuum : ∀ x : obj, Vacuum x → Sucks x
-- Vampires suck
axiom R_vampire : ∀ x : obj, Vampire x → Sucks x

-- Space is a vacuum
axiom Space : obj
axiom Space_is_vacuum : Vacuum Space

-- A duster is a household appliance that doesn't suck
axiom Duster : obj
axiom Duster_is_appliance : HouseholdAppliance Duster
axiom Duster_not_suck : ¬Sucks Duster

-- Negation of the statement: not all household appliances suck
theorem not_all_appliances_suck : ¬(∀ x : obj, HouseholdAppliance x → Sucks x) :=
by
  intro h
  have hD : Sucks Duster := h Duster Duster_is_appliance
  exact Duster_not_suck hD
</lean>

ANSWER: False
