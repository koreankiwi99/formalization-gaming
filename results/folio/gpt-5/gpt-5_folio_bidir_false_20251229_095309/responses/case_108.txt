<lean>
axiom obj : Type
axiom Peter : obj

axiom Spill : obj → Prop
axiom Tidy : obj → Prop
axiom CFF : obj → Prop -- clumsy foodie who goes out frequently to find new food restaurants to try
axiom Clean : obj → Prop
axiom Values : obj → Prop
axiom FamPrior : obj → Prop

axiom A1 : ∀ x : obj, Spill x → ¬ Tidy x
axiom A2 : ∀ x : obj, CFF x → Spill x
axiom A3 : ∀ x : obj, Clean x → Tidy x
axiom A4 : ∀ x : obj, Values x → Clean x
axiom A5 : ∀ x : obj, FamPrior x → Values x
axiom A6 : (Spill Peter ∧ Clean Peter) ∨ (¬ Spill Peter ∧ ¬ Clean Peter)

theorem negate_statement : ¬ (CFF Peter ∨ FamPrior Peter) :=
by
  cases A6 with
  | inl h =>
      have hs : Spill Peter := h.left
      have hc : Clean Peter := h.right
      have ht : Tidy Peter := A3 Peter hc
      have hnt : ¬ Tidy Peter := A1 Peter hs
      have f : False := hnt ht
      intro _
      exact False.elim f
  | inr h =>
      have hns : ¬ Spill Peter := h.left
      have hnc : ¬ Clean Peter := h.right
      intro q
      cases q with
      | inl hCFF =>
          have hs : Spill Peter := A2 Peter hCFF
          exact hns hs
      | inr hFam =>
          have hv : Values Peter := A5 Peter hFam
          have hc : Clean Peter := A4 Peter hv
          exact hnc hc
</lean>
