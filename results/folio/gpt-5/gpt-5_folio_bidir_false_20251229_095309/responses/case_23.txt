axiom Show : Type
axiom StrangerThings : Show
axiom BlackMirror : Show

axiom Netflix : Show → Prop
axiom Popular : Show → Prop
-- Karen-centric predicates on shows
axiom Binge : Show → Prop          -- Karen binge-watches the show
axiom Download : Show → Prop       -- Karen downloads the show
axiom ShareWithLisa : Show → Prop  -- Karen shares the show with Lisa

-- "Stranger Things" is a popular Netflix show.
axiom ST_netflix : Netflix StrangerThings
axiom ST_popular : Popular StrangerThings

-- If a Netflix show is popular, Karen will binge-watch it.
axiom R_pop_binge : ∀ s : Show, Netflix s → Popular s → Binge s

-- If and only if Karen binge-watches a Netflix show, she will download it.
axiom R_binge_iff_download : ∀ s : Show, Netflix s → (Binge s ↔ Download s)

-- Karen does not download "Black Mirror."
axiom BM_not_download : ¬Download BlackMirror

-- "Black Mirror" is a Netflix show.
axiom BM_netflix : Netflix BlackMirror

-- If Karen binge-watches a Netflix show, she will share it with Lisa.
axiom R_binge_share : ∀ s : Show, Netflix s → Binge s → ShareWithLisa s

-- From the iff and not-download, we can derive that Karen did not binge-watch Black Mirror.
theorem not_binge_BM : ¬Binge BlackMirror := by
  intro hb
  have hiff := R_binge_iff_download BlackMirror BM_netflix
  have hdl : Download BlackMirror := hiff.mp hb
  exact BM_not_download hdl
