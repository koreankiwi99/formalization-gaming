axiom obj : Type
axiom Mixture : obj

axiom Hydrocarbon : obj → Prop
axiom Organic : obj → Prop
axiom Alkane : obj → Prop
axiom Chemical : obj → Prop
axiom ContainsCarbon : obj → Prop
axiom OneElement : obj → Prop

-- Premises
axiom H1 : ∀ x : obj, Hydrocarbon x → Organic x
axiom H2 : ∀ x : obj, Alkane x → Hydrocarbon x
axiom H3 : ∀ x : obj, Organic x → Chemical x
axiom H4 : ∀ x : obj, Organic x → ContainsCarbon x
axiom H5 : ∀ x : obj, Chemical x → ¬OneElement x
axiom H6 : (Chemical Mixture ∧ OneElement Mixture) ∨ (¬Chemical Mixture ∧ ¬OneElement Mixture)

-- From H5, both Chemical and OneElement cannot hold simultaneously
lemma not_both_chem_one : ¬(Chemical Mixture ∧ OneElement Mixture) :=
  fun h => (H5 Mixture h.left) h.right

-- From the exclusive-or-like condition and the impossibility of both, we get neither
lemma neither_chem_nor_one : ¬Chemical Mixture ∧ ¬OneElement Mixture :=
  Or.elim H6
    (fun hboth => False.elim (not_both_chem_one hboth))
    (fun hneither => hneither)

-- Alkanes imply Chemicals via the given chains of implications
lemma alkane_implies_chemical : ∀ x : obj, Alkane x → Chemical x :=
  fun x hx => H3 x (H1 x (H2 x hx))

-- Therefore, Mixture is not an alkane
lemma mixture_not_alkane : ¬Alkane Mixture :=
  let hnotchem := (And.left neither_chem_nor_one)
  fun hA => hnotchem (alkane_implies_chemical Mixture hA)

-- Conclude the negation of the queried statement
theorem not_statement : ¬(Alkane Mixture ∧ ContainsCarbon Mixture) :=
  fun h => mixture_not_alkane h.left
</lean>
