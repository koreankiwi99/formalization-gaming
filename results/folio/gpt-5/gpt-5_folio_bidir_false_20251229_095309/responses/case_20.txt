<lean>
axiom obj : Type
axiom Rockie : obj

axiom Rabbit : obj → Prop
axiom Squirrel : obj → Prop
axiom Turtle : obj → Prop
axiom Spotted : obj → Prop
axiom Cute : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop

-- All rabbits that can be spotted near the campus are cute.
axiom A1 : ∀ x : obj, (Rabbit x ∧ Spotted x) → Cute x
-- Some turtles can be spotted near the campus.
axiom A2 : ∃ x : obj, Turtle x ∧ Spotted x
-- The only animals that can be spotted near the campus are rabbits and squirrels.
axiom A3 : ∀ x : obj, Spotted x → (Rabbit x ∨ Squirrel x)
-- If something is skittish, then it is not calm.
axiom A4 : ∀ x : obj, Skittish x → ¬Calm x
-- All the squirrels that can be spotted near the campus are skittish.
axiom A5 : ∀ x : obj, (Squirrel x ∧ Spotted x) → Skittish x
-- Rockie can be spotted near the campus, and it is calm.
axiom A6 : Spotted Rockie ∧ Calm Rockie

-- Negation of the statement: ¬((Cute Rockie ∧ Calm Rockie) → (Skittish Rockie ∧ Turtle Rockie))
theorem rockie_implication_false :
  ¬((Cute Rockie ∧ Calm Rockie) → (Skittish Rockie ∧ Turtle Rockie)) := by
  -- From A6
  have hSpotted : Spotted Rockie := And.left A6
  have hCalm : Calm Rockie := And.right A6
  -- Rockie is not skittish (since skittish → ¬calm and Rockie is calm)
  have hNotSkittish : ¬Skittish Rockie := by
    intro hSk
    have hNotCalm : ¬Calm Rockie := A4 Rockie hSk
    exact hNotCalm hCalm
  -- From "only spotted are rabbits or squirrels"
  have hDisj : Rabbit Rockie ∨ Squirrel Rockie := A3 Rockie hSpotted
  -- Rockie cannot be a squirrel (would be skittish, contradicting calm)
  have hRabbit : Rabbit Rockie := by
    cases hDisj with
    | inl hR => exact hR
    | inr hSq =>
      have hSk : Skittish Rockie := A5 Rockie (And.intro hSq hSpotted)
      have hNotCalm : ¬Calm Rockie := A4 Rockie hSk
      exact False.elim (hNotCalm hCalm)
  -- Therefore Rockie is cute
  have hCute : Cute Rockie := A1 Rockie (And.intro hRabbit hSpotted)
  -- Now negate the implication
  intro hImp
  have hConsequent : Skittish Rockie ∧ Turtle Rockie := hImp (And.intro hCute hCalm)
  have hSk : Skittish Rockie := And.left hConsequent
  exact hNotSkittish hSk
</lean>

ANSWER: False
