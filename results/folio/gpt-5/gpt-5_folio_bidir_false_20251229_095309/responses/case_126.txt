axiom obj : Type
axiom Rock : obj

axiom Pet : obj → Prop
axiom Mammal : obj → Prop
axiom Animal : obj → Prop
axiom Monkey : obj → Prop
axiom Bird : obj → Prop
axiom CanFly : obj → Prop
axiom CanBreathe : obj → Prop
axiom HasWings : obj → Prop

axiom A1 : ∀ x : obj, Pet x ∧ Mammal x → Animal x
axiom A2 : ∀ x : obj, Monkey x → Mammal x
axiom A3 : ∀ x : obj, Pet x → (Monkey x ∨ Bird x)
axiom A4 : ∀ x : obj, Pet x ∧ Bird x → CanFly x
axiom A5 : ∀ x : obj, Pet x ∧ Animal x → CanBreathe x
axiom A6 : ∀ x : obj, Pet x ∧ CanFly x → HasWings x
axiom A7 : Pet Rock
axiom A8 : CanFly Rock ∨ Bird Rock ∨ ¬CanBreathe Rock

theorem Rock_has_wings : HasWings Rock := by
  have hPet : Pet Rock := A7
  have hdisj : CanFly Rock ∨ Bird Rock ∨ ¬CanBreathe Rock := A8
  apply Or.elim hdisj
  · intro hFly
    exact A6 Rock (And.intro hPet hFly)
  · intro hRest
    apply Or.elim hRest
    · intro hBird
      have hFly : CanFly Rock := A4 Rock (And.intro hPet hBird)
      exact A6 Rock (And.intro hPet hFly)
    · intro hNotBreathe
      have hMB : Monkey Rock ∨ Bird Rock := A3 Rock hPet
      have hBird : Bird Rock := by
        apply Or.elim hMB
        · intro hMon
          have hMam : Mammal Rock := A2 Rock hMon
          have hAni : Animal Rock := A1 Rock (And.intro hPet hMam)
          have hBr : CanBreathe Rock := A5 Rock (And.intro hPet hAni)
          exact False.elim (hNotBreathe hBr)
        · intro hB
          exact hB
      have hFly : CanFly Rock := A4 Rock (And.intro hPet hBird)
      exact A6 Rock (And.intro hPet hFly)
</lean>
