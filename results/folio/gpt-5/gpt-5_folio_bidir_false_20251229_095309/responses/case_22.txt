<lean>
axiom obj : Type
axiom StrangerThings : obj
axiom BlackMirror : obj

axiom NetflixShow : obj → Prop
axiom Popular : obj → Prop
axiom BingeKaren : obj → Prop
axiom DownloadKaren : obj → Prop
axiom ShareKarenLisa : obj → Prop

-- Facts about Stranger Things
axiom ST_netflix : NetflixShow StrangerThings
axiom ST_popular : Popular StrangerThings

-- Rules
axiom R1 : ∀ x : obj, NetflixShow x → Popular x → BingeKaren x
axiom R2a : ∀ x : obj, NetflixShow x → (BingeKaren x → DownloadKaren x)  -- (binge → download)
axiom R2b : ∀ x : obj, NetflixShow x → (DownloadKaren x → BingeKaren x)  -- (download → binge)
axiom R3 : ∀ x : obj, NetflixShow x → BingeKaren x → ShareKarenLisa x

-- Facts about Black Mirror
axiom BM_not_download : ¬DownloadKaren BlackMirror
axiom BM_netflix : NetflixShow BlackMirror

-- Prove that Black Mirror is not popular
theorem not_popular_blackmirror : ¬Popular BlackMirror := by
  intro hpop
  have hbinge : BingeKaren BlackMirror := R1 BlackMirror BM_netflix hpop
  have hdownload : DownloadKaren BlackMirror := R2a BlackMirror BM_netflix hbinge
  exact BM_not_download hdownload
</lean>

ANSWER: False
