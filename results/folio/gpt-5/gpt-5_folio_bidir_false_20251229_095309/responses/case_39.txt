axiom obj : Type

-- Predicates
axiom ByMax : obj → Prop
axiom Adores : obj → Prop
axiom Zaha : obj → Prop
axiom Kelly : obj → Prop
axiom Interesting : obj → Prop
axiom Brutalist : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop

-- Context axioms
axiom C1 : ∀ x : obj, Zaha x ∧ Adores x → Interesting x
axiom C2 : ∀ x : obj, Brutalist x ∧ Adores x → ¬Interesting x
axiom C3 : ∀ x : obj, Adores x → Zaha x ∨ Kelly x
axiom C4 : ∀ x : obj, Kelly x ∧ Adores x → Evocative x
axiom C5 : ∀ x : obj, Kelly x ∧ Adores x → Dreamy x
axiom C6 : ∀ x : obj, ByMax x ∧ Adores x ∧ Interesting x → Brutalist x ∧ Evocative x

-- The statement to evaluate (interpreted as existential):
def Statement : Prop := ∃ x : obj, ByMax x ∧ Brutalist x

-- Some derivable consequences (do not yield ¬Statement)
theorem adored_brutalist_not_interesting (x : obj) (hB : Brutalist x) (hA : Adores x) : ¬Interesting x :=
  C2 x ⟨hB, hA⟩

theorem adored_zaha_interesting (x : obj) (hZ : Zaha x) (hA : Adores x) : Interesting x :=
  C1 x ⟨hZ, hA⟩

theorem adored_brutalist_not_zaha : ∀ x : obj, Adores x → Brutalist x → ¬Zaha x := by
  intro x hA hB hZ
  have hI : Interesting x := C1 x ⟨hZ, hA⟩
  have hNotI : ¬Interesting x := C2 x ⟨hB, hA⟩
  exact hNotI hI

theorem adored_brutalist_kelly : ∀ x : obj, Adores x → Brutalist x → Kelly x := by
  intro x hA hB
  have hzork : Zaha x ∨ Kelly x := C3 x hA
  cases hzork with
  | inl hZ =>
      have hzcontra : ¬Zaha x := adored_brutalist_not_zaha x hA hB
      have : False := hzcontra hZ
      exact False.elim this
  | inr hK =>
      exact hK
