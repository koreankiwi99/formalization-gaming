<lean>
axiom obj : Type
axiom Tom : obj

axiom NiceToAnimals : obj → Prop
axiom MeanToAnimals : obj → Prop
axiom Grumpy : obj → Prop
axiom AnimalLover : obj → Prop
axiom PetOwner : obj → Prop

-- No one nice to animals is also mean to animals.
axiom R1 : ∀ x : obj, NiceToAnimals x → ¬MeanToAnimals x
-- Some grumpy people are mean to animals.
axiom R2 : ∃ x : obj, Grumpy x ∧ MeanToAnimals x
-- All animal lovers are nice to animals.
axiom R3 : ∀ x : obj, AnimalLover x → NiceToAnimals x
-- All pet owners love animals (are animal lovers).
axiom R4 : ∀ x : obj, PetOwner x → AnimalLover x
-- Tom is a pet owner.
axiom T1 : PetOwner Tom

theorem tom_animalLover : AnimalLover Tom := R4 Tom T1
theorem tom_nice : NiceToAnimals Tom := R3 Tom tom_animalLover
theorem tom_not_mean : ¬MeanToAnimals Tom := R1 Tom tom_nice

-- Antecedent of the statement holds: ¬(Grumpy Tom ∧ MeanToAnimals Tom)
theorem antecedent_true : ¬(Grumpy Tom ∧ MeanToAnimals Tom) := by
  intro h
  exact tom_not_mean h.right

-- The statement to evaluate:
-- If Tom is not both grumpy and mean to animals, then Tom is neither grumpy nor an animal lover.
-- We prove its negation.
theorem negate_statement :
  ¬( (¬(Grumpy Tom ∧ MeanToAnimals Tom)) → (¬Grumpy Tom ∧ ¬AnimalLover Tom) ) := by
  intro hImp
  have hconseq : (¬Grumpy Tom ∧ ¬AnimalLover Tom) := hImp antecedent_true
  have hNotAL : ¬AnimalLover Tom := hconseq.right
  exact hNotAL tom_animalLover
</lean>
