axiom obj : Type

-- Entities
axiom Tom : obj
axiom Fluffy : obj
axiom OliveGarden : obj

-- Predicates and functions
axiom Managed : obj → Prop
axiom AllowsPets : obj → Prop
axiom Pet : obj → Prop
axiom Cat : obj → Prop
axiom AllowedToMoveInWith : obj → obj → obj → Prop  -- person, building, pet
axiom WillRent : obj → obj → Prop                   -- person, building
axiom MonthlyRent : obj → Nat
axiom SecurityDeposit : obj → Nat
axiom RequiresDeposit : obj → Prop

-- Given facts
-- Pets are allowed in some managed buildings.
axiom some_managed_allows_pets : ∃ b : obj, Managed b ∧ AllowsPets b

-- A deposit is required to rent an apartment in a managed building.
axiom dep_required : ∀ b : obj, Managed b → RequiresDeposit b

-- The security deposit can be either equal to the monthly rent at a managed building or more (i.e., ≥).
axiom deposit_ge_rent : ∀ b : obj, Managed b → MonthlyRent b ≤ SecurityDeposit b

-- Fluffy is Tom's cat. Cats are pets.
axiom fluffy_cat : Cat Fluffy
axiom cats_are_pets : ∀ x : obj, Cat x → Pet x

-- The Olive Garden is a managed building.
axiom og_managed : Managed OliveGarden

-- The monthly rent at the Olive Garden is $2000.
axiom rent_og : MonthlyRent OliveGarden = 2000

-- $2000 is more than $1500.
axiom twoThousand_gt_1500 : 1500 < 2000

-- Tom will rent an apartment in a managed building if he is allowed to move in with Fluffy,
-- and the security deposit is no more than $1500.
axiom tom_rent_condition :
  ∀ b : obj, Managed b → (AllowedToMoveInWith Tom b Fluffy ∧ SecurityDeposit b ≤ 1500) → WillRent Tom b

-- If a managed building allows pets, then people are allowed to move in with a pet.
axiom allows_pets_implies_allowed_move_in :
  ∀ p b a : obj, Managed b → AllowsPets b → Pet a → AllowedToMoveInWith p b a

-- Useful derived fact: at Olive Garden, the deposit is > $1500 (since deposit ≥ rent = 2000 > 1500).
lemma deposit_gt_1500_og : 1500 < SecurityDeposit OliveGarden := by
  have h1 : MonthlyRent OliveGarden ≤ SecurityDeposit OliveGarden :=
    deposit_ge_rent OliveGarden og_managed
  have h2 : 1500 < MonthlyRent OliveGarden := by
    simpa [rent_og] using twoThousand_gt_1500
  exact Nat.lt_of_lt_of_le h2 h1

lemma not_deposit_le_1500_og : ¬(SecurityDeposit OliveGarden ≤ 1500) := by
  exact Nat.not_le.mpr deposit_gt_1500_og

-- Target statement to negate: AllowedToMoveInWith Tom OliveGarden Fluffy.
-- We cannot derive ¬AllowedToMoveInWith Tom OliveGarden Fluffy from the given axioms,
-- since we do not have that Olive Garden disallows pets, and allowedness depends on pet policy, not deposit.
-- Cannot prove ¬AllowedToMoveInWith Tom OliveGarden Fluffy from given axioms.
</lean>
