-- If a man is taller than another man, the taller one can block the other's shooting.
axiom R1 : ∀ x y : obj, Man x → Man y → Taller x y → Block x y

-- Michael is a man who is taller than everyone else in his class.
axiom Tm : Man Michael
axiom R2 : ∀ y : obj, InClass y → y ≠ Michael → Taller Michael y

-- Transitivity of 'taller than'
axiom R3 : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom Tp : Man Peter
axiom Tpm : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom R4 : ∀ p : obj, ¬Jumps p → Block Michael p

-- Michael cannot block Windy's shooting.
axiom Tw : ¬Block Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom R5 : ∀ x : obj, Jumps x → GreatShooter x

-- Natural property of "taller than": asymmetry (no cycles)
axiom R6 : ∀ x y : obj, Taller x y → ¬Taller y x

-- Optional: Michael is in his class (not needed for the proof, but plausible)
axiom C1 : InClass Michael

theorem peter_not_shorter_than_any_man_in_michaels_class :
  ¬(∃ m : obj, Man m ∧ InClass m ∧ Taller m Peter) := by
