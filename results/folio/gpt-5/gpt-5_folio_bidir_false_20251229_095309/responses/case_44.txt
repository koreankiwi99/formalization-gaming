axiom obj : Type
axiom Coco : obj

axiom RankedHigh : obj → Prop
axiom MostActive : obj → Prop
axiom LostToIgaRG2022 : obj → Prop
axiom FemaleRG2022 : obj → Prop
axiom MaleRG2022 : obj → Prop
axiom LostToRafaRG2022 : obj → Prop
axiom TennisPlayerRG2022 : obj → Prop

-- If someone is ranked highly by the WTA, then they are one of the most active players in major tennis.
axiom A1 : ∀ x : obj, RankedHigh x → MostActive x
-- Everyone who lost to Iga Świątek at Roland Garros 2022 is ranked highly by the WTA.
axiom A2 : ∀ x : obj, LostToIgaRG2022 x → RankedHigh x
-- All female tennis players at Roland Garros 2022 lost to Iga Świątek.
axiom A3 : ∀ x : obj, FemaleRG2022 x → LostToIgaRG2022 x
-- All tennis players at Roland Garros 2022 are either female or male.
axiom A4 : ∀ x : obj, TennisPlayerRG2022 x → (FemaleRG2022 x ∨ MaleRG2022 x)
-- All male tennis players at Roland Garros in 2022 lost to Rafael Nadal.
axiom A5 : ∀ x : obj, MaleRG2022 x → LostToRafaRG2022 x
-- If Coco Gauff is ranked highly by the WTA or lost to Rafael Nadal, then Coco Gauff is not a male tennis player at Roland Garros 2022.
axiom A6 : (RankedHigh Coco ∨ LostToRafaRG2022 Coco) → ¬MaleRG2022 Coco
-- Coco Gauff is at Roland Garros 2022.
axiom A7 : TennisPlayerRG2022 Coco

-- We prove the negation of the statement "Coco is not both LostToIga and MostActive",
-- i.e., we prove that Coco both lost to Iga at RG2022 and is one of the most active players.
theorem coco_lost_and_active : LostToIgaRG2022 Coco ∧ MostActive Coco := by
  have hfm : FemaleRG2022 Coco ∨ MaleRG2022 Coco := A4 Coco A7
  cases hfm with
  | inl hf =>
      have hLost : LostToIgaRG2022 Coco := A3 Coco hf
      have hRank : RankedHigh Coco := A2 Coco hLost
      have hAct  : MostActive Coco := A1 Coco hRank
      exact And.intro hLost hAct
  | inr hm =>
      have hRafa : LostToRafaRG2022 Coco := A5 Coco hm
      have hdisj : RankedHigh Coco ∨ LostToRafaRG2022 Coco := Or.inr hRafa
      have hNotM : ¬MaleRG2022 Coco := A6 hdisj
      have contra : False := hNotM hm
      exact False.elim contra
