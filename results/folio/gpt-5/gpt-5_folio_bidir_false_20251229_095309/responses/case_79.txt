axiom FromBakedByMelissa : obj → Prop

-- No baked sweets are spicy.
axiom H1 : ∀ x : obj, BakedSweet x → ¬ Spicy x
-- All cupcakes are baked sweets.
axiom H2 : ∀ x : obj, Cupcake x → BakedSweet x
-- All mala hotpots are spicy.
axiom H3 : ∀ x : obj, MalaHotpot x → Spicy x
-- All products from Baked by Melissa are cupcakes.
axiom H4 : ∀ x : obj, FromBakedByMelissa x → Cupcake x
-- Dried Thai chilies are spicy or mala hotpots or not baked sweets.
axiom H5 : Spicy DriedThaiChilies ∨ (MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies)

-- In fact, from the premises we can prove the original statement:
theorem not_fromBBM : ¬ FromBakedByMelissa DriedThaiChilies := by
  intro hBBM
  have hCup : Cupcake DriedThaiChilies := H4 DriedThaiChilies hBBM
  have hBake : BakedSweet DriedThaiChilies := H2 DriedThaiChilies hCup
  have hNotSpicy : ¬ Spicy DriedThaiChilies := H1 DriedThaiChilies hBake
  cases H5 with
  | inl hSpicy =>
      exact hNotSpicy hSpicy
  | inr hrest =>
      cases hrest with
      | inl hMala =>
          have hSpicy' : Spicy DriedThaiChilies := H3 DriedThaiChilies hMala
          exact hNotSpicy hSpicy'
      | inr hNotBake =>
          exact hNotBake hBake
