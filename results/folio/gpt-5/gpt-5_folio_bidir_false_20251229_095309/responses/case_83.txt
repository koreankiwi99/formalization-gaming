axiom ManyNeg : obj → Prop

-- If a restaurant is listed in Yelp’s recommendations, then it has not received many negative reviews.
axiom A1 : ∀ r : obj, YelpRec r → ¬ManyNeg r
-- All restaurants with a rating greater than four are listed in Yelp’s recommendations.
axiom A2 : ∀ r : obj, RatingGt4 r → YelpRec r
-- Some restaurants that do not provide take-out service receive many negative reviews.
axiom A3 : ∃ r : obj, ¬TakeOut r ∧ ManyNeg r
-- All restaurants popular among local residents have ratings greater than four.
axiom A4 : ∀ r : obj, Popular r → RatingGt4 r
-- The Hamden Plaza Subway store has a rating greater than four, or it is popular among local residents.
axiom A5 : RatingGt4 Hamden ∨ Popular Hamden

-- From A4 and A5, Hamden has rating > 4.
theorem ratingHamden : RatingGt4 Hamden := by
  cases A5 with
  | inl h => exact h
  | inr hp => exact A4 Hamden hp

-- Therefore Hamden is listed, hence does not receive many negative reviews.

  have y : YelpRec Hamden := A2 Hamden ratingHamden
  exact A1 Hamden y

-- But we have ¬ManyNeg Hamden, so we cannot derive the antecedent TakeOut Hamden ∧ ManyNeg Hamden,
