<lean>
axiom obj : Type
axiom Hannah : obj

axiom StudentJob : obj → Prop
axiom NeedsMoney : obj → Prop
axiom OrdersTakeoutFrequently : obj → Prop
axiom EnjoyDiningHall : obj → Prop
axiom Picky : obj → Prop
axiom SpendTimeDiningHalls : obj → Prop

-- If people work in student jobs on campus, then they need to earn money
axiom R1 : ∀ x : obj, StudentJob x → NeedsMoney x
-- If people order takeout frequently, then they work in student jobs on campus
axiom R2 : ∀ x : obj, OrdersTakeoutFrequently x → StudentJob x
-- People order takeout frequently or enjoy dining hall
axiom R3 : ∀ x : obj, OrdersTakeoutFrequently x ∨ EnjoyDiningHall x
-- If people enjoy dining hall, then they are not picky
axiom R4 : ∀ x : obj, EnjoyDiningHall x → ¬Picky x
-- If people enjoy dining hall, then they spend time in dining halls
axiom R5 : ∀ x : obj, EnjoyDiningHall x → SpendTimeDiningHalls x

-- Hannah works in student jobs on campus
axiom T1 : StudentJob Hannah
-- If Hannah needs money, then she is neither picky nor needs money
axiom T2 : NeedsMoney Hannah → (¬Picky Hannah ∧ ¬NeedsMoney Hannah)

theorem hannah_not_needs_money : ¬NeedsMoney Hannah := by
  have hN : NeedsMoney Hannah := R1 Hannah T1
  have hConj : (¬Picky Hannah ∧ ¬NeedsMoney Hannah) := T2 hN
  exact hConj.right
</lean>

ANSWER: False
