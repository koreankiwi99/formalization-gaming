axiom obj : Type
axiom Taylor : obj

axiom Salad : obj → Prop
axiom Conscious : obj → Prop
axiom HCHome : obj → Prop
axiom Fulfill : obj → Prop
axiom Disregard : obj → Prop
axiom GymDaily : obj → Prop

-- All people who eat salads regularly are very conscious about their health and eating habits.
axiom A1 : ∀ x : obj, Salad x → Conscious x
-- All people who grew up in health-conscious childhood homes eat salads regularly.
axiom A2 : ∀ x : obj, HCHome x → Salad x
-- All people who fulfill their daily nutritional intakes grew up in health-conscious childhood homes.
axiom A3 : ∀ x : obj, Fulfill x → HCHome x
-- All people who disregard their physical well-being are not very conscious about their health and eating habits.
axiom A4 : ∀ x : obj, Disregard x → ¬Conscious x
-- If people visit the gym at least once a day, then they always fulfill their daily nutritional intakes.
axiom A5 : ∀ x : obj, GymDaily x → Fulfill x
-- Taylor either (grew up in a health-conscious childhood home and disregards) or neither.
axiom A6 : (HCHome Taylor ∧ Disregard Taylor) ∨ (¬HCHome Taylor ∧ ¬Disregard Taylor)

-- The left disjunct is impossible: it yields Conscious and ¬Conscious
theorem not_left : ¬(HCHome Taylor ∧ Disregard Taylor) := by
  intro h
  have hs : Salad Taylor := A2 Taylor h.left
  have hc : Conscious Taylor := A1 Taylor hs
  have hnc : ¬Conscious Taylor := A4 Taylor h.right
  exact hnc hc

-- Therefore Taylor did neither: ¬HCHome Taylor ∧ ¬Disregard Taylor
theorem taylor_neither_home_nor_disregard : ¬HCHome Taylor ∧ ¬Disregard Taylor := by
  cases A6 with
  | inl h => exact False.elim (not_left h)
  | inr h => exact h

-- From ¬HCHome and (GymDaily → Fulfill → HCHome), deduce ¬GymDaily (by mt)
theorem taylor_not_gym : ¬GymDaily Taylor := by
  have notHome : ¬HCHome Taylor := (taylor_neither_home_nor_disregard).left
  have g2h : GymDaily Taylor → HCHome Taylor := by
    intro g
    have f : Fulfill Taylor := A5 Taylor g
    exact A3 Taylor f
  exact mt g2h notHome

-- The statement "Taylor neither grew up in a health-conscious childhood home nor does she visit the gym at least once a day"
-- is actually provable from the premises:
theorem statement_true : ¬HCHome Taylor ∧ ¬GymDaily Taylor := by
  apply And.intro
  · exact (taylor_neither_home_nor_disregard).left
  · exact taylor_not_gym
