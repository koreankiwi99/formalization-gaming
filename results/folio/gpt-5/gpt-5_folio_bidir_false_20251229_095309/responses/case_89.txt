-- Entities and predicates
axiom Person : obj → Prop
axiom Book : obj → Prop
axiom Reads : obj → obj → Prop
axiom GainsKnowledge : obj → Prop

-- Rules
axiom ReadGivesKnowledge : ∀ p b : obj, Person p → Book b → Reads p b → GainsKnowledge p
axiom KnowledgeImpliesSmarter : ∀ p : obj, GainsKnowledge p → Smarter p

-- Specific facts

axiom Walden : obj
axiom HarryIsPerson : Person Harry
axiom WaldenIsBook : Book Walden

  ReadGivesKnowledge Harry Walden HarryIsPerson WaldenIsBook HarryReadWalden

  KnowledgeImpliesSmarter Harry HarryGainedKnowledge

  Exists.intro Harry (And.intro HarryIsPerson (And.intro HarryIsSmarter HarryGainedKnowledge))
