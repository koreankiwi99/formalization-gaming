<lean>
axiom obj : Type
axiom NiceToAnimals : obj → Prop
axiom MeanToAnimals : obj → Prop
axiom Grumpy : obj → Prop
axiom LoveAnimals : obj → Prop
axiom PetOwner : obj → Prop
axiom Tom : obj

-- Context axioms
axiom H1 : ∀ x : obj, NiceToAnimals x → ¬ MeanToAnimals x
axiom H2 : ∃ x : obj, Grumpy x ∧ MeanToAnimals x
axiom H3 : ∀ x : obj, LoveAnimals x → NiceToAnimals x
axiom H4 : ∀ x : obj, PetOwner x → LoveAnimals x
axiom H5 : PetOwner Tom

-- The statement is: (¬(Grumpy Tom ∧ MeanToAnimals Tom)) → (¬Grumpy Tom ∧ ¬LoveAnimals Tom)
-- We prove its negation.
theorem statement_false :
  ¬ ((¬ (Grumpy Tom ∧ MeanToAnimals Tom)) → (¬ Grumpy Tom ∧ ¬ LoveAnimals Tom)) := by
  intro h
  have L : LoveAnimals Tom := H4 Tom H5
  have N : NiceToAnimals Tom := H3 Tom L
  have nM : ¬ MeanToAnimals Tom := H1 Tom N
  have P : ¬ (Grumpy Tom ∧ MeanToAnimals Tom) := by
    intro hgm
    exact nM hgm.right
  have notQ : ¬ (¬ Grumpy Tom ∧ ¬ LoveAnimals Tom) := by
    intro hconj
    exact hconj.right L
  have Q := h P
  exact notQ Q
</lean>

ANSWER: False
