<lean>
axiom obj : Type
axiom DriedThaiChilies : obj

axiom Spicy : obj → Prop
axiom BakedSweet : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom FromBBM : obj → Prop

-- No baked sweets are spicy.
axiom NoBakedSweetSpicy : ∀ x : obj, BakedSweet x → ¬ Spicy x
-- All cupcakes are baked sweets.
axiom CupcakeIsBakedSweet : ∀ x : obj, Cupcake x → BakedSweet x
-- All mala hotpots are spicy.
axiom MalaIsSpicy : ∀ x : obj, MalaHotpot x → Spicy x
-- All products from Baked by Melissa are cupcakes.
axiom BBMProductsAreCupcakes : ∀ x : obj, FromBBM x → Cupcake x
-- Dried Thai chilies are spicy or mala hotpots or not baked sweets.
axiom DTC_disj : Spicy DriedThaiChilies ∨ MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies

theorem dried_thai_chilies_not_BBM : ¬ FromBBM DriedThaiChilies :=
by
  have hCup : FromBBM DriedThaiChilies → Cupcake DriedThaiChilies :=
    BBMProductsAreCupcakes DriedThaiChilies
  have hBake : FromBBM DriedThaiChilies → BakedSweet DriedThaiChilies :=
    fun h => CupcakeIsBakedSweet DriedThaiChilies (hCup h)
  have hNotSpicy : FromBBM DriedThaiChilies → ¬ Spicy DriedThaiChilies :=
    fun h => NoBakedSweetSpicy DriedThaiChilies (hBake h)
  cases DTC_disj with
  | inl hs =>
      intro hBBM
      exact (hNotSpicy hBBM) hs
  | inr h2 =>
      cases h2 with
      | inl hm =>
          have hs : Spicy DriedThaiChilies := MalaIsSpicy DriedThaiChilies hm
          intro hBBM
          exact (hNotSpicy hBBM) hs
      | inr hnotBake =>
          intro hBBM
          exact hnotBake (hBake hBBM)
</lean>

ANSWER: True
