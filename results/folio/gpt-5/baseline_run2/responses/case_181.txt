axiom MeanToAnimals : obj → Prop
axiom Grumpy : obj → Prop

axiom PetOwner : obj → Prop

axiom A1 : ∀ x : obj, NiceToAnimals x → ¬ MeanToAnimals x
-- Some grumpy people are mean to animals.
axiom A2 : ∃ x : obj, Grumpy x ∧ MeanToAnimals x

-- All pet owners love animals.
axiom A4 : ∀ x : obj, PetOwner x → LovesAnimals x
-- Tom is a pet owner.
axiom A5 : PetOwner Tom

theorem tom_loves : LovesAnimals Tom := by
  exact A4 Tom A5

theorem tom_nice : NiceToAnimals Tom := by
  exact A3 Tom tom_loves

theorem tom_not_mean : ¬ MeanToAnimals Tom := by
  exact A1 Tom tom_nice
