<lean>
axiom obj : Type
axiom Marvin : obj
axiom Alien : obj → Prop
axiom Extra : obj → Prop
axiom Human : obj → Prop
axiom HighInt : obj → Prop
axiom FromMars : obj → Prop
axiom FromEarth : obj → Prop

axiom A1 : ∀ x : obj, Alien x → Extra x
axiom A2 : ∀ x : obj, FromMars x → Alien x
axiom A3 : ∀ x : obj, Extra x → ¬ Human x
axiom A4 : ∀ x : obj, (HighInt x ∧ FromEarth x) → Human x
axiom T5 : HighInt Marvin
axiom D1 : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬ FromEarth Marvin ∧ ¬ FromMars Marvin)
axiom A6 : ¬ FromEarth Marvin → Extra Marvin

theorem marvin_neither_human_nor_from_mars : (¬ Human Marvin) ∧ (¬ FromMars Marvin) := by
  cases D1 with
  | inl hboth =>
      have hFE : FromEarth Marvin := And.left hboth
      have hFM : FromMars Marvin := And.right hboth
      have hHI : HighInt Marvin := T5
      have hHuman : Human Marvin := A4 Marvin (And.intro hHI hFE)
      have hAlien : Alien Marvin := A2 Marvin hFM
      have hExtra : Extra Marvin := A1 Marvin hAlien
      have hNotHuman : ¬ Human Marvin := A3 Marvin hExtra
      exact False.elim (hNotHuman hHuman)
  | inr hneither =>
      have hNotFE : ¬ FromEarth Marvin := And.left hneither
      have hNotFM : ¬ FromMars Marvin := And.right hneither
      have hExtra : Extra Marvin := A6 hNotFE
      have hNotHuman : ¬ Human Marvin := A3 Marvin hExtra
      exact And.intro hNotHuman hNotFM
</lean>

ANSWER: True
