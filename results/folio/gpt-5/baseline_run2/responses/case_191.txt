<lean>
axiom obj : Type
axiom Yuri : obj
axiom ProfessionalBasketballPlayer : obj → Prop
axiom AmericanNational : obj → Prop
axiom CanJumpShots : obj → Prop
axiom LeapsStraight : obj → Prop
axiom ActivatesLegMuscles : obj → Prop

-- Some professional basketball players are not American nationals.
axiom ex_not_american_pbp : ∃ x : obj, ProfessionalBasketballPlayer x ∧ ¬ AmericanNational x

-- All professional basketball players can do jump shots.
axiom pbp_implies_cjs : ∀ x : obj, ProfessionalBasketballPlayer x → CanJumpShots x

-- If someone can jump shots, they leap straight into the air.
axiom cjs_implies_ls : ∀ x : obj, CanJumpShots x → LeapsStraight x

-- If someone leaps straight into the air, they activate their leg muscles.
axiom ls_implies_alm : ∀ x : obj, LeapsStraight x → ActivatesLegMuscles x

-- Yuri does not activate his leg muscles.
axiom yuri_not_alm : ¬ ActivatesLegMuscles Yuri

-- We prove the negation of the statement:
-- If Yuri is not an American professional basketball player, then Yuri is a professional basketball player.
theorem not_statement : ¬ ( (¬ (AmericanNational Yuri ∧ ProfessionalBasketballPlayer Yuri)) → ProfessionalBasketballPlayer Yuri ) :=
  fun h =>
    -- First, show Yuri is not a professional basketball player
    let pbp_to_alm : ProfessionalBasketballPlayer Yuri → ActivatesLegMuscles Yuri :=
      fun hp =>
        ls_implies_alm Yuri (cjs_implies_ls Yuri (pbp_implies_cjs Yuri hp))
    have not_pbp : ¬ ProfessionalBasketballPlayer Yuri :=
      fun hp => yuri_not_alm (pbp_to_alm hp)
    -- Antecedent holds because Yuri is not a professional basketball player
    have antecedent : ¬ (AmericanNational Yuri ∧ ProfessionalBasketballPlayer Yuri) :=
      fun h' => not_pbp h'.right
    -- Using the implication, we get Yuri is a professional basketball player, contradiction
    have hp : ProfessionalBasketballPlayer Yuri := h antecedent
    show False from not_pbp hp
</lean>
