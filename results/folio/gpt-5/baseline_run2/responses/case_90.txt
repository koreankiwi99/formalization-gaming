<lean>
axiom obj : Type
axiom MachineTranslation : obj
axiom NLP : obj → Prop
axiom LangGen : obj → Prop
axiom LangUnd : obj → Prop
axiom OutputTextSeq : obj → Prop

-- A natural language processing task can be categorized as either language generation or language understanding.
axiom A1 : ∀ x : obj, NLP x → (LangGen x ∨ LangUnd x)

-- If the output of a natural language processing task has text sequence output, then it is a language generation task.
axiom A2 : ∀ x : obj, NLP x → OutputTextSeq x → LangGen x

-- Machine translation is an NLP task.
axiom A3 : NLP MachineTranslation

-- The output of machine translation is a text sequence.
axiom A4 : OutputTextSeq MachineTranslation

theorem mt_is_langgen : LangGen MachineTranslation := by
  exact A2 MachineTranslation A3 A4
</lean>

ANSWER: True
