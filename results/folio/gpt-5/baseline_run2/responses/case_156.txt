<lean>
axiom Student : Type
axiom James : Student

axiom WorksLibrary : Student → Prop
axiom FromCS : Student → Prop
axiom UniPartTime : Student → Prop
axiom TakingDB : Student → Prop
axiom WithProfDavid : Student → Prop
axiom WorksLab : Student → Prop

-- None of the students who currently work in the library are from CS
axiom A1 : ∀ x : Student, WorksLibrary x → ¬ FromCS x
-- All students with university part-time jobs work in the library
axiom A2 : ∀ x : Student, UniPartTime x → WorksLibrary x
-- All students taking the database course are from CS
axiom A3 : ∀ x : Student, TakingDB x → FromCS x
-- All students taking a class with Professor David are taking the database course
axiom A4 : ∀ x : Student, WithProfDavid x → TakingDB x
-- All students working in the lab are taking a class with Professor David
axiom A5 : ∀ x : Student, WorksLab x → WithProfDavid x
-- James is working in the lab
axiom A6 : WorksLab James
-- James doesn't work in the lab nor have a university part-time job
axiom A7 : ¬ WorksLab James ∧ ¬ UniPartTime James

theorem james_has_uni_job : UniPartTime James :=
by
  have h1 : WorksLab James := A6
  have hnot : ¬ WorksLab James := And.left A7
  have hfalse : False := hnot h1
  exact False.elim hfalse
</lean>

ANSWER: True
