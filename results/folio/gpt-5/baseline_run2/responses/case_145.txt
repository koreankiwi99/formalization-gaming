axiom TWCitizen : Person → Prop

-- Everyone who can register in the US can participate in the 2024 US presidential election.
axiom A1 : ∀ x : Person, CanRegisterUS x → CanParticipate2024 x
-- If someone has US citizenship, then they can register to vote in the US.
axiom A2 : ∀ x : Person, USCitizen x → CanRegisterUS x
-- Every person either has US citizenship or Taiwanese citizenship.
axiom A3 : ∀ x : Person, USCitizen x ∨ TWCitizen x
-- No Russian Federation officials hold Taiwanese citizenship.
axiom A4 : ∀ x : Person, RFCOfficial x → ¬ TWCitizen x
-- Vladimir neither holds Taiwanese citizenship nor is he a manager at Gazprom.
axiom A5 : ¬ TWCitizen Vladimir ∧ ¬ ManagerGazprom Vladimir
-- Ekaterina can register in the US, or she is a Russian federation official.
axiom A6 : CanRegisterUS Ekaterina ∨ RFCOfficial Ekaterina

-- From A3 and ¬TWCitizen Vladimir, we can derive USCitizen Vladimir.
theorem vlad_US : USCitizen Vladimir := by
  have disj : USCitizen Vladimir ∨ TWCitizen Vladimir := A3 Vladimir
  have notTW : ¬ TWCitizen Vladimir := And.left A5
  apply Or.elim disj
  · intro hUS; exact hUS
  · intro hTW; exact False.elim (notTW hTW)

theorem vlad_register : CanRegisterUS Vladimir := A2 Vladimir vlad_US
theorem vlad_participate : CanParticipate2024 Vladimir := A1 Vladimir vlad_register
