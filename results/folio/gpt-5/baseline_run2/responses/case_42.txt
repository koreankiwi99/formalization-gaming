<lean>
axiom obj : Type
axiom Coco : obj

axiom RankedWTA : obj → Prop
axiom MostActive : obj → Prop
axiom LostToIga : obj → Prop
axiom Female : obj → Prop   -- female tennis player at Roland Garros 2022
axiom Male : obj → Prop     -- male tennis player at Roland Garros 2022
axiom LostToRafa : obj → Prop
axiom AtRG : obj → Prop     -- at Roland Garros 2022

-- If someone is ranked highly by the WTA, then they are among the most active (Grand-Slam/major) players.
axiom A1 : ∀ x : obj, RankedWTA x → MostActive x
-- Everyone who lost to Iga at Roland Garros 2022 is ranked highly by the WTA.
axiom A2 : ∀ x : obj, LostToIga x → RankedWTA x
-- All female tennis players at Roland Garros 2022 lost to Iga Świątek.
axiom A3 : ∀ x : obj, Female x → LostToIga x
-- All tennis players at Roland Garros 2022 are either female or male.
axiom A4 : ∀ x : obj, AtRG x → Female x ∨ Male x
-- All male tennis players at Roland Garros 2022 lost to Rafael Nadal.
axiom A5 : ∀ x : obj, Male x → LostToRafa x
-- If Coco is ranked highly by the WTA or lost to Rafael Nadal, then Coco is not a male tennis player at Roland Garros 2022.
axiom A6 : (RankedWTA Coco ∨ LostToRafa Coco) → ¬ Male Coco
-- Coco Gauff is at Roland Garros 2022.
axiom A7 : AtRG Coco

theorem coco_most_active : MostActive Coco := by
  -- First, show Coco is not male.
  have hNotMale : ¬ Male Coco := by
    intro hMale
    have hLoseRafa : LostToRafa Coco := A5 Coco hMale
    have hNoMale : ¬ Male Coco := A6 (Or.inr hLoseRafa)
    exact hNoMale hMale
  -- From being at RG and not male, deduce Coco is female.
  have hFemOrMale : Female Coco ∨ Male Coco := A4 Coco A7
  have hFemale : Female Coco := by
    cases hFemOrMale with
    | inl hf => exact hf
    | inr hm => exact False.elim (hNotMale hm)
  -- Female ⇒ lost to Iga ⇒ ranked highly ⇒ most active.
  have hLoseIga : LostToIga Coco := A3 Coco hFemale
  have hRanked : RankedWTA Coco := A2 Coco hLoseIga
  exact A1 Coco hRanked
</lean>

ANSWER: True
