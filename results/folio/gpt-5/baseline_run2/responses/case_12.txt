<lean>
axiom obj : Type

axiom Zelda : obj

axiom JapaneseCompany : obj → Prop
axiom Created : obj → obj → Prop
axiom Top10 : obj → Prop
axiom SoldOver1M : obj → Prop

-- A Japanese game company created The Legend of Zelda
axiom A1 : ∃ c : obj, JapaneseCompany c ∧ Created c Zelda
-- All games on the Top 10 list are made by Japanese game companies
axiom A2 : ∀ x : obj, Top10 x → ∃ c : obj, JapaneseCompany c ∧ Created c x
-- If a game sells more than one million copies, then it is included in the Top 10 list
axiom A3 : ∀ x : obj, SoldOver1M x → Top10 x
-- The Legend of Zelda sold more than one million copies
axiom A4 : SoldOver1M Zelda

theorem zelda_on_top10 : Top10 Zelda := by
  exact A3 Zelda A4
</lean>

ANSWER: True
