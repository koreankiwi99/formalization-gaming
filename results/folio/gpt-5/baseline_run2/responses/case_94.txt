<lean>
axiom obj : Type
axiom OliveGarden : obj

axiom Managed : obj → Prop
axiom MonthlyRent : obj → Nat
axiom SecurityDeposit : obj → Nat

-- Policy: In any managed building, the security deposit is either equal to the monthly rent or greater.
axiom dep_policy :
  ∀ b : obj, Managed b → (SecurityDeposit b = MonthlyRent b ∨ MonthlyRent b < SecurityDeposit b)

-- Facts about the Olive Garden
axiom managed_og : Managed OliveGarden
axiom rent_og : MonthlyRent OliveGarden = 2000

-- Goal: The security deposit at the Olive Garden is either $2000 or more.
theorem deposit_og_2000_or_more :
  SecurityDeposit OliveGarden = 2000 ∨ 2000 < SecurityDeposit OliveGarden := by
  have h := dep_policy OliveGarden managed_og
  cases h with
  | inl hEq =>
      have h2 : SecurityDeposit OliveGarden = 2000 := by
        simpa [rent_og] using hEq
      exact Or.inl h2
  | inr hLt =>
      have h2 : 2000 < SecurityDeposit OliveGarden := by
        simpa [rent_og] using hLt
      exact Or.inr h2
</lean>

ANSWER: True
