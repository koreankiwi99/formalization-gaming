<lean>
axiom obj : Type
axiom George : obj

axiom LikesMusic : obj → Prop
axiom WantsCompose : obj → Prop
axiom AccessProgram : obj → Prop
axiom CanCompose : obj → Prop
axiom WillCompose : obj → Prop

axiom A1 : ∀ x : obj, LikesMusic x → WantsCompose x
axiom A2 : ∀ x : obj, AccessProgram x → CanCompose x
axiom A3 : ∀ x : obj, CanCompose x ∧ WantsCompose x → WillCompose x

theorem george_will_compose : (LikesMusic George ∧ AccessProgram George) → WillCompose George :=
by
  intro h
  have hm : LikesMusic George := And.left h
  have ha : AccessProgram George := And.right h
  have hw : WantsCompose George := A1 George hm
  have hc : CanCompose George := A2 George ha
  exact A3 George (And.intro hc hw)
</lean>

ANSWER: True
