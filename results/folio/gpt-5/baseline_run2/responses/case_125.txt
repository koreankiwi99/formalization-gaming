<lean>
axiom obj : Type
axiom Rock : obj
axiom Pet : obj → Prop
axiom Mammal : obj → Prop
axiom Animal : obj → Prop
axiom Monkey : obj → Prop
axiom Bird : obj → Prop
axiom CanFly : obj → Prop
axiom CanBreathe : obj → Prop
axiom HasWings : obj → Prop

axiom A1 : ∀ x : obj, Pet x ∧ Mammal x → Animal x
axiom A2 : ∀ x : obj, Monkey x → Mammal x
axiom A3 : ∀ x : obj, Pet x → (Monkey x ∨ Bird x)
axiom A4 : ∀ x : obj, Pet x ∧ Bird x → CanFly x
axiom A5 : ∀ x : obj, Pet x ∧ Animal x → CanBreathe x
axiom A6 : ∀ x : obj, Pet x ∧ CanFly x → HasWings x
axiom A7 : Pet Rock
axiom A8 : CanFly Rock ∨ Bird Rock ∨ ¬ CanBreathe Rock

theorem rock_has_wings : HasWings Rock := by
  have hPet : Pet Rock := A7
  have h : CanFly Rock ∨ Bird Rock ∨ ¬ CanBreathe Rock := A8
  cases h with
  | inl hFly =>
      exact A6 Rock (And.intro hPet hFly)
  | inr hrest =>
      cases hrest with
      | inl hBird =>
          have hCF : CanFly Rock := A4 Rock (And.intro hPet hBird)
          exact A6 Rock (And.intro hPet hCF)
      | inr hNotBreathe =>
          have hMB : Monkey Rock ∨ Bird Rock := A3 Rock hPet
          cases hMB with
          | inl hMonkey =>
              have hMammal : Mammal Rock := A2 Rock hMonkey
              have hAnimal : Animal Rock := A1 Rock (And.intro hPet hMammal)
              have hBreathe : CanBreathe Rock := A5 Rock (And.intro hPet hAnimal)
              have : False := hNotBreathe hBreathe
              exact False.elim this
          | inr hBird2 =>
              have hCF2 : CanFly Rock := A4 Rock (And.intro hPet hBird2)
              exact A6 Rock (And.intro hPet hCF2)
</lean>

ANSWER: True
