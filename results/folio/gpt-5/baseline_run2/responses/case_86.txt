<lean>
axiom obj : Type
axiom Movie : obj
axiom Digby : obj
axiom Nemesis : obj

axiom SuperheroMovie : obj → Prop
axiom NamedAfter : obj → obj → Prop
axiom InMovie : obj → obj → Prop
axiom Good : obj → Prop
axiom Bad : obj → Prop
axiom Fights : obj → obj → Prop
axiom Wins : obj → Prop

-- In superhero movies, the good guys always win.
axiom A1 : ∀ m : obj, SuperheroMovie m → ∀ x : obj, InMovie x m → Good x → Wins x
-- The Surprising Adventures of Sir Digby Chicken Caesar is a superhero movie.
axiom A2 : SuperheroMovie Movie
-- Good guys fight bad guys and vice versa.
axiom A3 : ∀ x y : obj, Good x → Bad y → Fights x y ∧ Fights y x
-- Sir Digby fights his nemesis.
axiom A4 : Fights Digby Nemesis
-- If a superhero movie is named after a character, that character is a good guy.
axiom A5 : ∀ m x : obj, SuperheroMovie m → NamedAfter m x → Good x
-- The Surprising Adventures of Sir Digby Chicken Caesar is named after Sir Digby.
axiom A6 : NamedAfter Movie Digby
-- If somebody wins a fight, the person they are fighting does not win.
axiom A7 : ∀ x y : obj, Fights x y → Wins x → ¬ Wins y
-- If a superhero movie is named after a character, that character is in the movie.
axiom A8 : ∀ m x : obj, SuperheroMovie m → NamedAfter m x → InMovie x m

theorem nemesis_does_not_win : ¬ Wins Nemesis := by
  have hGood : Good Digby := A5 Movie Digby A2 A6
  have hIn : InMovie Digby Movie := A8 Movie Digby A2 A6
  have hWin : Wins Digby := A1 Movie A2 Digby hIn hGood
  exact A7 Digby Nemesis A4 hWin
</lean>

ANSWER: True
