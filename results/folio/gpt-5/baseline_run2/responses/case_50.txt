<lean>
axiom obj : Type
axiom NineteenEightyFour : obj

axiom Digital : obj → Prop
axiom Analog : obj → Prop
axiom Printed : obj → Prop
axiom Streaming : obj → Prop
axiom Hardcover : obj → Prop

-- No digital media are analog.
axiom NoDigitalAreAnalog : ∀ x : obj, Digital x → ¬ Analog x
-- Every printed text is analog media.
axiom PrintedAreAnalog : ∀ x : obj, Printed x → Analog x
-- All streaming services are digital media.
axiom StreamingAreDigital : ∀ x : obj, Streaming x → Digital x
-- If an object is a hardcover book, then it is printed text.
axiom HardcoverArePrinted : ∀ x : obj, Hardcover x → Printed x
-- If 1984 is a streaming service, then 1984 is a hardcover book.
axiom IfStreamingThenHardcover1984 : Streaming NineteenEightyFour → Hardcover NineteenEightyFour

theorem not_streaming_1984 : ¬ Streaming NineteenEightyFour := by
  intro hS
  have hD : Digital NineteenEightyFour := StreamingAreDigital NineteenEightyFour hS
  have hHC : Hardcover NineteenEightyFour := IfStreamingThenHardcover1984 hS
  have hP : Printed NineteenEightyFour := HardcoverArePrinted NineteenEightyFour hHC
  have hA : Analog NineteenEightyFour := PrintedAreAnalog NineteenEightyFour hP
  have hNotA : ¬ Analog NineteenEightyFour := NoDigitalAreAnalog NineteenEightyFour hD
  exact hNotA hA
</lean>

ANSWER: True
