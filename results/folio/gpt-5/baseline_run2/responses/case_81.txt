<lean>
axiom obj : Type
axiom DriedThaiChilies : obj

axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom FromBBM : obj → Prop

-- No baked sweets are spicy.
axiom A1 : ∀ x : obj, BakedSweet x → ¬Spicy x
-- All cupcakes are baked sweets.
axiom A2 : ∀ x : obj, Cupcake x → BakedSweet x
-- All mala hotpots are spicy.
axiom A3 : ∀ x : obj, MalaHotpot x → Spicy x
-- All products from Baked by Melissa are cupcakes.
axiom A4 : ∀ x : obj, FromBBM x → Cupcake x
-- Dried Thai chilies are spicy or mala hotpots or not baked sweets.
axiom A5 : (Spicy DriedThaiChilies ∨ MalaHotpot DriedThaiChilies) ∨ ¬BakedSweet DriedThaiChilies

theorem dried_thai_chilies_neither :
  ¬FromBBM DriedThaiChilies ∧ ¬BakedSweet DriedThaiChilies := by
  -- First, show ¬ BakedSweet DriedThaiChilies
  have notBS : ¬BakedSweet DriedThaiChilies := by
    cases A5 with
    | inl h =>
      cases h with
      | inl spicy =>
        intro bs
        have notSp : ¬Spicy DriedThaiChilies := A1 DriedThaiChilies bs
        exact notSp spicy
      | inr mh =>
        intro bs
        have sp : Spicy DriedThaiChilies := A3 DriedThaiChilies mh
        have notSp : ¬Spicy DriedThaiChilies := A1 DriedThaiChilies bs
        exact notSp sp
    | inr notbs =>
      exact notbs
  -- Then, show ¬ FromBBM DriedThaiChilies using FromBBM → Cupcake → BakedSweet
  have notFrom : ¬FromBBM DriedThaiChilies := by
    intro from
    have cup : Cupcake DriedThaiChilies := A4 DriedThaiChilies from
    have bs : BakedSweet DriedThaiChilies := A2 DriedThaiChilies cup
    exact notBS bs
  exact And.intro notFrom notBS
</lean>

ANSWER: True
