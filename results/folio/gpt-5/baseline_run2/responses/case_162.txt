<lean>
axiom obj : Type
axiom Bird : obj → Prop
axiom Hawk : obj → Prop
axiom Swims : obj → Prop

axiom hawk_never_swims : ∀ x : obj, Hawk x → ¬ Swims x
axiom some_birds_are_hawks : ∃ x : obj, Bird x ∧ Hawk x

theorem not_all_birds_swim : ¬ (∀ x : obj, Bird x → Swims x) := by
  intro h
  cases some_birds_are_hawks with
  | intro x hx =>
    have bx : Bird x := And.left hx
    have hk : Hawk x := And.right hx
    have ns : ¬ Swims x := hawk_never_swims x hk
    have s : Swims x := h x bx
    exact ns s
</lean>

ANSWER: False
