axiom Airport : Type

axiom John : Person
axiom LGA : Airport

axiom From : Person → Airport → Prop
axiom To : Person → Airport → Prop

-- The departure and arrival cannot be at the same airport
axiom not_same_airport : ∀ p a, From p a → ¬ To p a

axiom John_from_LGA : From John LGA

-- Therefore, John does not fly to LGA
theorem john_not_to_LGA : ¬ To John LGA := by
  exact not_same_airport John LGA John_from_LGA
