axiom Michael : obj

axiom Taller : obj → obj → Prop
axiom CanBlockShooting : obj → obj → Prop
axiom CanJumpWhenShooting : obj → Prop

axiom GreatShooter : obj → Prop
axiom InMichaelsClass : obj → Prop

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom R_taller_blocks : ∀ x y : obj, Man x → Man y → Taller x y → CanBlockShooting x y

-- Michael is a man who is taller than everyone else in his class.
axiom T_Michael_man : Man Michael
axiom T_Michael_taller_class : ∀ y : obj, InMichaelsClass y → Taller Michael y

-- Transitivity of taller
axiom R_taller_trans : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom T_Peter_man : Man Peter
axiom T_Peter_taller_Michael : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom R_Michael_blocks_if_no_jump : ∀ y : obj, ¬CanJumpWhenShooting y → CanBlockShooting Michael y

-- Michael cannot block Windy's shooting.
axiom T_not_block_Windy : ¬ CanBlockShooting Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom R_jumpers_are_great : ∀ x : obj, Shooter x ∧ CanJumpWhenShooting x → GreatShooter x

-- Some derivable facts (not the target):
theorem Peter_blocks_Michael : CanBlockShooting Peter Michael :=
  R_taller_blocks Peter Michael T_Peter_man T_Michael_man T_Peter_taller_Michael

-- From Michael's blocking rule and the fact he cannot block Windy, we can derive ¬¬(Windy jumps).
theorem Windy_double_neg_jump : ¬¬ CanJumpWhenShooting Windy := by
  intro h_no_jump
  have h_block : CanBlockShooting Michael Windy := R_Michael_blocks_if_no_jump Windy h_no_jump
  exact T_not_block_Windy h_block
