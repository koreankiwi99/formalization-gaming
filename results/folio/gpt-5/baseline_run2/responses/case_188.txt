<lean>
axiom obj : Type
axiom Yuri : obj
axiom American : obj → Prop
axiom ProBasketballPlayer : obj → Prop
axiom CanJumpShots : obj → Prop
axiom LeapsStraightUp : obj → Prop
axiom ActivatesLegMuscles : obj → Prop

axiom S1 : ∃ x : obj, ProBasketballPlayer x ∧ ¬ American x
axiom R1 : ∀ x : obj, ProBasketballPlayer x → CanJumpShots x
axiom R2 : ∀ x : obj, CanJumpShots x → LeapsStraightUp x
axiom R3 : ∀ x : obj, LeapsStraightUp x → ActivatesLegMuscles x
axiom T_Yuri : ¬ ActivatesLegMuscles Yuri

theorem yuri_not_pro : ¬ ProBasketballPlayer Yuri := by
  intro hPro
  have hCJ : CanJumpShots Yuri := R1 Yuri hPro
  have hL : LeapsStraightUp Yuri := R2 Yuri hCJ
  have hA : ActivatesLegMuscles Yuri := R3 Yuri hL
  exact T_Yuri hA

theorem yuri_not_american_pro : ¬ (American Yuri ∧ ProBasketballPlayer Yuri) := by
  intro h
  exact yuri_not_pro (And.right h)
</lean>

ANSWER: True
