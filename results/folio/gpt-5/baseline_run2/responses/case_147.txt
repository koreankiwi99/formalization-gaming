<lean>
axiom Person : Type
axiom Ekaterina : Person
axiom Vladimir : Person

axiom CanRegisterUS : Person → Prop
axiom Participate2024 : Person → Prop
axiom UScit : Person → Prop
axiom TWcit : Person → Prop
axiom RFofficial : Person → Prop
axiom MgrGazprom : Person → Prop

-- Rules
axiom R1 : ∀ x : Person, CanRegisterUS x → Participate2024 x
axiom R2 : ∀ x : Person, UScit x → CanRegisterUS x
axiom R3 : ∀ x : Person, UScit x ∨ TWcit x
axiom R4 : ∀ x : Person, RFofficial x → ¬ TWcit x

-- Facts
axiom V1 : ¬ TWcit Vladimir
axiom V2 : ¬ MgrGazprom Vladimir
axiom E1 : CanRegisterUS Ekaterina ∨ RFofficial Ekaterina

theorem ekaterina_participate_or_vladimir_manager :
  Participate2024 Ekaterina ∨ MgrGazprom Vladimir := by
  have hPartEk : Participate2024 Ekaterina := by
    cases E1 with
    | inl hreg =>
      exact R1 Ekaterina hreg
    | inr hoff =>
      have hnotTW : ¬ TWcit Ekaterina := R4 Ekaterina hoff
      have hCit : UScit Ekaterina ∨ TWcit Ekaterina := R3 Ekaterina
      cases hCit with
      | inl hUS =>
        have hReg : CanRegisterUS Ekaterina := R2 Ekaterina hUS
        exact R1 Ekaterina hReg
      | inr hTW =>
        exact False.elim (hnotTW hTW)
  exact Or.inl hPartEk
</lean>

ANSWER: True
