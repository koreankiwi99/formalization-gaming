<lean>
axiom obj : Type
axiom Harry : obj

axiom Invertebrate : obj → Prop
axiom Vertebrate : obj → Prop
axiom Backbone : obj → Prop
axiom Mating : obj → Prop
axiom Bee : obj → Prop
axiom QueenBee : obj → Prop

-- Animals are either invertebrates or vertebrates (domain is animals)
axiom A1 : ∀ x : obj, Invertebrate x ∨ Vertebrate x
-- All animals with backbones reproduce by male-and-female mating
axiom A2 : ∀ x : obj, Backbone x → Mating x
-- All vertebrate animals have a backbone
axiom A3 : ∀ x : obj, Vertebrate x → Backbone x
-- All bees do not reproduce by male-and-female mating
axiom A4 : ∀ x : obj, Bee x → ¬ Mating x
-- All queen bees are bees
axiom A5 : ∀ x : obj, QueenBee x → Bee x
-- Harry is a bee
axiom A6 : Bee Harry

theorem harry_invertebrate_or_queenbee : Invertebrate Harry ∨ QueenBee Harry := by
  have nM : ¬ Mating Harry := A4 Harry A6
  have nB : ¬ Backbone Harry := by
    intro hb
    exact nM (A2 Harry hb)
  have nV : ¬ Vertebrate Harry := by
    intro vh
    exact nB (A3 Harry vh)
  have disj := A1 Harry
  have iH : Invertebrate Harry := by
    cases disj with
    | inl i => exact i
    | inr v => exact False.elim (nV v)
  exact Or.inl iH
</lean>

ANSWER: True
