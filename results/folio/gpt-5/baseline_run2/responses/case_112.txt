-- No Texas mountain range has a mine that has been donated
axiom no_TX_range_with_donated_mine :
  ∀ r : obj, MountainRange r → InTexas r → ¬ (∃ m : obj, Mine m ∧ LocatedIn m r ∧ Donated m)

-- From the donated mine in Picuris, deduce Picuris is not in Texas
theorem not_in_texas_picuris : ¬ InTexas Picuris := by
  intro htx
  have hnot : ¬ (∃ m, Mine m ∧ LocatedIn m Picuris ∧ Donated m) :=
    no_TX_range_with_donated_mine Picuris PM_is_mountain htx
  have hex : ∃ m, Mine m ∧ LocatedIn m Picuris ∧ Donated m := by
    refine Exists.intro HardingMine ?_
    exact And.intro HM_is_mine (And.intro HM_in_PM HM_donated)
  exact hnot hex

-- From "in New Mexico or Texas" and not in Texas, we can conclude in New Mexico
theorem in_new_mexico_picuris : InNewMexico Picuris := by
  cases PM_location with
  | inl hnm => exact hnm
  | inr htx =>
    exact False.elim (not_in_texas_picuris htx)
