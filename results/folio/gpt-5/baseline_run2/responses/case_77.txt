<lean>
axiom obj : Type
axiom At : obj → Prop
axiom Works : obj → Prop
axiom Needs : obj → Prop
axiom Orders : obj → Prop
axiom Enjoys : obj → Prop
axiom Picky : obj → Prop
axiom Spends : obj → Prop
axiom Hannah : obj

axiom R1 : ∀ x : obj, At x ∧ Works x → Needs x
axiom R2 : ∀ x : obj, At x ∧ Orders x → Works x
axiom R3 : ∀ x : obj, At x → (Orders x ∨ Enjoys x)
axiom R4 : ∀ x : obj, At x ∧ Enjoys x → ¬ Picky x
axiom R5 : ∀ x : obj, At x ∧ Enjoys x → Spends x
axiom TAt : At Hannah
axiom TWork : Works Hannah
axiom THimp : Needs Hannah → (¬ Picky Hannah ∧ ¬ Needs Hannah)

theorem hannah_statement :
  At Hannah ∧ (¬ Picky Hannah ∨ (Picky Hannah → Spends Hannah)) := by
  have hNeeds : Needs Hannah := by
    apply R1 Hannah
    exact And.intro TAt TWork
  have hNotPicky : ¬ Picky Hannah := by
    have h := THimp hNeeds
    exact And.left h
  apply And.intro
  · exact TAt
  · apply Or.inl
    exact hNotPicky
</lean>

ANSWER: True
