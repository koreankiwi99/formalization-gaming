<lean>
axiom obj : Type

-- Entities
axiom Picuris : obj
axiom Juan : obj
axiom HPM : obj

-- Predicates
axiom MountainRange : obj → Prop
axiom InNewMexico : obj → Prop
axiom InTexas : obj → Prop
axiom Visited : obj → obj → Prop
axiom Mine : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop

-- Context axioms
axiom T1 : MountainRange Picuris
axiom T2 : InNewMexico Picuris ∨ InTexas Picuris
axiom T3 : Visited Juan Picuris
axiom T4 : Mine HPM
axiom T5 : LocatedIn HPM Picuris
axiom T6 : Donated HPM
axiom R1 : ∀ x : obj, MountainRange x → InTexas x → ¬ (∃ y : obj, Mine y ∧ LocatedIn y x ∧ Donated y)

-- Goal: Juan de Onate visited a mountain range in New Mexico.
theorem visited_mountain_range_in_nm : ∃ x : obj, MountainRange x ∧ InNewMexico x ∧ Visited Juan x := by
  -- There exists a donated mine located in Picuris
  have hExists : ∃ y : obj, Mine y ∧ LocatedIn y Picuris ∧ Donated y := by
    exact Exists.intro HPM (And.intro T4 (And.intro T5 T6))
  -- Hence Picuris is not in Texas
  have hNotTX : ¬ InTexas Picuris := by
    intro hTX
    have hNoExist : ¬ (∃ y : obj, Mine y ∧ LocatedIn y Picuris ∧ Donated y) := R1 Picuris T1 hTX
    exact hNoExist hExists
  -- From "in NM or TX" and "not in TX", deduce "in NM"
  have hNM : InNewMexico Picuris := by
    cases T2 with
    | inl hnm => exact hnm
    | inr htx => exact False.elim (hNotTX htx)
  -- Conclude existence
  exact Exists.intro Picuris (And.intro T1 (And.intro hNM T3))
</lean>

ANSWER: True
