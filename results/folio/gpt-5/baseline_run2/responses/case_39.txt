axiom obj : Type

axiom KellyStyle : obj → Prop
axiom Adores : obj → Prop
axiom Interesting : obj → Prop
axiom Brutalist : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop
axiom DesignByMax : obj → Prop

-- All of Zaha Hadid's design styles that Max adores have interesting geometries.
axiom a1 : ∀ x : obj, ZahaStyle x ∧ Adores x → Interesting x

-- No brutalist buildings that Max adores have interesting geometries.
axiom a2 : ∀ x : obj, Brutalist x ∧ Adores x → ¬ Interesting x

-- Every style that Max adores is either Zaha Hadid's design style or Kelly Wearstler's design style.
axiom a3 : ∀ x : obj, Adores x → ZahaStyle x ∨ KellyStyle x

-- All of Kelly Wearstler's design styles that Max adores are evocative.
axiom a4 : ∀ x : obj, KellyStyle x ∧ Adores x → Evocative x

-- All of Kelly Wearstler's design styles that Max adores are dreamy.
axiom a5 : ∀ x : obj, KellyStyle x ∧ Adores x → Dreamy x

-- If a design by Max that he adores has interesting geometries, then the design is a brutalist building and evocative.
axiom a6 : ∀ x : obj, DesignByMax x ∧ Adores x ∧ Interesting x → Brutalist x ∧ Evocative x

-- A useful consequence: there is no design by Max that he adores which also has interesting geometries.
theorem no_adored_interesting_design_by_max :
  ∀ x : obj, DesignByMax x → Adores x → Interesting x → False :=
by
  intro x
  intro hx
  intro hA
  intro hI
  have h := a6 x ⟨hx, hA, hI⟩
  have hB : Brutalist x := And.left h
  have hNotI : ¬ Interesting x := a2 x ⟨hB, hA⟩
  exact hNotI hI
