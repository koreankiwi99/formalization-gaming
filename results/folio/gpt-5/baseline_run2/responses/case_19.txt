<lean>
axiom obj : Type
axiom Rabbit : obj → Prop
axiom Squirrel : obj → Prop
axiom Turtle : obj → Prop
axiom Near : obj → Prop
axiom Cute : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop

axiom Rockie : obj

-- All rabbits that can be spotted near the campus are cute.
axiom A1 : ∀ x : obj, Rabbit x → Near x → Cute x
-- Some turtles can be spotted near the campus.
axiom A2 : ∃ x : obj, Turtle x ∧ Near x
-- The only animals that can be spotted near the campus are rabbits and squirrels.
axiom A3 : ∀ x : obj, Near x → (Rabbit x ∨ Squirrel x)
-- If something is skittish, then it is not calm.
axiom A4 : ∀ x : obj, Skittish x → ¬ Calm x
-- All the squirrels that can be spotted near the campus are skittish.
axiom A5 : ∀ x : obj, Squirrel x → Near x → Skittish x
-- Rockie can be spotted near the campus, and it is calm.
axiom A6 : Near Rockie ∧ Calm Rockie

theorem rockie_cute_or_skittish : Cute Rockie ∨ Skittish Rockie := by
  have hNear : Near Rockie := And.left A6
  have hRS : Rabbit Rockie ∨ Squirrel Rockie := A3 Rockie hNear
  cases hRS with
  | inl hRab =>
      exact Or.inl (A1 Rockie hRab hNear)
  | inr hSq =>
      exact Or.inr (A5 Rockie hSq hNear)

theorem main : ¬(Turtle Rockie ∧ Squirrel Rockie) → (Cute Rockie ∨ Skittish Rockie) := by
  intro _
  exact rockie_cute_or_skittish
</lean>

ANSWER: True
