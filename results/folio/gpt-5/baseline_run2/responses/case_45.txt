<lean>
axiom obj : Type
axiom Cat : obj → Prop
axiom Mammal : obj → Prop
axiom Pet : obj → Prop

-- All cats are mammals.
axiom A1 : ∀ x : obj, Cat x → Mammal x

-- Some pets are not mammals.
axiom A2 : ∃ x : obj, Pet x ∧ ¬Mammal x

-- From the premises we can derive that some pet is not a cat.
-- (Because all cats are mammals, any non-mammal cannot be a cat.)
theorem some_pet_not_cat : ∃ x : obj, Pet x ∧ ¬Cat x := by
  rcases A2 with ⟨a, ha⟩
  rcases ha with ⟨haPet, haNotMammal⟩
  have hNotCat : ¬Cat a := fun hCat => haNotMammal (A1 a hCat)
  exact ⟨a, ⟨haPet, hNotCat⟩⟩
