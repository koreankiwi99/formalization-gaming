<lean>
axiom obj : Type
axiom HeinrichSchmidt : obj
axiom German : obj → Prop
axiom Politician : obj → Prop
axiom MemberPSP : obj → Prop
axiom MemberNR : obj → Prop

axiom HG : German HeinrichSchmidt
axiom HP : Politician HeinrichSchmidt
axiom HM_PSP : MemberPSP HeinrichSchmidt
axiom HM_NR : MemberNR HeinrichSchmidt

theorem some_german_politician_both :
  ∃ x : obj, German x ∧ Politician x ∧ MemberPSP x ∧ MemberNR x :=
by
  refine Exists.intro HeinrichSchmidt ?_
  exact And.intro HG (And.intro HP (And.intro HM_PSP HM_NR))
</lean>

ANSWER: True
