<lean>
axiom obj : Type
axiom BrownSwissCattle : obj → Prop
axiom Cow : obj → Prop
axiom Pet : obj → Prop
axiom Domesticated : obj → Prop
axiom Alligator : obj → Prop
axiom Ted : obj

axiom all_bsc_are_cows : ∀ x : obj, BrownSwissCattle x → Cow x
axiom some_pet_bsc : ∃ x : obj, Pet x ∧ BrownSwissCattle x
axiom all_cows_domesticated : ∀ x : obj, Cow x → Domesticated x
axiom alligators_not_domesticated : ∀ x : obj, Alligator x → ¬ Domesticated x
axiom ted_alligator : Alligator Ted

theorem ted_not_pet_if_bsc : BrownSwissCattle Ted → ¬ Pet Ted := by
  intro hBSC
  have hCow : Cow Ted := all_bsc_are_cows Ted hBSC
  have hDom : Domesticated Ted := all_cows_domesticated Ted hCow
  have hNotDom : ¬ Domesticated Ted := alligators_not_domesticated Ted ted_alligator
  have hFalse : False := hNotDom hDom
  intro _
  exact False.elim hFalse
</lean>

ANSWER: True
