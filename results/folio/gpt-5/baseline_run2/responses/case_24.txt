<lean>
axiom obj : Type
axiom Beijing : obj

axiom Event : Type
axiom Hosted : obj → Event → Prop

axiom SummerOlympics2008 : Event
axiom WinterOlympics : Event

axiom H1 : Hosted Beijing SummerOlympics2008
axiom H2 : Hosted Beijing WinterOlympics

  exact And.intro H1 H2
