axiom BlackMirror : obj

axiom Popular : obj → Prop
axiom Binge : obj → Prop        -- Karen binge-watches the show
axiom Download : obj → Prop     -- Karen downloads the show
axiom ShareWithLisa : obj → Prop -- Karen shares the show with Lisa

-- "Stranger Things" is a popular Netflix show.
axiom ST_netflix : NetflixShow StrangerThings
axiom ST_popular : Popular StrangerThings

-- If a Netflix show is popular, Karen will binge-watch it.
axiom R1 : ∀ s : obj, NetflixShow s → Popular s → Binge s

-- If and only if Karen binge-watches a Netflix show, she will download it.
axiom R2 : ∀ s : obj, NetflixShow s → (Binge s ↔ Download s)

-- Karen does not download "Black Mirror."
axiom R3 : ¬ Download BlackMirror

-- "Black Mirror" is a Netflix show.
axiom R4 : NetflixShow BlackMirror

-- If Karen binge-watches a Netflix show, she will share it with Lisa.
axiom R5 : ∀ s : obj, NetflixShow s → Binge s → ShareWithLisa s

lemma bm_not_binge : ¬ Binge BlackMirror := by
  have nfx : NetflixShow BlackMirror := R4
  have hiff : Binge BlackMirror ↔ Download BlackMirror := R2 BlackMirror nfx
  have nd : ¬ Download BlackMirror := R3
  intro hb
  have d : Download BlackMirror := (Iff.mp hiff) hb
  exact nd d

theorem bm_not_popular : ¬ Popular BlackMirror := by
  have nfx : NetflixShow BlackMirror := R4
  have nb : ¬ Binge BlackMirror := bm_not_binge

  have hb : Binge BlackMirror := R1 BlackMirror nfx p
