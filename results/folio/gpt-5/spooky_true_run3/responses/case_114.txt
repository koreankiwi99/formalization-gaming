<lean>
axiom obj : Type

axiom Man : obj → Prop
axiom Taller : obj → obj → Prop
axiom CanBlock : obj → obj → Prop
axiom InMClass : obj → Prop
axiom Jumps : obj → Prop
axiom Shooter : obj → Prop
axiom Great : obj → Prop

axiom Michael : obj
axiom Peter : obj
axiom Windy : obj

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom TallManBlocks : ∀ x y : obj, Man x → Man y → Taller x y → CanBlock x y

-- Michael is a man who is taller than everyone else in his class.
axiom Michael_is_man : Man Michael
axiom Michael_taller_class : ∀ y : obj, InMClass y → Taller Michael y

-- Transitivity of taller-than
axiom Taller_trans : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom Peter_is_man : Man Peter
axiom Peter_taller_Michael : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom Michael_blocks_if_no_jump : ∀ p : obj, ¬ Jumps p → CanBlock Michael p

-- Michael cannot block Windy's shooting.
axiom Michael_cannot_block_Windy : ¬ CanBlock Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom jump_shooter_great : ∀ s : obj, Shooter s → Jumps s → Great s

-- Derivable: Peter is taller than anyone in Michael's class.
theorem peter_taller_than_any_in_michael_class : ∀ y : obj, InMClass y → Taller Peter y := by
  intro y hy
  have hMy : Taller Michael y := Michael_taller_class y hy
  exact Taller_trans Peter Michael y Peter_taller_Michael hMy

-- Derivable: Peter can block any man in Michael's class (since he's taller than them).
theorem peter_can_block_any_man_in_michael_class : ∀ y : obj, Man y → InMClass y → CanBlock Peter y := by
  intro y hyMan hyIn
  have hTy : Taller Peter y := peter_taller_than_any_in_michael_class y hyIn
  exact TallManBlocks Peter y Peter_is_man hyMan hTy
