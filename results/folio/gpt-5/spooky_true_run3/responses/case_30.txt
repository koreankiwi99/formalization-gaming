<lean>
axiom obj : Type

axiom Mixer : obj → Prop
axiom GrandSlamChampion : obj → Prop
axiom OscarNominatedActor : obj → Prop
axiom ProTennisPlayer : obj → Prop
axiom Celebrity : obj → Prop
axiom Athlete : obj → Prop
axiom WellPaid : obj → Prop
axiom Famous : obj → Prop
axiom LivesInTaxHavens : obj → Prop

axiom Djokovic : obj

-- Premises
axiom P1 : ∀ x : obj, Mixer x → (GrandSlamChampion x ∨ OscarNominatedActor x)
axiom P2 : ∀ x : obj, Mixer x → GrandSlamChampion x → ProTennisPlayer x
axiom P3 : ∀ x : obj, Mixer x → OscarNominatedActor x → Celebrity x
axiom P4 : ∀ x : obj, Mixer x → ProTennisPlayer x → Athlete x
axiom P5 : ∀ x : obj, Mixer x → Celebrity x → WellPaid x
axiom P6 : ∀ x : obj, Mixer x → Athlete x → Famous x
axiom P7 : ∀ x : obj, Mixer x → WellPaid x → LivesInTaxHavens x
axiom P8 : Mixer Djokovic
axiom P9 : (Famous Djokovic ∧ Athlete Djokovic) → WellPaid Djokovic

-- Some derived consequences (do not establish that Djokovic is a Grand Slam champion)
theorem djokovic_gs_or_ona : GrandSlamChampion Djokovic ∨ OscarNominatedActor Djokovic := by
  exact P1 Djokovic P8

theorem djokovic_gs_implies_athlete : GrandSlamChampion Djokovic → Athlete Djokovic := by
  intro hgs
  have hpro : ProTennisPlayer Djokovic := P2 Djokovic P8 hgs
  exact P4 Djokovic P8 hpro

theorem djokovic_athlete_implies_famous : Athlete Djokovic → Famous Djokovic := by
  intro ha
  exact P6 Djokovic P8 ha

theorem djokovic_gs_implies_wellpaid : GrandSlamChampion Djokovic → WellPaid Djokovic := by
  intro hgs
  have ha : Athlete Djokovic := djokovic_gs_implies_athlete hgs
  have hf : Famous Djokovic := djokovic_athlete_implies_famous ha
  exact P9 ⟨hf, ha⟩

theorem djokovic_ona_implies_wellpaid : OscarNominatedActor Djokovic → WellPaid Djokovic := by
  intro hona
  have hcel : Celebrity Djokovic := P3 Djokovic P8 hona
  exact P5 Djokovic P8 hcel
