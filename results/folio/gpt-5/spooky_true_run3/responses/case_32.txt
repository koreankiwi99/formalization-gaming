<lean>
axiom obj : Type

-- Individuals
axiom Djokovic : obj

-- Predicates
axiom At : obj → Prop
axiom GrandSlamChampion : obj → Prop
axiom OscarNominatedActor : obj → Prop
axiom ProfessionalTennisPlayer : obj → Prop
axiom Celebrity : obj → Prop
axiom Athlete : obj → Prop
axiom Famous : obj → Prop
axiom WellPaid : obj → Prop
axiom LivesInTaxHaven : obj → Prop

-- Premises
-- Everyone at the mixer is a Grand Slam champion or an Oscar-nominated actor.
axiom P1 : ∀ x : obj, At x → (GrandSlamChampion x ∨ OscarNominatedActor x)
-- Every Grand Slam champion at the mixer is a professional tennis player.
axiom P2 : ∀ x : obj, At x → GrandSlamChampion x → ProfessionalTennisPlayer x
-- All Oscar-nominated actors at the mixer are celebrities.
axiom P3 : ∀ x : obj, At x → OscarNominatedActor x → Celebrity x
-- All professional tennis players at the mixer are athletes.
axiom P4 : ∀ x : obj, At x → ProfessionalTennisPlayer x → Athlete x
-- If a person at the mixer is a celebrity, then they are well paid.
axiom P5 : ∀ x : obj, At x → Celebrity x → WellPaid x
-- If a person at the mixer is an athlete, then they are famous.
axiom P6 : ∀ x : obj, At x → Athlete x → Famous x
-- All well-paid people at the mixer live in tax havens.
axiom P7 : ∀ x : obj, At x → WellPaid x → LivesInTaxHaven x

-- Djokovic is at the mixer
axiom P8 : At Djokovic
-- Djokovic-specific: if Djokovic is a famous athlete, then Djokovic is well-paid.
axiom P9 : Famous Djokovic → Athlete Djokovic → WellPaid Djokovic

-- Derivation: Djokovic lives in a tax haven.
theorem djokovic_tax_haven : LivesInTaxHaven Djokovic := by
  have hdisj : GrandSlamChampion Djokovic ∨ OscarNominatedActor Djokovic := P1 Djokovic P8
  cases hdisj with
  | inl hGS =>
      have hPro : ProfessionalTennisPlayer Djokovic := P2 Djokovic P8 hGS
      have hAth : Athlete Djokovic := P4 Djokovic P8 hPro
      have hFam : Famous Djokovic := P6 Djokovic P8 hAth
      have hWP : WellPaid Djokovic := P9 hFam hAth
      exact P7 Djokovic P8 hWP
  | inr hON =>
      have hCel : Celebrity Djokovic := P3 Djokovic P8 hON
      have hWP : WellPaid Djokovic := P5 Djokovic P8 hCel
      exact P7 Djokovic P8 hWP
