axiom Show : Type
axiom StrangerThings : Show
axiom BlackMirror : Show

axiom Netflix : Show → Prop
axiom Popular : Show → Prop
axiom BingeK : Show → Prop        -- Karen binge-watches the show
axiom DownloadK : Show → Prop     -- Karen downloads the show
axiom ShareWithLisa : Show → Prop -- Karen shares the show with Lisa

-- "Stranger Things" is a popular Netflix show.
axiom ST_netflix : Netflix StrangerThings
axiom ST_popular : Popular StrangerThings

-- If a Netflix show is popular, Karen will binge-watch it.
axiom R_pop_binge : ∀ s : Show, Netflix s → Popular s → BingeK s

-- If and only if Karen binge-watches a Netflix show, she will download it.
axiom R_binge_download_iff : ∀ s : Show, Netflix s → (BingeK s ↔ DownloadK s)

-- Karen does not download "Black Mirror."
axiom not_download_BM : ¬ DownloadK BlackMirror

-- "Black Mirror" is a Netflix show.
axiom BM_netflix : Netflix BlackMirror

-- If Karen binge-watches a Netflix show, she will share it with Lisa.
axiom R_binge_share : ∀ s : Show, Netflix s → BingeK s → ShareWithLisa s

-- From the iff and non-download, we can derive that Karen does not binge-watch Black Mirror.
theorem not_binge_BM : ¬ BingeK BlackMirror := by
  intro hb
  have hiff := R_binge_download_iff BlackMirror BM_netflix
  have hd : DownloadK BlackMirror := (Iff.mp hiff) hb
  exact not_download_BM hd
