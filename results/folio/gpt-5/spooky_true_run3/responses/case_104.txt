<lean>
axiom Person : Type
axiom Tom : Person

axiom GREApplicant : Person → Prop
axiom ProvedHardship : Person → Prop
axiom SingleParentFamily : Person → Prop
axiom FewResources : Person → Prop
axiom FinancialAidByETS : Person → Prop

-- ETS provides financial aid to those GRE applicants who prove economic hardship.
axiom AidRule : ∀ x : Person, GREApplicant x → ProvedHardship x → FinancialAidByETS x

-- Those living in single-parent families or having few resources can prove economic hardship.
axiom HardshipFromSingleParent : ∀ x : Person, SingleParentFamily x → ProvedHardship x
axiom HardshipFromFewResources : ∀ x : Person, FewResources x → ProvedHardship x

-- Facts about Tom
axiom TomSingleParent : SingleParentFamily Tom
axiom TomFewResources : FewResources Tom
axiom TomGREApplicant : GREApplicant Tom

theorem ETS_provides_financial_aid_to_Tom : FinancialAidByETS Tom := by
  apply AidRule Tom
  · exact TomGREApplicant
  · exact HardshipFromSingleParent Tom TomSingleParent
</lean>

ANSWER: True
