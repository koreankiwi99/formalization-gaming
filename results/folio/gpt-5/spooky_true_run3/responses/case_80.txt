<lean>
axiom obj : Type

axiom DriedThaiChilies : obj

axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom FromBBM : obj → Prop

-- Premises
axiom H1 : ∀ x : obj, BakedSweet x → ¬ Spicy x                    -- No baked sweets are spicy.
axiom H2 : ∀ x : obj, Cupcake x → BakedSweet x                    -- All cupcakes are baked sweets.
axiom H3 : ∀ x : obj, MalaHotpot x → Spicy x                      -- All mala hotpots are spicy.
axiom H4 : ∀ x : obj, FromBBM x → Cupcake x                       -- All products from BBM are cupcakes.
axiom H5 : Spicy DriedThaiChilies ∨ MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies
-- Dried Thai chilies are spicy or mala hotpots or not baked sweets.

-- Some derived consequences (do not yield the target conclusion).
theorem spicy_implies_not_baked : ∀ x : obj, Spicy x → ¬ BakedSweet x := by
  intro x hs
  -- From H1: BakedSweet x → ¬ Spicy x
  -- To show: ¬ BakedSweet x, i.e., BakedSweet x → False
  intro hb
  have hns : ¬ Spicy x := H1 x hb
  exact hns hs

theorem cupcakes_not_spicy : ∀ x : obj, Cupcake x → ¬ Spicy x := by
  intro x hc
  have hb : BakedSweet x := H2 x hc
  exact H1 x hb

theorem mala_not_baked : ∀ x : obj, MalaHotpot x → ¬ BakedSweet x := by
  intro x hm
  have hs : Spicy x := H3 x hm
  exact spicy_implies_not_baked x hs

-- We can simplify the given disjunction using spicy_implies_not_baked,
-- but still cannot conclude MalaHotpot DriedThaiChilies.
theorem reduced_disj :
  MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies := by
  have h := H5
  cases h with
  | inl hs =>
      have hnb : ¬ BakedSweet DriedThaiChilies := spicy_implies_not_baked DriedThaiChilies hs
      exact Or.inr hnb
  | inr h2 =>
      cases h2 with
      | inl hm => exact Or.inl hm
      | inr hnb => exact Or.inr hnb
