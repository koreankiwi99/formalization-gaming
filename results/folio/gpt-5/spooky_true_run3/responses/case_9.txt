<lean>
axiom obj : Type
axiom Tom : obj

axiom WildTurkey : obj → Prop
axiom EasternWildTurkey : obj → Prop
axiom OsceolaWildTurkey : obj → Prop
axiom GouldsWildTurkey : obj → Prop
axiom MerriamsWildTurkey : obj → Prop
axiom RioGrandeWildTurkey : obj → Prop
axiom OcellatedWildTurkey : obj → Prop

axiom six_types :
  ∀ x : obj,
    WildTurkey x →
    (EasternWildTurkey x ∨ OsceolaWildTurkey x ∨ GouldsWildTurkey x ∨
     MerriamsWildTurkey x ∨ RioGrandeWildTurkey x ∨ OcellatedWildTurkey x)

axiom Tom_not_Eastern : ¬ EasternWildTurkey Tom
axiom Tom_not_Osceola : ¬ OsceolaWildTurkey Tom
axiom Tom_not_Goulds : ¬ GouldsWildTurkey Tom
axiom Tom_not_Merriams : ¬ MerriamsWildTurkey Tom
axiom Tom_not_RioGrande : ¬ RioGrandeWildTurkey Tom
axiom Tom_is_WildTurkey : WildTurkey Tom

theorem tom_is_Ocellated : OcellatedWildTurkey Tom := by
  have h := six_types Tom Tom_is_WildTurkey
  cases h with
  | inl hE => exact False.elim (Tom_not_Eastern hE)
  | inr h1 =>
    cases h1 with
    | inl hO => exact False.elim (Tom_not_Osceola hO)
    | inr h2 =>
      cases h2 with
      | inl hG => exact False.elim (Tom_not_Goulds hG)
      | inr h3 =>
        cases h3 with
        | inl hM => exact False.elim (Tom_not_Merriams hM)
        | inr h4 =>
          cases h4 with
          | inl hR => exact False.elim (Tom_not_RioGrande hR)
          | inr hOc => exact hOc
</lean>

ANSWER: True
