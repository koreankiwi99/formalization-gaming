<lean>
axiom obj : Type

axiom Rabbit : obj → Prop
axiom Squirrel : obj → Prop
axiom Turtle : obj → Prop
axiom Spotted : obj → Prop
axiom Cute : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop

axiom Rockie : obj

-- All rabbits that can be spotted near the campus are cute.
axiom A1 : ∀ x : obj, Rabbit x ∧ Spotted x → Cute x
-- Some turtles can be spotted near the campus.
axiom A2 : ∃ x : obj, Turtle x ∧ Spotted x
-- The only animals that can be spotted near the campus are rabbits and squirrels.
axiom A3 : ∀ x : obj, Spotted x → (Rabbit x ∨ Squirrel x)
-- If something is skittish, then it is not calm.
axiom A4 : ∀ x : obj, Skittish x → ¬ Calm x
-- All the squirrels that can be spotted near the campus are skittish.
axiom A5 : ∀ x : obj, Squirrel x ∧ Spotted x → Skittish x
-- Rockie can be spotted near the campus, and it is calm.
axiom A6 : Spotted Rockie ∧ Calm Rockie

-- From calmness and spottedness, Rockie cannot be a squirrel.
theorem not_squirrel_Rockie : ¬ Squirrel Rockie := by
  intro hs
  have hsp : Spotted Rockie := And.left A6
  have hsk : Skittish Rockie := A5 Rockie (And.intro hs hsp)
  have hnc : ¬ Calm Rockie := A4 Rockie hsk
  exact hnc (And.right A6)

-- From "only spotted are rabbits or squirrels", and not being a squirrel, Rockie is a rabbit.
theorem rabbit_Rockie : Rabbit Rockie := by
  have hsp : Spotted Rockie := And.left A6
  have hor : Rabbit Rockie ∨ Squirrel Rockie := A3 Rockie hsp
  apply Or.elim hor
  · intro hr; exact hr
  · intro hs
    have : False := not_squirrel_Rockie hs
    exact False.elim this

-- Therefore Rockie is cute (but this does not help with being/not being a turtle).
theorem cute_Rockie : Cute Rockie := by
  have hsp : Spotted Rockie := And.left A6
  have hr : Rabbit Rockie := rabbit_Rockie
  exact A1 Rockie (And.intro hr hsp)
