<lean>
axiom obj : Type

axiom Mixture : obj

axiom Hydrocarbon : obj → Prop
axiom Organic : obj → Prop
axiom Chemical : obj → Prop
axiom ContainsCarbon : obj → Prop
axiom OnlyOneElement : obj → Prop
axiom Alkane : obj → Prop

axiom P1 : ∀ x : obj, Hydrocarbon x → Organic x
axiom P2 : ∀ x : obj, Alkane x → Hydrocarbon x
axiom P3 : ∀ x : obj, Organic x → Chemical x
axiom P4 : ∀ x : obj, Organic x → ContainsCarbon x
axiom P5 : ∀ x : obj, Chemical x → ¬ OnlyOneElement x
axiom P6 : (Chemical Mixture ∧ OnlyOneElement Mixture) ∨ (¬ Chemical Mixture ∧ ¬ OnlyOneElement Mixture)

theorem mixture_result :
  (OnlyOneElement Mixture ∨ ContainsCarbon Mixture) →
  (¬ Chemical Mixture ∧ ¬ Alkane Mixture) := by
  have h5 := P5 Mixture
  have hnot : ¬ Chemical Mixture ∧ ¬ OnlyOneElement Mixture := by
    cases P6 with
    | inl h =>
        have hc : Chemical Mixture := h.left
        have hone : OnlyOneElement Mixture := h.right
        have hno : ¬ OnlyOneElement Mixture := h5 hc
        have contra : False := hno hone
        exact And.intro (by intro _; exact False.elim contra) (by intro _; exact False.elim contra)
    | inr h =>
        exact h
  have hnotC : ¬ Chemical Mixture := hnot.left
  have hnotA : ¬ Alkane Mixture := by
    intro hA
    have hH : Hydrocarbon Mixture := P2 Mixture hA
    have hO : Organic Mixture := P1 Mixture hH
    have hC : Chemical Mixture := P3 Mixture hO
    exact hnotC hC
  intro _
  exact And.intro hnotC hnotA
</lean>

ANSWER: True
