<lean>
axiom obj : Type
axiom Listed : obj → Prop
axiom ManyNeg : obj → Prop
axiom RatingGt4 : obj → Prop
axiom TakeOut : obj → Prop
axiom Popular : obj → Prop
axiom Hamden : obj

axiom P1 : ∀ x : obj, Listed x → ¬ ManyNeg x
axiom P2 : ∀ x : obj, RatingGt4 x → Listed x
axiom P3 : ∃ x : obj, ¬ TakeOut x ∧ ManyNeg x
axiom P4 : ∀ x : obj, Popular x → RatingGt4 x
axiom P5 : RatingGt4 Hamden ∨ Popular Hamden

theorem hamden_statement :
  (TakeOut Hamden ∧ ManyNeg Hamden) → (RatingGt4 Hamden ∧ ¬ TakeOut Hamden) := by
  intro h
  have hMN : ManyNeg Hamden := And.right h
  have hRG4 : RatingGt4 Hamden :=
    Or.elim P5 (fun hr => hr) (fun hp => P4 Hamden hp)
  have hListed : Listed Hamden := P2 Hamden hRG4
  have hNotMN : ¬ ManyNeg Hamden := P1 Hamden hListed
  have f : False := hNotMN hMN
  have notTO : ¬ TakeOut Hamden := by
    intro _
    exact False.elim f
  exact And.intro hRG4 notTO
</lean>

ANSWER: True
