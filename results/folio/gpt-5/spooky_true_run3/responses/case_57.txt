axiom State : Type

-- Same state relation
def SameState (c d : City) : Prop := ∃ s : State, InState c s ∧ InState d s

-- "White Sulphur Springs and Butte are in the same state"
axiom WSS_same_as_Butte : SameState WhiteSulphurSprings Butte

-- "Any city in Butte is not in St Pierre" (interpretation: sharing Butte's state excludes sharing St Pierre's state)
axiom in_Butte_not_in_StPierre :
  ∀ c : City, SameState c Butte → ¬ SameState c StPierre

-- Uniqueness of state for cities except the four multi-state exceptions
axiom unique_state_except :
  ∀ c : City,
    c ≠ Bristol ∧ c ≠ Texarkana ∧ c ≠ Texhoma ∧ c ≠ UnionCity →
    ∀ s t : State, InState c s → InState c t → s = t

-- Butte is not one of the exceptions
axiom Butte_ne_Bristol : Butte ≠ Bristol
axiom Butte_ne_Texarkana : Butte ≠ Texarkana
axiom Butte_ne_Texhoma : Butte ≠ Texhoma
axiom Butte_ne_UnionCity : Butte ≠ UnionCity

-- From the facts, we can show Butte and St Pierre are not in the same state
theorem not_same_state_Butte_StPierre : ¬ SameState Butte StPierre :=
by
  intro h
  obtain ⟨s, hButte_s, hStPierre_s⟩ := h
  have hButte_M : InState Butte Montana := Butte_in_Montana
  have hneq :
    Butte ≠ Bristol ∧ Butte ≠ Texarkana ∧ Butte ≠ Texhoma ∧ Butte ≠ UnionCity :=
    And.intro Butte_ne_Bristol (And.intro Butte_ne_Texarkana (And.intro Butte_ne_Texhoma Butte_ne_UnionCity))
  have hs_eq : s = Montana := unique_state_except Butte hneq s Montana hButte_s hButte_M
  have hStPierre_M : InState StPierre Montana := by
    simpa [hs_eq] using hStPierre_s
  exact StPierre_not_in_Montana hStPierre_M
