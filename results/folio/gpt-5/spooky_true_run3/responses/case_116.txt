-- If a man is taller than another man, the taller one can block the other's shooting.
axiom R1 : ∀ x y : obj, Man x → Man y → Taller x y → CanBlock x y

-- Michael is a man who is taller than everyone else in his class.
axiom R2a : Man Michael
axiom R2b : ∀ y : obj, InClass y Michael ∧ y ≠ Michael → Taller Michael y

-- If x taller than y and y taller than z, then x taller than z.
axiom R3 : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom R4 : Man Peter ∧ Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom R5 : ∀ p : obj, ¬ Jump p → CanBlock Michael p

-- Michael cannot block Windy's shooting.
axiom R6 : ¬ CanBlock Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom R7 : ∀ s : obj, Jump s → Great s

-- Some useful derived facts
theorem peter_man : Man Peter := And.left R4
theorem peter_taller_michael : Taller Peter Michael := And.right R4

-- From R5 and R6 we can derive that Windy jumps (classically).
theorem windy_jumps : Jump Windy :=
by
  classical
  by_contra h
  have : CanBlock Michael Windy := R5 Windy h
  exact R6 this

theorem windy_great : Great Windy := R7 Windy windy_jumps
