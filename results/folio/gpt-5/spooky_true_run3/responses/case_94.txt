<lean>
axiom Building : Type
axiom Managed : Building → Prop
axiom OliveGarden : Building

axiom Rent : Building → Nat
axiom Deposit : Building → Nat

-- The security deposit at a managed building is either equal to its monthly rent or greater.
axiom Deposit_rule : ∀ b : Building, Managed b → (Deposit b = Rent b ∨ Deposit b > Rent b)

-- Facts about the Olive Garden
axiom OG_managed : Managed OliveGarden
axiom Rent_OG : Rent OliveGarden = 2000

-- Irrelevant numeric fact included in the context
axiom fact_gt : 2000 > 1500

theorem deposit_og_2000_or_more : Deposit OliveGarden = 2000 ∨ Deposit OliveGarden > 2000 :=
by
  have h := Deposit_rule OliveGarden OG_managed
  cases h with
  | inl hEq =>
      have : Deposit OliveGarden = 2000 := by
        simpa [Rent_OG] using hEq
      exact Or.inl this
  | inr hGt =>
      have : Deposit OliveGarden > 2000 := by
        simpa [Rent_OG] using hGt
      exact Or.inr this
</lean>

ANSWER: True
