axiom obj : Type

-- Predicates
axiom Animal : obj → Prop
axiom HeckCattle : obj → Prop
axiom BredBack : obj → Prop
axiom ArtificiallySelected : obj → Prop
axiom ToBeBredBack : obj → Prop
axiom Extinct : obj → Prop
axiom Resemble : obj → obj → Prop

-- A specific object representing "the aurochs"
axiom AurochsObj : obj

-- Premises
-- Breeding back is a form of artificial selection
axiom BB_is_AS : ∀ x : obj, BredBack x → ArtificiallySelected x

-- Heck cattle were bred back in the 1920s to resemble the aurochs
axiom Heck_bred_back_resemble_aurochs : ∃ x : obj, HeckCattle x ∧ BredBack x ∧ Resemble x AurochsObj

-- Heck cattle are animals
axiom Heck_are_animals : ∀ x : obj, HeckCattle x → Animal x

-- Aurochs are animals
axiom Aurochs_animal : Animal AurochsObj

-- Some animals to be bred back resemble extinct animals
axiom Some_animals_to_be_bred_back_resemble_extinct :
  ∃ x : obj, Animal x ∧ ToBeBredBack x ∧ ∃ y : obj, Extinct y ∧ Resemble x y

-- Goal: Some Heck cattle are artificially selected.
theorem some_heck_cattle_artificially_selected : ∃ x : obj, HeckCattle x ∧ ArtificiallySelected x :=
by
  apply Exists.elim Heck_bred_back_resemble_aurochs
  intro x hx
  apply And.elim hx
  intro hxHeck hrest
  apply And.elim hrest
  intro hxBred hxRes
  have hAS : ArtificiallySelected x := BB_is_AS x hxBred
  exact Exists.intro x (And.intro hxHeck hAS)
</lean>
