<lean>
axiom Show : Type
axiom StrangerThings : Show
axiom BlackMirror : Show

axiom Netflix : Show → Prop
axiom Popular : Show → Prop
axiom Binge : Show → Prop        -- Karen binge-watches the show
axiom Download : Show → Prop     -- Karen downloads the show
axiom ShareWithLisa : Show → Prop

-- "Stranger Things" is a popular Netflix show.
axiom ST_is_Netflix : Netflix StrangerThings
axiom ST_is_Popular : Popular StrangerThings

-- If a Netflix show is popular, Karen will binge-watch it.
axiom Rule1 : ∀ s : Show, Netflix s → Popular s → Binge s

-- If and only if Karen binge-watches a Netflix show, she will download it.
axiom Rule2 : ∀ s : Show, Netflix s → (Binge s ↔ Download s)

-- Karen does not download "Black Mirror."
axiom NotDownload_BM : ¬ Download BlackMirror

-- "Black Mirror" is a Netflix show.
axiom BM_is_Netflix : Netflix BlackMirror

-- If Karen binge-watches a Netflix show, she will share it with Lisa.
axiom Rule3 : ∀ s : Show, Netflix s → Binge s → ShareWithLisa s

-- Statement to evaluate: Karen will share "Stranger Things" with Lisa.
theorem share_ST_with_Lisa : ShareWithLisa StrangerThings := by
  have hBinge : Binge StrangerThings :=
    Rule1 StrangerThings ST_is_Netflix ST_is_Popular
  exact Rule3 StrangerThings ST_is_Netflix hBinge
</lean>

ANSWER: True
