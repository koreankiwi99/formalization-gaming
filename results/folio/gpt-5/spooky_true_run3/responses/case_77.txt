<lean>
axiom obj : Type
axiom Hannah : obj

axiom AMS : obj → Prop
axiom Work : obj → Prop
axiom Need : obj → Prop
axiom Order : obj → Prop
axiom Enjoy : obj → Prop
axiom Picky : obj → Prop
axiom Spend : obj → Prop

axiom R1 : ∀ x : obj, AMS x ∧ Work x → Need x
axiom R2 : ∀ x : obj, AMS x ∧ Order x → Work x
axiom R3 : ∀ x : obj, AMS x → (Order x ∨ Enjoy x)
axiom R4 : ∀ x : obj, AMS x ∧ Enjoy x → ¬ Picky x
axiom R5 : ∀ x : obj, AMS x ∧ Enjoy x → Spend x

axiom T6 : AMS Hannah
axiom T7a : Work Hannah
axiom T7b : Need Hannah → (¬ Picky Hannah ∧ ¬ Need Hannah)

theorem target : AMS Hannah ∧ (¬ Picky Hannah ∨ (Picky Hannah → Spend Hannah)) :=
by
  have hams : AMS Hannah := T6
  have disj : Order Hannah ∨ Enjoy Hannah := R3 Hannah hams
  have notPicky : ¬ Picky Hannah :=
    Or.elim disj
      (fun horder =>
        let hwork : Work Hannah := R2 Hannah ⟨hams, horder⟩
        let hneed : Need Hannah := R1 Hannah ⟨hams, hwork⟩
        (T7b hneed).left
      )
      (fun henjoy =>
        R4 Hannah ⟨hams, henjoy⟩
      )
  exact And.intro hams (Or.inl notPicky)
</lean>

ANSWER: True
