axiom obj : Type

axiom Hydrocarbon : obj → Prop
axiom Organic : obj → Prop
axiom Chemical : obj → Prop
axiom ContainsCarbon : obj → Prop
axiom Alkane : obj → Prop
axiom OneElement : obj → Prop

axiom Mixture : obj

-- Premises
axiom P1 : ∀ x : obj, Hydrocarbon x → Organic x
axiom P2 : ∀ x : obj, Alkane x → Hydrocarbon x
axiom P3 : ∀ x : obj, Organic x → Chemical x
axiom P4 : ∀ x : obj, Organic x → ContainsCarbon x
axiom P5 : ∀ x : obj, Chemical x → ¬ OneElement x
axiom P6 : (Chemical Mixture ∧ OneElement Mixture) ∨ (¬ Chemical Mixture ∧ ¬ OneElement Mixture)

-- Derived consequences from the “either both or neither” constraint and the incompatibility
-- of Chemical with OneElement.
theorem mixture_not_chemical : ¬ Chemical Mixture := by
  cases P6 with
  | inl hboth =>
    have hc : Chemical Mixture := hboth.left
    have ho : OneElement Mixture := hboth.right
    have hno : ¬ OneElement Mixture := P5 Mixture hc
    exact False.elim (hno ho)
  | inr hneither =>
    exact hneither.left

theorem mixture_not_oneelement : ¬ OneElement Mixture := by
  cases P6 with
  | inl hboth =>
    have hc : Chemical Mixture := hboth.left
    have ho : OneElement Mixture := hboth.right
    have hno : ¬ OneElement Mixture := P5 Mixture hc
    exact False.elim (hno ho)
  | inr hneither =>
    exact hneither.right

theorem mixture_not_organic : ¬ Organic Mixture := by
  intro horg
  have hc : Chemical Mixture := P3 Mixture horg
  exact mixture_not_chemical hc
