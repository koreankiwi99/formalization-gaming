<lean>
axiom Person : Type
axiom Taylor : Person

axiom EatsSaladsRegularly : Person → Prop
axiom VeryConscious : Person → Prop
axiom GrewUpHealthConsciousHome : Person → Prop
axiom FulfillDailyNutritionalIntakes : Person → Prop
axiom DisregardPhysicalWellbeing : Person → Prop
axiom VisitGymDaily : Person → Prop

axiom A1 : ∀ x : Person, EatsSaladsRegularly x → VeryConscious x
axiom A2 : ∀ x : Person, GrewUpHealthConsciousHome x → EatsSaladsRegularly x
axiom A3 : ∀ x : Person, FulfillDailyNutritionalIntakes x → GrewUpHealthConsciousHome x
axiom A4 : ∀ x : Person, DisregardPhysicalWellbeing x → ¬ VeryConscious x
axiom A5 : ∀ x : Person, VisitGymDaily x → FulfillDailyNutritionalIntakes x

axiom AT :
  (GrewUpHealthConsciousHome Taylor ∧ DisregardPhysicalWellbeing Taylor) ∨
  (¬ GrewUpHealthConsciousHome Taylor ∧ ¬ DisregardPhysicalWellbeing Taylor)

theorem taylor_neither_grewHC_nor_gym :
  ¬ GrewUpHealthConsciousHome Taylor ∧ ¬ VisitGymDaily Taylor := by
  -- Show the left disjunct is impossible
  have h_not_left : ¬ (GrewUpHealthConsciousHome Taylor ∧ DisregardPhysicalWellbeing Taylor) := by
    intro h
    have hG : GrewUpHealthConsciousHome Taylor := h.left
    have hD : DisregardPhysicalWellbeing Taylor := h.right
    have hVC : VeryConscious Taylor := A1 Taylor (A2 Taylor hG)
    have hNotVC : ¬ VeryConscious Taylor := A4 Taylor hD
    exact hNotVC hVC
  -- From the given disjunction, deduce the right conjunct
  have h_right : ¬ GrewUpHealthConsciousHome Taylor ∧ ¬ DisregardPhysicalWellbeing Taylor := by
    cases AT with
    | inl hleft =>
      exact False.elim (h_not_left hleft)
    | inr hright =>
      exact hright
  have h_notG : ¬ GrewUpHealthConsciousHome Taylor := h_right.left
  -- Show Taylor does not visit the gym daily
  have h_notGym : ¬ VisitGymDaily Taylor := by
    intro hGym
    have hFul : FulfillDailyNutritionalIntakes Taylor := A5 Taylor hGym
    have hG : GrewUpHealthConsciousHome Taylor := A3 Taylor hFul
    exact h_notG hG
  exact And.intro h_notG h_notGym
</lean>

ANSWER: True
