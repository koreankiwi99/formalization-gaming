axiom obj : Type
axiom NineteenEightyFour : obj

axiom Digital : obj → Prop
axiom Analog : obj → Prop
axiom PrintedText : obj → Prop
axiom StreamingService : obj → Prop
axiom HardcoverBook : obj → Prop

-- No digital media are analog.
axiom A1 : ∀ x : obj, Digital x → ¬ Analog x
-- Every printed text is analog media.
axiom A2 : ∀ x : obj, PrintedText x → Analog x
-- All streaming services are digital media.
axiom A3 : ∀ x : obj, StreamingService x → Digital x
-- If an object is a hardcover book, then it is printed text.
axiom A4 : ∀ x : obj, HardcoverBook x → PrintedText x
-- If 1984 is a streaming service, then 1984 is a hardcover book.
axiom A5 : StreamingService NineteenEightyFour → HardcoverBook NineteenEightyFour

-- From the premises we can derive that 1984 is not a streaming service.
theorem not_streaming_1984 : ¬ StreamingService NineteenEightyFour := by
  intro hs
  have d : Digital NineteenEightyFour := A3 NineteenEightyFour hs
  have hc : HardcoverBook NineteenEightyFour := A5 hs
  have p : PrintedText NineteenEightyFour := A4 NineteenEightyFour hc
  have a : Analog NineteenEightyFour := A2 NineteenEightyFour p
  have nd : ¬ Analog NineteenEightyFour := A1 NineteenEightyFour d
  exact nd a

-- We can also see that if 1984 were a streaming service, it would be printed text,
-- but that leads to a contradiction with A1 via A2 and A3, hence the negation above.
theorem streaming_implies_printed_1984 : StreamingService NineteenEightyFour → PrintedText NineteenEightyFour := by
  intro hs
  have hc : HardcoverBook NineteenEightyFour := A5 hs
  exact A4 NineteenEightyFour hc
