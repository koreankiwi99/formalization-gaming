  Manager James → ¬ (LunchHome James ∨ Remote James) := by
  intro m
  have hnr : ¬ Remote James := P6 James m
  intro h
  cases h with
  | inl lh =>
      have r : Remote James := P4 James EJ lh
      exact hnr r
  | inr r =>
      exact hnr r
