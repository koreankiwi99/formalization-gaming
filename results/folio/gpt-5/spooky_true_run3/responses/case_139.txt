<lean>
axiom obj : Type

axiom Film : obj → Prop
axiom RatedGA : obj → Prop
axiom AppropriateAllAges : obj → Prop
axiom ChildrenCanWatchWithoutGuidance : obj → Prop
axiom ContainsExcessiveEroticViolentContent : obj → Prop
axiom FamilyFriendlyAnimated : obj → Prop
axiom InFrozenSeries : obj → Prop

-- If a film is appropriate for people of all ages, then children can watch it without guidance from parents.
axiom A1 : ∀ x : obj, AppropriateAllAges x → ChildrenCanWatchWithoutGuidance x
-- If a film contains excessive erotic and violent content, children cannot watch it without guidance from their parents.
axiom A2 : ∀ x : obj, ContainsExcessiveEroticViolentContent x → ¬ ChildrenCanWatchWithoutGuidance x
-- If a film is rated General Audience, then it is appropriate for people of all ages.
axiom A3 : ∀ x : obj, RatedGA x → AppropriateAllAges x
-- All family-friendly animated films are rated General Audience.
axiom A4 : ∀ x : obj, FamilyFriendlyAnimated x → RatedGA x
-- All movies in the Frozen series are family-friendly animated films.
axiom A5 : ∀ x : obj, InFrozenSeries x → FamilyFriendlyAnimated x

-- “Hachi: A dog’s Tale” is a film.
axiom Hachi : obj
axiom Hachi_is_film : Film Hachi
-- “Hachi: A dog’s Tale”  is either a family-friendly animated film or is appropriate for people of all ages.
axiom Hachi_ff_or_appropriate : FamilyFriendlyAnimated Hachi ∨ AppropriateAllAges Hachi

-- We can prove that children can watch Hachi without guidance (but not that it is rated GA).
theorem hachi_children_can_watch : ChildrenCanWatchWithoutGuidance Hachi :=
  Or.elim Hachi_ff_or_appropriate
    (fun hff => A1 Hachi (A3 Hachi (A4 Hachi hff)))
    (fun happ => A1 Hachi happ)
