<lean>
axiom obj : Type
axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom FromBBM : obj → Prop
axiom DTC : obj

axiom A1 : ∀ x : obj, BakedSweet x → ¬ Spicy x
axiom A2 : ∀ x : obj, Cupcake x → BakedSweet x
axiom A3 : ∀ x : obj, MalaHotpot x → Spicy x
axiom A4 : ∀ x : obj, FromBBM x → Cupcake x
axiom A5 : Spicy DTC ∨ MalaHotpot DTC ∨ ¬ BakedSweet DTC

theorem dtc_not_bbm : ¬ FromBBM DTC := by
  intro hbbm
  have hc : Cupcake DTC := A4 DTC hbbm
  have hb : BakedSweet DTC := A2 DTC hc
  have hns : ¬ Spicy DTC := A1 DTC hb
  have hnmh : ¬ MalaHotpot DTC := by
    intro hm
    have hs : Spicy DTC := A3 DTC hm
    exact hns hs
  have contra : False := by
    cases A5 with
    | inl hs => exact hns hs
    | inr hr =>
      cases hr with
      | inl hm => exact hnmh hm
      | inr nbs => exact nbs hb
  exact contra
</lean>

ANSWER: True
