axiom obj : Type

-- Predicates
axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom FromBBM : obj → Prop

-- Constant for the specific item
axiom DriedThaiChilies : obj

-- Premises
axiom H1 : ∀ x : obj, BakedSweet x → ¬ Spicy x               -- No baked sweets are spicy
axiom H2 : ∀ x : obj, Cupcake x → BakedSweet x               -- All cupcakes are baked sweets
axiom H3 : ∀ x : obj, MalaHotpot x → Spicy x                 -- All mala hotpots are spicy
axiom H4 : ∀ x : obj, FromBBM x → Cupcake x                  -- All products from Baked by Melissa are cupcakes
axiom H5 : Spicy DriedThaiChilies ∨ MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies
  -- Dried Thai chilies are spicy or mala hotpots or not baked sweets

-- From the disjunction and the other premises, we can derive that Dried Thai chilies are not cupcakes.
theorem not_cupcake_DTC : ¬ Cupcake DriedThaiChilies := by
  intro hCup
  -- Case analysis on the given disjunction
  cases H5 with
  | inl hSpicy =>
      -- If spicy, cupcakes (hence baked sweets) cannot be spicy by H1 and H2
      have hNotSpicy : ¬ Spicy DriedThaiChilies := H1 DriedThaiChilies (H2 DriedThaiChilies hCup)
      exact hNotSpicy hSpicy
  | inr h =>
      cases h with
      | inl hMala =>
          -- If mala hotpot, then spicy; same contradiction as above
          have hSpicy : Spicy DriedThaiChilies := H3 DriedThaiChilies hMala
          have hNotSpicy : ¬ Spicy DriedThaiChilies := H1 DriedThaiChilies (H2 DriedThaiChilies hCup)
          exact hNotSpicy hSpicy
      | inr hNotSweet =>
          -- If not baked sweet, cupcakes (which are baked sweets) are impossible
          have hSweet : BakedSweet DriedThaiChilies := H2 DriedThaiChilies hCup
          exact hNotSweet hSweet

-- Therefore, the claimed conjunction cannot hold.
theorem not_claimed_conclusion : ¬ (Cupcake DriedThaiChilies ∧ FromBBM DriedThaiChilies) := by
  intro h
  exact not_cupcake_DTC h.left
