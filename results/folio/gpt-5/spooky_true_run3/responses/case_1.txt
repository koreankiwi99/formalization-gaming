<lean>
axiom obj : Type

axiom Club : obj → Prop
axiom PerformOften : obj → Prop
axiom AttendEvents : obj → Prop
axiom VeryEngaged : obj → Prop
axiom InactiveDisinterested : obj → Prop
axiom ChaperoneDances : obj → Prop
axiom StudentAtSchool : obj → Prop
axiom YoungChild : obj → Prop
axiom Teenager : obj → Prop
axiom WishesFurtherAcademics : obj → Prop

axiom Bonnie : obj

-- People in this club who perform in school talent shows often attend and are very engaged with school events.
axiom P1 : ∀ x : obj, Club x ∧ PerformOften x → AttendEvents x ∧ VeryEngaged x
-- People in this club either perform in school talent shows often or are inactive and disinterested community members.
axiom P2 : ∀ x : obj, Club x → PerformOften x ∨ InactiveDisinterested x
-- People in this club who chaperone high school dances are not students who attend the school.
axiom P3 : ∀ x : obj, Club x ∧ ChaperoneDances x → ¬ StudentAtSchool x
-- All people in this club who are inactive and disinterested members of their community chaperone high school dances.
axiom P4 : ∀ x : obj, Club x ∧ InactiveDisinterested x → ChaperoneDances x
-- All young children and teenagers in this club who wish to further their academic careers and educational opportunities are students who attend the school.
axiom P5 : ∀ x : obj, Club x ∧ ((YoungChild x ∨ Teenager x) ∧ WishesFurtherAcademics x) → StudentAtSchool x

-- Bonnie is in this club
axiom Bclub : Club Bonnie
-- Bonnie either ((attends ∧ is very engaged) ∧ is a student) or (¬(attends ∧ is very engaged) ∧ ¬student)
axiom Binfo :
  ((AttendEvents Bonnie ∧ VeryEngaged Bonnie) ∧ StudentAtSchool Bonnie) ∨
  (¬(AttendEvents Bonnie ∧ VeryEngaged Bonnie) ∧ ¬ StudentAtSchool Bonnie)

theorem statement :
  ( (((YoungChild Bonnie ∨ Teenager Bonnie) ∧ WishesFurtherAcademics Bonnie) ∧ ChaperoneDances Bonnie)
    ∨ ¬ ((YoungChild Bonnie ∨ Teenager Bonnie) ∧ WishesFurtherAcademics Bonnie) )
  → (StudentAtSchool Bonnie ∨ InactiveDisinterested Bonnie) := by
  intro _
  -- We show the conclusion unconditionally from club membership and Binfo.
  have h := P2 Bonnie Bclub
  cases h with
  | inl hperf =>
      have hAE : AttendEvents Bonnie ∧ VeryEngaged Bonnie :=
        P1 Bonnie (And.intro Bclub hperf)
      have hStu : StudentAtSchool Bonnie := by
        cases Binfo with
        | inl hgood =>
            -- ((Attend ∧ Engaged) ∧ Student)
            exact hgood.right
        | inr hbad =>
            -- ¬(Attend ∧ Engaged) contradicts hAE
            have : False := hbad.left hAE
            exact False.elim this
      exact Or.inl hStu
  | inr hinact =>
      exact Or.inr hinact
</lean>

ANSWER: True
