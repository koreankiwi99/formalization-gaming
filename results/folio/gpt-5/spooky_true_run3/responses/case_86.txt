<lean>
axiom obj : Type

-- Objects
axiom Movie1 : obj
axiom SirDigby : obj

-- Functions and Predicates
axiom Nemesis : obj → obj
axiom Fights : obj → obj → Prop
axiom Wins : obj → Prop
axiom SuperheroMovie : obj → Prop
axiom NamedAfter : obj → obj → Prop
axiom Good : obj → Prop
axiom In : obj → obj → Prop

-- Premises
-- In superhero movies, the good guys always win (if they are in the movie).
axiom P1 : ∀ m : obj, SuperheroMovie m → ∀ x : obj, In x m → Good x → Wins x
-- The Surprising Adventures of Sir Digby Chicken Caesar is a superhero movie.
axiom P2 : SuperheroMovie Movie1
-- Sir Digby fights his nemesis.
axiom P4 : Fights SirDigby (Nemesis SirDigby)
-- If a superhero movie is named after a character, that character is a good guy.
axiom P5 : ∀ m x, SuperheroMovie m → NamedAfter m x → Good x
-- The Surprising Adventures of Sir Digby Chicken Caesar is named after Sir Digby.
axiom P6 : NamedAfter Movie1 SirDigby
-- If somebody wins a fight, the person they are fighting does not win.
axiom P7 : ∀ x y, Fights x y → Wins x → ¬ Wins y
-- If a superhero movie is named after a character, that character is in the movie.
axiom P8 : ∀ m x, SuperheroMovie m → NamedAfter m x → In x m

-- Conclusion: Sir Digby’s nemesis does not win.
theorem nemesis_does_not_win : ¬ Wins (Nemesis SirDigby) := by
  have hGood : Good SirDigby := P5 Movie1 SirDigby P2 P6
  have hIn : In SirDigby Movie1 := P8 Movie1 SirDigby P2 P6
  have hWinDigby : Wins SirDigby := P1 Movie1 P2 SirDigby hIn hGood
  have hFight : Fights SirDigby (Nemesis SirDigby) := P4
  exact P7 SirDigby (Nemesis SirDigby) hFight hWinDigby
</lean>
ANSWER: True
