<lean>
axiom obj : Type
axiom RobertLewandowski : obj
axiom BayernMunchen : obj
axiom Striker : obj → Prop
axiom SoccerPlayer : obj → Prop
axiom Left : obj → obj → Prop
axiom PlaysFor : obj → obj → Prop

axiom T1 : Striker RobertLewandowski
axiom R1 : ∀ x : obj, Striker x → SoccerPlayer x
axiom T2 : Left RobertLewandowski BayernMunchen
axiom R2 : ∀ x t : obj, Left x t → ¬ PlaysFor x t

theorem RL_is_soccer_player : SoccerPlayer RobertLewandowski := by
  exact R1 RobertLewandowski T1
</lean>

ANSWER: True
