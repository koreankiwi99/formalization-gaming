def InEnglish : Obj → Prop := fun _ => True
def FromItalian : Obj → Prop
| NC => True
| PF => False

-- All of New Vessel Press's published books are in English.

  exact True.intro

-- Neapolitan Chronicles was translated from Italian.

-- Palace of Flies is published by New Vessel Press.
theorem pf_published : PublishedByNVP PF := True.intro

-- In this model, Palace of Flies was NOT translated from Italian.
-- Hence the target conclusion does not follow from the premises.
theorem not_pf_from_italian : ¬ FromItalian PF := by
  intro h
  -- h : False, so contradiction is immediate
