<lean>
axiom obj : Type
axiom H : obj

axiom Listed : obj → Prop
axiom ManyNeg : obj → Prop
axiom RatingGt4 : obj → Prop
axiom TakeOut : obj → Prop
axiom Popular : obj → Prop

axiom A1 : ∀ x : obj, Listed x → ¬ ManyNeg x
axiom A2 : ∀ x : obj, RatingGt4 x → Listed x
axiom A3 : ∃ x : obj, ¬ TakeOut x ∧ ManyNeg x
axiom A4 : ∀ x : obj, Popular x → RatingGt4 x
axiom A5 : RatingGt4 H ∨ Popular H

theorem hamden_takeout_or_notneg : TakeOut H ∨ ¬ ManyNeg H := by
  have hRating : RatingGt4 H := by
    apply Or.elim A5
    · intro hRating; exact hRating
    · intro hPop; exact A4 H hPop
  have hListed : Listed H := A2 H hRating
  have hNotNeg : ¬ ManyNeg H := A1 H hListed
  exact Or.inr hNotNeg
</lean>

ANSWER: True
