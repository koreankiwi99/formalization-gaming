<lean>
axiom obj : Type
axiom GRE : obj
axiom Below300 : obj → Prop
axiom Cost205 : obj → Prop
axiom T_cost205 : Cost205 GRE
axiom R_cost205_implies_below300 : ∀ x : obj, Cost205 x → Below300 x

theorem gre_cost_below_300 : Below300 GRE := by
  apply R_cost205_implies_below300 GRE
  exact T_cost205
</lean>

ANSWER: True
