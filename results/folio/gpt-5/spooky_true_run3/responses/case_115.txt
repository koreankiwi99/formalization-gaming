axiom obj : Type

axiom Man : obj → Prop
axiom Taller : obj → obj → Prop
axiom CanBlock : obj → obj → Prop
axiom Shooter : obj → Prop
axiom JumpsWhenShooting : obj → Prop
axiom GreatShooter : obj → Prop

axiom Michael : obj
axiom Peter : obj
axiom Windy : obj

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom A1 : ∀ x y : obj, Man x → Man y → Taller x y → CanBlock x y
-- Michael is a man
axiom A2 : Man Michael
-- Transitivity of Taller
axiom A3 : ∀ x y z : obj, Taller x y → Taller y z → Taller x z
-- Peter is a man who is taller than Michael
axiom A4 : Man Peter ∧ Taller Peter Michael
-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom A5 : ∀ p : obj, ¬ JumpsWhenShooting p → CanBlock Michael p
-- Michael cannot block Windy's shooting, and Windy is a shooter.
axiom A6 : Shooter Windy ∧ ¬ CanBlock Michael Windy
-- Every shooter who can jump when shooting is a great shooter.
axiom A7 : ∀ x : obj, Shooter x → JumpsWhenShooting x → GreatShooter x

theorem Windy_great : GreatShooter Windy := by
  have hShooter : Shooter Windy := And.left A6
  have hNotBlock : ¬ CanBlock Michael Windy := And.right A6
  have hNNJ : ¬¬ JumpsWhenShooting Windy := by
    intro hNJ
    have : CanBlock Michael Windy := A5 Windy hNJ
    exact hNotBlock this
  classical
  have hJ : JumpsWhenShooting Windy := by
    by_contra h
    exact hNNJ h
  exact A7 Windy hShooter hJ
</lean>

ANSWER: True

  have hNotBlock : ¬ CanBlock Michael Windy := And.right A6
  have hNNJ : ¬¬ JumpsWhenShooting Windy :=

    match Classical.em (JumpsWhenShooting Windy) with
    | Or.inl h => h
