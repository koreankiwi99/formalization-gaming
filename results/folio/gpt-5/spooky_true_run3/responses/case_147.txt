<lean>
axiom obj : Type

axiom Ekaterina : obj
axiom Vladimir : obj

axiom CanRegisterUS : obj → Prop
axiom CanParticipateElection2024 : obj → Prop
axiom USCitizenship : obj → Prop
axiom TaiwaneseCitizenship : obj → Prop
axiom RussianFederationOfficial : obj → Prop
axiom ManagerAtGazprom : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, CanRegisterUS x → CanParticipateElection2024 x
axiom P2 : ∀ x : obj, USCitizenship x → CanRegisterUS x
axiom P3 : ∀ x : obj, USCitizenship x ∨ TaiwaneseCitizenship x
axiom P4 : ∀ x : obj, RussianFederationOfficial x → ¬ TaiwaneseCitizenship x
axiom P5 : ¬ TaiwaneseCitizenship Vladimir ∧ ¬ ManagerAtGazprom Vladimir
axiom P6 : CanRegisterUS Ekaterina ∨ RussianFederationOfficial Ekaterina

theorem main :
  CanParticipateElection2024 Ekaterina ∨ ManagerAtGazprom Vladimir :=
by
  apply Or.inl
  cases P6 with
  | inl hreg =>
      exact P1 Ekaterina hreg
  | inr hrof =>
      have hnotTW : ¬ TaiwaneseCitizenship Ekaterina := P4 Ekaterina hrof
      have hdisj : USCitizenship Ekaterina ∨ TaiwaneseCitizenship Ekaterina := P3 Ekaterina
      have hus : USCitizenship Ekaterina :=
        by
          cases hdisj with
          | inl hus => exact hus
          | inr htw => exact False.elim (hnotTW htw)
      have hreg2 : CanRegisterUS Ekaterina := P2 Ekaterina hus
      exact P1 Ekaterina hreg2
</lean>

ANSWER: True
