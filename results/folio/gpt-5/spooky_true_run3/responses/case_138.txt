<lean>
axiom obj : Type

-- Entities
axiom SOTA : obj  -- the state-of-the-art text summarization model

-- Predicates
axiom Supervised : obj → Prop
axiom Unsupervised : obj → Prop
axiom Reinforcement : obj → Prop
axiom MLAlg : obj → Prop
axiom UsedToTrain : obj → obj → Prop
axiom RequiresLabels : obj → Prop

-- Premises
-- 1) ML algorithms are categorized among supervised, unsupervised, or reinforcement learning
axiom Cat : ∀ a : obj, MLAlg a → (Supervised a ∨ Unsupervised a ∨ Reinforcement a)

-- 2) Unsupervised learning algorithms do not require labeled data
axiom UL_no_labels : ∀ a : obj, Unsupervised a → ¬ RequiresLabels a

-- 3) The SOTA text summarization model is trained with machine learning algorithms
axiom SOTA_trained_by_ML : ∃ a : obj, MLAlg a ∧ UsedToTrain a SOTA

-- 4) Reinforcement learning is not used to train the SOTA text summarization model
axiom RL_not_used : ∀ a : obj, Reinforcement a → ¬ UsedToTrain a SOTA

-- 5) The ML algorithm for training text summarization models requires labeled data (specialized to SOTA)
axiom Train_requires_labels : ∀ a : obj, UsedToTrain a SOTA → RequiresLabels a

-- Target statement to evaluate:
-- "Unsupervised learning is used to train the SOTA text summarization model."
-- Formalization: ∃ a, Unsupervised a ∧ UsedToTrain a SOTA

-- We show the negation holds from the premises (so the statement is not provable).
theorem no_unsupervised_used : ¬ (∃ a : obj, Unsupervised a ∧ UsedToTrain a SOTA) := by
  intro h
  cases h with
  | intro a ha =>
    have hu : Unsupervised a := And.left ha
    have ht : UsedToTrain a SOTA := And.right ha
    have rl : RequiresLabels a := Train_requires_labels a ht
    have nrl : ¬ RequiresLabels a := UL_no_labels a hu
    exact nrl rl
</lean>
