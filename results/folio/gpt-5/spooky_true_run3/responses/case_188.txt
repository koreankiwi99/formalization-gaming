<lean>
axiom obj : Type
axiom Yuri : obj

axiom Pro : obj → Prop                -- Professional basketball player
axiom American : obj → Prop           -- American national
axiom JumpShots : obj → Prop          -- Can do jump shots
axiom LeapsStraight : obj → Prop      -- Leaps straight into the air
axiom ActivateLeg : obj → Prop        -- Activates leg muscles

-- Premises
axiom S1 : ∃ x : obj, Pro x ∧ ¬ American x
axiom A1 : ∀ x : obj, Pro x → JumpShots x
axiom A2 : ∀ x : obj, JumpShots x → LeapsStraight x
axiom A3 : ∀ x : obj, LeapsStraight x → ActivateLeg x
axiom A4 : ¬ ActivateLeg Yuri

theorem yuri_not_american_pro : ¬ (American Yuri ∧ Pro Yuri) := by
  -- Pro Yuri → ActivateLeg Yuri
  have hchain : Pro Yuri → ActivateLeg Yuri := by
    intro hpro
    have hj : JumpShots Yuri := A1 Yuri hpro
    have hl : LeapsStraight Yuri := A2 Yuri hj
    have ha : ActivateLeg Yuri := A3 Yuri hl
    exact ha
  -- From ¬ActivateLeg Yuri, deduce ¬Pro Yuri
  have hnotpro : ¬ Pro Yuri := by
    intro hpro
    exact A4 (hchain hpro)
  -- Hence, not (American Yuri ∧ Pro Yuri)
  intro h
  exact hnotpro h.right
</lean>
ANSWER: True
