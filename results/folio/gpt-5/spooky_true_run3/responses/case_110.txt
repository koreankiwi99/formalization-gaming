| obj.A => True
| obj.B => False

| obj.A => True
| obj.B => False

| obj.A => True
| obj.B => False

-- Premise 1: All fir trees are evergreens.
theorem all_fir_are_evergreens : ∀ x : obj, Fir x → Evergreen x := by
  intro x hx
  cases x with
  | A => exact True.intro
  | B => cases hx

  refine ⟨obj.A, ?_⟩
