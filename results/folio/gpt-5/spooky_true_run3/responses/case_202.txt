<lean>
axiom obj : Type
axiom AiltonSilva : obj
axiom Ailton : obj
axiom Braga : obj
axiom Nautico : obj
axiom Fluminense : obj

axiom FootballPlayer : obj → Prop
axiom BrazilianFootballer : obj → Prop
axiom PlaysFor : obj → obj → Prop
axiom LoanedTo : obj → obj → Prop
axiom FootballClub : obj → Prop

-- Ailton Silva is commonly known as Ailton (treat as the same person)
axiom alias : AiltonSilva = Ailton

-- Ailton was loaned out to Braga
axiom loaned_Ailton_Braga : LoanedTo Ailton Braga

-- Náutico and Braga (and Fluminense) are football clubs
axiom club_Nautico : FootballClub Nautico
axiom club_Braga : FootballClub Braga
axiom club_Fluminense : FootballClub Fluminense

-- Additional context (not needed for the proof but faithful to the text)
axiom fp_Ailton : FootballPlayer Ailton
axiom braz_AiltonSilva : BrazilianFootballer AiltonSilva
axiom plays_AiltonSilva_Nautico : PlaysFor AiltonSilva Nautico

theorem ailton_silva_loaned_to_football_club : ∃ c : obj, FootballClub c ∧ LoanedTo AiltonSilva c :=
by
  have h : LoanedTo AiltonSilva Braga := by
    simpa [alias.symm] using loaned_Ailton_Braga
  exact Exists.intro Braga (And.intro club_Braga h)
</lean>

ANSWER: True
