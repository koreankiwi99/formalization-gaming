<lean>
axiom obj : Type

axiom MachineTranslation : obj

axiom NLPTask : obj → Prop
axiom LanguageGenerationTask : obj → Prop
axiom LanguageUnderstandingTask : obj → Prop
axiom OutputIsTextSequence : obj → Prop

-- A natural language processing task can be categorized as either generation or understanding.
axiom A1 : ∀ x : obj, NLPTask x → (LanguageGenerationTask x ∨ LanguageUnderstandingTask x)

-- If the output of a natural language processing task is a text sequence, then it is a language generation task.
axiom A2 : ∀ x : obj, NLPTask x → OutputIsTextSequence x → LanguageGenerationTask x

-- Machine translation is a natural language processing task.
axiom A3 : NLPTask MachineTranslation

-- The output of machine translation is a text sequence.
axiom A4 : OutputIsTextSequence MachineTranslation

theorem mt_is_language_generation_task : LanguageGenerationTask MachineTranslation := by
  exact A2 MachineTranslation A3 A4
</lean>

ANSWER: True
