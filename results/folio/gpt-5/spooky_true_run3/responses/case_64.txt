<lean>
axiom obj : Type
axiom BrownSwissCattle : obj → Prop
axiom Cow : obj → Prop
axiom Domesticated : obj → Prop
axiom Pet : obj → Prop
axiom Alligator : obj → Prop
axiom Ted : obj

-- Premises
axiom P1 : ∀ x : obj, BrownSwissCattle x → Cow x
axiom P2 : ∀ x : obj, Cow x → Domesticated x
axiom P3 : ∀ x : obj, Alligator x → ¬ Domesticated x
axiom P4 : Alligator Ted
axiom P5 : ∃ x : obj, Pet x ∧ BrownSwissCattle x

-- Conclusion to prove
theorem ted_bsc_implies_not_pet : BrownSwissCattle Ted → ¬ Pet Ted := by
  intro hBSC
  have hCow : Cow Ted := P1 Ted hBSC
  have hDom : Domesticated Ted := P2 Ted hCow
  have hNotDom : ¬ Domesticated Ted := P3 Ted P4
  have hFalse : False := hNotDom hDom
  exact (fun _ => False.elim hFalse)
</lean>

ANSWER: True
