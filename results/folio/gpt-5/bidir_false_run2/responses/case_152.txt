theorem mixture_not_pair : ¬(Chemical Mixture ∧ OnlyOneElement Mixture) := by
  intro h
  have hC : Chemical Mixture := And.left h
  have hO : OnlyOneElement Mixture := And.right h
  have : ¬OnlyOneElement Mixture := A5 Mixture hC
  exact this hO

  cases hdisj with
  | inl hboth =>
      have : False := mixture_not_pair hboth
      exact False.elim this
  | inr hneither =>
      exact hneither
