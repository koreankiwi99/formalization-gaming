<lean>
axiom obj : Type
axiom Bird : obj → Prop
axiom Hawk : obj → Prop
axiom Swims : obj → Prop

-- A hawk never swims.
axiom Hawks_never_swim : ∀ x : obj, Hawk x → ¬Swims x
-- Some birds are hawks.
axiom Some_birds_are_hawks : ∃ x : obj, Bird x ∧ Hawk x

-- We prove the negation of "All birds swim."
theorem not_all_birds_swim : ¬(∀ x : obj, Bird x → Swims x) := by
  intro hAll
  apply Exists.elim Some_birds_are_hawks
  intro a
  intro h
  have hBa : Bird a := And.left h
  have hHa : Hawk a := And.right h
  have hSw : Swims a := hAll a hBa
  have hNotSw : ¬Swims a := Hawks_never_swim a hHa
  exact hNotSw hSw
</lean>

ANSWER: False
