<lean>
axiom obj : Type
axiom Human : obj → Prop
axiom Horse : obj → Prop
axiom Hooves : obj → Prop

-- Premises
axiom H1 : ∀ x : obj, Horse x → Hooves x
axiom H2 : ∀ x : obj, Human x → ¬Hooves x

-- Conclusion is false: there cannot exist a human who is a horse
theorem no_humans_are_horses : ¬(∃ x : obj, Human x ∧ Horse x) := by
  intro h
  apply Exists.elim h
  intro x hx
  have hx_hooves : Hooves x := H1 x hx.right
  have hx_not_hooves : ¬Hooves x := H2 x hx.left
  exact hx_not_hooves hx_hooves
</lean>

ANSWER: False
