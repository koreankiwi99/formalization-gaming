<lean>
axiom obj : Type
axiom PicurisMountains : obj
axiom JuanDeOnate : obj
axiom HardingPegmatiteMine : obj

axiom MountainRange : obj → Prop
axiom Mine : obj → Prop
axiom InTexas : obj → Prop
axiom InNewMexico : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop
axiom Visited : obj → obj → Prop

-- Context axioms
axiom A1 : MountainRange PicurisMountains
axiom A2 : InNewMexico PicurisMountains ∨ InTexas PicurisMountains
axiom A3 : Visited JuanDeOnate PicurisMountains
axiom A4 : Mine HardingPegmatiteMine
axiom A5 : LocatedIn HardingPegmatiteMine PicurisMountains
axiom A6 : Donated HardingPegmatiteMine
axiom A7 : ∀ r : obj, MountainRange r → InTexas r → ¬∃ m : obj, Mine m ∧ LocatedIn m r ∧ Donated m

-- From the donated mine located in Picuris and the "no such Texas ranges" rule, Picuris cannot be in Texas
theorem picuris_not_in_texas : ¬InTexas PicurisMountains := by
  intro hTX
  have hNo := A7 PicurisMountains A1 hTX
  have hEx : ∃ m : obj, Mine m ∧ LocatedIn m PicurisMountains ∧ Donated m := by
    exact Exists.intro HardingPegmatiteMine (And.intro A4 (And.intro A5 A6))
  exact hNo hEx

-- Negation of the statement: "Juan de Onate visited a mountain range in Texas"
-- Interpreted as: he visited the Picuris Mountains and they are in Texas
theorem not_statement : ¬(Visited JuanDeOnate PicurisMountains ∧ InTexas PicurisMountains) := by
  intro h
  have hTX : InTexas PicurisMountains := And.right h
  exact picuris_not_in_texas hTX
</lean>
