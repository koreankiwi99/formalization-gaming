<lean>
axiom obj : Type

axiom AtMS : obj → Prop
axiom Work : obj → Prop
axiom Need : obj → Prop
axiom OrderTakeout : obj → Prop
axiom EnjoyDining : obj → Prop
axiom NotPicky : obj → Prop
axiom SpendDiningHalls : obj → Prop

axiom Hannah : obj

-- Premises
axiom R1 : ∀ x : obj, AtMS x ∧ Work x → Need x
axiom R2 : ∀ x : obj, AtMS x ∧ OrderTakeout x → Work x
axiom R3 : ∀ x : obj, AtMS x → (OrderTakeout x ∨ EnjoyDining x)
axiom R4 : ∀ x : obj, AtMS x ∧ EnjoyDining x → NotPicky x
axiom R5 : ∀ x : obj, AtMS x ∧ EnjoyDining x → SpendDiningHalls x
axiom T6 : AtMS Hannah
axiom T7 : Work Hannah ∧ (Need Hannah → (NotPicky Hannah ∧ ¬Need Hannah))

-- Derive a contradiction from the premises
theorem contradiction_from_premises : False := by
  have hAt : AtMS Hannah := T6
  have hWork : Work Hannah := And.left T7
  have hNeed : Need Hannah := R1 Hannah ⟨hAt, hWork⟩
  have hPair : NotPicky Hannah ∧ ¬Need Hannah := (And.right T7) hNeed
  have hNotNeed : ¬Need Hannah := And.right hPair
  exact hNotNeed hNeed

-- Statement to evaluate:
-- Hannah is at Mary's school and she either is not a picky eater or, if she is,
-- then she spends a lot of her time eating and catching up with friends in the campus dining halls.
def Statement : Prop :=
  AtMS Hannah ∧ (NotPicky Hannah ∨ ((¬ NotPicky Hannah) → SpendDiningHalls Hannah))

-- Prove the negation of the statement (since premises are inconsistent)
theorem negate_Statement : ¬Statement := by
  intro _
  exact contradiction_from_premises
</lean>
