  have hSBT : SiblingsBornTogether Luke := R3 Luke hC

  have hSpend : SpendWithSiblings Luke := R1 Luke hMB
  have hHome : LiveAtHome Luke := R5 Luke hSpend
  have hNotStr : ¬LiveWithStrangers Luke := R4 Luke hHome
