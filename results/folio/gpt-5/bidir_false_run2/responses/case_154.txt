axiom OneElement : obj → Prop
axiom Alkane : obj → Prop

axiom R1 : ∀ x : obj, Hydrocarbon x → Organic x
axiom R2 : ∀ x : obj, Alkane x → Hydrocarbon x
axiom R3 : ∀ x : obj, Organic x → Chemical x
axiom R4 : ∀ x : obj, Organic x → ContainsCarbon x
axiom R5 : ∀ x : obj, Chemical x → ¬OneElement x
axiom R6 : (Chemical Mixture ∧ OneElement Mixture) ∨ (¬Chemical Mixture ∧ ¬OneElement Mixture)

-- From R5 and R6 we can deduce the mixture is not a chemical compound and does not have only one element
theorem mixture_not_chem_and_not_one : ¬Chemical Mixture ∧ ¬OneElement Mixture := by
  cases R6 with
  | inl h =>
    have hC : Chemical Mixture := h.left
    have hOne : OneElement Mixture := h.right
    have hNoOne : ¬OneElement Mixture := R5 Mixture hC
    have contra : False := hNoOne hOne
    exact And.intro (False.elim contra) (False.elim contra)
  | inr h =>
    exact h

-- Therefore the mixture is not an alkane (since alkane → hydrocarbon → organic → chemical)
theorem mixture_not_alkane : ¬Alkane Mixture := by
  intro hAlk
  have hHydro : Hydrocarbon Mixture := R2 Mixture hAlk
  have hOrg : Organic Mixture := R1 Mixture hHydro
  have hChem : Chemical Mixture := R3 Mixture hOrg
  exact (mixture_not_chem_and_not_one.left) hChem

-- Consequently, the conclusion of the conditional holds unconditionally:
theorem conclusion_holds_unconditionally : ¬Chemical Mixture ∧ ¬Alkane Mixture := by
  exact And.intro (mixture_not_chem_and_not_one.left) mixture_not_alkane
