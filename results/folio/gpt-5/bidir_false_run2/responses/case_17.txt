axiom obj : Type
axiom Rabbit : obj → Prop
axiom Turtle : obj → Prop
axiom Squirrel : obj → Prop
axiom Cute : obj → Prop
axiom Spotted : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop
axiom Rockie : obj

axiom A1 : ∀ x : obj, Rabbit x ∧ Spotted x → Cute x
axiom A2 : ∃ x : obj, Turtle x ∧ Spotted x
axiom A3 : ∀ x : obj, Spotted x → (Rabbit x ∨ Squirrel x)
axiom A4 : ∀ x : obj, Skittish x → ¬ Calm x
axiom A5 : ∀ x : obj, Squirrel x ∧ Spotted x → Skittish x
axiom A6 : Spotted Rockie ∧ Calm Rockie

-- From the premises we can derive that Rockie is not a squirrel
theorem Rockie_not_squirrel : ¬ Squirrel Rockie :=
  by
    intro hs
    have hsp : Spotted Rockie := And.left A6
    have sk : Skittish Rockie := A5 Rockie (And.intro hs hsp)
    have ncalm : ¬ Calm Rockie := A4 Rockie sk
    exact ncalm (And.right A6)

-- And thus Rockie is a rabbit
theorem Rockie_is_rabbit : Rabbit Rockie :=
  by
    have hsp : Spotted Rockie := And.left A6
    have disj : Rabbit Rockie ∨ Squirrel Rockie := A3 Rockie hsp
    cases disj with
    | inl hr => exact hr
    | inr hs => exact False.elim (Rockie_not_squirrel hs)
