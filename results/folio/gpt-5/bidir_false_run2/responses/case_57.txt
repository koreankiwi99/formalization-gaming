  ∀ c : City,
    (∃ s : State, InState c s ∧ InState Butte s) →
    ¬ (∃ s : State, InState c s ∧ InState StPierre s)

-- Conclusion to refute: Butte and St Pierre are in the same state
theorem not_same_state_Butte_StPierre :
  ¬ (∃ s : State, InState Butte s ∧ InState StPierre s) :=
by
  have hButteInSomeState : ∃ s : State, InState Butte s ∧ InState Butte s :=
    ⟨Montana, And.intro T2 T2⟩
