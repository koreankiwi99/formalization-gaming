theorem oneSeater_implies_not_tesla : ∀ x : obj, OneSeater x → ¬TeslaModel3 x :=
  fun x hx => by
    intro hTesla
    have hMP : MultiPassenger x := A2 x hTesla
    have hNotOne : ¬OneSeater x := A1 x hMP
    exact hNotOne hx

theorem soloEV_implies_oneSeater_John : SoloEV JohnCar → OneSeater JohnCar :=
  by
    intro h
    have hS : SingleSeatElectricCar JohnCar := A4 JohnCar h
    exact A3 JohnCar hS

theorem soloEV_implies_not_tesla_John : SoloEV JohnCar → ¬TeslaModel3 JohnCar :=
  by

    have hMP : MultiPassenger JohnCar := A2 JohnCar hT
    have hNotOne : ¬OneSeater JohnCar := A1 JohnCar hMP
