axiom obj : Type
axiom Happy : obj → Prop
axiom Sad : obj → Prop

-- Events are either happy or sad (and not both)
axiom Either : ∀ x : obj, Happy x ∨ Sad x
axiom Exclusive : ∀ x : obj, Happy x → ¬Sad x

-- At least one event is happy
axiom ExistsHappy : ∃ x : obj, Happy x

-- Statement to evaluate: All events are sad (we prove its negation)
theorem not_all_sad : ¬(∀ x : obj, Sad x) := by
  intro hAllSad
  obtain ⟨e, heHappy⟩ := ExistsHappy
  have heSad : Sad e := hAllSad e
  have hNotSad : ¬Sad e := Exclusive e heHappy
  exact hNotSad heSad
</lean>
