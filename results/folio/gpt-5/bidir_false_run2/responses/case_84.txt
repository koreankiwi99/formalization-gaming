axiom HPS : obj  -- Hamden Plaza Subway store

axiom ManyNeg : obj → Prop

axiom RatingGT4 : obj → Prop
axiom Popular : obj → Prop

-- If listed in Yelp’s recommendations, then not many negative reviews.
axiom A1 : ∀ x : obj, Listed x → ¬ ManyNeg x
-- All with rating > 4 are listed.
axiom A2 : ∀ x : obj, RatingGT4 x → Listed x
-- Some do not provide take-out and receive many negative reviews.
axiom A3 : ∃ x : obj, ¬ Takeout x ∧ ManyNeg x
-- All popular have rating > 4.
axiom A4 : ∀ x : obj, Popular x → RatingGT4 x
-- HPS has rating > 4 or is popular.
axiom A5 : RatingGT4 HPS ∨ Popular HPS

-- The statement to evaluate: Takeout HPS ∨ ¬ ManyNeg HPS
-- In fact, we can prove it holds (so we cannot prove its negation).
theorem statement_true : Takeout HPS ∨ ¬ ManyNeg HPS := by
  cases A5 with
  | inl hR =>
      have hL : Listed HPS := A2 HPS hR
      have hNM : ¬ ManyNeg HPS := A1 HPS hL
      exact Or.inr hNM
  | inr hPop =>
      have hR : RatingGT4 HPS := A4 HPS hPop
      have hL : Listed HPS := A2 HPS hR
      have hNM : ¬ ManyNeg HPS := A1 HPS hL
      exact Or.inr hNM
