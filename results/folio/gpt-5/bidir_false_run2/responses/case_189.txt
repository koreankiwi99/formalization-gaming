<lean>
axiom obj : Type
axiom Yuri : obj

axiom American : obj → Prop
axiom ProBBall : obj → Prop
axiom CanJumpShots : obj → Prop
axiom LeapsStraight : obj → Prop
axiom ActivateLegs : obj → Prop

-- Some professional basketball players are not American nationals.
axiom E1 : ∃ x : obj, ProBBall x ∧ ¬American x
-- All professional basketball players can do jump shots.
axiom R1 : ∀ x : obj, ProBBall x → CanJumpShots x
-- If someone can jump shots, they leap straight into the air.
axiom R2 : ∀ x : obj, CanJumpShots x → LeapsStraight x
-- If someone leaps straight into the air, they activate their leg muscles.
axiom R3 : ∀ x : obj, LeapsStraight x → ActivateLegs x
-- Yuri does not activate his leg muscles.
axiom T1 : ¬ActivateLegs Yuri

-- Statement to evaluate: Yuri is an American professional basketball player.
-- We prove its negation.
theorem yuri_not_american_pro : ¬(American Yuri ∧ ProBBall Yuri) := by
  intro h
  have hypro : ProBBall Yuri := And.right h
  have hjs : CanJumpShots Yuri := R1 Yuri hypro
  have hleap : LeapsStraight Yuri := R2 Yuri hjs
  have hact : ActivateLegs Yuri := R3 Yuri hleap
  exact T1 hact
</lean>

ANSWER: False
