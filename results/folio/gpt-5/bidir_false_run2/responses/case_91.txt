<lean>
axiom obj : Type
axiom MachineTranslation : obj
axiom NLPTask : obj → Prop
axiom LangGen : obj → Prop
axiom LangUnderstand : obj → Prop
axiom TextSeqOutput : obj → Prop

-- Every NLP task is either generation or understanding (coverage)
axiom COV : ∀ x : obj, NLPTask x → (LangGen x ∨ LangUnderstand x)
-- NLP task categories are exclusive (no task is both)
axiom EXCL : ∀ x : obj, NLPTask x → ¬(LangGen x ∧ LangUnderstand x)

-- If the output is a text sequence, then it is a language generation task
axiom R1 : ∀ x : obj, TextSeqOutput x → LangGen x
-- Machine translation is an NLP task
axiom T1 : NLPTask MachineTranslation
-- The output of machine translation is a text sequence
axiom T2 : TextSeqOutput MachineTranslation

theorem mt_not_understanding : ¬LangUnderstand MachineTranslation := by
  have hgen : LangGen MachineTranslation := R1 MachineTranslation T2
  have hnotboth : ¬(LangGen MachineTranslation ∧ LangUnderstand MachineTranslation) :=
    EXCL MachineTranslation T1
  intro hunder
  exact hnotboth (And.intro hgen hunder)
</lean>

ANSWER: False
