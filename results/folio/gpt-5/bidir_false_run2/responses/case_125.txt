<lean>
axiom obj : Type
axiom Rock : obj

axiom PeterPet : obj → Prop
axiom Mammal : obj → Prop
axiom Animal : obj → Prop
axiom Monkey : obj → Prop
axiom Bird : obj → Prop
axiom CanFly : obj → Prop
axiom CanBreathe : obj → Prop
axiom HasWings : obj → Prop

-- All of Peter's pets that are mammals are animals
axiom R1 : ∀ x : obj, PeterPet x ∧ Mammal x → Animal x
-- All monkeys are mammals
axiom R2 : ∀ x : obj, Monkey x → Mammal x
-- Peter's pets are all either monkeys or birds
axiom R3 : ∀ x : obj, PeterPet x → (Monkey x ∨ Bird x)
-- Peter's birds can fly
axiom R4 : ∀ x : obj, PeterPet x ∧ Bird x → CanFly x
-- All animals that are Peter's pets can breathe
axiom R5 : ∀ x : obj, PeterPet x ∧ Animal x → CanBreathe x
-- If Peter's pet can fly, then it has wings
axiom R6 : ∀ x : obj, PeterPet x ∧ CanFly x → HasWings x

-- Rock is Peter's pet
axiom T1 : PeterPet Rock
-- Rock can fly, or Rock is a bird, or Rock cannot breathe
axiom T2 : CanFly Rock ∨ (Bird Rock ∨ ¬ CanBreathe Rock)

-- From the premises, Rock has wings (so we cannot prove ¬HasWings Rock)
theorem rock_has_wings : HasWings Rock := by
  have pet : PeterPet Rock := T1
  have disj : CanFly Rock ∨ (Bird Rock ∨ ¬ CanBreathe Rock) := T2
  cases disj with
  | inl fly =>
      exact R6 Rock ⟨pet, fly⟩
  | inr rest =>
      cases rest with
      | inl bird =>
          have fly2 : CanFly Rock := R4 Rock ⟨pet, bird⟩
          exact R6 Rock ⟨pet, fly2⟩
      | inr not_breathe =>
          have mb : Monkey Rock ∨ Bird Rock := R3 Rock pet
          cases mb with
          | inl monkey =>
              have mammal : Mammal Rock := R2 Rock monkey
              have animal : Animal Rock := R1 Rock ⟨pet, mammal⟩
              have breathe : CanBreathe Rock := R5 Rock ⟨pet, animal⟩
              exact False.elim (not_breathe breathe)
          | inr bird =>
              have fly2 : CanFly Rock := R4 Rock ⟨pet, bird⟩
              exact R6 Rock ⟨pet, fly2⟩
