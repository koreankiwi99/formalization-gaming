<lean>
axiom obj : Type
axiom Taylor : obj

axiom SaladRegular : obj → Prop
axiom HealthConscious : obj → Prop
axiom GrewUpHC : obj → Prop
axiom FulfillDaily : obj → Prop
axiom Disregard : obj → Prop
axiom VisitGymDaily : obj → Prop

-- Rules from the context
axiom R1 : ∀ x : obj, SaladRegular x → HealthConscious x
axiom R2 : ∀ x : obj, GrewUpHC x → SaladRegular x
axiom R3 : ∀ x : obj, FulfillDaily x → GrewUpHC x
axiom R4 : ∀ x : obj, Disregard x → ¬HealthConscious x
axiom R5 : ∀ x : obj, VisitGymDaily x → FulfillDaily x

-- Taylor either both grew up HC and disregards, or neither
axiom T1 : (GrewUpHC Taylor ∧ Disregard Taylor) ∨ (¬GrewUpHC Taylor ∧ ¬Disregard Taylor)

theorem taylor_not_visitgym : ¬VisitGymDaily Taylor := by
  intro H
  have FD : FulfillDaily Taylor := R5 Taylor H
  have GU : GrewUpHC Taylor := R3 Taylor FD
  have SR : SaladRegular Taylor := R2 Taylor GU
  have HC : HealthConscious Taylor := R1 Taylor SR
  cases T1 with
  | inl h =>
      have d : Disregard Taylor := h.right
      have notHC : ¬HealthConscious Taylor := R4 Taylor d
      exact notHC HC
  | inr h =>
      have notGU : ¬GrewUpHC Taylor := h.left
      exact notGU GU
</lean>

ANSWER: False
