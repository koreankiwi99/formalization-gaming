axiom Cleanly : obj → Prop
axiom ValuesOrder : obj → Prop
axiom FamilyPrioritizeOrder : obj → Prop
axiom ClumsyFoodie : obj → Prop
axiom GoesOutFrequentlyForNewFood : obj → Prop

-- If people spill a lot of food on their clothing, then they are not notably tidy.
axiom R1 : ∀ x : obj, Spills x → ¬Tidy x
-- All clumsy foodies who go out frequently to find new food restaurants to try spill a lot of food.
axiom R2 : ∀ x : obj, ClumsyFoodie x ∧ GoesOutFrequentlyForNewFood x → Spills x
-- If people are cleanly, then they are notably tidy.
axiom R3 : ∀ x : obj, Cleanly x → Tidy x
-- If people value order and spotlessness, then they are cleanly.
axiom R4 : ∀ x : obj, ValuesOrder x → Cleanly x
-- All people whose families prioritize order and spotlessness value order and spotlessness.
axiom R5 : ∀ x : obj, FamilyPrioritizeOrder x → ValuesOrder x

-- Peter is either (spills ∧ cleanly) or (¬spills ∧ ¬cleanly)
axiom Pdisj : (Spills Peter ∧ Cleanly Peter) ∨ (¬Spills Peter ∧ ¬Cleanly Peter)

-- From the rules, (Spills ∧ Cleanly) leads to a contradiction (Tidy ∧ ¬Tidy).
theorem not_both : ¬(Spills Peter ∧ Cleanly Peter) := by
  intro h
  have hs : Spills Peter := h.left
  have hc : Cleanly Peter := h.right
  have ht : Tidy Peter := R3 Peter hc
  have hnt : ¬Tidy Peter := R1 Peter hs
  exact hnt ht

-- Hence the right disjunct must hold: ¬Spills ∧ ¬Cleanly for Peter.
theorem peter_not_spills_and_not_cleanly : ¬Spills Peter ∧ ¬Cleanly Peter := by
  have h := Pdisj
  have hna := not_both
  cases h with
  | inl hA => exact False.elim (hna hA)
  | inr hB => exact hB
