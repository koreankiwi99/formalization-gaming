axiom Range : Type
axiom Mine : Type
axiom Person : Type

axiom Picuris : Range
axiom Harding : Mine
axiom Juan : Person

axiom InNM : Range → Prop
axiom InTX : Range → Prop
axiom LocatedIn : Mine → Range → Prop
axiom Donated : Mine → Prop
axiom Visited : Person → Range → Prop

-- Context axioms
axiom A1 : InNM Picuris ∨ InTX Picuris
axiom A2 : LocatedIn Harding Picuris
axiom A3 : Donated Harding
axiom A4 : Visited Juan Picuris
axiom A5 : ∀ r : Range, InTX r → ¬∃ m : Mine, LocatedIn m r ∧ Donated m

theorem not_inTX_picuris : ¬InTX Picuris := by
  intro htx
  have h := A5 Picuris htx
  have hex : ∃ m : Mine, LocatedIn m Picuris ∧ Donated m := by
    exact Exists.intro Harding (And.intro A2 A3)
  exact h hex

theorem picuris_inNM : InNM Picuris := by
  cases A1 with
  | inl hnm => exact hnm
  | inr htx =>
      have contra : False := not_inTX_picuris htx
      exact False.elim contra

-- Negation of the statement: there exists a mountain range in NM containing the Harding Mine
theorem harding_in_nm_range : ∃ r : Range, InNM r ∧ LocatedIn Harding r :=
  Exists.intro Picuris (And.intro picuris_inNM A2)
