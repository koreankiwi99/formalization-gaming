<lean>
axiom obj : Type
axiom Yale : obj
axiom PrivateIvyLeagueResearchUniversity : obj → Prop
axiom MovedToNewHaven : obj → Prop

-- From the context:
axiom T1 : PrivateIvyLeagueResearchUniversity Yale
axiom T2 : MovedToNewHaven Yale

-- Claim: There exists a private Ivy League research university that moved to New Haven.
theorem exists_private_ivy_moved :
  ∃ x : obj, PrivateIvyLeagueResearchUniversity x ∧ MovedToNewHaven x := by
  exact Exists.intro Yale (And.intro T1 T2)
</lean>

ANSWER: True
