<lean>
axiom obj : Type
axiom James : obj

-- Predicates
axiom WorksInLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom HasPTUniJob : obj → Prop
axiom TakingDatabase : obj → Prop
axiom TakingWithDavid : obj → Prop
axiom WorksInLab : obj → Prop

-- Axioms from the context
-- None of the students who currently work in the library are from the computer science department.
axiom A1 : ∀ x : obj, WorksInLibrary x → ¬ FromCS x
-- All of the students with part-time jobs offered by the university currently work in the library.
axiom A2 : ∀ x : obj, HasPTUniJob x → WorksInLibrary x
-- All of the students who are taking the database course are from the computer science department.
axiom A3 : ∀ x : obj, TakingDatabase x → FromCS x
-- All of the students taking a class with Professor David are taking the database course.
axiom A4 : ∀ x : obj, TakingWithDavid x → TakingDatabase x
-- All students working in the lab are taking a class with Professor David.
axiom A5 : ∀ x : obj, WorksInLab x → TakingWithDavid x
-- James is a student working in the lab.
axiom A6 : WorksInLab James
-- James doesn't work in the lab or have a part-time job offered by the university. (one plausible formalization)
axiom A7 : ¬WorksInLab James ∨ ¬HasPTUniJob James

-- Goal: James either takes the database course or has a part-time job offered by the university.
theorem james_db_or_pt : TakingDatabase James ∨ HasPTUniJob James :=
by
  have hWD : TakingWithDavid James := A5 James A6
  have hDB : TakingDatabase James := A4 James hWD
  exact Or.inl hDB
</lean>

ANSWER: True
