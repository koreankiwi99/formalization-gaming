<lean>
axiom obj : Type
axiom James : obj

axiom WorksLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom UniPTJob : obj → Prop
axiom TakesDB : obj → Prop
axiom TakesWithDavid : obj → Prop
axiom WorksLab : obj → Prop

-- None of the students who currently work in the library are from the CS department.
axiom R1 : ∀ x : obj, WorksLibrary x → ¬ FromCS x
-- All students with university part-time jobs work in the library.
axiom R2 : ∀ x : obj, UniPTJob x → WorksLibrary x
-- All students taking the database course are from the CS department.
axiom R3 : ∀ x : obj, TakesDB x → FromCS x
-- All students taking a class with Professor David are taking the database course.
axiom R4 : ∀ x : obj, TakesWithDavid x → TakesDB x
-- All students working in the lab are taking a class with Professor David.
axiom R5 : ∀ x : obj, WorksLab x → TakesWithDavid x
-- James is working in the lab.
axiom T6 : WorksLab James
-- James doesn't work in the lab or have a university part-time job. (interpreted as ¬(lab ∨ uni-PT))
axiom T7 : ¬ (WorksLab James ∨ UniPTJob James)

theorem james_takes_db : TakesDB James := by
  have hDavid : TakesWithDavid James := R5 James T6
  exact R4 James hDavid
</lean>

ANSWER: True
