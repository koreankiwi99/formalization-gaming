<lean>
axiom obj : Type
axiom Rockie : obj

axiom Rabbit : obj → Prop
axiom Turtle : obj → Prop
axiom Squirrel : obj → Prop
axiom Spotted : obj → Prop
axiom Cute : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop

-- All rabbits that can be spotted near the campus are cute.
axiom R1 : ∀ x : obj, Rabbit x → Spotted x → Cute x
-- Some turtles can be spotted near the campus.
axiom E1 : ∃ x : obj, Turtle x ∧ Spotted x
-- The only animals that can be spotted near the campus are rabbits and squirrels.
axiom OnlySpottedAreRabbitOrSquirrel : ∀ x : obj, Spotted x → (Rabbit x ∨ Squirrel x)
-- If something is skittish, then it is not calm.
axiom SkittishNotCalm : ∀ x : obj, Skittish x → ¬Calm x
-- All the squirrels that can be spotted near the campus are skittish.
axiom SquirrelSpottedSkittish : ∀ x : obj, Squirrel x → Spotted x → Skittish x
-- Rockie can be spotted near the campus, and it is calm.
axiom spotted_Rockie : Spotted Rockie
axiom calm_Rockie : Calm Rockie

theorem rockie_turtle_or_cute : Turtle Rockie ∨ Cute Rockie := by
  have hOr : Rabbit Rockie ∨ Squirrel Rockie :=
    OnlySpottedAreRabbitOrSquirrel Rockie spotted_Rockie
  have hRabbit : Rabbit Rockie :=
    Or.elim hOr
      (fun hr => hr)
      (fun hs =>
        have hsk : Skittish Rockie := SquirrelSpottedSkittish Rockie hs spotted_Rockie
        have hnc : ¬Calm Rockie := SkittishNotCalm Rockie hsk
        have contra : False := hnc calm_Rockie
        False.elim contra
      )
  have hCute : Cute Rockie := R1 Rockie hRabbit spotted_Rockie
  exact Or.inr hCute
</lean>

ANSWER: True
