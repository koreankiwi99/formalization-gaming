<lean>
axiom obj : Type
axiom KO : obj

axiom Growth : obj → Prop
axiom BoughtRapid : obj → Prop
axiom Suitable : obj → Prop
axiom Mature : obj → Prop
axiom Stock : obj → Prop
axiom Volatile : obj → Prop

-- All growth stocks are bought to earn profits from rapid price appreciation.
axiom A1 : ∀ x : obj, Growth x → BoughtRapid x
-- If a stock is bought to earn profits from rapid price appreciation, then it is not suitable for a retirement fund.
axiom A2 : ∀ x : obj, BoughtRapid x → ¬Suitable x
-- Some stocks are growth stocks.
axiom A3 : ∃ x : obj, Stock x ∧ Growth x
-- All mature stocks are suitable for a retirement fund.
axiom A4 : ∀ x : obj, Mature x → Suitable x
-- KO is a mature stock.
axiom A5 : Mature KO

-- Claim: If KO is a growth stock or bought to earn profits from rapid price appreciation,
-- then KO is neither a stock nor is its price volatile.
theorem ko_implication :
  (Growth KO ∨ BoughtRapid KO) → (¬Stock KO ∧ ¬Volatile KO) := by
  intro h
  have S : Suitable KO := A4 KO A5
  cases h with
  | inl hg =>
      have br : BoughtRapid KO := A1 KO hg
      have ns : ¬Suitable KO := A2 KO br
      have f : False := ns S
      have nsStock : ¬Stock KO := by intro hs; exact False.elim f
      have nsVol : ¬Volatile KO := by intro hv; exact False.elim f
      exact And.intro nsStock nsVol
  | inr hb =>
      have ns : ¬Suitable KO := A2 KO hb
      have f : False := ns S
      have nsStock : ¬Stock KO := by intro hs; exact False.elim f
      have nsVol : ¬Volatile KO := by intro hv; exact False.elim f
      exact And.intro nsStock nsVol
</lean>

ANSWER: True
