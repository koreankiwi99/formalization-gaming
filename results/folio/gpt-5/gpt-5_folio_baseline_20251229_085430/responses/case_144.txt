axiom Person : Type
axiom Vladimir : Person
axiom Ekaterina : Person

axiom CanRegisterUS : Person → Prop
axiom Participate2024 : Person → Prop
axiom USCitizen : Person → Prop
axiom TaiwaneseCitizen : Person → Prop
axiom RussianOfficial : Person → Prop
axiom ManagerGazprom : Person → Prop

-- 1. Everyone who can register to vote in the US can participate in the 2024 US presidential election.
axiom R1 : ∀ x : Person, CanRegisterUS x → Participate2024 x
-- 2. If someone has US citizenship, then they can register to vote in the US.
axiom R2 : ∀ x : Person, USCitizen x → CanRegisterUS x
-- 3. A person either has US citizenship or Taiwanese citizenship.
axiom R3 : ∀ x : Person, USCitizen x ∨ TaiwaneseCitizen x
-- 4. No Russian Federation officials hold Taiwanese citizenship.
axiom R4 : ∀ x : Person, RussianOfficial x → ¬TaiwaneseCitizen x
-- 5. Vladimir neither holds Taiwanese citizenship nor is he a manager at Gazprom.
axiom F1 : ¬TaiwaneseCitizen Vladimir ∧ ¬ManagerGazprom Vladimir
-- 6. Ekaterina can register to vote in the US, or she is a Russian federation official.
axiom E1 : CanRegisterUS Ekaterina ∨ RussianOfficial Ekaterina

-- Some consequences from the context
theorem vladimir_not_taiwanese : ¬TaiwaneseCitizen Vladimir := by
  exact And.left F1

theorem vladimir_us_citizen : USCitizen Vladimir := by
  have hnt : ¬TaiwaneseCitizen Vladimir := And.left F1
  have h := R3 Vladimir
  cases h with
  | inl hUS => exact hUS
  | inr hTW =>
      have : False := hnt hTW
      exact False.elim this

theorem vladimir_can_register : CanRegisterUS Vladimir := by
  exact R2 Vladimir vladimir_us_citizen

theorem vladimir_can_participate : Participate2024 Vladimir := by
  exact R1 Vladimir vladimir_can_register
