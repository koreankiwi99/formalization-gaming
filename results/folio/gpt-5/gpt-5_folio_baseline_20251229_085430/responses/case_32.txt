<lean>
axiom obj : Type
axiom Djokovic : obj

axiom AtMixer : obj → Prop
axiom GrandSlamChampion : obj → Prop
axiom OscarNominatedActor : obj → Prop
axiom ProfessionalTennisPlayer : obj → Prop
axiom Celebrity : obj → Prop
axiom Athlete : obj → Prop
axiom WellPaid : obj → Prop
axiom LivesInTaxHaven : obj → Prop
axiom Famous : obj → Prop

-- Everyone at the mixer is a Grand Slam champion or an Oscar-nominated actor.
axiom A1 : ∀ x : obj, AtMixer x → (GrandSlamChampion x ∨ OscarNominatedActor x)
-- Every Grand Slam champion at the mixer is a professional tennis player.
axiom A2 : ∀ x : obj, AtMixer x → GrandSlamChampion x → ProfessionalTennisPlayer x
-- All Oscar-nominated actors at the mixer are celebrities.
axiom A3 : ∀ x : obj, AtMixer x → OscarNominatedActor x → Celebrity x
-- All professional tennis players at the mixer are athletes.
axiom A4 : ∀ x : obj, AtMixer x → ProfessionalTennisPlayer x → Athlete x
-- If a person at the mixer is a celebrity, then they are well paid.
axiom A5 : ∀ x : obj, AtMixer x → Celebrity x → WellPaid x
-- If a person at the mixer is an athlete, then they are famous.
axiom A6 : ∀ x : obj, AtMixer x → Athlete x → Famous x
-- All well-paid people at the mixer live in tax havens.
axiom A7 : ∀ x : obj, AtMixer x → WellPaid x → LivesInTaxHaven x

-- Djokovic is at the mixer
axiom T_Dj : AtMixer Djokovic
-- If Djokovic is a famous athlete, then Djokovic is well-paid.
axiom A8 : (Famous Djokovic ∧ Athlete Djokovic) → WellPaid Djokovic

theorem Djokovic_lives_tax_haven : LivesInTaxHaven Djokovic :=
by
  have hAt : AtMixer Djokovic := T_Dj
  have hdisj : GrandSlamChampion Djokovic ∨ OscarNominatedActor Djokovic := A1 Djokovic hAt
  cases hdisj with
  | inl hGSC =>
    have hPTP : ProfessionalTennisPlayer Djokovic := A2 Djokovic hAt hGSC
    have hAth : Athlete Djokovic := A4 Djokovic hAt hPTP
    have hFam : Famous Djokovic := A6 Djokovic hAt hAth
    have hWell : WellPaid Djokovic := A8 (And.intro hFam hAth)
    exact A7 Djokovic hAt hWell
  | inr hONA =>
    have hCeleb : Celebrity Djokovic := A3 Djokovic hAt hONA
    have hWell : WellPaid Djokovic := A5 Djokovic hAt hCeleb
    exact A7 Djokovic hAt hWell
</lean>

ANSWER: False
