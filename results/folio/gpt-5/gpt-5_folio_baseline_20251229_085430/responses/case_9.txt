<lean>
axiom obj : Type
axiom Tom : obj

axiom WildTurkey : obj → Prop
axiom Eastern : obj → Prop
axiom Osceola : obj → Prop
axiom Goulds : obj → Prop
axiom Merriams : obj → Prop
axiom RioGrande : obj → Prop
axiom Ocellated : obj → Prop

-- Tom's facts
axiom T_wild : WildTurkey Tom
axiom N_eastern : ¬ Eastern Tom
axiom N_osceola : ¬ Osceola Tom
axiom N_goulds : ¬ Goulds Tom
axiom N_merriams : ¬ Merriams Tom
axiom N_riogrande : ¬ RioGrande Tom

-- Exhaustiveness of types: every wild turkey is one of these six types
axiom types_exhaustive :
  ∀ x : obj, WildTurkey x →
    Eastern x ∨ (Osceola x ∨ (Goulds x ∨ (Merriams x ∨ (RioGrande x ∨ Ocellated x))))

theorem tom_is_ocellated : Ocellated Tom :=
by
  have h0 := types_exhaustive Tom T_wild
  have h1 := Or.resolve_left h0 N_eastern
  have h2 := Or.resolve_left h1 N_osceola
  have h3 := Or.resolve_left h2 N_goulds
  have h4 := Or.resolve_left h3 N_merriams
  exact Or.resolve_left h4 N_riogrande
</lean>

ANSWER: True
