<lean>
axiom obj : Type
axiom Djokovic : obj

axiom At : obj → Prop
axiom GrandSlamChampion : obj → Prop
axiom OscarNominatedActor : obj → Prop
axiom ProTennisPlayer : obj → Prop
axiom Celebrity : obj → Prop
axiom Athlete : obj → Prop
axiom WellPaid : obj → Prop
axiom Famous : obj → Prop
axiom LiveTaxHaven : obj → Prop

axiom A1 : ∀ x : obj, At x → (GrandSlamChampion x ∨ OscarNominatedActor x)
axiom A2 : ∀ x : obj, At x → GrandSlamChampion x → ProTennisPlayer x
axiom A3 : ∀ x : obj, At x → OscarNominatedActor x → Celebrity x
axiom A4 : ∀ x : obj, At x → ProTennisPlayer x → Athlete x
axiom A5 : ∀ x : obj, At x → Celebrity x → WellPaid x
axiom A6 : ∀ x : obj, At x → Athlete x → Famous x
axiom A7 : ∀ x : obj, At x → WellPaid x → LiveTaxHaven x
axiom A8 : At Djokovic
axiom A9 : (Famous Djokovic ∧ Athlete Djokovic) → WellPaid Djokovic

theorem djokovic_lives_in_tax_haven : LiveTaxHaven Djokovic := by
  have hAt : At Djokovic := A8
  have hdisj : GrandSlamChampion Djokovic ∨ OscarNominatedActor Djokovic := A1 Djokovic hAt
  cases hdisj with
  | inl hG =>
    have hPTP : ProTennisPlayer Djokovic := A2 Djokovic hAt hG
    have hAth : Athlete Djokovic := A4 Djokovic hAt hPTP
    have hFam : Famous Djokovic := A6 Djokovic hAt hAth
    have hWP : WellPaid Djokovic := A9 (And.intro hFam hAth)
    exact A7 Djokovic hAt hWP
  | inr hO =>
    have hCel : Celebrity Djokovic := A3 Djokovic hAt hO
    have hWP : WellPaid Djokovic := A5 Djokovic hAt hCel
    exact A7 Djokovic hAt hWP
</lean>

ANSWER: True
