<lean>
axiom obj : Type

-- Entities
axiom DiamondMine : obj
axiom RoderickStrong : obj
axiom CreedBrothers : obj
axiom IvyNile : obj
axiom Imperium : obj

-- Predicates/Relations
axiom WrestlingStable : obj → Prop
axiom Leads : obj → obj → Prop
axiom Includes : obj → obj → Prop
axiom FeudsWith : obj → obj → Prop

-- Given facts
axiom DM_stable : WrestlingStable DiamondMine
axiom Strong_leads_DM : Leads RoderickStrong DiamondMine
axiom DM_includes_Creed : Includes DiamondMine CreedBrothers
axiom DM_includes_Ivy : Includes DiamondMine IvyNile
axiom Imperium_feuds_DM : FeudsWith Imperium DiamondMine

-- Goal: Roderick Strong leads a professional wrestling stable
theorem strong_leads_stable : ∃ s : obj, WrestlingStable s ∧ Leads RoderickStrong s := by
  refine Exists.intro DiamondMine ?_
  exact And.intro DM_stable Strong_leads_DM
</lean>

ANSWER: True
