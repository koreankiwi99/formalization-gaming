axiom A4 : ∀ x : obj, Bee x → ¬ ReproMF x
axiom A5 : ∀ x : obj, QueenBee x → Bee x
axiom T_beeHarry : Bee Harry

-- Useful consequence: Harry cannot be an animal with a backbone
theorem not_animal_backbone_Harry : ¬ (Animal Harry ∧ Backbone Harry) :=
by
  intro hAB
  have hrep : ReproMF Harry := A2 Harry hAB
  have hnot : ¬ ReproMF Harry := A4 Harry T_beeHarry
  exact hnot hrep

-- If Harry were both a vertebrate and an animal with a backbone, the desired conclusion follows (by explosion)
theorem branch1 :
  (Vertebrate Harry ∧ Animal Harry ∧ Backbone Harry) → (¬Invertebrate Harry ∧ ¬QueenBee Harry) :=
by
  intro h
  have hv : Vertebrate Harry := And.left h
  have ha : Animal Harry := And.left (And.right h)
  have hb : Backbone Harry := And.right (And.right h)
  have hrep : ReproMF Harry := A2 Harry (And.intro ha hb)
  have hnot : ¬ ReproMF Harry := A4 Harry T_beeHarry
  have contra : False := hnot hrep
  have nInv : ¬ Invertebrate Harry := fun _ => contra
  have nQB  : ¬ QueenBee Harry  := fun _ => contra
  exact And.intro nInv nQB
