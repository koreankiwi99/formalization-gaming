<lean>
axiom obj : Type
axiom Yuri : obj

axiom Pro : obj → Prop           -- professional basketball player
axiom American : obj → Prop
axiom CanJump : obj → Prop       -- can do jump shots
axiom Leap : obj → Prop          -- leaps straight into the air
axiom Activate : obj → Prop      -- activate their leg muscles

-- Context axioms
axiom some_pro_not_american : ∃ x : obj, Pro x ∧ (American x → False)
axiom pro_jump : ∀ x : obj, Pro x → CanJump x
axiom jump_leap : ∀ x : obj, CanJump x → Leap x
axiom leap_activate : ∀ x : obj, Leap x → Activate x
axiom yuri_not_activate : Activate Yuri → False

-- We show the negation of the queried implication.
-- Queried implication: (¬ Leap Yuri) → (American Yuri ∧ Pro Yuri)
-- We avoid '¬' and write it as (Leap Yuri → False).
theorem yuri_statement_false :
  ((Leap Yuri → False) → (American Yuri ∧ Pro Yuri)) → False := by
  intro hImp
  -- From the chain Pro → CanJump → Leap → Activate and ¬Activate Yuri we get ¬Pro Yuri
  have not_pro : Pro Yuri → False := by
    intro hPro
    have hCJ : CanJump Yuri := pro_jump Yuri hPro
    have hL  : Leap Yuri := jump_leap Yuri hCJ
    have hA  : Activate Yuri := leap_activate Yuri hL
    exact yuri_not_activate hA
  -- From Leap → Activate and ¬Activate Yuri we get ¬Leap Yuri
  have not_leap : Leap Yuri → False := by
    intro hL
    have hA : Activate Yuri := leap_activate Yuri hL
    exact yuri_not_activate hA
  -- Apply the implication to get Pro Yuri, contradicting ¬Pro Yuri
  have hAP : American Yuri ∧ Pro Yuri := hImp not_leap
  exact not_pro hAP.right
</lean>

ANSWER: False
