axiom obj : Type
axiom Listed : obj → Prop
axiom ManyNeg : obj → Prop
axiom RatingGT4 : obj → Prop
axiom TakeOut : obj → Prop
axiom Popular : obj → Prop
axiom HPS : obj

axiom A1 : ∀ x : obj, Listed x → ¬ ManyNeg x
axiom A2 : ∀ x : obj, RatingGT4 x → Listed x
axiom A3 : ∃ x : obj, ¬ TakeOut x ∧ ManyNeg x
axiom A4 : ∀ x : obj, Popular x → RatingGT4 x
axiom A5 : RatingGT4 HPS ∨ Popular HPS

lemma rg4_HPS : RatingGT4 HPS := by
  cases A5 with
  | inl h => exact h
  | inr h => exact A4 HPS h

lemma not_manyneg_HPS : ¬ ManyNeg HPS := by
  have listed_HPS : Listed HPS := A2 HPS rg4_HPS
  exact A1 HPS listed_HPS

theorem goal : (TakeOut HPS ∧ ManyNeg HPS) → (RatingGT4 HPS ∧ ¬ TakeOut HPS) := by
  intro h
  apply And.intro
  · exact rg4_HPS
  · intro _
    exact not_manyneg_HPS h.right
</lean>

ANSWER: True

axiom obj : Type
axiom Listed : obj → Prop
axiom ManyNeg : obj → Prop
axiom RatingGT4 : obj → Prop
axiom TakeOut : obj → Prop
axiom Popular : obj → Prop
axiom HPS : obj

axiom A1 : ∀ x : obj, Listed x → ¬ ManyNeg x
axiom A2 : ∀ x : obj, RatingGT4 x → Listed x
axiom A3 : ∃ x : obj, ¬ TakeOut x ∧ ManyNeg x
axiom A4 : ∀ x : obj, Popular x → RatingGT4 x
axiom A5 : RatingGT4 HPS ∨ Popular HPS

theorem rg4_HPS : RatingGT4 HPS :=
  Or.elim A5 (fun h => h) (fun h => A4 HPS h)

theorem not_manyneg_HPS : ¬ ManyNeg HPS :=
  fun mn => (A1 HPS (A2 HPS rg4_HPS)) mn

theorem goal : (TakeOut HPS ∧ ManyNeg HPS) → (RatingGT4 HPS ∧ ¬ TakeOut HPS) :=
  fun h =>
    let mn : ManyNeg HPS := And.right h
    let contra : False := not_manyneg_HPS mn
    And.intro rg4_HPS (fun _ => False.elim contra)
</lean>
