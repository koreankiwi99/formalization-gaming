<lean>
axiom obj : Type
axiom James : obj

axiom WorkLibrary : obj → Prop
axiom PartTime : obj → Prop
axiom CS : obj → Prop
axiom Database : obj → Prop
axiom WithDavid : obj → Prop
axiom WorkLab : obj → Prop

-- None of the students who currently work in the library are from the CS department.
axiom R1 : ∀ x : obj, WorkLibrary x → ¬ CS x
-- All students with part-time university jobs currently work in the library.
axiom R2 : ∀ x : obj, PartTime x → WorkLibrary x
-- All students taking the database course are from CS.
axiom R3 : ∀ x : obj, Database x → CS x
-- All students taking a class with Professor David are taking the database course.
axiom R4 : ∀ x : obj, WithDavid x → Database x
-- All students working in the lab are taking a class with Professor David.
axiom R5 : ∀ x : obj, WorkLab x → WithDavid x
-- James is a student working in the lab.
axiom T1 : WorkLab James
-- James doesn't work in the lab or have a part-time job offered by the university.
axiom T2 : ¬ (WorkLab James ∨ PartTime James)

theorem james_not_parttime : ¬ PartTime James :=
  fun hpt =>
    have hlib : WorkLibrary James := R2 James hpt
    have hnotcs : ¬ CS James := R1 James hlib
    have hdavid : WithDavid James := R5 James T1
    have hdb : Database James := R4 James hdavid
    have hcs : CS James := R3 James hdb
    show False from hnotcs hcs
</lean>

ANSWER: True
