<lean>
axiom Person : Type
axiom Tom : Person

axiom GREApplicant : Person → Prop
axiom EconomicHardship : Person → Prop
axiom SingleParentFamily : Person → Prop
axiom FewResources : Person → Prop
axiom ParentOutOfWork : Person → Prop
axiom FinancialAid : Person → Prop

-- Rules from the context
axiom R1 : ∀ x : Person, GREApplicant x → EconomicHardship x → FinancialAid x
axiom R2 : ∀ x : Person, (SingleParentFamily x ∨ FewResources x) → EconomicHardship x

-- Facts about Tom
axiom T_single : SingleParentFamily Tom
axiom T_out_of_work_parent : ParentOutOfWork Tom
axiom T_few_resources : FewResources Tom
axiom T_applicant : GREApplicant Tom

theorem ets_provides_financial_aid_to_tom : FinancialAid Tom := by
  have hEH : EconomicHardship Tom := R2 Tom (Or.inl T_single)
  exact R1 Tom T_applicant hEH
</lean>

ANSWER: True
