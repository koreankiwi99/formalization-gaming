axiom Airport : Type

-- Rule: departure and arrival cannot be at the same airport

theorem susan_not_from_lga : ¬ From Susan LGA := by
  intro hfrom
  have hnotTo : ¬ To Susan LGA := no_same_airport Susan LGA hfrom
  exact hnotTo susan_to_lga
