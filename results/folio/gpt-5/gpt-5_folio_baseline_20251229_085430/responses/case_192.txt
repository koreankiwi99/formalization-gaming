<lean>
axiom obj : Type
axiom Happy : obj → Prop
axiom Sad : obj → Prop
-- Every event is either happy or sad (coverage)
axiom either_happy_or_sad : ∀ x : obj, Happy x ∨ Sad x
-- No event is both happy and sad (exclusivity)
axiom exclusive_happy_sad : ∀ x : obj, ¬ (Happy x ∧ Sad x)
-- At least one event is happy
axiom exists_happy : ∃ x : obj, Happy x

-- The statement "All events are sad" is false
theorem not_all_events_are_sad : ¬ (∀ x : obj, Sad x) := by
  intro allSad
  obtain ⟨x, hxHappy⟩ := exists_happy
  have hxSad : Sad x := allSad x
  have hnotboth : ¬ (Happy x ∧ Sad x) := exclusive_happy_sad x
  exact hnotboth ⟨hxHappy, hxSad⟩
</lean>

ANSWER: False
