-- From Mia_diff_Emma and Emma_fav, we can derive that Mia's favorite is not Summer.
theorem mia_not_summer : Favorite Mia ≠ Summer := by
  intro h
  have hME : Favorite Mia = Favorite Emma := Eq.trans h (Eq.symm Emma_fav)
  exact Mia_diff_Emma hME
