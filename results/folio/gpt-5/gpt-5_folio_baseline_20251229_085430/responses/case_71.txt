axiom obj : Type
axiom KO : obj

axiom Stock : obj → Prop
axiom Growth : obj → Prop
axiom RapidProfit : obj → Prop
axiom Suitable : obj → Prop
axiom Mature : obj → Prop

-- All growth stocks are bought to earn profits from rapid price appreciation.
axiom growth_implies_rapid : ∀ x : obj, Growth x → RapidProfit x
-- If a stock is bought to earn rapid profits, then it is not suitable for a retirement fund.
axiom rapid_implies_not_suitable : ∀ x : obj, RapidProfit x → ¬ Suitable x
-- Some stocks are growth stocks.
axiom some_stock_is_growth : ∃ x : obj, Stock x ∧ Growth x
-- All mature stocks are suitable for a retirement fund.
axiom mature_implies_suitable : ∀ x : obj, Mature x → Suitable x
-- KO is a mature stock.
axiom KO_mature : Mature KO

-- From the axioms, KO cannot be a growth stock (since KO is suitable).
theorem KO_not_growth : ¬ Growth KO := by
  intro hG
  have hRapid : RapidProfit KO := growth_implies_rapid KO hG
  have hNotSuitable : ¬ Suitable KO := rapid_implies_not_suitable KO hRapid
  have hSuitable : Suitable KO := mature_implies_suitable KO KO_mature
  exact hNotSuitable hSuitable

-- Therefore, KO cannot be both a stock and a growth stock.
theorem KO_not_stock_and_growth : ¬ (Stock KO ∧ Growth KO) := by
  intro h
  have hG : Growth KO := h.right
  exact KO_not_growth hG
