axiom obj : Type
axiom Rock : obj

axiom Pet : obj → Prop
axiom Mammal : obj → Prop
axiom Animal : obj → Prop
axiom Monkey : obj → Prop
axiom Bird : obj → Prop
axiom CanFly : obj → Prop
axiom HasWings : obj → Prop
axiom CanBreathe : obj → Prop

-- All of Peter's pets that are mammals are also animals.
axiom A1 : ∀ x : obj, Pet x ∧ Mammal x → Animal x
-- All monkeys are mammals.
axiom A2 : ∀ x : obj, Monkey x → Mammal x
-- Peter's pets are all either monkeys or birds.
axiom A3 : ∀ x : obj, Pet x → (Monkey x ∨ Bird x)
-- Peter's birds can fly.
axiom A4 : ∀ x : obj, Pet x ∧ Bird x → CanFly x
-- All animals that are Peter's pets can breathe.
axiom A5 : ∀ x : obj, Animal x ∧ Pet x → CanBreathe x
-- If Peter's pet can fly, then it has wings.
axiom A6 : ∀ x : obj, Pet x ∧ CanFly x → HasWings x
-- Rock is Peter's pet.
axiom A7 : Pet Rock
-- Rock can fly, or Rock is a bird, or Rock cannot breathe.
axiom A8 : CanFly Rock ∨ Bird Rock ∨ ¬CanBreathe Rock

-- Some consequences derivable from the axioms:

-- If Rock is a monkey, then Rock can breathe.
theorem rock_monkey_implies_breathe : Monkey Rock → CanBreathe Rock := by
  intro hM
  have hMam : Mammal Rock := A2 Rock hM
  have hAn : Animal Rock := A1 Rock (And.intro A7 hMam)
  exact A5 Rock (And.intro hAn A7)

-- From the given disjunction, Rock can fly or Rock cannot breathe.
theorem rock_fly_or_not_breathe : CanFly Rock ∨ ¬CanBreathe Rock := by
  cases A8 with
  | inl hFly => exact Or.inl hFly
  | inr h =>
    cases h with
    | inl hBird =>
      have hFly : CanFly Rock := A4 Rock (And.intro A7 hBird)
      exact Or.inl hFly
    | inr hNotBreathe =>
      exact Or.inr hNotBreathe

-- If Rock cannot breathe, then Rock is not a monkey.
theorem not_monkey_if_not_breathe : ¬CanBreathe Rock → ¬Monkey Rock := by
  intro hNB
  intro hM
  have hB : CanBreathe Rock := rock_monkey_implies_breathe hM
  exact hNB hB
