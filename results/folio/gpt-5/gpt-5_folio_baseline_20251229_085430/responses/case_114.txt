axiom obj : Type

-- Individuals
axiom Michael : obj
axiom Peter : obj
axiom Windy : obj

-- Predicates and relations
axiom Man : obj → Prop
axiom Taller : obj → obj → Prop
axiom Block : obj → obj → Prop
-- InClass x means "x is in Michael's class"
axiom InClass : obj → Prop
axiom Jump : obj → Prop
axiom Shooter : obj → Prop
axiom GreatShooter : obj → Prop

-- 1) If a man is taller than another man, the taller one can block the other's shooting.
axiom block_of_taller_men : ∀ x y : obj, Man x → Man y → Taller x y → Block x y

-- 2) Michael is a man who is taller than everyone else in his class.
axiom michael_man : Man Michael
axiom michael_taller_than_all_in_class : ∀ y : obj, InClass y → Taller Michael y

-- 3) Transitivity of "taller than".
axiom taller_trans : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- 4) Peter is a man who is taller than Michael.
axiom peter_man : Man Peter
axiom peter_taller_michael : Taller Peter Michael

-- 5) Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom michael_blocks_nonjumpers : ∀ y : obj, ¬ Jump y → Block Michael y

-- 6) Michael cannot block Windy's shooting.
axiom not_michael_blocks_windy : ¬ Block Michael Windy

-- 7) Every shooter who can jump when shooting is a great shooter.
axiom great_if_shooter_jumps : ∀ x : obj, Shooter x ∧ Jump x → GreatShooter x

-- Useful derived fact: Peter is taller than everyone in Michael's class.
theorem peter_taller_than_all_in_michaels_class : ∀ z : obj, InClass z → Taller Peter z :=
by
  intro z hz
  have h1 : Taller Peter Michael := peter_taller_michael
  have h2 : Taller Michael z := michael_taller_than_all_in_class z hz
  exact taller_trans Peter Michael z h1 h2

-- From (5) and (6) we can derive ¬¬(Jump Windy), but not Jump Windy constructively.
theorem not_not_jump_windy : ¬ ¬ Jump Windy :=
by
  intro hnj
  have hb : Block Michael Windy := michael_blocks_nonjumpers Windy hnj
  exact not_michael_blocks_windy hb
