<lean>
axiom Country : Type
axiom Event : Type

axiom UnitedStates : Country
axiom Tokyo : Event
axiom LastSummerOlympics : Event

axiom WonMostMedalsAt : Country → Event → Prop

-- The last summer Olympic games was in Tokyo
axiom last_in_tokyo : LastSummerOlympics = Tokyo

-- The United States won the most medals in Tokyo
axiom us_won_tokyo : WonMostMedalsAt UnitedStates Tokyo

-- Therefore, the United States won the most medals in the last summer Olympic games
theorem us_won_last : WonMostMedalsAt UnitedStates LastSummerOlympics := by
  simpa [last_in_tokyo] using us_won_tokyo
</lean>

ANSWER: True
