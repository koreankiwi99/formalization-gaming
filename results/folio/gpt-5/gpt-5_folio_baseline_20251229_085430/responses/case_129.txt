-- If John has a larger family, then either John has a well-attended wedding or invites friends to their ceremony.
axiom A6 : LargerFamily John → (WellAttended John ∨ InvitesFriends John)

-- Useful derived fact: anyone with a well-attended wedding is engaged.
theorem well_attended_implies_engaged : ∀ x : obj, WellAttended x → Engaged x := by

  have hio : InvitesOthers x := A3 x hx
  have hmw : MakesWeddingPlans x := A2 x hio
