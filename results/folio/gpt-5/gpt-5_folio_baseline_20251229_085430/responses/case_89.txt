axiom Person : Type
axiom Book : Type

axiom Reads : Person → Book → Prop
axiom GainsKnowledge : Person → Prop
axiom Smarter : Person → Prop

-- Rules from the context
axiom reading_gives_knowledge : ∀ p : Person, ∀ b : Book, Reads p b → GainsKnowledge p
axiom knowledge_implies_smarter : ∀ p : Person, GainsKnowledge p → Smarter p

-- Specific facts
axiom Harry : Person
axiom Walden : Book
axiom harry_read_walden : Reads Harry Walden

-- Deductions about Harry
theorem harry_gains_knowledge : GainsKnowledge Harry := by
  exact reading_gives_knowledge Harry Walden harry_read_walden

theorem harry_is_smarter : Smarter Harry := by
  exact knowledge_implies_smarter Harry harry_gains_knowledge
