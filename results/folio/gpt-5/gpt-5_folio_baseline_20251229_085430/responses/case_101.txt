axiom obj : Type
axiom Luke : obj

-- Predicates
axiom MultipleBirth : obj → Prop
axiom SpendTimeWithSiblings : obj → Prop
axiom HaveSibsBornTogether : obj → Prop
axiom ComplainsOftenAboutAnnoyingSiblings : obj → Prop
axiom LiveAtHome : obj → Prop
axiom LiveWithStrangers : obj → Prop
axiom Baby : obj → Prop

-- Rules from the context
axiom R1 : ∀ x : obj, MultipleBirth x → SpendTimeWithSiblings x
axiom R2 : ∀ x : obj, HaveSibsBornTogether x → MultipleBirth x
axiom R3 : ∀ x : obj, ComplainsOftenAboutAnnoyingSiblings x → HaveSibsBornTogether x
axiom R4 : ∀ x : obj, LiveAtHome x → ¬LiveWithStrangers x
axiom R5 : ∀ x : obj, SpendTimeWithSiblings x → LiveAtHome x

-- Luke-specific disjunction
axiom Luke_disj :
  (Baby Luke ∧ (MultipleBirth Luke ∧ LiveWithStrangers Luke)) ∨
  ((¬Baby Luke) ∧ (¬MultipleBirth Luke ∧ ¬LiveWithStrangers Luke))

-- The first disjunct is impossible (it would force both LiveWithStrangers and ¬LiveWithStrangers)
theorem not_first :
  ¬ (Baby Luke ∧ (MultipleBirth Luke ∧ LiveWithStrangers Luke)) := by
  intro h
  have hMB : MultipleBirth Luke := And.left (And.right h)
  have hLS : LiveWithStrangers Luke := And.right (And.right h)
  have hST : SpendTimeWithSiblings Luke := R1 Luke hMB
  have hHome : LiveAtHome Luke := R5 Luke hST
  have hNotLS : ¬LiveWithStrangers Luke := R4 Luke hHome
  exact hNotLS hLS

-- Therefore the second disjunct holds
theorem luke_second_disjunct :
  (¬Baby Luke) ∧ (¬MultipleBirth Luke ∧ ¬LiveWithStrangers Luke) := by
  have h := Luke_disj
  have nf := not_first
  cases h with
  | inl hA => exact False.elim (nf hA)
  | inr hB => exact hB

theorem luke_not_multiple : ¬MultipleBirth Luke :=
  And.left (And.right luke_second_disjunct)

theorem luke_not_live_with_strangers : ¬LiveWithStrangers Luke :=
  And.right (And.right luke_second_disjunct)
