<lean>
axiom Person : Type
axiom Taylor : Person

axiom ESR : Person → Prop -- eats salads regularly
axiom VCH : Person → Prop -- very conscious about health and eating habits
axiom GHC : Person → Prop -- grew up in health-conscious childhood home
axiom FDN : Person → Prop -- fulfill daily nutritional intakes
axiom DPW : Person → Prop -- disregard physical well-being
axiom VG  : Person → Prop -- visit the gym at least once a day

-- Rules
axiom R1 : ∀ x : Person, ESR x → VCH x
axiom R2 : ∀ x : Person, GHC x → ESR x
axiom R3 : ∀ x : Person, FDN x → GHC x
axiom R4 : ∀ x : Person, DPW x → ¬VCH x
axiom R5 : ∀ x : Person, VG x → FDN x

-- Taylor's given disjunction
axiom Taylor_case : (GHC Taylor ∧ DPW Taylor) ∨ (¬GHC Taylor ∧ ¬DPW Taylor)

theorem taylor_neither_ghc_nor_vg : ¬GHC Taylor ∧ ¬VG Taylor := by
  -- Show Taylor cannot both have GHC and DPW
  have not_both : ¬(GHC Taylor ∧ DPW Taylor) := by
    intro h
    have hG : GHC Taylor := h.left
    have hD : DPW Taylor := h.right
    have hE : ESR Taylor := R2 Taylor hG
    have hV : VCH Taylor := R1 Taylor hE
    have hNV : ¬VCH Taylor := R4 Taylor hD
    exact hNV hV
  -- From the given disjunction and impossibility of the first case, deduce the second case
  have hNotHCNotDPW : ¬GHC Taylor ∧ ¬DPW Taylor := by
    cases Taylor_case with
    | inl hgd =>
        exact False.elim (not_both hgd)
    | inr hnn =>
        exact hnn
  have notHC : ¬GHC Taylor := hNotHCNotDPW.left
  -- From ¬GHC and (VG → FDN → GHC), get ¬VG
  have notVG : ¬VG Taylor := by
    intro hVG
    have hGHC : GHC Taylor := R3 Taylor (R5 Taylor hVG)
    exact notHC hGHC
  exact And.intro notHC notVG
</lean>

ANSWER: True
