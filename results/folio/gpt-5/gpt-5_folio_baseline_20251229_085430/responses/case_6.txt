<lean>
axiom obj : Type

axiom Animal : obj → Prop
axiom Mammal : obj → Prop
axiom Human : obj → Prop

axiom OccursMPXV : obj → Prop
axiom HasMonkeypox : obj → Prop

axiom Fever : obj → Prop
axiom Headache : obj → Prop
axiom MusclePains : obj → Prop
axiom Tired : obj → Prop
axiom Flu : obj → Prop

-- When the Monkeypox virus occurs in a being, it may get Monkeypox.
axiom R1 : ∀ x : obj, OccursMPXV x → HasMonkeypox x

-- Monkeypox virus can occur in certain animals.
axiom A1 : ∃ x : obj, Animal x ∧ OccursMPXV x

-- Humans are mammals.
axiom A2 : ∀ x : obj, Human x → Mammal x

-- Mammals are animals.
axiom A3 : ∀ x : obj, Mammal x → Animal x

-- Symptoms of Monkeypox include fever, headache, muscle pains, and tiredness.
axiom A4 : ∀ x : obj, HasMonkeypox x → Fever x ∧ Headache x ∧ MusclePains x ∧ Tired x

-- People feel tired when they get the flu.
axiom A5 : ∀ x : obj, Flu x → Tired x

theorem there_is_an_animal : ∃ x : obj, Animal x := by
  obtain ⟨x, hx⟩ := A1
  exact ⟨x, And.left hx⟩
</lean>

ANSWER: True
