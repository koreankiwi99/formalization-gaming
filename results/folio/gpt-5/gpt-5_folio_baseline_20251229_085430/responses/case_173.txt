<lean>
axiom Person : Type
axiom Taylor : Person

axiom EatsSaladsRegularly : Person → Prop
axiom VeryConscious : Person → Prop
axiom GrewUpHealthConsciousHome : Person → Prop
axiom FulfillDailyNutritionalIntakes : Person → Prop
axiom DisregardPhysicalWellBeing : Person → Prop
axiom VisitsGymAtLeastOnceADay : Person → Prop

-- All people who eat salads regularly are very conscious.
axiom R1 : ∀ x : Person, EatsSaladsRegularly x → VeryConscious x
-- All people who grew up in health-conscious childhood homes eat salads regularly.
axiom R2 : ∀ x : Person, GrewUpHealthConsciousHome x → EatsSaladsRegularly x
-- All people who fulfill their daily nutritional intakes grew up in health-conscious childhood homes.
axiom R3 : ∀ x : Person, FulfillDailyNutritionalIntakes x → GrewUpHealthConsciousHome x
-- All people who disregard their physical well-being are not very conscious.
axiom R4 : ∀ x : Person, DisregardPhysicalWellBeing x → ¬ VeryConscious x
-- If people visit the gym at least once a day, then they fulfill their daily nutritional intakes.
axiom R5 : ∀ x : Person, VisitsGymAtLeastOnceADay x → FulfillDailyNutritionalIntakes x
-- Taylor either (grew up in a health-conscious home and disregards well-being) or (neither).
axiom R6 :
  (GrewUpHealthConsciousHome Taylor ∧ DisregardPhysicalWellBeing Taylor) ∨
  (¬GrewUpHealthConsciousHome Taylor ∧ ¬DisregardPhysicalWellBeing Taylor)

-- From growing up in a health-conscious home, one is very conscious.
lemma grewUp_implies_very : ∀ x : Person, GrewUpHealthConsciousHome x → VeryConscious x :=
  fun x hx => R1 x (R2 x hx)

-- It is impossible that Taylor both grew up health-conscious and disregards well-being.
lemma not_both_grewup_and_disregard :
  ¬ (GrewUpHealthConsciousHome Taylor ∧ DisregardPhysicalWellBeing Taylor) :=
by
  intro h
  have hg : GrewUpHealthConsciousHome Taylor := h.left
  have hd : DisregardPhysicalWellBeing Taylor := h.right
  have hv : VeryConscious Taylor := grewUp_implies_very Taylor hg
  have hnv : ¬ VeryConscious Taylor := R4 Taylor hd
  exact hnv hv

-- Therefore the second disjunct must hold: Taylor did neither.
lemma not_grewup_and_not_disregard :
  (¬ GrewUpHealthConsciousHome Taylor ∧ ¬ DisregardPhysicalWellBeing Taylor) :=
by
  have hdisj := R6
  cases hdisj with
  | inl h1 => exact False.elim (not_both_grewup_and_disregard h1)
  | inr h2 => exact h2

-- Conclude Taylor does not visit the gym at least once a day.
theorem taylor_not_visit : ¬ VisitsGymAtLeastOnceADay Taylor :=
by
  intro hvisit
  have hfulfill : FulfillDailyNutritionalIntakes Taylor := R5 Taylor hvisit
  have hgrow : GrewUpHealthConsciousHome Taylor := R3 Taylor hfulfill
  have hnotgrow : ¬ GrewUpHealthConsciousHome Taylor := (not_grewup_and_not_disregard).left
  exact hnotgrow hgrow
</lean>

ANSWER: False
