<lean>
axiom obj : Type
axiom Animal : obj → Prop
axiom Heck : obj → Prop
axiom Aurochs : obj → Prop
axiom BredBack : obj → Prop
axiom ArtificiallySelected : obj → Prop

-- Breeding back is a form of artificial selection
axiom R1 : ∀ x : obj, BredBack x → ArtificiallySelected x

-- Heck cattle were bred back in the 1920s
axiom T2 : ∃ x : obj, Heck x ∧ BredBack x

-- Heck cattle are animals
axiom R2 : ∀ x : obj, Heck x → Animal x

-- Aurochs are animals
axiom R3 : ∀ x : obj, Aurochs x → Animal x

theorem some_heck_artificial : ∃ x : obj, Heck x ∧ ArtificiallySelected x := by
  obtain ⟨x, hxH, hxB⟩ := T2
  exact ⟨x, hxH, R1 x hxB⟩
</lean>

ANSWER: True
