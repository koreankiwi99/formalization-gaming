<lean>
axiom obj : Type
axiom Peter : obj

axiom Spills : obj → Prop
axiom Tidy : obj → Prop
axiom Cleanly : obj → Prop
axiom ValuesOrder : obj → Prop
axiom FamilyPrioritizes : obj → Prop
axiom ClumsyFoodie : obj → Prop

axiom R1 : ∀ x : obj, Spills x → ¬ Tidy x
axiom R2 : ∀ x : obj, ClumsyFoodie x → Spills x
axiom R3 : ∀ x : obj, Cleanly x → Tidy x
axiom R4 : ∀ x : obj, ValuesOrder x → Cleanly x
axiom R5 : ∀ x : obj, FamilyPrioritizes x → ValuesOrder x

axiom HPeter : (Spills Peter ∧ Cleanly Peter) ∨ (¬ Spills Peter ∧ ¬ Cleanly Peter)

theorem peter_not_clumsy_or_family : ¬ (ClumsyFoodie Peter ∨ FamilyPrioritizes Peter) := by
  -- Spill ∧ Cleanly is impossible
  have notSC : ¬ (Spills Peter ∧ Cleanly Peter) := by
    intro h
    have ht : Tidy Peter := R3 Peter h.right
    have hnt : ¬ Tidy Peter := R1 Peter h.left
    exact hnt ht
  -- From the given disjunction and notSC, get ¬Spills ∧ ¬Cleanly
  have hneg : ¬ Spills Peter ∧ ¬ Cleanly Peter := by
    cases HPeter with
    | inl hsc => exact False.elim (notSC hsc)
    | inr hn => exact hn
  have hns : ¬ Spills Peter := hneg.left
  have hnc : ¬ Cleanly Peter := hneg.right
  -- Modus tollens from ClumsyFoodie → Spills and ¬Spills
  have notCF : ¬ ClumsyFoodie Peter := by
    intro hcf
    have hs : Spills Peter := R2 Peter hcf
    exact hns hs
  -- Modus tollens from Family → Values → Cleanly and ¬Cleanly
  have notFam : ¬ FamilyPrioritizes Peter := by
    intro hfam
    have hc : Cleanly Peter := R4 Peter (R5 Peter hfam)
    exact hnc hc
  -- Conclude ¬(ClumsyFoodie ∨ FamilyPrioritizes)
  intro h
  cases h with
  | inl hcf => exact False.elim (notCF hcf)
  | inr hfam => exact False.elim (notFam hfam)
</lean>

ANSWER: False
