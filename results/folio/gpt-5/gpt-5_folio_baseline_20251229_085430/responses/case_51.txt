<lean>
axiom obj : Type
axiom Romance : obj → Prop
axiom IndoEuropean : obj → Prop
axiom Related : obj → obj → Prop

axiom French : obj
axiom Spanish : obj
axiom German : obj
axiom Basque : obj

-- All Romance languages are Indo-European languages.
axiom A1 : ∀ x : obj, Romance x → IndoEuropean x

-- Romance languages form a family: any two Romance languages are related.
axiom A2 : ∀ x y : obj, Romance x → Romance y → Related x y

-- French and Spanish are Romance languages.
axiom A3 : Romance French
axiom A4 : Romance Spanish

-- German is related to Spanish.
axiom A5 : Related German Spanish

-- Basque is not related to any other language.
axiom A6 : ∀ y : obj, y ≠ Basque → ¬ Related Basque y

-- Spanish is a different language than Basque.
axiom A7 : Spanish ≠ Basque

theorem basque_not_romance : ¬ Romance Basque := by
  intro h
  have hrel : Related Basque Spanish := A2 Basque Spanish h A4
  have hnot : ¬ Related Basque Spanish := A6 Spanish A7
  exact hnot hrel
</lean>

ANSWER: False
