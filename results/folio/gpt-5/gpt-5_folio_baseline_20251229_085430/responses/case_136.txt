<lean>
axiom obj : Type
axiom Sucks : obj → Prop
axiom Plunger : obj → Prop
axiom Vacuum : obj → Prop
axiom Vampire : obj → Prop
axiom HouseholdAppliance : obj → Prop

-- Universals from the context
axiom R_plunger : ∀ x : obj, Plunger x → Sucks x
axiom R_vacuum : ∀ x : obj, Vacuum x → Sucks x
axiom R_vampire : ∀ x : obj, Vampire x → Sucks x

-- Specific objects and facts
axiom Space : obj
axiom Space_is_vacuum : Vacuum Space

axiom Duster : obj
axiom Duster_is_appliance : HouseholdAppliance Duster
axiom Duster_not_suck : ¬ Sucks Duster

-- Negation of the queried universal: there exists a household appliance that doesn't suck
theorem exists_appliance_not_suck : ∃ x : obj, HouseholdAppliance x ∧ ¬ Sucks x := by
  exact Exists.intro Duster (And.intro Duster_is_appliance Duster_not_suck)

-- Therefore, the universal statement "if something is a household appliance, it sucks" is false
theorem not_all_appliances_suck : ¬ (∀ x : obj, HouseholdAppliance x → Sucks x) := by
  intro h
  have hD : Sucks Duster := h Duster Duster_is_appliance
  exact Duster_not_suck hD
</lean>

ANSWER: False
