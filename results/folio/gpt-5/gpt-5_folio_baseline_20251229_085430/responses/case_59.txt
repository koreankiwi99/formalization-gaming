<lean>
axiom City : Type
axiom State : Type

axiom InState : City → State → Prop

-- Cities
axiom Billings : City
axiom Butte : City
axiom Helena : City
axiom Missoula : City
axiom WhiteSulphurSprings : City
axiom StPierre : City
axiom Bristol : City
axiom Texarkana : City
axiom Texhoma : City
axiom UnionCity : City

-- State
axiom Montana : State

-- Given facts
axiom T1 : InState Billings Montana
axiom T2 : InState Butte Montana
axiom T3 : InState Helena Montana
axiom T4 : InState Missoula Montana
axiom T5 : ∃ s : State, InState WhiteSulphurSprings s ∧ InState Butte s
axiom T6 : ¬ InState StPierre Montana

-- Uniqueness of state for cities except the listed exceptions
axiom unique_state_except :
  ∀ c : City, c ≠ Bristol ∧ c ≠ Texarkana ∧ c ≠ Texhoma ∧ c ≠ UnionCity →
    ∀ s1 s2 : State, InState c s1 → InState c s2 → s1 = s2

-- Goal: Montana is home to the city of Missoula.
theorem montana_home_missoula : InState Missoula Montana := by
  exact T4
</lean>

ANSWER: True
