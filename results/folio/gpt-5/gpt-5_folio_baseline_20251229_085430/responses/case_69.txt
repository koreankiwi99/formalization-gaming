<lean>
axiom Person : Type
axiom Show : Type

axiom MemberOfPappys : Person → Prop
axiom Starred : Person → Show → Prop
axiom PilotedOnBBCTwo : Show → Prop
axiom PilotedOnBBCThree : Show → Prop

axiom Badults : Show
-- Badults was piloted on BBC Three
axiom Badults_piloted_three : PilotedOnBBCThree Badults
-- Badults starred (at least one) member of Pappy's
axiom exists_member_starred_Badults : ∃ p : Person, MemberOfPappys p ∧ Starred p Badults

-- The claim: No members of Pappy's have starred in a show piloting on BBC Two or BBC Three.
-- We prove its negation from the context.
theorem claim_is_false :
  ¬ (∀ p : Person, MemberOfPappys p → ¬ (∃ s : Show, Starred p s ∧ (PilotedOnBBCTwo s ∨ PilotedOnBBCThree s))) := by
  intro h
  rcases exists_member_starred_Badults with ⟨p, hpMem, hpStar⟩
  have hex : ∃ s : Show, Starred p s ∧ (PilotedOnBBCTwo s ∨ PilotedOnBBCThree s) := by
    refine ⟨Badults, ?_⟩
    apply And.intro
    · exact hpStar
    · exact Or.inr Badults_piloted_three
  have hneg := h p hpMem
  exact hneg hex
</lean>

ANSWER: False
