<lean>
axiom obj : Type
axiom StrangerThings : obj
axiom BlackMirror : obj

axiom Netflix : obj → Prop
axiom Popular : obj → Prop
axiom Binge : obj → Prop        -- Karen binge-watches the show
axiom Download : obj → Prop     -- Karen downloads the show
axiom ShareLisa : obj → Prop    -- Karen shares the show with Lisa

-- Facts about the shows
axiom ST_netflix : Netflix StrangerThings
axiom ST_popular : Popular StrangerThings
axiom BM_netflix : Netflix BlackMirror

-- Rules
axiom R_pop_binge : ∀ x : obj, Netflix x → Popular x → Binge x
axiom R_binge_download : ∀ x : obj, Netflix x → (Binge x → Download x)
axiom R_download_binge : ∀ x : obj, Netflix x → (Download x → Binge x)
axiom BM_not_download : ¬ Download BlackMirror
axiom R_binge_share : ∀ x : obj, Netflix x → Binge x → ShareLisa x

theorem share_ST : ShareLisa StrangerThings := by
  have hB : Binge StrangerThings := R_pop_binge StrangerThings ST_netflix ST_popular
  exact R_binge_share StrangerThings ST_netflix hB
</lean>

ANSWER: True
