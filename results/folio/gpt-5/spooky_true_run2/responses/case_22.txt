<lean>
axiom obj : Type
axiom StrangerThings : obj
axiom BlackMirror : obj

axiom NetflixShow : obj → Prop
axiom Popular : obj → Prop
axiom Binge : obj → Prop
axiom Download : obj → Prop
axiom ShareWithLisa : obj → Prop

-- "Stranger Things" is a popular Netflix show.
axiom ST_netflix : NetflixShow StrangerThings
axiom ST_popular : Popular StrangerThings

-- If a Netflix show is popular, Karen will binge-watch it.
axiom popular_implies_binge_for_netflix : ∀ s : obj, NetflixShow s → Popular s → Binge s

-- If and only if Karen binge-watches a Netflix show, she will download it. (both directions)
axiom binge_iff_download_for_netflix_left  : ∀ s : obj, NetflixShow s → (Binge s → Download s)
axiom binge_iff_download_for_netflix_right : ∀ s : obj, NetflixShow s → (Download s → Binge s)

-- Karen does not download "Black Mirror."
axiom BM_not_download : ¬ Download BlackMirror

-- "Black Mirror" is a Netflix show.
axiom BM_netflix : NetflixShow BlackMirror

-- If Karen binge-watches a Netflix show, she will share it with Lisa.
axiom binge_implies_share_with_lisa : ∀ s : obj, NetflixShow s → Binge s → ShareWithLisa s

-- From the premises we can derive that "Black Mirror" is not popular.
theorem not_popular_BlackMirror : ¬ Popular BlackMirror :=
by
  intro hpop
  have hnf : NetflixShow BlackMirror := BM_netflix
  have hbinge : Binge BlackMirror := popular_implies_binge_for_netflix BlackMirror hnf hpop
  have hdownload : Download BlackMirror := binge_iff_download_for_netflix_left BlackMirror hnf hbinge
  exact BM_not_download hdownload
