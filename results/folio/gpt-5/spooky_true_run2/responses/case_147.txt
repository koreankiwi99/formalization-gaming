<lean>
axiom obj : Type
axiom Ekaterina : obj
axiom Vladimir : obj

axiom CanRegisterUS : obj → Prop
axiom CanParticipate2024 : obj → Prop
axiom USCitizen : obj → Prop
axiom TaiwaneseCitizen : obj → Prop
axiom RFOfficial : obj → Prop
axiom ManagerGazprom : obj → Prop

axiom A1 : ∀ x : obj, CanRegisterUS x → CanParticipate2024 x
axiom A2 : ∀ x : obj, USCitizen x → CanRegisterUS x
axiom A3 : ∀ x : obj, USCitizen x ∨ TaiwaneseCitizen x
axiom A4 : ∀ x : obj, RFOfficial x → ¬ TaiwaneseCitizen x
axiom A5 : ¬ TaiwaneseCitizen Vladimir
axiom A6 : ¬ ManagerGazprom Vladimir
axiom A7 : CanRegisterUS Ekaterina ∨ RFOfficial Ekaterina

theorem statement : CanParticipate2024 Ekaterina ∨ ManagerGazprom Vladimir := by
  left
  have h : CanRegisterUS Ekaterina ∨ RFOfficial Ekaterina := A7
  cases h with
  | inl hreg =>
      exact A1 Ekaterina hreg
  | inr hrf =>
      have notTW : ¬ TaiwaneseCitizen Ekaterina := A4 Ekaterina hrf
      have citOr : USCitizen Ekaterina ∨ TaiwaneseCitizen Ekaterina := A3 Ekaterina
      cases citOr with
      | inl hus =>
          exact A1 Ekaterina (A2 Ekaterina hus)
      | inr htw =>
          exact False.elim (notTW htw)
</lean>

ANSWER: True
