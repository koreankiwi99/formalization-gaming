<lean>
axiom obj : Type
axiom Taylor : obj

axiom EatsSaladsRegularly : obj → Prop
axiom VeryConsciousHealthEating : obj → Prop
axiom GrewUpHealthConsciousHome : obj → Prop
axiom DisregardPhysicalWellBeing : obj → Prop
axiom FulfillDailyNutritionalIntakes : obj → Prop
axiom VisitGymAtLeastOnceADay : obj → Prop

axiom A1 : ∀ x : obj, EatsSaladsRegularly x → VeryConsciousHealthEating x
axiom A2 : ∀ x : obj, GrewUpHealthConsciousHome x → EatsSaladsRegularly x
axiom A3 : ∀ x : obj, FulfillDailyNutritionalIntakes x → GrewUpHealthConsciousHome x
axiom A4 : ∀ x : obj, DisregardPhysicalWellBeing x → ¬ VeryConsciousHealthEating x
axiom A5 : ∀ x : obj, VisitGymAtLeastOnceADay x → FulfillDailyNutritionalIntakes x
axiom A6 : (GrewUpHealthConsciousHome Taylor ∧ DisregardPhysicalWellBeing Taylor) ∨ (¬ GrewUpHealthConsciousHome Taylor ∧ ¬ DisregardPhysicalWellBeing Taylor)

theorem grew_up_implies_very : ∀ x : obj, GrewUpHealthConsciousHome x → VeryConsciousHealthEating x := by
  intro x
  intro hg
  have hE : EatsSaladsRegularly x := A2 x hg
  exact A1 x hE

theorem visit_implies_grew : ∀ x : obj, VisitGymAtLeastOnceADay x → GrewUpHealthConsciousHome x := by
  intro x
  intro hv
  have hfd : FulfillDailyNutritionalIntakes x := A5 x hv
  exact A3 x hfd

theorem taylor_neither_grew_nor_visit : ¬ GrewUpHealthConsciousHome Taylor ∧ ¬ VisitGymAtLeastOnceADay Taylor := by
  have hnot_left : ¬ (GrewUpHealthConsciousHome Taylor ∧ DisregardPhysicalWellBeing Taylor) := by
    intro h
    have hg : GrewUpHealthConsciousHome Taylor := And.left h
    have hd : DisregardPhysicalWellBeing Taylor := And.right h
    have hv : VeryConsciousHealthEating Taylor := grew_up_implies_very Taylor hg
    have hnv : ¬ VeryConsciousHealthEating Taylor := A4 Taylor hd
    exact hnv hv
  have hpair : (¬ GrewUpHealthConsciousHome Taylor ∧ ¬ DisregardPhysicalWellBeing Taylor) :=
    Or.elim A6 (fun h => False.elim (hnot_left h)) (fun h => h)
  have hng : ¬ GrewUpHealthConsciousHome Taylor := And.left hpair
  have hnviz : ¬ VisitGymAtLeastOnceADay Taylor := by
    intro hv
    exact hng (visit_implies_grew Taylor hv)
  exact And.intro hng hnviz
</lean>

ANSWER: True
