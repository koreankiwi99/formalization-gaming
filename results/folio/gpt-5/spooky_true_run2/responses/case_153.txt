-- Premises
axiom A1 : ∀ x : obj, Hydrocarbon x → Organic x
axiom A2 : ∀ x : obj, Alkane x → Hydrocarbon x
axiom A3 : ∀ x : obj, Organic x → Chemical x
axiom A4 : ∀ x : obj, Organic x → ContainsCarbon x
axiom A5 : ∀ x : obj, Chemical x → ¬ OneElement x
axiom A6 : (Chemical mixture ∧ OneElement mixture) ∨ (¬ Chemical mixture ∧ ¬ OneElement mixture)

-- From the "both or neither" condition together with A5, we get that mixture is not a chemical compound
theorem not_chemical_mixture : ¬ Chemical mixture :=
  fun hchem =>
    Or.elim A6
      (fun hboth =>
        let hone := And.right hboth
        let hnotone := A5 mixture hchem
        show False from hnotone hone)
      (fun hneither =>
        (And.left hneither) hchem)

-- Alkanes are chemical (via the given chains of implications)
theorem alkane_implies_chemical : ∀ x : obj, Alkane x → Chemical x :=
  fun x hx => A3 x (A1 x (A2 x hx))

-- Hence mixture is not an alkane
theorem not_alkane_mixture : ¬ Alkane mixture :=
  fun halk => not_chemical_mixture (alkane_implies_chemical mixture halk)

-- Therefore the target conjunction cannot hold
theorem not_target : ¬ (Alkane mixture ∧ ContainsCarbon mixture) :=
  fun h => not_alkane_mixture h.left
