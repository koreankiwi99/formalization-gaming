axiom obj : Type

axiom Man : obj → Prop
axiom Taller : obj → obj → Prop
axiom CanBlock : obj → obj → Prop
axiom Jumps : obj → Prop
axiom Shooter : obj → Prop
axiom Great : obj → Prop

axiom Michael : obj
axiom Peter : obj
axiom Windy : obj

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom R1 : ∀ x y : obj, Man x → Man y → Taller x y → CanBlock x y

-- Michael is a man (the "taller than everyone else in his class" detail is not needed for the conclusion here).
axiom M_man : Man Michael

-- If x taller than y and y taller than z then x taller than z.
axiom taller_trans : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom P_man : Man Peter
axiom P_taller_M : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom M_blocks_if_not_jump : ∀ y : obj, ¬ Jumps y → CanBlock Michael y

-- Michael cannot block Windy's shooting.
axiom not_M_blocks_W : ¬ CanBlock Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom great_of_shooter_jump : ∀ x : obj, Shooter x → Jumps x → Great x

-- Windy is a shooter.
axiom W_shooter : Shooter Windy

open Classical

theorem windy_great : Great Windy := by
  have hJ : Jumps Windy := by
    by_contra hnj
    exact not_M_blocks_W (M_blocks_if_not_jump Windy hnj)
  exact great_of_shooter_jump Windy W_shooter hJ
</lean>

ANSWER: True

  let hJ : Jumps Windy :=
    match Classical.em (Jumps Windy) with
    | Or.inl h => h
