<lean>
axiom obj : Type
axiom Taylor : obj

axiom EatsSaladsRegularly : obj → Prop
axiom VeryConscious : obj → Prop
axiom GrewUpHealthConsciousHome : obj → Prop
axiom FulfillDailyNutrition : obj → Prop
axiom DisregardWellbeing : obj → Prop
axiom VisitGymDaily : obj → Prop

-- 1. All people who eat salads regularly are very conscious about their health and eating habits.
axiom A1 : ∀ x : obj, EatsSaladsRegularly x → VeryConscious x
-- 2. All people who grew up in health-conscious childhood homes eat salads regularly.
axiom A2 : ∀ x : obj, GrewUpHealthConsciousHome x → EatsSaladsRegularly x
-- 3. All people who fulfill their daily nutritional intakes grew up in health-conscious childhood homes.
axiom A3 : ∀ x : obj, FulfillDailyNutrition x → GrewUpHealthConsciousHome x
-- 4. All people who disregard their physical well-being are not very conscious about their health and eating habits.
axiom A4 : ∀ x : obj, DisregardWellbeing x → ¬ VeryConscious x
-- 5. If people visit the gym at least once a day, then they always fulfill their daily nutritional intakes.
axiom A5 : ∀ x : obj, VisitGymDaily x → FulfillDailyNutrition x

-- Taylor either grew up in a health-conscious childhood home and disregards her physical well-being, or she did neither.
axiom Tdisj :
  (GrewUpHealthConsciousHome Taylor ∧ DisregardWellbeing Taylor) ∨
  (¬ GrewUpHealthConsciousHome Taylor ∧ ¬ DisregardWellbeing Taylor)

-- From the universal statements, the first disjunct is impossible.
theorem not_case1 : ¬ (GrewUpHealthConsciousHome Taylor ∧ DisregardWellbeing Taylor) := by
  intro h
  have hG : GrewUpHealthConsciousHome Taylor := And.left h
  have hD : DisregardWellbeing Taylor := And.right h
  have hEat : EatsSaladsRegularly Taylor := A2 Taylor hG
  have hVC : VeryConscious Taylor := A1 Taylor hEat
  have hNotVC : ¬ VeryConscious Taylor := A4 Taylor hD
  exact hNotVC hVC

-- Therefore, the second disjunct must hold: Taylor did neither.
theorem taylor_neither : ¬ GrewUpHealthConsciousHome Taylor ∧ ¬ DisregardWellbeing Taylor :=
  Or.resolve_left Tdisj not_case1
