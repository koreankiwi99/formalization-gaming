axiom P5 : ∀ x : obj, Chemical x → ¬ OneElement x
axiom P6 : (Chemical Mixture ∧ OneElement Mixture) ∨ (¬ Chemical Mixture ∧ ¬ OneElement Mixture)

-- From P5 and P6 we can deduce that the first disjunct in P6 is impossible,
-- so Mixture is neither a chemical compound nor contains only one element.
theorem mixture_not_chemical_and_not_one : ¬ Chemical Mixture ∧ ¬ OneElement Mixture :=
by
  have h := P6
  cases h with
  | inl h1 =>
      have hc : Chemical Mixture := And.left h1
      have ho : OneElement Mixture := And.right h1
      have hno : ¬ OneElement Mixture := P5 Mixture hc
      have contra : False := hno ho
      exact False.elim contra
  | inr h2 =>
      exact h2
