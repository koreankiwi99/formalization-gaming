<lean>
axiom obj : Type
axiom YaleUniversity : obj
axiom PrivateIvyLeagueResearchUniversity : obj → Prop
axiom MovedToNewHaven : obj → Prop

axiom H1 : PrivateIvyLeagueResearchUniversity YaleUniversity
axiom H2 : MovedToNewHaven YaleUniversity

theorem exists_private_ivy_league_research_university_moved_to_new_haven :
  ∃ x : obj, PrivateIvyLeagueResearchUniversity x ∧ MovedToNewHaven x := by
  apply Exists.intro YaleUniversity
  apply And.intro
  · exact H1
  · exact H2
</lean>

ANSWER: True
