<lean>
axiom Person : Type
axiom Book : Type

axiom Harry : Person
axiom Walden : Book

axiom Reads : Person → Book → Prop
axiom GainsKnowledge : Person → Prop
axiom SmarterThanBefore : Person → Prop

-- Optional context facts (not needed for the proof)
axiom ContainsTonsOfKnowledge : Book → Prop
axiom AllBooksContainKnowledge : ∀ b : Book, ContainsTonsOfKnowledge b

-- Core rules
axiom ReadImpliesGain : ∀ p : Person, ∀ b : Book, Reads p b → GainsKnowledge p
axiom GainImpliesSmarter : ∀ p : Person, GainsKnowledge p → SmarterThanBefore p

-- Fact: Harry read Walden
axiom HarryReadWalden : Reads Harry Walden

theorem harry_smarter : SmarterThanBefore Harry := by
  have hGain : GainsKnowledge Harry := ReadImpliesGain Harry Walden HarryReadWalden
  exact GainImpliesSmarter Harry hGain
</lean>

ANSWER: True
