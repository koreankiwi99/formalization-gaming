axiom PicurisMountains : obj
axiom HardingPegmatiteMine : obj
axiom Juan : obj

axiom InNewMexico : obj → Prop
axiom InTexas : obj → Prop
axiom Visited : obj → obj → Prop
axiom Mine : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop

-- Premises from the text
axiom PM_is_mountain : MountainRange PicurisMountains
axiom PM_in_NM_or_TX : InNewMexico PicurisMountains ∨ InTexas PicurisMountains
axiom Juan_visited_PM : Visited Juan PicurisMountains
axiom HPM_is_mine : Mine HardingPegmatiteMine
axiom HPM_located_PM : LocatedIn HardingPegmatiteMine PicurisMountains
axiom HPM_donated : Donated HardingPegmatiteMine
axiom No_TX_mountain_with_donated_mine :
  ∀ r : obj, MountainRange r → InTexas r → ¬ (∃ m : obj, Mine m ∧ LocatedIn m r ∧ Donated m)

-- From the donated mine located in Picuris, deduce Picuris is not in Texas
theorem not_in_Texas_Picuris : ¬ InTexas PicurisMountains := by
  intro htx
  have hno : ¬ (∃ m : obj, Mine m ∧ LocatedIn m PicurisMountains ∧ Donated m) :=
    No_TX_mountain_with_donated_mine PicurisMountains PM_is_mountain htx
  have hex : ∃ m : obj, Mine m ∧ LocatedIn m PicurisMountains ∧ Donated m := by
    refine Exists.intro HardingPegmatiteMine ?_
    exact And.intro HPM_is_mine (And.intro HPM_located_PM HPM_donated)
  exact hno hex

-- Therefore, from "in NM or TX" and not in TX, conclude Picuris is in New Mexico
theorem in_NewMexico_Picuris : InNewMexico PicurisMountains := by
  cases PM_in_NM_or_TX with
  | inl hnm => exact hnm
  | inr htx =>
      have : False := not_in_Texas_Picuris htx
      exact False.elim this
