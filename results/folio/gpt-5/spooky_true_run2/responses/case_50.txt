<lean>
axiom obj : Type

axiom NineteenEightyFour : obj

axiom DigitalMedia : obj → Prop
axiom AnalogMedia : obj → Prop
axiom PrintedText : obj → Prop
axiom StreamingService : obj → Prop
axiom HardcoverBook : obj → Prop

-- No digital media are analog.
axiom H1 : ∀ x : obj, DigitalMedia x → ¬ AnalogMedia x
-- Every printed text is analog media.
axiom H2 : ∀ x : obj, PrintedText x → AnalogMedia x
-- All streaming services are digital media.
axiom H3 : ∀ x : obj, StreamingService x → DigitalMedia x
-- If an object is a hardcover book, then it is printed text.
axiom H4 : ∀ x : obj, HardcoverBook x → PrintedText x
-- If 1984 is a streaming service, then 1984 is a hardcover book.
axiom H5 : StreamingService NineteenEightyFour → HardcoverBook NineteenEightyFour

theorem nineteenEightyFour_not_streaming : ¬ StreamingService NineteenEightyFour := by
  intro hS
  have hD : DigitalMedia NineteenEightyFour := H3 NineteenEightyFour hS
  have hH : HardcoverBook NineteenEightyFour := H5 hS
  have hP : PrintedText NineteenEightyFour := H4 NineteenEightyFour hH
  have hA : AnalogMedia NineteenEightyFour := H2 NineteenEightyFour hP
  have hNotA : ¬ AnalogMedia NineteenEightyFour := H1 NineteenEightyFour hD
  exact hNotA hA
</lean>

ANSWER: True
