<lean>
axiom obj : Type

axiom Show : obj → Prop
axiom AirsMon8 : obj → Prop
axiom GivesRoses : obj → Prop
axiom EpisodeOfBachelor : obj → Prop
axiom PortraysReal : obj → Prop
axiom RealityTV : obj → Prop

axiom BreakingBad : obj

-- Some show airing at 8 pm on Monday gives out roses on TV.
axiom Ex_some_show : ∃ s : obj, Show s ∧ AirsMon8 s ∧ GivesRoses s

-- If a show gives out roses on TV, then the show is an episode of The Bachelor.
axiom R_gives_episode : ∀ x : obj, GivesRoses x → EpisodeOfBachelor x

-- The Bachelor portrays the lives of real people (applied to its episodes).
axiom R_episode_portrays : ∀ x : obj, EpisodeOfBachelor x → PortraysReal x

-- All shows portraying the lives of real people are reality TV shows.
axiom R_real_implies_reality : ∀ x : obj, Show x → PortraysReal x → RealityTV x

-- Breaking Bad is a show.
axiom BB_show : Show BreakingBad

-- Breaking Bad is not a reality TV show.
axiom BB_not_reality : ¬RealityTV BreakingBad

theorem breakingbad_roses_then_monday8 : GivesRoses BreakingBad → AirsMon8 BreakingBad := by
  have not_roses : ¬GivesRoses BreakingBad := by
    intro gr
    have ep : EpisodeOfBachelor BreakingBad := R_gives_episode BreakingBad gr
    have pr : PortraysReal BreakingBad := R_episode_portrays BreakingBad ep
    have rt : RealityTV BreakingBad := R_real_implies_reality BreakingBad BB_show pr
    exact BB_not_reality rt
  intro gr
  exact False.elim (not_roses gr)
</lean>

ANSWER: True
