<lean>
axiom obj : Type

-- Entities
axiom PM : obj          -- Picuris Mountains
axiom JdO : obj         -- Juan de Onate
axiom HPM : obj         -- Harding Pegmatite Mine

-- Predicates and relations
axiom MountainRange : obj → Prop
axiom InNewMexico : obj → Prop
axiom InTexas : obj → Prop
axiom Visited : obj → obj → Prop
axiom Mine : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop

-- Premises
axiom PM_is_range : MountainRange PM
axiom PM_in_NM_or_TX : InNewMexico PM ∨ InTexas PM
axiom Onate_visited_PM : Visited JdO PM
axiom HPM_is_mine : Mine HPM
axiom HPM_loc_PM : LocatedIn HPM PM
axiom HPM_donated : Donated HPM
axiom no_TX_range_with_donated_mine :
  ∀ r m, MountainRange r → InTexas r → Mine m → LocatedIn m r → Donated m → False

-- Conclusion: Juan de Onate visited a mountain range in New Mexico
theorem Onate_visited_some_NM_range :
  ∃ r : obj, MountainRange r ∧ InNewMexico r ∧ Visited JdO r := by
  -- Show PM is not in Texas using the donated mine located there
  have not_in_TX_PM : ¬ InTexas PM := by
    intro htx
    exact no_TX_range_with_donated_mine PM HPM PM_is_range htx HPM_is_mine HPM_loc_PM HPM_donated
  -- From (InNewMexico PM ∨ InTexas PM) and ¬InTexas PM, deduce InNewMexico PM
  have PM_in_NM : InNewMexico PM := by
    cases PM_in_NM_or_TX with
    | inl nm => exact nm
    | inr tx => exact False.elim (not_in_TX_PM tx)
  -- Witness PM
  exact Exists.intro PM (And.intro PM_is_range (And.intro PM_in_NM Onate_visited_PM))
</lean>

ANSWER: True
