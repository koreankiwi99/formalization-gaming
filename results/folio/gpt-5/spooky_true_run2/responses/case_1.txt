<lean>
axiom obj : Type
axiom Bonnie : obj

axiom InClub : obj → Prop
axiom PerformOften : obj → Prop
axiom AttendEngaged : obj → Prop
axiom Inactive : obj → Prop
axiom Chaperone : obj → Prop
axiom Student : obj → Prop
axiom YoungChild : obj → Prop
axiom Teenager : obj → Prop
axiom WishFurther : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, InClub x ∧ PerformOften x → AttendEngaged x
axiom A2 : ∀ x : obj, InClub x → PerformOften x ∨ Inactive x
axiom A3 : ∀ x : obj, InClub x ∧ Chaperone x → ¬ Student x
axiom A4 : ∀ x : obj, InClub x ∧ Inactive x → Chaperone x
axiom A5 : ∀ x : obj, InClub x ∧ (YoungChild x ∨ Teenager x) ∧ WishFurther x → Student x

axiom BonnieIn : InClub Bonnie
axiom BonnieEquiv : (AttendEngaged Bonnie ∧ Student Bonnie) ∨ (¬ AttendEngaged Bonnie ∧ ¬ Student Bonnie)

theorem bonnie_statement :
  (((YoungChild Bonnie ∨ Teenager Bonnie) ∧ WishFurther Bonnie ∧ Chaperone Bonnie) ∨ ¬((YoungChild Bonnie ∨ Teenager Bonnie) ∧ WishFurther Bonnie))
  → (Student Bonnie ∨ Inactive Bonnie) := by
  intro _
  have hdisj := A2 Bonnie BonnieIn
  cases hdisj
  case inl hperf =>
    have hatt : AttendEngaged Bonnie := A1 Bonnie (And.intro BonnieIn hperf)
    have hiff : AttendEngaged Bonnie ↔ Student Bonnie := by
      cases BonnieEquiv
      case inl hboth =>
        exact Iff.intro (fun _ => hboth.right) (fun _ => hboth.left)
      case inr hbothneg =>
        have hnp : ¬ AttendEngaged Bonnie := hbothneg.left
        have hnq : ¬ Student Bonnie := hbothneg.right
        exact Iff.intro
          (fun hp => False.elim (hnp hp))
          (fun hq => False.elim (hnq hq))
    have hstu : Student Bonnie := (Iff.mp hiff) hatt
    exact Or.inl hstu
  case inr hinactive =>
    exact Or.inr hinactive
</lean>
ANSWER: True
