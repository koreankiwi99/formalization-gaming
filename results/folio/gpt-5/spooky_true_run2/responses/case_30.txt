<lean>
axiom obj : Type

-- Entities
axiom Djokovic : obj

-- Predicates
axiom AtMixer : obj → Prop
axiom GrandSlamChampion : obj → Prop
axiom OscarNominatedActor : obj → Prop
axiom ProTennisPlayer : obj → Prop
axiom Celebrity : obj → Prop
axiom Athlete : obj → Prop
axiom WellPaid : obj → Prop
axiom Famous : obj → Prop
axiom LivesInTaxHaven : obj → Prop

-- Premises
-- 1. Everyone at the mixer is a Grand Slam champion or an Oscar-nominated actor.
axiom A1 : ∀ x : obj, AtMixer x → (GrandSlamChampion x ∨ OscarNominatedActor x)

-- 2. Every Grand Slam champion at the mixer is a professional tennis player.
axiom A2 : ∀ x : obj, AtMixer x → GrandSlamChampion x → ProTennisPlayer x

-- 3. All Oscar-nominated actors at the mixer are celebrities.
axiom A3 : ∀ x : obj, AtMixer x → OscarNominatedActor x → Celebrity x

-- 4. All professional tennis players at the mixer are athletes.
axiom A4 : ∀ x : obj, AtMixer x → ProTennisPlayer x → Athlete x

-- 5. If a person at the mixer is a celebrity, then they are well paid.
axiom A5 : ∀ x : obj, AtMixer x → Celebrity x → WellPaid x

-- 6. If a person at the mixer is an athlete, then they are famous.
axiom A6 : ∀ x : obj, AtMixer x → Athlete x → Famous x

-- 7. All well-paid people at the mixer live in tax havens.
axiom A7 : ∀ x : obj, AtMixer x → WellPaid x → LivesInTaxHaven x

-- 8. Djokovic is at the mixer; special implication about famous athlete -> well-paid.
axiom DJ_at : AtMixer Djokovic
axiom DJ_special : Famous Djokovic ∧ Athlete Djokovic → WellPaid Djokovic

-- Some derived facts we can prove (but they do not establish that Djokovic is a Grand Slam champion).

theorem djokovic_GS_or_Oscar : GrandSlamChampion Djokovic ∨ OscarNominatedActor Djokovic :=
  A1 Djokovic DJ_at

theorem djokovic_GS_of_not_Oscar : ¬ OscarNominatedActor Djokovic → GrandSlamChampion Djokovic := by
  intro hno
  have h := A1 Djokovic DJ_at
  cases h with
  | inl hgs => exact hgs
  | inr hoscar => exact False.elim (hno hoscar)

theorem djokovic_WellPaid_if_GS : GrandSlamChampion Djokovic → WellPaid Djokovic := by
  intro hgs
  have hpro := A2 Djokovic DJ_at hgs
  have hath := A4 Djokovic DJ_at hpro
  have hfam := A6 Djokovic DJ_at hath
  have hconj : Famous Djokovic ∧ Athlete Djokovic := And.intro hfam hath
  exact DJ_special hconj

theorem djokovic_WellPaid_if_Oscar : OscarNominatedActor Djokovic → WellPaid Djokovic := by
  intro hoscar
  have hcel := A3 Djokovic DJ_at hoscar
  exact A5 Djokovic DJ_at hcel

theorem djokovic_TaxHaven_if_GS : GrandSlamChampion Djokovic → LivesInTaxHaven Djokovic := by
  intro hgs
  have hwp := djokovic_WellPaid_if_GS hgs
  exact A7 Djokovic DJ_at hwp

theorem djokovic_TaxHaven_if_Oscar : OscarNominatedActor Djokovic → LivesInTaxHaven Djokovic := by
  intro hoscar
  have hwp := djokovic_WellPaid_if_Oscar hoscar
  exact A7 Djokovic DJ_at hwp
