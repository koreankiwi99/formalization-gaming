<lean>
axiom obj : Type
axiom George : obj

axiom LikesMusic : obj → Prop
axiom WantsCompose : obj → Prop
axiom HasAccessProgram : obj → Prop
axiom CanCompose : obj → Prop
axiom WillCompose : obj → Prop

-- Premises
axiom P1 : LikesMusic George → WantsCompose George
axiom P2 : HasAccessProgram George → CanCompose George
axiom P3 : CanCompose George ∧ WantsCompose George → WillCompose George

-- Conclusion
theorem george_will_compose :
  (LikesMusic George ∧ HasAccessProgram George) → WillCompose George :=
by
  intro h
  have hm : LikesMusic George := And.left h
  have ha : HasAccessProgram George := And.right h
  have w : WantsCompose George := P1 hm
  have c : CanCompose George := P2 ha
  have hwc : CanCompose George ∧ WantsCompose George := And.intro c w
  exact P3 hwc
</lean>

ANSWER: True
