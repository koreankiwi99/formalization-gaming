<lean>
axiom obj : Type
axiom Hannah : obj

axiom At : obj → Prop
axiom Work : obj → Prop
axiom Need : obj → Prop
axiom OrderTakeout : obj → Prop
axiom EnjoyDining : obj → Prop
axiom Picky : obj → Prop
axiom SpendTimeDining : obj → Prop

axiom R1 : ∀ x : obj, At x ∧ Work x → Need x
axiom R2 : ∀ x : obj, At x ∧ OrderTakeout x → Work x
axiom R3 : ∀ x : obj, At x → (OrderTakeout x ∨ EnjoyDining x)
axiom R4 : ∀ x : obj, At x ∧ EnjoyDining x → ¬ Picky x
axiom R5 : ∀ x : obj, At x ∧ EnjoyDining x → SpendTimeDining x

axiom T1 : At Hannah
axiom T2 : Work Hannah ∧ (Need Hannah → (¬ Picky Hannah ∧ ¬ Need Hannah))

theorem hannah_conclusion :
  At Hannah ∧ ¬ Picky Hannah ∧ SpendTimeDining Hannah := by
  have atH : At Hannah := T1
  have disj : OrderTakeout Hannah ∨ EnjoyDining Hannah := R3 Hannah atH
  have enjoyH : EnjoyDining Hannah := by
    cases disj with
    | inl hOrder =>
        have workH : Work Hannah := R2 Hannah ⟨atH, hOrder⟩
        have needH : Need Hannah := R1 Hannah ⟨atH, workH⟩
        have cond : Need Hannah → (¬ Picky Hannah ∧ ¬ Need Hannah) := And.right T2
        have both := cond needH
        have notNeed : ¬ Need Hannah := And.right both
        exact False.elim (notNeed needH)
    | inr hEnjoy =>
        exact hEnjoy
  have notPickyH : ¬ Picky Hannah := R4 Hannah ⟨atH, enjoyH⟩
  have spendH : SpendTimeDining Hannah := R5 Hannah ⟨atH, enjoyH⟩
  exact And.intro atH (And.intro notPickyH spendH)
</lean>

ANSWER: True
