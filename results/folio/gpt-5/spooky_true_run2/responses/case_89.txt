<lean>
axiom obj : Type

axiom Person : obj → Prop
axiom Book : obj → Prop
axiom Reads : obj → obj → Prop
axiom GainedKnowledge : obj → Prop
axiom Smarter : obj → Prop

axiom R_read_gain : ∀ x y : obj, Person x → Book y → Reads x y → GainedKnowledge x
axiom R_gain_smarter : ∀ x : obj, GainedKnowledge x → Smarter x

axiom Harry : obj
axiom Walden : obj

axiom Harry_is_person : Person Harry
axiom Walden_is_book : Book Walden
axiom Harry_read_Walden : Reads Harry Walden

theorem smarter_person_has_gained_knowledge : ∃ x : obj, Smarter x ∧ GainedKnowledge x := by
  have gh : GainedKnowledge Harry := R_read_gain Harry Walden Harry_is_person Walden_is_book Harry_read_Walden
  have sh : Smarter Harry := R_gain_smarter Harry gh
  exact Exists.intro Harry (And.intro sh gh)
</lean>

ANSWER: True
