axiom obj : Type
axiom Coco : obj

axiom HighWTA : obj → Prop
axiom ActiveMajor : obj → Prop
axiom LostIga : obj → Prop
axiom LostRafa : obj → Prop
axiom Female : obj → Prop
axiom Male : obj → Prop
axiom AtRG2022 : obj → Prop

-- If someone is ranked highly by the WTA, then they are one of the most active players in major tennis.
axiom A1 : ∀ x : obj, HighWTA x → ActiveMajor x
-- Everyone who lost to Iga Świątek at Roland Garros 2022 is ranked highly by the WTA.
axiom A2 : ∀ x : obj, LostIga x → HighWTA x
-- All female tennis players at Roland Garros 2022 lost to Iga Świątek.
axiom A3 : ∀ x : obj, Female x → LostIga x
-- All tennis players at Roland Garros 2022 are either female or male.
axiom A4 : ∀ x : obj, AtRG2022 x → Female x ∨ Male x
-- All male tennis players at Roland Garros in 2022 lost to Rafael Nadal.
axiom A5 : ∀ x : obj, Male x → LostRafa x
-- If Coco Gauff is ranked highly by the WTA or lost to Rafael Nadal, then Coco Gauff is not a male tennis player at Roland Garros 2022.
axiom A6 : (HighWTA Coco ∨ LostRafa Coco) → ¬ Male Coco
-- Coco Gauff is at Roland Garros 2022.
axiom A7 : AtRG2022 Coco

-- From the premises we can derive that Coco both lost to Iga and is one of the most active players.
theorem coco_lost_and_active : LostIga Coco ∧ ActiveMajor Coco := by
  have hfm : Female Coco ∨ Male Coco := A4 Coco A7
  have hdisj : HighWTA Coco ∨ LostRafa Coco := by
    cases hfm with
    | inl hf =>
      have hli : LostIga Coco := A3 Coco hf
      have hh : HighWTA Coco := A2 Coco hli
      exact Or.inl hh
    | inr hm =>
      have lr : LostRafa Coco := A5 Coco hm
      exact Or.inr lr
  have hnotmale : ¬ Male Coco := A6 hdisj
  have hfem : Female Coco := by
    cases hfm with
    | inl hf => exact hf
    | inr hm => exact False.elim (hnotmale hm)
  have hli : LostIga Coco := A3 Coco hfem
  have hh : HighWTA Coco := A2 Coco hli
  have ha : ActiveMajor Coco := A1 Coco hh
  exact And.intro hli ha
