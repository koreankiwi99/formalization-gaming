<lean>
axiom obj : Type
axiom Rabbit : obj → Prop
axiom Turtle : obj → Prop
axiom Squirrel : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop
axiom Spotted : obj → Prop
axiom Cute : obj → Prop
axiom Rockie : obj

axiom A1 : ∀ x : obj, Rabbit x ∧ Spotted x → Cute x
axiom A2 : ∃ x : obj, Turtle x ∧ Spotted x
axiom A3 : ∀ x : obj, Spotted x → Rabbit x ∨ Squirrel x
axiom A4 : ∀ x : obj, Skittish x → ¬ Calm x
axiom A5 : ∀ x : obj, Squirrel x ∧ Spotted x → Skittish x
axiom A6 : Spotted Rockie ∧ Calm Rockie

theorem rockie_cute_or_skittish :
  (¬ (Turtle Rockie ∧ Squirrel Rockie)) → (Cute Rockie ∨ Skittish Rockie) := by
  intro _
  have SR : Spotted Rockie := And.left A6
  have CR : Calm Rockie := And.right A6
  have notS : ¬ Squirrel Rockie := by
    intro hS
    have hSk : Skittish Rockie := A5 Rockie (And.intro hS SR)
    have hNotC : ¬ Calm Rockie := A4 Rockie hSk
    exact hNotC CR
  have hEither : Rabbit Rockie ∨ Squirrel Rockie := A3 Rockie SR
  have hCute : Cute Rockie := by
    cases hEither with
    | inl hRab =>
        exact A1 Rockie (And.intro hRab SR)
    | inr hSq =>
        exact False.elim (notS hSq)
  exact Or.inl hCute
</lean>

ANSWER: True
