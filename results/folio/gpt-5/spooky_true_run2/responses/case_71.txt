axiom obj : Type
axiom Stock : obj → Prop
axiom Growth : obj → Prop
axiom Mature : obj → Prop
axiom Suitable : obj → Prop
axiom BoughtRapid : obj → Prop

axiom KO : obj

-- Premises
-- All growth stocks are bought to earn profits from rapid price appreciation.
axiom P1 : ∀ x : obj, (Stock x ∧ Growth x) → BoughtRapid x
-- If a stock is bought to earn profits from rapid price appreciation, then it is not suitable for a retirement fund.
axiom P2 : ∀ x : obj, BoughtRapid x → ¬ Suitable x
-- Some stocks are growth stocks.
axiom P3 : ∃ x : obj, Stock x ∧ Growth x
-- All mature stocks are suitable for a retirement fund.
axiom P4 : ∀ x : obj, (Stock x ∧ Mature x) → Suitable x
-- KO is a mature stock (i.e., KO is a stock and KO is mature).
axiom P5_stock : Stock KO
axiom P5_mature : Mature KO

theorem KO_suitable : Suitable KO :=
  P4 KO ⟨P5_stock, P5_mature⟩

theorem KO_not_growth : ¬ Growth KO :=
by
  intro gKO
  have br : BoughtRapid KO := P1 KO ⟨P5_stock, gKO⟩
  have ns : ¬ Suitable KO := P2 KO br
  exact ns KO_suitable

theorem not_stock_and_growth_KO : ¬ (Stock KO ∧ Growth KO) :=
by
  intro h
  have g : Growth KO := h.right
  exact KO_not_growth g
