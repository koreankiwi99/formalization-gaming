<lean>
axiom obj : Type

axiom Tom : obj

axiom WildTurkey : obj → Prop
axiom Eastern : obj → Prop
axiom Osceola : obj → Prop
axiom Goulds : obj → Prop
axiom Merriams : obj → Prop
axiom RioGrande : obj → Prop
axiom Ocellated : obj → Prop

-- Every wild turkey is one of the six types listed
axiom cover_types :
  ∀ x : obj,
    WildTurkey x →
    (Eastern x ∨ Osceola x ∨ Goulds x ∨ Merriams x ∨ RioGrande x ∨ Ocellated x)

-- Given facts about Tom
axiom Tom_not_Eastern : ¬ Eastern Tom
axiom Tom_not_Osceola : ¬ Osceola Tom
axiom Tom_not_Goulds : ¬ Goulds Tom
axiom Tom_not_Merriams : ¬ Merriams Tom
axiom Tom_not_RioGrande : ¬ RioGrande Tom
axiom Tom_is_WildTurkey : WildTurkey Tom

theorem tom_is_ocellated : Ocellated Tom := by
  have h := cover_types Tom Tom_is_WildTurkey
  cases h
  case inl hE =>
    exact False.elim (Tom_not_Eastern hE)
  case inr h2 =>
    cases h2
    case inl hOs =>
      exact False.elim (Tom_not_Osceola hOs)
    case inr h3 =>
      cases h3
      case inl hG =>
        exact False.elim (Tom_not_Goulds hG)
      case inr h4 =>
        cases h4
        case inl hM =>
          exact False.elim (Tom_not_Merriams hM)
        case inr h5 =>
          cases h5
          case inl hR =>
            exact False.elim (Tom_not_RioGrande hR)
          case inr hOc =>
            exact hOc
</lean>

ANSWER: True
