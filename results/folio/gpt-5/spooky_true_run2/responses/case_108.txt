axiom Person : Type
axiom Peter : Person

axiom Spill : Person → Prop
axiom Tidy : Person → Prop
axiom Clean : Person → Prop
axiom Value : Person → Prop
axiom FamilyPrior : Person → Prop
axiom ClumsyFoodie : Person → Prop
axiom GoesOut : Person → Prop

-- Premises
axiom A1 : ∀ x : Person, Spill x → ¬ Tidy x
axiom A2 : ∀ x : Person, (ClumsyFoodie x ∧ GoesOut x) → Spill x
axiom A3 : ∀ x : Person, Clean x → Tidy x
axiom A4 : ∀ x : Person, Value x → Clean x
axiom A5 : ∀ x : Person, FamilyPrior x → Value x
axiom A6 : (Spill Peter ∧ Clean Peter) ∨ ((¬ Spill Peter) ∧ (¬ Clean Peter))

-- From the first disjunct we can derive any conclusion (explosion)
theorem target_from_case1 :
  (Spill Peter ∧ Clean Peter) →
  ((ClumsyFoodie Peter ∧ GoesOut Peter) ∨ FamilyPrior Peter) :=
by
  intro h
  have ht : Tidy Peter := A3 Peter h.right
  have hnt : ¬ Tidy Peter := A1 Peter h.left
  exact False.elim (hnt ht)

-- From the second disjunct we can refute the target conclusion
theorem not_target_from_case2 :
  ((¬ Spill Peter) ∧ (¬ Clean Peter)) →
  ¬ ((ClumsyFoodie Peter ∧ GoesOut Peter) ∨ FamilyPrior Peter) :=
by
  intro h
  intro hdisj
  cases hdisj with
  | inl hcf =>
      have hs : Spill Peter := A2 Peter hcf
      exact h.left hs
  | inr hfam =>
      have hv : Value Peter := A5 Peter hfam
      have hc : Clean Peter := A4 Peter hv
      exact h.right hc
