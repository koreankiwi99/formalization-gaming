axiom obj : Type

-- Predicates
axiom BusinessOrg : obj → Prop
axiom Company : obj → Prop
axiom PrivateCompany : obj → Prop
axiom LegalEntity : obj → Prop
axiom CreatedUnderLaw : obj → Prop
axiom LegalObligations : obj → Prop

-- Constant for the Harvard Weekly Book Club
axiom HWBC : obj

-- Premises
axiom A1 : ∀ x : obj, BusinessOrg x → LegalEntity x
axiom A2 : ∀ x : obj, Company x → BusinessOrg x
axiom A3 : ∀ x : obj, PrivateCompany x → Company x
axiom A4 : ∀ x : obj, LegalEntity x → CreatedUnderLaw x
axiom A5 : ∀ x : obj, LegalEntity x → LegalObligations x
axiom A6 : CreatedUnderLaw HWBC → ¬ PrivateCompany HWBC

-- From the premises we can derive that HWBC is not a private company.
theorem not_private_company_HWBC : ¬ PrivateCompany HWBC := by
  intro hp
  have hcomp : Company HWBC := A3 HWBC hp
  have hbo : BusinessOrg HWBC := A2 HWBC hcomp
  have hle : LegalEntity HWBC := A1 HWBC hbo
  have hcul : CreatedUnderLaw HWBC := A4 HWBC hle
  have hnotpc : ¬ PrivateCompany HWBC := A6 hcul
  exact hnotpc hp

-- Therefore, the conjunction "HWBC has legal obligations and is a private company" is impossible.
theorem not_statement : ¬ (LegalObligations HWBC ∧ PrivateCompany HWBC) := by
  intro h
  exact not_private_company_HWBC (And.right h)
