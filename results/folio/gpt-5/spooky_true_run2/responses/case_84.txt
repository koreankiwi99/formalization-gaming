<lean>
axiom obj : Type

axiom Hamden : obj

axiom Listed : obj → Prop
axiom ManyNeg : obj → Prop
axiom RatingGT4 : obj → Prop
axiom TakeOut : obj → Prop
axiom Popular : obj → Prop

-- If listed in Yelp’s recommendations, then not many negative reviews.
axiom A1 : ∀ x : obj, Listed x → ¬ ManyNeg x
-- All with rating > 4 are listed.
axiom A2 : ∀ x : obj, RatingGT4 x → Listed x
-- Some restaurants without take-out receive many negative reviews.
axiom A3 : ∃ x : obj, ¬ TakeOut x ∧ ManyNeg x
-- All popular among locals have rating > 4.
axiom A4 : ∀ x : obj, Popular x → RatingGT4 x
-- Hamden has rating > 4 or is popular.
axiom A5 : RatingGT4 Hamden ∨ Popular Hamden

theorem hamden_takeout_or_notneg : TakeOut Hamden ∨ ¬ ManyNeg Hamden := by
  have hRG4 : RatingGT4 Hamden := by
    cases A5 with
    | inl h => exact h
    | inr hp => exact A4 Hamden hp
  have hL : Listed Hamden := A2 Hamden hRG4
  have hNotNeg : ¬ ManyNeg Hamden := A1 Hamden hL
  exact Or.inr hNotNeg
</lean>
ANSWER: True
