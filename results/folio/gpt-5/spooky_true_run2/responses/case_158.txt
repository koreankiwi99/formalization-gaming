<lean>
axiom obj : Type
axiom Student : obj → Prop
axiom WorksLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom HasUniPTJob : obj → Prop
axiom TakingDB : obj → Prop
axiom TakingDavid : obj → Prop
axiom WorksLab : obj → Prop
axiom James : obj

-- None of the students who currently work in the library are from CS
axiom A1 : ∀ x : obj, Student x → WorksLibrary x → ¬ FromCS x
-- All students with university part-time jobs work in the library
axiom A2 : ∀ x : obj, Student x → HasUniPTJob x → WorksLibrary x
-- All students taking the database course are from CS
axiom A3 : ∀ x : obj, Student x → TakingDB x → FromCS x
-- All students taking a class with Prof. David are taking the database course
axiom A4 : ∀ x : obj, Student x → TakingDavid x → TakingDB x
-- All students working in the lab are taking a class with Prof. David
axiom A5 : ∀ x : obj, Student x → WorksLab x → TakingDavid x
-- James is a student working in the lab
axiom A6 : Student James
axiom A7 : WorksLab James
-- James doesn't work in the lab and doesn't have a university part-time job
axiom A8 : ¬ WorksLab James
axiom A9 : ¬ HasUniPTJob James

theorem james_db_or_pt : TakingDB James ∨ HasUniPTJob James := by
  have hS : Student James := A6
  have hLab : WorksLab James := A7
  have hDavid : TakingDavid James := A5 James hS hLab
  have hDB : TakingDB James := A4 James hS hDavid
  exact Or.inl hDB
</lean>

ANSWER: True
