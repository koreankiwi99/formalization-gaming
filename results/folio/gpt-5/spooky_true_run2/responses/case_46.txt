<lean>
axiom Student : Type

axiom Emma : Student
axiom Mia : Student
axiom James : Student

axiom Season : Type
axiom Spring : Season
axiom Summer : Season
axiom Fall : Season
axiom Winter : Season

axiom Favorite : Student → Season
axiom WantsLongVacation : Student → Prop

axiom R1 : ∀ s : Student, WantsLongVacation s → Favorite s = Summer

axiom EmmaFav : Favorite Emma = Summer
axiom MiaFavNeEmmaFav : Favorite Mia ≠ Favorite Emma

axiom JamesWants : WantsLongVacation James

theorem james_favorite_season_is_summer : Favorite James = Summer := by
  exact R1 James JamesWants
</lean>

ANSWER: True
