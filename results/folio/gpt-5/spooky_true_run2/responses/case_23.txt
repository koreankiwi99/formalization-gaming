<lean>
axiom Show : Type
axiom StrangerThings : Show
axiom BlackMirror : Show

axiom Netflix : Show → Prop
axiom Popular : Show → Prop
axiom Binge : Show → Prop
axiom Download : Show → Prop
axiom ShareWithLisa : Show → Prop

-- "Stranger Things" is a popular Netflix show.
axiom ST_Netflix : Netflix StrangerThings
axiom ST_Popular : Popular StrangerThings

-- If a Netflix show is popular, Karen will binge-watch it.
axiom popular_netflix_implies_binge : ∀ s : Show, (Netflix s ∧ Popular s) → Binge s

-- If and only if Karen binge-watches a Netflix show, she will download it.
axiom download_iff_binge_on_netflix : ∀ s : Show, Netflix s → (Binge s ↔ Download s)

-- Karen does not download "Black Mirror."
axiom not_download_BM : ¬ Download BlackMirror

-- "Black Mirror" is a Netflix show.
axiom BM_Netflix : Netflix BlackMirror

-- If Karen binge-watches a Netflix show, she will share it with Lisa.
axiom binge_netflix_implies_share : ∀ s : Show, Netflix s → Binge s → ShareWithLisa s

-- Consequence: Karen does not binge-watch "Black Mirror."
theorem not_binge_black_mirror : ¬ Binge BlackMirror := by
  intro hb
  have hiff := download_iff_binge_on_netflix BlackMirror BM_Netflix
  have hb_to_dl : Binge BlackMirror → Download BlackMirror := Iff.mp hiff
  have hdl : Download BlackMirror := hb_to_dl hb
  exact not_download_BM hdl

-- Unrelated positive consequence: she shares "Stranger Things" with Lisa.
theorem share_stranger_things : ShareWithLisa StrangerThings := by
  have hnet := ST_Netflix
  have hpop := ST_Popular
  have hb : Binge StrangerThings := popular_netflix_implies_binge StrangerThings (And.intro hnet hpop)
  exact binge_netflix_implies_share StrangerThings hnet hb
