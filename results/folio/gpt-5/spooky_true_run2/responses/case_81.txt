<lean>
axiom obj : Type

axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom BBMProduct : obj → Prop

axiom DTC : obj  -- Dried Thai chilies

axiom no_baked_are_spicy : ∀ x : obj, BakedSweet x → ¬ Spicy x
axiom cupcakes_baked : ∀ x : obj, Cupcake x → BakedSweet x
axiom mala_spicy : ∀ x : obj, MalaHotpot x → Spicy x
axiom bbm_cupcakes : ∀ x : obj, BBMProduct x → Cupcake x
axiom dtc_disj : Spicy DTC ∨ MalaHotpot DTC ∨ ¬ BakedSweet DTC

theorem dtc_neither_bbm_nor_baked : ¬ BBMProduct DTC ∧ ¬ BakedSweet DTC := by
  -- Spicy implies not BakedSweet
  have spicy_not_baked : ∀ x : obj, Spicy x → ¬ BakedSweet x := by
    intro x hx
    intro hb
    have hns : ¬ Spicy x := no_baked_are_spicy x hb
    exact hns hx
  -- Derive ¬BakedSweet DTC from the disjunction
  have not_baked : ¬ BakedSweet DTC := by
    have h := dtc_disj
    cases h with
    | inl hsp =>
      exact spicy_not_baked DTC hsp
    | inr h2 =>
      cases h2 with
      | inl hmh =>
        have hsp := mala_spicy DTC hmh
        exact spicy_not_baked DTC hsp
      | inr hnb =>
        exact hnb
  -- From ¬BakedSweet DTC, derive ¬BBMProduct DTC
  have not_bbm : ¬ BBMProduct DTC := by
    intro hbbm
    have hcup := bbm_cupcakes DTC hbbm
    have hbaked := cupcakes_baked DTC hcup
    exact not_baked hbaked
  exact And.intro not_bbm not_baked
</lean>

ANSWER: True
