<lean>
axiom obj : Type
axiom Peter : obj
axiom Spill : obj → Prop
axiom Tidy : obj → Prop
axiom Clean : obj → Prop
axiom Value : obj → Prop
axiom FamilyPrioritize : obj → Prop
axiom ClumsyFoodieGoOut : obj → Prop

axiom A1 : ∀ x : obj, Spill x → ¬ Tidy x
axiom A2 : ∀ x : obj, ClumsyFoodieGoOut x → Spill x
axiom A3 : ∀ x : obj, Clean x → Tidy x
axiom A4 : ∀ x : obj, Value x → Clean x
axiom A5 : ∀ x : obj, FamilyPrioritize x → Value x
axiom A6 : (Spill Peter ∧ Clean Peter) ∨ (¬ Spill Peter ∧ ¬ Clean Peter)

theorem peter_either_both_or_neither :
  (ClumsyFoodieGoOut Peter ∧ FamilyPrioritize Peter) ∨
  (¬ ClumsyFoodieGoOut Peter ∧ ¬ FamilyPrioritize Peter) := by
  have hnot_first : ¬ (Spill Peter ∧ Clean Peter) := by
    intro h
    have hs : Spill Peter := h.left
    have hc : Clean Peter := h.right
    have hnotT : ¬ Tidy Peter := A1 Peter hs
    have hT : Tidy Peter := A3 Peter hc
    exact hnotT hT
  have hsecond : (¬ Spill Peter ∧ ¬ Clean Peter) := by
    exact Or.resolve_left A6 hnot_first
  have hnotS : ¬ Spill Peter := hsecond.left
  have hnotC : ¬ Clean Peter := hsecond.right
  have hnotCF : ¬ ClumsyFoodieGoOut Peter := by
    intro hcf
    apply hnotS
    exact A2 Peter hcf
  have hFtoC : FamilyPrioritize Peter → Clean Peter := by
    intro hf
    have hv : Value Peter := A5 Peter hf
    exact A4 Peter hv
  have hnotF : ¬ FamilyPrioritize Peter := by
    intro hf
    apply hnotC
    exact hFtoC hf
  exact Or.inr ⟨hnotCF, hnotF⟩
</lean>

ANSWER: True
