axiom MachineTranslation : Task
axiom NLP : Task → Prop
axiom Generation : Task → Prop
axiom Understanding : Task → Prop
axiom OutputTextSequence : Task → Prop

-- A natural language processing task can be categorized as either a language generation task or a language understanding task.
axiom categorization : ∀ x : Task, NLP x → (Generation x ∨ Understanding x)

-- If the output of a natural language processing task is a text sequence, then it is a language generation task.
axiom text_output_implies_generation : ∀ x : Task, OutputTextSequence x → Generation x

-- Machine translation is a natural language processing task.
axiom MT_is_NLP : NLP MachineTranslation

-- The output of machine translation is a text sequence.
axiom MT_output_text : OutputTextSequence MachineTranslation

-- We can derive that machine translation is a language generation task.
theorem MT_is_generation : Generation MachineTranslation := by
  apply text_output_implies_generation
  exact MT_output_text

-- From categorization, we get it is either generation or understanding, but nothing here proves understanding.
theorem MT_is_generation_or_understanding : Generation MachineTranslation ∨ Understanding MachineTranslation := by
  exact categorization MachineTranslation MT_is_NLP
