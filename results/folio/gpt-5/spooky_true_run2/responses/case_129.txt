axiom obj : Type
axiom John : obj

-- Predicates
axiom MakePlans : obj → Prop
axiom Engaged : obj → Prop
axiom InviteOthers : obj → Prop
axiom WellAttended : obj → Prop
axiom LargerFamily : obj → Prop
axiom InviteFriends : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, MakePlans x → Engaged x
axiom A2 : ∀ x : obj, InviteOthers x → MakePlans x
axiom A3 : ∀ x : obj, WellAttended x → InviteOthers x
axiom A4 : ∃ x : obj, WellAttended x ∧ LargerFamily x
axiom A5 : Engaged John → (¬ InviteFriends John) ∧ MakePlans John
axiom A6 : LargerFamily John → (WellAttended John ∨ InviteFriends John)

-- Some derived consequences
theorem engaged_implies_makeplans : Engaged John → MakePlans John :=
  fun h => (A5 h).right

theorem makeplans_implies_engaged (x : obj) : MakePlans x → Engaged x :=
  A1 x

theorem engaged_iff_makeplans : Engaged John ↔ MakePlans John := by
  constructor
  · exact engaged_implies_makeplans
  · intro h; exact A1 John h

theorem largerfamily_john_implies_engaged_or_invitefriends :
  LargerFamily John → Engaged John ∨ InviteFriends John := by
  intro hLF
  have h := A6 hLF
  cases h with
  | inl hWA =>
      have hIO : InviteOthers John := A3 John hWA
      have hMP : MakePlans John := A2 John hIO
      have hE  : Engaged John := A1 John hMP
      exact Or.inl hE
  | inr hIF =>
      exact Or.inr hIF
