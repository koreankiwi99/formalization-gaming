-- Jane does not like any animals that bark.
axiom A1 : ∀ x : obj, Animal x → Barks x → ¬ Likes Jane x

-- All dogs are animals that bark.
axiom A2 : ∀ x : obj, Dog x → Animal x
axiom A3 : ∀ x : obj, Dog x → Barks x

-- Jane likes all animals that jump.
axiom A4 : ∀ x : obj, Animal x → Jumps x → Likes Jane x

-- If an animal has legs, then it jumps.
axiom A5 : ∀ x : obj, Animal x → HasLegs x → Jumps x

-- If an animal is terricolous, then it has legs.
axiom A6 : ∀ x : obj, Animal x → Terricolous x → HasLegs x

-- KiKi is an animal.
axiom A7 : Animal KiKi

-- If KiKi neither jumps nor has legs, then KiKi is terricolous or has legs.
axiom A8 : (¬ Jumps KiKi ∧ ¬ HasLegs KiKi) → (Terricolous KiKi ∨ HasLegs KiKi)

-- A derived fact (unrelated to barking/dog): KiKi cannot both fail to jump and lack legs.
theorem no_both_not_jump_and_not_legs : ¬(¬Jumps KiKi ∧ ¬HasLegs KiKi) := by
  intro h
  have disj := A8 h
  cases disj with
  | inl t =>
      have legs : HasLegs KiKi := A6 KiKi A7 t
      exact h.right legs
  | inr legs =>
      exact h.right legs
