axiom James : obj

axiom WorksInLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom HasUniPartTimeJob : obj → Prop
axiom TakesDatabaseCourse : obj → Prop
axiom TakesClassWithDavid : obj → Prop
axiom WorksInLab : obj → Prop

-- Context axioms
axiom A1 : ∀ x : obj, WorksInLibrary x → ¬FromCS x
axiom A2 : ∀ x : obj, HasUniPartTimeJob x → WorksInLibrary x
axiom A3 : ∀ x : obj, TakesDatabaseCourse x → FromCS x
axiom A4 : ∀ x : obj, TakesClassWithDavid x → TakesDatabaseCourse x
axiom A5 : ∀ x : obj, WorksInLab x → TakesClassWithDavid x
axiom A6 : WorksInLab James
-- "James doesn't work in the lab or have a part-time job" interpreted as neither:
axiom A7 : ¬WorksInLab James ∧ ¬HasUniPartTimeJob James

-- Derive a contradiction from A6 and A7
theorem contradiction : False := by
  have hnotlab : ¬WorksInLab James := And.left A7
  havehlab : WorksInLab James := A6
  exact hnotlab havehlab

-- Prove the negation of the statement: ¬(TakesDatabaseCourse James ∨ HasUniPartTimeJob James)
theorem negate_statement : ¬(TakesDatabaseCourse James ∨ HasUniPartTimeJob James) := by
  intro _
  exact False.elim contradiction
