axiom DisregardPhysicalWellBeing : obj → Prop

axiom A2 : ∀ x : obj, GrewUpHealthHome x → EatSaladsRegularly x
axiom A3 : ∀ x : obj, FulfillDailyNutritionalIntakes x → GrewUpHealthHome x
axiom A4 : ∀ x : obj, DisregardPhysicalWellBeing x → ¬VeryConscious x

theorem taylor_not_visit : ¬VisitGymDaily Taylor := by

  cases T6 with

      exact A1 Taylor hESR
    have hNotVCon : ¬VeryConscious Taylor := A4 Taylor h.right
    exact hNotVCon hVCon
  | inr h =>
    have hGHH_from_gym : GrewUpHealthHome Taylor := by
