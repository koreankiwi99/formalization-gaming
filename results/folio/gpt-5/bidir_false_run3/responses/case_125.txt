axiom obj : Type
axiom Rock : obj

axiom Pet : obj → Prop
axiom Mammal : obj → Prop
axiom Animal : obj → Prop
axiom Monkey : obj → Prop
axiom Bird : obj → Prop
axiom CanFly : obj → Prop
axiom CanBreathe : obj → Prop
axiom HasWings : obj → Prop

-- All of Peter's pets that are mammals are also animals.
axiom T1 : ∀ x : obj, Pet x → Mammal x → Animal x
-- All monkeys are mammals.
axiom T2 : ∀ x : obj, Monkey x → Mammal x
-- Peter's pets are all either monkeys or birds.
axiom T3 : ∀ x : obj, Pet x → Monkey x ∨ Bird x
-- Peter's birds can fly.
axiom T4 : ∀ x : obj, Pet x ∧ Bird x → CanFly x
-- All animals that are Peter's pets can breathe.
axiom T5 : ∀ x : obj, Pet x ∧ Animal x → CanBreathe x
-- If Peter's pet can fly, then it has wings.
axiom T6 : ∀ x : obj, Pet x ∧ CanFly x → HasWings x
-- Rock is Peter's pet.
axiom T7 : Pet Rock
-- Rock can fly, or Rock is a bird, or Rock cannot breathe.
axiom T8 : CanFly Rock ∨ Bird Rock ∨ ¬CanBreathe Rock

-- In fact, Rock has wings (hence we cannot prove ¬HasWings Rock).
theorem rock_has_wings : HasWings Rock := by
  have pR : Pet Rock := T7
  have disj : CanFly Rock ∨ Bird Rock ∨ ¬CanBreathe Rock := T8
  cases disj with
  | inl hFly =>
    exact T6 Rock (And.intro pR hFly)
  | inr rest =>
    cases rest with
    | inl hBird =>
      have hCanFly : CanFly Rock := T4 Rock (And.intro pR hBird)
      exact T6 Rock (And.intro pR hCanFly)
    | inr hNotBreath =>
      have pm : Monkey Rock ∨ Bird Rock := T3 Rock pR
      cases pm with
      | inl hMonkey =>
        have hMammal : Mammal Rock := T2 Rock hMonkey
        have hAnimal : Animal Rock := T1 Rock pR hMammal
        have hBreath : CanBreathe Rock := T5 Rock (And.intro pR hAnimal)
        have f : False := hNotBreath hBreath
        exact False.elim f
      | inr hBird2 =>
        have hCanFly : CanFly Rock := T4 Rock (And.intro pR hBird2)
        exact T6 Rock (And.intro pR hCanFly)
