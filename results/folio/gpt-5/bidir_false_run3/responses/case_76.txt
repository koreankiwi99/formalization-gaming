<lean>
axiom obj : Type
axiom Hannah : obj

axiom At : obj → Prop
axiom Works : obj → Prop
axiom Needs : obj → Prop
axiom Orders : obj → Prop
axiom Enjoys : obj → Prop
axiom Spends : obj → Prop
axiom Picky : obj → Prop

-- If people at Mary's school work in student jobs on campus, then they need to earn money to help pay for their college tuition.
axiom R1 : ∀ x : obj, At x → Works x → Needs x
-- If people at Mary's school order takeout frequently in college, then they work in student jobs on campus.
axiom R2 : ∀ x : obj, At x → Orders x → Works x
-- People at Mary's school order takeout frequently in college or enjoy the dining hall meals and recipes.
axiom R3 : ∀ x : obj, At x → (Orders x ∨ Enjoys x)
-- If people at Mary's school enjoy the dining hall meals and recipes, then they are not picky eaters.
axiom R4 : ∀ x : obj, At x → Enjoys x → ¬ Picky x
-- If people at Mary's school enjoy the dining hall meals and recipes, then they spend a lot of their time eating and catching up with friends in the campus dining halls.
axiom R5 : ∀ x : obj, At x → Enjoys x → Spends x

-- Hannah is at Mary's school.
axiom T1 : At Hannah
-- Hannah works; and if she needs money, then she is neither picky nor needs money.
axiom T2 : Works Hannah ∧ (Needs Hannah → (¬ Picky Hannah ∧ ¬ Needs Hannah))

-- The premises are inconsistent: derive False.
theorem contradiction : False :=
by
  have Wh : Works Hannah := And.left T2
  have Nh : Needs Hannah := R1 Hannah T1 Wh
  have hcond : Needs Hannah → (¬ Picky Hannah ∧ ¬ Needs Hannah) := And.right T2
  have nNh : ¬ Needs Hannah := And.right (hcond Nh)
  exact nNh Nh

-- Negation of the statement to evaluate:
-- "Hannah is at Mary's school and she is not a picky eater and spends a lot of her time in the dining halls."
-- We can prove its negation since we already have False from the premises.
theorem neg_statement :
  ¬(At Hannah ∧ ¬ Picky Hannah ∧ Spends Hannah) :=
by
  intro _
  exact False.elim contradiction
</lean>

ANSWER: False
