<lean>
axiom obj : Type
axiom AtMixer : obj → Prop
axiom GrandSlam : obj → Prop
axiom OscarNom : obj → Prop
axiom TennisPro : obj → Prop
axiom Celebrity : obj → Prop
axiom Athlete : obj → Prop
axiom Famous : obj → Prop
axiom WellPaid : obj → Prop
axiom TaxHaven : obj → Prop

axiom Djokovic : obj

-- Context axioms
axiom A1 : ∀ x : obj, AtMixer x → (GrandSlam x ∨ OscarNom x)
axiom A2 : ∀ x : obj, AtMixer x → GrandSlam x → TennisPro x
axiom A3 : ∀ x : obj, AtMixer x → OscarNom x → Celebrity x
axiom A4 : ∀ x : obj, AtMixer x → TennisPro x → Athlete x
axiom A5 : ∀ x : obj, AtMixer x → Celebrity x → WellPaid x
axiom A6 : ∀ x : obj, AtMixer x → Athlete x → Famous x
axiom A7 : ∀ x : obj, AtMixer x → WellPaid x → TaxHaven x

axiom Dj_at : AtMixer Djokovic
axiom Dj_fam_ath_implies_wp : Famous Djokovic → Athlete Djokovic → WellPaid Djokovic

theorem djokovic_lives_in_tax_haven : TaxHaven Djokovic :=
by
  have DjAt : AtMixer Djokovic := Dj_at
  exact Or.elim (A1 Djokovic DjAt)
    (fun g =>
      let tp := A2 Djokovic DjAt g
      let ath := A4 Djokovic DjAt tp
      let fam := A6 Djokovic DjAt ath
      let wp := Dj_fam_ath_implies_wp fam ath
      A7 Djokovic DjAt wp)
    (fun o =>
      let celeb := A3 Djokovic DjAt o
      let wp := A5 Djokovic DjAt celeb
      A7 Djokovic DjAt wp)
</lean>
