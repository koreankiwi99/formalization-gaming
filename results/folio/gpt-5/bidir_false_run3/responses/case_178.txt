<lean>
axiom obj : Type
axiom Human : obj → Prop
axiom Horse : obj → Prop
axiom Hooves : obj → Prop

-- All horses have hooves.
axiom H1 : ∀ x : obj, Horse x → Hooves x
-- No humans have hooves.
axiom H2 : ∀ x : obj, Human x → ¬Hooves x

-- Negation of "Some humans are horses."
theorem no_human_is_horse : ¬ (∃ x : obj, Human x ∧ Horse x) := by
  intro h
  cases h with
  | intro x hx =>
    cases hx with
    | intro hxHuman hxHorse =>
      have hxHooves : Hooves x := H1 x hxHorse
      have hxNotHooves : ¬Hooves x := H2 x hxHuman
      exact hxNotHooves hxHooves
</lean>

ANSWER: False
