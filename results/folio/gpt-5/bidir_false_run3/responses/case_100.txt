<lean>
axiom obj : Type

axiom Harry : obj

axiom Bee : obj → Prop
axiom QueenBee : obj → Prop
axiom Vertebrate : obj → Prop
axiom Invertebrate : obj → Prop
axiom Backbone : obj → Prop
axiom MFRepro : obj → Prop

-- Animals are either invertebrates or vertebrates.
axiom R1 : ∀ x : obj, Invertebrate x ∨ Vertebrate x
-- All animals with backbones reproduce by male-and-female mating.
axiom R2 : ∀ x : obj, Backbone x → MFRepro x
-- All vertebrate animals have a backbone.
axiom R3 : ∀ x : obj, Vertebrate x → Backbone x
-- All bees do not reproduce by male-and-female mating.
axiom R4 : ∀ x : obj, Bee x → ¬MFRepro x
-- All queen bees are bees.
axiom R5 : ∀ x : obj, QueenBee x → Bee x
-- Harry is a bee.
axiom T1 : Bee Harry

lemma not_vertebrate_Harry : ¬Vertebrate Harry := by
  intro hv
  have hb : Backbone Harry := R3 Harry hv
  have hm : MFRepro Harry := R2 Harry hb
  have hnm : ¬MFRepro Harry := R4 Harry T1
  exact hnm hm

lemma not_backbone_Harry : ¬Backbone Harry := by
  intro hb
  have hm : MFRepro Harry := R2 Harry hb
  have hnm : ¬MFRepro Harry := R4 Harry T1
  exact hnm hm

lemma invertebrate_Harry : Invertebrate Harry := by
  have h := R1 Harry
  cases h with
  | inl hinv => exact hinv
  | inr hv => exact False.elim (not_vertebrate_Harry hv)

-- Negation of the statement: ¬(P → Q)
-- P: (Vertebrate Harry ∧ Backbone Harry) ∨ (¬Vertebrate Harry ∧ ¬Backbone Harry)
-- Q: ¬Invertebrate Harry ∧ ¬QueenBee Harry
theorem neg_statement :
  ¬( ((Vertebrate Harry ∧ Backbone Harry) ∨ (¬Vertebrate Harry ∧ ¬Backbone Harry))
     → (¬Invertebrate Harry ∧ ¬QueenBee Harry) ) := by
  intro hImp
  have p : ( (Vertebrate Harry ∧ Backbone Harry) ∨ (¬Vertebrate Harry ∧ ¬Backbone Harry) ) :=
    Or.inr ⟨not_vertebrate_Harry, not_backbone_Harry⟩
  have q : (¬Invertebrate Harry ∧ ¬QueenBee Harry) := hImp p
  have hNotInv : ¬Invertebrate Harry := q.left
  exact hNotInv invertebrate_Harry
</lean>
