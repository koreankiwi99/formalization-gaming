axiom AllowsPets : obj → Prop
axiom Pet : obj → Prop

axiom AllowedMoveIn : obj → obj → obj → Prop -- person → pet → building
axiom WillRent : obj → obj → Prop            -- person → building
axiom MonthlyRent : obj → Nat
axiom SecurityDeposit : obj → Nat

-- Context axioms
-- Pets are allowed in some managed buildings.
axiom some_managed_allows_pets : ∃ b : obj, Managed b ∧ AllowsPets b

-- A deposit is required to rent an apartment in a managed building.
-- (We model the deposit amount function; requirement implicit.)

-- The security deposit is at least the monthly rent at a managed building.
axiom deposit_at_least_rent : ∀ b : obj, Managed b → MonthlyRent b ≤ SecurityDeposit b

-- Fluffy is Tom's cat.
axiom Fluffy_cat : Cat Fluffy

-- Cats are pets.
axiom cats_are_pets : ∀ x : obj, Cat x → Pet x

-- The Olive Garden is a managed building.
axiom OG_managed : Managed OliveGarden

-- The monthly rent at the Olive Garden is $2000.
axiom OG_rent_2000 : MonthlyRent OliveGarden = 2000

-- $2000 is more than $1500.
axiom two_thousand_gt_1500 : 2000 > 1500

-- Tom will rent an apartment in a managed building if he is allowed to move in with Fluffy,
-- and the security deposit is no more than $1500.
axiom tom_rents_if_allowed_and_deposit_small :
  ∀ b : obj, Managed b → AllowedMoveIn Tom Fluffy b → SecurityDeposit b ≤ 1500 → WillRent Tom b

-- If a managed building allows pets, then people are allowed to move in with a pet.
axiom allows_pets_implies_allowed_move_in :
  ∀ b : obj, Managed b → AllowsPets b → ∀ person pet, Pet pet → AllowedMoveIn person pet b
