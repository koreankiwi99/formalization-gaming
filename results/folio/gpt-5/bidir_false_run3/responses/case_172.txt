axiom obj : Type
axiom Taylor : obj

axiom EatsSaladsRegularly : obj → Prop
axiom VeryConscious : obj → Prop
axiom GrewUpHealthConsciousHome : obj → Prop
axiom FulfillsDailyNutrition : obj → Prop
axiom DisregardsPhysicalWellBeing : obj → Prop
axiom VisitsGymDaily : obj → Prop

-- All who eat salads regularly are very conscious.
axiom A1 : ∀ x : obj, EatsSaladsRegularly x → VeryConscious x
-- All who grew up in health-conscious childhood homes eat salads regularly.
axiom A2 : ∀ x : obj, GrewUpHealthConsciousHome x → EatsSaladsRegularly x
-- All who fulfill their daily nutritional intakes grew up in health-conscious childhood homes.
axiom A3 : ∀ x : obj, FulfillsDailyNutrition x → GrewUpHealthConsciousHome x
-- All who disregard their physical well-being are not very conscious.
axiom A4 : ∀ x : obj, DisregardsPhysicalWellBeing x → ¬VeryConscious x
-- If people visit the gym at least once a day, then they fulfill their daily nutritional intakes.
axiom A5 : ∀ x : obj, VisitsGymDaily x → FulfillsDailyNutrition x

-- Taylor either (grew up health-conscious ∧ disregards) or (neither).
axiom A6 : (GrewUpHealthConsciousHome Taylor ∧ DisregardsPhysicalWellBeing Taylor)
        ∨ (¬GrewUpHealthConsciousHome Taylor ∧ ¬DisregardsPhysicalWellBeing Taylor)

-- From (grew up health-conscious ∧ disregards) we derive a contradiction.
theorem not_case1 : ¬(GrewUpHealthConsciousHome Taylor ∧ DisregardsPhysicalWellBeing Taylor) :=
by
  intro h
  have hH : GrewUpHealthConsciousHome Taylor := h.left
  have hD : DisregardsPhysicalWellBeing Taylor := h.right
  have hS : EatsSaladsRegularly Taylor := A2 Taylor hH
  have hC : VeryConscious Taylor := A1 Taylor hS
  have hNC : ¬VeryConscious Taylor := A4 Taylor hD
  exact hNC hC

-- Hence, from the given disjunction, the second case must hold.
theorem case2 : ¬GrewUpHealthConsciousHome Taylor ∧ ¬DisregardsPhysicalWellBeing Taylor :=
by
  cases A6 with
  | inl h =>
      exact False.elim (not_case1 h)
  | inr h =>
      exact h
