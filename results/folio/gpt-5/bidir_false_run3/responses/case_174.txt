axiom Conscious : obj → Prop
axiom GrewHome : obj → Prop
axiom Fulfill : obj → Prop
axiom Disregard : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, EatSalad x → Conscious x
axiom A2 : ∀ x : obj, GrewHome x → EatSalad x
axiom A3 : ∀ x : obj, Fulfill x → GrewHome x
axiom A4 : ∀ x : obj, Disregard x → ¬Conscious x
axiom A5 : ∀ x : obj, GymDaily x → Fulfill x
axiom A6 : (GrewHome Taylor ∧ Disregard Taylor) ∨ (¬GrewHome Taylor ∧ ¬Disregard Taylor)

-- From GrewHome and the premises, one becomes Conscious; Disregard implies not Conscious, so both cannot hold.
theorem no_both_Taylor : ¬(GrewHome Taylor ∧ Disregard Taylor) := by
  intro h
  have hgh : GrewHome Taylor := And.left h
  have hdis : Disregard Taylor := And.right h
  have hcons : Conscious Taylor := A1 Taylor (A2 Taylor hgh)
  have hncons : ¬Conscious Taylor := A4 Taylor hdis
  exact hncons hcons

-- Using the exclusive-style premise A6 and the impossibility above, we deduce Taylor neither grew up in such a home nor disregards well-being.
theorem taylor_neither_grew_nor_disregard : ¬GrewHome Taylor ∧ ¬Disregard Taylor := by
  have hb := no_both_Taylor
  cases A6 with
  | inl h =>
      exact False.elim (hb h)
  | inr h =>
      exact h
