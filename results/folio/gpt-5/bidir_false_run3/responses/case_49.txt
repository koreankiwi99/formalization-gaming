<lean>
axiom obj : Type
axiom N1984 : obj

axiom Digital : obj → Prop
axiom Analog : obj → Prop
axiom PrintedText : obj → Prop
axiom Streaming : obj → Prop
axiom Hardcover : obj → Prop

-- No digital media are analog.
axiom R1 : ∀ x : obj, Digital x → ¬Analog x
-- Every printed text is analog media.
axiom R2 : ∀ x : obj, PrintedText x → Analog x
-- All streaming services are digital media.
axiom R3 : ∀ x : obj, Streaming x → Digital x
-- If an object is a hardcover book, then it is printed text.
axiom R4 : ∀ x : obj, Hardcover x → PrintedText x
-- If 1984 is a streaming service, then 1984 is a hardcover book.
axiom R5 : Streaming N1984 → Hardcover N1984

theorem not_streaming_1984 : ¬Streaming N1984 := by
  intro hS
  have hD : Digital N1984 := R3 N1984 hS
  have hNotA : ¬Analog N1984 := R1 N1984 hD
  have hHC : Hardcover N1984 := R5 hS
  have hPT : PrintedText N1984 := R4 N1984 hHC
  have hA : Analog N1984 := R2 N1984 hPT
  exact hNotA hA
</lean>

ANSWER: False
