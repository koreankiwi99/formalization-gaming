<lean>
axiom Person : Type
axiom Emma : Person
axiom Mia : Person
axiom James : Person

inductive Season where
| Spring : Season
| Summer : Season
| Fall : Season
| Winter : Season

axiom Favorite : Person → Season
axiom Student : Person → Prop
axiom WantLongVacation : Person → Prop

-- All students who want a long vacation have summer as their favorite season.
axiom R1 : ∀ x : Person, Student x → WantLongVacation x → Favorite x = Season.Summer

-- Emma's favorite season is summer.
axiom T1 : Favorite Emma = Season.Summer

-- Mia's favorite season is not the same as Emma's.
axiom T2 : Favorite Mia ≠ Favorite Emma

-- James wants to have a long vacation.
axiom T3 : WantLongVacation James

-- Derived fact: Mia's favorite season is not summer.
theorem Mia_not_summer : Favorite Mia ≠ Season.Summer := by
  intro h
  have : Favorite Mia = Favorite Emma := Eq.trans h (Eq.symm T1)
  exact T2 this

-- We cannot prove ¬(Favorite James = Season.Summer) from the given premises,
-- since we do not know that James is a student (needed to apply R1),
-- nor do we have any information pushing James's favorite away from summer.
</lean>
