<lean>
axiom obj : Type

axiom Zaha : obj → Prop
axiom Kelly : obj → Prop
axiom Style : obj → Prop
axiom DesignByMax : obj → Prop
axiom MaxAdores : obj → Prop
axiom Interesting : obj → Prop
axiom Brutalist : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop

-- All of Zaha Hadid's design styles that Max adores have interesting geometries.
axiom A1 : ∀ x : obj, Zaha x ∧ MaxAdores x → Interesting x
-- No brutalist buildings that Max adores have interesting geometries.
axiom A2 : ∀ x : obj, Brutalist x ∧ MaxAdores x → ¬Interesting x
-- Every style that Max adores is either Zaha Hadid's design style or Kelly Wearstler's design style.
axiom A3 : ∀ x : obj, Style x ∧ MaxAdores x → (Zaha x ∨ Kelly x)
-- All of Kelly Wearstler's design styles that Max adores are evocative.
axiom A4 : ∀ x : obj, Kelly x ∧ MaxAdores x → Evocative x
-- All of Kelly Wearstler's design styles that Max adores are dreamy.
axiom A5 : ∀ x : obj, Kelly x ∧ MaxAdores x → Dreamy x
-- If a design by Max that he adores has interesting geometries, then the design is a brutalist building and evocative.
axiom A6 : ∀ x : obj, DesignByMax x ∧ MaxAdores x ∧ Interesting x → Brutalist x ∧ Evocative x

-- Target statement to negate would be: ∃ x, DesignByMax x ∧ Brutalist x
-- From the given axioms, we cannot derive ¬(∃ x, DesignByMax x ∧ Brutalist x).
-- There is no link forbidding a design by Max from being brutalist unless it is also adored and interesting.

-- An example derivable contradiction (not the target):
-- No object can be simultaneously a design by Max, a Zaha style, and adored by Max.
theorem no_design_by_max_zaha_adored : ∀ x : obj, DesignByMax x ∧ Zaha x ∧ MaxAdores x → False := by
  intro x hx
  have hDM : DesignByMax x := And.left hx
  have hZ : Zaha x := And.left (And.right hx)
  have hM : MaxAdores x := And.right (And.right hx)
  have hI : Interesting x := A1 x (And.intro hZ hM)
  have hBE : Brutalist x ∧ Evocative x := A6 x (And.intro hDM (And.intro hM hI))
  have hB : Brutalist x := And.left hBE
  have hNotI : ¬Interesting x := A2 x (And.intro hB hM)
  exact hNotI hI

-- However, this does not rule out the existence of a design by Max that is brutalist (without the Zaha/adored/interesting conditions).
-- Therefore we cannot prove the negation of the statement "A design by Max is a brutalist building."
