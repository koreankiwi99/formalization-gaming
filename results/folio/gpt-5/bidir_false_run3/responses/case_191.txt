axiom Pro : obj → Prop
axiom Jump : obj → Prop

axiom SomeProNotAmerican : ∃ x : obj, Pro x ∧ ¬American x
-- All professional basketball players can do jump shots.
axiom Pro_implies_Jump : ∀ x : obj, Pro x → Jump x
-- If someone can jump shots, they leap straight into the air.
axiom Jump_implies_Leap : ∀ x : obj, Jump x → Leap x
-- If someone leaps straight into the air, they activate their leg muscles.
axiom Leap_implies_Activate : ∀ x : obj, Leap x → Activate x
-- Yuri does not activate his leg muscles.
axiom Yuri_not_Activate : ¬Activate Yuri

theorem not_Pro_Yuri : ¬Pro Yuri := by
  intro hPro
  have hJump : Jump Yuri := Pro_implies_Jump Yuri hPro
  have hLeap : Leap Yuri := Jump_implies_Leap Yuri hJump
  have hAct : Activate Yuri := Leap_implies_Activate Yuri hLeap
  exact Yuri_not_Activate hAct

theorem negate_statement : ¬(¬(American Yuri ∧ Pro Yuri) → Pro Yuri) := by
  intro hImp
  have hNotPro : ¬Pro Yuri := not_Pro_Yuri
  have hAnte : ¬(American Yuri ∧ Pro Yuri) := by
    intro hAnd
    have hPro : Pro Yuri := And.right hAnd
    exact hNotPro hPro
  have hProFromImp : Pro Yuri := hImp hAnte
  exact hNotPro hProFromImp
