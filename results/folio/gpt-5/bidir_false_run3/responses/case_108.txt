axiom obj : Type
axiom Peter : obj

axiom Spill : obj → Prop
axiom Tidy : obj → Prop
axiom CFG : obj → Prop  -- clumsy foodie who goes out frequently
axiom Cleanly : obj → Prop
axiom Value : obj → Prop  -- values order and spotlessness
axiom Family : obj → Prop -- family prioritizes order and spotlessness

-- Premises
axiom A1 : ∀ x : obj, Spill x → ¬ Tidy x
axiom A2 : ∀ x : obj, CFG x → Spill x
axiom A3 : ∀ x : obj, Cleanly x → Tidy x
axiom A4 : ∀ x : obj, Value x → Cleanly x
axiom A5 : ∀ x : obj, Family x → Value x
axiom A6 : (Spill Peter ∧ Cleanly Peter) ∨ (¬ Spill Peter ∧ ¬ Cleanly Peter)

lemma not_both : ¬ (Spill Peter ∧ Cleanly Peter) := by
  intro h
  have hs : Spill Peter := And.left h
  have hc : Cleanly Peter := And.right h
  have ht : Tidy Peter := A3 Peter hc
  have hnt : ¬ Tidy Peter := A1 Peter hs
  exact hnt ht

lemma not_spill_not_clean : ¬ Spill Peter ∧ ¬ Cleanly Peter := by
  cases A6 with
  | inl h => exact False.elim (not_both h)
  | inr h => exact h

lemma not_cfg : ¬ CFG Peter := by
  intro hcfg
  have hs : Spill Peter := A2 Peter hcfg
  have hns : ¬ Spill Peter := And.left not_spill_not_clean
  exact hns hs

lemma not_family : ¬ Family Peter := by
  intro hfam
  have hv : Value Peter := A5 Peter hfam
  have hc : Cleanly Peter := A4 Peter hv
  have hnc : ¬ Cleanly Peter := And.right not_spill_not_clean
  exact hnc hc

theorem peter_statement_false : ¬ (CFG Peter ∨ Family Peter) := by
  intro h
  cases h with
  | inl hcfg => exact not_cfg hcfg
  | inr hfam => exact not_family hfam
