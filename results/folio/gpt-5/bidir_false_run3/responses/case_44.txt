<lean>
axiom obj : Type
axiom Coco : obj

axiom RankedHigh : obj → Prop
axiom MostActive : obj → Prop
axiom LostToIga : obj → Prop
axiom LostToNadal : obj → Prop
axiom PlayerRG : obj → Prop
axiom Female : obj → Prop
axiom Male : obj → Prop

-- If someone is ranked highly by the WTA, then they are one of the most active players in major tennis.
axiom A1 : ∀ x : obj, RankedHigh x → MostActive x
-- Everyone who lost to Iga Świątek at Roland Garros 2022 is ranked highly by the WTA.
axiom A2 : ∀ x : obj, LostToIga x → RankedHigh x
-- All female tennis players at Roland Garros 2022 lost to Iga Świątek.
axiom A3 : ∀ x : obj, PlayerRG x → Female x → LostToIga x
-- All tennis players at Roland Garros 2022 are either female or male.
axiom A4 : ∀ x : obj, PlayerRG x → (Female x ∨ Male x)
-- All male tennis players at Roland Garros in 2022 lost to Rafael Nadal.
axiom A5 : ∀ x : obj, PlayerRG x → Male x → LostToNadal x
-- If Coco Gauff is ranked highly by the WTA or lost to Rafael Nadal, then Coco Gauff is not a male tennis player at Roland Garros 2022.
axiom A6 : (RankedHigh Coco ∨ LostToNadal Coco) → ¬(PlayerRG Coco ∧ Male Coco)
-- Coco Gauff is at Roland Garros 2022 (as a tennis player there).
axiom A7 : PlayerRG Coco

-- Negation of the statement to evaluate: Coco both lost to Iga and is one of the most active players.
theorem coco_lost_and_active : LostToIga Coco ∧ MostActive Coco := by
  have hP : PlayerRG Coco := A7
  have hdisj : Female Coco ∨ Male Coco := A4 Coco hP
  cases hdisj with
  | inl hf =>
      have hLI : LostToIga Coco := A3 Coco hP hf
      have hRH : RankedHigh Coco := A2 Coco hLI
      have hMA : MostActive Coco := A1 Coco hRH
      exact And.intro hLI hMA
  | inr hm =>
      have hLN : LostToNadal Coco := A5 Coco hP hm
      have hnot : ¬(PlayerRG Coco ∧ Male Coco) := A6 (Or.inr hLN)
      have hFalse : False := hnot (And.intro hP hm)
      exact False.elim hFalse
</lean>

ANSWER: False
