theorem not_ted_pet_and_bsc : ¬(Pet Ted ∧ BrownSwissCattle Ted) :=

  have hBSC : BrownSwissCattle Ted := And.right h
  have hCow : Cow Ted := R1 Ted hBSC
  have hDom : Domesticated Ted := R2 Ted hCow

  exact hNotDom hDom
