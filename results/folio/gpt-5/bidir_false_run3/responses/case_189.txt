<lean>
axiom obj : Type
axiom Yuri : obj

axiom American : obj → Prop
axiom PBP : obj → Prop
axiom JumpShot : obj → Prop
axiom LeapStraight : obj → Prop
axiom ActivateLeg : obj → Prop

-- Some professional basketball players are not American nationals.
axiom S1 : ∃ x : obj, PBP x ∧ ¬American x

-- All professional basketball players can do jump shots.
axiom A1 : ∀ x : obj, PBP x → JumpShot x

-- If someone can do jump shots, they leap straight into the air.
axiom A2 : ∀ x : obj, JumpShot x → LeapStraight x

-- If someone leaps straight into the air, they activate their leg muscles.
axiom A3 : ∀ x : obj, LeapStraight x → ActivateLeg x

-- Yuri does not activate his leg muscles.
axiom Y0 : ¬ActivateLeg Yuri

-- Negation of the statement: Yuri is an American professional basketball player.
theorem yuri_not_american_pbp : ¬(American Yuri ∧ PBP Yuri) :=
by
  intro h
  have hy_pbp : PBP Yuri := And.right h
  have hj : JumpShot Yuri := A1 Yuri hy_pbp
  have hl : LeapStraight Yuri := A2 Yuri hj
  have ha : ActivateLeg Yuri := A3 Yuri hl
  exact Y0 ha
</lean>

ANSWER: False
