<lean>
axiom obj : Type

axiom DriedThaiChilies : obj

axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom FromBBM : obj → Prop

-- No baked sweets are spicy.
axiom H1 : ∀ x : obj, BakedSweet x → ¬ Spicy x
-- All cupcakes are baked sweets.
axiom H2 : ∀ x : obj, Cupcake x → BakedSweet x
-- All mala hotpots are spicy.
axiom H3 : ∀ x : obj, MalaHotpot x → Spicy x
-- All products from Baked by Melissa are cupcakes.
axiom H4 : ∀ x : obj, FromBBM x → Cupcake x
-- Dried Thai chilies are spicy or mala hotpots or not baked sweets.
axiom H5 : Spicy DriedThaiChilies ∨ MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies

theorem dried_thai_chilies_not_fromBBM : ¬ FromBBM DriedThaiChilies := by
  intro hFrom
  have hCup : Cupcake DriedThaiChilies := H4 DriedThaiChilies hFrom
  have hBaked : BakedSweet DriedThaiChilies := H2 DriedThaiChilies hCup
  have hNotSpicy : ¬ Spicy DriedThaiChilies := H1 DriedThaiChilies hBaked
  have contra : False := by
    cases H5 with
    | inl hSpicy =>
        exact hNotSpicy hSpicy
    | inr hrest =>
        cases hrest with
        | inl hMala =>
            have hSpicy' : Spicy DriedThaiChilies := H3 DriedThaiChilies hMala
            exact hNotSpicy hSpicy'
        | inr hNotBaked =>
            exact hNotBaked hBaked
  exact contra
</lean>

ANSWER: True
