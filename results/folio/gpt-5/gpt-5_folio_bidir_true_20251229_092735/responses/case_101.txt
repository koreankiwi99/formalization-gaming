axiom MultipleBirth : Person → Prop
axiom SpendTime : Person → Prop
axiom HaveSiblingsBornTogether : Person → Prop
axiom ComplainSiblings : Person → Prop
axiom LiveAtHome : Person → Prop
axiom LiveWithStrangers : Person → Prop
axiom Baby : Person → Prop

-- Premises
axiom A1 : ∀ x : Person, MultipleBirth x → SpendTime x
axiom A2 : ∀ x : Person, HaveSiblingsBornTogether x → MultipleBirth x
axiom A3 : ∀ x : Person, ComplainSiblings x → HaveSiblingsBornTogether x
axiom A4 : ∀ x : Person, LiveAtHome x → ¬ LiveWithStrangers x
axiom A5 : ∀ x : Person, SpendTime x → LiveAtHome x

-- Luke either: (baby ∧ multiple birth ∧ lives with strangers) or (not (baby ∧ multiple birth) ∧ not lives with strangers)
axiom A6 : (Baby Luke ∧ MultipleBirth Luke ∧ LiveWithStrangers Luke) ∨ ((¬ (Baby Luke ∧ MultipleBirth Luke)) ∧ (¬ LiveWithStrangers Luke))

-- The first disjunct of A6 contradicts A1, A5, A4
theorem not_first_disjunct : ¬ (Baby Luke ∧ MultipleBirth Luke ∧ LiveWithStrangers Luke) :=
by
  intro h
  have hMB : MultipleBirth Luke := And.left h.2
  have hLWS : LiveWithStrangers Luke := And.right h.2
  have hST : SpendTime Luke := A1 Luke hMB
  have hLAH : LiveAtHome Luke := A5 Luke hST
  have hNotLWS : ¬ LiveWithStrangers Luke := A4 Luke hLAH
  exact hNotLWS hLWS

-- Hence Luke does not live with strangers
theorem luke_not_live_with_strangers : ¬ LiveWithStrangers Luke :=
by
  cases A6 with
  | inl hA =>
    have : False := not_first_disjunct hA
    exact False.elim this
  | inr hB =>
    exact hB.right
