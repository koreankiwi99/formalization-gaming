<lean>
axiom obj : Type
axiom Yuri : obj

axiom ProfessionalBasketballPlayer : obj → Prop
axiom American : obj → Prop
axiom CanJumpShots : obj → Prop
axiom LeapsStraight : obj → Prop
axiom ActivatesLegMuscles : obj → Prop

-- Context axioms
axiom SomePBPNotAmerican : ∃ x : obj, ProfessionalBasketballPlayer x ∧ ¬ American x
axiom PBP_implies_CanJump : ∀ x : obj, ProfessionalBasketballPlayer x → CanJumpShots x
axiom CanJump_implies_Leap : ∀ x : obj, CanJumpShots x → LeapsStraight x
axiom Leap_implies_Activate : ∀ x : obj, LeapsStraight x → ActivatesLegMuscles x
axiom Yuri_not_Activate : ¬ ActivatesLegMuscles Yuri

-- Goal: Yuri is not an American professional basketball player
theorem yuri_not_american_pbp : ¬ (American Yuri ∧ ProfessionalBasketballPlayer Yuri) := by
  intro h
  have hp : ProfessionalBasketballPlayer Yuri := And.right h
  have hj : CanJumpShots Yuri := PBP_implies_CanJump Yuri hp
  have hl : LeapsStraight Yuri := CanJump_implies_Leap Yuri hj
  have hm : ActivatesLegMuscles Yuri := Leap_implies_Activate Yuri hl
  exact Yuri_not_Activate hm
</lean>

ANSWER: True
