<lean>
axiom obj : Type

-- Individuals and predicates
axiom Hannah : obj
axiom AtMS : obj → Prop
axiom Work : obj → Prop
axiom Need : obj → Prop
axiom OT : obj → Prop
axiom Enjoy : obj → Prop
axiom Picky : obj → Prop
axiom Spend : obj → Prop

-- Premises
axiom R1 : ∀ x : obj, AtMS x → Work x → Need x
axiom R2 : ∀ x : obj, AtMS x → OT x → Work x
axiom R3 : ∀ x : obj, AtMS x → (OT x ∨ Enjoy x)
axiom R4 : ∀ x : obj, AtMS x → Enjoy x → ¬ Picky x
axiom R5 : ∀ x : obj, AtMS x → Enjoy x → Spend x

axiom T6 : AtMS Hannah
axiom T7w : Work Hannah
axiom T7imp : Need Hannah → (¬ Picky Hannah ∧ ¬ Need Hannah)

-- Goal: Hannah needs to earn money to help pay for her college tuition.
theorem hannah_needs_money : Need Hannah := by
  have hwork : Work Hannah := T7w
  have hneed : Need Hannah := R1 Hannah T6 hwork
  exact hneed
</lean>

ANSWER: True
