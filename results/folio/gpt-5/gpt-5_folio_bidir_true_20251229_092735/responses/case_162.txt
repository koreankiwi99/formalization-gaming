<lean>
axiom obj : Type
axiom Bird : obj → Prop
axiom Hawk : obj → Prop
axiom Swims : obj → Prop

axiom Hawks_never_swim : ∀ x : obj, Hawk x → ¬ Swims x
axiom some_bird_is_hawk : ∃ x : obj, Bird x ∧ Hawk x

-- The conclusion to evaluate is: ∀ x, Bird x → Swims x
-- This does not follow; in fact, its negation is derivable.
theorem not_all_birds_swim : ¬ (∀ x : obj, Bird x → Swims x) := by
  intro h
  obtain ⟨w, hw⟩ := some_bird_is_hawk
  have hwBird : Bird w := And.left hw
  have hwHawk : Hawk w := And.right hw
  have hns : ¬ Swims w := Hawks_never_swim w hwHawk
  have hs : Swims w := h w hwBird
  exact hns hs
