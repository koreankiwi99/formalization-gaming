<lean>
axiom obj : Type
axiom Taylor : obj

axiom EatSaladsRegularly : obj → Prop
axiom VeryConsciousAboutHealthAndEatingHabits : obj → Prop
axiom GrewUpInHealthConsciousChildhoodHome : obj → Prop
axiom FulfillDailyNutritionalIntakes : obj → Prop
axiom DisregardPhysicalWellBeing : obj → Prop
axiom VisitGymAtLeastOnceADay : obj → Prop

axiom A1 : ∀ x : obj, EatSaladsRegularly x → VeryConsciousAboutHealthAndEatingHabits x
axiom A2 : ∀ x : obj, GrewUpInHealthConsciousChildhoodHome x → EatSaladsRegularly x
axiom A3 : ∀ x : obj, FulfillDailyNutritionalIntakes x → GrewUpInHealthConsciousChildhoodHome x
axiom A4 : ∀ x : obj, DisregardPhysicalWellBeing x → ¬ VeryConsciousAboutHealthAndEatingHabits x
axiom A5 : ∀ x : obj, VisitGymAtLeastOnceADay x → FulfillDailyNutritionalIntakes x
axiom A6 : (GrewUpInHealthConsciousChildhoodHome Taylor ∧ DisregardPhysicalWellBeing Taylor) ∨ (¬ GrewUpInHealthConsciousChildhoodHome Taylor ∧ ¬ DisregardPhysicalWellBeing Taylor)

theorem taylor_neither_grew_nor_visits : (¬ GrewUpInHealthConsciousChildhoodHome Taylor) ∧ (¬ VisitGymAtLeastOnceADay Taylor) := by
  have h_not_both : ¬ (GrewUpInHealthConsciousChildhoodHome Taylor ∧ DisregardPhysicalWellBeing Taylor) := by
    intro h
    cases h with
    | intro hG hD =>
      have hEat : EatSaladsRegularly Taylor := A2 Taylor hG
      have hVC : VeryConsciousAboutHealthAndEatingHabits Taylor := A1 Taylor hEat
      have hNotVC : ¬ VeryConsciousAboutHealthAndEatingHabits Taylor := A4 Taylor hD
      exact hNotVC hVC
  have hNotG : ¬ GrewUpInHealthConsciousChildhoodHome Taylor := by
    cases A6 with
    | inl hBoth =>
      exact False.elim (h_not_both hBoth)
    | inr hNeither =>
      exact hNeither.left
  have hVisitImpliesGrewUp : VisitGymAtLeastOnceADay Taylor → GrewUpInHealthConsciousChildhoodHome Taylor := fun hVisit => by
    have hFulfill : FulfillDailyNutritionalIntakes Taylor := A5 Taylor hVisit
    exact A3 Taylor hFulfill
  have hNotVisit : ¬ VisitGymAtLeastOnceADay Taylor := fun hVisit => hNotG (hVisitImpliesGrewUp hVisit)
  exact And.intro hNotG hNotVisit
</lean>

ANSWER: True
