<lean>
axiom obj : Type
axiom GRETest : obj
axiom Cost205 : obj → Prop
axiom CheaperThan300 : obj → Prop
axiom CostBelow300 : obj → Prop
axiom T1 : Cost205 GRETest
axiom R1 : ∀ x : obj, Cost205 x → CheaperThan300 x
axiom R2 : ∀ x : obj, CheaperThan300 x → CostBelow300 x

theorem gre_cost_below_300 : CostBelow300 GRETest := by
  have hcheap : CheaperThan300 GRETest := R1 GRETest T1
  exact R2 GRETest hcheap
</lean>

ANSWER: True
