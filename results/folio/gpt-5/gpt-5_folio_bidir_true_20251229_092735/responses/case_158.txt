<lean>
axiom obj : Type
axiom James : obj
axiom WorksLibrary : obj → Prop
axiom CS : obj → Prop
axiom UniPT : obj → Prop
axiom DB : obj → Prop
axiom WithDavid : obj → Prop
axiom Lab : obj → Prop

axiom A1 : ∀ x : obj, WorksLibrary x → ¬ CS x
axiom A2 : ∀ x : obj, UniPT x → WorksLibrary x
axiom A3 : ∀ x : obj, DB x → CS x
axiom A4 : ∀ x : obj, WithDavid x → DB x
axiom A5 : ∀ x : obj, Lab x → WithDavid x
axiom A6 : Lab James
axiom A7 : ¬ Lab James ∧ ¬ UniPT James

theorem james_db_or_unipt : DB James ∨ UniPT James := by
  left
  have hWD : WithDavid James := A5 James A6
  exact A4 James hWD
</lean>

ANSWER: True
