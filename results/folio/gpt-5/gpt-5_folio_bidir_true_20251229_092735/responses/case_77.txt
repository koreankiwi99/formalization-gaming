<lean>
axiom obj : Type

axiom At : obj → Prop
axiom Work : obj → Prop
axiom Need : obj → Prop
axiom OrderTakeout : obj → Prop
axiom EnjoyMeals : obj → Prop
axiom SpendDiningHalls : obj → Prop
axiom Picky : obj → Prop

axiom R1 : ∀ x : obj, At x → Work x → Need x
axiom R2 : ∀ x : obj, At x → OrderTakeout x → Work x
axiom R3 : ∀ x : obj, At x → (OrderTakeout x ∨ EnjoyMeals x)
axiom R4 : ∀ x : obj, At x → EnjoyMeals x → ¬ Picky x
axiom R5 : ∀ x : obj, At x → EnjoyMeals x → SpendDiningHalls x

axiom Hannah : obj
axiom T6 : At Hannah
axiom T7_work : Work Hannah
axiom T7_cond : Need Hannah → (¬ Picky Hannah ∧ ¬ Need Hannah)

theorem hannah_statement :
  At Hannah ∧ (¬ Picky Hannah ∨ (Picky Hannah → SpendDiningHalls Hannah)) := by
  have hNeed : Need Hannah := R1 Hannah T6 T7_work
  have hNotPicky : ¬ Picky Hannah := And.left (T7_cond hNeed)
  exact And.intro T6 (Or.inl hNotPicky)
</lean>

ANSWER: True
