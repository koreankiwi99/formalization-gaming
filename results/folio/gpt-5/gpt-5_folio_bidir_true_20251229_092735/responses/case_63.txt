axiom Cow : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, BrownSwissCattle x → Cow x                 -- All Brown Swiss cattle are cows
axiom A2 : ∃ x : obj, Pet x ∧ BrownSwissCattle x                 -- Some pets are Brown Swiss cattle
axiom A3 : ∀ x : obj, Cow x → Domesticated x                     -- All cows are domesticated animals
axiom A4 : ∀ x : obj, Alligator x → ¬ Domesticated x             -- Alligators are not domesticated animals
axiom A5 : Alligator Ted                                         -- Ted is an alligator

-- From the premises, Ted cannot be Brown Swiss cattle (would imply domesticated, contradicting A4 and A5)
theorem not_brownSwiss_Ted : ¬ BrownSwissCattle Ted := by
  intro hbs
  have hCow : Cow Ted := A1 Ted hbs
  have hDom : Domesticated Ted := A3 Ted hCow
  have hNotDom : ¬ Domesticated Ted := A4 Ted A5
  exact hNotDom hDom

-- Therefore, the conjunction "Ted is a pet and Brown Swiss cattle" is impossible
theorem not_statement : ¬ (Pet Ted ∧ BrownSwissCattle Ted) := by
  intro h
  have hbs : BrownSwissCattle Ted := And.right h
  exact not_brownSwiss_Ted hbs
