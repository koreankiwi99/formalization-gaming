<lean>
axiom obj : Type
axiom James : obj

axiom Student : obj → Prop
axiom WorksInLibrary : obj → Prop
axiom FromCSDept : obj → Prop
axiom HasUnivPartTimeJob : obj → Prop
axiom TakingDatabaseCourse : obj → Prop
axiom TakingClassWithDavid : obj → Prop
axiom WorkingInLab : obj → Prop

-- None of the students who currently work in the library are from the CS department.
axiom A1 : ∀ x : obj, Student x ∧ WorksInLibrary x → ¬ FromCSDept x
-- All of the students with university part-time jobs currently work in the library.
axiom A2 : ∀ x : obj, Student x ∧ HasUnivPartTimeJob x → WorksInLibrary x
-- All of the students who are taking the database course are from the CS department.
axiom A3 : ∀ x : obj, Student x ∧ TakingDatabaseCourse x → FromCSDept x
-- All of the students taking a class with Professor David are taking the database course.
axiom A4 : ∀ x : obj, Student x ∧ TakingClassWithDavid x → TakingDatabaseCourse x
-- All students working in the lab are taking a class with Professor David.
axiom A5 : ∀ x : obj, Student x ∧ WorkingInLab x → TakingClassWithDavid x
-- James is a student working in the lab.
axiom A6 : Student James ∧ WorkingInLab James
-- James doesn't work in the lab or have a part-time job offered by the university. (included as a disjunction to mirror the wording; not needed for the proof)
axiom A7 : ¬ WorkingInLab James ∨ ¬ HasUnivPartTimeJob James

theorem james_no_univ_part_time_job : ¬ HasUnivPartTimeJob James := by
  intro hPT
  have hStu : Student James := And.left A6
  have hLab : WorkingInLab James := And.right A6
  have hDavid : TakingClassWithDavid James := A5 James (And.intro hStu hLab)
  have hDB : TakingDatabaseCourse James := A4 James (And.intro hStu hDavid)
  have hCS : FromCSDept James := A3 James (And.intro hStu hDB)
  have hLib : WorksInLibrary James := A2 James (And.intro hStu hPT)
  have hNotCS : ¬ FromCSDept James := A1 James (And.intro hStu hLib)
  exact hNotCS hCS
</lean>

ANSWER: True
