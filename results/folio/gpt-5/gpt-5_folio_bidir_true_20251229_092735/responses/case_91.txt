axiom Task : Type
axiom MachineTranslation : Task

axiom NLP : Task → Prop
axiom Generation : Task → Prop
axiom Understanding : Task → Prop
axiom TextSeqOutput : Task → Prop

-- A natural language processing task is either generation or understanding
axiom A1 : ∀ t : Task, NLP t → (Generation t ∨ Understanding t)
-- If an NLP task outputs a text sequence, then it is a generation task
axiom A2 : ∀ t : Task, NLP t → (TextSeqOutput t → Generation t)
-- Machine translation is an NLP task
axiom A3 : NLP MachineTranslation
-- The output of machine translation is a text sequence
axiom A4 : TextSeqOutput MachineTranslation

-- We can prove it is a generation task
theorem MT_is_generation : Generation MachineTranslation := by
  have h := A2 MachineTranslation
  have h' := h A3
  exact h' A4
