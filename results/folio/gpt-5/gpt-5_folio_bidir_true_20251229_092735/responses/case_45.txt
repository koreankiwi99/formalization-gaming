axiom Mammal : obj → Prop
axiom Pet : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, Cat x → Mammal x
axiom A2 : ∃ x : obj, Pet x ∧ ¬ Mammal x

-- From the premises we can derive: Some pet is not a cat.
theorem some_pet_not_cat : ∃ x : obj, Pet x ∧ ¬ Cat x :=
  Exists.elim A2 (fun x hx =>
    Exists.intro x (
      And.intro (And.left hx)
        (fun hxCat => (And.right hx) (A1 x hxCat))
    )
  )

-- The stronger statement "No pets are cats", i.e. ∀ x, Pet x → ¬ Cat x, does not follow from the premises.
