<lean>
axiom obj : Type

-- Entities
axiom SirDigby : obj
axiom TheMovie : obj
axiom nemesis : obj → obj

-- Predicates
axiom SuperheroMovie : obj → Prop
axiom NamedAfter : obj → obj → Prop
axiom InMovie : obj → obj → Prop
axiom GoodGuy : obj → Prop
axiom BadGuy : obj → Prop
axiom Fights : obj → obj → Prop
axiom Wins : obj → Prop

-- Facts from the context
axiom T1 : SuperheroMovie TheMovie
axiom T2 : NamedAfter TheMovie SirDigby
axiom Tfight : Fights SirDigby (nemesis SirDigby)

-- Rules from the context
-- In superhero movies, the good guys always win.
axiom R1 : ∀ m x, SuperheroMovie m → InMovie x m → GoodGuy x → Wins x
-- If a superhero movie is named after a character, that character is a good guy.
axiom R2 : ∀ m x, SuperheroMovie m → NamedAfter m x → GoodGuy x
-- If a superhero movie is named after a character, that character is in the movie.
axiom R3 : ∀ m x, SuperheroMovie m → NamedAfter m x → InMovie x m
-- If somebody wins a fight, the person they are fighting does not win.
axiom R4 : ∀ x y, Wins x → Fights x y → ¬ Wins y

theorem nemesis_does_not_win : ¬ Wins (nemesis SirDigby) := by
  have hGood : GoodGuy SirDigby := R2 TheMovie SirDigby T1 T2
  have hIn : InMovie SirDigby TheMovie := R3 TheMovie SirDigby T1 T2
  have hWin : Wins SirDigby := R1 TheMovie SirDigby T1 hIn hGood
  have hFight : Fights SirDigby (nemesis SirDigby) := Tfight
  exact R4 SirDigby (nemesis SirDigby) hWin hFight
</lean>

ANSWER: True
