axiom obj : Type
axiom Rockie : obj
axiom Rabbit : obj → Prop
axiom Squirrel : obj → Prop
axiom Turtle : obj → Prop
axiom Spotted : obj → Prop
axiom Cute : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, Rabbit x → Spotted x → Cute x
axiom A2 : ∃ x : obj, Turtle x ∧ Spotted x
axiom A3 : ∀ x : obj, Spotted x → (Rabbit x ∨ Squirrel x)
axiom A4 : ∀ x : obj, Skittish x → ¬ Calm x
axiom A5 : ∀ x : obj, Squirrel x → Spotted x → Skittish x
axiom A6 : Spotted Rockie ∧ Calm Rockie

-- From premises, Rockie is not a squirrel
theorem not_squirrel_rockie : ¬ Squirrel Rockie :=
by
  intro hsq
  have hS : Spotted Rockie := And.left A6
  have sk : Skittish Rockie := A5 Rockie hsq hS
  have nc : ¬ Calm Rockie := A4 Rockie sk
  exact nc (And.right A6)

-- Therefore Rockie is a rabbit
theorem rabbit_rockie : Rabbit Rockie :=
by
  have hS : Spotted Rockie := And.left A6
  have h : Rabbit Rockie ∨ Squirrel Rockie := A3 Rockie hS
  exact Or.resolve_right h not_squirrel_rockie
