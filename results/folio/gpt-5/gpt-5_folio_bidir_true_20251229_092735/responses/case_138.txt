axiom used_requires_label : ∀ a : Algorithm, UsedToTrain SOTA a → RequiresLabeled a

-- From the axioms we can derive its negation:
theorem no_unsupervised_used : ¬ (∃ a : Algorithm, Unsupervised a ∧ UsedToTrain SOTA a) :=
  fun h =>
    match h with
    | ⟨a, ⟨hU, hUsed⟩⟩ =>
      let hReq : RequiresLabeled a := used_requires_label a hUsed
      let hNoReq : ¬ RequiresLabeled a := unsup_no_label a hU
