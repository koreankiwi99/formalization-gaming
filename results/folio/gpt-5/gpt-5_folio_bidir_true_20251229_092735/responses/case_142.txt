<lean>
axiom obj : Type

axiom Animal : obj → Prop
axiom HeckCattle : obj → Prop
axiom Aurochs : obj → Prop
axiom Extinct : obj → Prop
axiom Resembles : obj → obj → Prop
axiom BredBack : obj → Prop
axiom ArtificiallySelected : obj → Prop

-- Breeding back is a form of artificial selection
axiom R1 : ∀ x : obj, BredBack x → ArtificiallySelected x

-- Heck cattle were bred back (witnessing existence)
axiom T1 : ∃ x : obj, HeckCattle x ∧ BredBack x

-- Heck cattle are animals
axiom T2 : ∀ x : obj, HeckCattle x → Animal x

-- Aurochs are animals
axiom T3 : ∀ x : obj, Aurochs x → Animal x

-- Some animals to be bred back resemble extinct animals (modeled via existence + resemblance to some extinct animal)
axiom T4 : ∃ x : obj, Animal x ∧ (∃ y : obj, Extinct y ∧ Resembles x y)

theorem some_heck_artificial : ∃ x : obj, HeckCattle x ∧ ArtificiallySelected x :=
by
  cases T1 with
  | intro x hx =>
    cases hx with
    | intro hxHC hxBB =>
      have hAS : ArtificiallySelected x := R1 x hxBB
      exact Exists.intro x (And.intro hxHC hAS)
</lean>

ANSWER: True
