axiom obj : Type
axiom Taylor : obj

axiom EatSalads : obj → Prop
axiom VeryConscious : obj → Prop
axiom HealthHome : obj → Prop
axiom Fulfill : obj → Prop
axiom Disregard : obj → Prop
axiom GymDaily : obj → Prop

-- All people who eat salads regularly are very conscious about their health and eating habits.
axiom A1 : ∀ x : obj, EatSalads x → VeryConscious x
-- All people who grew up in health-conscious childhood homes eat salads regularly.
axiom A2 : ∀ x : obj, HealthHome x → EatSalads x
-- All people who fulfill their daily nutritional intakes grew up in health-conscious childhood homes.
axiom A3 : ∀ x : obj, Fulfill x → HealthHome x
-- All people who disregard their physical well-being are not very conscious about their health and eating habits.
axiom A4 : ∀ x : obj, Disregard x → ¬ VeryConscious x
-- If people visit the gym at least once a day, then they always fulfill their daily nutritional intakes.
axiom A5 : ∀ x : obj, GymDaily x → Fulfill x
-- Taylor either grew up in a health-conscious childhood home and disregards her physical well-being, or she did neither.
axiom A6 : (HealthHome Taylor ∧ Disregard Taylor) ∨ (¬ HealthHome Taylor ∧ ¬ Disregard Taylor)

-- From the premises we can derive that Taylor does not visit the gym at least once a day.
theorem taylor_not_gym : ¬ GymDaily Taylor := by
  -- Show the first disjunct leads to a contradiction
  have hnotAB : ¬ (HealthHome Taylor ∧ Disregard Taylor) := by
    intro hab
    have hA : HealthHome Taylor := hab.left
    have hB : Disregard Taylor := hab.right
    have e : EatSalads Taylor := A2 Taylor hA
    have v : VeryConscious Taylor := A1 Taylor e
    have nv : ¬ VeryConscious Taylor := A4 Taylor hB
    exact nv v
  -- Hence, from the disjunction we obtain the second conjunct
  have hneither : ¬ HealthHome Taylor ∧ ¬ Disregard Taylor := by
    cases A6 with
    | inl hab =>
        exact False.elim (hnotAB hab)
    | inr h =>
        exact h
  have nhA : ¬ HealthHome Taylor := hneither.left
  -- From ¬HealthHome and (Fulfill → HealthHome), we get ¬Fulfill
  have nFul : ¬ Fulfill Taylor := by
    intro hFul
    exact nhA (A3 Taylor hFul)
  -- From (GymDaily → Fulfill) and ¬Fulfill, we get ¬GymDaily
  intro hGym
  exact nFul (A5 Taylor hGym)
</lean>
