-- Derived: Born in a multiple birth implies not living with strangers
theorem born_multiple_implies_not_strangers : ∀ x : obj, BornMultiple x → ¬ LiveWithStrangers x :=
  fun x hBM => by
    have hSpend : SpendTimeWithSiblings x := A1 x hBM
    have hHome : LiveAtHome x := A5 x hSpend
    exact A4 x hHome

-- From the disjunction and the derived lemma, we can show Luke does not live with strangers.
theorem luke_not_live_with_strangers : ¬ LiveWithStrangers Luke := by
  cases A6 with
  | inl h1 =>
      have hBM : BornMultiple Luke := And.left (And.right h1)
      exact born_multiple_implies_not_strangers Luke hBM
  | inr h2 =>
      exact And.right h2
