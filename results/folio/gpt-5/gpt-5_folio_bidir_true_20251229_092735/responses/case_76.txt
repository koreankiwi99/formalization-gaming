<lean>
axiom obj : Type
axiom Hannah : obj

axiom AtSchool : obj → Prop
axiom Work : obj → Prop
axiom NeedMoney : obj → Prop
axiom OrderTakeout : obj → Prop
axiom EnjoyDiningHall : obj → Prop
axiom NotPicky : obj → Prop
axiom SpendTimeDining : obj → Prop

-- If people at Mary's school work in student jobs on campus, then they need to earn money
axiom R1 : ∀ x : obj, AtSchool x → Work x → NeedMoney x
-- If people at Mary's school order takeout frequently in college, then they work in student jobs
axiom R2 : ∀ x : obj, AtSchool x → OrderTakeout x → Work x
-- People at Mary's school order takeout frequently in college or enjoy the dining hall meals and recipes
axiom R3 : ∀ x : obj, AtSchool x → (OrderTakeout x ∨ EnjoyDiningHall x)
-- If people at Mary's school enjoy the dining hall meals and recipes, then they are not picky eaters
axiom R4 : ∀ x : obj, AtSchool x → EnjoyDiningHall x → NotPicky x
-- If people at Mary's school enjoy the dining hall meals and recipes, then they spend a lot of their time in the dining halls
axiom R5 : ∀ x : obj, AtSchool x → EnjoyDiningHall x → SpendTimeDining x

-- Hannah is at Mary's school
axiom T6 : AtSchool Hannah
-- Hannah works; and if she needs money, then she is not picky and does not need money
axiom T7 : Work Hannah ∧ (NeedMoney Hannah → (NotPicky Hannah ∧ ¬ NeedMoney Hannah))

theorem conclude :
  AtSchool Hannah ∧ NotPicky Hannah ∧ SpendTimeDining Hannah := by
  have AtH : AtSchool Hannah := T6
  have hWorkAndImp : Work Hannah ∧ (NeedMoney Hannah → (NotPicky Hannah ∧ ¬ NeedMoney Hannah)) := T7
  have WorkH : Work Hannah := And.left hWorkAndImp
  have NeedH : NeedMoney Hannah := R1 Hannah AtH WorkH
  have hImp := And.right hWorkAndImp
  have hNP_and_notNeed : NotPicky Hannah ∧ ¬ NeedMoney Hannah := hImp NeedH
  have NotPickyH : NotPicky Hannah := And.left hNP_and_notNeed
  have NotNeedH : ¬ NeedMoney Hannah := And.right hNP_and_notNeed
  have contra : False := NotNeedH NeedH
  have SpendH : SpendTimeDining Hannah := False.elim contra
  exact And.intro AtH (And.intro NotPickyH SpendH)
</lean>

ANSWER: True
