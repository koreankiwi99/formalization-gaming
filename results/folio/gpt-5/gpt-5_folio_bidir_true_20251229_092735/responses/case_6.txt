<lean>
axiom obj : Type

axiom Animal : obj → Prop
axiom Mammal : obj → Prop
axiom Human : obj → Prop
axiom Person : obj → Prop

axiom Fever : obj → Prop
axiom Headache : obj → Prop
axiom MusclePains : obj → Prop
axiom Tired : obj → Prop
axiom Flu : obj → Prop

axiom MonkeypoxOccurs : obj → Prop
axiom GetsMonkeypox : obj → Prop

-- When the Monkeypox virus occurs in a being, it gets Monkeypox
axiom A1 : ∀ x : obj, MonkeypoxOccurs x → GetsMonkeypox x

-- Monkeypox virus can occur in certain animals (existential witness)
axiom A_occurs_animals : ∃ x : obj, Animal x ∧ MonkeypoxOccurs x

-- Humans are mammals; mammals are animals
axiom A_hm : ∀ x : obj, Human x → Mammal x
axiom A_ma : ∀ x : obj, Mammal x → Animal x

-- Symptoms of Monkeypox include fever, headache, muscle pains, and tiredness
axiom A_symptoms : ∀ x : obj, GetsMonkeypox x → Fever x ∧ Headache x ∧ MusclePains x ∧ Tired x

-- People feel tired when they get the flu
axiom A_flu_tired : ∀ x : obj, Person x → Flu x → Tired x

-- Goal: There is an animal.
theorem exists_animal : ∃ x : obj, Animal x := by
  apply Exists.elim A_occurs_animals
  intro a h
  exact Exists.intro a (And.left h)
</lean>

ANSWER: True
