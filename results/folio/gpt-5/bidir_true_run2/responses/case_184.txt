<lean>
axiom obj : Type

-- Predicates
axiom Person : obj → Prop
axiom Book : obj → Prop
axiom Novel : obj → Prop
axiom Podcast : obj → Prop
axiom AmericanCity : obj → Prop
axiom American : obj → Prop
axiom BornIn : obj → obj → Prop
axiom Writer : obj → Prop
axiom WrittenBy : obj → obj → Prop
axiom NovelWriter : obj → Prop
axiom Created : obj → obj → Prop

-- Constants
axiom DaniShapiro : obj
axiom FamilyHistory : obj
axiom FamilySecrets : obj
axiom Boston : obj

-- Axioms from the context
axiom A1 : ∀ x : obj, Podcast x → ¬ Novel x
axiom A2 : ∀ p c : obj, Person p → AmericanCity c → BornIn p c → American p
axiom A3 : ∀ b p : obj, Book b → Novel b → Person p → WrittenBy b p → NovelWriter p
axiom A4 : American DaniShapiro
axiom A5 : Writer DaniShapiro
axiom A6 : WrittenBy FamilyHistory DaniShapiro
axiom A7 : Novel FamilyHistory
axiom A8 : ∀ x : obj, Writer x → Person x
axiom A9 : Podcast FamilySecrets
axiom A10 : Created DaniShapiro FamilySecrets
axiom A11 : AmericanCity Boston
axiom A12 : ∀ x : obj, Novel x → Book x

-- Goal: Dani Shapiro is a novel writer.
theorem dani_novel_writer : NovelWriter DaniShapiro := by
  have HF_book : Book FamilyHistory := A12 FamilyHistory A7
  have Dani_person : Person DaniShapiro := A8 DaniShapiro A5
  exact A3 FamilyHistory DaniShapiro HF_book A7 Dani_person A6
</lean>

ANSWER: True
