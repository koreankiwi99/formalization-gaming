-- Axioms from the context

axiom A2 : ∀ x : Person, ClumsyFoodieGoesOutFrequently x → Spills x
axiom A3 : ∀ x : Person, Cleanly x → Tidy x
axiom A4 : ∀ x : Person, ValuesOrderSpotlessness x → Cleanly x
axiom A5 : ∀ x : Person, FamilyPrioritizesOrderSpotlessness x → ValuesOrderSpotlessness x

-- Given disjunction about Peter
axiom Peter_either :
  (Spills Peter ∧ Cleanly Peter) ∨ (¬ Spills Peter ∧ ¬ Cleanly Peter)

-- Some derived lemmas (not sufficient to prove the target conclusion)
theorem spills_and_cleanly_impossible :
  (Spills Peter ∧ Cleanly Peter) → False := by
  intro h
  have hs : Spills Peter := h.left
  have hc : Cleanly Peter := h.right
  have ht : Tidy Peter := A3 Peter hc
  have hnt : ¬ Tidy Peter := A1 Peter hs
  exact hnt ht

theorem family_implies_cleanly_all :
  ∀ x : Person, FamilyPrioritizesOrderSpotlessness x → Cleanly x := by
  intro x hf
  have hv : ValuesOrderSpotlessness x := A5 x hf
  exact A4 x hv
