<lean>
axiom Building : Type
axiom Amt : Type

axiom Managed : Building → Prop
axiom Rent : Building → Amt
axiom Deposit : Building → Amt
axiom More : Amt → Amt → Prop

axiom OliveGarden : Building
axiom twoK : Amt
axiom fifteenHundred : Amt

-- Context axioms
axiom OG_managed : Managed OliveGarden
axiom rent_OG : Rent OliveGarden = twoK
axiom more_2000_1500 : More twoK fifteenHundred

-- The security deposit at a managed building is equal to the monthly rent or more
axiom deposit_rule : ∀ b : Building, Managed b → (Deposit b = Rent b ∨ More (Deposit b) (Rent b))

-- Statement to prove: The security deposit at the Olive Garden is either $2000 or more.
theorem deposit_OG_2000_or_more : Deposit OliveGarden = twoK ∨ More (Deposit OliveGarden) twoK := by
  have h := deposit_rule OliveGarden OG_managed
  cases h with
  | inl hEq =>
      left
      simpa [rent_OG] using hEq
  | inr hMore =>
      right
      simpa [rent_OG] using hMore
</lean>

ANSWER: True
