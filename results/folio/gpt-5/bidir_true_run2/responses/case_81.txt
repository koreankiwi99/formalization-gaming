<lean>
axiom obj : Type
axiom DTC : obj
axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom FromBBM : obj → Prop

axiom no_baked_spicy : ∀ x : obj, BakedSweet x → ¬ Spicy x
axiom cupcakes_baked : ∀ x : obj, Cupcake x → BakedSweet x
axiom mala_spicy : ∀ x : obj, MalaHotpot x → Spicy x
axiom bbm_cupcakes : ∀ x : obj, FromBBM x → Cupcake x
axiom dtc_info : Spicy DTC ∨ MalaHotpot DTC ∨ ¬ BakedSweet DTC

theorem dtc_neither_bbm_nor_baked : ¬ FromBBM DTC ∧ ¬ BakedSweet DTC :=
by
  have hb : ¬ BakedSweet DTC :=
    match dtc_info with
    | Or.inl hSp =>
        fun hbs => (no_baked_spicy DTC hbs) hSp
    | Or.inr (Or.inl hMala) =>
        let hSp : Spicy DTC := mala_spicy DTC hMala
        fun hbs => (no_baked_spicy DTC hbs) hSp
    | Or.inr (Or.inr hNotB) => hNotB
  have hnotfbm : ¬ FromBBM DTC :=
    by
      intro hfbm
      have hc : Cupcake DTC := bbm_cupcakes DTC hfbm
      have hbs : BakedSweet DTC := cupcakes_baked DTC hc
      exact hb hbs
  exact And.intro hnotfbm hb
</lean>

ANSWER: True
