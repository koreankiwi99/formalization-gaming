<lean>
axiom obj : Type
axiom Tom : obj

axiom GREApplicant : obj → Prop
axiom ProvesEconomicHardship : obj → Prop
axiom ProvidesFinancialAid : obj → Prop
axiom SingleParentFamily : obj → Prop
axiom FewResources : obj → Prop
axiom DadOutOfWork : obj → Prop

-- ETS provides financial aid to GRE applicants who prove economic hardship
axiom R1 : ∀ x : obj, GREApplicant x → ProvesEconomicHardship x → ProvidesFinancialAid x
-- Living in a single-parent family or having few resources suffices to prove economic hardship
axiom R2 : ∀ x : obj, (SingleParentFamily x ∨ FewResources x) → ProvesEconomicHardship x

-- Facts about Tom
axiom T1 : SingleParentFamily Tom
axiom T2 : DadOutOfWork Tom
axiom T3 : FewResources Tom
axiom T4 : GREApplicant Tom

theorem ets_financial_aid_Tom : ProvidesFinancialAid Tom := by
  have hEH : ProvesEconomicHardship Tom := R2 Tom (Or.inl T1)
  exact R1 Tom T4 hEH
</lean>

ANSWER: True
