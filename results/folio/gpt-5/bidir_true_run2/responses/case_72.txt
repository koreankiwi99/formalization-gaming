<lean>
axiom obj : Type
axiom KO : obj

axiom Growth : obj → Prop
axiom BoughtRapid : obj → Prop
axiom Suitable : obj → Prop
axiom Mature : obj → Prop
axiom Stock : obj → Prop
axiom Volatile : obj → Prop

-- All growth stocks are bought to earn profits from rapid price appreciation.
axiom R1 : ∀ x : obj, Growth x → BoughtRapid x
-- If a stock is bought to earn profits from rapid price appreciation, then it is not suitable for a retirement fund.
axiom R2 : ∀ x : obj, BoughtRapid x → ¬ Suitable x
-- Some stocks are growth stocks.
axiom E1 : ∃ x : obj, Stock x ∧ Growth x
-- All mature stocks are suitable for a retirement fund.
axiom R3 : ∀ x : obj, Stock x ∧ Mature x → Suitable x
-- KO is a mature stock (both stock and mature).
axiom T1 : Stock KO
axiom T2 : Mature KO

theorem target : (Growth KO ∨ BoughtRapid KO) → (¬ Stock KO ∧ ¬ Volatile KO) :=
by
  intro h
  have contra : False :=
    by
      cases h with
      | inl g =>
          have br : BoughtRapid KO := R1 KO g
          have ns : ¬ Suitable KO := R2 KO br
          have s : Suitable KO := R3 KO (And.intro T1 T2)
          exact ns s
      | inr br =>
          have ns : ¬ Suitable KO := R2 KO br
          have s : Suitable KO := R3 KO (And.intro T1 T2)
          exact ns s
  have notStockKO : ¬ Stock KO := by
    intro _
    exact False.elim contra
  have notVolKO : ¬ Volatile KO := by
    intro _
    exact False.elim contra
  exact And.intro notStockKO notVolKO
</lean>

ANSWER: True
