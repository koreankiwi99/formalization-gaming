<lean>
axiom obj : Type
axiom Beijing : obj

-- Predicates for hosting specific events
axiom Hosted2008SummerOlympics : obj → Prop
axiom HostedWinterOlympics : obj → Prop

-- Facts from the context
axiom H1 : Hosted2008SummerOlympics Beijing
axiom H2 : HostedWinterOlympics Beijing

-- Conclusion: Beijing hosted both the 2008 Summer Olympics and the Winter Olympics
theorem beijing_hosted_both : Hosted2008SummerOlympics Beijing ∧ HostedWinterOlympics Beijing :=
  And.intro H1 H2
</lean>

ANSWER: True
