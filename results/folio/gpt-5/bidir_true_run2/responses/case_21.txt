<lean>
axiom obj : Type

axiom ST : obj
axiom BM : obj

axiom Popular : obj → Prop
axiom Netflix : obj → Prop
axiom Binge : obj → Prop
axiom Download : obj → Prop
axiom ShareLisa : obj → Prop

-- "Stranger Things" is a popular Netflix show.
axiom ST_pop : Popular ST
axiom ST_net : Netflix ST

-- If a Netflix show is popular, Karen will binge-watch it.
axiom R1 : ∀ x : obj, Netflix x → Popular x → Binge x

-- If and only if Karen binge-watches a Netflix show, she will download it.
axiom R2 : ∀ x : obj, Netflix x → (Binge x ↔ Download x)

-- Karen does not download "Black Mirror." And "Black Mirror" is a Netflix show.
axiom BM_not_dl : ¬ Download BM
axiom BM_net : Netflix BM

-- If Karen binge-watches a Netflix show, she will share it with Lisa.
axiom R3 : ∀ x : obj, Netflix x → Binge x → ShareLisa x

-- Goal: Karen will share "Stranger Things" with Lisa.
theorem share_ST : ShareLisa ST := by
  have b : Binge ST := R1 ST ST_net ST_pop
  exact R3 ST ST_net b
</lean>

ANSWER: True
