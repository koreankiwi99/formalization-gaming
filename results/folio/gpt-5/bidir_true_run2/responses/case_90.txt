<lean>
axiom obj : Type
axiom MachineTranslation : obj
axiom NLPTask : obj → Prop
axiom LanguageGeneration : obj → Prop
axiom LanguageUnderstanding : obj → Prop
axiom OutputTextSeq : obj → Prop

-- A natural language processing task can be categorized as either generation or understanding
axiom R0 : ∀ x : obj, NLPTask x → (LanguageGeneration x ∨ LanguageUnderstanding x)
-- If the output of a natural language processing task is a text sequence, then it is a language generation task
axiom R1 : ∀ x : obj, NLPTask x → OutputTextSeq x → LanguageGeneration x

-- Facts about machine translation
axiom T1 : NLPTask MachineTranslation
axiom T2 : OutputTextSeq MachineTranslation

theorem mt_is_generation : LanguageGeneration MachineTranslation := by
  exact R1 MachineTranslation T1 T2
</lean>

ANSWER: True
