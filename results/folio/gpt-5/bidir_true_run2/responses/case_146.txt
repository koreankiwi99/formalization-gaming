<lean>
axiom obj : Type

-- Individuals
axiom Ekaterina : obj
axiom Vladimir : obj

-- Predicates
axiom CanRegisterUS : obj → Prop
axiom CanParticipateUS2024 : obj → Prop
axiom USCitizenship : obj → Prop
axiom TaiwaneseCitizenship : obj → Prop
axiom RussianOfficial : obj → Prop
axiom ManagerGazprom : obj → Prop

-- Axioms from the context
axiom A1 : ∀ x : obj, CanRegisterUS x → CanParticipateUS2024 x
axiom A2 : ∀ x : obj, USCitizenship x → CanRegisterUS x
axiom A3 : ∀ x : obj, USCitizenship x ∨ TaiwaneseCitizenship x
axiom A4 : ∀ x : obj, RussianOfficial x → ¬ TaiwaneseCitizenship x
axiom A5a : ¬ TaiwaneseCitizenship Vladimir
axiom A5b : ¬ ManagerGazprom Vladimir
axiom A6 : CanRegisterUS Ekaterina ∨ RussianOfficial Ekaterina

theorem ekaterina_participates_or_manager :
  CanParticipateUS2024 Ekaterina ∨ ManagerGazprom Ekaterina := by
  apply Or.inl
  -- Prove Ekaterina can participate
  cases A6 with
  | inl hReg =>
      exact A1 Ekaterina hReg
  | inr hRO =>
      have hNotTw : ¬ TaiwaneseCitizenship Ekaterina := A4 Ekaterina hRO
      have hCitOrTw : USCitizenship Ekaterina ∨ TaiwaneseCitizenship Ekaterina := A3 Ekaterina
      cases hCitOrTw with
      | inl hUS =>
          exact A1 Ekaterina (A2 Ekaterina hUS)
      | inr hTw =>
          exact False.elim (hNotTw hTw)
</lean>

ANSWER: True
