<lean>
axiom obj : Type
axiom James : obj

axiom WorksInLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom PartTimeUniJob : obj → Prop
axiom TakesDB : obj → Prop
axiom TakesClassWithDavid : obj → Prop
axiom WorksInLab : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, WorksInLibrary x → ¬ FromCS x
axiom A2 : ∀ x : obj, PartTimeUniJob x → WorksInLibrary x
axiom A3 : ∀ x : obj, TakesDB x → FromCS x
axiom A4 : ∀ x : obj, TakesClassWithDavid x → TakesDB x
axiom A5 : ∀ x : obj, WorksInLab x → TakesClassWithDavid x
axiom A6 : WorksInLab James
axiom A7 : ¬ (WorksInLab James ∨ PartTimeUniJob James)

theorem james_takes_db : TakesDB James := by
  have hDavid : TakesClassWithDavid James := A5 James A6
  exact A4 James hDavid
</lean>

ANSWER: True
