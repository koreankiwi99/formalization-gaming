axiom obj : Type
axiom Human : obj → Prop
axiom Horse : obj → Prop
axiom Hooves : obj → Prop

-- Premises
axiom AllHorseHaveHooves : ∀ x : obj, Horse x → Hooves x
axiom NoHumanHasHooves : ∀ x : obj, Human x → ¬ Hooves x

-- From the premises, no individual can be both a human and a horse
theorem no_human_is_horse_at_all : ∀ x : obj, ¬ (Human x ∧ Horse x) := by
  intro x
  intro h
  cases h with
  | intro hHuman hHorse =>
      have hHooves : Hooves x := AllHorseHaveHooves x hHorse
      have nHooves : ¬ Hooves x := NoHumanHasHooves x hHuman
      exact nHooves hHooves

-- Hence, there does not exist a human who is a horse
theorem not_some_humans_are_horses : ¬ (∃ x : obj, Human x ∧ Horse x) := by
  intro h
  cases h with
  | intro x hx =>
      exact (no_human_is_horse_at_all x) hx
