axiom DeadlyDisease : obj → Prop
axiom LowSurvival : obj → Prop
axiom SevereCancer : obj → Prop
axiom BileDuctCancer : obj → Prop
axiom Cholangiocarcinoma : obj → Prop

-- Entities
axiom MildFlu : obj
axiom ColorectalCancer : obj

-- Axioms from the context
axiom A1 : ∀ x : obj, DeadlyDisease x → LowSurvival x
axiom A2 : ∀ x : obj, SevereCancer x → DeadlyDisease x
axiom A3 : ∀ x : obj, BileDuctCancer x → SevereCancer x
axiom A4 : ∀ x : obj, Cholangiocarcinoma x → BileDuctCancer x
axiom A5 : LowSurvival MildFlu
axiom A6 : ¬ (BileDuctCancer ColorectalCancer ∧ LowSurvival ColorectalCancer)

-- Useful derived lemma: bile duct cancer implies low survival
theorem bileduct_implies_low : ∀ x : obj, BileDuctCancer x → LowSurvival x := by
  intro x
  intro hb
  have hs : SevereCancer x := A3 x hb
  have hd : DeadlyDisease x := A2 x hs
  have hl : LowSurvival x := A1 x hd
  exact hl

-- From A6 and the above, we can show ColorectalCancer is not a bile duct cancer
theorem not_bile_colorectal : ¬ BileDuctCancer ColorectalCancer := by
  intro hb
  have hl : LowSurvival ColorectalCancer := bileduct_implies_low ColorectalCancer hb
  have hAnd : BileDuctCancer ColorectalCancer ∧ LowSurvival ColorectalCancer := And.intro hb hl
  exact A6 hAnd
