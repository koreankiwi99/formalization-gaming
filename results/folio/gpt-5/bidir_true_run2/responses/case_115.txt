<lean>
-- Entities and predicates
axiom obj : Type
axiom Man : obj → Prop
axiom Taller : obj → obj → Prop
axiom Block : obj → obj → Prop        -- Block x y means x can block y's shooting
axiom JumpWhenShooting : obj → Prop
axiom GreatShooter : obj → Prop

-- Specific individuals
axiom Michael : obj
axiom Peter : obj
axiom Windy : obj

-- Context axioms
-- 1) If a man is taller than another man, the taller one can block the other's shooting.
axiom R_tall_block : ∀ x y : obj, Man x → Man y → Taller x y → Block x y

-- 2) Michael is a man who is taller than everyone else in his class.
axiom Michael_man : Man Michael
axiom ClassmateOfMichael : obj → Prop
axiom Michael_taller_class : ∀ y : obj, ClassmateOfMichael y → Taller Michael y

-- 3) Transitivity of taller
axiom Taller_trans : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- 4) Peter is a man who is taller than Michael.
axiom Peter_man : Man Peter
axiom Peter_taller_Michael : Taller Peter Michael

-- 5) Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom Michael_blocks_if_not_jump : ∀ y : obj, ¬ JumpWhenShooting y → Block Michael y

-- 6) Michael cannot block Windy's shooting.
axiom Michael_cannot_block_Windy : ¬ Block Michael Windy

-- 7) Every shooter who can jump when shooting is a great shooter.
-- We encode "can jump when shooting" as the property JumpWhenShooting.
axiom Jump_implies_great : ∀ x : obj, JumpWhenShooting x → GreatShooter x

open Classical

theorem Windy_is_great_shooter : GreatShooter Windy := by
  -- From Michael cannot block Windy and Michael blocks all non-jumpers,
  -- deduce that Windy jumps when shooting.
  have Windy_jumps : JumpWhenShooting Windy := by
    by_contra h_not_jump
    apply Michael_cannot_block_Windy
    exact Michael_blocks_if_not_jump Windy h_not_jump
  -- Then apply the rule that anyone who jumps when shooting is a great shooter.
  exact Jump_implies_great Windy Windy_jumps
</lean>

ANSWER: True

  classical
  have Windy_jumps : JumpWhenShooting Windy := by
    by_cases h : JumpWhenShooting Windy
    · exact h
    ·
      have hb : Block Michael Windy := Michael_blocks_if_not_jump Windy h
      exact False.elim (Michael_cannot_block_Windy hb)
  exact Jump_implies_great Windy Windy_jumps
