<lean>
axiom obj : Type
axiom Tom : obj

axiom WildTurkey : obj → Prop
axiom Eastern : obj → Prop
axiom Osceola : obj → Prop
axiom Goulds : obj → Prop
axiom Merriams : obj → Prop
axiom RioGrande : obj → Prop
axiom Ocellated : obj → Prop

-- Facts about Tom
axiom NotEastern : ¬ Eastern Tom
axiom NotOsceola : ¬ Osceola Tom
axiom NotGoulds : ¬ Goulds Tom
axiom NotMerriams : ¬ Merriams Tom
axiom NotRioGrande : ¬ RioGrande Tom
axiom TomWild : WildTurkey Tom

-- Classification: every wild turkey is one of the six types
axiom SixTypes :
  ∀ x : obj,
    WildTurkey x →
    (Eastern x ∨ Osceola x ∨ Goulds x ∨ Merriams x ∨ RioGrande x ∨ Ocellated x)

theorem tom_is_ocellated : Ocellated Tom := by
  have h := SixTypes Tom TomWild
  cases h with
  | inl hE =>
      exact False.elim (NotEastern hE)
  | inr h1 =>
      cases h1 with
      | inl hO =>
          exact False.elim (NotOsceola hO)
      | inr h2 =>
          cases h2 with
          | inl hG =>
              exact False.elim (NotGoulds hG)
          | inr h3 =>
              cases h3 with
              | inl hM =>
                  exact False.elim (NotMerriams hM)
              | inr h4 =>
                  cases h4 with
                  | inl hR =>
                      exact False.elim (NotRioGrande hR)
                  | inr hOc =>
                      exact hOc
</lean>

ANSWER: True
