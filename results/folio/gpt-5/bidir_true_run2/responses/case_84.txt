<lean>
axiom obj : Type
axiom HPS : obj

axiom InYelpRec : obj → Prop
axiom NotManyNeg : obj → Prop
axiom Rating4plus : obj → Prop
axiom TakeOut : obj → Prop
axiom ManyNeg : obj → Prop
axiom Popular : obj → Prop

-- If a restaurant is listed in Yelp’s recommendations, then it has not received many negative reviews.
axiom A1 : ∀ x : obj, InYelpRec x → NotManyNeg x
-- All restaurants with a rating greater than four are listed in Yelp’s recommendations.
axiom A2 : ∀ x : obj, Rating4plus x → InYelpRec x
-- Some restaurants that do not provide take-out service receive many negative reviews.
axiom A3 : ∃ x : obj, ¬ TakeOut x ∧ ManyNeg x
-- All restaurants popular among local residents have ratings greater than four.
axiom A4 : ∀ x : obj, Popular x → Rating4plus x
-- The Hamden Plaza Subway store has rating > 4 or is popular among local residents.
axiom A5 : Rating4plus HPS ∨ Popular HPS

theorem HPS_takeout_or_notmanyneg : TakeOut HPS ∨ NotManyNeg HPS :=
by
  have hR4_to_Not : Rating4plus HPS → NotManyNeg HPS :=
    fun h => A1 HPS (A2 HPS h)
  have hPop_to_Not : Popular HPS → NotManyNeg HPS :=
    fun hp => hR4_to_Not (A4 HPS hp)
  cases A5 with
  | inl hr4 =>
      exact Or.inr (hR4_to_Not hr4)
  | inr hpop =>
      exact Or.inr (hPop_to_Not hpop)
</lean>

ANSWER: True
