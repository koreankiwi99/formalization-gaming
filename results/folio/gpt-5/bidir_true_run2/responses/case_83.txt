<lean>
axiom Restaurant : Type
axiom HPS : Restaurant
axiom Listed : Restaurant → Prop
axiom ManyNeg : Restaurant → Prop
axiom RatingGT4 : Restaurant → Prop
axiom ProvideTakeout : Restaurant → Prop
axiom Popular : Restaurant → Prop

axiom A1 : ∀ x : Restaurant, Listed x → ¬ ManyNeg x
axiom A2 : ∀ x : Restaurant, RatingGT4 x → Listed x
axiom A3 : ∃ x : Restaurant, ¬ ProvideTakeout x ∧ ManyNeg x
axiom A4 : ∀ x : Restaurant, Popular x → RatingGT4 x
axiom A5 : RatingGT4 HPS ∨ Popular HPS

theorem stmt :
  (ProvideTakeout HPS ∧ ManyNeg HPS) → (RatingGT4 HPS ∧ ¬ ProvideTakeout HPS) := by
  have ratingHPS : RatingGT4 HPS := by
    cases A5 with
    | inl r => exact r
    | inr p => exact A4 HPS p
  have notManyHPS : ¬ ManyNeg HPS := by
    have listed : Listed HPS := A2 HPS ratingHPS
    exact A1 HPS listed
  intro h
  have mn : ManyNeg HPS := And.right h
  have contra : False := notManyHPS mn
  have notProv : ¬ ProvideTakeout HPS := False.elim contra
  exact And.intro ratingHPS notProv
</lean>

ANSWER: True
