<lean>
axiom Person : Type
axiom Series : Type
axiom Title : Type

axiom AndrewCollins : Person
axiom Badults : Series
axiom SecretDudeSociety : Title

axiom WorkingTitle : Series → Title → Prop
axiom ScriptEditor : Person → Series → Prop

axiom WT_badults : WorkingTitle Badults SecretDudeSociety
axiom SE_collins_badults : ScriptEditor AndrewCollins Badults

theorem collins_editor_secret_title_series :
  ∃ s : Series, WorkingTitle s SecretDudeSociety ∧ ScriptEditor AndrewCollins s :=
by
  apply Exists.intro Badults
  apply And.intro
  · exact WT_badults
  · exact SE_collins_badults
</lean>

ANSWER: True
