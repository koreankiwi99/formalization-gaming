<lean>
axiom obj : Type
axiom James : obj

axiom WorksLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom PartTimeUni : obj → Prop
axiom TakingDatabase : obj → Prop
axiom TakingWithDavid : obj → Prop
axiom WorksLab : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, WorksLibrary x → ¬ FromCS x
axiom A2 : ∀ x : obj, PartTimeUni x → WorksLibrary x
axiom A3 : ∀ x : obj, TakingDatabase x → FromCS x
axiom A4 : ∀ x : obj, TakingWithDavid x → TakingDatabase x
axiom A5 : ∀ x : obj, WorksLab x → TakingWithDavid x
axiom A6 : WorksLab James
axiom A7 : ¬ WorksLab James ∧ ¬ PartTimeUni James

theorem james_has_university_part_time_job : PartTimeUni James :=
by
  have hWL : WorksLab James := A6
  have hNotWL : ¬ WorksLab James := And.left A7
  have hFalse : False := hNotWL hWL
  exact False.elim hFalse
</lean>

ANSWER: True
