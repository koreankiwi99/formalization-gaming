axiom A5 : ¬Activate Yuri

-- Derived: Yuri does not leap straight into the air
theorem not_leap_yuri : ¬Leap Yuri :=
  fun h => A5 (A4 Yuri h)

-- Derived: Any professional basketball player activates leg muscles
theorem pbp_implies_activate : ∀ x : obj, PBP x → Activate x :=
  fun x hp => A4 x (A3 x (A2 x hp))

-- Derived: Yuri is not a professional basketball player

  fun hp => A5 (pbp_implies_activate Yuri hp)

-- The target statement cannot be proven; in fact it is refuted by the premises:
-- Since ¬Leap Yuri holds, the implication would yield American Yuri ∧ PBP Yuri,
-- but we have ¬PBP Yuri, a contradiction.
theorem target_contradicts : ¬((¬Leap Yuri) → (American Yuri ∧ PBP Yuri)) :=
by
  intro h
  have hnl : ¬Leap Yuri := not_leap_yuri
  have hcons : American Yuri ∧ PBP Yuri := h hnl
  have hp : PBP Yuri := hcons.right
  exact not_pbp_yuri hp
