<lean>
axiom obj : Type

-- Entities
axiom SummerOlympicGames : obj
axiom LastSummerOlympics : obj
axiom Tokyo : obj
axiom UnitedStates : obj

-- Predicates/Relations
axiom SportingEvent : obj → Prop
axiom WonMostMedalsIn : obj → obj → Prop

-- Given facts
axiom T0 : SportingEvent SummerOlympicGames
axiom T1 : LastSummerOlympics = Tokyo
axiom T2 : WonMostMedalsIn UnitedStates Tokyo

-- Goal: The United States won the most medals in the last summer Olympic games.
theorem us_won_most_last : WonMostMedalsIn UnitedStates LastSummerOlympics := by
  simpa [T1] using T2
</lean>

ANSWER: True
