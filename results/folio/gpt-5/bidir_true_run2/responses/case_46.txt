<lean>
axiom obj : Type
axiom Emma : obj
axiom Mia : obj
axiom James : obj

inductive Season where
| Spring : Season
| Summer : Season
| Fall : Season
| Winter : Season

axiom FavoriteSeason : obj → Season
axiom WantsLongVacation : obj → Prop

-- All (students) who want a long vacation have Summer as their favorite season
axiom R1 : ∀ x : obj, WantsLongVacation x → FavoriteSeason x = Season.Summer

-- Emma's favorite season is Summer
axiom T2 : FavoriteSeason Emma = Season.Summer

-- Mia's favorite season is not the same as Emma's
axiom T3 : FavoriteSeason Mia ≠ FavoriteSeason Emma

-- James wants to have a long vacation
axiom T4 : WantsLongVacation James

-- Therefore, James's favorite season is Summer
theorem james_favorite_season_is_summer : FavoriteSeason James = Season.Summer := by
  apply R1 James
  exact T4
</lean>

ANSWER: True
