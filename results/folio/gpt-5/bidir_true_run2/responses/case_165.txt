<lean>
axiom obj : Type
axiom Show : obj → Prop
axiom Monday8pm : obj → Prop
axiom GivesRoses : obj → Prop
axiom EpisodeOfBachelor : obj → Prop
axiom PortraysRealPeople : obj → Prop
axiom RealityTV : obj → Prop

-- Some show airing at 8 pm on Monday gives out roses on TV.
axiom ExMondayRoses : ∃ s : obj, Show s ∧ Monday8pm s ∧ GivesRoses s

-- If a show gives out roses on TV, then the show is an episode of The Bachelor.
axiom RosesImpliesBachelor : ∀ s : obj, GivesRoses s → EpisodeOfBachelor s

-- The Bachelor portrays the lives of real people (modeled as: its episodes do).
axiom BachelorPortraysReal : ∀ s : obj, EpisodeOfBachelor s → PortraysRealPeople s

-- All shows portraying the lives of real people are reality TV shows.
axiom RealPeopleImpliesReality : ∀ s : obj, PortraysRealPeople s → RealityTV s

-- Breaking Bad facts
axiom BreakingBad : obj
axiom BB_is_show : Show BreakingBad
axiom BB_not_reality : ¬ RealityTV BreakingBad

-- Statement to evaluate: If roses are given out during Breaking Bad, then it is on Monday at 8 pm.
theorem breaking_bad_roses_implies_monday8 : GivesRoses BreakingBad → Monday8pm BreakingBad := by
  intro hRoses
  have hBachelor : EpisodeOfBachelor BreakingBad := RosesImpliesBachelor BreakingBad hRoses
  have hPortrays : PortraysRealPeople BreakingBad := BachelorPortraysReal BreakingBad hBachelor
  have hReality : RealityTV BreakingBad := RealPeopleImpliesReality BreakingBad hPortrays
  exact False.elim (BB_not_reality hReality)
</lean>

ANSWER: True
