<lean>
axiom obj : Type
axiom Picuris : obj
axiom Harding : obj
axiom Juan : obj

axiom MountainRange : obj → Prop
axiom InNewMexico : obj → Prop
axiom InTexas : obj → Prop
axiom Visited : obj → obj → Prop
axiom Mine : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop

-- Picuris is a mountain range, and it is either in New Mexico or in Texas
axiom MR_Picuris : MountainRange Picuris
axiom NM_or_TX_Picuris : InNewMexico Picuris ∨ InTexas Picuris

-- Juan de Onate visited the Picuris Mountains
axiom JuanVisitedPicuris : Visited Juan Picuris

-- The Harding Pegmatite Mine is a mine located in the Picuris Mountains and was donated
axiom Harding_is_mine : Mine Harding
axiom Harding_in_Picuris : LocatedIn Harding Picuris
axiom Harding_donated : Donated Harding

-- There are no Texas mountain ranges that have donated mines located in them
axiom NoTxMRDonated :
  ∀ x : obj, MountainRange x → InTexas x → ¬ (∃ m : obj, Mine m ∧ LocatedIn m x ∧ Donated m)

theorem Juan_visited_a_NM_mountain_range :
  ∃ r : obj, MountainRange r ∧ InNewMexico r ∧ Visited Juan r := by
  have ex_harding : ∃ m : obj, Mine m ∧ LocatedIn m Picuris ∧ Donated m := by
    refine Exists.intro Harding ?_
    apply And.intro
    · exact Harding_is_mine
    · apply And.intro
      · exact Harding_in_Picuris
      · exact Harding_donated
  have notInTexas_Picuris : ¬ InTexas Picuris := by
    intro htx
    have hnone : ¬ (∃ m : obj, Mine m ∧ LocatedIn m Picuris ∧ Donated m) :=
      NoTxMRDonated Picuris MR_Picuris htx
    exact hnone ex_harding
  have inNM_Picuris : InNewMexico Picuris := by
    cases NM_or_TX_Picuris with
    | inl hnm => exact hnm
    | inr htx => exact False.elim (notInTexas_Picuris htx)
  refine Exists.intro Picuris ?_
  apply And.intro
  · exact MR_Picuris
  · apply And.intro
    · exact inNM_Picuris
    · exact JuanVisitedPicuris
</lean>

ANSWER: True
