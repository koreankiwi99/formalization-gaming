  ∀ s : obj, NetflixShow s → Popular s → Binge s

  ∀ s : obj, NetflixShow s → (Binge s ↔ Download s)

axiom bm_netflix : NetflixShow BlackMirror

-- From the biconditional we extract both directions
theorem binge_implies_download (s : obj) (hN : NetflixShow s) :
  Binge s → Download s :=
  fun hb =>
    (binge_iff_download s hN).mp hb

theorem download_implies_binge (s : obj) (hN : NetflixShow s) :
  Download s → Binge s :=
  fun hd =>
    (binge_iff_download s hN).mpr hd

-- Derive that Black Mirror is not popular

by
  intro hpop
  have hb : Binge BlackMirror :=
    popular_implies_binge BlackMirror bm_netflix hpop
  have hd : Download BlackMirror :=
    binge_implies_download BlackMirror bm_netflix hb
  exact not_download_bm hd
