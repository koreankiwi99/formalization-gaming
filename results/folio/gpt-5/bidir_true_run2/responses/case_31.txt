<lean>
axiom obj : Type

-- Entities
axiom Djokovic : obj

-- Predicates
axiom At : obj → Prop
axiom GrandSlamChamp : obj → Prop
axiom OscarNominatedActor : obj → Prop
axiom ProTennisPlayer : obj → Prop
axiom Celebrity : obj → Prop
axiom Athlete : obj → Prop
axiom Famous : obj → Prop
axiom WellPaid : obj → Prop
axiom LivesInTaxHaven : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, At x → (GrandSlamChamp x ∨ OscarNominatedActor x)
axiom P2 : ∀ x : obj, At x ∧ GrandSlamChamp x → ProTennisPlayer x
axiom P3 : ∀ x : obj, At x ∧ OscarNominatedActor x → Celebrity x
axiom P4 : ∀ x : obj, At x ∧ ProTennisPlayer x → Athlete x
axiom P5 : ∀ x : obj, At x ∧ Celebrity x → WellPaid x
axiom P6 : ∀ x : obj, At x ∧ Athlete x → Famous x
axiom P7 : ∀ x : obj, At x ∧ WellPaid x → LivesInTaxHaven x
axiom Djokovic_at : At Djokovic
axiom Djokovic_special : Famous Djokovic ∧ Athlete Djokovic → WellPaid Djokovic

theorem djokovic_tax_haven : LivesInTaxHaven Djokovic := by
  have AtDj : At Djokovic := Djokovic_at
  have h := P1 Djokovic AtDj
  cases h with
  | inl hGSC =>
      have pro : ProTennisPlayer Djokovic := P2 Djokovic ⟨AtDj, hGSC⟩
      have ath : Athlete Djokovic := P4 Djokovic ⟨AtDj, pro⟩
      have fam : Famous Djokovic := P6 Djokovic ⟨AtDj, ath⟩
      have wp : WellPaid Djokovic := Djokovic_special ⟨fam, ath⟩
      exact P7 Djokovic ⟨AtDj, wp⟩
  | inr hOsc =>
      have cel : Celebrity Djokovic := P3 Djokovic ⟨AtDj, hOsc⟩
      have wp : WellPaid Djokovic := P5 Djokovic ⟨AtDj, cel⟩
      exact P7 Djokovic ⟨AtDj, wp⟩
</lean>

ANSWER: True
