<lean>
axiom obj : Type

-- Entities
axiom Billings : obj
axiom Montana : obj
axiom Butte : obj
axiom Helena : obj
axiom Missoula : obj
axiom WhiteSulphurSprings : obj
axiom StPierre : obj
axiom Bristol : obj
axiom Texarkana : obj
axiom Texhoma : obj
axiom UnionCity : obj

-- Predicates and relations
axiom City : obj → Prop
axiom State : obj → Prop
axiom In : obj → obj → Prop

-- Given facts
-- Billings is a city in the state of Montana in U.S.
axiom CityBillings : City Billings
axiom StateMontana : State Montana
axiom InBillingsMontana : In Billings Montana

-- The state of Montana includes the cities of Butte, Helena, and Missoula.
axiom CityButte : City Butte
axiom CityHelena : City Helena
axiom CityMissoula : City Missoula
axiom InButteMontana : In Butte Montana
axiom InHelenaMontana : In Helena Montana
axiom InMissoulaMontana : In Missoula Montana

-- White Sulphur Springs and Butte are cities in the same state in U.S.
axiom CityWhiteSulphurSprings : City WhiteSulphurSprings
axiom SameState_WS_Butte : ∃ s : obj, State s ∧ In WhiteSulphurSprings s ∧ In Butte s

-- The city of St Pierre is not in the state of Montana.
axiom CityStPierre : City StPierre
axiom NotInStPierreMontana : ¬ In StPierre Montana

-- Any city in Butte is not in St Pierre.
axiom NoCityInButteInStPierre : ∀ x : obj, In x Butte → ¬ In x StPierre

-- A city can only be in one state in U.S. except for Bristol, Texarkana, Texhoma and Union City.
axiom UniqueStateUnlessExceptions :
  ∀ x y z : obj,
    City x → State y → State z → In x y → In x z →
    x ≠ Bristol → x ≠ Texarkana → x ≠ Texhoma → x ≠ UnionCity →
    y = z

-- Statement to evaluate: Montana is home to the city of Missoula.
theorem montana_home_missoula : In Missoula Montana := by
  exact InMissoulaMontana
</lean>

ANSWER: True
