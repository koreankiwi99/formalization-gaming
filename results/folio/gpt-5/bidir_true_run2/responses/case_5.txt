axiom James : Person

axiom Employee : Person → Prop
axiom Manager : Person → Prop

axiom LunchHome : Person → Prop
axiom LunchBuilding : Person → Prop
axiom SchedulesMeeting : Person → Prop
axiom AppearCompany : Person → Prop
axiom RemoteFromHome : Person → Prop

-- Axioms from the context
-- 1) All employees who schedule a meeting with their customers will go to the company building today.
axiom A1 : ∀ x : Person, Employee x → SchedulesMeeting x → AppearCompany x

-- 2) Everyone who has lunch in the company building schedules meetings with their customers.
axiom A2 : ∀ x : Person, LunchBuilding x → SchedulesMeeting x

-- 3) Employees have lunch either in the company building or at home.
axiom A3 : ∀ x : Person, Employee x → (LunchBuilding x ∨ LunchHome x)

-- 4) If an employee has lunch at home, they are working remotely from home.
axiom A4 : ∀ x : Person, Employee x → LunchHome x → RemoteFromHome x

-- 5) All employees who are in other countries work remotely from home.
axiom A5 : ∀ x : Person, Employee x → InOtherCountries x → RemoteFromHome x

-- 6) No managers work remotely from home.
axiom A6 : ∀ x : Person, Manager x → ¬ RemoteFromHome x

-- 7) James will appear in the company today iff he is a manager.
axiom A7 : AppearCompany James ↔ Manager James

-- Some consequences we can derive:

-- Remote implies not Manager (constructively derived from A6)
theorem remote_implies_not_manager (x : Person) : RemoteFromHome x → ¬ Manager x :=
fun r m => (A6 x m) r

-- Specializing to James
theorem james_remote_implies_not_manager : RemoteFromHome James → ¬ Manager James :=
remote_implies_not_manager James

-- If James is an employee and has lunch at home, then he is not a manager
theorem james_lunchHome_employee_implies_not_manager :
  Employee James → LunchHome James → ¬ Manager James :=
fun eJ lhJ mJ => (A6 James mJ) ((A4 James eJ) lhJ)

-- (2) (LunchHome James ∨ RemoteFromHome James) → (¬ Manager James ∧ ¬ InOtherCountries James)
-- From the given axioms, we cannot derive ¬LunchHome James from Manager James,
