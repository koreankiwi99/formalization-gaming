<lean>
axiom obj : Type

-- Individuals
axiom Coco : obj
axiom Iga : obj
axiom Nadal : obj

-- Predicates
axiom RankedHighlyWTA : obj → Prop
axiom MostActiveGS : obj → Prop
axiom LostToIgaRG : obj → Prop
axiom LostToNadal : obj → Prop
axiom PlayerRG : obj → Prop
axiom FemaleRG : obj → Prop
axiom MaleRG : obj → Prop

-- Axioms from the context
-- If someone is ranked highly by the WTA, then they are among the most active Grand-Slam players.
axiom A1 : ∀ x : obj, RankedHighlyWTA x → MostActiveGS x
-- Everyone who lost to Iga Świątek at Roland Garros 2022 is ranked highly by the WTA.
axiom A2 : ∀ x : obj, LostToIgaRG x → RankedHighlyWTA x
-- All female tennis players at Roland Garros 2022 lost to Iga Świątek.
axiom A3 : ∀ x : obj, FemaleRG x → LostToIgaRG x
-- All tennis players at Roland Garros 2022 are either female or male.
axiom A4 : ∀ x : obj, PlayerRG x → (FemaleRG x ∨ MaleRG x)
-- All male tennis players at Roland Garros in 2022 lost to Rafael Nadal.
axiom A5 : ∀ x : obj, MaleRG x → LostToNadal x
-- If Coco is ranked highly by the WTA or lost to Rafael Nadal, then Coco is not a male tennis player at Roland Garros 2022.
axiom A6 : (RankedHighlyWTA Coco ∨ LostToNadal Coco) → ¬ MaleRG Coco
-- Coco Gauff is at Roland Garros 2022 (is a tennis player there).
axiom A7 : PlayerRG Coco

theorem coco_most_active : MostActiveGS Coco :=
by
  -- First, show Coco is not male at RG using A5 and A6
  have notMale : ¬ MaleRG Coco :=
    by
      intro hmale
      have l : LostToNadal Coco := A5 Coco hmale
      have nm : ¬ MaleRG Coco := A6 (Or.inr l)
      exact nm hmale
  -- From being a player at RG and the female/male dichotomy, deduce Coco is female
  have femOrMale : FemaleRG Coco ∨ MaleRG Coco := A4 Coco A7
  have fem : FemaleRG Coco := Or.resolve_right femOrMale notMale
  -- Hence Coco lost to Iga at RG, is ranked highly, and thus among the most active GS players
  have lostIga : LostToIgaRG Coco := A3 Coco fem
  have ranked : RankedHighlyWTA Coco := A2 Coco lostIga
  exact A1 Coco ranked
</lean>

ANSWER: True
