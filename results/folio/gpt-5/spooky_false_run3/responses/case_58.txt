axiom obj : Type

-- Entities (cities and a state)
axiom Billings : obj
axiom Butte : obj
axiom Helena : obj
axiom Missoula : obj
axiom WhiteSulphurSprings : obj
axiom StPierre : obj
axiom Bismarck : obj
axiom Bristol : obj
axiom Texarkana : obj
axiom Texhoma : obj
axiom UnionCity : obj
axiom Montana : obj

-- Relation: a city is in a (U.S.) state
axiom InState : obj → obj → Prop

-- Given facts
axiom Billings_in_Montana : InState Billings Montana
axiom Butte_in_Montana : InState Butte Montana
axiom Helena_in_Montana : InState Helena Montana
axiom Missoula_in_Montana : InState Missoula Montana

-- White Sulphur Springs and Butte are in the same state (we fix it as Montana)
axiom WSS_in_Montana : InState WhiteSulphurSprings Montana

-- St Pierre is not in Montana
axiom StPierre_not_in_Montana : ¬ InState StPierre Montana

-- Any city sharing a state with Butte does not share a state with St Pierre
axiom not_in_Butte_vs_StPierre :
  ∀ x : obj,
    (∃ s1 : obj, InState x s1 ∧ InState Butte s1) →
    ¬ (∃ s2 : obj, InState x s2 ∧ InState StPierre s2)

-- Uniqueness of state membership except for the listed exceptions
axiom unique_state_except :
  ∀ c s1 s2 : obj,
    InState c s1 → InState c s2 →
    (s1 = s2 ∨ c = Bristol ∨ c = Texarkana ∨ c = Texhoma ∨ c = UnionCity)

-- "SameState" means there exists a state both cities are in
def SameState (c1 c2 : obj) : Prop := ∃ s : obj, InState c1 s ∧ InState c2 s

-- A derivable consequence: Butte and St Pierre are not in the same state
theorem Butte_not_same_state_as_StPierre : ¬ SameState Butte StPierre := by
  unfold SameState
  have hButteShare : ∃ s1 : obj, InState Butte s1 ∧ InState Butte s1 := by
    exact ⟨Montana, And.intro Butte_in_Montana Butte_in_Montana⟩
  have h := not_in_Butte_vs_StPierre Butte
  exact h hButteShare
