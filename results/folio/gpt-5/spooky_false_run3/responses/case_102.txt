<lean>
axiom obj : Type
axiom Luke : obj

axiom BornInMultiple : obj → Prop
axiom SpendTimeWithSiblings : obj → Prop
axiom HaveSiblingsBornTogether : obj → Prop
axiom ComplainsOftenAboutAnnoyingSiblings : obj → Prop
axiom LiveAtHome : obj → Prop
axiom LiveWithStrangers : obj → Prop

-- People who are born in a multiple birth spend time with their siblings
axiom A1 : ∀ x : obj, BornInMultiple x → SpendTimeWithSiblings x
-- If people have siblings who were born together, then they were born in a multiple birth
axiom A2 : ∀ x : obj, HaveSiblingsBornTogether x → BornInMultiple x
-- If people complain often about annoying siblings, then they have siblings who were born together
axiom A3 : ∀ x : obj, ComplainsOftenAboutAnnoyingSiblings x → HaveSiblingsBornTogether x
-- If people live at home, then they do not live with strangers
axiom A4 : ∀ x : obj, LiveAtHome x → ¬LiveWithStrangers x
-- If people spend a lot of time with their siblings, then they live at home
axiom A5 : ∀ x : obj, SpendTimeWithSiblings x → LiveAtHome x
-- Luke either is born in a multiple birth and lives with strangers, or is not born in a multiple birth and does not live with strangers
axiom Luke_cases : (BornInMultiple Luke ∧ LiveWithStrangers Luke) ∨ (¬BornInMultiple Luke ∧ ¬LiveWithStrangers Luke)

theorem luke_not_complain : ¬ComplainsOftenAboutAnnoyingSiblings Luke := by
  intro hC
  have hHS : HaveSiblingsBornTogether Luke := A3 Luke hC
  have hBorn : BornInMultiple Luke := A2 Luke hHS
  have hSpend : SpendTimeWithSiblings Luke := A1 Luke hBorn
  have hHome : LiveAtHome Luke := A5 Luke hSpend
  have hNotStr : ¬LiveWithStrangers Luke := A4 Luke hHome
  have hStr : LiveWithStrangers Luke := by
    cases Luke_cases with
    | inl h =>
      exact h.right
    | inr h =>
      have hFalse : False := h.left hBorn
      exact False.elim hFalse
  exact hNotStr hStr
</lean>
