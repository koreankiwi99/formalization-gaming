axiom obj : Type
axiom Yuri : obj

axiom American : obj → Prop
axiom Pro : obj → Prop
axiom Jump : obj → Prop
axiom Leap : obj → Prop
axiom Activate : obj → Prop

axiom S1 : ∃ x : obj, Pro x ∧ ¬ American x
axiom R1 : ∀ x : obj, Pro x → Jump x
axiom R2 : ∀ x : obj, Jump x → Leap x
axiom R3 : ∀ x : obj, Leap x → Activate x
axiom T1 : ¬ Activate Yuri

theorem negate_statement : ¬ ((¬ (American Yuri ∧ Pro Yuri)) → Pro Yuri) := by
  -- derive Pro -> Activate
  have pro_to_activate : ∀ x : obj, Pro x → Activate x := by
    intro x hx
    have hj : Jump x := R1 x hx
    have hl : Leap x := R2 x hj
    have ha : Activate x := R3 x hl
    exact ha
  -- Yuri is not a professional basketball player
  have not_pro_yuri : ¬ Pro Yuri := by
    intro hpy
    have ha : Activate Yuri := pro_to_activate Yuri hpy
    exact T1 ha
  -- hence Yuri is not an American professional basketball player
  have not_american_and_pro : ¬ (American Yuri ∧ Pro Yuri) := by
    intro h
    exact not_pro_yuri h.right
  -- negate the implication
  intro himp
  have hpro : Pro Yuri := himp not_american_and_pro
  exact not_pro_yuri hpro
