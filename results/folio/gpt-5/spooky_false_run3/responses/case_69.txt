<lean>
axiom Person : Type
axiom Show : Type
axiom Channel : Type

axiom MemberOfPappys : Person → Prop
axiom StarredIn : Person → Show → Prop
axiom PilotedOn : Show → Channel → Prop

axiom Badults : Show
axiom BBCTwo : Channel
axiom BBCThree : Channel

-- Context axioms from the text
axiom BadultsStarsPappys : ∃ p : Person, MemberOfPappys p ∧ StarredIn p Badults
axiom BadultsPilotedOnBBCThree : PilotedOn Badults BBCThree

-- The statement to evaluate: No members of Pappy's have starred in a show piloting on BBC Two or BBC Three.
def Statement : Prop :=
  ∀ p : Person, MemberOfPappys p → ¬ ∃ s : Show, StarredIn p s ∧ (PilotedOn s BBCTwo ∨ PilotedOn s BBCThree)

-- We prove the negation of the statement from the premises.
theorem statement_false : ¬Statement := by
  intro h
  cases BadultsStarsPappys with
  | intro p hp =>
    cases hp with
    | intro hpappys hstar =>
      have hexists : ∃ s : Show, StarredIn p s ∧ (PilotedOn s BBCTwo ∨ PilotedOn s BBCThree) := by
        refine ⟨Badults, And.intro hstar (Or.inr BadultsPilotedOnBBCThree)⟩
      have hneg := h p hpappys
      exact hneg hexists
</lean>

ANSWER: False
