axiom obj : Type

axiom Person : obj → Prop
axiom Book : obj → Prop
axiom Reads : obj → obj → Prop
axiom GainsKnowledge : obj → Prop
axiom Smarter : obj → Prop

axiom Harry : obj
axiom Walden : obj

-- Premises
axiom HarryIsPerson : Person Harry
axiom WaldenIsBook : Book Walden
axiom HarryReadWalden : Reads Harry Walden
axiom ReadImpliesGain : ∀ p b, Person p → Book b → Reads p b → GainsKnowledge p
axiom GainImpliesSmarter : ∀ p, GainsKnowledge p → Smarter p

-- From the premises, Harry gained knowledge and is smarter
theorem harry_gained : GainsKnowledge Harry :=
  ReadImpliesGain Harry Walden HarryIsPerson WaldenIsBook HarryReadWalden

theorem harry_smarter : Smarter Harry :=
  GainImpliesSmarter Harry harry_gained

-- Therefore, there exists a smarter person who has gained knowledge
theorem exists_smarter_has_gained : ∃ x, Smarter x ∧ GainsKnowledge x := by
  exact Exists.intro Harry (And.intro harry_smarter harry_gained)

-- Hence, the negation of the statement is impossible
theorem neg_statement_impossible : (¬ (∃ x, Smarter x ∧ GainsKnowledge x)) → False := by
  intro h
  exact h exists_smarter_has_gained
