<lean>
axiom obj : Type

-- Predicates
axiom InClub : obj → Prop
axiom PerformsOften : obj → Prop
axiom AttendsEvents : obj → Prop
axiom VeryEngaged : obj → Prop
axiom Inactive : obj → Prop
axiom Chaperone : obj → Prop
axiom Student : obj → Prop
axiom YoungChild : obj → Prop
axiom Teenager : obj → Prop
axiom WishesFurther : obj → Prop

-- Individual
axiom Bonnie : obj

-- Premises
-- 1. People in this club who perform often attend and are very engaged with school events.
axiom P1 : ∀ x : obj, InClub x ∧ PerformsOften x → AttendsEvents x ∧ VeryEngaged x

-- 2. People in this club either perform often or are inactive and disinterested.
axiom P2 : ∀ x : obj, InClub x → PerformsOften x ∨ Inactive x

-- 3. People in this club who chaperone high school dances are not students who attend the school.
axiom P3 : ∀ x : obj, InClub x ∧ Chaperone x → ¬ Student x

-- 4. All people in this club who are inactive and disinterested chaperone high school dances.
axiom P4 : ∀ x : obj, InClub x ∧ Inactive x → Chaperone x

-- 5. All young children and teenagers in this club who wish to further their academic careers are students.
axiom P5 : ∀ x : obj, InClub x ∧ (YoungChild x ∨ Teenager x) ∧ WishesFurther x → Student x

-- 6. Bonnie is in this club.
axiom B_in : InClub Bonnie

-- 7. Bonnie either both attends and is very engaged and is a student, or is not someone who both attends and is very engaged and is not a student.
axiom B_special : (AttendsEvents Bonnie ∧ VeryEngaged Bonnie ∧ Student Bonnie) ∨ ¬(AttendsEvents Bonnie ∧ VeryEngaged Bonnie ∧ ¬Student Bonnie)

-- Definitions to structure the statement
def YTWish (x : obj) : Prop := (YoungChild x ∨ Teenager x) ∧ WishesFurther x
def Antecedent : Prop := (YTWish Bonnie ∧ Chaperone Bonnie) ∨ ¬(YTWish Bonnie)
def Consequent : Prop := Student Bonnie ∨ Inactive Bonnie
def Statement : Prop := Antecedent → Consequent

-- A useful lemma: the left disjunct of the antecedent leads to a contradiction for Bonnie.
theorem left_disj_impossible : (((YoungChild Bonnie ∨ Teenager Bonnie) ∧ WishesFurther Bonnie) ∧ Chaperone Bonnie) → False :=
by
  intro h
  have hYT : (YoungChild Bonnie ∨ Teenager Bonnie) := And.left h.left
  have hWish : WishesFurther Bonnie := And.right h.left
  have hCh : Chaperone Bonnie := And.right h
  have s : Student Bonnie := P5 Bonnie (And.intro B_in (And.intro hYT hWish))
  have ns : ¬ Student Bonnie := P3 Bonnie (And.intro B_in hCh)
  exact ns s
</lean>
