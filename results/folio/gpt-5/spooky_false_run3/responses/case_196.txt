axiom Bark : obj → Prop

axiom HasLegs : obj → Prop
axiom Terricolous : obj → Prop

axiom A1 : ∀ x : obj, Animal x ∧ Bark x → ¬Likes Jane x
axiom A2 : ∀ x : obj, Dog x → Animal x ∧ Bark x
axiom A3 : ∀ x : obj, Animal x ∧ Jumps x → Likes Jane x
axiom A4 : ∀ x : obj, Animal x ∧ HasLegs x → Jumps x
axiom A5 : ∀ x : obj, Animal x ∧ Terricolous x → HasLegs x
axiom A6 : Animal KiKi
axiom A7 : (¬Jumps KiKi ∧ ¬HasLegs KiKi) → (Terricolous KiKi ∨ HasLegs KiKi)

theorem kiki_jumps : Jumps KiKi := by
  classical
  by_contra hNotJ
  by_cases hL : HasLegs KiKi
  · exact hNotJ (A4 KiKi (And.intro A6 hL))
  · have hTorL : Terricolous KiKi ∨ HasLegs KiKi := A7 (And.intro hNotJ hL)
    have hL2 : HasLegs KiKi := Or.elim hTorL
      (fun t => A5 KiKi (And.intro A6 t))
      (fun l => l)
    exact hL hL2

theorem kiki_not_barking_dog : ¬(Dog KiKi ∧ Bark KiKi) := by

  have hb : Bark KiKi := h.right
  have hNotLike : ¬Likes Jane KiKi := A1 KiKi (And.intro A6 hb)
  have hLike : Likes Jane KiKi := A3 KiKi (And.intro A6 kiki_jumps)
