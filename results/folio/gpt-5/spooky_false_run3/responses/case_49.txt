<lean>
axiom obj : Type
axiom NineteenEightyFour : obj

axiom DigitalMedia : obj → Prop
axiom AnalogMedia : obj → Prop
axiom PrintedText : obj → Prop
axiom StreamingService : obj → Prop
axiom HardcoverBook : obj → Prop

-- No digital media are analog.
axiom P1 : ∀ x : obj, DigitalMedia x → ¬AnalogMedia x
-- Every printed text is analog media.
axiom P2 : ∀ x : obj, PrintedText x → AnalogMedia x
-- All streaming services are digital media.
axiom P3 : ∀ x : obj, StreamingService x → DigitalMedia x
-- If an object is a hardcover book, then it is printed text.
axiom P4 : ∀ x : obj, HardcoverBook x → PrintedText x
-- If 1984 is a streaming service, then 1984 is a hardcover book.
axiom P5 : StreamingService NineteenEightyFour → HardcoverBook NineteenEightyFour

theorem not_streaming_1984 : ¬StreamingService NineteenEightyFour := by
  intro hS
  have hD : DigitalMedia NineteenEightyFour := P3 NineteenEightyFour hS
  have hHB : HardcoverBook NineteenEightyFour := P5 hS
  have hPT : PrintedText NineteenEightyFour := P4 NineteenEightyFour hHB
  have hA : AnalogMedia NineteenEightyFour := P2 NineteenEightyFour hPT
  have hNotA : ¬AnalogMedia NineteenEightyFour := P1 NineteenEightyFour hD
  exact hNotA hA
</lean>

ANSWER: False
