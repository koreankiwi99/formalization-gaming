-- Premises
axiom AllCatsAreMammals : ∀ x : obj, Cat x → Mammal x
axiom SomePetsNotMammals : ∃ x : obj, Pet x ∧ ¬Mammal x

-- From the premises we can derive that there exists a pet that is not a cat.
-- This does not suffice to prove that "No pets are cats" is false.
theorem exists_pet_not_cat : ∃ x : obj, Pet x ∧ ¬Cat x := by
  apply Exists.elim SomePetsNotMammals
  intro x hx
  cases hx with
  | intro hxPet hxNotMammal =>
    have hnotcat : ¬Cat x := by
      intro hcat
      exact hxNotMammal (AllCatsAreMammals x hcat)
    exact Exists.intro x (And.intro hxPet hnotcat)
