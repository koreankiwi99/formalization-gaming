axiom Cow : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, BrownSwissCattle x → Cow x
axiom A2 : ∃ x : obj, Pet x ∧ BrownSwissCattle x
axiom A3 : ∀ x : obj, Cow x → Domesticated x
axiom A4 : ∀ x : obj, Alligator x → ¬Domesticated x
axiom A5 : Alligator Ted

-- Derivation: Ted is not domesticated
theorem ted_not_domesticated : ¬Domesticated Ted := by
  exact A4 Ted A5

-- Derivation: There exists a domesticated pet (from the existential and the two universal premises)
theorem exists_domesticated_pet : ∃ x : obj, Pet x ∧ Domesticated x := by
  cases A2 with
  | intro w hw =>
    have hw_pet : Pet w := And.left hw
    have hw_bs  : BrownSwissCattle w := And.right hw
    have hw_cow : Cow w := A1 w hw_bs
    have hw_dom : Domesticated w := A3 w hw_cow
    exact Exists.intro w (And.intro hw_pet hw_dom)
