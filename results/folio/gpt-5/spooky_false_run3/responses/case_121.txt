axiom P5 : ¬ TeslaModel3 johns_car → (¬ TeslaModel3 johns_car ∨ SingleSeatEV johns_car)

-- A derived fact: any Tesla Model 3 is not a one-seater.
theorem Tesla_not_OneSeater : ∀ x : obj, TeslaModel3 x → ¬ OneSeater x := by

  have hxMulti : Multi x := P2 x hx
