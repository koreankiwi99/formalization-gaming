<lean>
axiom obj : Type
axiom Human : obj → Prop
axiom Horse : obj → Prop
axiom HasHooves : obj → Prop

axiom all_horses_have_hooves : ∀ x : obj, Horse x → HasHooves x
axiom no_humans_have_hooves : ∀ x : obj, Human x → ¬HasHooves x

theorem no_human_is_horse : ∀ x : obj, Human x → ¬Horse x :=
by
  intro x hx hhorse
  have hh : HasHooves x := all_horses_have_hooves x hhorse
  have nhh : ¬ HasHooves x := no_humans_have_hooves x hx
  exact nhh hh

theorem some_humans_are_horses_false : ¬ (∃ x : obj, Human x ∧ Horse x) :=
by
  intro h
  cases h with
  | intro w hw =>
    have hh : HasHooves w := all_horses_have_hooves w hw.right
    have nhh : ¬ HasHooves w := no_humans_have_hooves w hw.left
    exact nhh hh
</lean>

ANSWER: False
