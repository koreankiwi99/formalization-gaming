axiom Aurochs : obj → Prop

axiom ArtificialSelection : obj → Prop

axiom Resembles : obj → obj → Prop

-- Premises from the text
-- Heck cattle are animals.
axiom HeckAreAnimals : ∀ x : obj, HeckCattle x → Animal x
-- Aurochs are animals.
axiom AurochsAreAnimals : ∀ x : obj, Aurochs x → Animal x
-- Breeding back is a form of artificial selection.
axiom BreedingBackIsArtificialSelection : ∀ x : obj, BredBack x → ArtificialSelection x
-- Heck cattle were bred back (modeled as: all Heck cattle are bred back).
axiom HeckWereBredBack : ∀ x : obj, HeckCattle x → BredBack x
-- Some animals to be bred back resemble extinct animals.
axiom SomeAnimalsToBeBredBackResembleExtinctAnimals :
  ∃ x : obj, Animal x ∧ ToBeBredBack x ∧ ∃ y : obj, Extinct y ∧ Animal y ∧ Resembles x y

-- A conditional consequence of the premises (not the negation of the target statement):
-- If some Heck cattle exist, then some Heck cattle are artificially selected.
theorem if_heck_exist_then_some_artsel :
  (∃ x : obj, HeckCattle x) → (∃ x : obj, HeckCattle x ∧ ArtificialSelection x) := by
  intro hex
  rcases hex with ⟨x, hxH⟩
  have hb : BredBack x := HeckWereBredBack x hxH
  have ha : ArtificialSelection x := BreedingBackIsArtificialSelection x hb
  exact ⟨x, And.intro hxH ha⟩
