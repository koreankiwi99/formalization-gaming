axiom le_trans_nat : ∀ a b c : Nat, a ≤ b → b ≤ c → a ≤ c

-- A derived fact: At The Olive Garden, the security deposit is not ≤ 1500
theorem olive_secdep_not_le_1500 : ¬ (SecDep OliveGarden ≤ 1500) := by
  intro h
  have hMng := olive_is_managed
  have hRentLeSec : Rent OliveGarden ≤ SecDep OliveGarden :=
    secdep_at_least_rent OliveGarden hMng
  have h2000_le_sec : 2000 ≤ SecDep OliveGarden := by
    have h' : Rent OliveGarden ≤ SecDep OliveGarden := hRentLeSec
    -- rewrite Rent OliveGarden = 2000
    have h'' := h'
    -- rewrite in place
    have : 2000 ≤ SecDep OliveGarden := by
      -- use rewriting on h''
      -- Note: rw modifies goals; we produce the term by rewriting and returning it
      exact (Eq.rec (motive := fun x => x ≤ SecDep OliveGarden) hRentLeSec rent_olive)
    exact this
  have h2000_le_1500 : 2000 ≤ 1500 := le_trans_nat 2000 (SecDep OliveGarden) 1500 h2000_le_sec h
  exact (not_le_of_gt_nat 2000 1500 twoThousand_gt_oneThousandFiveHundred) h2000_le_1500

-- The sufficient condition for Tom renting at OliveGarden fails (due to deposit bound)
theorem not_cond_for_tom_rent_olive :
  ¬ (AllowedWith Tom OliveGarden Fluffy ∧ SecDep OliveGarden ≤ 1500) := by
  intro h
  exact olive_secdep_not_le_1500 h.right
