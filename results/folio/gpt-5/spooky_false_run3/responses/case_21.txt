axiom ST : Show
axiom BM : Show

axiom Netflix : Show → Prop
axiom Popular : Show → Prop
axiom BingeK : Show → Prop
axiom DownloadK : Show → Prop
axiom ShareWithLisa : Show → Prop

-- Premises
axiom ST_popular : Popular ST
axiom ST_netflix : Netflix ST
axiom popular_netflix_implies_binge :
  ∀ s : Show, (Netflix s ∧ Popular s) → BingeK s
-- Interpreting "iff Karen binge-watches a Netflix show, she will download it"
-- as: for any Netflix show s, she downloads s iff there exists some Netflix show she binge-watches
axiom download_iff_exists_binged_netflix :
  ∀ s : Show, Netflix s → (DownloadK s ↔ ∃ t : Show, Netflix t ∧ BingeK t)
axiom BM_not_download : ¬DownloadK BM
axiom BM_netflix : Netflix BM
axiom binge_implies_share_with_lisa :
  ∀ s : Show, Netflix s → (BingeK s → ShareWithLisa s)

theorem not_share_ST : ¬ShareWithLisa ST := by
  intro _hshare
  -- From "Stranger Things" is a popular Netflix show and the rule,
  -- Karen binge-watches ST.
  have binge_ST : BingeK ST := by
    have h : Netflix ST ∧ Popular ST := And.intro ST_netflix ST_popular
    exact popular_netflix_implies_binge ST h
  -- Therefore, there exists a Netflix show Karen binge-watches (namely, ST).
  have exists_binged : ∃ t : Show, Netflix t ∧ BingeK t :=
    Exists.intro ST (And.intro ST_netflix binge_ST)
  -- By the iff premise applied to BM (a Netflix show), she downloads BM.
  have hiff : DownloadK BM ↔ ∃ t : Show, Netflix t ∧ BingeK t :=
    download_iff_exists_binged_netflix BM BM_netflix
  have dlBM : DownloadK BM := (Iff.mpr hiff) exists_binged
  -- Contradiction with "Karen does not download BM".
  exact BM_not_download dlBM
</lean>

ANSWER: False
