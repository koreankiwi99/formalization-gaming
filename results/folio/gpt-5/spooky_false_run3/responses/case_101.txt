axiom obj : Type
axiom Luke : obj

axiom Baby : obj → Prop
axiom MultipleBirth : obj → Prop
axiom SpendTime : obj → Prop
axiom SiblingsBornTogether : obj → Prop
axiom Complain : obj → Prop
axiom LiveAtHome : obj → Prop
axiom LiveWithStrangers : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, MultipleBirth x → SpendTime x
axiom P2 : ∀ x : obj, SiblingsBornTogether x → MultipleBirth x
axiom P3 : ∀ x : obj, Complain x → SiblingsBornTogether x
axiom P4 : ∀ x : obj, LiveAtHome x → ¬LiveWithStrangers x
axiom P5 : ∀ x : obj, SpendTime x → LiveAtHome x

-- Disjunctive fact about Luke:
-- Either (Baby ∧ MultipleBirth ∧ LiveWithStrangers) or (¬Baby ∧ ¬MultipleBirth ∧ ¬LiveWithStrangers)
axiom P6 : (Baby Luke ∧ MultipleBirth Luke ∧ LiveWithStrangers Luke) ∨
            (¬Baby Luke ∧ ¬MultipleBirth Luke ∧ ¬LiveWithStrangers Luke)

-- SpendTime implies not living with strangers (via LiveAtHome)
theorem spendtime_implies_not_strangers (x : obj) : SpendTime x → ¬LiveWithStrangers x :=
  fun h => P4 x (P5 x h)

-- The first disjunct in P6 is impossible
theorem not_first_branch : ¬(Baby Luke ∧ MultipleBirth Luke ∧ LiveWithStrangers Luke) := by
  intro h
  have hM : MultipleBirth Luke := And.left (And.right h)
  have hS : LiveWithStrangers Luke := And.right (And.right h)
  have hST : SpendTime Luke := P1 Luke hM
  have hNotS : ¬LiveWithStrangers Luke := spendtime_implies_not_strangers Luke hST
  exact hNotS hS

-- Therefore Luke does not live with strangers
theorem luke_not_with_strangers : ¬LiveWithStrangers Luke := by
  have h2 : ¬Baby Luke ∧ ¬MultipleBirth Luke ∧ ¬LiveWithStrangers Luke :=
    Or.resolve_left P6 not_first_branch
  exact And.right (And.right h2)

-- And Luke is not born in a multiple birth
theorem luke_not_multiple : ¬MultipleBirth Luke := by
  have h2 : ¬Baby Luke ∧ ¬MultipleBirth Luke ∧ ¬LiveWithStrangers Luke :=
    Or.resolve_left P6 not_first_branch
  exact And.left (And.right h2)
