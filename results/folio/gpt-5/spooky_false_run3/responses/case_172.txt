axiom obj : Type
axiom Taylor : obj

axiom EatsSaladsRegularly : obj → Prop
axiom VeryConsciousAboutHealthAndEatingHabits : obj → Prop
axiom GrewUpInHealthConsciousChildhoodHome : obj → Prop
axiom FulfillDailyNutritionalIntakes : obj → Prop
axiom DisregardPhysicalWellBeing : obj → Prop
axiom VisitGymAtLeastOnceADay : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, EatsSaladsRegularly x → VeryConsciousAboutHealthAndEatingHabits x
axiom P2 : ∀ x : obj, GrewUpInHealthConsciousChildhoodHome x → EatsSaladsRegularly x
axiom P3 : ∀ x : obj, FulfillDailyNutritionalIntakes x → GrewUpInHealthConsciousChildhoodHome x
axiom P4 : ∀ x : obj, DisregardPhysicalWellBeing x → ¬VeryConsciousAboutHealthAndEatingHabits x
axiom P5 : ∀ x : obj, VisitGymAtLeastOnceADay x → FulfillDailyNutritionalIntakes x
axiom P6 : (GrewUpInHealthConsciousChildhoodHome Taylor ∧ DisregardPhysicalWellBeing Taylor)
        ∨ (¬GrewUpInHealthConsciousChildhoodHome Taylor ∧ ¬DisregardPhysicalWellBeing Taylor)

-- Showing the first disjunct in P6 is impossible
theorem taylor_AD_impossible :
  (GrewUpInHealthConsciousChildhoodHome Taylor ∧ DisregardPhysicalWellBeing Taylor) → False := by
  intro h
  have hA : GrewUpInHealthConsciousChildhoodHome Taylor := And.left h
  have hD : DisregardPhysicalWellBeing Taylor := And.right h
  have hE : EatsSaladsRegularly Taylor := P2 Taylor hA
  have hV : VeryConsciousAboutHealthAndEatingHabits Taylor := P1 Taylor hE
  have hNotV : ¬VeryConsciousAboutHealthAndEatingHabits Taylor := P4 Taylor hD
  exact hNotV hV

-- Therefore Taylor did neither: ¬A ∧ ¬D
theorem taylor_notA_and_notD :
  ¬GrewUpInHealthConsciousChildhoodHome Taylor ∧ ¬DisregardPhysicalWellBeing Taylor := by
  have hdisj := P6
  have hnotAD : ¬(GrewUpInHealthConsciousChildhoodHome Taylor ∧ DisregardPhysicalWellBeing Taylor) :=
    taylor_AD_impossible
  cases hdisj with
  | inl hAD =>
      exact False.elim (hnotAD hAD)
  | inr hneither =>
      exact hneither

theorem taylor_notA : ¬GrewUpInHealthConsciousChildhoodHome Taylor :=
  And.left taylor_notA_and_notD

theorem taylor_notD : ¬DisregardPhysicalWellBeing Taylor :=
  And.right taylor_notA_and_notD

-- From ¬A and P3 (Fulfill → A), Taylor does not fulfill daily nutritional intakes
theorem taylor_notFulfill : ¬FulfillDailyNutritionalIntakes Taylor := by
  intro hF
  have hA : GrewUpInHealthConsciousChildhoodHome Taylor := P3 Taylor hF
  exact taylor_notA hA

-- From ¬Fulfill and P5 (VisitGym → Fulfill), Taylor does not visit the gym daily
theorem taylor_notVisitGym : ¬VisitGymAtLeastOnceADay Taylor := by
  intro hG
  have hF : FulfillDailyNutritionalIntakes Taylor := P5 Taylor hG
  exact taylor_notFulfill hF
