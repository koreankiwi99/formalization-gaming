by
  intro h
  have hS : SingleSeatElectric JCar := R4 JCar h
  exact R3 JCar hS

by
  intro h
  intro hM
  have hNotOne : ¬ OneSeater JCar := R1 JCar hM
  have hOne : OneSeater JCar := R3 JCar (R4 JCar h)
  exact hNotOne hOne
