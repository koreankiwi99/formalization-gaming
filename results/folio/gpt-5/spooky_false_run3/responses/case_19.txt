<lean>
axiom obj : Type

-- Entities
axiom Rockie : obj

-- Predicates
axiom Rabbit : obj → Prop
axiom Squirrel : obj → Prop
axiom Turtle : obj → Prop
axiom Spotted : obj → Prop
axiom Cute : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop

-- Premises
-- All rabbits that can be spotted near the campus are cute.
axiom A1 : ∀ x : obj, Rabbit x ∧ Spotted x → Cute x
-- Some turtles can be spotted near the campus.
axiom A2 : ∃ x : obj, Turtle x ∧ Spotted x
-- The only animals that can be spotted near the campus are rabbits and squirrels.
axiom A3 : ∀ x : obj, Spotted x → (Rabbit x ∨ Squirrel x)
-- If something is skittish, then it is not calm.
axiom A4 : ∀ x : obj, Skittish x → ¬Calm x
-- All the squirrels that can be spotted near the campus are skittish.
axiom A5 : ∀ x : obj, Squirrel x ∧ Spotted x → Skittish x
-- Rockie can be spotted near the campus, and it is calm.
axiom A6 : Spotted Rockie ∧ Calm Rockie

-- Derive: Rockie is either cute or skittish (unconditionally).
theorem cute_or_skittish_Rockie : Cute Rockie ∨ Skittish Rockie := by
  have sp : Spotted Rockie := And.left A6
  have h : Rabbit Rockie ∨ Squirrel Rockie := A3 Rockie sp
  cases h with
  | inl r =>
    apply Or.inl
    have rs : Rabbit Rockie ∧ Spotted Rockie := And.intro r sp
    exact A1 Rockie rs
  | inr s =>
    apply Or.inr
    have ss : Squirrel Rockie ∧ Spotted Rockie := And.intro s sp
    exact A5 Rockie ss

-- The given statement holds: If Rockie is not both a turtle and a squirrel, then Rockie is either cute or skittish.
theorem statement_true :
  (¬(Turtle Rockie ∧ Squirrel Rockie)) → (Cute Rockie ∨ Skittish Rockie) := by
  intro _
  exact cute_or_skittish_Rockie

-- Hence, the negation of the statement cannot hold.
theorem negation_impossible :
  ¬( (¬(Turtle Rockie ∧ Squirrel Rockie)) ∧ ¬(Cute Rockie ∨ Skittish Rockie) ) := by
  intro h
  have notQ : ¬(Cute Rockie ∨ Skittish Rockie) := And.right h
  exact notQ cute_or_skittish_Rockie
