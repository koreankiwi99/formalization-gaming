<lean>
axiom obj : Type

-- Predicates
axiom At : obj → Prop
axiom Work : obj → Prop
axiom Need : obj → Prop
axiom OrderTakeout : obj → Prop
axiom EnjoyDining : obj → Prop
axiom Picky : obj → Prop
axiom SpendTimeDining : obj → Prop

-- Premises
axiom R1 : ∀ x : obj, At x ∧ Work x → Need x
axiom R2 : ∀ x : obj, At x ∧ OrderTakeout x → Work x
axiom R3 : ∀ x : obj, At x → (OrderTakeout x ∨ EnjoyDining x)
axiom R4 : ∀ x : obj, At x ∧ EnjoyDining x → ¬Picky x
axiom R5 : ∀ x : obj, At x ∧ EnjoyDining x → SpendTimeDining x

-- Hannah-specific facts
axiom Hannah : obj
axiom HAt : At Hannah
axiom HWork : Work Hannah
axiom HImp : Need Hannah → (¬Picky Hannah ∧ ¬Need Hannah)

-- We prove the negation of the statement: Hannah does not need to earn money.
theorem hannah_not_need : ¬Need Hannah := by
  intro hneed
  have hpair : ¬Picky Hannah ∧ ¬Need Hannah := HImp hneed
  have hnot : ¬Need Hannah := And.right hpair
  exact hnot hneed

-- Additionally, from the general rule and facts we can derive Need Hannah,
-- yielding a contradiction (showing the statement is false under the premises).
theorem hannah_need : Need Hannah := by
  have hw : At Hannah ∧ Work Hannah := And.intro HAt HWork
  exact R1 Hannah hw

theorem statement_contradiction : False := by
  exact hannah_not_need hannah_need
</lean>
