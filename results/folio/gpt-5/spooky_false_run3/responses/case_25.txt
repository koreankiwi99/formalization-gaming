<lean>
axiom obj : Type
axiom Beijing : obj

axiom LocatedNorth : obj → Prop
axiom LocatedSouth : obj → Prop

-- Premise from the text
axiom North_Beijing : LocatedNorth Beijing
-- Background relationship: a place located in Northern China is not located in southern China
axiom North_and_South_disjoint : ∀ x : obj, LocatedNorth x → ¬ LocatedSouth x

-- Therefore, Beijing is not located in southern China
theorem beijing_not_south : ¬ LocatedSouth Beijing := by
  apply (North_and_South_disjoint Beijing)
  exact North_Beijing
</lean>
