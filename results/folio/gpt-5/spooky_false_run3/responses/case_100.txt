  have hm : Mating Harry := P2 Harry h
  have hnm : ¬ Mating Harry := P4 Harry P6
  exact hnm hm

theorem harry_not_animal_and_vertebrate : ¬(Animal Harry ∧ Vertebrate Harry) := by
  intro h
  have hv : Vertebrate Harry := h.right
  have ha : Animal Harry := h.left
  have hb : Backbone Harry := P3 Harry ⟨hv, ha⟩
  have hm : Mating Harry := P2 Harry ⟨ha, hb⟩
