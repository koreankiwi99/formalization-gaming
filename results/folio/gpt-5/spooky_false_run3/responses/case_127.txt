<lean>
axiom Person : Type
axiom John : Person

axiom MakesPlans : Person → Prop
axiom Engaged : Person → Prop
axiom InvitesOthers : Person → Prop
axiom WellAttended : Person → Prop
axiom LargerFamily : Person → Prop
axiom InvitesFriends : Person → Prop

-- All people who make wedding plans are engaged.
axiom A1 : ∀ x : Person, MakesPlans x → Engaged x
-- All people who invite others to come to their ceremony make wedding plans.
axiom A2 : ∀ x : Person, InvitesOthers x → MakesPlans x
-- Anyone who has a well-attended wedding invites others to come to their ceremony.
axiom A3 : ∀ x : Person, WellAttended x → InvitesOthers x
-- Some people that have well-attended weddings have larger families.
axiom A4 : ∃ x : Person, WellAttended x ∧ LargerFamily x

-- If John is engaged, then John does not invite friends to their ceremony and does not make wedding plans.
axiom A5 : Engaged John → ¬InvitesFriends John ∧ ¬MakesPlans John

-- If John has a larger family, then either John has a well-attended wedding or invites friends to their ceremony.
axiom A6 : LargerFamily John → WellAttended John ∨ InvitesFriends John

-- Bridging: inviting friends counts as inviting others.
axiom FriendsAreOthers : ∀ x : Person, InvitesFriends x → InvitesOthers x

theorem not_larger_family_John : ¬LargerFamily John := by
  intro hLF
  have hdisj : WellAttended John ∨ InvitesFriends John := A6 hLF
  cases hdisj with
  | inl hWA =>
    have hIO : InvitesOthers John := A3 John hWA
    have hMP : MakesPlans John := A2 John hIO
    have hEN : Engaged John := A1 John hMP
    have hNotMP : ¬MakesPlans John := (A5 hEN).right
    exact hNotMP hMP
  | inr hIF =>
    have hIO : InvitesOthers John := FriendsAreOthers John hIF
    have hMP : MakesPlans John := A2 John hIO
    have hEN : Engaged John := A1 John hMP
    have hNotMP : ¬MakesPlans John := (A5 hEN).right
    exact hNotMP hMP
</lean>

ANSWER: False
