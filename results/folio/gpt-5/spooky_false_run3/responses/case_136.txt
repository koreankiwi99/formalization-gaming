<lean>
axiom obj : Type

axiom Sucks : obj → Prop
axiom Plunger : obj → Prop
axiom Vacuum : obj → Prop
axiom Vampire : obj → Prop
axiom HouseholdAppliance : obj → Prop

axiom Space : obj
axiom Duster : obj

-- Premises
axiom P1 : ∀ x : obj, Plunger x → Sucks x          -- Plungers suck.
axiom P2 : ∀ x : obj, Vacuum x → Sucks x           -- Vacuums suck.
axiom P3 : ∀ x : obj, Vampire x → Sucks x          -- Vampires suck.
axiom P4 : Vacuum Space                             -- Space is a vacuum.
axiom P5 : HouseholdAppliance Duster                -- Duster is a household appliance
axiom P6 : ¬ Sucks Duster                           -- ...that doesn't suck.

-- Negation of the statement to evaluate: not (for all x, if x is a household appliance then x sucks)
theorem not_all_household_appliances_suck : ¬ (∀ x : obj, HouseholdAppliance x → Sucks x) :=
by
  intro h
  have hs : Sucks Duster := h Duster P5
  exact P6 hs
</lean>

ANSWER: False
