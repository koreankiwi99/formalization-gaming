<lean>
axiom obj : Type
axiom James : obj

axiom Employee : obj → Prop
axiom Manager : obj → Prop
axiom SchedulesMeetings : obj → Prop
axiom GoesToBuildingToday : obj → Prop
axiom LunchInCompany : obj → Prop
axiom LunchAtHome : obj → Prop
axiom RemoteFromHome : obj → Prop
axiom InOtherCountries : obj → Prop
axiom AppearsInCompanyToday : obj → Prop

-- 1. All employees who schedule a meeting with their customers will go to the company building today.
axiom A1 : ∀ x : obj, Employee x → SchedulesMeetings x → GoesToBuildingToday x
-- 2. Everyone who has lunch in the company building schedules meetings with their customers.
axiom A2 : ∀ x : obj, LunchInCompany x → SchedulesMeetings x
-- 3. Employees have lunch either in the company building or at home.
axiom A3 : ∀ x : obj, Employee x → (LunchInCompany x ∨ LunchAtHome x)
-- 4. If an employee has lunch at home, they are working remotely from home.
axiom A4 : ∀ x : obj, Employee x → LunchAtHome x → RemoteFromHome x
-- 5. All employees who are in other countries work remotely from home.
axiom A5 : ∀ x : obj, Employee x → InOtherCountries x → RemoteFromHome x
-- 6. No managers work remotely from home.
axiom A6 : ∀ x : obj, Manager x → ¬RemoteFromHome x
-- 7. James will appear in the company today iff he is a manager.
axiom A7 : AppearsInCompanyToday James ↔ Manager James

-- Useful derived facts (do not conclude the target negation unconditionally):

-- If James is a manager and an employee, then he must have lunch in the company building.
theorem james_lunch_in_company_if_manager_and_employee :
  Manager James → Employee James → LunchInCompany James := by
  intro jm je
  have disj := A3 James je
  cases disj with
  | inl linco =>
      exact linco
  | inr lhome =>
      have rfh := A4 James je lhome
      have nrfh := A6 James jm
      exact False.elim (nrfh rfh)

-- If James appears in the company today and is an employee, then he must have lunch in the company building.
theorem james_lunch_in_company_if_appears_and_employee :
  Employee James → AppearsInCompanyToday James → LunchInCompany James := by
  intro je ja
  have jm : Manager James := (Iff.mp A7) ja
  exact james_lunch_in_company_if_manager_and_employee jm je
