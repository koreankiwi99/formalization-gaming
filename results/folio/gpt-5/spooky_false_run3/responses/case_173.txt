<lean>
axiom obj : Type

-- Individuals
axiom Taylor : obj

-- Predicates
axiom SaladReg : obj → Prop
axiom HealthConscious : obj → Prop
axiom HCChildHome : obj → Prop
axiom FulfillDaily : obj → Prop
axiom Disregard : obj → Prop
axiom GymDaily : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, SaladReg x → HealthConscious x
axiom A2 : ∀ x : obj, HCChildHome x → SaladReg x
axiom A3 : ∀ x : obj, FulfillDaily x → HCChildHome x
axiom A4 : ∀ x : obj, Disregard x → ¬HealthConscious x
axiom A5 : ∀ x : obj, GymDaily x → FulfillDaily x
axiom A6 : (HCChildHome Taylor ∧ Disregard Taylor) ∨ (¬HCChildHome Taylor ∧ ¬Disregard Taylor)

theorem taylor_not_gym : ¬GymDaily Taylor := by
  intro hGym
  -- From gym daily to home
  have hFul : FulfillDaily Taylor := A5 Taylor hGym
  have hHome : HCChildHome Taylor := A3 Taylor hFul
  -- Show the first disjunct (home ∧ disregard) is impossible
  have h_not_AB : ¬(HCChildHome Taylor ∧ Disregard Taylor) := by
    intro hAB
    have hHC : HealthConscious Taylor := by
      have hSal : SaladReg Taylor := A2 Taylor hAB.left
      exact A1 Taylor hSal
    have hNotHC : ¬HealthConscious Taylor := A4 Taylor hAB.right
    exact hNotHC hHC
  -- Use the disjunction to get the second conjunct (¬home ∧ ¬disregard)
  have hSecond : (¬HCChildHome Taylor ∧ ¬Disregard Taylor) := by
    cases A6 with
    | inl hAB =>
        exact False.elim (h_not_AB hAB)
    | inr hNeither =>
        exact hNeither
  have hNotHome : ¬HCChildHome Taylor := hSecond.left
  exact hNotHome hHome
</lean>
