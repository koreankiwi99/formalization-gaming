axiom DisregardsWellbeing : person → Prop
axiom VisitsGymDaily : person → Prop

axiom A1 : ∀ x : person, EatsSaladsRegularly x → VeryConscious x
axiom A2 : ∀ x : person, GrewUpHealthHome x → EatsSaladsRegularly x
axiom A3 : ∀ x : person, FulfillsDailyIntake x → GrewUpHealthHome x
axiom A4 : ∀ x : person, DisregardsWellbeing x → ¬ VeryConscious x
axiom A5 : ∀ x : person, VisitsGymDaily x → FulfillsDailyIntake x
axiom A6 : (GrewUpHealthHome Taylor ∧ DisregardsWellbeing Taylor) ∨ (¬ GrewUpHealthHome Taylor ∧ ¬ DisregardsWellbeing Taylor)

theorem grewUp_implies_conscious : ∀ x : person, GrewUpHealthHome x → VeryConscious x := by
  intro x hx
  apply A1 x
  exact A2 x hx

theorem gym_implies_grewUp : ∀ x : person, VisitsGymDaily x → GrewUpHealthHome x := by
  intro x hg
  apply A3 x
  exact A5 x hg

theorem taylor_neither_grewUp_nor_gym : ¬ GrewUpHealthHome Taylor ∧ ¬ VisitsGymDaily Taylor := by
  have h1 : ¬ GrewUpHealthHome Taylor ∧ ¬ DisregardsWellbeing Taylor := by
    cases A6 with
    | inl hgd =>
        have hv : VeryConscious Taylor := grewUp_implies_conscious Taylor hgd.left
        have hnv : ¬ VeryConscious Taylor := A4 Taylor hgd.right
        exact False.elim (hnv hv)
    | inr h =>
        exact h
  have nh : ¬ GrewUpHealthHome Taylor := h1.left
  have ng : ¬ VisitsGymDaily Taylor := by
    intro hg
    have hh : GrewUpHealthHome Taylor := gym_implies_grewUp Taylor hg
    exact nh hh
  exact And.intro nh ng
