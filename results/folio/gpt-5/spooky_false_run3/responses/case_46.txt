<lean>
axiom Person : Type
axiom Season : Type

axiom Spring : Season
axiom Summer : Season
axiom Fall : Season
axiom Winter : Season

axiom Favorite : Person → Season

axiom Emma : Person
axiom Mia : Person
axiom James : Person

axiom Student : Person → Prop
axiom WantsLongVacation : Person → Prop

-- Premises
axiom AllStudentsWhoWantLongVacationHaveSummerFav :
  ∀ x : Person, Student x ∧ WantsLongVacation x → Favorite x = Summer

axiom EmmaFav : Favorite Emma = Summer
axiom MiaFavNeEmmaFav : Favorite Mia ≠ Favorite Emma
axiom JamesWants : WantsLongVacation James

-- A derivable fact about Mia (not directly related to James)
theorem mia_not_summer : Favorite Mia ≠ Summer := by
  intro h
  have h' : Favorite Mia = Favorite Emma := by
    exact Eq.trans h (Eq.symm EmmaFav)
  exact MiaFavNeEmmaFav h'

-- What we can say about James if he were a student
theorem james_summer_if_student : Student James → Favorite James = Summer := by
  intro hstu
  have hboth : Student James ∧ WantsLongVacation James := And.intro hstu JamesWants
  exact AllStudentsWhoWantLongVacationHaveSummerFav James hboth
