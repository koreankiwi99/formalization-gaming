axiom obj : Type

-- Predicates
axiom Show : obj → Prop
axiom AirsMon8 : obj → Prop
axiom GivesRosesOnTV : obj → Prop
axiom Bachelor : obj → Prop
axiom PortraysReal : obj → Prop
axiom RealityTV : obj → Prop

-- Facts from the context
axiom exists_rose_show : ∃ s : obj, Show s ∧ AirsMon8 s ∧ GivesRosesOnTV s
axiom R1 : ∀ x : obj, GivesRosesOnTV x → Bachelor x
axiom R2 : ∀ x : obj, Bachelor x → PortraysReal x
axiom R3 : ∀ x : obj, PortraysReal x → RealityTV x

-- Breaking Bad
axiom BreakingBad : obj
axiom BB_is_show : Show BreakingBad
axiom BB_not_reality : ¬RealityTV BreakingBad

-- If Breaking Bad gave out roses on TV, it would be reality TV (contradiction).
theorem bb_not_roses : ¬GivesRosesOnTV BreakingBad := by
  intro h
  have hb : Bachelor BreakingBad := R1 BreakingBad h
  have hp : PortraysReal BreakingBad := R2 BreakingBad hb
  have hr : RealityTV BreakingBad := R3 BreakingBad hp
  exact BB_not_reality hr

-- Negation of the statement to evaluate:
-- "Breaking Bad is a TV show in which roses are given out on Monday at 8 pm."
-- We formalize this as: Show BreakingBad ∧ AirsMon8 BreakingBad ∧ GivesRosesOnTV BreakingBad
theorem breakingbad_statement_false :
  ¬(Show BreakingBad ∧ AirsMon8 BreakingBad ∧ GivesRosesOnTV BreakingBad) := by
  intro h
  have hroses : GivesRosesOnTV BreakingBad := (And.right h).right
  exact bb_not_roses hroses
