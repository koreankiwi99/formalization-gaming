axiom Airport : Type

theorem susan_not_from_LGA : ¬ FliesFrom Susan LGA := by
  intro h
  have hAnd : FliesFrom Susan LGA ∧ FliesTo Susan LGA := And.intro h Susan_to_LGA
  exact (not_same_airport Susan LGA) hAnd
