<lean>
axiom Show : Type
axiom StrangerThings : Show
axiom BlackMirror : Show

axiom NetflixShow : Show → Prop
axiom Popular : Show → Prop
axiom Binge : Show → Prop
axiom Download : Show → Prop
axiom ShareWithLisa : Show → Prop

axiom ST_netflix : NetflixShow StrangerThings
axiom ST_popular : Popular StrangerThings

axiom R1 : ∀ s : Show, NetflixShow s → Popular s → Binge s
axiom R2 : ∀ s : Show, NetflixShow s → (Binge s ↔ Download s)
axiom BM_not_download : ¬Download BlackMirror
axiom BM_netflix : NetflixShow BlackMirror
axiom R3 : ∀ s : Show, NetflixShow s → Binge s → ShareWithLisa s

theorem not_pop_blackmirror : ¬Popular BlackMirror := by
  intro hpop
  have hnet : NetflixShow BlackMirror := BM_netflix
  have hbinge : Binge BlackMirror := R1 BlackMirror hnet hpop
  have hiff : Binge BlackMirror ↔ Download BlackMirror := R2 BlackMirror hnet
  have hdl : Download BlackMirror := (Iff.mp hiff) hbinge
  exact BM_not_download hdl
</lean>

ANSWER: False
