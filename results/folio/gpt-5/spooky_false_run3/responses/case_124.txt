axiom Animal : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, Pet x ∧ Mammal x → Animal x
axiom P2 : ∀ x : obj, Monkey x → Mammal x
axiom P3 : ∀ x : obj, Pet x → (Monkey x ∨ Bird x)
axiom P4 : ∀ x : obj, Bird x ∧ Pet x → Fly x
axiom P5 : ∀ x : obj, Animal x ∧ Pet x → Breathe x
axiom P6 : ∀ x : obj, Pet x ∧ Fly x → Wings x
axiom P7 : Pet Rock
axiom P8 : Fly Rock ∨ Bird Rock ∨ ¬Breathe Rock

-- Derived facts
theorem pet_monkey_animal : ∀ x : obj, Pet x → Monkey x → Animal x := by
  intro x px mx
  have mm : Mammal x := P2 x mx
  have h : Pet x ∧ Mammal x := And.intro px mm
  exact P1 x h

theorem pet_monkey_breathe : ∀ x : obj, Pet x → Monkey x → Breathe x := by
  intro x px mx
  have ax : Animal x := pet_monkey_animal x px mx
  have h : Animal x ∧ Pet x := And.intro ax px
  exact P5 x h

-- If Rock were a monkey, then Rock must fly or be a bird (from the disjunction plus breathing)
theorem rock_monkey_implies_fly_or_bird : Monkey Rock → (Fly Rock ∨ Bird Rock) := by
  intro m
  have br : Breathe Rock := pet_monkey_breathe Rock P7 m
  have h := P8
  cases h with
  | inl f => exact Or.inl f
  | inr t =>
    cases t with
    | inl b => exact Or.inr b
    | inr nb =>
      have contra : False := nb br
      exact False.elim contra
