<lean>
axiom Person : Type
axiom Djokovic : Person

axiom AtMixer : Person → Prop
axiom GrandSlamChampion : Person → Prop
axiom OscarNominatedActor : Person → Prop
axiom ProTennisPlayer : Person → Prop
axiom Celebrity : Person → Prop
axiom Athlete : Person → Prop
axiom WellPaid : Person → Prop
axiom Famous : Person → Prop
axiom LivesInTaxHaven : Person → Prop

-- Everyone at the mixer is a Grand Slam champion or an Oscar-nominated actor.
axiom A1 : ∀ x : Person, AtMixer x → (GrandSlamChampion x ∨ OscarNominatedActor x)
-- Every Grand Slam champion at the mixer is a professional tennis player.
axiom A2 : ∀ x : Person, AtMixer x ∧ GrandSlamChampion x → ProTennisPlayer x
-- All Oscar-nominated actors at the mixer are celebrities.
axiom A3 : ∀ x : Person, AtMixer x ∧ OscarNominatedActor x → Celebrity x
-- All professional tennis players at the mixer are athletes.
axiom A4 : ∀ x : Person, AtMixer x ∧ ProTennisPlayer x → Athlete x
-- If a person at the mixer is a celebrity, then they are well paid.
axiom A5 : ∀ x : Person, AtMixer x ∧ Celebrity x → WellPaid x
-- If a person at the mixer is an athlete, then they are famous.
axiom A6 : ∀ x : Person, AtMixer x ∧ Athlete x → Famous x
-- All well-paid people at the mixer live in tax havens.
axiom A7 : ∀ x : Person, AtMixer x ∧ WellPaid x → LivesInTaxHaven x
-- Djokovic is at the mixer
axiom A8 : AtMixer Djokovic
-- If Djokovic is a famous athlete, then Djokovic is well-paid.
axiom A9 : Famous Djokovic ∧ Athlete Djokovic → WellPaid Djokovic

-- From the premises, we can derive that Djokovic lives in a tax haven.
theorem djokovic_lives_in_tax_haven : LivesInTaxHaven Djokovic := by
  have atD : AtMixer Djokovic := A8
  have h := A1 Djokovic atD
  cases h with
  | inl gs =>
    have pro : ProTennisPlayer Djokovic := A2 Djokovic (And.intro atD gs)
    have ath : Athlete Djokovic := A4 Djokovic (And.intro atD pro)
    have fam : Famous Djokovic := A6 Djokovic (And.intro atD ath)
    have wp : WellPaid Djokovic := A9 (And.intro fam ath)
    exact A7 Djokovic (And.intro atD wp)
  | inr oscar =>
    have cel : Celebrity Djokovic := A3 Djokovic (And.intro atD oscar)
    have wp : WellPaid Djokovic := A5 Djokovic (And.intro atD cel)
    exact A7 Djokovic (And.intro atD wp)
</lean>
