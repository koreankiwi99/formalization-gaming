-- Dried Thai chilies are spicy or mala hotpots or not baked sweets

  | inr h2 =>
      cases h2 with
      | inl hM =>
          have hS : Spicy DriedThaiChilies := R3 DriedThaiChilies hM
          exact Or.inl hS
      | inr hNotB =>
          exact Or.inr hNotB

-- Spicy DTC implies not BakedSweet DTC

  cases h with
  | inl hS =>
      exact spicy_implies_notB hS
  | inr hNotB =>
      exact hNotB
