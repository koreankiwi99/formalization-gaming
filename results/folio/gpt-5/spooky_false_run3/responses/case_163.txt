<lean>
axiom obj : Type

-- Entities
axiom BreakingBad : obj
axiom Mon8pm : obj   -- the show airing at 8 pm on Monday

-- Predicates
axiom Show : obj → Prop
axiom GivesRosesOnTV : obj → Prop
axiom EpisodeOfTheBachelor : obj → Prop
axiom PortraysRealLives : obj → Prop
axiom RealityTV : obj → Prop

-- Premises
-- Some (the) show airing at 8 pm on Monday gives out roses on TV.
axiom P1 : GivesRosesOnTV Mon8pm
-- If a show gives out roses on TV, then it is an episode of The Bachelor.
axiom P2 : ∀ x : obj, GivesRosesOnTV x → EpisodeOfTheBachelor x
-- The Bachelor portrays the lives of real people: any episode of it portrays real lives.
axiom P3 : ∀ x : obj, EpisodeOfTheBachelor x → PortraysRealLives x
-- All shows portraying the lives of real people are reality TV shows.
axiom P4 : ∀ x : obj, PortraysRealLives x → RealityTV x
-- Breaking Bad is a show.
axiom P5 : Show BreakingBad
-- Breaking Bad is not a reality TV show.
axiom P6 : ¬RealityTV BreakingBad

-- Conclusion to negate: "Breaking Bad is on Monday at 8 pm" is modeled as BreakingBad = Mon8pm.
theorem breakingbad_not_mon8pm : BreakingBad ≠ Mon8pm := by
  intro h
  have e1 : EpisodeOfTheBachelor Mon8pm := P2 Mon8pm P1
  have e2 : PortraysRealLives Mon8pm := P3 Mon8pm e1
  have e3 : RealityTV Mon8pm := P4 Mon8pm e2
  have : RealityTV BreakingBad := by
    -- rewrite the goal using h: BreakingBad = Mon8pm
    simpa [h] using e3
  exact P6 this
</lean>

ANSWER: False
