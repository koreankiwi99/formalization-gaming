axiom Evergreen : obj → Prop
axiom Worship : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, Fir x → Evergreen x
axiom A2 : ∃ x : obj, Worship x ∧ Fir x

-- From the premises, we can derive that some evergreen is an object of worship.
theorem exists_evergreen_worship : ∃ x : obj, Evergreen x ∧ Worship x := by
  rcases A2 with ⟨x, hx⟩
  have hxW : Worship x := And.left hx
  have hxF : Fir x := And.right hx
  have hxE : Evergreen x := A1 x hxF
  exact ⟨x, And.intro hxE hxW⟩
