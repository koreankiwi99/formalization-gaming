axiom obj : Type
axiom Marvin : obj

axiom Alien : obj → Prop
axiom Extraterrestrial : obj → Prop
axiom Human : obj → Prop
axiom FromMars : obj → Prop
axiom FromEarth : obj → Prop
axiom High : obj → Prop

axiom A1 : ∀ x : obj, Alien x → Extraterrestrial x
axiom A2 : ∀ x : obj, FromMars x → Alien x
axiom A3 : ∀ x : obj, Extraterrestrial x → ¬Human x
axiom A4 : ∀ x : obj, High x ∧ FromEarth x → Human x
axiom T1 : High Marvin
axiom T2 : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬FromEarth Marvin ∧ ¬FromMars Marvin)
axiom A5 : ¬FromEarth Marvin → Extraterrestrial Marvin

-- Marvin cannot be both from Earth and from Mars
theorem not_both_earth_mars : ¬(FromEarth Marvin ∧ FromMars Marvin) := by
  intro h
  have hFE : FromEarth Marvin := h.left
  have hFM : FromMars Marvin := h.right
  have hAlien : Alien Marvin := A2 Marvin hFM
  have hEx : Extraterrestrial Marvin := A1 Marvin hAlien
  have hHuman : Human Marvin := A4 Marvin (And.intro T1 hFE)
  have hNotHuman : ¬Human Marvin := A3 Marvin hEx
  exact hNotHuman hHuman

-- Therefore Marvin is from neither Earth nor Mars
theorem from_neither : ¬FromEarth Marvin ∧ ¬FromMars Marvin := by
  cases T2 with
  | inl h =>
      have : False := not_both_earth_mars h
      exact False.elim this
  | inr h =>
      exact h

-- From not being from Earth, Marvin is extraterrestrial
theorem marvin_extraterrestrial : Extraterrestrial Marvin := by
  apply A5
  exact from_neither.left
