<lean>
axiom obj : Type
axiom colorectal : obj

axiom Deadly : obj → Prop
axiom LowSurvival : obj → Prop
axiom SevereCancer : obj → Prop
axiom BileDuctCancer : obj → Prop
axiom Cholangiocarcinoma : obj → Prop
axiom MildFlu : obj → Prop

-- If something is a deadly disease, then it comes with a low survival rate.
axiom R0 : ∀ x : obj, Deadly x → LowSurvival x
-- Severe cancers are deadly diseases.
axiom R1 : ∀ x : obj, SevereCancer x → Deadly x
-- Bile duct cancer is a severe form cancer.
axiom R2 : ∀ x : obj, BileDuctCancer x → SevereCancer x
-- All Cholangiocarcinoma is bile duct cancer.
axiom R3 : ∀ x : obj, Cholangiocarcinoma x → BileDuctCancer x
-- Mild flu comes with a low survival rate.
axiom R4 : ∀ x : obj, MildFlu x → LowSurvival x
-- Colorectal cancer is not both a bile duct cancer and with a low survival rate.
axiom Tcol : ¬(BileDuctCancer colorectal ∧ LowSurvival colorectal)

theorem not_cholangio_colorectal : ¬Cholangiocarcinoma colorectal := by
  intro hP
  have hR : BileDuctCancer colorectal := R3 colorectal hP
  have hS : SevereCancer colorectal := R2 colorectal hR
  have hD : Deadly colorectal := R1 colorectal hS
  have hL : LowSurvival colorectal := R0 colorectal hD
  exact Tcol (And.intro hR hL)

-- Statement:
-- Colorectal cancer is a form of Cholangiocarcinoma and
-- it is a kind of mild flu or a kind of bile duct cancer,
-- or all of the above.
-- Formalized as:
-- (Cholangiocarcinoma colorectal ∧ (MildFlu colorectal ∨ BileDuctCancer colorectal))
-- ∨ (Cholangiocarcinoma colorectal ∧ MildFlu colorectal ∧ BileDuctCancer colorectal)
theorem negate_statement :
  ¬(
    (Cholangiocarcinoma colorectal ∧ (MildFlu colorectal ∨ BileDuctCancer colorectal))
    ∨
    (Cholangiocarcinoma colorectal ∧ MildFlu colorectal ∧ BileDuctCancer colorectal)
  ) := by
  intro h
  have hP : Cholangiocarcinoma colorectal :=
    Or.elim h
      (fun h1 => And.left h1)
      (fun h2 => And.left h2)
  exact not_cholangio_colorectal hP
</lean>
