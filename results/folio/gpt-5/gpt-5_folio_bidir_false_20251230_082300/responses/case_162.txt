<lean>
axiom obj : Type
axiom Bird : obj → Prop
axiom Hawk : obj → Prop
axiom Swims : obj → Prop

-- Premises
axiom Hawk_never_swims : ∀ x : obj, Hawk x → ¬Swims x
axiom some_bird_is_hawk : ∃ x : obj, Bird x ∧ Hawk x

-- Negation of the conclusion: not all birds swim
theorem not_all_birds_swim : ¬(∀ x : obj, Bird x → Swims x) :=
by
  intro all_birds_swim
  cases some_bird_is_hawk with
  | intro a h =>
    have Ba : Bird a := And.left h
    have Ha : Hawk a := And.right h
    have nSa : ¬Swims a := Hawk_never_swims a Ha
    have Sa : Swims a := all_birds_swim a Ba
    exact nSa Sa
</lean>

ANSWER: False
