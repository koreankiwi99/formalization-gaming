<lean>
axiom obj : Type
axiom James : obj

axiom Student : obj → Prop
axiom WorkLib : obj → Prop
axiom FromCS : obj → Prop
axiom PartTimeUni : obj → Prop
axiom TakingDB : obj → Prop
axiom TakingWithDavid : obj → Prop
axiom WorkLab : obj → Prop

-- None of the students who currently work in the library are from CS.
axiom A1 : ∀ x : obj, Student x ∧ WorkLib x → ¬FromCS x
-- All students with university part-time jobs currently work in the library.
axiom A2 : ∀ x : obj, Student x ∧ PartTimeUni x → WorkLib x
-- All students taking the database course are from CS.
axiom A3 : ∀ x : obj, Student x ∧ TakingDB x → FromCS x
-- All students taking a class with Professor David are taking the database course.
axiom A4 : ∀ x : obj, Student x ∧ TakingWithDavid x → TakingDB x
-- All students working in the lab are taking a class with Professor David.
axiom A5 : ∀ x : obj, Student x ∧ WorkLab x → TakingWithDavid x
-- James is a student working in the lab.
axiom A6 : Student James ∧ WorkLab James
-- James doesn't work in the lab or have a part-time job offered by the university. (¬(Lab ∨ PartTime))
axiom A7 : ¬(WorkLab James ∨ PartTimeUni James)

-- We prove the negation of the statement: James has a part-time job offered by the university.
theorem james_not_parttime : ¬PartTimeUni James := by
  intro hpt
  have hS : Student James := And.left A6
  have hWL : WorkLab James := And.right A6
  have hDavid : TakingWithDavid James := A5 James (And.intro hS hWL)
  have hDB : TakingDB James := A4 James (And.intro hS hDavid)
  have hCS : FromCS James := A3 James (And.intro hS hDB)
  have hLib : WorkLib James := A2 James (And.intro hS hpt)
  have hNotCS : ¬FromCS James := A1 James (And.intro hS hLib)
  exact hNotCS hCS
</lean>

ANSWER: False
