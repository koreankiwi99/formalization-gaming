-- Useful consequences
theorem rockie_not_skittish : ¬Skittish Rockie := by
  intro hsk
  have hnc : ¬Calm Rockie := R4 Rockie hsk
  have hcalm : Calm Rockie := And.right R6
  exact hnc hcalm

theorem rockie_not_squirrel : ¬Squirrel Rockie := by
  intro hsq
  have hnear : Near Rockie := And.left R6
  have hsk : Skittish Rockie := R5 Rockie (And.intro hsq hnear)
  exact rockie_not_skittish hsk

theorem rockie_rabbit : Rabbit Rockie := by

  have h : Rabbit Rockie ∨ Squirrel Rockie := R3 Rockie hnear
  exact Or.elim h (fun hr => hr) (fun hs => False.elim (rockie_not_squirrel hs))

  have hnear : Near Rockie := And.left R6
  have hrab : Rabbit Rockie := rockie_rabbit
  exact R1 Rockie (And.intro hrab hnear)
