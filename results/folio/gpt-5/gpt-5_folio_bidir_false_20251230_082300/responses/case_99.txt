axiom obj : Type
axiom Animal : obj → Prop
axiom Invertebrate : obj → Prop
axiom Vertebrate : obj → Prop
axiom Backbone : obj → Prop
axiom ReproMF : obj → Prop
axiom Bee : obj → Prop
axiom QueenBee : obj → Prop
axiom Harry : obj

-- Animals are either invertebrates or vertebrates.
axiom A1 : ∀ x : obj, Animal x → (Invertebrate x ∨ Vertebrate x)
-- All animals with backbones reproduce by male-and-female mating.
axiom A2 : ∀ x : obj, Animal x → Backbone x → ReproMF x
-- All vertebrate animals have a backbone.
axiom A3 : ∀ x : obj, Animal x → Vertebrate x → Backbone x
-- All bees do not reproduce by male-and-female mating.
axiom A4 : ∀ x : obj, Bee x → ¬ReproMF x
-- All queen bees are bees.
axiom A5 : ∀ x : obj, QueenBee x → Bee x
-- Harry is a bee.
axiom T1 : Bee Harry

-- From the given axioms we can derive some incompatibilities involving Harry:
theorem not_animal_backbone_Harry : ¬(Animal Harry ∧ Backbone Harry) := by
  intro h
  have ha : Animal Harry := h.left
  have hb : Backbone Harry := h.right
  have repro : ReproMF Harry := A2 Harry ha hb
  have nrepro : ¬ReproMF Harry := A4 Harry T1
  exact nrepro repro

theorem not_animal_vertebrate_Harry : ¬(Animal Harry ∧ Vertebrate Harry) := by
  intro h
  have ha : Animal Harry := h.left
  have hv : Vertebrate Harry := h.right
  have hb : Backbone Harry := A3 Harry ha hv
  exact not_animal_backbone_Harry ⟨ha, hb⟩
