<lean>
axiom obj : Type
axiom Susan : obj
axiom John : obj
axiom LGA : obj
axiom FliesTo : obj → obj → Prop
axiom FliesFrom : obj → obj → Prop

-- Premises
axiom T1 : FliesTo Susan LGA
axiom T2 : FliesFrom John LGA
axiom R1 : ∀ p a : obj, ¬(FliesFrom p a ∧ FliesTo p a)

-- Goal: prove that John does not fly to LGA
theorem john_not_to_lga : ¬FliesTo John LGA := by
  intro hTo
  have hnot : ¬(FliesFrom John LGA ∧ FliesTo John LGA) := R1 John LGA
  apply hnot
  exact And.intro T2 hTo
</lean>

ANSWER: False
