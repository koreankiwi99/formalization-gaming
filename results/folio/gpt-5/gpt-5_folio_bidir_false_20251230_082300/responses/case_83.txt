axiom ManyNeg : obj → Prop

-- If a restaurant is listed in Yelp’s recommendations, then it has not received many negative reviews.
axiom A1 : ∀ x : obj, Listed x → ¬ManyNeg x
-- All restaurants with a rating greater than four are listed in Yelp’s recommendations.
axiom A2 : ∀ x : obj, RatingGT4 x → Listed x
-- Some restaurants that do not provide take-out service receive many negative reviews.
axiom A3 : ∃ x : obj, ¬ProvidesTakeout x ∧ ManyNeg x
-- All restaurants popular among local residents have ratings greater than four.
axiom A4 : ∀ x : obj, Popular x → RatingGT4 x
-- The Hamden Plaza Subway store has a rating > 4, or it is popular among local residents.
axiom A5 : RatingGT4 H ∨ Popular H

-- From A5 and A4, we can derive RatingGT4 H in any case.
theorem rating_H : RatingGT4 H := by
  cases A5 with
  | inl r => exact r
  | inr p => exact A4 H p

-- Hence H is listed and thus cannot have many negative reviews.
theorem notManyNeg_H : ¬ManyNeg H := by
  have l : Listed H := A2 H rating_H
  exact A1 H l

-- The given statement is actually true vacuously since ManyNeg H is impossible.
theorem statement_true :
  (ProvidesTakeout H ∧ ManyNeg H) → (RatingGT4 H ∧ ¬ProvidesTakeout H) := by
  intro h
  have m : ManyNeg H := And.right h
  have nm : ¬ManyNeg H := notManyNeg_H
  have f : False := nm m
  exact False.elim f
