axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop

axiom MalaHotpot : obj → Prop
axiom FromBBM : obj → Prop

-- No baked sweets are spicy.
axiom R1 : ∀ x : obj, BakedSweet x → ¬ Spicy x
-- All cupcakes are baked sweets.
axiom R2 : ∀ x : obj, Cupcake x → BakedSweet x
-- All mala hotpots are spicy.
axiom R3 : ∀ x : obj, MalaHotpot x → Spicy x
-- All products from Baked by Melissa are cupcakes.
axiom R4 : ∀ x : obj, FromBBM x → Cupcake x
-- Dried Thai chilies are spicy or mala hotpots or not baked sweets.
axiom T5 : Spicy DriedThaiChilies ∨ MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies

-- From the premises we can actually prove the original statement (not its negation):
theorem dtc_not_frombbm : ¬ FromBBM DriedThaiChilies :=
by
  intro hfrom
  have hcup : Cupcake DriedThaiChilies := R4 DriedThaiChilies hfrom
  have hbs : BakedSweet DriedThaiChilies := R2 DriedThaiChilies hcup
  have hnsp : ¬ Spicy DriedThaiChilies := R1 DriedThaiChilies hbs
  have hnmh : ¬ MalaHotpot DriedThaiChilies := fun hm => hnsp (R3 DriedThaiChilies hm)
  have hnnbs : ¬ ¬ BakedSweet DriedThaiChilies := fun hnb => hnb hbs
  have contra : False :=
    Or.elim T5
      (fun hsp => hnsp hsp)
      (fun h =>
        Or.elim h
          (fun hmh => hnmh hmh)
          (fun hnotbs => hnnbs hnotbs))
  exact contra
