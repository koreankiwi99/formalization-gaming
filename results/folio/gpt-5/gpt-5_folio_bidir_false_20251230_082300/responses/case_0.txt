<lean>
axiom obj : Type

-- Predicates
axiom InClub : obj → Prop
axiom PerformOften : obj → Prop
axiom AttendEngaged : obj → Prop
axiom InactiveDisinterested : obj → Prop
axiom ChaperoneDances : obj → Prop
axiom StudentAtSchool : obj → Prop
axiom YoungChild : obj → Prop
axiom Teenager : obj → Prop
axiom WishFurtherAcademics : obj → Prop

-- Individuals
axiom Bonnie : obj

-- Premises
-- 1. Club members who perform often attend and are very engaged with school events.
axiom R1 : ∀ x : obj, InClub x ∧ PerformOften x → AttendEngaged x
-- 2. Club members either perform often or are inactive and disinterested.
axiom R2 : ∀ x : obj, InClub x → PerformOften x ∨ InactiveDisinterested x
-- 3. Club members who chaperone dances are not students at the school.
axiom R3 : ∀ x : obj, InClub x ∧ ChaperoneDances x → ¬ StudentAtSchool x
-- 4. Inactive and disinterested club members chaperone high school dances.
axiom R4 : ∀ x : obj, InClub x ∧ InactiveDisinterested x → ChaperoneDances x
-- 5. Young children or teenagers in the club who wish to further academics are students.
axiom R5 : ∀ x : obj, InClub x ∧ (YoungChild x ∨ Teenager x) ∧ WishFurtherAcademics x → StudentAtSchool x

-- 6. Bonnie is in the club, and (AttendEngaged ↔ StudentAtSchool) given as two-way disjunction.
axiom Bonnie_inClub : InClub Bonnie
axiom Bonnie_equiv : (AttendEngaged Bonnie ∧ StudentAtSchool Bonnie) ∨ (¬AttendEngaged Bonnie ∧ ¬StudentAtSchool Bonnie)

-- Some derived consequences about Bonnie (do not yield ¬PerformOften Bonnie)
theorem Bonnie_perf_implies_student : PerformOften Bonnie → StudentAtSchool Bonnie := by
  intro hPerf
  have hA : AttendEngaged Bonnie := R1 Bonnie (And.intro Bonnie_inClub hPerf)
  cases Bonnie_equiv with
  | inl h =>
      exact h.right
  | inr h =>
      have : False := h.left hA
      exact False.elim this

theorem Bonnie_inactive_implies_not_student : InactiveDisinterested Bonnie → ¬StudentAtSchool Bonnie := by
  intro hInact
  have hChap : ChaperoneDances Bonnie := R4 Bonnie (And.intro Bonnie_inClub hInact)
  have hNotStu : ¬StudentAtSchool Bonnie := R3 Bonnie (And.intro Bonnie_inClub hChap)
  exact hNotStu
