-- If a film is appropriate for people of all ages, then children can watch it without guidance.
axiom R1 : ∀ x : obj, AppropriateForAllAges x → ChildrenCanWatchWithoutGuidance x
-- If a film contains excessive erotic and violent content, children cannot watch it without guidance.
axiom R2 : ∀ x : obj, ContainsExcessiveEroticViolentContent x → ¬ ChildrenCanWatchWithoutGuidance x
-- If a film is rated General Audience, then it is appropriate for people of all ages.
axiom R3 : ∀ x : obj, RatedGeneralAudience x → AppropriateForAllAges x
-- All family-friendly animated films are rated General Audience.
axiom R4 : ∀ x : obj, FamilyFriendlyAnimated x → RatedGeneralAudience x
-- All movies in the Frozen series are family-friendly animated films.
axiom R5 : ∀ x : obj, InFrozenSeries x → FamilyFriendlyAnimated x

-- “Hachi: A dog’s Tale” is a film.
axiom T_HachiFilm : IsFilm Hachi
-- “Hachi: A dog’s Tale” is either a family-friendly animated film or is appropriate for people of all ages.
axiom T_Hachi_FFAF_or_AFAA : FamilyFriendlyAnimated Hachi ∨ AppropriateForAllAges Hachi

-- From the disjunction, Hachi is watchable by children without guidance.
theorem Hachi_CWG : ChildrenCanWatchWithoutGuidance Hachi := by
  apply Or.elim T_Hachi_FFAF_or_AFAA
  · intro hFFAF
    have hRGA : RatedGeneralAudience Hachi := R4 Hachi hFFAF
    have hAFAA : AppropriateForAllAges Hachi := R3 Hachi hRGA
    exact R1 Hachi hAFAA
  · intro hAFAA
    exact R1 Hachi hAFAA

-- Therefore, Hachi does not contain excessive erotic and violent content.

  intro hCEEV
  have hNotCWG : ¬ ChildrenCanWatchWithoutGuidance Hachi := R2 Hachi hCEEV
  exact hNotCWG Hachi_CWG
