axiom obj : Type
axiom Tom : obj

axiom Nice : obj → Prop
axiom Mean : obj → Prop
axiom Grumpy : obj → Prop
axiom AnimalLover : obj → Prop
axiom PetOwner : obj → Prop

-- No one nice to animals is also mean to animals.
axiom R1 : ∀ x : obj, Nice x → ¬Mean x
-- Some grumpy people are mean to animals.
axiom R2 : ∃ x : obj, Grumpy x ∧ Mean x
-- All animal lovers are nice to animals.
axiom R3 : ∀ x : obj, AnimalLover x → Nice x
-- All pet owners love animals.
axiom R4 : ∀ x : obj, PetOwner x → AnimalLover x
-- Tom is a pet owner.
axiom T1 : PetOwner Tom

def Statement : Prop :=
  (¬(Grumpy Tom ∧ Mean Tom)) → (¬Grumpy Tom ∧ ¬AnimalLover Tom)

lemma TomAnimalLover : AnimalLover Tom :=
  R4 Tom T1

lemma TomNice : Nice Tom :=
  R3 Tom TomAnimalLover

lemma TomNotMean : ¬Mean Tom :=
  R1 Tom TomNice

lemma not_both_Grumpy_and_Mean_Tom : ¬(Grumpy Tom ∧ Mean Tom) := by
  intro h
  have hm : Mean Tom := h.right
  exact TomNotMean hm

theorem negate_Statement : ¬Statement := by
  intro h
  have ante : ¬(Grumpy Tom ∧ Mean Tom) := not_both_Grumpy_and_Mean_Tom
  have concl : ¬Grumpy Tom ∧ ¬AnimalLover Tom := h ante
  have notAL : ¬AnimalLover Tom := concl.right
  exact notAL TomAnimalLover
