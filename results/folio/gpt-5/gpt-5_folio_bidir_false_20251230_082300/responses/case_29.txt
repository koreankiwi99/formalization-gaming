<lean>
axiom obj : Type
axiom Marvin : obj

axiom Alien : obj → Prop
axiom Extraterrestrial : obj → Prop
axiom Human : obj → Prop
axiom FromEarth : obj → Prop
axiom FromMars : obj → Prop
axiom HighlyIntelligent : obj → Prop

-- All aliens are extraterrestrials.
axiom A1 : ∀ x : obj, Alien x → Extraterrestrial x
-- If someone is from Mars, then they are an alien.
axiom A2 : ∀ x : obj, FromMars x → Alien x
-- No extraterrestrials are human.
axiom A3 : ∀ x : obj, Extraterrestrial x → ¬Human x
-- All highly intelligent beings from Earth are humans.
axiom A4 : ∀ x : obj, (HighlyIntelligent x ∧ FromEarth x) → Human x
-- Marvin is highly intelligent.
axiom A5 : HighlyIntelligent Marvin
-- Marvin is either from Earth and from Mars, or from neither.
axiom A6 : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬FromEarth Marvin ∧ ¬FromMars Marvin)
-- If Marvin is not from Earth, then Marvin is an extraterrestrial.
axiom A7 : ¬FromEarth Marvin → Extraterrestrial Marvin

-- Prove the negation of: If Marvin is not from Mars, then Marvin is a human.
theorem not_imp : ¬(¬FromMars Marvin → Human Marvin) := by
  -- First, rule out the case that Marvin is from both Earth and Mars
  have hRight : (¬FromEarth Marvin ∧ ¬FromMars Marvin) :=
    Or.elim A6
      (fun hboth => by
        have hFE : FromEarth Marvin := And.left hboth
        have hFM : FromMars Marvin := And.right hboth
        have hAlien : Alien Marvin := A2 Marvin hFM
        have hExt : Extraterrestrial Marvin := A1 Marvin hAlien
        have hNotHuman : ¬Human Marvin := A3 Marvin hExt
        have hHuman : Human Marvin := A4 Marvin (And.intro A5 hFE)
        exact False.elim (hNotHuman hHuman)
      )
      (fun hneither => hneither)
  have hnE : ¬FromEarth Marvin := And.left hRight
  have hnM : ¬FromMars Marvin := And.right hRight
  have hExt2 : Extraterrestrial Marvin := A7 hnE
  have hNotHuman : ¬Human Marvin := A3 Marvin hExt2
  intro imp
  have hHuman : Human Marvin := imp hnM
  exact hNotHuman hHuman
</lean>

ANSWER: False
