<lean>
axiom obj : Type

-- Entities
axiom PicurisMountains : obj
axiom HardingPegmatiteMine : obj

-- Predicates
axiom MountainRange : obj → Prop
axiom InNewMexico : obj → Prop
axiom InTexas : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop

-- Context axioms
axiom PM_is_MR : MountainRange PicurisMountains
axiom PM_in_NM_or_TX : InNewMexico PicurisMountains ∨ InTexas PicurisMountains
axiom HPM_in_PM : LocatedIn HardingPegmatiteMine PicurisMountains
axiom HPM_donated : Donated HardingPegmatiteMine
axiom No_TX_MR_with_donated_mine :
  ∀ (r m : obj), MountainRange r → InTexas r → LocatedIn m r → Donated m → False

-- We prove the negation of the statement:
-- Statement: "The Harding Pegmatite Mine is not located in a mountain range in New Mexico."
-- Formalized statement: ¬∃ r, MountainRange r ∧ InNewMexico r ∧ LocatedIn HardingPegmatiteMine r
-- We prove its negation: ∃ r, MountainRange r ∧ InNewMexico r ∧ LocatedIn HardingPegmatiteMine r
theorem HPM_located_in_NM_range :
  ∃ r : obj, MountainRange r ∧ InNewMexico r ∧ LocatedIn HardingPegmatiteMine r := by
  -- First, show Picuris Mountains are not in Texas
  have not_in_tx : ¬InTexas PicurisMountains := by
    intro htx
    exact No_TX_MR_with_donated_mine PicurisMountains HardingPegmatiteMine
      PM_is_MR htx HPM_in_PM HPM_donated
  -- From "in NM or in TX" and "not in TX", deduce "in NM"
  have in_nm : InNewMexico PicurisMountains :=
    Or.elim PM_in_NM_or_TX
      (fun hnm => hnm)
      (fun htx => False.elim (not_in_tx htx))
  -- Exhibit Picuris Mountains as the required NM mountain range containing the mine
  refine Exists.intro PicurisMountains ?_
  apply And.intro
  · exact PM_is_MR
  · apply And.intro
    · exact in_nm
    · exact HPM_in_PM
</lean>

ANSWER: False
