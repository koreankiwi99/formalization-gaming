axiom Alien : obj → Prop
axiom Extraterrestrial : obj → Prop
axiom Human : obj → Prop
axiom FromMars : obj → Prop
axiom FromEarth : obj → Prop
axiom HighIntelligent : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, Alien x → Extraterrestrial x
axiom A2 : ∀ x : obj, FromMars x → Alien x
axiom A3 : ∀ x : obj, Extraterrestrial x → ¬ Human x
axiom A4 : ∀ x : obj, HighIntelligent x ∧ FromEarth x → Human x
axiom A5 : HighIntelligent Marvin
axiom A6 : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬ FromEarth Marvin ∧ ¬ FromMars Marvin)
axiom A7 : ¬ FromEarth Marvin → Extraterrestrial Marvin

-- From "either both or neither" plus ¬ FromMars Marvin, deduce ¬ FromEarth Marvin
theorem notMars_implies_notEarth : ¬ FromMars Marvin → ¬ FromEarth Marvin := by
  intro hNM
  cases A6 with
  | inl hBoth =>
      have hFM : FromMars Marvin := hBoth.right
      have hFalse : False := hNM hFM
      intro hFE
      exact False.elim hFalse
  | inr hNeither =>
      exact hNeither.left

-- Consequently, ¬ FromMars Marvin implies ¬ Human Marvin (not the target statement)
theorem notMars_implies_notHuman : ¬ FromMars Marvin → ¬ Human Marvin := by
  intro hNM
  have hNFE : ¬ FromEarth Marvin := notMars_implies_notEarth hNM
  have hET : Extraterrestrial Marvin := A7 hNFE
  have hNoHuman : ¬ Human Marvin := A3 Marvin hET
  exact hNoHuman
