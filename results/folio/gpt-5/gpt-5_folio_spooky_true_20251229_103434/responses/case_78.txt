axiom ProductOfBakedByMelissa : obj → Prop

-- No baked sweets are spicy.

-- All cupcakes are baked sweets.
axiom A2 : ∀ x : obj, Cupcake x → BakedSweet x
-- All mala hotpots are spicy.
axiom A3 : ∀ x : obj, MalaHotpot x → Spicy x
-- All products from Baked by Melissa are cupcakes.
axiom A4 : ∀ x : obj, ProductOfBakedByMelissa x → Cupcake x
-- Dried Thai chilies are spicy or mala hotpots or not baked sweets.
axiom A5 : Spicy DriedThaiChilies ∨ MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies

-- From the premises, we can derive that Dried Thai chilies are NOT products of Baked by Melissa.
theorem not_BBM_DTC : ¬ ProductOfBakedByMelissa DriedThaiChilies := by

  have hCup : Cupcake DriedThaiChilies := A4 DriedThaiChilies h

  have hNotMala : ¬ MalaHotpot DriedThaiChilies := fun hm => hNotSpicy (A3 DriedThaiChilies hm)
  have hOr : Spicy DriedThaiChilies ∨ MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies := A5
  cases hOr with
  | inl hs => exact hNotSpicy hs
  | inr h2 =>
    cases h2 with
    | inl hm => exact hNotMala hm
    | inr hnb => exact hnb hBake
