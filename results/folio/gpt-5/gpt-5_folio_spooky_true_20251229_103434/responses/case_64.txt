<lean>
axiom obj : Type

axiom Ted : obj

axiom BrownSwissCattle : obj → Prop
axiom Cow : obj → Prop
axiom Pet : obj → Prop
axiom Domesticated : obj → Prop
axiom Alligator : obj → Prop

-- All Brown Swiss cattle are cows.
axiom A1 : ∀ x : obj, BrownSwissCattle x → Cow x
-- All cows are domesticated animals.
axiom A2 : ∀ x : obj, Cow x → Domesticated x
-- Alligators are not domesticated animals.
axiom A3 : ∀ x : obj, Alligator x → ¬ Domesticated x
-- Ted is an alligator.
axiom A4 : Alligator Ted
-- Some pets are Brown Swiss cattle.
axiom A5 : ∃ x : obj, Pet x ∧ BrownSwissCattle x

theorem ted_not_pet_if_bsc : BrownSwissCattle Ted → ¬ Pet Ted := by
  intro hBSC
  have hCow : Cow Ted := A1 Ted hBSC
  have hDom : Domesticated Ted := A2 Ted hCow
  have hNotDom : ¬ Domesticated Ted := A3 Ted A4
  intro hPet
  exact hNotDom hDom
</lean>

ANSWER: True
