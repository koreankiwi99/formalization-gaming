<lean>
axiom obj : Type

-- Individuals
axiom Ekaterina : obj
axiom Vladimir : obj

-- Predicates
axiom CanRegisterUS : obj → Prop
axiom CanParticipate2024 : obj → Prop
axiom USCitizen : obj → Prop
axiom TaiwaneseCitizen : obj → Prop
axiom RussianFederationOfficial : obj → Prop
axiom ManagerAtGazprom : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, CanRegisterUS x → CanParticipate2024 x
axiom P2 : ∀ x : obj, USCitizen x → CanRegisterUS x
axiom P3 : ∀ x : obj, USCitizen x ∨ TaiwaneseCitizen x
axiom P4 : ∀ x : obj, RussianFederationOfficial x → ¬ TaiwaneseCitizen x
axiom P5 : ¬ TaiwaneseCitizen Vladimir ∧ ¬ ManagerAtGazprom Vladimir
axiom P6 : CanRegisterUS Ekaterina ∨ RussianFederationOfficial Ekaterina

-- Goal: Ekaterina can participate in the 2024 US presidential election or is a manager at Gazprom.
theorem ekaterina_participate_or_manager :
  CanParticipate2024 Ekaterina ∨ ManagerAtGazprom Ekaterina := by
  cases P6 with
  | inl hreg =>
      exact Or.inl (P1 Ekaterina hreg)
  | inr hrfo =>
      have hnotTW : ¬ TaiwaneseCitizen Ekaterina := P4 Ekaterina hrfo
      have hUSorTW : USCitizen Ekaterina ∨ TaiwaneseCitizen Ekaterina := P3 Ekaterina
      have hUS : USCitizen Ekaterina := by
        cases hUSorTW with
        | inl hus => exact hus
        | inr htw => exact False.elim (hnotTW htw)
      have hreg2 : CanRegisterUS Ekaterina := P2 Ekaterina hUS
      have hpart : CanParticipate2024 Ekaterina := P1 Ekaterina hreg2
      exact Or.inl hpart
</lean>

ANSWER: True
