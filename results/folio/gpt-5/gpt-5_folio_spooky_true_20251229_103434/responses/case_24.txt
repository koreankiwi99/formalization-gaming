<lean>
axiom obj : Type
axiom Beijing : obj

axiom Hosted2008SummerOlympics : obj → Prop
axiom HostedWinterOlympics : obj → Prop
axiom Hosted2008SummerParalympics : obj → Prop
axiom HostedSummerOlympics : obj → Prop
axiom HostedSummerParalympics : obj → Prop
axiom HostedWinterParalympics : obj → Prop

-- From the context:
-- "Beijing hosted the 2008 Summer Olympics and 2008 Summer Paralympics Games."
axiom H1 : Hosted2008SummerOlympics Beijing
axiom H2 : Hosted2008SummerParalympics Beijing

-- "Beijing has hosted the Summer and Winter Olympics and the Summer and Winter Paralympics."
axiom H3 : HostedSummerOlympics Beijing
axiom H4 : HostedWinterOlympics Beijing
axiom H5 : HostedSummerParalympics Beijing
axiom H6 : HostedWinterParalympics Beijing

theorem beijing_hosted_2008_summer_and_winter_olympics :
  Hosted2008SummerOlympics Beijing ∧ HostedWinterOlympics Beijing := by
  apply And.intro
  · exact H1
  · exact H4
</lean>

ANSWER: True
