axiom obj : Type

axiom DaniShapiro : obj
axiom FamilyHistory : obj
axiom FamilySecrets : obj
axiom Boston : obj

axiom Podcast : obj → Prop
axiom Novel : obj → Prop
axiom Book : obj → Prop
axiom Person : obj → Prop
axiom Writer : obj → Prop
axiom American : obj → Prop
axiom AmericanCity : obj → Prop
axiom BornIn : obj → obj → Prop
axiom WrittenBy : obj → obj → Prop
axiom NovelWriter : obj → Prop
axiom Created : obj → obj → Prop
axiom WrittenIn2003 : obj → Prop

axiom podcast_not_novel : ∀ x : obj, Podcast x → ¬ Novel x
axiom born_in_american_city : ∀ x y : obj, Person x → BornIn x y → AmericanCity y → American x
axiom novel_rule : ∀ b p : obj, Book b → Novel b → WrittenBy b p → Person p → NovelWriter p

axiom dani_american : American DaniShapiro
axiom dani_writer : Writer DaniShapiro

axiom family_history_written_by : WrittenBy FamilyHistory DaniShapiro
axiom family_history_novel : Novel FamilyHistory
axiom family_history_2003 : WrittenIn2003 FamilyHistory

axiom dani_created_podcast : Created DaniShapiro FamilySecrets
axiom family_secrets_podcast : Podcast FamilySecrets

axiom boston_american_city : AmericanCity Boston

axiom writer_is_person : ∀ x : obj, Writer x → Person x
axiom novel_is_book : ∀ x : obj, Novel x → Book x

theorem dani_is_novel_writer : NovelWriter DaniShapiro := by
  have p1 : Person DaniShapiro := writer_is_person DaniShapiro dani_writer
  have nb : Novel FamilyHistory := family_history_novel
  have bb : Book FamilyHistory := novel_is_book FamilyHistory nb
  have wb : WrittenBy FamilyHistory DaniShapiro := family_history_written_by
  exact novel_rule FamilyHistory DaniShapiro bb nb wb p1
