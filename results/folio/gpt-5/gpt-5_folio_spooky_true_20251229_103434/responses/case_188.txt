<lean>
axiom obj : Type
axiom Yuri : obj
axiom American : obj → Prop
axiom ProBasketballPlayer : obj → Prop
axiom CanDoJumpShots : obj → Prop
axiom LeapsStraightIntoAir : obj → Prop
axiom ActivatesLegMuscles : obj → Prop

axiom premise1 : ∃ x : obj, ProBasketballPlayer x ∧ ¬ American x
axiom premise2 : ∀ x : obj, ProBasketballPlayer x → CanDoJumpShots x
axiom premise3 : ∀ x : obj, CanDoJumpShots x → LeapsStraightIntoAir x
axiom premise4 : ∀ x : obj, LeapsStraightIntoAir x → ActivatesLegMuscles x
axiom premise5 : ¬ ActivatesLegMuscles Yuri

theorem yuri_not_american_pro : ¬ (American Yuri ∧ ProBasketballPlayer Yuri) := by
  intro h
  have hPro : ProBasketballPlayer Yuri := And.right h
  have hJump : CanDoJumpShots Yuri := premise2 Yuri hPro
  have hLeap : LeapsStraightIntoAir Yuri := premise3 Yuri hJump
  have hAct : ActivatesLegMuscles Yuri := premise4 Yuri hLeap
  exact premise5 hAct
</lean>

ANSWER: True
