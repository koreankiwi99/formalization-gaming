<lean>
axiom obj : Type

axiom Invertebrate : obj → Prop
axiom Vertebrate : obj → Prop
axiom Backbone : obj → Prop
axiom MF : obj → Prop
axiom Bee : obj → Prop
axiom QueenBee : obj → Prop

axiom Harry : obj

-- Premises
axiom P1 : ∀ x : obj, Invertebrate x ∨ Vertebrate x
axiom P2 : ∀ x : obj, Backbone x → MF x
axiom P3 : ∀ x : obj, Vertebrate x → Backbone x
axiom P4 : ∀ x : obj, Bee x → ¬ MF x
axiom P5 : ∀ x : obj, QueenBee x → Bee x
axiom P6 : Bee Harry

theorem harry_invertebrate_or_queenbee : Invertebrate Harry ∨ QueenBee Harry := by
  have hNotMF : ¬ MF Harry := P4 Harry P6
  have hInv : Invertebrate Harry := by
    have hdisj := P1 Harry
    cases hdisj with
    | inl hinv => exact hinv
    | inr hvert =>
      have hbb : Backbone Harry := P3 Harry hvert
      have hmf : MF Harry := P2 Harry hbb
      have contra : False := hNotMF hmf
      exact False.elim contra
  exact Or.inl hInv
</lean>

ANSWER: True
