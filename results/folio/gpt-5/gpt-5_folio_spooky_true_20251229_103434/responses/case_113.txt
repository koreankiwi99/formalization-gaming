-- we infer Picuris is not in Texas.

  intro htx
  have hno : ¬ (∃ m, Mine m ∧ LocatedIn m Picuris ∧ Donated m) :=
    A_no_TX_mr_with_donated_mine Picuris A_mr_P htx

    refine Exists.intro HPMine ?_
    exact And.intro A_mine_H (And.intro A_loc_H_P A_don_H)
  exact hno hex

-- Using the disjunction "in NM or in TX" together with ¬ InTX, we conclude InNM.

  | inl hnm => exact hnm
  | inr htx => exact False.elim (Picuris_not_TX htx)
