<lean>
axiom obj : Type

axiom Adores : obj → Prop
axiom Zaha : obj → Prop
axiom Kelly : obj → Prop
axiom Brutalist : obj → Prop
axiom Interesting : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop

axiom T1 : ∀ x : obj, Zaha x ∧ Adores x → Interesting x
axiom T2 : ∀ x : obj, Brutalist x ∧ Adores x → ¬ Interesting x
axiom T3 : ∀ x : obj, Adores x → Zaha x ∨ Kelly x
axiom T4 : ∀ x : obj, Kelly x ∧ Adores x → Evocative x
axiom T5 : ∀ x : obj, Kelly x ∧ Adores x → Dreamy x
axiom T6 : ∀ x : obj, Adores x ∧ Interesting x → Brutalist x ∧ Evocative x

theorem max_design_evocative_or_dreamy : ∀ x : obj, Adores x → Evocative x ∨ Dreamy x := by
  intro x hx
  have hdisj := T3 x hx
  cases hdisj with
  | inl hz =>
      have hi := T1 x (And.intro hz hx)
      have hbe := T6 x (And.intro hx hi)
      exact Or.inl (And.right hbe)
  | inr hk =>
      have hev := T4 x (And.intro hk hx)
      exact Or.inl hev
</lean>

ANSWER: True
