<lean>
axiom obj : Type
axiom Coco : obj

axiom PlayerRG2022 : obj → Prop
axiom FemaleRG2022 : obj → Prop
axiom MaleRG2022 : obj → Prop
axiom LostToIgaRG2022 : obj → Prop
axiom LostToRafaRG2022 : obj → Prop
axiom RankedHighWTA : obj → Prop
axiom MostActiveMajor : obj → Prop
axiom MostActiveGrandSlam : obj → Prop

-- If someone is ranked highly by the WTA, then they are one of the most active players in major tennis.
axiom A1 : ∀ x : obj, RankedHighWTA x → MostActiveMajor x
-- Everyone who lost to Iga Świątek at Roland Garros 2022 is ranked highly by the WTA.
axiom A2 : ∀ x : obj, LostToIgaRG2022 x → RankedHighWTA x
-- All female tennis players at Roland Garros 2022 lost to Iga Świątek.
axiom A3 : ∀ x : obj, FemaleRG2022 x → LostToIgaRG2022 x
-- All tennis players at Roland Garros 2022 are either female or male.
axiom A4 : ∀ x : obj, PlayerRG2022 x → FemaleRG2022 x ∨ MaleRG2022 x
-- All male tennis players at Roland Garros in 2022 lost to Rafael Nadal.
axiom A5 : ∀ x : obj, MaleRG2022 x → LostToRafaRG2022 x
-- If Coco Gauff is ranked highly by the WTA or lost to Rafael Nadal, then Coco Gauff is not a male tennis player at Roland Garros 2022.
axiom A6 : (RankedHighWTA Coco ∨ LostToRafaRG2022 Coco) → ¬ MaleRG2022 Coco
-- Coco Gauff is at Roland Garros 2022.
axiom A7 : PlayerRG2022 Coco
-- Bridge: most active in major tennis implies most active in Grand Slam.
axiom A8 : ∀ x : obj, MostActiveMajor x → MostActiveGrandSlam x

theorem coco_active : MostActiveGrandSlam Coco := by
  have hdisj : FemaleRG2022 Coco ∨ MaleRG2022 Coco := A4 Coco A7
  cases hdisj with
  | inl hfem =>
      have hIga : LostToIgaRG2022 Coco := A3 Coco hfem
      have hrank : RankedHighWTA Coco := A2 Coco hIga
      have hmaj : MostActiveMajor Coco := A1 Coco hrank
      exact A8 Coco hmaj
  | inr hmale =>
      have hRafa : LostToRafaRG2022 Coco := A5 Coco hmale
      have hnotMale : ¬ MaleRG2022 Coco := A6 (Or.inr hRafa)
      exact False.elim (hnotMale hmale)
</lean>

ANSWER: True
