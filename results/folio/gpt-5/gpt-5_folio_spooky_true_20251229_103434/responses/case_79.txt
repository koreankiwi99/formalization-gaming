<lean>
axiom obj : Type

axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom FromBBM : obj → Prop

axiom DTC : obj  -- Dried Thai chilies

-- Premises
axiom H1 : ∀ x : obj, BakedSweet x → ¬ Spicy x
axiom H2 : ∀ x : obj, Cupcake x → BakedSweet x
axiom H3 : ∀ x : obj, MalaHotpot x → Spicy x
axiom H4 : ∀ x : obj, FromBBM x → Cupcake x
axiom H5 : Spicy DTC ∨ MalaHotpot DTC ∨ ¬ BakedSweet DTC

theorem dtc_not_bbm : ¬ FromBBM DTC :=
by
  intro hbbm
  have hcup : Cupcake DTC := H4 DTC hbbm
  have hbaked : BakedSweet DTC := H2 DTC hcup
  have hnotspicy : ¬ Spicy DTC := H1 DTC hbaked
  have hFalse : False :=
    Or.elim H5
      (fun hspicy => hnotspicy hspicy)
      (fun h =>
        Or.elim h
          (fun hm => hnotspicy (H3 DTC hm))
          (fun hnotbaked => hnotbaked hbaked)
      )
  exact hFalse
</lean>
ANSWER: True
