<lean>
axiom obj : Type
axiom James : obj

axiom Student : obj → Prop
axiom WorksLibrary : obj → Prop
axiom HasPartTimeJob : obj → Prop
axiom TakingDB : obj → Prop
axiom TakingWithDavid : obj → Prop
axiom WorksLab : obj → Prop
axiom FromCS : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, (Student x ∧ WorksLibrary x) → ¬ FromCS x
axiom P2 : ∀ x : obj, (Student x ∧ HasPartTimeJob x) → WorksLibrary x
axiom P3 : ∀ x : obj, (Student x ∧ TakingDB x) → FromCS x
axiom P4 : ∀ x : obj, (Student x ∧ TakingWithDavid x) → TakingDB x
axiom P5 : ∀ x : obj, (Student x ∧ WorksLab x) → TakingWithDavid x
axiom P6 : Student James ∧ WorksLab James
-- Ambiguous sentence formalized weakly; not used in the proof
axiom P7 : ¬ WorksLab James ∨ ¬ HasPartTimeJob James

theorem james_no_university_part_time_job : ¬ HasPartTimeJob James := by
  intro hJob
  have hStu : Student James := And.left P6
  have hLab : WorksLab James := And.right P6
  have hDavid : TakingWithDavid James := P5 James (And.intro hStu hLab)
  have hDB : TakingDB James := P4 James (And.intro hStu hDavid)
  have hCS : FromCS James := P3 James (And.intro hStu hDB)
  have hLib : WorksLibrary James := P2 James (And.intro hStu hJob)
  have hNotCS : ¬ FromCS James := P1 James (And.intro hStu hLib)
  exact hNotCS hCS
</lean>

ANSWER: True
