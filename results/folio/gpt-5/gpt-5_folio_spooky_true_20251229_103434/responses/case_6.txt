<lean>
axiom obj : Type

-- Predicates
axiom Animal : obj → Prop
axiom Mammal : obj → Prop
axiom Human : obj → Prop
axiom MonkeypoxVirusOccurs : obj → Prop
axiom CanMonkeypoxVirusOccur : obj → Prop
axiom GetsMonkeypox : obj → Prop
axiom Fever : obj → Prop
axiom Headache : obj → Prop
axiom MusclePains : obj → Prop
axiom Tired : obj → Prop
axiom Flu : obj → Prop

-- Axioms from the textual context
axiom R_occurs_gets : ∀ x : obj, MonkeypoxVirusOccurs x → GetsMonkeypox x
axiom R_animals_with_pox_potential : ∃ x : obj, Animal x ∧ CanMonkeypoxVirusOccur x
axiom R_humans_mammals : ∀ x : obj, Human x → Mammal x
axiom R_mammals_animals : ∀ x : obj, Mammal x → Animal x
axiom R_symptoms_of_monkeypox : ∀ x : obj, GetsMonkeypox x → Fever x ∧ Headache x ∧ MusclePains x ∧ Tired x
axiom R_flu_tired : ∀ x : obj, Flu x → Tired x

-- Goal: There is an animal.
theorem there_is_an_animal : ∃ x : obj, Animal x :=
by
  apply Exists.elim R_animals_with_pox_potential
  intro x hx
  exact Exists.intro x (And.left hx)
</lean>

ANSWER: True
