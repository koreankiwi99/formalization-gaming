<lean>
axiom Test : Type
axiom GRE : Test
axiom Cost : Test → Nat

def Below300 (t : Test) : Prop := Cost t < 300

axiom cost_GRE : Cost GRE = 205
axiom lt205_300 : 205 < 300

theorem gre_cost_below_300 : Below300 GRE := by
  unfold Below300
  have h : Cost GRE = 205 := cost_GRE
  simpa [h] using lt205_300
</lean>

ANSWER: True
