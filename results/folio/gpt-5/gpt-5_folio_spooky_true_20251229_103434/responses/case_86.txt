<lean>
axiom obj : Type

-- Entities
axiom Movie_S : obj
axiom SirDigby : obj
axiom Nemesis : obj

-- Predicates and relations
axiom SuperheroMovie : obj → Prop
axiom GoodGuy : obj → Prop
axiom BadGuy : obj → Prop
axiom Fights : obj → obj → Prop
axiom Wins : obj → Prop
axiom NamedAfter : obj → obj → Prop
axiom InMovie : obj → obj → Prop

-- Premises
axiom P1 : SuperheroMovie Movie_S
axiom P2 : ∀ m x, SuperheroMovie m → NamedAfter m x → GoodGuy x
axiom P3 : NamedAfter Movie_S SirDigby
axiom P4 : ∀ m x, SuperheroMovie m → InMovie m x → GoodGuy x → Wins x
axiom P5 : ∀ m x, SuperheroMovie m → NamedAfter m x → InMovie m x
axiom P6 : Fights SirDigby Nemesis
axiom P7 : ∀ x y, Fights x y → Wins x → ¬ Wins y
axiom P8 : ∀ x y, GoodGuy x → BadGuy y → Fights x y
axiom P9 : ∀ x y, BadGuy x → GoodGuy y → Fights x y

-- Conclusion: Sir Digby's nemesis does not win.
theorem nemesis_does_not_win : ¬ Wins Nemesis := by
  have hIn : InMovie Movie_S SirDigby := P5 Movie_S SirDigby P1 P3
  have hGood : GoodGuy SirDigby := P2 Movie_S SirDigby P1 P3
  have hWin : Wins SirDigby := P4 Movie_S SirDigby P1 hIn hGood
  have hFight : Fights SirDigby Nemesis := P6
  exact P7 SirDigby Nemesis hFight hWin
</lean>

ANSWER: True
