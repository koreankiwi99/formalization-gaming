axiom obj : Type
axiom Film : obj → Prop
axiom AppropriateForAllAges : obj → Prop

axiom RatedGeneralAudience : obj → Prop

axiom P1 : ∀ f : obj, AppropriateForAllAges f → ChildrenCanWatchWithoutGuidance f
axiom P2 : ∀ f : obj, ContainsExcessiveEroticAndViolentContent f → ¬ ChildrenCanWatchWithoutGuidance f
axiom P3 : ∀ f : obj, RatedGeneralAudience f → AppropriateForAllAges f
axiom P4 : ∀ f : obj, FamilyFriendlyAnimatedFilm f → RatedGeneralAudience f
axiom P5 : ∀ f : obj, InFrozenSeries f → FamilyFriendlyAnimatedFilm f

axiom P7 : FamilyFriendlyAnimatedFilm Hachi ∨ AppropriateForAllAges Hachi

theorem appropriate_all_ages_hachi : AppropriateForAllAges Hachi := by

    have hrga : RatedGeneralAudience Hachi := P4 Hachi hffa
    exact P3 Hachi hrga
  | inr hafa =>
    exact hafa

theorem can_watch_hachi : ChildrenCanWatchWithoutGuidance Hachi := by
  have hafa : AppropriateForAllAges Hachi := appropriate_all_ages_hachi
  exact P1 Hachi hafa

theorem not_cevc_hachi : ¬ ContainsExcessiveEroticAndViolentContent Hachi := by
  intro hcevc
  have hnotcan : ¬ ChildrenCanWatchWithoutGuidance Hachi := P2 Hachi hcevc
