  have hnotTW : ¬ TWCitizen Vladimir := And.left A5
  have hdisj : USCitizen Vladimir ∨ TWCitizen Vladimir := A3 Vladimir
  exact Or.elim hdisj (fun hUS => hUS) (fun hTW => False.elim (hnotTW hTW))

theorem Vladimir_CanRegister : CanRegisterUS Vladimir :=
  A2 Vladimir Vladimir_USCitizen
