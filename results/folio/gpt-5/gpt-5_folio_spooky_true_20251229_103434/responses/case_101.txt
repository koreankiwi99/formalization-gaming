axiom obj : Type
axiom Luke : obj

-- Predicates
axiom Baby : obj → Prop
axiom MultipleBirth : obj → Prop
axiom SpendTimeWithSiblings : obj → Prop
axiom SiblingsBornTogether : obj → Prop
axiom ComplainOftenAboutAnnoyingSiblings : obj → Prop
axiom LiveAtHome : obj → Prop
axiom LiveWithStrangers : obj → Prop

-- Axioms from the context
-- People born in a multiple birth (with siblings) spend a lot of time with their siblings.
axiom A1 : ∀ x : obj, MultipleBirth x → SpendTimeWithSiblings x
-- If people have siblings who were born together, then they were born in a multiple birth.
axiom A2 : ∀ x : obj, SiblingsBornTogether x → MultipleBirth x
-- If people complain often about annoying siblings, then they have siblings who were born together.
axiom A3 : ∀ x : obj, ComplainOftenAboutAnnoyingSiblings x → SiblingsBornTogether x
-- If people live at home, then they do not live with strangers.
axiom A4 : ∀ x : obj, LiveAtHome x → ¬ LiveWithStrangers x
-- If people spend a lot of time with their siblings, then they often live at home.
axiom A5 : ∀ x : obj, SpendTimeWithSiblings x → LiveAtHome x

-- Luke either is a baby born in a multiple birth and lives with strangers,
-- or is not a baby born in a multiple birth and does not live with strangers.
axiom L :
  (Baby Luke ∧ MultipleBirth Luke ∧ LiveWithStrangers Luke) ∨
  (¬(Baby Luke ∧ MultipleBirth Luke) ∧ ¬ LiveWithStrangers Luke)

-- Derived facts (do not yield the target conclusion)
theorem multiple_birth_implies_not_strangers : ∀ x : obj, MultipleBirth x → ¬ LiveWithStrangers x := by
  intro x hMB
  have hSpend : SpendTimeWithSiblings x := A1 x hMB
  have hHome : LiveAtHome x := A5 x hSpend
  have hNotStr : ¬ LiveWithStrangers x := A4 x hHome
  exact hNotStr

theorem not_first_disjunct : ¬ (Baby Luke ∧ MultipleBirth Luke ∧ LiveWithStrangers Luke) := by
  intro h
  cases h with
  | intro hBaby hrest =>
    cases hrest with
    | intro hMB hStr =>
      have hNotStr : ¬ LiveWithStrangers Luke := multiple_birth_implies_not_strangers Luke hMB
      exact hNotStr hStr

theorem deduce_second_disjunct : (¬(Baby Luke ∧ MultipleBirth Luke) ∧ ¬ LiveWithStrangers Luke) := by
  have h := L
  cases h with
  | inl h1 => exact False.elim (not_first_disjunct h1)
  | inr h2 => exact h2
