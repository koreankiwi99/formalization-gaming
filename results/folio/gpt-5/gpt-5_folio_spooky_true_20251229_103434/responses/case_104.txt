<lean>
axiom obj : Type

axiom Tom : obj

axiom GREApplicant : obj → Prop
axiom ApplyingToGRE : obj → Prop
axiom ProvesEconomicHardship : obj → Prop
axiom SingleParentFamily : obj → Prop
axiom FewResources : obj → Prop
axiom ProvidesFinancialAid : obj → Prop

axiom R_applicant : ∀ x : obj, ApplyingToGRE x → GREApplicant x
axiom R_aid : ∀ x : obj, GREApplicant x → ProvesEconomicHardship x → ProvidesFinancialAid x
axiom R_prove : ∀ x : obj, (SingleParentFamily x ∨ FewResources x) → ProvesEconomicHardship x

axiom T_single : SingleParentFamily Tom
axiom T_few : FewResources Tom
axiom T_apply : ApplyingToGRE Tom

theorem ets_financial_aid_Tom : ProvidesFinancialAid Tom :=
  let tom_app : GREApplicant Tom := R_applicant Tom T_apply
  let tom_hard : ProvesEconomicHardship Tom := R_prove Tom (Or.inl T_single)
  R_aid Tom tom_app tom_hard
</lean>

ANSWER: True
