axiom Peter : obj
axiom Windy : obj

axiom Man : obj → Prop
axiom Taller : obj → obj → Prop
axiom CanBlock : obj → obj → Prop
axiom InClass : obj → Prop
axiom JumpsWhenShooting : obj → Prop
axiom GreatShooter : obj → Prop

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom R1 : ∀ x y : obj, Man x → Man y → Taller x y → CanBlock x y

-- Michael is a man who is taller than everyone else in his class.
axiom Michael_man : Man Michael
axiom Michael_taller_than_class : ∀ x : obj, InClass x → Taller Michael x

-- Transitivity of taller-than
axiom taller_trans : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom Peter_man : Man Peter
axiom Peter_taller_Michael : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom Michael_blocks_if_no_jump : ∀ x : obj, ¬ JumpsWhenShooting x → CanBlock Michael x

-- Michael cannot block Windy's shooting.
axiom Michael_cannot_block_Windy : ¬ CanBlock Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom great_if_jump_shooter : ∀ x : obj, JumpsWhenShooting x → GreatShooter x

-- A useful derived fact: Windy must jump when shooting (by contraposition on Michael's blocking rule).
lemma Windy_jumps : JumpsWhenShooting Windy := by
  classical
  by_contra h
  have hb : CanBlock Michael Windy := Michael_blocks_if_no_jump Windy h
  exact Michael_cannot_block_Windy hb

-- Peter can block Michael (but this does not yield the desired conclusion about Windy).
lemma Peter_blocks_Michael : CanBlock Peter Michael :=
  R1 Peter Michael Peter_man Michael_man Peter_taller_Michael
