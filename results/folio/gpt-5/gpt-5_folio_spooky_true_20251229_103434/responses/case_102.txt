<lean>
axiom obj : Type
axiom Luke : obj

axiom MultipleBirth : obj → Prop
axiom SiblingsBornTogether : obj → Prop
axiom SpendTimeWithSiblings : obj → Prop
axiom LiveHome : obj → Prop
axiom LiveWithStrangers : obj → Prop
axiom ComplainsOften : obj → Prop

-- People who are born in a multiple birth with siblings spend time with siblings.
axiom A1 : ∀ x : obj, SiblingsBornTogether x → SpendTimeWithSiblings x
-- If people have siblings who were born together, then they were born in a multiple birth.
axiom A2 : ∀ x : obj, SiblingsBornTogether x → MultipleBirth x
-- If people complain often about annoying siblings, then they have siblings who were born together.
axiom A3 : ∀ x : obj, ComplainsOften x → SiblingsBornTogether x
-- If people live at home, then they do not live with strangers.
axiom A4 : ∀ x : obj, LiveHome x → ¬ LiveWithStrangers x
-- If people spend time with siblings, then they often live at home.
axiom A5 : ∀ x : obj, SpendTimeWithSiblings x → LiveHome x
-- Luke either is born in a multiple birth and lives with strangers, 
-- or is not born in a multiple birth and does not live with strangers.
axiom A6 : (MultipleBirth Luke ∧ LiveWithStrangers Luke) ∨ (¬ MultipleBirth Luke ∧ ¬ LiveWithStrangers Luke)

-- From the premises we can derive that Luke does not complain often about annoying siblings.
theorem not_complains_often_Luke : ¬ ComplainsOften Luke := by
  intro hC
  have hS : SiblingsBornTogether Luke := A3 Luke hC
  have hT : SpendTimeWithSiblings Luke := A1 Luke hS
  have hH : LiveHome Luke := A5 Luke hT
  have hNotLW : ¬ LiveWithStrangers Luke := A4 Luke hH
  have hMB : MultipleBirth Luke := A2 Luke hS
  cases A6 with
  | inl h =>
    exact hNotLW h.right
  | inr h =>
    exact h.left hMB
</lean>
