<lean>
axiom obj : Type

axiom Symphony9 : obj
axiom Beethoven : obj
axiom ViennaMusicSociety : obj

axiom MusicPiece : obj → Prop
axiom Composer : obj → Prop
axiom Orchestra : obj → Prop
axiom Conductor : obj → Prop

axiom Write : obj → obj → Prop
axiom Premiered : obj → obj → Prop
axiom Leads : obj → obj → Prop

-- Given facts
axiom T1 : MusicPiece Symphony9
axiom R1 : ∀ x : obj, Composer x → ∀ y : obj, MusicPiece y → Write x y
axiom T2 : Write Beethoven Symphony9
axiom T3 : Premiered ViennaMusicSociety Symphony9
axiom T4 : Orchestra ViennaMusicSociety
axiom T5 : Leads Beethoven ViennaMusicSociety
axiom R2 : ∀ o : obj, Orchestra o → ∃ c : obj, Conductor c ∧ Leads c o

-- Goal: Some orchestras premiered music pieces.
theorem some_orchestra_premiered_music_piece :
  ∃ o : obj, Orchestra o ∧ ∃ m : obj, MusicPiece m ∧ Premiered o m :=
by
  refine Exists.intro ViennaMusicSociety ?_
  refine And.intro T4 ?_
  refine Exists.intro Symphony9 ?_
  exact And.intro T1 T3
</lean>

ANSWER: True
