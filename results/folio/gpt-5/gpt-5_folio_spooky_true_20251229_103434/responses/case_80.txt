<lean>
axiom obj : Type

-- Predicates
axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom FromBakedByMelissa : obj → Prop

-- Constant for the entity in question
axiom DriedThaiChilies : obj

-- Premises
-- 1. No baked sweets are spicy.
axiom H1 : ∀ x : obj, BakedSweet x → ¬ Spicy x
-- 2. All cupcakes are baked sweets.
axiom H2 : ∀ x : obj, Cupcake x → BakedSweet x
-- 3. All mala hotpots are spicy.
axiom H3 : ∀ x : obj, MalaHotpot x → Spicy x
-- 4. All products from Baked by Melissa are cupcakes.
axiom H4 : ∀ x : obj, FromBakedByMelissa x → Cupcake x
-- 5. Dried Thai chilies are spicy or mala hotpots or not baked sweets.
axiom H5 : Spicy DriedThaiChilies ∨ MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies

-- A derived fact (not sufficient to prove the goal)
theorem cupcakes_not_spicy : ∀ x : obj, Cupcake x → ¬ Spicy x :=
by
  intro x hx
  have hbs : BakedSweet x := H2 x hx
  exact H1 x hbs
