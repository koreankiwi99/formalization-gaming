axiom Event : Type
axiom Happy : Event → Prop
axiom Sad : Event → Prop

-- Interpret "either ... or ..." as exclusive or:
axiom either_happy_or_sad : ∀ e : Event, (Happy e ∧ ¬ Sad e) ∨ (Sad e ∧ ¬ Happy e)
axiom exists_happy : ∃ e : Event, Happy e

-- From the premises it follows that not all events are sad.
theorem not_all_events_sad : ¬ (∀ e : Event, Sad e) := by
  intro Hall
  cases exists_happy with
  | intro w Hw =>
    have Sw : Sad w := Hall w
    cases either_happy_or_sad w with
    | inl h =>
      exact h.right Sw
    | inr h =>
      exact h.right Hw
