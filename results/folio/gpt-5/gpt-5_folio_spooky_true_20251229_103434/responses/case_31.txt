<lean>
axiom obj : Type

axiom At : obj → Prop
axiom GrandSlam : obj → Prop
axiom OscarNominated : obj → Prop
axiom ProTennis : obj → Prop
axiom Celebrity : obj → Prop
axiom Athlete : obj → Prop
axiom Famous : obj → Prop
axiom WellPaid : obj → Prop
axiom TaxHaven : obj → Prop

axiom A1 : ∀ x : obj, At x → (GrandSlam x ∨ OscarNominated x)
axiom A2 : ∀ x : obj, At x ∧ GrandSlam x → ProTennis x
axiom A3 : ∀ x : obj, At x ∧ OscarNominated x → Celebrity x
axiom A4 : ∀ x : obj, At x ∧ ProTennis x → Athlete x
axiom A5 : ∀ x : obj, At x ∧ Celebrity x → WellPaid x
axiom A6 : ∀ x : obj, At x ∧ Athlete x → Famous x
axiom A7 : ∀ x : obj, At x ∧ WellPaid x → TaxHaven x

axiom Djokovic : obj
axiom A8 : At Djokovic
axiom A9 : (Famous Djokovic ∧ Athlete Djokovic) → WellPaid Djokovic

theorem djokovic_taxhaven : TaxHaven Djokovic := by
  have hdisj : GrandSlam Djokovic ∨ OscarNominated Djokovic := A1 Djokovic A8
  cases hdisj with
  | inl hgs =>
      have hpro : ProTennis Djokovic := A2 Djokovic (And.intro A8 hgs)
      have hath : Athlete Djokovic := A4 Djokovic (And.intro A8 hpro)
      have hfam : Famous Djokovic := A6 Djokovic (And.intro A8 hath)
      have hwp : WellPaid Djokovic := A9 (And.intro hfam hath)
      exact A7 Djokovic (And.intro A8 hwp)
  | inr hosc =>
      have hcel : Celebrity Djokovic := A3 Djokovic (And.intro A8 hosc)
      have hwp : WellPaid Djokovic := A5 Djokovic (And.intro A8 hcel)
      exact A7 Djokovic (And.intro A8 hwp)
</lean>

ANSWER: True
