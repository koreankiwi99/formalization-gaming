<lean>
axiom obj : Type

axiom Listed : obj → Prop
axiom ManyNegReviews : obj → Prop
axiom RatingGT4 : obj → Prop
axiom ProvidesTakeOut : obj → Prop
axiom PopularLocal : obj → Prop

axiom HamdenPlazaSubway : obj

-- Premises
axiom P1 : ∀ x : obj, Listed x → ¬ ManyNegReviews x
axiom P2 : ∀ x : obj, RatingGT4 x → Listed x
axiom P3 : ∃ x : obj, ¬ ProvidesTakeOut x ∧ ManyNegReviews x
axiom P4 : ∀ x : obj, PopularLocal x → RatingGT4 x
axiom P5 : RatingGT4 HamdenPlazaSubway ∨ PopularLocal HamdenPlazaSubway

theorem target :
  (ProvidesTakeOut HamdenPlazaSubway ∧ ManyNegReviews HamdenPlazaSubway) →
  (RatingGT4 HamdenPlazaSubway ∧ ¬ ProvidesTakeOut HamdenPlazaSubway) := by
  -- derive that Hamden has rating > 4
  have hrating : RatingGT4 HamdenPlazaSubway := by
    cases P5 with
    | inl hrg => exact hrg
    | inr hpop => exact P4 HamdenPlazaSubway hpop
  -- from rating, listed; from listed, not many negative reviews
  have hlisted : Listed HamdenPlazaSubway := P2 HamdenPlazaSubway hrating
  have hnotneg : ¬ ManyNegReviews HamdenPlazaSubway := P1 HamdenPlazaSubway hlisted
  intro h
  have hmn : ManyNegReviews HamdenPlazaSubway := And.right h
  have contra : False := hnotneg hmn
  have not_takeout : ¬ ProvidesTakeOut HamdenPlazaSubway := by
    intro ht
    exact False.elim contra
  exact And.intro hrating not_takeout
</lean>

ANSWER: True
