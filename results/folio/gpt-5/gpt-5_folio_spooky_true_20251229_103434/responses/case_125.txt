<lean>
axiom obj : Type

axiom Rock : obj

axiom Pet : obj → Prop
axiom Mammal : obj → Prop
axiom Animal : obj → Prop
axiom Monkey : obj → Prop
axiom Bird : obj → Prop
axiom CanFly : obj → Prop
axiom Breathe : obj → Prop
axiom HasWings : obj → Prop

-- All of Peter's pets that are mammals are also animals.
axiom A1 : ∀ x : obj, Pet x → Mammal x → Animal x
-- All monkeys are mammals.
axiom A2 : ∀ x : obj, Monkey x → Mammal x
-- Peter's pets are all either monkeys or birds.
axiom A3 : ∀ x : obj, Pet x → Monkey x ∨ Bird x
-- Peter's birds can fly.
axiom A4 : ∀ x : obj, Pet x → Bird x → CanFly x
-- All animals that are Peter's pets can breathe.
axiom A5 : ∀ x : obj, Animal x → Pet x → Breathe x
-- If Peter's pet can fly, then it has wings.
axiom A6 : ∀ x : obj, Pet x → CanFly x → HasWings x
-- Rock is Peter's pet.
axiom A7 : Pet Rock
-- Rock can fly, or Rock is a bird, or Rock cannot breathe.
axiom A8 : (CanFly Rock ∨ Bird Rock) ∨ ¬Breathe Rock

theorem Rock_has_wings : HasWings Rock :=
by
  have pRock : Pet Rock := A7
  cases A8 with
  | inl h =>
      cases h with
      | inl fly =>
          exact A6 Rock pRock fly
      | inr bird =>
          have fly : CanFly Rock := A4 Rock pRock bird
          exact A6 Rock pRock fly
  | inr nbreathe =>
      -- Derive that Rock is a bird, using nbreathe to rule out Monkey
      have bird : Bird Rock :=
        by
          have hb : Monkey Rock ∨ Bird Rock := A3 Rock pRock
          cases hb with
          | inl hmonk =>
              have mammal : Mammal Rock := A2 Rock hmonk
              have animal : Animal Rock := A1 Rock pRock mammal
              have br : Breathe Rock := A5 Rock animal pRock
              exact False.elim (nbreathe br)
          | inr hbird =>
              exact hbird
      have fly : CanFly Rock := A4 Rock pRock bird
      exact A6 Rock pRock fly
</lean>

ANSWER: True
