<lean>
axiom obj : Type

axiom Person : obj → Prop
axiom Book : obj → Prop
axiom Reads : obj → obj → Prop
axiom ContainsKnowledge : obj → Prop
axiom GainsKnowledge : obj → Prop
axiom Smarter : obj → Prop

axiom Harry : obj
axiom Walden : obj

-- Books contain tons of knowledge
axiom T1 : ∀ x : obj, Book x → ContainsKnowledge x
-- When a person reads a book, that person gains knowledge
axiom T2 : ∀ p b : obj, Person p → Book b → Reads p b → GainsKnowledge p
-- If a person gains knowledge, they become smarter
axiom T3 : ∀ p : obj, GainsKnowledge p → Smarter p
-- Harry read the book “Walden”
axiom T4 : Book Walden
axiom T5 : Person Harry
axiom T6 : Reads Harry Walden

theorem walden_contains_knowledge : ContainsKnowledge Walden := by
  apply T1
  exact T4
</lean>

ANSWER: True
