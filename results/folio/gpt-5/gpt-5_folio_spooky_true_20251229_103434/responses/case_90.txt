<lean>
axiom obj : Type
axiom MachineTranslation : obj

axiom NLPTask : obj → Prop
axiom LangGen : obj → Prop
axiom LangUnd : obj → Prop
axiom OutputText : obj → Prop

-- Every NLP task can be categorized as either generation or understanding
axiom C1 : ∀ x : obj, NLPTask x → (LangGen x ∨ LangUnd x)

-- If the output of a natural language processing task is a text sequence, then it is a language generation task
axiom C2 : ∀ x : obj, NLPTask x → OutputText x → LangGen x

-- Machine translation is an NLP task
axiom F1 : NLPTask MachineTranslation

-- The output of machine translation is a text sequence
axiom F2 : OutputText MachineTranslation

theorem machine_translation_is_language_generation : LangGen MachineTranslation := by
  apply C2 MachineTranslation
  · exact F1
  · exact F2
</lean>

ANSWER: True
