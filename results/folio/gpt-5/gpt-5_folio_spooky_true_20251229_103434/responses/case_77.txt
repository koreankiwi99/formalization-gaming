<lean>
axiom obj : Type

axiom At : obj → Prop
axiom Work : obj → Prop
axiom Need : obj → Prop
axiom Order : obj → Prop
axiom Enjoy : obj → Prop
axiom Picky : obj → Prop
axiom Spend : obj → Prop

-- If people at Mary's school work in student jobs on campus, then they need to earn money.
axiom A1 : ∀ x : obj, (At x ∧ Work x) → Need x
-- If people at Mary's school order takeout frequently in college, then they work in student jobs on campus.
axiom A2 : ∀ x : obj, (At x ∧ Order x) → Work x
-- People at Mary's school order takeout frequently in college or enjoy the dining hall meals and recipes.
axiom A3 : ∀ x : obj, At x → (Order x ∨ Enjoy x)
-- If people at Mary's school enjoy the dining hall meals and recipes, then they are not picky eaters.
axiom A4 : ∀ x : obj, Enjoy x → ¬ Picky x
-- If people at Mary's school enjoy the dining hall meals and recipes, then they spend a lot of their time in the campus dining halls.
axiom A5 : ∀ x : obj, Enjoy x → Spend x

axiom Hannah : obj
axiom H_at : At Hannah
-- Hannah works in student jobs on campus and if she needs to earn money, then she is neither picky nor needs to earn money.
axiom H_work_and_cond : Work Hannah ∧ (Need Hannah → (¬ Picky Hannah ∧ ¬ Need Hannah))

theorem target :
  At Hannah ∧ (¬ Picky Hannah ∨ (Picky Hannah → Spend Hannah)) := by
  have hWork : Work Hannah := And.left H_work_and_cond
  have hCond : Need Hannah → (¬ Picky Hannah ∧ ¬ Need Hannah) := And.right H_work_and_cond
  have hNeed : Need Hannah := A1 Hannah (And.intro H_at hWork)
  have hNotPicky : ¬ Picky Hannah := And.left (hCond hNeed)
  exact And.intro H_at (Or.inl hNotPicky)
</lean>

ANSWER: True
