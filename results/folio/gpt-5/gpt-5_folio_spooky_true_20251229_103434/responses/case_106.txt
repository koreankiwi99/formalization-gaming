axiom ETS : obj

axiom FewResources : obj → Prop

-- Rules from the text
axiom R1 : ∀ x : obj, GREApplicant x → ProvesHardship x → ProvideAid ETS x
axiom R2 : ∀ x : obj, (SingleParent x ∨ FewResources x) → ProvesHardship x

axiom T_FR : FewResources Tom
axiom T_App : GREApplicant Tom

-- Derivations: Tom can prove hardship, so ETS provides aid to Tom (as an applicant)
theorem tom_proves_hardship : ProvesHardship Tom := by
  have h : SingleParent Tom ∨ FewResources Tom := Or.inl T_SP
  exact R2 Tom h

theorem ets_provides_aid_to_tom : ProvideAid ETS Tom := by
  exact R1 Tom T_App tom_proves_hardship

theorem exists_provider_for_tom : ∃ y : obj, ProvideAid y Tom := by
  exact Exists.intro ETS ets_provides_aid_to_tom
