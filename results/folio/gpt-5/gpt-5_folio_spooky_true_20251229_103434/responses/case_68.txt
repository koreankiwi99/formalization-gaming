<lean>
axiom obj : Type
axiom Badults : obj
axiom AndrewCollins : obj
axiom SecretDudeSociety : obj
axiom Series : obj → Prop
axiom WorkingTitle : obj → obj → Prop
axiom ScriptEditor : obj → obj → Prop

axiom H1 : Series Badults
axiom H2 : WorkingTitle Badults SecretDudeSociety
axiom H3 : ScriptEditor AndrewCollins Badults

theorem claim : ∃ s : obj, Series s ∧ WorkingTitle s SecretDudeSociety ∧ ScriptEditor AndrewCollins s :=
by
  apply Exists.intro Badults
  apply And.intro
  · exact H1
  · apply And.intro
    · exact H2
    · exact H3
</lean>

ANSWER: True
