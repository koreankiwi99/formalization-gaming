<lean>
axiom obj : Type

-- Entities
axiom Hannah : obj

-- Predicates
axiom At : obj → Prop
axiom WorkOnCampus : obj → Prop
axiom NeedsMoney : obj → Prop
axiom OrdersTakeout : obj → Prop
axiom EnjoysDiningHall : obj → Prop
axiom PickyEater : obj → Prop
axiom SpendsTimeDining : obj → Prop

-- Premises
axiom R1 : ∀ x : obj, At x → (WorkOnCampus x → NeedsMoney x)
axiom R2 : ∀ x : obj, At x → (OrdersTakeout x → WorkOnCampus x)
axiom R3 : ∀ x : obj, At x → (OrdersTakeout x ∨ EnjoysDiningHall x)
axiom R4 : ∀ x : obj, At x → (EnjoysDiningHall x → ¬ PickyEater x)
axiom R5 : ∀ x : obj, At x → (EnjoysDiningHall x → SpendsTimeDining x)
axiom T6 : At Hannah
axiom T7 : WorkOnCampus Hannah ∧ (NeedsMoney Hannah → (¬ PickyEater Hannah ∧ ¬ NeedsMoney Hannah))

-- Conclusion
theorem hannah_needs_money : NeedsMoney Hannah := by
  have hAt : At Hannah := T6
  have hWork : WorkOnCampus Hannah := And.left T7
  have hImp : At Hannah → (WorkOnCampus Hannah → NeedsMoney Hannah) := R1 Hannah
  exact hImp hAt hWork
</lean>

ANSWER: True
