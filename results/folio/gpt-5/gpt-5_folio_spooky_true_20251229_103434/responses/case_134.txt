<lean>
axiom obj : Type

axiom Space : obj
axiom Duster : obj

axiom Sucks : obj → Prop
axiom Vacuum : obj → Prop
axiom Plunger : obj → Prop
axiom Vampire : obj → Prop
axiom HouseholdAppliance : obj → Prop

-- Premises
axiom plungers_suck : ∀ x : obj, Plunger x → Sucks x
axiom vacuums_suck : ∀ x : obj, Vacuum x → Sucks x
axiom vampires_suck : ∀ x : obj, Vampire x → Sucks x
axiom space_is_vacuum : Vacuum Space
axiom duster_is_appliance : HouseholdAppliance Duster
axiom duster_doesnt_suck : ¬ Sucks Duster

-- Conclusion to prove: Space sucks.
theorem space_sucks : Sucks Space := by
  apply vacuums_suck
  exact space_is_vacuum
</lean>

ANSWER: True
