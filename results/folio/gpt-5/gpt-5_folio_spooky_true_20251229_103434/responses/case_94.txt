<lean>
axiom obj : Type

-- Entities and amounts
axiom OliveGarden : obj
axiom Tom : obj
axiom Fluffy : obj
axiom D2000 : obj
axiom D1500 : obj

-- Predicates and functions
axiom Managed : obj → Prop
axiom AllowsPets : obj → Prop
axiom Cat : obj → Prop
axiom Pet : obj → Prop
axiom Rent : obj → obj
axiom Deposit : obj → obj
axiom More : obj → obj → Prop

-- Contextual axioms
axiom some_managed_allows_pets : ∃ b : obj, Managed b ∧ AllowsPets b
axiom cats_are_pets : ∀ x : obj, Cat x → Pet x
axiom og_managed : Managed OliveGarden
axiom rent_og_2000 : Rent OliveGarden = D2000
axiom m2000_gt_1500 : More D2000 D1500

-- Key policy: In any managed building, the security deposit is equal to the monthly rent or more.
axiom deposit_ge_rent : ∀ b : obj, Managed b → (Deposit b = Rent b ∨ More (Deposit b) (Rent b))

-- Target statement: The security deposit at the Olive Garden is either $2000 or more.
theorem deposit_og_2000_or_more : Deposit OliveGarden = D2000 ∨ More (Deposit OliveGarden) D2000 := by
  have h := deposit_ge_rent OliveGarden og_managed
  have hr := rent_og_2000
  cases h with
  | inl hEq =>
      apply Or.inl
      simpa [hr] using hEq
  | inr hMore =>
      apply Or.inr
      simpa [hr] using hMore
</lean>

ANSWER: True
