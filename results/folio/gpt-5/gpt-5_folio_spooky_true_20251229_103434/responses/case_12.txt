<lean>
axiom obj : Type
axiom Zelda : obj
axiom JapaneseMade : obj → Prop
axiom OnTop10 : obj → Prop
axiom SellsMillion : obj → Prop

-- Premises
axiom P1 : JapaneseMade Zelda
axiom P2 : ∀ g : obj, OnTop10 g → JapaneseMade g
axiom P3 : ∀ g : obj, SellsMillion g → OnTop10 g
axiom P4 : SellsMillion Zelda

-- Conclusion: The Legend of Zelda is on the Top 10 list.
theorem zelda_on_top10 : OnTop10 Zelda := by
  exact (P3 Zelda) P4
</lean>

ANSWER: True
