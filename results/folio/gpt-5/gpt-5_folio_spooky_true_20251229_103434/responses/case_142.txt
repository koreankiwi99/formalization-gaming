<lean>
axiom obj : Type

axiom Animal : obj → Prop
axiom HeckCattle : obj → Prop
axiom Aurochs : obj → Prop
axiom BredBack : obj → Prop
axiom ArtificiallySelected : obj → Prop

-- Breeding back is a form of artificial selection
axiom R1 : ∀ x : obj, BredBack x → ArtificiallySelected x

-- There exist Heck cattle that were bred back
axiom T1 : ∃ x : obj, HeckCattle x ∧ BredBack x

theorem some_heck_cattle_artificially_selected : ∃ x : obj, HeckCattle x ∧ ArtificiallySelected x :=
by
  apply Exists.elim T1
  intro x hx
  apply Exists.intro x
  apply And.intro
  · exact And.left hx
  · exact R1 x (And.right hx)
</lean>

ANSWER: True
