-- In superhero movies, the good guys always win.
axiom P1 : ∀ m : Movie, SuperheroMovie m →
           ∀ p : Person, InMovie p m → Good p →
           ∀ q : Person, Fights p q → Wins p q

axiom P2 : SuperheroMovie SDMovie

-- Good guys fight bad guys and vice versa.
axiom P3a : ∀ p q : Person, Good p → Fights p q → Bad q
axiom P3b : ∀ p q : Person, Bad p → Fights p q → Good q

-- Sir Digby fights his nemesis.
axiom P4 : Fights Digby (Nemesis Digby)

-- If a superhero movie is named after a character, that character is a good guy.
axiom P5 : ∀ m : Movie, ∀ p : Person, SuperheroMovie m → NamedAfter m p → Good p

-- The Surprising Adventures of Sir Digby Chicken Caesar is named after Sir Digby.
axiom P6 : NamedAfter SDMovie Digby

-- If somebody wins a fight, the person they are fighting does not win.
axiom P7 : ∀ p q : Person, Wins p q → ¬ Wins q p

-- If a superhero movie is named after a character, that character is in the movie.
axiom P8 : ∀ m : Movie, ∀ p : Person, SuperheroMovie m → NamedAfter m p → InMovie p m

-- From the premises, we can prove the original statement:
theorem nemesis_does_not_win : ¬ Wins (Nemesis Digby) Digby := by
  have inMovieDigby : InMovie Digby SDMovie := P8 SDMovie Digby P2 P6
  have goodDigby : Good Digby := P5 SDMovie Digby P2 P6
  have digbyWins : Wins Digby (Nemesis Digby) := P1 SDMovie P2 Digby inMovieDigby goodDigby (Nemesis Digby) P4
