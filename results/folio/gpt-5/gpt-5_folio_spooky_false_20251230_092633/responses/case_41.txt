<lean>
axiom obj : Type

-- Predicates
axiom ByMax : obj → Prop
axiom Adores : obj → Prop
axiom Zaha : obj → Prop
axiom Kelly : obj → Prop
axiom Brutalist : obj → Prop
axiom Interesting : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, Zaha x ∧ Adores x → Interesting x
axiom P2 : ∀ x : obj, Brutalist x ∧ Adores x → ¬Interesting x
axiom P3 : ∀ x : obj, Adores x → (Zaha x ∨ Kelly x)
axiom P4 : ∀ x : obj, Kelly x ∧ Adores x → Evocative x
axiom P5 : ∀ x : obj, Kelly x ∧ Adores x → Dreamy x
axiom P6 : ∀ x : obj, ByMax x ∧ Adores x ∧ Interesting x → Brutalist x ∧ Evocative x

-- Some consequences we can prove from the premises

theorem no_bymax_adored_zaha : ∀ x : obj, ByMax x ∧ Adores x ∧ Zaha x → False :=
fun x h =>
  have hBy : ByMax x := h.left
  have hAdZa : Adores x ∧ Zaha x := h.right
  have hAd : Adores x := hAdZa.left
  have hZa : Zaha x := hAdZa.right
  have hInt : Interesting x := P1 x (And.intro hZa hAd)
  have hBrEv : Brutalist x ∧ Evocative x := P6 x (And.intro hBy (And.intro hAd hInt))
  have hBr : Brutalist x := hBrEv.left
  have hNotInt : ¬Interesting x := P2 x (And.intro hBr hAd)
  show False from hNotInt hInt

theorem bymax_adored_implies_kelly : ∀ x : obj, ByMax x ∧ Adores x → Kelly x :=
fun x h =>
  have hBy : ByMax x := h.left
  have hAd : Adores x := h.right
  have hZorK : Zaha x ∨ Kelly x := P3 x hAd
  Or.elim hZorK
    (fun hZ : Zaha x => False.elim (no_bymax_adored_zaha x (And.intro hBy (And.intro hAd hZ))))
    (fun hK : Kelly x => hK)

theorem bymax_adored_evocative_dreamy : ∀ x : obj, ByMax x ∧ Adores x → Evocative x ∧ Dreamy x :=
fun x h =>
  have hBy : ByMax x := h.left
  have hAd : Adores x := h.right
  have hK : Kelly x := bymax_adored_implies_kelly x (And.intro hBy hAd)
  have hEv : Evocative x := P4 x (And.intro hK hAd)
  have hDr : Dreamy x := P5 x (And.intro hK hAd)
  And.intro hEv hDr
