axiom obj : Type

-- Entities
axiom Picuris : obj
axiom Juan : obj
axiom Harding : obj

-- Predicates/relations
axiom MountainRange : obj → Prop
axiom InNM : obj → Prop
axiom InTX : obj → Prop
axiom Visited : obj → obj → Prop
axiom Mine : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop

-- Premises
axiom Picuris_range : MountainRange Picuris
axiom Picuris_in_NM_or_TX : InNM Picuris ∨ InTX Picuris
axiom Juan_visited_Picuris : Visited Juan Picuris
axiom Harding_mine : Mine Harding
axiom Harding_located_Picuris : LocatedIn Harding Picuris
axiom Harding_donated : Donated Harding
axiom no_TX_ranges_with_donated_mines :
  ∀ r : obj, MountainRange r → InTX r →
  ∀ m : obj, Mine m → LocatedIn m r → ¬Donated m

-- From the donated mine in Picuris, Picuris cannot be in Texas
theorem Picuris_not_TX : ¬InTX Picuris := by
  intro htx
  have hnd : ¬Donated Harding :=
    no_TX_ranges_with_donated_mines Picuris Picuris_range htx Harding Harding_mine Harding_located_Picuris
  exact hnd Harding_donated

-- Hence, by the disjunction, Picuris is in New Mexico
theorem Picuris_in_NM : InNM Picuris := by
  cases Picuris_in_NM_or_TX with
  | inl hnm => exact hnm
  | inr htx =>
    have : False := Picuris_not_TX htx
    exact False.elim this

-- Therefore, Juan visited a mountain range in New Mexico
theorem conclusion_true : ∃ r : obj, MountainRange r ∧ InNM r ∧ Visited Juan r := by
  apply Exists.intro Picuris
  apply And.intro
  · exact Picuris_range
  · apply And.intro
    · exact Picuris_in_NM
    · exact Juan_visited_Picuris
