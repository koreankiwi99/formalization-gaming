<lean>
axiom obj : Type

-- Predicates
axiom ZahaStyle : obj → Prop
axiom KellyStyle : obj → Prop
axiom Adores : obj → Prop
axiom Interesting : obj → Prop
axiom Brutalist : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop
axiom DesignByMax : obj → Prop

-- Premises
-- 1. All of Zaha Hadid's design styles that Max adores have interesting geometries.
axiom A1 : ∀ x : obj, ZahaStyle x ∧ Adores x → Interesting x

-- 2. No brutalist buildings that Max adores have interesting geometries.
axiom A2 : ∀ x : obj, Brutalist x ∧ Adores x → ¬ Interesting x

-- 3. Every style that Max adores is either Zaha Hadid's design style or Kelly Wearstler's design style.
axiom A3 : ∀ x : obj, Adores x → (ZahaStyle x ∨ KellyStyle x)

-- 4. All of Kelly Wearstler's design styles that Max adores are evocative.
axiom A4 : ∀ x : obj, KellyStyle x ∧ Adores x → Evocative x

-- 5. All of Kelly Wearstler's design styles that Max adores are dreamy.
axiom A5 : ∀ x : obj, KellyStyle x ∧ Adores x → Dreamy x

-- 6. If a design by Max that he adores has interesting geometries, then the design is a brutalist building and evocative.
axiom A6 : ∀ x : obj, DesignByMax x ∧ Adores x ∧ Interesting x → Brutalist x ∧ Evocative x

-- A derived contradiction: there is no x that is simultaneously a design by Max, adored by Max, and a Zaha style.
theorem no_design_by_max_adored_zaha : ∀ x : obj, ¬ (DesignByMax x ∧ Adores x ∧ ZahaStyle x) := by
  intro x
  intro h
  -- unpack h : DesignByMax x ∧ (Adores x ∧ ZahaStyle x)
  have h_design : DesignByMax x := And.left h
  have h_ad_zaha : Adores x ∧ ZahaStyle x := And.right h
  have h_ador : Adores x := And.left h_ad_zaha
  have h_zaha : ZahaStyle x := And.right h_ad_zaha
  -- ZahaStyle ∧ Adores ⇒ Interesting
  have h_int : Interesting x := A1 x (And.intro h_zaha h_ador)
  -- DesignByMax ∧ Adores ∧ Interesting ⇒ Brutalist ∧ Evocative
  have h_b_ev : Brutalist x ∧ Evocative x := A6 x (And.intro h_design (And.intro h_ador h_int))
  have h_brut : Brutalist x := And.left h_b_ev
  -- Brutalist ∧ Adores ⇒ ¬Interesting
  have h_not_int : ¬ Interesting x := A2 x (And.intro h_brut h_ador)
  exact h_not_int h_int
