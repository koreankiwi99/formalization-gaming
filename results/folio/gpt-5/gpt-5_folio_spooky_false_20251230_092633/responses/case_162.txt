<lean>
axiom obj : Type
axiom Bird : obj → Prop
axiom Hawk : obj → Prop
axiom Swims : obj → Prop

-- A hawk never swims.
axiom H1 : ∀ x : obj, Hawk x → ¬Swims x
-- Some birds are hawks.
axiom H2 : ∃ x : obj, Bird x ∧ Hawk x

-- Statement to evaluate: All birds swim.
-- We prove its negation from the premises.
theorem not_all_birds_swim : ¬(∀ x : obj, Bird x → Swims x) := by
  intro hAll
  cases H2 with
  | intro a hpair =>
    cases hpair with
    | intro hBa hHa =>
      have hSa : Swims a := hAll a hBa
      have hNotSa : ¬Swims a := H1 a hHa
      exact hNotSa hSa
</lean>

ANSWER: False
