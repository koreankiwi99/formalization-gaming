axiom Nice : obj → Prop
axiom Mean : obj → Prop
axiom Grumpy : obj → Prop
axiom Lover : obj → Prop
axiom Owner : obj → Prop

axiom Tom : obj

-- Premises
axiom P1 : ∀ x : obj, Nice x → ¬Mean x                          -- No one nice to animals is also mean to animals.
axiom P2 : ∃ x : obj, Grumpy x ∧ Mean x                         -- Some grumpy people are mean to animals.
axiom P3 : ∀ x : obj, Lover x → Nice x                          -- All animal lovers are nice to animals.
axiom P4 : ∀ x : obj, Owner x → Lover x                         -- All pet owners love animals.
axiom P5 : Owner Tom                                           -- Tom is a pet owner.

-- From the premises, we can show Tom is nice and thus not mean to animals.
theorem Tom_nice : Nice Tom := P3 Tom (P4 Tom P5)
theorem Tom_not_mean : ¬Mean Tom := P1 Tom Tom_nice

-- Therefore, Tom is not both grumpy and mean to animals (the given statement is true).
theorem tom_not_grumpy_and_mean : ¬ (Grumpy Tom ∧ Mean Tom) :=
  by
    intro h
    have hmean : Mean Tom := h.right
    exact Tom_not_mean hmean
