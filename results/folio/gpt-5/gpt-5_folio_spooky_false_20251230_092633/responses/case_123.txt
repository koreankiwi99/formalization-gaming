axiom TeslaModel3 : obj → Prop

-- All Tesla Model 3 can transport multiple passengers.
axiom H2 : ∀ x : obj, TeslaModel3 x → TransportMultiplePassengers x
-- All single-seat electric cars are one-seaters.
axiom H3 : ∀ x : obj, SingleSeatElectricCar x → OneSeater x
-- All Solo EV cars are single-seat electric cars.
axiom H4 : ∀ x : obj, SoloEV x → SingleSeatElectricCar x
-- If John's car is not a Tesla Model 3, then John's car is not a Tesla Model 3 or a single-seat electric car.
axiom H5 : ¬ TeslaModel3 JohnCar → (¬ TeslaModel3 JohnCar ∨ SingleSeatElectricCar JohnCar)

-- Derived: Tesla Model 3 are not one-seaters.

  intro x hx
  have hmp : TransportMultiplePassengers x := H2 x hx
  exact H1 x hmp

theorem SoloEV_one : ∀ x : obj, SoloEV x → OneSeater x := by

  have hse : SingleSeatElectricCar x := H4 x hs
  exact H3 x hse

-- Derived incompatibility: No Tesla Model 3 is a Solo EV.
