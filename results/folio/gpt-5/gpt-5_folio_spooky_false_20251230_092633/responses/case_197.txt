axiom obj : Type
axiom Jane : obj
axiom KiKi : obj

axiom Animal : obj → Prop
axiom Barks : obj → Prop
axiom Dog : obj → Prop
axiom Jumps : obj → Prop
axiom Legs : obj → Prop
axiom Terricolous : obj → Prop
axiom Likes : obj → obj → Prop

-- Jane does not like any animals that bark.
axiom A1 : ∀ x : obj, Animal x ∧ Barks x → ¬Likes Jane x
-- All dogs are animals that bark.
axiom A2 : ∀ x : obj, Dog x → Animal x ∧ Barks x
-- Jane likes all animals that jump.
axiom A3 : ∀ x : obj, Animal x ∧ Jumps x → Likes Jane x
-- If an animal has legs, then it jumps.
axiom A4 : ∀ x : obj, Animal x ∧ Legs x → Jumps x
-- If an animal is terricolous, then it has legs.
axiom A5 : ∀ x : obj, Animal x ∧ Terricolous x → Legs x
-- KiKi is an animal.
axiom A6 : Animal KiKi
-- If KiKi neither jumps nor has legs, then KiKi is terricolous or has legs.
axiom A7 : (¬Jumps KiKi ∧ ¬Legs KiKi) → (Terricolous KiKi ∨ Legs KiKi)

-- From legs to jumps for KiKi.
theorem legs_implies_jumps_KiKi : Legs KiKi → Jumps KiKi :=
by
  intro hL
  have h := A4 KiKi
  exact h (And.intro A6 hL)

-- From (¬J ∧ ¬L) derive L for KiKi, using terricolous→legs.
theorem conj_implies_legs_KiKi : (¬Jumps KiKi ∧ ¬Legs KiKi) → Legs KiKi :=
by
  intro h
  have hTL := A7 h
  cases hTL with
  | inl hT =>
      have h' := A5 KiKi
      exact h' (And.intro A6 hT)
  | inr hL =>
      exact hL

-- Therefore, not both (¬J ∧ ¬L) for KiKi.
theorem not_both_notJ_notL_KiKi : ¬(¬Jumps KiKi ∧ ¬Legs KiKi) :=
by
  intro h
  have hL := conj_implies_legs_KiKi h
  exact h.right hL

-- Hence ¬¬Jumps KiKi (constructively).
theorem notnot_jumps_KiKi : ¬¬Jumps KiKi :=
by
  intro hNotJ
  have hNotL : ¬Legs KiKi := fun hL => hNotJ (legs_implies_jumps_KiKi hL)
  have hConj : (¬Jumps KiKi ∧ ¬Legs KiKi) := And.intro hNotJ hNotL
  exact not_both_notJ_notL_KiKi hConj
