<lean>
axiom obj : Type

axiom Tom : obj
axiom Nice : obj → Prop
axiom Mean : obj → Prop
axiom Grumpy : obj → Prop
axiom AnimalLover : obj → Prop
axiom PetOwner : obj → Prop

-- No one nice to animals is also mean to animals.
axiom H1 : ∀ x : obj, Nice x → ¬Mean x
-- Some grumpy people are mean to animals.
axiom H2 : ∃ x : obj, Grumpy x ∧ Mean x
-- All animal lovers are nice to animals.
axiom H3 : ∀ x : obj, AnimalLover x → Nice x
-- All pet owners love animals (are animal lovers).
axiom H4 : ∀ x : obj, PetOwner x → AnimalLover x
-- Tom is a pet owner.
axiom H5 : PetOwner Tom

-- Negation of the statement:
-- If ¬(Grumpy Tom ∧ Mean Tom) then (¬Grumpy Tom ∧ ¬AnimalLover Tom).
-- We prove this implication is false.
theorem negate_statement :
  ¬ ((¬(Grumpy Tom ∧ Mean Tom)) → (¬Grumpy Tom ∧ ¬AnimalLover Tom)) := by
  intro hImp
  have hLover : AnimalLover Tom := H4 Tom H5
  have hNice : Nice Tom := H3 Tom hLover
  have hNotMean : ¬ Mean Tom := H1 Tom hNice
  have hAnte : ¬ (Grumpy Tom ∧ Mean Tom) := by
    intro hgm
    exact hNotMean hgm.right
  have hConseq : (¬Grumpy Tom ∧ ¬AnimalLover Tom) := hImp hAnte
  have hNotLover : ¬ AnimalLover Tom := hConseq.right
  exact hNotLover hLover
</lean>

ANSWER: False
