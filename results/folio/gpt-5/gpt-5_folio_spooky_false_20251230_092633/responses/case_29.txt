axiom FromEarth : obj → Prop

axiom A2 : ∀ x : obj, FromMars x → Alien x
axiom A3 : ∀ x : obj, ET x → ¬Human x
axiom A4 : ∀ x : obj, Intelligent x ∧ FromEarth x → Human x
axiom A5 : Intelligent Marvin
axiom A6 : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬FromEarth Marvin ∧ ¬FromMars Marvin)
axiom A7 : ¬FromEarth Marvin → ET Marvin

theorem negate_statement : ¬(¬FromMars Marvin → Human Marvin) := by

    intro h
    have hEarth : FromEarth Marvin := h.left
    have hMars : FromMars Marvin := h.right
    have hAlien : Alien Marvin := A2 Marvin hMars
    have hET : ET Marvin := A1 Marvin hAlien
    have hNotHuman : ¬Human Marvin := A3 Marvin hET
    have hHuman : Human Marvin := A4 Marvin (And.intro A5 hEarth)
    exact hNotHuman hHuman
  have hne : ¬FromEarth Marvin ∧ ¬FromMars Marvin := by
    cases A6 with
    | inl hBoth =>
        exact False.elim (h_not hBoth)
    | inr hNeither =>
        exact hNeither
  have hNotEarth : ¬FromEarth Marvin := hne.left
  have hNotMars : ¬FromMars Marvin := hne.right
  have hETM : ET Marvin := A7 hNotEarth
  have hNotHuman : ¬Human Marvin := A3 Marvin hETM
  intro hImp
  have hHuman : Human Marvin := hImp hNotMars
