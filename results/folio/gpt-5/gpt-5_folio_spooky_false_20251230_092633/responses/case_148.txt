axiom Ekaterina : obj
axiom Vladimir : obj

axiom CanRegisterUS : obj → Prop
axiom CanParticipate2024 : obj → Prop
axiom USCitizen : obj → Prop
axiom TaiwaneseCitizen : obj → Prop
axiom RussianOfficial : obj → Prop
axiom ManagerGazprom : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, CanRegisterUS x → CanParticipate2024 x
axiom P2 : ∀ x : obj, USCitizen x → CanRegisterUS x
axiom P3 : ∀ x : obj, USCitizen x ∨ TaiwaneseCitizen x
axiom P4 : ∀ x : obj, RussianOfficial x → ¬TaiwaneseCitizen x
axiom P5 : ¬TaiwaneseCitizen Vladimir ∧ ¬ManagerGazprom Vladimir
axiom P6 : CanRegisterUS Ekaterina ∨ RussianOfficial Ekaterina

-- We prove the negation of the statement, i.e., we prove:
-- CanRegisterUS Ekaterina ∧ CanParticipate2024 Vladimir
theorem neg_statement_provable :
  CanRegisterUS Ekaterina ∧ CanParticipate2024 Vladimir :=
by
  constructor
  · -- CanRegisterUS Ekaterina
    cases P6 with
    | inl hreg =>
      exact hreg
    | inr hro =>
      have hnotTW : ¬TaiwaneseCitizen Ekaterina := P4 Ekaterina hro
      have hdisj : USCitizen Ekaterina ∨ TaiwaneseCitizen Ekaterina := P3 Ekaterina
      cases hdisj with
      | inl hUS =>
        exact P2 Ekaterina hUS
      | inr hTW =>
        exact False.elim (hnotTW hTW)
  · -- CanParticipate2024 Vladimir
    have hnotTW : ¬TaiwaneseCitizen Vladimir := And.left P5
    have hdisj : USCitizen Vladimir ∨ TaiwaneseCitizen Vladimir := P3 Vladimir
    have hUS : USCitizen Vladimir :=
      by
        cases hdisj with
        | inl h =>
          exact h
        | inr hTW =>
          exact False.elim (hnotTW hTW)
    have hReg : CanRegisterUS Vladimir := P2 Vladimir hUS
    exact P1 Vladimir hReg
