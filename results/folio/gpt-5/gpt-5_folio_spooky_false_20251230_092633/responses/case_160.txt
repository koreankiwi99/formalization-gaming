<lean>
axiom obj : Type
axiom George : obj

axiom LikesMusic : obj → Prop
axiom WantsCompose : obj → Prop
axiom CanCompose : obj → Prop
axiom HasAccessProgram : obj → Prop
axiom WillCompose : obj → Prop

-- Premises from the textual context
axiom P1 : LikesMusic George → WantsCompose George
axiom P2 : HasAccessProgram George → CanCompose George
axiom P3 : CanCompose George ∧ WantsCompose George → WillCompose George

-- The statement to evaluate is derivable from the premises
theorem will_from_likes_and_access :
  LikesMusic George ∧ HasAccessProgram George → WillCompose George := by
  intro h
  have lm : LikesMusic George := And.left h
  have ap : HasAccessProgram George := And.right h
  have want : WantsCompose George := P1 lm
  have can : CanCompose George := P2 ap
  have cw : CanCompose George ∧ WantsCompose George := And.intro can want
  exact P3 cw

-- Consequently, the negation of the statement leads to a contradiction
theorem negation_impossible :
  (LikesMusic George ∧ HasAccessProgram George ∧ ¬WillCompose George) → False := by
  intro h
  have lm : LikesMusic George := And.left h
  have rest : HasAccessProgram George ∧ ¬WillCompose George := And.right h
  have ap : HasAccessProgram George := And.left rest
  have notW : ¬WillCompose George := And.right rest
  have W : WillCompose George := will_from_likes_and_access (And.intro lm ap)
  exact notW W
