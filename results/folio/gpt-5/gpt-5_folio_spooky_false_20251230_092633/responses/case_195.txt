axiom obj : Type

axiom KiKi : obj

axiom Animal : obj → Prop
axiom Dog : obj → Prop
axiom Bark : obj → Prop
axiom Jump : obj → Prop
axiom HasLegs : obj → Prop
axiom Terricolous : obj → Prop
axiom Likes : obj → obj → Prop

-- Jane does not like any animals that bark.
axiom A1 : ∀ x : obj, (Animal x ∧ Bark x) → ¬Likes Jane x
-- All dogs are animals that bark.
axiom A2 : ∀ x : obj, Dog x → (Animal x ∧ Bark x)
-- Jane likes all animals that jump.
axiom A3 : ∀ x : obj, (Animal x ∧ Jump x) → Likes Jane x
-- If an animal has legs, then it jumps.
axiom A4 : ∀ x : obj, (Animal x ∧ HasLegs x) → Jump x
-- If an animal is terricolous, then it has legs.
axiom A5 : ∀ x : obj, (Animal x ∧ Terricolous x) → HasLegs x
-- KiKi is an animal.
axiom A6 : Animal KiKi
-- If KiKi neither jumps nor has legs, then KiKi is terricolous or has legs.
axiom A7 : (¬Jump KiKi ∧ ¬HasLegs KiKi) → (Terricolous KiKi ∨ HasLegs KiKi)

-- Some derived consequences (do not yield ¬Terricolous KiKi).
theorem terri_implies_legs_KiKi : Terricolous KiKi → HasLegs KiKi := by
  intro hT
  exact A5 KiKi (And.intro A6 hT)

theorem legs_implies_jump_KiKi : HasLegs KiKi → Jump KiKi := by
  intro hL
  exact A4 KiKi (And.intro A6 hL)

theorem terri_implies_jump_KiKi : Terricolous KiKi → Jump KiKi := by
  intro hT
  have hL : HasLegs KiKi := A5 KiKi (And.intro A6 hT)
  exact A4 KiKi (And.intro A6 hL)

theorem terri_implies_liked_KiKi : Terricolous KiKi → Likes Jane KiKi := by
  intro hT
  have hJ : Jump KiKi := A4 KiKi (And.intro A6 (A5 KiKi (And.intro A6 hT)))
  exact A3 KiKi (And.intro A6 hJ)

-- We can show that KiKi cannot simultaneously fail to jump and fail to have legs.
theorem not_not_jump_and_not_legs : ¬(¬Jump KiKi ∧ ¬HasLegs KiKi) := by
  intro h
  have disj : Terricolous KiKi ∨ HasLegs KiKi := A7 h
  cases disj with
  | inl hT =>
    have hL : HasLegs KiKi := A5 KiKi (And.intro A6 hT)
    exact h.right hL
  | inr hL =>
