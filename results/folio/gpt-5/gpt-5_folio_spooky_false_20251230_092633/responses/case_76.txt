axiom Work : obj → Prop
axiom Need : obj → Prop

axiom Enjoy : obj → Prop
axiom Picky : obj → Prop

axiom R1 : ∀ x : obj, At x → Work x → Need x
axiom R2 : ∀ x : obj, At x → Order x → Work x
axiom R3 : ∀ x : obj, At x → (Order x ∨ Enjoy x)

axiom R5 : ∀ x : obj, At x → Enjoy x → Spend x
axiom T_at : At Hannah
axiom T7_work : Work Hannah
axiom T7_imp : Need Hannah → (¬Picky Hannah ∧ ¬Need Hannah)

-- Try to prove the negation of the statement:
-- ¬(At Hannah ∧ ¬Picky Hannah ∧ Spend Hannah)
theorem neg_statement : ¬(At Hannah ∧ ¬Picky Hannah ∧ Spend Hannah) := by
  intro h
  have hNeed : Need Hannah := R1 Hannah T_at T7_work
  have hNotNeed : ¬Need Hannah := (T7_imp hNeed).right
  exact hNotNeed hNeed
