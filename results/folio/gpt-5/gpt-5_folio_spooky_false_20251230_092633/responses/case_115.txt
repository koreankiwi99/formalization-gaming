axiom obj : Type

axiom Man : obj → Prop
axiom Taller : obj → obj → Prop
axiom Block : obj → obj → Prop
axiom Jump : obj → Prop
axiom GreatShooter : obj → Prop

axiom Michael : obj
axiom Peter : obj
axiom Windy : obj

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom P1 : ∀ x y : obj, Man x → Man y → Taller x y → Block x y

-- Michael is a man.
axiom P2 : Man Michael

-- Transitivity of taller-than
axiom P3 : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom P4a : Man Peter
axiom P4b : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom P5 : ∀ y : obj, ¬Jump y → Block Michael y

-- Michael cannot block Windy's shooting.
axiom P6 : ¬Block Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom P7 : ∀ x : obj, Jump x → GreatShooter x

-- From the premises, we can in fact derive that Windy is a great shooter.
theorem windy_is_great : GreatShooter Windy := by
  have hnn : ¬¬Jump Windy := fun hnj => P6 (P5 Windy hnj)
  classical
  by_cases hJ : Jump Windy
  · exact P7 Windy hJ
  · exact False.elim (hnn hJ)
