<lean>
axiom obj : Type

-- Entities
axiom Picuris : obj
axiom Juan : obj
axiom HardingMine : obj

-- Predicates
axiom MountainRange : obj → Prop
axiom InTexas : obj → Prop
axiom InNewMexico : obj → Prop
axiom Mine : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop
axiom Visited : obj → obj → Prop

-- Premises
axiom MR_Picuris : MountainRange Picuris
axiom OrNMorTX_Picuris : InNewMexico Picuris ∨ InTexas Picuris
axiom Visited_Juan_Picuris : Visited Juan Picuris
axiom Mine_Harding : Mine HardingMine
axiom LocatedIn_Harding_Picuris : LocatedIn HardingMine Picuris
axiom Donated_Harding : Donated HardingMine
axiom NoTexasDonated :
  ∀ x : obj, MountainRange x → InTexas x →
  ∀ y : obj, Mine y → LocatedIn y x → ¬Donated y

-- From the donated mine located in Picuris and the "no donated-mine Texas ranges" rule,
-- Picuris cannot be in Texas.
theorem Picuris_not_in_Texas : ¬InTexas Picuris := by
  intro hTX
  have h := NoTexasDonated Picuris MR_Picuris hTX HardingMine Mine_Harding LocatedIn_Harding_Picuris
  exact h Donated_Harding

-- Using the disjunction "in New Mexico or Texas" and the fact it's not in Texas,
-- we can conclude it is in New Mexico.
theorem Picuris_in_NewMexico : InNewMexico Picuris := by
  cases OrNMorTX_Picuris with
  | inl hNM => exact hNM
  | inr hTX =>
      exact False.elim (Picuris_not_in_Texas hTX)
