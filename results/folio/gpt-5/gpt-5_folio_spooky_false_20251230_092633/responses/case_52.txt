<lean>
axiom Lang : Type

axiom French : Lang
axiom Spanish : Lang
axiom German : Lang
axiom Basque : Lang

axiom Related : Lang → Lang → Prop
axiom Romance : Lang → Prop
axiom IndoEuropean : Lang → Prop

-- "Romance languages are a language family."
axiom LanguageFamily : (Lang → Prop) → Prop
axiom Romance_is_family : LanguageFamily Romance

-- "All languages within a language family are related to each other."
axiom Family_related : ∀ F : Lang → Prop, LanguageFamily F → ∀ x y : Lang, F x → F y → Related x y

-- "All Romance languages are Indo-European languages."
axiom Romance_implies_IE : ∀ x : Lang, Romance x → IndoEuropean x

-- "French and Spanish are both Romance languages."
axiom French_Romance : Romance French
axiom Spanish_Romance : Romance Spanish

-- "German is related to Spanish."
axiom German_related_Spanish : Related German Spanish

-- "Basque is not related to any other language."
axiom Basque_is_isolate : ∀ x : Lang, x ≠ Basque → (¬ Related Basque x ∧ ¬ Related x Basque)

-- Some derivable consequences (do not yield a contradiction with "German is a Romance language").
theorem Spanish_related_French : Related Spanish French := by
  exact Family_related Romance Romance_is_family Spanish French Spanish_Romance French_Romance

theorem German_related_French_if_German_Romance : Romance German → Related German French := by
  intro h
  exact Family_related Romance Romance_is_family German French h French_Romance
