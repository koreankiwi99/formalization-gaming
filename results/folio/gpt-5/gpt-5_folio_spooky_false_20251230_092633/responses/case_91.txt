<lean>
axiom Task : Type
axiom MachineTranslation : Task
axiom NLP : Task → Prop
axiom Gen : Task → Prop
axiom Und : Task → Prop
axiom OutputText : Task → Prop

-- A natural language processing task is categorized as either generation or understanding, but not both.
axiom Categ : ∀ x : Task, NLP x → (Gen x ∨ Und x) ∧ ¬(Gen x ∧ Und x)
-- If the output of an NLP task is a text sequence, then it is a generation task.
axiom GenIfText : ∀ x : Task, NLP x → OutputText x → Gen x
-- Machine translation is an NLP task.
axiom MT_is_NLP : NLP MachineTranslation
-- The output of machine translation is a text sequence.
axiom MT_output_text : OutputText MachineTranslation

theorem mt_not_understanding : ¬Und MachineTranslation := by
  intro hUnd
  have hNLP : NLP MachineTranslation := MT_is_NLP
  have hGen : Gen MachineTranslation := GenIfText MachineTranslation hNLP MT_output_text
  have hNoBoth : ¬(Gen MachineTranslation ∧ Und MachineTranslation) :=
    (And.right (Categ MachineTranslation hNLP))
  exact hNoBoth ⟨hGen, hUnd⟩
</lean>

ANSWER: False
