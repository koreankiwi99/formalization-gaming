axiom obj : Type
axiom James : obj

axiom WorksInLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom UniPTJob : obj → Prop
axiom TakingDB : obj → Prop
axiom ClassWithDavid : obj → Prop
axiom WorksInLab : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, WorksInLibrary x → ¬FromCS x
axiom A2 : ∀ x : obj, UniPTJob x → WorksInLibrary x
axiom A3 : ∀ x : obj, TakingDB x → FromCS x
axiom A4 : ∀ x : obj, ClassWithDavid x → TakingDB x
axiom A5 : ∀ x : obj, WorksInLab x → ClassWithDavid x
axiom A6 : WorksInLab James
axiom A7 : ¬WorksInLab James ∨ ¬UniPTJob James

-- Derived: James is from the CS department
theorem james_from_cs : FromCS James := by
  have hCD : ClassWithDavid James := A5 James A6
  have hDB : TakingDB James := A4 James hCD
  exact A3 James hDB

-- Therefore, James does not have a university part-time job
theorem james_not_uni_job : ¬UniPTJob James := by
  intro hJob
  have hLib : WorksInLibrary James := A2 James hJob
  have hNotCS : ¬FromCS James := A1 James hLib
  exact hNotCS james_from_cs
