axiom obj : Type

axiom DesignByMax : obj → Prop
axiom Adored : obj → Prop
axiom Zaha : obj → Prop
axiom Kelly : obj → Prop
axiom Interesting : obj → Prop
axiom Brutalist : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop

-- All of Zaha Hadid's design styles that Max adores have interesting geometries.
axiom A1 : ∀ x : obj, Zaha x ∧ Adored x → Interesting x
-- No brutalist buildings that Max adores have interesting geometries.
axiom A2 : ∀ x : obj, Brutalist x ∧ Adored x → ¬ Interesting x
-- Every style that Max adores is either Zaha Hadid's design style or Kelly Wearstler's design style.
axiom A3 : ∀ x : obj, Adored x → (Zaha x ∨ Kelly x)
-- All of Kelly Wearstler's design styles that Max adores are evocative.
axiom A4 : ∀ x : obj, Kelly x ∧ Adored x → Evocative x
-- All of Kelly Wearstler's design styles that Max adores are dreamy.
axiom A5 : ∀ x : obj, Kelly x ∧ Adored x → Dreamy x
-- If a design by Max that he adores has interesting geometries, then it is a brutalist building and evocative.
axiom A6 : ∀ x : obj, DesignByMax x ∧ Adored x ∧ Interesting x → Brutalist x ∧ Evocative x

-- Useful consequences of the premises.

-- No design by Max that he adores can be a Zaha design (since that would force Interesting and ¬Interesting).
theorem no_zaha_design_by_max_adored : ∀ x : obj, DesignByMax x ∧ Adored x ∧ Zaha x → False := by
  intro x
  intro h
  have dmx : DesignByMax x := And.left h
  have had : Adored x := And.left (And.right h)
  have hz  : Zaha x := And.right (And.right h)
  have hi  : Interesting x := A1 x (And.intro hz had)
  have hbe : Brutalist x ∧ Evocative x := A6 x (And.intro dmx (And.intro had hi))
  have hb  : Brutalist x := And.left hbe
  have hni : ¬ Interesting x := A2 x (And.intro hb had)
  exact hni hi

-- Any design by Max that he adores must be a Kelly design.
theorem adored_design_by_max_is_kelly : ∀ x : obj, DesignByMax x ∧ Adored x → Kelly x := by
  intro x
  intro h
  have had : Adored x := And.right h
  have disj : Zaha x ∨ Kelly x := A3 x had
  cases disj with
  | inl hz =>
      have contra : False := no_zaha_design_by_max_adored x (And.intro (And.left h) (And.intro had hz))
      exact False.elim contra
  | inr hk =>
      exact hk

-- Any design by Max that he adores is both evocative and dreamy.
theorem adored_design_by_max_evocative_dreamy : ∀ x : obj, DesignByMax x ∧ Adored x → Evocative x ∧ Dreamy x := by
  intro x
  intro h
  have had : Adored x := And.right h
  have hk  : Kelly x := adored_design_by_max_is_kelly x h
  have hev : Evocative x := A4 x (And.intro hk had)
  have hdr : Dreamy x := A5 x (And.intro hk had)
  exact And.intro hev hdr

-- No design by Max that he adores can have interesting geometries.
theorem no_adored_interesting_design_by_max : ∀ x : obj, DesignByMax x ∧ Adored x ∧ Interesting x → False := by
  intro x
  intro h
  have dmx : DesignByMax x := And.left h
  have had : Adored x := And.left (And.right h)
  have hi  : Interesting x := And.right (And.right h)
  have hbe : Brutalist x ∧ Evocative x := A6 x (And.intro dmx (And.intro had hi))
  have hb  : Brutalist x := And.left hbe
  have hni : ¬ Interesting x := A2 x (And.intro hb had)
  exact hni hi
