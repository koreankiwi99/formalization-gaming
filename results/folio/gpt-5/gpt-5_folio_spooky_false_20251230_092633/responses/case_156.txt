<lean>
axiom obj : Type
axiom James : obj

axiom WorkInLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom UnivPT : obj → Prop
axiom TakingDB : obj → Prop
axiom TakingWithDavid : obj → Prop
axiom WorkInLab : obj → Prop

-- None who work in the library are from CS
axiom A1 : ∀ x : obj, WorkInLibrary x → ¬ FromCS x
-- All with university-offered part-time jobs work in the library
axiom A2 : ∀ x : obj, UnivPT x → WorkInLibrary x
-- All taking DB are from CS
axiom A3 : ∀ x : obj, TakingDB x → FromCS x
-- All taking a class with Prof. David are taking DB
axiom A4 : ∀ x : obj, TakingWithDavid x → TakingDB x
-- All working in the lab are taking a class with Prof. David
axiom A5 : ∀ x : obj, WorkInLab x → TakingWithDavid x
-- James works in the lab
axiom A6 : WorkInLab James
-- James doesn't work in the lab or have a university-offered part-time job
axiom A7 : ¬WorkInLab James ∨ ¬UnivPT James

theorem james_no_univpt : ¬UnivPT James := by
  intro hpt
  have hlib : WorkInLibrary James := A2 James hpt
  have hncs : ¬FromCS James := A1 James hlib
  have hwd : TakingWithDavid James := A5 James A6
  have hdb : TakingDB James := A4 James hwd
  have hcs : FromCS James := A3 James hdb
  exact hncs hcs
</lean>

ANSWER: False
