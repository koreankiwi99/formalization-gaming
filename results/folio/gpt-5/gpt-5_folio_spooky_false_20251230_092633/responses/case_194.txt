axiom Descampe : Person

axiom OnLeaderboard1992 : Person → Prop
axiom FromBelgium : Person → Prop
axiom InSixWayTie1992 : Person → Prop

-- Premises
axiom P1 : Winner1992 Steinhauer
axiom P2 : Participated1992 Steinhauer
-- "One six-way tie was on the leaderboard" modeled as: anyone in the six-way tie is on the leaderboard
axiom P3a : ∀ x : Person, InSixWayTie1992 x → OnLeaderboard1992 x
-- "one person in the six-way tie was from Belgium"
axiom P3b : ∃ x : Person, InSixWayTie1992 x ∧ FromBelgium x
-- Descampe facts
axiom P4 : FromBelgium Descampe
axiom P5 : OnLeaderboard1992 Descampe
-- All people on the leaderboard participated
axiom P6 : ∀ x : Person, OnLeaderboard1992 x → Participated1992 x

-- A derived fact we can prove: Descampe participated.
theorem descampe_participated : Participated1992 Descampe := P6 Descampe P5

-- From P3b and P3a, we can derive that there exists a Belgian on the leaderboard,
-- but this does not entail that Descampe is (or is not) in the six-way tie.
theorem exists_belgian_on_leaderboard : ∃ x : Person, OnLeaderboard1992 x ∧ FromBelgium x := by
  rcases P3b with ⟨x, hxTie, hxBel⟩
  exact ⟨x, And.intro (P3a x hxTie) hxBel⟩
