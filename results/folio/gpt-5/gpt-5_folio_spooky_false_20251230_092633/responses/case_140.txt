<lean>
axiom obj : Type

axiom Film : obj → Prop
axiom AppropriateAllAges : obj → Prop
axiom CanWatchWithoutGuidance : obj → Prop
axiom ExcessiveEroticViolent : obj → Prop
axiom RatedGeneralAudience : obj → Prop
axiom FamilyFriendlyAnimated : obj → Prop
axiom FrozenSeries : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, Film x → AppropriateAllAges x → CanWatchWithoutGuidance x
axiom P2 : ∀ x : obj, Film x → ExcessiveEroticViolent x → ¬CanWatchWithoutGuidance x
axiom P3 : ∀ x : obj, RatedGeneralAudience x → AppropriateAllAges x
axiom P4 : ∀ x : obj, FamilyFriendlyAnimated x → RatedGeneralAudience x
axiom P5 : ∀ x : obj, FrozenSeries x → FamilyFriendlyAnimated x

-- Specific object: Hachi: A dog's Tale
axiom Hachi : obj
axiom Hachi_is_film : Film Hachi
axiom Hachi_ffa_or_app : FamilyFriendlyAnimated Hachi ∨ AppropriateAllAges Hachi

-- Derived: Hachi can be watched without parental guidance
theorem Hachi_can_watch : CanWatchWithoutGuidance Hachi := by
  cases Hachi_ffa_or_app with
  | inl hffa =>
      have hRG : RatedGeneralAudience Hachi := P4 Hachi hffa
      have happ : AppropriateAllAges Hachi := P3 Hachi hRG
      exact P1 Hachi Hachi_is_film happ
  | inr happ =>
      exact P1 Hachi Hachi_is_film happ

-- Therefore Hachi does not contain excessive erotic and violent content
theorem not_Hachi_EEVC : ¬ExcessiveEroticViolent Hachi := by
  intro hee
  have hnotCW : ¬CanWatchWithoutGuidance Hachi := P2 Hachi Hachi_is_film hee
  exact hnotCW Hachi_can_watch
