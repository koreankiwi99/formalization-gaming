<lean>
axiom obj : Type
axiom Happy : obj → Prop
axiom Sad : obj → Prop
-- Events are either happy or sad, exclusively
axiom either_exclusive : ∀ x : obj, (Happy x ∨ Sad x) ∧ ¬(Happy x ∧ Sad x)
axiom exists_happy : ∃ x : obj, Happy x

theorem not_all_events_sad : ¬(∀ x : obj, Sad x) := by
  intro allSad
  have ⟨w, Hw⟩ := exists_happy
  have Sw : Sad w := allSad w
  have not_both : ¬(Happy w ∧ Sad w) := (either_exclusive w).right
  exact not_both ⟨Hw, Sw⟩
</lean>

ANSWER: False
