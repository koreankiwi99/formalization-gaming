<lean>
axiom obj : Type

axiom AtMS : obj → Prop
axiom SJC : obj → Prop
axiom NeedMoney : obj → Prop
axiom OrderTakeout : obj → Prop
axiom EnjoyDining : obj → Prop
axiom Picky : obj → Prop
axiom SpendTime : obj → Prop

axiom R1 : ∀ x : obj, AtMS x ∧ SJC x → NeedMoney x
axiom R2 : ∀ x : obj, AtMS x ∧ OrderTakeout x → SJC x
axiom R3 : ∀ x : obj, AtMS x → (OrderTakeout x ∨ EnjoyDining x)
axiom R4 : ∀ x : obj, AtMS x ∧ EnjoyDining x → ¬Picky x
axiom R5 : ∀ x : obj, AtMS x ∧ EnjoyDining x → SpendTime x

axiom Hannah : obj
axiom T6 : AtMS Hannah
axiom T7a : SJC Hannah
axiom T7b : NeedMoney Hannah → (¬Picky Hannah ∧ ¬NeedMoney Hannah)

theorem contradiction_from_premises : False := by
  have hNeed : NeedMoney Hannah := by
    have h1 : AtMS Hannah ∧ SJC Hannah := And.intro T6 T7a
    exact R1 Hannah h1
  have hNoNeed : ¬NeedMoney Hannah := by
    have h := T7b hNeed
    exact h.right
  exact hNoNeed hNeed

def Statement : Prop :=
  AtMS Hannah ∧ (¬Picky Hannah ∨ (Picky Hannah → SpendTime Hannah))

theorem neg_Statement : ¬Statement := by
  intro h
  exact contradiction_from_premises
</lean>

ANSWER: False
