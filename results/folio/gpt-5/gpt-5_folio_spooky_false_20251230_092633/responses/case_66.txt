<lean>
axiom obj : Type
axiom Yale : obj

-- Predicates
axiom EducationalInstitution : obj → Prop
axiom LargerEndowment : obj → obj → Prop

-- Facts from the context
axiom Yale_is_educational : EducationalInstitution Yale

-- Basic property about a strict "larger endowment than" relation: irreflexivity
axiom LargerEndowment_irrefl : ∀ x : obj, ¬LargerEndowment x x

-- Formalizing: "has the largest university endowment of any educational institution"
def LargestOfAnyEducationalInstitution (y : obj) : Prop :=
  ∀ x : obj, EducationalInstitution x → LargerEndowment y x

-- Prove the negation: such a claim would force Yale's endowment to be larger than itself.
theorem yale_not_largest : ¬LargestOfAnyEducationalInstitution Yale := by
  intro h
  have hself : LargerEndowment Yale Yale := h Yale Yale_is_educational
  exact (LargerEndowment_irrefl Yale) hself
</lean>
