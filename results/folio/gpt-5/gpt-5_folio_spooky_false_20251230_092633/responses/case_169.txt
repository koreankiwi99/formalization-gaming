axiom LowSurvival : obj → Prop
axiom Severe : obj → Prop
axiom BileDuct : obj → Prop
axiom Cholangiocarcinoma : obj → Prop

axiom Colorectal : obj

-- If something is a deadly disease, then it comes with a low survival rate.
axiom A1 : ∀ x : obj, Deadly x → LowSurvival x
-- Severe cancers are deadly diseases.
axiom A2 : ∀ x : obj, Severe x → Deadly x
-- Bile duct cancer is a severe form cancer.
axiom A3 : ∀ x : obj, BileDuct x → Severe x
-- All Cholangiocarcinoma is bile duct cancer.
axiom A4 : ∀ x : obj, Cholangiocarcinoma x → BileDuct x
-- Mild flu comes with a low survival rate.
axiom A5 : LowSurvival MildFlu
-- Colorectal cancer is not both a bile duct cancer and with a low survival rate.
axiom A6 : ¬(BileDuct Colorectal ∧ LowSurvival Colorectal)

-- From the chain: BileDuct → Severe → Deadly → LowSurvival
theorem bileduct_implies_lowsurvival : ∀ x : obj, BileDuct x → LowSurvival x :=
  fun x hb => A1 x (A2 x (A3 x hb))

-- Hence, Colorectal cancer cannot be a bile duct cancer (or it would contradict A6).
theorem not_bileduct_colorectal : ¬BileDuct Colorectal :=
  fun hb => A6 (And.intro hb (bileduct_implies_lowsurvival Colorectal hb))
