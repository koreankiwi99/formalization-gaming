-- If a license plate begins with the number 35, then it does not begin with the number 34.
axiom A5 : ∀ x : obj, Begins35 x → ¬Begins34 x

-- Negation of the statement to evaluate: Tom's license plate is not from Istanbul.
theorem tom_not_from_istanbul : ¬FromIstanbul Tom := A2 Tom (A5 Tom A4)
