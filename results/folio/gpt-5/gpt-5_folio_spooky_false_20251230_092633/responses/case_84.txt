<lean>
axiom obj : Type
axiom Hamden : obj

axiom Listed : obj → Prop
axiom ManyNegReviews : obj → Prop
axiom RatingGt4 : obj → Prop
axiom Popular : obj → Prop
axiom TakeOut : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, Listed x → ¬ManyNegReviews x
axiom P2 : ∀ x : obj, RatingGt4 x → Listed x
axiom P3 : ∃ x : obj, ¬TakeOut x ∧ ManyNegReviews x
axiom P4 : ∀ x : obj, Popular x → RatingGt4 x
axiom P5 : RatingGt4 Hamden ∨ Popular Hamden

-- The given statement is actually derivable from the premises:
-- Hamden provides take-out or does not receive many negative reviews.
theorem hamden_takeout_or_not_many_neg_reviews : TakeOut Hamden ∨ ¬ManyNegReviews Hamden :=
by
  -- From P5 and the chains P4 → P2 → P1, we get ¬ManyNegReviews Hamden
  have notMNR : ¬ManyNegReviews Hamden :=
    match P5 with
    | Or.inl r => P1 Hamden (P2 Hamden r)
    | Or.inr p => P1 Hamden (P2 Hamden (P4 Hamden p))
  exact Or.inr notMNR
</lean>
