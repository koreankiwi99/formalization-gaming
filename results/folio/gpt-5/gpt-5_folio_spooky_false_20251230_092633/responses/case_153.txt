<lean>
axiom obj : Type
axiom Mixture : obj

axiom Hydrocarbon : obj → Prop
axiom Organic : obj → Prop
axiom Chemical : obj → Prop
axiom ContainsCarbon : obj → Prop
axiom OnlyOneElement : obj → Prop
axiom Alkane : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, Hydrocarbon x → Organic x
axiom P2 : ∀ x : obj, Alkane x → Hydrocarbon x
axiom P3 : ∀ x : obj, Organic x → Chemical x
axiom P4 : ∀ x : obj, Organic x → ContainsCarbon x
axiom P5 : ∀ x : obj, Chemical x → ¬OnlyOneElement x
axiom P6 : (Chemical Mixture ∧ OnlyOneElement Mixture) ∨ (¬Chemical Mixture ∧ ¬OnlyOneElement Mixture)

-- Goal: The statement "Mixture is an alkane and contains carbon" is false.
theorem mixture_not_alkane_and_contains_carbon : ¬(Alkane Mixture ∧ ContainsCarbon Mixture) := by
  intro h
  have hAlk : Alkane Mixture := And.left h
  have hChem : Chemical Mixture := by
    have hHc : Hydrocarbon Mixture := P2 Mixture hAlk
    have hOrg : Organic Mixture := P1 Mixture hHc
    exact P3 Mixture hOrg
  have hOne : OnlyOneElement Mixture := by
    cases P6 with
    | inl hBoth =>
      exact And.right hBoth
    | inr hNeither =>
      have hNotChem : ¬Chemical Mixture := And.left hNeither
      exact False.elim (hNotChem hChem)
  have hNotOne : ¬OnlyOneElement Mixture := P5 Mixture hChem
  exact hNotOne hOne
</lean>

ANSWER: False
