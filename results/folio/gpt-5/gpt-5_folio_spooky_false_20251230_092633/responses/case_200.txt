<lean>
axiom obj : Type
axiom Ailton : obj
axiom AiltonSilva : obj
axiom Braga : obj
axiom Nautico : obj
axiom Fluminense : obj

axiom FootballClub : obj → Prop
axiom LoanedOutTo : obj → obj → Prop

-- Premises
axiom Loan_Ailton_Braga : LoanedOutTo Ailton Braga
axiom FC_Braga : FootballClub Braga
axiom FC_Nautico : FootballClub Nautico
axiom FC_Fluminense : FootballClub Fluminense

-- The statement to evaluate: "Ailton was not loaned out to a football club."
-- Formalized as: ¬ ∃ c, FootballClub c ∧ LoanedOutTo Ailton c
def Statement : Prop := ¬∃ c : obj, FootballClub c ∧ LoanedOutTo Ailton c

-- We prove the negation of the statement (i.e., that there exists such a club).
theorem neg_Statement : ¬Statement := by
  intro h
  have ex : ∃ c : obj, FootballClub c ∧ LoanedOutTo Ailton c := by
    apply Exists.intro Braga
    apply And.intro
    · exact FC_Braga
    · exact Loan_Ailton_Braga
  exact h ex
</lean>

ANSWER: False
