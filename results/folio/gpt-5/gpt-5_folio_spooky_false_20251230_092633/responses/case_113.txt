<lean>
axiom obj : Type

axiom PicurisMountains : obj
axiom HardingPegmatiteMine : obj
axiom Juan_de_Onate : obj

axiom MountainRange : obj → Prop
axiom InNewMexico : obj → Prop
axiom InTexas : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Mine : obj → Prop
axiom Donated : obj → Prop
axiom Visited : obj → obj → Prop

-- Premises
axiom P_range : MountainRange PicurisMountains
axiom P_in_NM_or_TX : InNewMexico PicurisMountains ∨ InTexas PicurisMountains
axiom J_visited_P : Visited Juan_de_Onate PicurisMountains
axiom H_is_mine : Mine HardingPegmatiteMine
axiom H_in_P : LocatedIn HardingPegmatiteMine PicurisMountains
axiom H_donated : Donated HardingPegmatiteMine

axiom no_tx_ranges_with_donated_mines :
  ∀ r m : obj, MountainRange r → InTexas r → Mine m → LocatedIn m r → Donated m → False

theorem harding_mine_is_in_a_nm_mountain_range :
  ∃ r : obj, MountainRange r ∧ InNewMexico r ∧ LocatedIn HardingPegmatiteMine r := by
  have not_texas_P : ¬ InTexas PicurisMountains := by
    intro htx
    exact no_tx_ranges_with_donated_mines PicurisMountains HardingPegmatiteMine
      P_range htx H_is_mine H_in_P H_donated
  have P_in_NM : InNewMexico PicurisMountains := by
    cases P_in_NM_or_TX with
    | inl hnm => exact hnm
    | inr htx =>
      exact False.elim (not_texas_P htx)
  exact ⟨PicurisMountains, ⟨P_range, ⟨P_in_NM, H_in_P⟩⟩⟩
</lean>

ANSWER: False
