axiom obj : Type
axiom Harry : obj

axiom Invertebrate : obj → Prop
axiom Vertebrate : obj → Prop
axiom Backbone : obj → Prop
axiom ReproduceByMating : obj → Prop
axiom Bee : obj → Prop
axiom QueenBee : obj → Prop

-- Animals are either invertebrates or vertebrates.
axiom A1 : ∀ x : obj, Invertebrate x ∨ Vertebrate x
-- All vertebrate animals have a backbone.
axiom A2 : ∀ x : obj, Vertebrate x → Backbone x
-- All animals with backbones reproduce by male-and-female mating.
axiom A3 : ∀ x : obj, Backbone x → ReproduceByMating x
-- All bees do not reproduce by male-and-female mating.
axiom A4 : ∀ x : obj, Bee x → ¬ ReproduceByMating x
-- All queen bees are bees.
axiom A5 : ∀ x : obj, QueenBee x → Bee x
-- Harry is a bee.
axiom A6 : Bee Harry

-- From being a bee and the mating chain, Harry cannot be a vertebrate.
theorem harry_not_vertebrate : ¬ Vertebrate Harry := by
  intro hv
  have hb : Backbone Harry := A2 Harry hv
  have hm : ReproduceByMating Harry := A3 Harry hb
  have hnb : ¬ ReproduceByMating Harry := A4 Harry A6
  exact hnb hm

-- Using the dichotomy (invertebrate or vertebrate), Harry must be an invertebrate.
theorem harry_invertebrate : Invertebrate Harry := by
  have disj : Invertebrate Harry ∨ Vertebrate Harry := A1 Harry
  exact Or.elim disj
    (fun hi => hi)
    (fun hv => False.elim (harry_not_vertebrate hv))

-- Therefore, the statement "Harry is an invertebrate or a queen bee" holds.
theorem statement_true : Invertebrate Harry ∨ QueenBee Harry :=
  Or.inl harry_invertebrate
