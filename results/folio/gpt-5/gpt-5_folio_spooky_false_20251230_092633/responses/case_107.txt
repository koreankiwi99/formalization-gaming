axiom obj : Type
axiom Peter : obj

axiom Spill : obj → Prop
axiom Tidy : obj → Prop
axiom Cleanly : obj → Prop
axiom ValueOrder : obj → Prop
axiom FamilyPrioritize : obj → Prop
axiom Clumsy : obj → Prop
axiom Foodie : obj → Prop
axiom GoOutFrequently : obj → Prop

-- If people spill a lot of food on their clothing, then they are not notably tidy.
axiom R1 : ∀ x : obj, Spill x → ¬Tidy x
-- All clumsy foodies who go out frequently are people who spill a lot of food on their clothing.
axiom R2 : ∀ x : obj, Clumsy x ∧ Foodie x ∧ GoOutFrequently x → Spill x
-- If people are cleanly, then they are notably tidy.
axiom R3 : ∀ x : obj, Cleanly x → Tidy x
-- If people value order and spotlessness, then they are cleanly.
axiom R4 : ∀ x : obj, ValueOrder x → Cleanly x
-- All people whose families prioritize order and spotlessness themselves value order and spotlessness.
axiom R5 : ∀ x : obj, FamilyPrioritize x → ValueOrder x
-- Peter is either (spills and is cleanly) or (does not spill and is not cleanly).
axiom R6 : (Spill Peter ∧ Cleanly Peter) ∨ (¬Spill Peter ∧ ¬Cleanly Peter)

theorem left_branch_impossible : ¬(Spill Peter ∧ Cleanly Peter) := by
  intro h
  have hs : Spill Peter := And.left h
  have hc : Cleanly Peter := And.right h
  have hnotT : ¬Tidy Peter := R1 Peter hs
  have hT : Tidy Peter := R3 Peter hc
  exact hnotT hT

theorem peter_not_spill_and_not_cleanly : ¬Spill Peter ∧ ¬Cleanly Peter := by
  have hdisj : (Spill Peter ∧ Cleanly Peter) ∨ (¬Spill Peter ∧ ¬Cleanly Peter) := R6
  apply Or.elim hdisj
  · intro hleft
    have : False := left_branch_impossible hleft
    exact False.elim this
  · intro hright
    exact hright

theorem peter_not_spill : ¬Spill Peter := And.left peter_not_spill_and_not_cleanly
theorem peter_not_cleanly : ¬Cleanly Peter := And.right peter_not_spill_and_not_cleanly

-- Note: From the given premises we cannot derive ¬Tidy Peter.
-- We can show Tidy Peter → ¬Spill Peter, but this is consistent with the derived ¬Spill Peter.
theorem tidy_implies_not_spill : Tidy Peter → ¬Spill Peter := by
  intro hT
  intro hS
  exact (R1 Peter hS) hT
