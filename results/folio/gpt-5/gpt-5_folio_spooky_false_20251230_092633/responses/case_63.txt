<lean>
axiom obj : Type

axiom Pet : obj → Prop
axiom BrownSwissCattle : obj → Prop
axiom Cow : obj → Prop
axiom Domesticated : obj → Prop
axiom Alligator : obj → Prop

axiom Ted : obj

-- Premises
axiom A1 : ∀ x : obj, BrownSwissCattle x → Cow x
axiom A2 : ∃ x : obj, Pet x ∧ BrownSwissCattle x
axiom A3 : ∀ x : obj, Cow x → Domesticated x
axiom A4 : ∀ x : obj, Alligator x → ¬Domesticated x
axiom A5 : Alligator Ted

-- Conclusion to refute: Ted is a pet and Brown Swiss cattle
theorem ted_not_pet_and_brownSwiss : ¬(Pet Ted ∧ BrownSwissCattle Ted) := by
  intro h
  have hbs : BrownSwissCattle Ted := And.right h
  have hcow : Cow Ted := A1 Ted hbs
  have hdom : Domesticated Ted := A3 Ted hcow
  have hndom : ¬Domesticated Ted := A4 Ted A5
  exact hndom hdom
</lean>

ANSWER: False
