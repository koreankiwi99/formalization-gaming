<lean>
axiom obj : Type
axiom Peter : obj

axiom Spill : obj → Prop
axiom Tidy : obj → Prop
axiom CFO : obj → Prop
axiom Cleanly : obj → Prop
axiom ValueOrder : obj → Prop
axiom FamilyPrioritize : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, Spill x → ¬ Tidy x
axiom P2 : ∀ x : obj, CFO x → Spill x
axiom P3 : ∀ x : obj, Cleanly x → Tidy x
axiom P4 : ∀ x : obj, ValueOrder x → Cleanly x
axiom P5 : ∀ x : obj, FamilyPrioritize x → ValueOrder x
axiom P6 : (Spill Peter ∧ Cleanly Peter) ∨ (¬Spill Peter ∧ ¬Cleanly Peter)

-- Useful consequences
theorem peter_not_spill : ¬ Spill Peter := by
  intro s
  cases P6 with
  | inl h =>
      have t : Tidy Peter := P3 Peter h.right
      have nt : ¬ Tidy Peter := P1 Peter s
      exact nt t
  | inr h =>
      exact h.left s

theorem peter_not_cleanly : ¬ Cleanly Peter := by
  intro c
  cases P6 with
  | inl h =>
      have t : Tidy Peter := P3 Peter c
      have nt : ¬ Tidy Peter := P1 Peter h.left
      exact nt t
  | inr h =>
      exact h.right c

theorem peter_not_cfo : ¬ CFO Peter := by
  intro hcfo
  have s : Spill Peter := P2 Peter hcfo
  exact peter_not_spill s

theorem peter_not_fp : ¬ FamilyPrioritize Peter := by
  intro hfp
  have vo : ValueOrder Peter := P5 Peter hfp
  have c : Cleanly Peter := P4 Peter vo
  exact peter_not_cleanly c

-- The statement to evaluate is actually true
theorem statement_true :
  (CFO Peter ∧ FamilyPrioritize Peter) ∨ (¬CFO Peter ∧ ¬FamilyPrioritize Peter) :=
  Or.inr (And.intro peter_not_cfo peter_not_fp)
