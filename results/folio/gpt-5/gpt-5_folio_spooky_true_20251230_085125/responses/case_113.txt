axiom JuanDeOnate : obj

axiom InNewMexico : obj → Prop
axiom InTexas : obj → Prop
axiom Visited : obj → obj → Prop
axiom Mine : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop

-- Premises
axiom P1 : MountainRange Picuris
axiom P2 : InNewMexico Picuris ∨ InTexas Picuris
axiom P3 : Visited JuanDeOnate Picuris
axiom P4 : Mine HardingPegmatiteMine
axiom P5 : LocatedIn HardingPegmatiteMine Picuris
axiom P6 : Donated HardingPegmatiteMine
-- No Texas mountain ranges have donated mines located in them
axiom P7 : ∀ x : obj, MountainRange x → InTexas x → ∀ m : obj, Mine m → LocatedIn m x → ¬ Donated m

-- From the premises, the Picuris Mountains must be in New Mexico (not Texas),
-- because there is a donated mine located in them.
theorem picuris_in_new_mexico : InNewMexico Picuris := by
  cases P2 with
  | inl hnm => exact hnm
  | inr htx =>
      have hnotdon : ¬ Donated HardingPegmatiteMine := P7 Picuris P1 htx HardingPegmatiteMine P4 P5
      have hfalse : False := hnotdon P6
      exact False.elim hfalse

-- Therefore, the Harding Pegmatite Mine is located in a mountain range in New Mexico.
theorem exists_nm_range_for_hpm : ∃ r : obj, MountainRange r ∧ InNewMexico r ∧ LocatedIn HardingPegmatiteMine r := by
  exact Exists.intro Picuris (And.intro P1 (And.intro picuris_in_new_mexico P5))

-- Hence, the statement "The Harding Pegmatite Mine is not located in a mountain range in New Mexico"
-- contradicts the derived fact above.
theorem statement_contradiction :
  (¬ ∃ r : obj, MountainRange r ∧ InNewMexico r ∧ LocatedIn HardingPegmatiteMine r) → False := by
  intro h
  exact h exists_nm_range_for_hpm
