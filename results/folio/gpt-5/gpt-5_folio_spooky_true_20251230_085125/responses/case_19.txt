<lean>
axiom obj : Type
axiom Rockie : obj

axiom Rabbit : obj → Prop
axiom Turtle : obj → Prop
axiom Squirrel : obj → Prop
axiom Near : obj → Prop
axiom Cute : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop

-- All rabbits near campus are cute
axiom RabbitsNearCute : ∀ x : obj, Rabbit x ∧ Near x → Cute x
-- Some turtles can be spotted near the campus
axiom SomeTurtlesNear : ∃ x : obj, Turtle x ∧ Near x
-- Only animals near campus are rabbits or squirrels
axiom OnlyNearAreRabbitOrSquirrel : ∀ x : obj, Near x → Rabbit x ∨ Squirrel x
-- If something is skittish, then it is not calm
axiom SkittishNotCalm : ∀ x : obj, Skittish x → ¬ Calm x
-- All squirrels near campus are skittish
axiom SquirrelsNearSkittish : ∀ x : obj, Squirrel x ∧ Near x → Skittish x
-- Rockie is near and calm
axiom RockieNearCalm : Near Rockie ∧ Calm Rockie

theorem rockie_imp : (¬ (Turtle Rockie ∧ Squirrel Rockie)) → (Cute Rockie ∨ Skittish Rockie) := by
  have hNear : Near Rockie := And.left RockieNearCalm
  have hCalm : Calm Rockie := And.right RockieNearCalm
  -- Rockie cannot be a squirrel (since squirrels near are skittish, and skittish implies not calm)
  have notSquirrel : ¬ Squirrel Rockie := by
    intro hSq
    have hSk : Skittish Rockie := SquirrelsNearSkittish Rockie (And.intro hSq hNear)
    have hNotCalm : ¬ Calm Rockie := SkittishNotCalm Rockie hSk
    exact hNotCalm hCalm
  -- From "near" and "only near are rabbits or squirrels" plus notSquirrel, deduce Rockie is a rabbit
  have hDisj : Rabbit Rockie ∨ Squirrel Rockie := OnlyNearAreRabbitOrSquirrel Rockie hNear
  have hRabbit : Rabbit Rockie := by
    cases hDisj with
    | inl hR => exact hR
    | inr hS => exact False.elim (notSquirrel hS)
  -- Rabbits near are cute, so Rockie is cute
  have hCute : Cute Rockie := RabbitsNearCute Rockie (And.intro hRabbit hNear)
  intro _
  exact Or.inl hCute
</lean>

ANSWER: True
