axiom Tidy : obj → Prop
axiom ClumsyFoodie : obj → Prop
axiom GoesOut : obj → Prop
axiom Clean : obj → Prop
axiom ValuesOrder : obj → Prop
axiom FamilyPrioritizes : obj → Prop

axiom A1 : ∀ x : obj, Spill x → ¬ Tidy x
axiom A2 : ∀ x : obj, (ClumsyFoodie x ∧ GoesOut x) → Spill x
axiom A3 : ∀ x : obj, Clean x → Tidy x
axiom A4 : ∀ x : obj, ValuesOrder x → Clean x
axiom A5 : ∀ x : obj, FamilyPrioritizes x → ValuesOrder x
axiom A6 : (Spill Peter ∧ Clean Peter) ∨ (¬ Spill Peter ∧ ¬ Clean Peter)

-- The left disjunct in A6 leads to a contradiction
theorem not_both_spill_clean : ¬ (Spill Peter ∧ Clean Peter) := by
  intro h
  have t : Tidy Peter := A3 Peter h.right
  have nt : ¬ Tidy Peter := A1 Peter h.left
  exact nt t

-- From A6 and the previous lemma, we get the right disjunct
theorem not_spill_and_not_clean : ¬ Spill Peter ∧ ¬ Clean Peter := by
  have h := A6
  cases h with
  | inl h1 =>
      have contra : False := not_both_spill_clean h1
      exact False.elim contra
  | inr h2 =>
      exact h2

theorem not_clumsy_and_goesout : ¬ (ClumsyFoodie Peter ∧ GoesOut Peter) := by
  intro hC
  have hs : Spill Peter := A2 Peter hC
  have hns : ¬ Spill Peter := (not_spill_and_not_clean).left
  exact hns hs

theorem not_family_prioritizes : ¬ FamilyPrioritizes Peter := by
  intro hF
  have v : ValuesOrder Peter := A5 Peter hF
  have c : Clean Peter := A4 Peter v
  have nc : ¬ Clean Peter := (not_spill_and_not_clean).right
  exact nc c

-- Therefore, the target disjunction does not hold
theorem target_false : ¬ ((ClumsyFoodie Peter ∧ GoesOut Peter) ∨ FamilyPrioritizes Peter) := by
  intro h
  cases h with
  | inl hc =>
      exact not_clumsy_and_goesout hc
  | inr hf =>
      exact not_family_prioritizes hf
