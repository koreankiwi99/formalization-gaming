<lean>
axiom obj : Type

-- Entities
axiom DaniShapiro : obj
axiom FamilyHistory : obj
axiom FamilySecrets : obj
axiom Boston : obj

-- Predicates
axiom Podcast : obj → Prop
axiom Novel : obj → Prop
axiom WrittenBy : obj → obj → Prop
axiom NovelWriter : obj → Prop
axiom American : obj → Prop
axiom Writer : obj → Prop
axiom AmericanCity : obj → Prop
axiom BornIn : obj → obj → Prop
axiom CreatedBy : obj → obj → Prop
axiom WrittenIn2003 : obj → Prop

-- Axioms from the context
axiom P1 : ∀ x : obj, Podcast x → ¬ Novel x
axiom P2 : ∀ p c : obj, BornIn p c → AmericanCity c → American p
axiom P3 : ∀ b p : obj, Novel b → WrittenBy b p → NovelWriter p

axiom A1 : American DaniShapiro
axiom A2 : Writer DaniShapiro
axiom A3 : WrittenBy FamilyHistory DaniShapiro
axiom A4 : Novel FamilyHistory
axiom A5 : WrittenIn2003 FamilyHistory
axiom A6 : CreatedBy FamilySecrets DaniShapiro
axiom A7 : Podcast FamilySecrets
axiom A8 : AmericanCity Boston

-- Goal: Dani Shapiro is a novel writer.
theorem dani_shapiro_novel_writer : NovelWriter DaniShapiro := by
  have hNovelFH : Novel FamilyHistory := A4
  have hWroteFH : WrittenBy FamilyHistory DaniShapiro := A3
  exact P3 FamilyHistory DaniShapiro hNovelFH hWroteFH
</lean>

ANSWER: True
