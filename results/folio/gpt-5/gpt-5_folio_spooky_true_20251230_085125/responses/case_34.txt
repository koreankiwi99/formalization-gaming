<lean>
axiom obj : Type

-- Entities
axiom DiamondMine : obj
axiom Imperium : obj
axiom RoderickStrong : obj
axiom CreedBrothers : obj
axiom IvyNile : obj

-- Relations
axiom Leads : obj → obj → Prop
axiom Includes : obj → obj → Prop
axiom Feud : obj → obj → Prop

-- Contextual facts
axiom H1 : Leads RoderickStrong DiamondMine
axiom H2 : Includes DiamondMine CreedBrothers
axiom H3 : Includes DiamondMine IvyNile
axiom H4 : Feud Imperium DiamondMine

-- Semantic bridge: a leader of a group leads its included members/subgroups
axiom leader_of_group_leads_members : ∀ l g m, Leads l g → Includes g m → Leads l m

theorem roderick_leads_creed_brothers : Leads RoderickStrong CreedBrothers := by
  exact leader_of_group_leads_members RoderickStrong DiamondMine CreedBrothers H1 H2
</lean>

ANSWER: True
