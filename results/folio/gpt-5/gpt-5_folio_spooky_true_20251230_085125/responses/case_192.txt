axiom Event : Type
axiom Happy : Event → Prop
axiom Sad : Event → Prop

-- Interpreting "either happy or sad" as mutually exclusive:
axiom exclusive_either : ∀ e : Event, (Happy e ∧ ¬ Sad e) ∨ (Sad e ∧ ¬ Happy e)
axiom exists_happy : ∃ e : Event, Happy e

-- The statement to evaluate is: ∀ e, Sad e
-- From the premises, we can show this is not the case.
theorem not_all_events_are_sad : ¬ (∀ e : Event, Sad e) := by
  intro allSad
  obtain ⟨w, hwHappy⟩ := exists_happy
  have h := exclusive_either w
  cases h with
  | inl hHappy =>
      have hNotSad : ¬ Sad w := hHappy.right
      exact hNotSad (allSad w)
  | inr hSad =>
      have hNotHappy : ¬ Happy w := hSad.right
      exact hNotHappy hwHappy
