-- What follows from the premises: if George will not compose, then he either cannot compose or does not want to compose.
theorem not_will_implies_not_can_or_not_wants : ¬WillCompose → (¬CanCompose ∨ ¬WantsCompose) := by

  have hcontra : ¬(CanCompose ∧ WantsCompose) := by
    intro hBoth
    apply hnotw
    exact P3 hBoth
  classical
  by_cases hcan : CanCompose
  · right
    intro hwants
    apply hcontra
    exact And.intro hcan hwants
  · left
    exact hcan

-- If we additionally knew George wants to compose, then the target statement would follow.

  intro hnotw
  intro hcan
  have hwill : WillCompose := P3 (And.intro hcan hwants)
