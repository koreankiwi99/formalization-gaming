-- Premises
axiom P1 : ∀ x : obj, Employee x → Schedules x → GoToBuildingToday x
axiom P2 : ∀ x : obj, LunchInBuilding x → Schedules x
axiom P3 : ∀ x : obj, Employee x → (LunchInBuilding x ∨ LunchAtHome x)
axiom P4 : ∀ x : obj, Employee x → LunchAtHome x → Remote x
axiom P5 : ∀ x : obj, Employee x → InOtherCountries x → Remote x
axiom P6 : ∀ x : obj, Manager x → ¬ Remote x
axiom P7 : AppearsToday James ↔ Manager James
axiom P8 : Employee James

-- What we can prove: if James is a manager, he must have lunch in the company building
theorem manager_implies_lunch_company : Manager James → LunchInBuilding James := by
  intro hM
  have hNotRemote : ¬ Remote James := P6 James hM
  have lunch_either : LunchInBuilding James ∨ LunchAtHome James := P3 James P8
  have not_home : ¬ LunchAtHome James := by
    intro hHome
    have hRemote : Remote James := P4 James P8 hHome
    exact hNotRemote hRemote
  cases lunch_either with
  | inl hIn => exact hIn
  | inr hHome => exact False.elim (not_home hHome)
