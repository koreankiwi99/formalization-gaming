<lean>
axiom obj : Type
axiom DriedThaiChilies : obj

axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom BBM : obj → Prop

-- No baked sweets are spicy.
axiom A1 : ∀ x : obj, BakedSweet x → ¬ Spicy x
-- All cupcakes are baked sweets.
axiom A2 : ∀ x : obj, Cupcake x → BakedSweet x
-- All mala hotpots are spicy.
axiom A3 : ∀ x : obj, MalaHotpot x → Spicy x
-- All products from Baked by Melissa are cupcakes.
axiom A4 : ∀ x : obj, BBM x → Cupcake x
-- Dried Thai chilies are spicy or mala hotpots or not baked sweets.
axiom A5 : Spicy DriedThaiChilies ∨ MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies

theorem driedThai_not_bbm_nor_baked :
  ¬ BBM DriedThaiChilies ∧ ¬ BakedSweet DriedThaiChilies := by
  have hnotB : ¬ BakedSweet DriedThaiChilies := by
    cases A5 with
    | inl hs =>
        intro hb
        have ns := A1 DriedThaiChilies hb
        exact ns hs
    | inr h =>
        cases h with
        | inl hm =>
            intro hb
            have hs := A3 DriedThaiChilies hm
            have ns := A1 DriedThaiChilies hb
            exact ns hs
        | inr hnb =>
            exact hnb
  have hnotBBM : ¬ BBM DriedThaiChilies := by
    intro hbbm
    have hc := A4 DriedThaiChilies hbbm
    have hb := A2 DriedThaiChilies hc
    exact hnotB hb
  exact And.intro hnotBBM hnotB
</lean>

ANSWER: True
