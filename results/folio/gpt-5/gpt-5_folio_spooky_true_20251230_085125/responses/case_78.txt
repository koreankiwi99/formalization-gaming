<lean>
axiom obj : Type

-- Entities
axiom DriedThaiChilies : obj

-- Predicates
axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom BBMProduct : obj → Prop

-- Premises
-- 1. No baked sweets are spicy.
axiom P1 : ∀ x : obj, BakedSweet x → ¬ Spicy x
-- 2. All cupcakes are baked sweets.
axiom P2 : ∀ x : obj, Cupcake x → BakedSweet x
-- 3. All mala hotpots are spicy.
axiom P3 : ∀ x : obj, MalaHotpot x → Spicy x
-- 4. All products from Baked by Melissa are cupcakes.
axiom P4 : ∀ x : obj, BBMProduct x → Cupcake x
-- 5. Dried Thai chilies are spicy or mala hotpots or not baked sweets.
axiom P5 : Spicy DriedThaiChilies ∨ MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies

-- Derived: Dried Thai chilies are not baked sweets.
theorem not_baked_sweet_dtc : ¬ BakedSweet DriedThaiChilies := by
  intro bs
  cases P5 with
  | inl s =>
      exact (P1 DriedThaiChilies bs) s
  | inr rest =>
      cases rest with
      | inl mh =>
          have s : Spicy DriedThaiChilies := P3 DriedThaiChilies mh
          exact (P1 DriedThaiChilies bs) s
      | inr not_bs =>
          exact not_bs bs

-- Consequently: Dried Thai chilies are not products of Baked by Melissa.
theorem not_bbm_dtc : ¬ BBMProduct DriedThaiChilies := by
  intro bbm
  have cup : Cupcake DriedThaiChilies := P4 DriedThaiChilies bbm
  have bs : BakedSweet DriedThaiChilies := P2 DriedThaiChilies cup
  exact not_baked_sweet_dtc bs
