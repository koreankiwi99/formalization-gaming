<lean>
axiom obj : Type
axiom HeinrichSchmidt : obj

axiom GermanPolitician : obj → Prop
axiom InPrussianStateParliament : obj → Prop
axiom InNaziReichstag : obj → Prop

axiom HS_german : GermanPolitician HeinrichSchmidt
axiom HS_prussian : InPrussianStateParliament HeinrichSchmidt
axiom HS_reichstag : InNaziReichstag HeinrichSchmidt

theorem exists_politician_in_both :
  ∃ x : obj, GermanPolitician x ∧ InPrussianStateParliament x ∧ InNaziReichstag x :=
by
  refine Exists.intro HeinrichSchmidt ?_
  exact And.intro HS_german (And.intro HS_prussian HS_reichstag)
</lean>

ANSWER: True
