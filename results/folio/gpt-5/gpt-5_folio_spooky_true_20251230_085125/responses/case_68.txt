<lean>
axiom Series : Type
axiom Person : Type
axiom Title : Type

axiom Badults : Series
axiom AndrewCollins : Person
axiom SecretDudeSociety : Title

axiom WorkingTitle : Series → Title → Prop
axiom ScriptEditor : Person → Series → Prop

-- Premises from the context
axiom WT_Badults : WorkingTitle Badults SecretDudeSociety
axiom AC_editor : ScriptEditor AndrewCollins Badults

-- Conclusion: There exists a series with working title "The Secret Dude Society" for which Andrew Collins was the script editor.
theorem claim : ∃ s : Series, WorkingTitle s SecretDudeSociety ∧ ScriptEditor AndrewCollins s := by
  refine Exists.intro Badults ?_
  refine And.intro ?ht ?he
  · exact WT_Badults
  · exact AC_editor
</lean>

ANSWER: True
