<lean>
axiom obj : Type

axiom YaleUniversity : obj
axiom Private : obj → Prop
axiom IvyLeague : obj → Prop
axiom ResearchUniversity : obj → Prop
axiom MovedToNewHaven : obj → Prop

axiom H1 : Private YaleUniversity
axiom H2 : IvyLeague YaleUniversity
axiom H3 : ResearchUniversity YaleUniversity
axiom H4 : MovedToNewHaven YaleUniversity

theorem exists_private_ivy_research_moved :
  ∃ x : obj, Private x ∧ IvyLeague x ∧ ResearchUniversity x ∧ MovedToNewHaven x := by
  refine Exists.intro YaleUniversity ?_
  exact And.intro H1 (And.intro H2 (And.intro H3 H4))
</lean>

ANSWER: True
