axiom obj : Type
axiom Cat : obj → Prop
axiom Pet : obj → Prop
axiom Mammal : obj → Prop

-- Premises:
-- 1. All cats are mammals.
axiom A1 : ∀ x : obj, Cat x → Mammal x
-- 2. Some pets are not mammals.
axiom A2 : ∃ x : obj, Pet x ∧ ¬ Mammal x

-- What we can derive from the premises:
-- There exists a pet that is not a cat.
theorem some_pet_not_cat : ∃ x : obj, Pet x ∧ ¬ Cat x := by
  rcases A2 with ⟨x, hxPet, hxNotMammal⟩
  have hNotCat : ¬ Cat x := by
    intro hCat
    exact hxNotMammal (A1 x hCat)
  exact ⟨x, hxPet, hNotCat⟩

-- A concrete countermodel showing the conclusion "No pets are cats" does not follow.
inductive U where | a | b
open U

def CatM : U → Prop
| a => True
| b => False

def MammalM : U → Prop
| a => True
| b => False

def PetM : U → Prop
| a => True
| b => True

theorem model_all_cats_mammals : ∀ x : U, CatM x → MammalM x := by
  intro x
  cases x with
  | a =>
      intro _; exact True.intro
  | b =>
      intro hb; cases hb

theorem model_some_pets_not_mammals : ∃ x : U, PetM x ∧ ¬ MammalM x := by
  refine ⟨b, ?_, ?_⟩
  · exact True.intro
  · intro hM; cases hM

-- In this model, there is a pet that is a cat, falsifying "No pets are cats".
theorem model_exists_pet_cat : ∃ x : U, PetM x ∧ CatM x := by
  exact ⟨a, True.intro, True.intro⟩
