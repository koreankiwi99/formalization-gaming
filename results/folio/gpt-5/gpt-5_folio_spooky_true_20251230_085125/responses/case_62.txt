axiom obj : Type

axiom Pet : obj → Prop
axiom BrownSwiss : obj → Prop
axiom Cow : obj → Prop
axiom Domesticated : obj → Prop
axiom Alligator : obj → Prop

axiom Ted : obj

-- Premises
axiom P1 : ∀ x : obj, BrownSwiss x → Cow x                  -- All Brown Swiss cattle are cows.
axiom P2 : ∃ x : obj, Pet x ∧ BrownSwiss x                  -- Some pets are Brown Swiss Cattle.
axiom P3 : ∀ x : obj, Cow x → Domesticated x                -- All cows are domesticated animals.
axiom P4 : ∀ x : obj, Alligator x → ¬ Domesticated x        -- Alligators are not domesticated animals.
axiom P5 : Alligator Ted                                    -- Ted is an alligator.

-- Consequence: Ted is not domesticated.
theorem ted_not_domesticated : ¬ Domesticated Ted := by
  have hAll : Alligator Ted := P5
  exact P4 Ted hAll

-- Consequence: There exists a pet that is domesticated (from P2, P1, P3).
theorem exists_pet_domesticated : ∃ x : obj, Pet x ∧ Domesticated x := by
  cases P2 with
  | intro x hx =>
    have hxPet : Pet x := And.left hx
    have hxBS  : BrownSwiss x := And.right hx
    have hxCow : Cow x := P1 x hxBS
    have hxDom : Domesticated x := P3 x hxCow
    exact Exists.intro x (And.intro hxPet hxDom)
