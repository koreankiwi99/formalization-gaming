<lean>
axiom obj : Type

axiom DiamondMine : obj
axiom RoderickStrong : obj
axiom Imperium : obj
axiom CreedBrothers : obj
axiom IvyNile : obj

axiom ProfessionalWrestlingStable : obj → Prop
axiom Leads : obj → obj → Prop
axiom Includes : obj → obj → Prop
axiom FeudsWith : obj → obj → Prop

axiom H1 : ProfessionalWrestlingStable DiamondMine
axiom H2 : Leads RoderickStrong DiamondMine
axiom H3 : Includes DiamondMine CreedBrothers
axiom H4 : Includes DiamondMine IvyNile
axiom H5 : FeudsWith Imperium DiamondMine

theorem roderick_leads_a_professional_wrestling_stable :
  ∃ s : obj, ProfessionalWrestlingStable s ∧ Leads RoderickStrong s :=
by
  apply Exists.intro DiamondMine
  apply And.intro
  · exact H1
  · exact H2
</lean>

ANSWER: True
