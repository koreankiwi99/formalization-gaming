<lean>
axiom obj : Type

-- Entities
axiom UnitedStates : obj
axiom LastSummerOlympics : obj
axiom Tokyo : obj

-- Predicates/relations
axiom SummerOlympics : obj → Prop
axiom SportingEvent : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom WonMostMedalsIn : obj → obj → Prop

-- Axioms from the text
-- The summer Olympic games is a sporting event.
axiom R1 : ∀ x : obj, SummerOlympics x → SportingEvent x
-- The last summer Olympic games was in Tokyo.
axiom T1 : SummerOlympics LastSummerOlympics
axiom T2 : LocatedIn LastSummerOlympics Tokyo
-- The United States won the most medals in the (summer Olympic games) in Tokyo.
axiom R2 : ∀ e : obj, SummerOlympics e ∧ LocatedIn e Tokyo → WonMostMedalsIn UnitedStates e

-- Conclusion: The United States won the most medals in the last summer Olympic games.
theorem us_won_most_last_summer_olympics : WonMostMedalsIn UnitedStates LastSummerOlympics :=
by
  have h : SummerOlympics LastSummerOlympics ∧ LocatedIn LastSummerOlympics Tokyo :=
    And.intro T1 T2
  exact R2 LastSummerOlympics h
</lean>

ANSWER: True
