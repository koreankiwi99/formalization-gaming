<lean>
axiom obj : Type

axiom Suck : obj → Prop
axiom Plunger : obj → Prop
axiom Vacuum : obj → Prop
axiom Vampire : obj → Prop
axiom HouseholdAppliance : obj → Prop

axiom Space : obj
axiom Duster : obj

-- Premises
axiom P1 : ∀ x : obj, Plunger x → Suck x
axiom P2 : ∀ x : obj, Vacuum x → Suck x
axiom P3 : ∀ x : obj, Vampire x → Suck x
axiom P4 : Vacuum Space
axiom P5 : HouseholdAppliance Duster
axiom P6 : ¬ Suck Duster

-- Conclusion
theorem space_sucks : Suck Space := by
  exact P2 Space P4
</lean>

ANSWER: True
