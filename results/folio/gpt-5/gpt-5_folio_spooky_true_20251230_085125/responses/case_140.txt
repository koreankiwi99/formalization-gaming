<lean>
axiom obj : Type

-- Entities and predicates
axiom Hachi : obj
axiom Film : obj → Prop
axiom AppropriateAllAges : obj → Prop
axiom WatchWithoutGuidance : obj → Prop
axiom ExcessiveEroticViolentContent : obj → Prop
axiom RatedGeneralAudience : obj → Prop
axiom FamilyFriendlyAnimated : obj → Prop
axiom InFrozenSeries : obj → Prop

-- Premises
-- If a film is appropriate for all ages, then children can watch it without guidance.
axiom P1 : ∀ x : obj, Film x → AppropriateAllAges x → WatchWithoutGuidance x
-- If a film contains excessive erotic and violent content, children cannot watch it without guidance.
axiom P2 : ∀ x : obj, ExcessiveEroticViolentContent x → ¬ WatchWithoutGuidance x
-- If a film is rated General Audience, then it is appropriate for all ages.
axiom P3 : ∀ x : obj, RatedGeneralAudience x → AppropriateAllAges x
-- All family-friendly animated films are rated General Audience.
axiom P4 : ∀ x : obj, FamilyFriendlyAnimated x → RatedGeneralAudience x
-- All movies in the Frozen series are family-friendly animated films.
axiom P5 : ∀ x : obj, InFrozenSeries x → FamilyFriendlyAnimated x
-- “Hachi: A dog’s Tale” is a film.
axiom P6 : Film Hachi
-- “Hachi: A dog’s Tale” is either a family-friendly animated film or appropriate for all ages.
axiom P7 : FamilyFriendlyAnimated Hachi ∨ AppropriateAllAges Hachi

-- Derived facts
theorem Hachi_watch_without_guidance : WatchWithoutGuidance Hachi := by
  have hFilm : Film Hachi := P6
  cases P7 with
  | inl hFF =>
      have hGA : RatedGeneralAudience Hachi := P4 Hachi hFF
      have hAA : AppropriateAllAges Hachi := P3 Hachi hGA
      exact P1 Hachi hFilm hAA
  | inr hAA =>
      exact P1 Hachi hFilm hAA

theorem not_excessive_for_Hachi : ¬ ExcessiveEroticViolentContent Hachi := by
  intro hEx
  have hnotW : ¬ WatchWithoutGuidance Hachi := P2 Hachi hEx
  exact hnotW Hachi_watch_without_guidance
