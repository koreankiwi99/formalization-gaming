axiom obj : Type

axiom Ekaterina : obj
axiom Vladimir : obj

axiom CanRegisterUS : obj → Prop
axiom Participate2024 : obj → Prop
axiom USCitizen : obj → Prop
axiom TaiwaneseCitizen : obj → Prop
axiom RussianOfficial : obj → Prop
axiom ManagerGazprom : obj → Prop

-- Everyone who can register to vote in the US can participate in the 2024 US presidential election.
axiom A1 : ∀ x : obj, CanRegisterUS x → Participate2024 x
-- If someone has US citizenship, then they can register to vote in the US.
axiom A2 : ∀ x : obj, USCitizen x → CanRegisterUS x
-- A person either has US citizenship or Taiwanese citizenship.
axiom A3 : ∀ x : obj, USCitizen x ∨ TaiwaneseCitizen x
-- No Russian Federation officials hold Taiwanese citizenship.
axiom A4 : ∀ x : obj, RussianOfficial x → ¬ TaiwaneseCitizen x
-- Vladimir neither holds Taiwanese citizenship nor is he a manager at Gazprom.
axiom A5 : ¬ TaiwaneseCitizen Vladimir ∧ ¬ ManagerGazprom Vladimir
-- Ekaterina can register to vote in the US, or she is a Russian federation official.
axiom A6 : CanRegisterUS Ekaterina ∨ RussianOfficial Ekaterina

theorem vladimir_US : USCitizen Vladimir := by
  have hdisj := A3 Vladimir
  have hnotTW : ¬ TaiwaneseCitizen Vladimir := And.left A5
  cases hdisj with
  | inl hUS => exact hUS
  | inr hTW => exact False.elim (hnotTW hTW)

theorem vladimir_can_register : CanRegisterUS Vladimir :=
  A2 Vladimir vladimir_US

theorem vladimir_participates : Participate2024 Vladimir :=
  A1 Vladimir vladimir_can_register

theorem ekaterina_can_register : CanRegisterUS Ekaterina := by
  have hx := A6
  cases hx with
  | inl hreg => exact hreg
  | inr hroff =>
      have hnotTW : ¬ TaiwaneseCitizen Ekaterina := A4 Ekaterina hroff
      have hdisj := A3 Ekaterina
      cases hdisj with
      | inl hUS => exact A2 Ekaterina hUS
      | inr hTW => exact False.elim (hnotTW hTW)

theorem both_true : CanRegisterUS Ekaterina ∧ Participate2024 Vladimir :=
  And.intro ekaterina_can_register vladimir_participates

-- Hence the statement "¬ (CanRegisterUS Ekaterina ∧ Participate2024 Vladimir)" is false.
theorem negate_of_statement : ¬ (¬ (CanRegisterUS Ekaterina ∧ Participate2024 Vladimir)) := by
  intro h
  exact h both_true
