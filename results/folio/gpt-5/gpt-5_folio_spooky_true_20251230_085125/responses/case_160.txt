<lean>
axiom LikesMusic : Prop
axiom HasAccessToProgram : Prop
axiom WantsToCompose : Prop
axiom CanCompose : Prop
axiom WillCompose : Prop

axiom R1 : LikesMusic → WantsToCompose
axiom R2 : HasAccessToProgram → CanCompose
axiom R3 : CanCompose ∧ WantsToCompose → WillCompose

theorem compose_from_likes_and_access :
  LikesMusic ∧ HasAccessToProgram → WillCompose :=
by
  intro h
  have hL : LikesMusic := And.left h
  have hA : HasAccessToProgram := And.right h
  have hW : WantsToCompose := R1 hL
  have hC : CanCompose := R2 hA
  have hCW : CanCompose ∧ WantsToCompose := And.intro hC hW
  exact R3 hCW
</lean>

ANSWER: True
