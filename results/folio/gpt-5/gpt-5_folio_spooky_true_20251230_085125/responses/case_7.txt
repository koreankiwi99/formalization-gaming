-- Premises
-- When the Monkeypox virus occurs in a being, it may get Monkeypox.
axiom A1 : ∀ x : obj, MonkeypoxVirusOccurs x → Monkeypox x
-- Monkeypox virus can occur in certain animals. (existential)
axiom A2 : ∃ x : obj, Animal x ∧ MonkeypoxVirusOccurs x
-- Humans are mammals.
axiom A3 : ∀ x : obj, Human x → Mammal x
-- Mammals are animals.
axiom A4 : ∀ x : obj, Mammal x → Animal x
-- Symptoms of Monkeypox include fever, headache, muscle pains, and tiredness.
axiom A5 : ∀ x : obj, Monkeypox x → Fever x ∧ Headache x ∧ MusclePains x ∧ Tired x
-- People (humans) feel tired when they get the flu.
axiom A6 : ∀ x : obj, Human x → Flu x → Tired x

-- Some derived consequences (do not yield the target conclusion)
theorem monkeypox_causes_tired : ∀ x : obj, Monkeypox x → Tired x := by
  intro x hx
  have h := A5 x hx
  exact h.right.right.right

theorem virus_occurs_tired : ∀ x : obj, MonkeypoxVirusOccurs x → Tired x := by
  intro x hv
  have hm : Monkeypox x := A1 x hv
  exact monkeypox_causes_tired x hm

theorem human_animal : ∀ x : obj, Human x → Animal x := by

  have hm : Mammal x := A3 x hh
  exact A4 x hm
