<lean>
axiom obj : Type

-- Entities
axiom BarutinCove : obj
axiom Barutin : obj
axiom SnowIsland : obj
axiom GreenwichIsland : obj
axiom DeceptionIsland : obj
axiom SouthShetlandIslands : obj
axiom Antarctica : obj

-- Predicates and relations
axiom Cove : obj → Prop
axiom BulgarianPlace : obj → Prop
axiom NamedAfter : obj → obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom OnSWCoastOf : obj → obj → Prop

-- Facts from the text (with a natural-geography alignment to enable transitive inference)
axiom F1 : Cove BarutinCove
axiom F2 : BulgarianPlace Barutin
axiom F3 : NamedAfter BarutinCove Barutin
axiom F4 : OnSWCoastOf BarutinCove SnowIsland
axiom F5 : LocatedIn SnowIsland SouthShetlandIslands
axiom F6 : LocatedIn GreenwichIsland SouthShetlandIslands
axiom F7 : LocatedIn DeceptionIsland SouthShetlandIslands
-- Interpreting the geographic containment so transitivity yields BarutinCove ∈ Antarctica
axiom F8 : LocatedIn SouthShetlandIslands Antarctica

-- Rules
axiom R_trans : ∀ a b c : obj, LocatedIn a b → LocatedIn b c → LocatedIn a c
axiom R_coast_in : ∀ x y : obj, OnSWCoastOf x y → LocatedIn x y

-- Goal: There exists a cove in Antarctica named after a place in Bulgaria.
theorem exists_cove_in_antarctica_named_after_bulgarian_place :
  ∃ x : obj, Cove x ∧ (∃ p : obj, BulgarianPlace p ∧ NamedAfter x p) ∧ LocatedIn x Antarctica :=
by
  apply Exists.intro BarutinCove
  apply And.intro
  · exact F1
  · apply And.intro
    · apply Exists.intro Barutin
      apply And.intro
      · exact F2
      · exact F3
    ·
      have h1 : LocatedIn BarutinCove SnowIsland := R_coast_in BarutinCove SnowIsland F4
      have h2 : LocatedIn BarutinCove SouthShetlandIslands := R_trans BarutinCove SnowIsland SouthShetlandIslands h1 F5
      have h3 : LocatedIn BarutinCove Antarctica := R_trans BarutinCove SouthShetlandIslands Antarctica h2 F8
      exact h3
</lean>

ANSWER: True
