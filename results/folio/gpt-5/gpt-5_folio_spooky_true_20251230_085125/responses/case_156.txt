<lean>
axiom obj : Type
axiom James : obj

axiom Student : obj → Prop
axiom WorksLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom PartTimeUni : obj → Prop
axiom TakingDatabase : obj → Prop
axiom TakingWithDavid : obj → Prop
axiom WorkingLab : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, Student x ∧ WorksLibrary x → ¬ FromCS x
axiom P2 : ∀ x : obj, Student x ∧ PartTimeUni x → WorksLibrary x
axiom P3 : ∀ x : obj, Student x ∧ TakingDatabase x → FromCS x
axiom P4 : ∀ x : obj, Student x ∧ TakingWithDavid x → TakingDatabase x
axiom P5 : ∀ x : obj, Student x ∧ WorkingLab x → TakingWithDavid x
axiom P6a : Student James
axiom P6b : WorkingLab James
axiom P7a : ¬ WorkingLab James
axiom P7b : ¬ PartTimeUni James

theorem james_has_part_time : PartTimeUni James := by
  have hFalse : False := P7a P6b
  exact False.elim hFalse
</lean>

ANSWER: True
