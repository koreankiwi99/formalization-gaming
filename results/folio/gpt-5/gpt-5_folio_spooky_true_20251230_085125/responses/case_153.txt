axiom SingleElement : obj → Prop

axiom Mixture : obj

-- Premises
axiom P1 : ∀ x : obj, Hydrocarbon x → Organic x
axiom P2 : ∀ x : obj, Alkane x → Hydrocarbon x
axiom P3 : ∀ x : obj, Organic x → Chemical x
axiom P4 : ∀ x : obj, Organic x → ContainsCarbon x
axiom P5 : ∀ x : obj, Chemical x → ¬ SingleElement x
axiom P6 : (Chemical Mixture ∧ SingleElement Mixture) ∨ (¬ Chemical Mixture ∧ ¬ SingleElement Mixture)

-- From the chain Alkane → Hydrocarbon → Organic → Chemical
lemma alkane_to_chemical : ∀ x : obj, Alkane x → Chemical x := by
  intro x
  intro hA
  have hH : Hydrocarbon x := P2 x hA
  have hO : Organic x := P1 x hH
  have hC : Chemical x := P3 x hO
  exact hC

-- From the exclusivity in P5 and the equivalence-style P6, Mixture is not a chemical compound
lemma not_chemical_mixture : ¬ Chemical Mixture := by
  cases P6 with
  | inl h =>
      have hc : Chemical Mixture := h.left
      have hs : SingleElement Mixture := h.right
      have hns : ¬ SingleElement Mixture := P5 Mixture hc
      exact hns hs
  | inr h =>
      exact h.left

-- Also Mixture does not contain only one element
lemma not_singleelement_mixture : ¬ SingleElement Mixture := by
  cases P6 with
  | inl h =>
      have hc : Chemical Mixture := h.left
      have hs : SingleElement Mixture := h.right
      have hns : ¬ SingleElement Mixture := P5 Mixture hc
      exact hns hs
  | inr h =>
      exact h.right

-- Hence Mixture cannot be an alkane (since alkanes are chemicals)
theorem not_mixture_alkane : ¬ Alkane Mixture := by
  intro hA
  have hc : Chemical Mixture := alkane_to_chemical Mixture hA
  have nc : ¬ Chemical Mixture := not_chemical_mixture
  exact nc hc

-- Therefore the target conjunction is false
theorem not_target : ¬ (Alkane Mixture ∧ ContainsCarbon Mixture) := by
  intro h
  have hna : ¬ Alkane Mixture := not_mixture_alkane
