<lean>
axiom obj : Type

axiom Student : obj → Prop
axiom WorksInLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom PartTimeUniJob : obj → Prop
axiom TakesDatabase : obj → Prop
axiom ClassWithDavid : obj → Prop
axiom WorksInLab : obj → Prop

-- Premises
-- None of the students who currently work in the library are from the computer science department.
axiom P1 : ∀ x : obj, Student x → WorksInLibrary x → ¬ FromCS x
-- All of the students with part-time jobs offered by the university currently work in the library.
axiom P2 : ∀ x : obj, Student x → PartTimeUniJob x → WorksInLibrary x
-- All of the students who are taking the database course are from the computer science department.
axiom P3 : ∀ x : obj, Student x → TakesDatabase x → FromCS x
-- All of the students taking a class with Professor David are taking the database course.
axiom P4 : ∀ x : obj, Student x → ClassWithDavid x → TakesDatabase x
-- All students working in the lab are taking a class with Professor David.
axiom P5 : ∀ x : obj, Student x → WorksInLab x → ClassWithDavid x

-- Facts about James
axiom James : obj
axiom JStudent : Student James
axiom JWorksLab : WorksInLab James
-- "James doesn't work in the lab or have a part-time job offered by the university."
axiom JNoLabOrNoPT : ¬ WorksInLab James ∨ ¬ PartTimeUniJob James

-- Goal: James takes the database course.
theorem james_takes_database : TakesDatabase James := by
  have hClass : ClassWithDavid James := P5 James JStudent JWorksLab
  exact P4 James JStudent hClass
</lean>

ANSWER: True
