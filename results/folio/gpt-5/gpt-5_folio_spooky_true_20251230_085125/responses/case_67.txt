<lean>
axiom obj : Type

axiom YaleUniversity : obj
axiom PiersonCollege : obj

axiom ResidentialCollegeAtYale : obj → Prop
axiom InResidentialCollegesList : obj → Prop

-- From the text: everything in the given list is a residential college at Yale
axiom list_implies_residential :
  ∀ x : obj, InResidentialCollegesList x → ResidentialCollegeAtYale x

-- From the text: Pierson College is in that list
axiom Pierson_in_list : InResidentialCollegesList PiersonCollege

theorem pierson_is_residential : ResidentialCollegeAtYale PiersonCollege := by
  apply list_implies_residential
  exact Pierson_in_list
</lean>

ANSWER: True
