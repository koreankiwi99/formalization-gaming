axiom SpendTimeWithSibs : obj → Prop
axiom HaveSibsBornTogether : obj → Prop
axiom ComplainsOften : obj → Prop
axiom LivesAtHome : obj → Prop
axiom LivesWithStrangers : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, MultipleBirth x → SpendTimeWithSibs x

axiom P3 : ∀ x : obj, ComplainsOften x → HaveSibsBornTogether x
axiom P4 : ∀ x : obj, LivesAtHome x → ¬ LivesWithStrangers x
axiom P5 : ∀ x : obj, SpendTimeWithSibs x → LivesAtHome x
axiom P6 : (MultipleBirth Luke ∧ LivesWithStrangers Luke) ∨ (¬ MultipleBirth Luke ∧ ¬ LivesWithStrangers Luke)

-- Useful derived lemma: Multiple birth implies not living with strangers
lemma MB_implies_not_LWS (x : obj) : MultipleBirth x → ¬ LivesWithStrangers x := by
  intro hMB
  have hST : SpendTimeWithSibs x := P1 x hMB
  have hHome : LivesAtHome x := P5 x hST
  exact P4 x hHome

-- From the disjunction and the lemma, deduce Luke is not multiple-birth and not living with strangers
theorem luke_not_MB_and_not_LWS : ¬ MultipleBirth Luke ∧ ¬ LivesWithStrangers Luke := by
  have hnotleft : ¬ (MultipleBirth Luke ∧ LivesWithStrangers Luke) := by
    intro h
    have hMB := h.left
    have hLWS := h.right
    have hNotLWS := MB_implies_not_LWS Luke hMB
    exact hNotLWS hLWS
  cases P6 with
  | inl hl =>
      have : False := hnotleft hl
      exact False.elim this
  | inr hr =>
      exact hr

-- From the chain Complains → HaveSibsBornTogether → MultipleBirth and ¬MultipleBirth, deduce ¬Complains

  have hnotMB : ¬ MultipleBirth Luke := (luke_not_MB_and_not_LWS).left
  have hImp : ComplainsOften Luke → MultipleBirth Luke := by
