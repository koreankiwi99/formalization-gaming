<lean>
axiom obj : Type

axiom Hannah : obj

axiom AtSchool : obj → Prop
axiom Work : obj → Prop
axiom Need : obj → Prop
axiom Order : obj → Prop
axiom Enjoy : obj → Prop
axiom Picky : obj → Prop
axiom SpendTime : obj → Prop

-- If people at Mary's school work in student jobs on campus, then they need to earn money.
axiom R1 : ∀ x : obj, AtSchool x → Work x → Need x
-- If people at Mary's school order takeout frequently in college, then they work in student jobs on campus.
axiom R2 : ∀ x : obj, AtSchool x → Order x → Work x
-- People at Mary's school order takeout frequently in college or enjoy the dining hall meals and recipes.
axiom R3 : ∀ x : obj, AtSchool x → (Order x ∨ Enjoy x)
-- If people at Mary's school enjoy the dining hall meals and recipes, then they are not picky eaters.
axiom R4 : ∀ x : obj, AtSchool x → Enjoy x → ¬ Picky x
-- If people at Mary's school enjoy the dining hall meals and recipes, then they spend a lot of their time in the campus dining halls.
axiom R5 : ∀ x : obj, AtSchool x → Enjoy x → SpendTime x

-- Hannah is at Mary's school.
axiom T6 : AtSchool Hannah
-- Hannah works in student jobs on campus and if she needs to earn money, then she is neither picky nor needs to earn money.
axiom T7 : Work Hannah ∧ (Need Hannah → (¬ Picky Hannah ∧ ¬ Need Hannah))

theorem Hannah_needs_money : Need Hannah := by
  have hwork : Work Hannah := And.left T7
  exact R1 Hannah T6 hwork
</lean>

ANSWER: True
