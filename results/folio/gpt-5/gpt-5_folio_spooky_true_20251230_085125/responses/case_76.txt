<lean>
axiom obj : Type

-- Predicates
axiom AtSchool : obj → Prop
axiom Work : obj → Prop
axiom Need : obj → Prop
axiom Order : obj → Prop
axiom Enjoy : obj → Prop
axiom Picky : obj → Prop
axiom Spend : obj → Prop

-- Axioms from the context
axiom R1 : ∀ x : obj, AtSchool x → Work x → Need x
axiom R2 : ∀ x : obj, AtSchool x → Order x → Work x
axiom R3 : ∀ x : obj, AtSchool x → (Order x ∨ Enjoy x)
axiom R4 : ∀ x : obj, AtSchool x → Enjoy x → ¬ Picky x
axiom R5 : ∀ x : obj, AtSchool x → Enjoy x → Spend x

-- Hannah and her properties
axiom Hannah : obj
axiom H_at : AtSchool Hannah
axiom H_work_and_cond : Work Hannah ∧ (Need Hannah → (¬ Picky Hannah ∧ ¬ Need Hannah))

theorem hannah_conclusion : AtSchool Hannah ∧ ¬ Picky Hannah ∧ Spend Hannah := by
  have hAt : AtSchool Hannah := H_at
  have hWork : Work Hannah := And.left H_work_and_cond
  have hcond : Need Hannah → (¬ Picky Hannah ∧ ¬ Need Hannah) := And.right H_work_and_cond
  have hNeed : Need Hannah := R1 Hannah hAt hWork
  have hnegPair : ¬ Picky Hannah ∧ ¬ Need Hannah := hcond hNeed
  have hNotPicky : ¬ Picky Hannah := And.left hnegPair
  have hNotNeed : ¬ Need Hannah := And.right hnegPair
  have hFalse : False := hNotNeed hNeed
  have hSpend : Spend Hannah := False.elim hFalse
  exact And.intro hAt (And.intro hNotPicky hSpend)
</lean>

ANSWER: True
