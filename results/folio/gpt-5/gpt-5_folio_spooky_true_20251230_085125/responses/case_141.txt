axiom obj : Type

axiom Film : obj → Prop
axiom AppropriateForAllAges : obj → Prop
axiom CanWatchWithoutGuidance : obj → Prop
axiom ExcessiveEroticViolentContent : obj → Prop
axiom RatedGeneralAudience : obj → Prop
axiom FamilyFriendlyAnimated : obj → Prop
axiom FrozenSeries : obj → Prop

axiom Hachi : obj

-- Premises
axiom P1 : ∀ x : obj, AppropriateForAllAges x → CanWatchWithoutGuidance x
axiom P2 : ∀ x : obj, ExcessiveEroticViolentContent x → ¬ CanWatchWithoutGuidance x
axiom P3 : ∀ x : obj, RatedGeneralAudience x → AppropriateForAllAges x
axiom P4 : ∀ x : obj, FamilyFriendlyAnimated x → RatedGeneralAudience x
axiom P5 : ∀ x : obj, FrozenSeries x → FamilyFriendlyAnimated x
axiom P6 : Film Hachi
axiom P7 : FamilyFriendlyAnimated Hachi ∨ AppropriateForAllAges Hachi

-- From the premises we can show Hachi can be watched without guidance
theorem hachi_can_watch : CanWatchWithoutGuidance Hachi := by
  have haa : AppropriateForAllAges Hachi := by
    cases P7 with
    | inl hffa =>
      have hga : RatedGeneralAudience Hachi := P4 Hachi hffa
      exact P3 Hachi hga
    | inr haa => exact haa
  exact P1 Hachi haa

-- Hence Hachi does not contain excessive erotic and violent content
theorem hachi_not_excessive : ¬ ExcessiveEroticViolentContent Hachi := by
  intro hex
  have hnot : ¬ CanWatchWithoutGuidance Hachi := P2 Hachi hex
  exact hnot hachi_can_watch
