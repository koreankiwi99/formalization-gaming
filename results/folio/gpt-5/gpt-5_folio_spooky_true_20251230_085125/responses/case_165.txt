<lean>
axiom obj : Type
axiom Show : obj → Prop
axiom AirsMon8 : obj → Prop
axiom Roses : obj → Prop
axiom EpisodeOfBachelor : obj → Prop
axiom PortraysReal : obj → Prop
axiom RealityTV : obj → Prop
axiom TheBachelor : obj
axiom BreakingBad : obj

-- Some show airing at 8 pm on Monday gives out roses on TV.
axiom ExMon8Roses : ∃ s : obj, Show s ∧ AirsMon8 s ∧ Roses s

-- If a show gives out roses on TV, then the show is an episode of The Bachelor.
axiom Roses_implies_BachelorEp : ∀ s : obj, Roses s → EpisodeOfBachelor s

-- The Bachelor portrays the lives of real people.
axiom Bachelor_portrays_real : PortraysReal TheBachelor

-- Episodes of The Bachelor portray the lives of real people (subtle inheritance).
axiom Ep_portrays_real : ∀ s : obj, EpisodeOfBachelor s → PortraysReal s

-- All shows portraying the lives of real people are reality TV shows.
axiom Portray_implies_Reality : ∀ s : obj, PortraysReal s → RealityTV s

-- Breaking Bad is a show.
axiom BB_is_show : Show BreakingBad

-- Breaking Bad is not a reality TV show.
axiom BB_not_reality : ¬ RealityTV BreakingBad

-- Statement to evaluate: If roses are given out during Breaking Bad, then it is on Monday at 8 pm.
theorem bb_roses_implies_mon8 : Roses BreakingBad → AirsMon8 BreakingBad := by
  intro hR
  have hEp : EpisodeOfBachelor BreakingBad := Roses_implies_BachelorEp BreakingBad hR
  have hPR : PortraysReal BreakingBad := Ep_portrays_real BreakingBad hEp
  have hRT : RealityTV BreakingBad := Portray_implies_Reality BreakingBad hPR
  have contra : False := BB_not_reality hRT
  exact False.elim contra
</lean>

ANSWER: True
