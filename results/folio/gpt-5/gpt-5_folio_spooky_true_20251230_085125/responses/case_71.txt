axiom Stock : obj → Prop
axiom Growth : obj → Prop
axiom BoughtRapid : obj → Prop
axiom Suitable : obj → Prop
axiom Mature : obj → Prop

axiom KO : obj

-- Premises
axiom P1 : ∀ x : obj, Growth x → BoughtRapid x
axiom P2 : ∀ x : obj, BoughtRapid x → ¬ Suitable x
axiom P3 : ∃ x : obj, Stock x ∧ Growth x
axiom P4 : ∀ x : obj, Stock x ∧ Mature x → Suitable x
axiom P5 : Stock KO
axiom P6 : Mature KO

-- Derived facts
theorem KO_suitable : Suitable KO := by
  apply P4 KO
  exact And.intro P5 P6

theorem not_Growth_KO : ¬ Growth KO := by
  intro hG
  have hns : ¬ Suitable KO := by
    have hbr : BoughtRapid KO := P1 KO hG
    exact P2 KO hbr
  exact hns KO_suitable

theorem not_stock_and_growth_KO : ¬ (Stock KO ∧ Growth KO) := by
  intro h
  have hS : Suitable KO := by
    apply P4 KO
    exact And.intro (And.left h) P6
  have hns : ¬ Suitable KO := by
    have hbr : BoughtRapid KO := P1 KO (And.right h)
    exact P2 KO hbr
  exact hns hS
