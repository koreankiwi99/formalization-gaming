axiom obj : Type

axiom EatsSaladsRegularly : obj → Prop
axiom VeryConscious : obj → Prop
axiom GrewUpHealthConsciousHome : obj → Prop
axiom FulfillsDailyNutrition : obj → Prop
axiom DisregardsPhysicalWellbeing : obj → Prop
axiom VisitsGymDaily : obj → Prop

axiom A1 : ∀ x : obj, EatsSaladsRegularly x → VeryConscious x
axiom A2 : ∀ x : obj, GrewUpHealthConsciousHome x → EatsSaladsRegularly x
axiom A3 : ∀ x : obj, FulfillsDailyNutrition x → GrewUpHealthConsciousHome x
axiom A4 : ∀ x : obj, DisregardsPhysicalWellbeing x → ¬ VeryConscious x
axiom A5 : ∀ x : obj, VisitsGymDaily x → FulfillsDailyNutrition x

axiom Taylor : obj
axiom A6 :
  (GrewUpHealthConsciousHome Taylor ∧ DisregardsPhysicalWellbeing Taylor) ∨
  (¬ GrewUpHealthConsciousHome Taylor ∧ ¬ DisregardsPhysicalWellbeing Taylor)

theorem ghc_and_dpw_impossible (x : obj) :
  GrewUpHealthConsciousHome x ∧ DisregardsPhysicalWellbeing x → False := by
  intro h
  have hg : GrewUpHealthConsciousHome x := And.left h
  have hd : DisregardsPhysicalWellbeing x := And.right h
  have hESR : EatsSaladsRegularly x := A2 x hg
  have hVC : VeryConscious x := A1 x hESR
  have hNVC : ¬ VeryConscious x := A4 x hd
  exact hNVC hVC

theorem taylor_neither : ¬ GrewUpHealthConsciousHome Taylor ∧ ¬ DisregardsPhysicalWellbeing Taylor := by
  have hnotfirst : ¬ (GrewUpHealthConsciousHome Taylor ∧ DisregardsPhysicalWellbeing Taylor) := by
    intro h
    exact ghc_and_dpw_impossible Taylor h
  have hdisj := A6
  cases hdisj with
  | inl hfirst =>
      exact False.elim (hnotfirst hfirst)
  | inr hsecond =>
      exact hsecond

theorem taylor_not_visits : ¬ VisitsGymDaily Taylor := by
  intro hvg
  have hfd : FulfillsDailyNutrition Taylor := A5 Taylor hvg
  have hghc : GrewUpHealthConsciousHome Taylor := A3 Taylor hfd
  have hnotghc : ¬ GrewUpHealthConsciousHome Taylor := And.left taylor_neither
  exact hnotghc hghc
