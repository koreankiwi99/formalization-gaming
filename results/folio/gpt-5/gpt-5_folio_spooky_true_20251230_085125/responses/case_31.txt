<lean>
axiom obj : Type

-- Constants
axiom Djokovic : obj

-- Predicates
axiom At : obj → Prop
axiom GSC : obj → Prop                -- Grand Slam champion
axiom OscarNom : obj → Prop           -- Oscar-nominated actor
axiom ProTennis : obj → Prop
axiom Celebrity : obj → Prop
axiom Athlete : obj → Prop
axiom WellPaid : obj → Prop
axiom Famous : obj → Prop
axiom TaxHaven : obj → Prop

-- Premises restricted to those at the mixer
axiom P1 : ∀ x : obj, At x → (GSC x ∨ OscarNom x)
axiom P2 : ∀ x : obj, At x → GSC x → ProTennis x
axiom P3 : ∀ x : obj, At x → OscarNom x → Celebrity x
axiom P4 : ∀ x : obj, At x → ProTennis x → Athlete x
axiom P5 : ∀ x : obj, At x → Celebrity x → WellPaid x
axiom P6 : ∀ x : obj, At x → Athlete x → Famous x
axiom P7 : ∀ x : obj, At x → WellPaid x → TaxHaven x

-- Djokovic-specific facts
axiom DjAt : At Djokovic
axiom DjFA_imp_WP : (Famous Djokovic ∧ Athlete Djokovic) → WellPaid Djokovic

theorem djokovic_tax_haven : TaxHaven Djokovic := by
  have hAt : At Djokovic := DjAt
  have hDisj : GSC Djokovic ∨ OscarNom Djokovic := P1 Djokovic hAt
  have hWP : WellPaid Djokovic := by
    cases hDisj with
    | inl hGSC =>
        have hPT : ProTennis Djokovic := P2 Djokovic hAt hGSC
        have hAth : Athlete Djokovic := P4 Djokovic hAt hPT
        have hFam : Famous Djokovic := P6 Djokovic hAt hAth
        exact DjFA_imp_WP ⟨hFam, hAth⟩
    | inr hOscar =>
        have hCeleb : Celebrity Djokovic := P3 Djokovic hAt hOscar
        exact P5 Djokovic hAt hCeleb
  exact P7 Djokovic hAt hWP
</lean>

ANSWER: True
