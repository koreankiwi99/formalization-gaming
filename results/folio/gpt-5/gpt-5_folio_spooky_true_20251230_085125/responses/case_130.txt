<lean>
axiom obj : Type
axiom John : obj

axiom MakePlans : obj → Prop
axiom Engaged : obj → Prop
axiom InviteOthers : obj → Prop
axiom InviteFriends : obj → Prop
axiom WellAttended : obj → Prop
axiom LargerFamily : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, MakePlans x → Engaged x
axiom P2 : ∀ x : obj, InviteOthers x → MakePlans x
axiom P3 : ∀ x : obj, WellAttended x → InviteOthers x
axiom P4 : ∃ x : obj, WellAttended x ∧ LargerFamily x
axiom P5 : Engaged John → (¬ InviteFriends John ∧ ¬ MakePlans John)
axiom P6 : LargerFamily John → (WellAttended John ∨ InviteFriends John)

-- Bridge: inviting friends implies inviting others
axiom Bridge : ∀ x : obj, InviteFriends x → InviteOthers x

theorem john_neither_large_nor_invites_others :
  (¬ LargerFamily John) ∧ (¬ InviteOthers John) := by
  -- First, show John does not invite others
  have hNoInviteOthers : ¬ InviteOthers John := by
    intro hIO
    have hMP : MakePlans John := P2 John hIO
    have hEng : Engaged John := P1 John hMP
    have hNotMP : ¬ MakePlans John := (P5 hEng).right
    exact hNotMP hMP
  -- Then, show John does not have a larger family
  have hNoLarger : ¬ LargerFamily John := by
    intro hLF
    have hdisj : WellAttended John ∨ InviteFriends John := P6 hLF
    cases hdisj with
    | inl hWA =>
        have hIO : InviteOthers John := P3 John hWA
        exact hNoInviteOthers hIO
    | inr hIF =>
        have hIO : InviteOthers John := Bridge John hIF
        exact hNoInviteOthers hIO
  exact And.intro hNoLarger hNoInviteOthers
</lean>
ANSWER: True
