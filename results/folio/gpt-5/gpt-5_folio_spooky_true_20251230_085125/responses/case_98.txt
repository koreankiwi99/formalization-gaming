axiom obj : Type
axiom Harry : obj

axiom Animal : obj → Prop
axiom Invertebrate : obj → Prop
axiom Vertebrate : obj → Prop
axiom Backbone : obj → Prop
axiom MFMate : obj → Prop
axiom Bee : obj → Prop
axiom QueenBee : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, Animal x → (Invertebrate x ∨ Vertebrate x)
axiom P2 : ∀ x : obj, Animal x → Backbone x → MFMate x
axiom P3 : ∀ x : obj, Animal x → Vertebrate x → Backbone x
axiom P4 : ∀ x : obj, Bee x → ¬ MFMate x
axiom P5 : ∀ x : obj, QueenBee x → Bee x
axiom P6 : Bee Harry

-- Some consequences we can derive (but not the target statement)
theorem harry_not_animal_with_backbone : ¬ (Animal Harry ∧ Backbone Harry) := by
  intro h
  have hA : Animal Harry := And.left h
  have hB : Backbone Harry := And.right h
  have hM : MFMate Harry := P2 Harry hA hB
  have hN : ¬ MFMate Harry := P4 Harry P6
  exact hN hM

theorem harry_not_animal_and_vertebrate : ¬ (Animal Harry ∧ Vertebrate Harry) := by
  intro h
  have hA : Animal Harry := And.left h
  have hV : Vertebrate Harry := And.right h
  have hB : Backbone Harry := P3 Harry hA hV
  have hM : MFMate Harry := P2 Harry hA hB
  have hN : ¬ MFMate Harry := P4 Harry P6
  exact hN hM
