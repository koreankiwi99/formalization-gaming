<lean>
axiom obj : Type

axiom WorksLib : obj → Prop
axiom FromCS : obj → Prop
axiom PTJob : obj → Prop
axiom TakingDB : obj → Prop
axiom TakingWithDavid : obj → Prop
axiom WorksLab : obj → Prop

axiom James : obj

-- Premises
axiom A1 : ∀ x : obj, WorksLib x → ¬ FromCS x
axiom A2 : ∀ x : obj, PTJob x → WorksLib x
axiom A3 : ∀ x : obj, TakingDB x → FromCS x
axiom A4 : ∀ x : obj, TakingWithDavid x → TakingDB x
axiom A5 : ∀ x : obj, WorksLab x → TakingWithDavid x
axiom A6 : WorksLab James
axiom A7 : ¬ (WorksLab James ∨ PTJob James)

-- Conclusion: James does not have a part-time job offered by the university.
theorem james_no_ptjob : ¬ PTJob James :=
  fun hpt => A7 (Or.inr hpt)
</lean>

ANSWER: True
