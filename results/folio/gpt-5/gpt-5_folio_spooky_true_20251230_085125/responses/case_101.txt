axiom A4 : ∀ x : obj, LiveAtHome x → ¬ LiveWithStrangers x

-- Disjunction about Luke
axiom A7 : (BabyMultiBirth Luke ∧ LiveWithStrangers Luke) ∨ (¬ BabyMultiBirth Luke ∧ ¬ LiveWithStrangers Luke)

-- From A7 and the other axioms, the left disjunct is impossible
theorem luke_not_left_disjunct : ¬ (BabyMultiBirth Luke ∧ LiveWithStrangers Luke) := by
  intro h
  have hb : BabyMultiBirth Luke := And.left h
  have hs : LiveWithStrangers Luke := And.right h
  have hmb : MultiBirth Luke := A6 Luke hb
  have hsp : SpendTimeWithSiblings Luke := A1 Luke hmb
  have hhome : LiveAtHome Luke := A5 Luke hsp
  have hnstr : ¬ LiveWithStrangers Luke := A4 Luke hhome
  exact hnstr hs

-- Therefore, the right disjunct must hold
theorem luke_right_disjunct : (¬ BabyMultiBirth Luke ∧ ¬ LiveWithStrangers Luke) := by
  cases A7 with
  | inl h =>
      exact False.elim (luke_not_left_disjunct h)
  | inr h =>
      exact h

-- Consequences we can derive
