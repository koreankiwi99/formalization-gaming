axiom obj : Type
axiom Bird : obj → Prop
axiom Hawk : obj → Prop
axiom Swim : obj → Prop

-- A hawk never swims.
axiom H1 : ∀ x : obj, Hawk x → ¬ Swim x
-- Some birds are hawks.
axiom H2 : ∃ x : obj, Bird x ∧ Hawk x

-- From the premises, we can derive that not all birds swim.
theorem not_all_birds_swim : ¬ (∀ x : obj, Bird x → Swim x) := by
  intro h
  cases H2 with
  | intro x hx =>
    have hxB : Bird x := And.left hx
    have hxH : Hawk x := And.right hx
    have hxS : Swim x := h x hxB
    have hxNotS : ¬ Swim x := H1 x hxH
    exact hxNotS hxS

-- We can also derive that there exists a bird that does not swim.
theorem exists_bird_not_swim : ∃ x : obj, Bird x ∧ ¬ Swim x := by
  cases H2 with
  | intro x hx =>
    exact Exists.intro x (And.intro (And.left hx) (H1 x (And.right hx)))
