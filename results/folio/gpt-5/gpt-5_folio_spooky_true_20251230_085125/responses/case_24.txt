<lean>
axiom obj : Type
axiom Beijing : obj
axiom O2008Summer : obj
axiom WinterOlympics : obj
axiom Hosted : obj → obj → Prop

-- From the text:
-- "Beijing hosted the 2008 Summer Olympics"
axiom H_summer2008 : Hosted Beijing O2008Summer
-- "Beijing has hosted the ... Winter Olympics"
axiom H_winter : Hosted Beijing WinterOlympics

theorem beijing_hosted_2008_and_winter : Hosted Beijing O2008Summer ∧ Hosted Beijing WinterOlympics := by
  exact And.intro H_summer2008 H_winter
</lean>

ANSWER: True
