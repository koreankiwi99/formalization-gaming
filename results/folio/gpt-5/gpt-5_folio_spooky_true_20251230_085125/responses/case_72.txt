<lean>
axiom obj : Type
axiom KO : obj

-- Predicates
axiom Growth : obj → Prop
axiom Rapid : obj → Prop
axiom Suitable : obj → Prop
axiom Mature : obj → Prop
axiom Stock : obj → Prop
axiom Volatile : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, Growth x → Rapid x
axiom A2 : ∀ x : obj, Rapid x → ¬ Suitable x
axiom A3 : ∃ x : obj, Stock x ∧ Growth x
axiom A4 : ∀ x : obj, Mature x → Suitable x
axiom A5 : Mature KO

-- Target statement
theorem target :
  (Growth KO ∨ Rapid KO) → (¬ Stock KO ∧ ¬ Volatile KO) := by
  intro h
  have hS : Suitable KO := A4 KO A5
  have f : False := by
    cases h with
    | inl g =>
      have r : Rapid KO := A1 KO g
      have ns : ¬ Suitable KO := A2 KO r
      exact ns hS
    | inr r =>
      have ns : ¬ Suitable KO := A2 KO r
      exact ns hS
  apply And.intro
  · intro s
    exact False.elim f
  · intro v
    exact False.elim f
</lean>

ANSWER: True
