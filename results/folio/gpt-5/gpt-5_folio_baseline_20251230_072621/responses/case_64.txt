<lean>
axiom obj : Type

axiom BrownSwissCattle : obj → Prop
axiom Cow : obj → Prop
axiom Pet : obj → Prop
axiom Domesticated : obj → Prop
axiom Alligator : obj → Prop

axiom all_BSC_are_cows : ∀ x : obj, BrownSwissCattle x → Cow x
axiom some_pets_BSC : ∃ x : obj, Pet x ∧ BrownSwissCattle x
axiom all_cows_domesticated : ∀ x : obj, Cow x → Domesticated x
axiom alligators_not_domesticated : ∀ x : obj, Alligator x → ¬ Domesticated x

axiom Ted : obj
axiom Ted_is_alligator : Alligator Ted

theorem ted_bsc_implies_not_pet : BrownSwissCattle Ted → ¬ Pet Ted := by
  intro hBSC
  have hCow : Cow Ted := all_BSC_are_cows Ted hBSC
  have hDom : Domesticated Ted := all_cows_domesticated Ted hCow
  have hNotDom : ¬ Domesticated Ted := alligators_not_domesticated Ted Ted_is_alligator
  intro _hPet
  exact hNotDom hDom
</lean>

ANSWER: True
