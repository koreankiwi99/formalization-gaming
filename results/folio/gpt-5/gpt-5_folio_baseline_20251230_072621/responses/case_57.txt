<lean>
axiom City : Type
axiom State : Type

axiom Billings : City
axiom Butte : City
axiom Helena : City
axiom Missoula : City
axiom WhiteSulphurSprings : City
axiom StPierre : City

axiom Montana : State

axiom InState : City → State → Prop

-- Billings is in Montana
axiom T1 : InState Billings Montana
-- Montana includes Butte, Helena, Missoula
axiom T2 : InState Butte Montana
axiom T3 : InState Helena Montana
axiom T4 : InState Missoula Montana
-- White Sulphur Springs and Butte are in the same state
axiom T5 : ∃ s : State, InState WhiteSulphurSprings s ∧ InState Butte s
-- St Pierre is not in Montana
axiom T6 : ¬ InState StPierre Montana
-- Any city in the same state as Butte is not in the same state as St Pierre
axiom T7 : ∀ (c : City) (s1 s2 : State), InState c s1 → InState Butte s1 → InState StPierre s2 → ¬ InState c s2

-- Claim to assess: ∃ s, InState Butte s ∧ InState StPierre s
-- We prove its negation.
theorem butte_stpierre_not_same_state : ¬ (∃ s : State, InState Butte s ∧ InState StPierre s) :=
by
  intro h
  apply Exists.elim h
  intro s hs
  have hB : InState Butte s := And.left hs
  have hP : InState StPierre s := And.right hs
  have hcontra : ¬ InState Butte s := T7 Butte s s hB hB hP
  exact hcontra hB
</lean>
