<lean>
axiom Building : Type
axiom OliveGarden : Building
axiom Managed : Building → Prop
axiom PetsAllowed : Building → Prop
axiom Rent : Building → Nat
axiom SecurityDeposit : Building → Nat

-- Contextual axioms
axiom some_managed_allows_pets : ∃ b : Building, Managed b ∧ PetsAllowed b
axiom sd_ge_rent : ∀ b : Building, Managed b → SecurityDeposit b ≥ Rent b
axiom OG_managed : Managed OliveGarden
axiom OG_rent_2000 : Rent OliveGarden = 2000
axiom gt_fact : 1500 < 2000  -- not used in the proof

-- Irrelevant extra context (not used in the proof)
axiom Person : Type
axiom Tom : Person
axiom Animal : Type
axiom Fluffy : Animal
axiom Cat : Animal → Prop
axiom Pet : Animal → Prop
axiom cat_is_pet : ∀ a : Animal, Cat a → Pet a
axiom Fluffy_cat : Cat Fluffy
axiom AllowedMoveInWithPet : Building → Person → Animal → Prop
axiom TomWillRent : Building → Prop
axiom allow_pets_implies_allowed_move_in :
  ∀ b : Building, ∀ p : Person, ∀ a : Animal,
    Managed b → PetsAllowed b → Pet a → AllowedMoveInWithPet b p a
axiom tom_rents_if_allowed_and_sd_le_1500 :
  ∀ b : Building, Managed b → AllowedMoveInWithPet b Tom Fluffy → SecurityDeposit b ≤ 1500 → TomWillRent b

-- Target statement: The security deposit at the Olive Garden is either $2000 or more (i.e., ≥ 2000).
theorem OG_security_deposit_ge_2000 : SecurityDeposit OliveGarden ≥ 2000 := by
  have h1 : SecurityDeposit OliveGarden ≥ Rent OliveGarden := sd_ge_rent OliveGarden OG_managed
  have h2 : Rent OliveGarden = 2000 := OG_rent_2000
  simpa [h2] using h1
</lean>

ANSWER: True
