axiom obj : Type
-- The algorithm used to train the state-of-the-art text summarization model
axiom SummAlgo : obj

-- Predicates
axiom IsML : obj → Prop
axiom Supervised : obj → Prop
axiom Unsupervised : obj → Prop
axiom Reinforcement : obj → Prop
axiom RequiresLabeled : obj → Prop

-- Context axioms
-- ML algorithms can be categorized as supervised, unsupervised, or reinforcement learning
axiom ML_categorization : ∀ a : obj, IsML a → (Supervised a ∨ Unsupervised a ∨ Reinforcement a)
-- Unsupervised learning algorithms do not require labeled data
axiom Unsupervised_no_labels : ∀ a : obj, Unsupervised a → ¬ RequiresLabeled a
-- The SOTA text summarization model is trained with an ML algorithm
axiom Summ_is_ML : IsML SummAlgo
-- Reinforcement learning is not used to train the SOTA text summarization model
axiom Not_RL_Summ : ¬ Reinforcement SummAlgo
-- The ML algorithm for training text summarization models requires labeled data
axiom Summ_requires_labels : RequiresLabeled SummAlgo

-- The statement in question would be: Unsupervised SummAlgo
-- We prove its negation from the given context.
theorem not_unsupervised_summ : ¬ Unsupervised SummAlgo :=
by
  intro hUnsup
  have hNoLabels : ¬ RequiresLabeled SummAlgo := Unsupervised_no_labels SummAlgo hUnsup
  exact hNoLabels Summ_requires_labels
