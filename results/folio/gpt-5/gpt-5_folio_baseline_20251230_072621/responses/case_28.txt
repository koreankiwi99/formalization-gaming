<lean>
axiom obj : Type
axiom Marvin : obj

axiom Alien : obj → Prop
axiom Extraterrestrial : obj → Prop
axiom Human : obj → Prop
axiom FromEarth : obj → Prop
axiom FromMars : obj → Prop
axiom HighlyIntelligent : obj → Prop

-- Axioms from the context
axiom A1 : ∀ x : obj, Alien x → Extraterrestrial x
axiom A2 : ∀ x : obj, FromMars x → Alien x
axiom A3 : ∀ x : obj, Extraterrestrial x → ¬ Human x
axiom A4 : ∀ x : obj, (HighlyIntelligent x ∧ FromEarth x) → Human x
axiom A5 : HighlyIntelligent Marvin
axiom A6 : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬ FromEarth Marvin ∧ ¬ FromMars Marvin)
axiom A7 : ¬ FromEarth Marvin → Extraterrestrial Marvin

theorem marvin_neither : ¬ Human Marvin ∧ ¬ FromMars Marvin := by
  -- First, show Marvin cannot be both from Earth and from Mars
  have not_both : ¬ (FromEarth Marvin ∧ FromMars Marvin) := by
    intro h
    have fe : FromEarth Marvin := h.left
    have fm : FromMars Marvin := h.right
    have alienM : Alien Marvin := A2 Marvin fm
    have extraM : Extraterrestrial Marvin := A1 Marvin alienM
    have notHumanM : ¬ Human Marvin := A3 Marvin extraM
    have hiM : HighlyIntelligent Marvin := A5
    have humanM : Human Marvin := A4 Marvin (And.intro hiM fe)
    exact notHumanM humanM
  -- Use the given disjunction to conclude Marvin is from neither
  have neitherEFM : ¬ FromEarth Marvin ∧ ¬ FromMars Marvin :=
    Or.elim A6
      (fun hA => False.elim (not_both hA))
      (fun hB => hB)
  have nfe : ¬ FromEarth Marvin := neitherEFM.left
  have nfm : ¬ FromMars Marvin := neitherEFM.right
  -- From not from Earth, deduce extraterrestrial, hence not human
  have extra2 : Extraterrestrial Marvin := A7 nfe
  have notHuman2 : ¬ Human Marvin := A3 Marvin extra2
  exact And.intro notHuman2 nfm
</lean>

ANSWER: True
