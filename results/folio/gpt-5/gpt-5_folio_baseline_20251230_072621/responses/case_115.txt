axiom obj : Type

axiom Michael : obj
axiom Peter : obj
axiom Windy : obj

axiom Man : obj → Prop
axiom Taller : obj → obj → Prop
axiom Block : obj → obj → Prop
axiom Jump : obj → Prop
axiom Shooter : obj → Prop
axiom Great : obj → Prop
axiom InClassOf : obj → obj → Prop

-- If a man is taller than another man, the taller one can block the other's shooting.
axiom man_taller_blocks : ∀ x y : obj, Man x → Man y → Taller x y → Block x y

-- Michael is a man who is taller than everyone else in his class.
axiom Michael_man : Man Michael
axiom Michael_taller_than_class : ∀ x : obj, InClassOf x Michael → x ≠ Michael → Taller Michael x

-- Transitivity of taller
axiom taller_trans : ∀ x y z : obj, Taller x y → Taller y z → Taller x z

-- Peter is a man who is taller than Michael.
axiom Peter_man : Man Peter
axiom Peter_taller_Michael : Taller Peter Michael

-- Michael can block any shooting from a person as long as the person does not jump when shooting.
axiom Michael_blocks_if_not_jump : ∀ x : obj, ¬ Jump x → Block Michael x

-- Michael cannot block Windy's shooting.
axiom Michael_cannot_block_Windy : ¬ Block Michael Windy

-- Every shooter who can jump when shooting is a great shooter.
axiom great_if_shooter_and_jump : ∀ x : obj, Shooter x → Jump x → Great x

-- From referring to "Windy's shooting", we take Windy to be a shooter.
axiom Windy_shooter : Shooter Windy

open classical

theorem windy_is_great : Great Windy := by
  -- From "Michael can block any shooting from a person as long as the person does not jump when shooting"
  -- and "Michael cannot block Windy's shooting", deduce that Windy must jump when shooting.
  have h_imp : ¬ Jump Windy → Block Michael Windy := Michael_blocks_if_not_jump Windy
  have h_not : ¬ Block Michael Windy := Michael_cannot_block_Windy
  have h_jump : Jump Windy := by
    by_contra hnj
    exact h_not (h_imp hnj)
  -- Then apply the rule that any shooter who can jump when shooting is a great shooter.
  exact great_if_shooter_and_jump Windy Windy_shooter h_jump
</lean>

ANSWER: True
