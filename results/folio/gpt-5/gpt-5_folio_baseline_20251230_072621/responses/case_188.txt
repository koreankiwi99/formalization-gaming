<lean>
axiom obj : Type
axiom Yuri : obj

axiom ProfessionalBasketballPlayer : obj → Prop
axiom American : obj → Prop
axiom CanDoJumpShots : obj → Prop
axiom LeapsStraight : obj → Prop
axiom ActivatesLegMuscles : obj → Prop

axiom some_pro_not_american : ∃ x, ProfessionalBasketballPlayer x ∧ ¬ American x
axiom pros_can_jump : ∀ x, ProfessionalBasketballPlayer x → CanDoJumpShots x
axiom jumpers_leap : ∀ x, CanDoJumpShots x → LeapsStraight x
axiom leapers_activate : ∀ x, LeapsStraight x → ActivatesLegMuscles x
axiom yuri_not_activate : ¬ ActivatesLegMuscles Yuri

theorem yuri_not_american_pro : ¬ (American Yuri ∧ ProfessionalBasketballPlayer Yuri) := by
  intro h
  have hPro : ProfessionalBasketballPlayer Yuri := And.right h
  have notPro : ¬ ProfessionalBasketballPlayer Yuri := by
    intro hP
    apply yuri_not_activate
    have hAct : ActivatesLegMuscles Yuri := by
      have hLeap : LeapsStraight Yuri := by
        have hJump : CanDoJumpShots Yuri := pros_can_jump Yuri hP
        exact jumpers_leap Yuri hJump
      exact leapers_activate Yuri hLeap
    exact hAct
  exact notPro hPro
</lean>

ANSWER: True
