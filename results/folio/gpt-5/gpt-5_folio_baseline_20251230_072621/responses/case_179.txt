<lean>
axiom Person : Type
axiom Airport : Type

axiom Susan : Person
axiom John : Person
axiom LGA : Airport

axiom FliesTo : Person → Airport → Prop
axiom FliesFrom : Person → Airport → Prop

-- Susan flies to LGA
axiom T1 : FliesTo Susan LGA
-- Departure and arrival cannot be at the same airport
axiom R1 : ∀ p : Person, ∀ a : Airport, ¬ (FliesFrom p a ∧ FliesTo p a)
-- John flies from LGA (irrelevant to Susan's case)
axiom T2 : FliesFrom John LGA

theorem susan_not_from_LGA : ¬ FliesFrom Susan LGA := by
  intro hfrom
  have hnot : ¬ (FliesFrom Susan LGA ∧ FliesTo Susan LGA) := R1 Susan LGA
  exact hnot (And.intro hfrom T1)
</lean>

ANSWER: False
