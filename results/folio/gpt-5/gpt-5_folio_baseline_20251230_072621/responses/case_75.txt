<lean>
axiom obj : Type
axiom AtSchool : obj → Prop
axiom Work : obj → Prop
axiom NeedMoney : obj → Prop
axiom OrderTakeout : obj → Prop
axiom EnjoyDining : obj → Prop
axiom Picky : obj → Prop
axiom SpendTimeDiningHalls : obj → Prop

-- Rules about people at Mary's school
axiom R1 : ∀ x : obj, AtSchool x ∧ Work x → NeedMoney x
axiom R2 : ∀ x : obj, AtSchool x ∧ OrderTakeout x → Work x
axiom R3 : ∀ x : obj, AtSchool x → (OrderTakeout x ∨ EnjoyDining x)
axiom R4 : ∀ x : obj, AtSchool x ∧ EnjoyDining x → ¬ Picky x
axiom R5 : ∀ x : obj, AtSchool x ∧ EnjoyDining x → SpendTimeDiningHalls x

-- Hannah-specific facts
axiom Hannah : obj
axiom HannahAt : AtSchool Hannah
axiom HannahWork : Work Hannah
axiom HannahRule : NeedMoney Hannah → (¬ Picky Hannah ∧ ¬ NeedMoney Hannah)

-- Target: Hannah needs to earn money to help pay for her college tuition
theorem hannah_needs_money : NeedMoney Hannah := by
  apply R1 Hannah
  exact And.intro HannahAt HannahWork

-- Note: From HannahRule we can also derive ¬ NeedMoney Hannah
theorem hannah_not_need_money : ¬ NeedMoney Hannah := by
  intro hneed
  have h := HannahRule hneed
  exact h.right hneed
</lean>

ANSWER: True
