<lean>
axiom obj : Type

-- Predicates
axiom City : obj → Prop
axiom State : obj → Prop
axiom USState : obj → Prop
axiom In : obj → obj → Prop  -- In city state

-- Entities
axiom Billings Butte Helena Missoula WhiteSulphurSprings StPierre Montana : obj
axiom Bristol Texarkana Texhoma UnionCity : obj

-- Context axioms
-- Billings is a city in the state of Montana in U.S.
axiom City_Billings : City Billings
axiom S_Montana : State Montana
axiom US_Montana : USState Montana
axiom In_Billings_Montana : In Billings Montana

-- The state of Montana includes the cities of Butte, Helena, and Missoula.
axiom City_Butte : City Butte
axiom City_Helena : City Helena
axiom City_Missoula : City Missoula
axiom In_Butte_Montana : In Butte Montana
axiom In_Helena_Montana : In Helena Montana
axiom In_Missoula_Montana : In Missoula Montana

-- White Sulphur Springs and Butte are cities in the same state in U.S.
axiom City_WSS : City WhiteSulphurSprings
axiom Exists_same_state_WSS_Butte :
  ∃ s : obj, USState s ∧ In WhiteSulphurSprings s ∧ In Butte s

-- The city of St Pierre is not in the state of Montana.
axiom City_StPierre : City StPierre
axiom NotIn_StPierre_Montana : ¬ In StPierre Montana

-- Any city in Butte is not in St Pierre.
axiom Rule_city_in_Butte_not_in_StPierre :
  ∀ x : obj, City x ∧ In x Butte → ¬ In x StPierre

-- A city can only be in one state in U.S. except for Bristol, Texarkana, Texhoma and Union City.
axiom Unique_USState_membership_except_exceptions :
  ∀ x y z : obj, City x → USState y → USState z → In x y → In x z →
    x ≠ Bristol → x ≠ Texarkana → x ≠ Texhoma → x ≠ UnionCity → y = z

-- Goal: Montana is home to the city of Missoula.
theorem missoula_in_montana : In Missoula Montana := by
  exact In_Missoula_Montana
</lean>

ANSWER: True
