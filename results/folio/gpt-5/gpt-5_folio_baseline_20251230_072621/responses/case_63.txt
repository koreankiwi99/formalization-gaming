<lean>
axiom obj : Type

axiom Ted : obj

axiom BrownSwiss : obj → Prop
axiom Cow : obj → Prop
axiom Pet : obj → Prop
axiom Domesticated : obj → Prop
axiom Alligator : obj → Prop

-- All Brown Swiss cattle are cows.
axiom A1 : ∀ x : obj, BrownSwiss x → Cow x
-- Some pets are Brown Swiss cattle.
axiom A2 : ∃ x : obj, Pet x ∧ BrownSwiss x
-- All cows are domesticated animals.
axiom A3 : ∀ x : obj, Cow x → Domesticated x
-- Alligators are not domesticated animals.
axiom A4 : ∀ x : obj, Alligator x → ¬ Domesticated x
-- Ted is an alligator.
axiom A5 : Alligator Ted

theorem not_pet_and_brown_swiss_ted : ¬ (Pet Ted ∧ BrownSwiss Ted) :=
  let not_dom_ted : ¬ Domesticated Ted := A4 Ted A5
  let not_brown_swiss_ted : ¬ BrownSwiss Ted :=
    fun h => not_dom_ted (A3 Ted (A1 Ted h))
  fun h => not_brown_swiss_ted h.right
</lean>

ANSWER: False
