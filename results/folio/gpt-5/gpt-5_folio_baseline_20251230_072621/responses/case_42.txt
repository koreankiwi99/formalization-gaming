<lean>
axiom obj : Type
axiom Coco : obj

axiom RankedHighWTA : obj → Prop
axiom MostActiveGS : obj → Prop
axiom LostToIgaRG2022 : obj → Prop
axiom FemaleRG2022 : obj → Prop
axiom MaleRG2022 : obj → Prop
axiom LostToNadalRG2022 : obj → Prop
axiom AtRG2022 : obj → Prop

-- If someone is ranked highly by the WTA, then they are among the most active Grand-Slam players.
axiom A1 : ∀ x : obj, RankedHighWTA x → MostActiveGS x
-- Everyone who lost to Iga Świątek at Roland Garros 2022 is ranked highly by the WTA.
axiom A2 : ∀ x : obj, LostToIgaRG2022 x → RankedHighWTA x
-- All female tennis players at Roland Garros 2022 lost to Iga Świątek.
axiom A3 : ∀ x : obj, FemaleRG2022 x → LostToIgaRG2022 x
-- All tennis players at Roland Garros 2022 are either female or male.
axiom A4 : ∀ x : obj, AtRG2022 x → FemaleRG2022 x ∨ MaleRG2022 x
-- All male tennis players at Roland Garros 2022 lost to Rafael Nadal.
axiom A5 : ∀ x : obj, MaleRG2022 x → LostToNadalRG2022 x
-- If Coco Gauff is ranked highly by the WTA or lost to Rafael Nadal, then Coco Gauff is not male at Roland Garros 2022.
axiom A6 : (RankedHighWTA Coco ∨ LostToNadalRG2022 Coco) → ¬ MaleRG2022 Coco
-- Coco Gauff is at Roland Garros 2022.
axiom A7 : AtRG2022 Coco

theorem coco_most_active : MostActiveGS Coco := by
  have hdisj : FemaleRG2022 Coco ∨ MaleRG2022 Coco := A4 Coco A7
  cases hdisj with
  | inl hf =>
      have hI : LostToIgaRG2022 Coco := A3 Coco hf
      have hH : RankedHighWTA Coco := A2 Coco hI
      exact A1 Coco hH
  | inr hm =>
      have hN : LostToNadalRG2022 Coco := A5 Coco hm
      have hNotM : ¬ MaleRG2022 Coco := A6 (Or.inr hN)
      exact False.elim (hNotM hm)
</lean>

ANSWER: True
