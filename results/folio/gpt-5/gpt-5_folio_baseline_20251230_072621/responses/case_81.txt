<lean>
axiom obj : Type
axiom DriedThaiChilies : obj

axiom BakedSweet : obj → Prop
axiom Spicy : obj → Prop
axiom Cupcake : obj → Prop
axiom MalaHotpot : obj → Prop
axiom FromBBM : obj → Prop

-- No baked sweets are spicy.
axiom A1 : ∀ x : obj, BakedSweet x → ¬ Spicy x
-- All cupcakes are baked sweets.
axiom A2 : ∀ x : obj, Cupcake x → BakedSweet x
-- All mala hotpots are spicy.
axiom A3 : ∀ x : obj, MalaHotpot x → Spicy x
-- All products from Baked by Melissa are cupcakes.
axiom A4 : ∀ x : obj, FromBBM x → Cupcake x
-- Dried Thai chilies are spicy or mala hotpots or not baked sweets.
axiom A5 : Spicy DriedThaiChilies ∨ MalaHotpot DriedThaiChilies ∨ ¬ BakedSweet DriedThaiChilies

theorem dried_thai_neither_bbm_nor_baked : ¬ FromBBM DriedThaiChilies ∧ ¬ BakedSweet DriedThaiChilies := by
  -- First, show Dried Thai chilies are not baked sweets.
  have not_baked : ¬ BakedSweet DriedThaiChilies := by
    apply Or.elim A5
    · intro hS
      intro hBS
      have hNS : ¬ Spicy DriedThaiChilies := A1 DriedThaiChilies hBS
      exact hNS hS
    · intro h2
      apply Or.elim h2
      · intro hM
        intro hBS
        have hS : Spicy DriedThaiChilies := A3 DriedThaiChilies hM
        have hNS : ¬ Spicy DriedThaiChilies := A1 DriedThaiChilies hBS
        exact hNS hS
      · intro hNB
        exact hNB
  -- Then, use that to show they are not from Baked by Melissa.
  have not_bbm : ¬ FromBBM DriedThaiChilies := by
    intro hBBM
    have hCup : Cupcake DriedThaiChilies := A4 DriedThaiChilies hBBM
    have hBS : BakedSweet DriedThaiChilies := A2 DriedThaiChilies hCup
    exact not_baked hBS
  exact And.intro not_bbm not_baked
</lean>

ANSWER: True
