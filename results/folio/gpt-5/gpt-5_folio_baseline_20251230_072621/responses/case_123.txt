axiom obj : Type

axiom CanTransportMultiplePassengers : obj → Prop
axiom OneSeater : obj → Prop
axiom TeslaModel3 : obj → Prop
axiom SingleSeatElectricCar : obj → Prop

-- If something can transport multiple passengers, then it is not a one-seater.
axiom A1 : ∀ x : obj, CanTransportMultiplePassengers x → ¬ OneSeater x
-- All Tesla Model 3 can transport multiple passengers.
axiom A2 : ∀ x : obj, TeslaModel3 x → CanTransportMultiplePassengers x
-- All single-seat electric cars are one-seaters.
axiom A3 : ∀ x : obj, SingleSeatElectricCar x → OneSeater x
-- All Solo EV cars are single-seat electric cars.
axiom A4 : ∀ x : obj, SoloEV x → SingleSeatElectricCar x
-- If John's car is not a Tesla Model 3, then John's car is not a Tesla Model 3 or a single-seat electric car.
axiom A5 : ¬ TeslaModel3 JohnCar → (¬ TeslaModel3 JohnCar ∨ SingleSeatElectricCar JohnCar)

-- Derived: Tesla Model 3 are not one-seaters.
theorem tm3_not_oneseater : ∀ x : obj, TeslaModel3 x → ¬ OneSeater x := by
  intro x
  intro hTM3
  have hCMP : CanTransportMultiplePassengers x := A2 x hTM3
  exact A1 x hCMP

-- Derived: Solo EV are one-seaters.

  intro x
  intro hSEV
  have hSSEC : SingleSeatElectricCar x := A4 x hSEV
  exact A3 x hSSEC

-- They cannot be both Tesla Model 3 and Solo EV (for John's car).
theorem not_both_tm3_and_soloev_for_john : ¬ (TeslaModel3 JohnCar ∧ SoloEV JohnCar) := by
  intro h
  have hTM3 : TeslaModel3 JohnCar := And.left h
  have hSEV : SoloEV JohnCar := And.right h
  have hNotOne : ¬ OneSeater JohnCar := tm3_not_oneseater JohnCar hTM3
  have hOne : OneSeater JohnCar := soloev_oneseater JohnCar hSEV
  exact hNotOne hOne
