axiom obj : Type

-- Entities
axiom Peter : obj

-- Predicates
axiom Spills : obj → Prop
axiom Tidy : obj → Prop
axiom Cleanly : obj → Prop
axiom ValuesOrderSpotlessness : obj → Prop
axiom FamilyPrioritizesOrderSpotlessness : obj → Prop
axiom ClumsyFoodieWhoGoesOutFrequently : obj → Prop

-- Axioms from the context
axiom A1 : ∀ x : obj, Spills x → ¬ Tidy x
axiom A2 : ∀ x : obj, ClumsyFoodieWhoGoesOutFrequently x → Spills x
axiom A3 : ∀ x : obj, Cleanly x → Tidy x
axiom A4 : ∀ x : obj, ValuesOrderSpotlessness x → Cleanly x
axiom A5 : ∀ x : obj, FamilyPrioritizesOrderSpotlessness x → ValuesOrderSpotlessness x
axiom A6 : (Spills Peter ∧ Cleanly Peter) ∨ (¬ Spills Peter ∧ ¬ Cleanly Peter)

-- From A1 and A3, no one can be both spills and cleanly
theorem no_spills_and_cleanly_all : ∀ x : obj, ¬ (Spills x ∧ Cleanly x) := by
  intro x
  intro h
  have hSpill : Spills x := And.left h
  have hClean : Cleanly x := And.right h
  have hNotTidy : ¬ Tidy x := A1 x hSpill
  have hTidy : Tidy x := A3 x hClean
  exact hNotTidy hTidy

theorem peter_not_both : ¬ (Spills Peter ∧ Cleanly Peter) :=
  no_spills_and_cleanly_all Peter

-- Using A6 and peter_not_both, we deduce the second disjunct
theorem peter_not_spills_and_not_cleanly : (¬ Spills Peter ∧ ¬ Cleanly Peter) := by
  have hnotA : ¬ (Spills Peter ∧ Cleanly Peter) := no_spills_and_cleanly_all Peter
  cases A6 with
  | inl hA => exact False.elim (hnotA hA)
  | inr hB => exact hB
