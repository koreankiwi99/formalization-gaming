  fun x h => R4 x (R5 x (R1 x h))

-- Using the above, the first disjunct in Luke_disj is impossible, so we obtain the second disjunct.

by

    ¬ (Baby Luke ∧ MultipleBirth Luke ∧ LiveWithStrangers Luke) :=
  by
    intro h
    have hMB : MultipleBirth Luke := h.right.left
    have hStr : LiveWithStrangers Luke := h.right.right
    have hNotStr : ¬ LiveWithStrangers Luke := multipleBirth_no_strangers Luke hMB

  cases Luke_disj with
  | inl hA =>
