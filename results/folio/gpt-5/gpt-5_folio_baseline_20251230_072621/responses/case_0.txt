<lean>
axiom obj : Type

-- Predicates
axiom InClub : obj → Prop
axiom PerformOften : obj → Prop
axiom EngagedEvents : obj → Prop
axiom InactiveDisinterested : obj → Prop
axiom ChaperoneDances : obj → Prop
axiom StudentAtSchool : obj → Prop
axiom YoungChild : obj → Prop
axiom Teenager : obj → Prop
axiom WishFurtherAcademic : obj → Prop

-- Axioms from the context
-- People in this club who perform in school talent shows often attend and are very engaged with school events.
axiom A1 : ∀ x : obj, InClub x ∧ PerformOften x → EngagedEvents x

-- People in this club either perform in school talent shows often or are inactive and disinterested community members.
axiom A2 : ∀ x : obj, InClub x → (PerformOften x ∨ InactiveDisinterested x)

-- People in this club who chaperone high school dances are not students who attend the school.
axiom A3 : ∀ x : obj, InClub x ∧ ChaperoneDances x → ¬ StudentAtSchool x

-- All people in this club who are inactive and disinterested members of their community chaperone high school dances.
axiom A4 : ∀ x : obj, InClub x ∧ InactiveDisinterested x → ChaperoneDances x

-- All young children and teenagers in this club who wish to further their academic careers and educational opportunities are students who attend the school.
axiom A5 : ∀ x : obj, InClub x ∧ (YoungChild x ∨ Teenager x) ∧ WishFurtherAcademic x → StudentAtSchool x

-- Bonnie and her properties
axiom Bonnie : obj
axiom B1 : InClub Bonnie
-- Bonnie either (engaged with school events and is a student) or is not someone who both is engaged and not a student.
-- This formalizes as: (Engaged ∧ Student) ∨ ¬(Engaged ∧ ¬Student)
axiom B2 : (EngagedEvents Bonnie ∧ StudentAtSchool Bonnie) ∨ ¬(EngagedEvents Bonnie ∧ ¬ StudentAtSchool Bonnie)

-- Some derived lemmas showing the available reasoning

-- From B2, we can derive that Bonnie cannot be both engaged and not a student.
lemma bonnie_not_engaged_and_not_student : ¬ (EngagedEvents Bonnie ∧ ¬ StudentAtSchool Bonnie) := by
  cases B2 with
  | inl hES =>
      intro h
      exact h.right hES.right
  | inr hNot => 
      exact hNot

-- If Bonnie is inactive (and in the club), then she is not a student.
lemma inactive_implies_not_student (x : obj) (hc : InClub x) (hi : InactiveDisinterested x) : ¬ StudentAtSchool x := by
  have hch : ChaperoneDances x := A4 x ⟨hc, hi⟩
  exact A3 x ⟨hc, hch⟩

-- If Bonnie is not a student, then she is not engaged (using the previous lemma).
lemma bonnie_not_student_implies_not_engaged (hns : ¬ StudentAtSchool Bonnie) : ¬ EngagedEvents Bonnie := by
  intro hE
  exact bonnie_not_engaged_and_not_student ⟨hE, hns⟩

-- If Bonnie is in the club and not engaged, then she does not perform often.
lemma bonnie_not_engaged_implies_not_perform (hne : ¬ EngagedEvents Bonnie) : ¬ PerformOften Bonnie := by
  intro hp
  have hE : EngagedEvents Bonnie := A1 Bonnie ⟨B1, hp⟩
  exact hne hE

-- From the club disjunction, we can do case analysis: either Bonnie performs, or if she is inactive then she doesn't perform.
lemma bonnie_perform_or_not_perform : PerformOften Bonnie ∨ ¬ PerformOften Bonnie := by
  have h := A2 Bonnie B1
  cases h with
  | inl hp => exact Or.inl hp
  | inr hi =>
      have hns : ¬ StudentAtSchool Bonnie := inactive_implies_not_student Bonnie B1 hi
      have hne : ¬ EngagedEvents Bonnie := bonnie_not_student_implies_not_engaged hns
      have hnp : ¬ PerformOften Bonnie := bonnie_not_engaged_implies_not_perform hne
      exact Or.inr hnp

-- Cannot prove PerformOften Bonnie or its negation from the given axioms.
-- Therefore, the truth of "Bonnie performs in school talent shows often" is undecidable from the context.
</lean>
