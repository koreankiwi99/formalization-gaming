<lean>
axiom obj : Type
axiom Marvin : obj

axiom Alien : obj → Prop
axiom Extraterrestrial : obj → Prop
axiom Human : obj → Prop
axiom FromMars : obj → Prop
axiom FromEarth : obj → Prop
axiom HighlyIntelligent : obj → Prop

-- Axioms from the context
axiom A1 : ∀ x : obj, Alien x → Extraterrestrial x                 -- All aliens are extraterrestrials
axiom A2 : ∀ x : obj, FromMars x → Alien x                         -- From Mars ⇒ Alien
axiom A3 : ∀ x : obj, Extraterrestrial x → ¬ Human x               -- No extraterrestrials are human
axiom A4 : ∀ x : obj, HighlyIntelligent x ∧ FromEarth x → Human x  -- HI ∧ From Earth ⇒ Human
axiom A5 : HighlyIntelligent Marvin                                 -- Marvin is highly intelligent
axiom A6 : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬ FromEarth Marvin ∧ ¬ FromMars Marvin)
axiom A7 : ¬ FromEarth Marvin → Extraterrestrial Marvin             -- If Marvin not from Earth ⇒ Extraterrestrial

-- We prove the negation of the queried implication:
-- ¬ (¬ FromMars Marvin → Human Marvin)
theorem marvin_implication_false : ¬ (¬ FromMars Marvin → Human Marvin) := by
  -- First, show that the left disjunct of A6 leads to a contradiction
  have hNotEAndNotM : ¬ FromEarth Marvin ∧ ¬ FromMars Marvin := by
    cases A6 with
    | inl hEM =>
        -- From Earth ∧ From Mars leads to Human and Extraterrestrial, hence contradiction
        have hHuman : Human Marvin := A4 Marvin (And.intro A5 (And.left hEM))
        have hET : Extraterrestrial Marvin := A1 Marvin (A2 Marvin (And.right hEM))
        have hNotHuman : ¬ Human Marvin := A3 Marvin hET
        exact False.elim (hNotHuman hHuman)
    | inr hne =>
        exact hne
  have hNotE : ¬ FromEarth Marvin := And.left hNotEAndNotM
  have hNotM : ¬ FromMars Marvin := And.right hNotEAndNotM
  have hET2 : Extraterrestrial Marvin := A7 hNotE
  have hNotHuman2 : ¬ Human Marvin := A3 Marvin hET2
  intro hImp
  have hH : Human Marvin := hImp hNotM
  exact hNotHuman2 hH
</lean>

ANSWER: False
