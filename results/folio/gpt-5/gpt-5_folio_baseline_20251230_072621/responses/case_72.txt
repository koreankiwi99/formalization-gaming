<lean>
axiom obj : Type
axiom KO : obj

axiom Stock : obj → Prop
axiom Growth : obj → Prop
axiom BoughtRapid : obj → Prop
axiom Suitable : obj → Prop
axiom Mature : obj → Prop
axiom Volatile : obj → Prop

-- All growth stocks are bought to earn profits from rapid price appreciation.
axiom R1 : ∀ x : obj, Growth x → BoughtRapid x
-- If a stock is bought to earn profits from rapid price appreciation, then it is not suitable for a retirement fund.
axiom R2 : ∀ x : obj, BoughtRapid x → ¬ Suitable x
-- Some stocks are growth stocks.
axiom R3 : ∃ x : obj, Stock x ∧ Growth x
-- All mature stocks are suitable for a retirement fund.
axiom R4 : ∀ x : obj, Mature x → Suitable x

-- KO is a mature stock.
axiom T_stock_KO : Stock KO
axiom T_mature_KO : Mature KO

theorem target :
  (Growth KO ∨ BoughtRapid KO) → (¬ Stock KO ∧ ¬ Volatile KO) := by
  intro h
  have hSuitable : Suitable KO := R4 KO T_mature_KO
  have hFalse : False := by
    apply Or.elim h
    · intro hG
      have hB : BoughtRapid KO := R1 KO hG
      have hNotS : ¬ Suitable KO := R2 KO hB
      exact hNotS hSuitable
    · intro hB
      have hNotS : ¬ Suitable KO := R2 KO hB
      exact hNotS hSuitable
  have notStock : ¬ Stock KO := fun _ => False.elim hFalse
  have notVolatile : ¬ Volatile KO := fun _ => False.elim hFalse
  exact And.intro notStock notVolatile
</lean>

ANSWER: True
