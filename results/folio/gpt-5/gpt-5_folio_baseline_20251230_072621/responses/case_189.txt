<lean>
axiom obj : Type
axiom Yuri : obj

axiom Professional : obj → Prop
axiom American : obj → Prop
axiom CanJumpShots : obj → Prop
axiom LeapsStraight : obj → Prop
axiom ActivatesLegMuscles : obj → Prop

-- Some professional basketball players are not American nationals.
axiom some_pro_not_american : ∃ x : obj, Professional x ∧ ¬American x

-- All professional basketball players can do jump shots.
axiom pro_can_jump : ∀ x : obj, Professional x → CanJumpShots x

-- If someone can jump shots, they leap straight into the air.
axiom jump_implies_leap : ∀ x : obj, CanJumpShots x → LeapsStraight x

-- If someone leaps straight into the air, they activate their leg muscles.
axiom leap_implies_activate : ∀ x : obj, LeapsStraight x → ActivatesLegMuscles x

-- Yuri does not activate his leg muscles.
axiom yuri_not_activate : ¬ ActivatesLegMuscles Yuri

theorem yuri_not_american_pro : ¬ (American Yuri ∧ Professional Yuri) := by
  intro h
  have hpro : Professional Yuri := h.right
  have hjump : CanJumpShots Yuri := pro_can_jump Yuri hpro
  have hleap : LeapsStraight Yuri := jump_implies_leap Yuri hjump
  have hact : ActivatesLegMuscles Yuri := leap_implies_activate Yuri hleap
  exact yuri_not_activate hact
</lean>

ANSWER: False
