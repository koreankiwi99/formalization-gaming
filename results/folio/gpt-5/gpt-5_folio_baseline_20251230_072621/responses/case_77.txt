<lean>
axiom Person : Type
axiom Hannah : Person

axiom At : Person → Prop
axiom Work : Person → Prop
axiom Need : Person → Prop
axiom Order : Person → Prop
axiom Enjoy : Person → Prop
axiom Picky : Person → Prop
axiom Spend : Person → Prop

axiom R1 : ∀ x : Person, (At x ∧ Work x) → Need x
axiom R2 : ∀ x : Person, (At x ∧ Order x) → Work x
axiom R3 : ∀ x : Person, At x → (Order x ∨ Enjoy x)
axiom R4 : ∀ x : Person, Enjoy x → ¬Picky x
axiom R5 : ∀ x : Person, Enjoy x → Spend x

axiom T1 : At Hannah
axiom T2 : Work Hannah ∧ (Need Hannah → (¬Picky Hannah ∧ ¬Need Hannah))

theorem hannah_statement :
  At Hannah ∧ (¬Picky Hannah ∨ (Picky Hannah → Spend Hannah)) := by
  have hAt : At Hannah := T1
  have hOE : Order Hannah ∨ Enjoy Hannah := R3 Hannah hAt
  have hNotPicky : ¬Picky Hannah := by
    cases hOE with
    | inl hOrder =>
        have hWork : Work Hannah := R2 Hannah (And.intro hAt hOrder)
        have hNeed : Need Hannah := R1 Hannah (And.intro hAt hWork)
        have hImp : Need Hannah → (¬Picky Hannah ∧ ¬Need Hannah) := And.right T2
        have hPair : ¬Picky Hannah ∧ ¬Need Hannah := hImp hNeed
        exact And.left hPair
    | inr hEnjoy =>
        exact R4 Hannah hEnjoy
  exact And.intro hAt (Or.inl hNotPicky)
</lean>

ANSWER: True
