<lean>
axiom obj : Type
axiom Taylor : obj

axiom EatSaladsRegularly : obj → Prop
axiom VeryConsciousAboutHealth : obj → Prop
axiom GrewUpHealthConsciousHome : obj → Prop
axiom FulfillDailyNutritionalIntakes : obj → Prop
axiom DisregardPhysicalWellbeing : obj → Prop
axiom VisitGymOnceADay : obj → Prop

axiom A1 : ∀ x : obj, EatSaladsRegularly x → VeryConsciousAboutHealth x
axiom A2 : ∀ x : obj, GrewUpHealthConsciousHome x → EatSaladsRegularly x
axiom A3 : ∀ x : obj, FulfillDailyNutritionalIntakes x → GrewUpHealthConsciousHome x
axiom A4 : ∀ x : obj, DisregardPhysicalWellbeing x → ¬ VeryConsciousAboutHealth x
axiom A5 : ∀ x : obj, VisitGymOnceADay x → FulfillDailyNutritionalIntakes x
axiom A6 : (GrewUpHealthConsciousHome Taylor ∧ DisregardPhysicalWellbeing Taylor) ∨
            (¬ GrewUpHealthConsciousHome Taylor ∧ ¬ DisregardPhysicalWellbeing Taylor)

theorem taylor_neither_grewUp_nor_visitGym :
  ¬ GrewUpHealthConsciousHome Taylor ∧ ¬ VisitGymOnceADay Taylor := by
  have not_case1 : ¬ (GrewUpHealthConsciousHome Taylor ∧ DisregardPhysicalWellbeing Taylor) := by
    intro h
    have g : GrewUpHealthConsciousHome Taylor := h.left
    have d : DisregardPhysicalWellbeing Taylor := h.right
    have vc : VeryConsciousAboutHealth Taylor := A1 Taylor (A2 Taylor g)
    have nvc : ¬ VeryConsciousAboutHealth Taylor := A4 Taylor d
    exact nvc vc
  have h2 : (¬ GrewUpHealthConsciousHome Taylor ∧ ¬ DisregardPhysicalWellbeing Taylor) :=
    Or.resolve_left A6 not_case1
  have not_g : ¬ GrewUpHealthConsciousHome Taylor := h2.left
  have not_visit : ¬ VisitGymOnceADay Taylor := by
    intro v
    apply not_g
    have f : FulfillDailyNutritionalIntakes Taylor := A5 Taylor v
    have g2 : GrewUpHealthConsciousHome Taylor := A3 Taylor f
    exact g2
  exact And.intro not_g not_visit
</lean>

ANSWER: True
