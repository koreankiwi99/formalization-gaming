<lean>
axiom obj : Type
axiom Rabbit : obj → Prop
axiom Turtle : obj → Prop
axiom Squirrel : obj → Prop
axiom Spotted : obj → Prop
axiom Cute : obj → Prop
axiom Skittish : obj → Prop
axiom Calm : obj → Prop
axiom Rockie : obj

axiom A1 : ∀ x : obj, (Rabbit x ∧ Spotted x) → Cute x
axiom A2 : ∃ x : obj, Turtle x ∧ Spotted x
axiom A3 : ∀ x : obj, Spotted x → (Rabbit x ∨ Squirrel x)
axiom A4 : ∀ x : obj, Skittish x → ¬ Calm x
axiom A5 : ∀ x : obj, (Squirrel x ∧ Spotted x) → Skittish x
axiom A6 : Spotted Rockie ∧ Calm Rockie

theorem rockie_cute_or_skittish_of_not_turtle_and_squirrel :
  (¬ (Turtle Rockie ∧ Squirrel Rockie)) → (Cute Rockie ∨ Skittish Rockie) := by
  intro _
  have hS : Spotted Rockie := And.left A6
  have hC : Calm Rockie := And.right A6
  have notSk : ¬ Skittish Rockie := by
    intro hsk
    have hnotC : ¬ Calm Rockie := A4 Rockie hsk
    exact hnotC hC
  have rs : Rabbit Rockie ∨ Squirrel Rockie := A3 Rockie hS
  have notSq : ¬ Squirrel Rockie := by
    intro hsq
    have hsk : Skittish Rockie := A5 Rockie (And.intro hsq hS)
    exact notSk hsk
  have hRabbit : Rabbit Rockie :=
    Or.elim rs (fun hr => hr) (fun hs => False.elim (notSq hs))
  have hCute : Cute Rockie := A1 Rockie (And.intro hRabbit hS)
  exact Or.inl hCute
</lean>

ANSWER: True
