<lean>
axiom obj : Type
axiom Tom : obj

axiom GREApplicant : obj → Prop
axiom ProvesHardship : obj → Prop
axiom SingleParentFamily : obj → Prop
axiom FewResources : obj → Prop
axiom FinancialAid : obj → Prop

-- ETS provides financial aid to GRE applicants who prove economic hardship.
axiom A1 : ∀ x : obj, GREApplicant x → ProvesHardship x → FinancialAid x

-- Living in a single-parent family or having few resources suffices to prove hardship.
axiom A2 : ∀ x : obj, SingleParentFamily x → ProvesHardship x
axiom A3 : ∀ x : obj, FewResources x → ProvesHardship x

-- Facts about Tom
axiom T1 : SingleParentFamily Tom
axiom T2 : FewResources Tom
axiom T3 : GREApplicant Tom

theorem ets_financial_aid_Tom : FinancialAid Tom :=
  A1 Tom T3 (A2 Tom T1)
</lean>

ANSWER: True
