<lean>
axiom obj : Type
axiom James : obj

-- Predicates
axiom WorksLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom PartTimeJob : obj → Prop
axiom TakesDB : obj → Prop
axiom ClassWithDavid : obj → Prop
axiom WorksLab : obj → Prop

-- Axioms from the context
axiom A1 : ∀ x : obj, WorksLibrary x → ¬ FromCS x
axiom A2 : ∀ x : obj, PartTimeJob x → WorksLibrary x
axiom A3 : ∀ x : obj, TakesDB x → FromCS x
axiom A4 : ∀ x : obj, ClassWithDavid x → TakesDB x
axiom A5 : ∀ x : obj, WorksLab x → ClassWithDavid x
axiom A6 : WorksLab James
-- "James doesn't work in the lab or have a part-time job offered by the university."
-- Interpreted as: ¬ (WorksLab James ∨ PartTimeJob James)
axiom A7 : ¬ (WorksLab James ∨ PartTimeJob James)

-- We can derive that James is from CS via the lab → David → DB → CS chain
theorem james_from_cs : FromCS James := by
  have hLab : WorksLab James := A6
  have hDavid : ClassWithDavid James := A5 James hLab
  have hDB : TakesDB James := A4 James hDavid
  exact A3 James hDB

-- From PartTimeJob → WorksLibrary and WorksLibrary → ¬FromCS, together with FromCS James,
-- we conclude James does not have a part-time job.
theorem james_no_part_time_job : ¬ PartTimeJob James := by
  intro hPT
  have hLib : WorksLibrary James := A2 James hPT
  have hNotCS : ¬ FromCS James := A1 James hLib
  exact hNotCS james_from_cs
</lean>

ANSWER: False
