-- 6. Ekaterina can register to vote in the United States, or she is a Russian federation official.
axiom A6 : CanRegisterUSVote Ekaterina ∨ RussianOfficial Ekaterina

-- Some derivable facts about Vladimir (do not resolve official status)
lemma v_not_taiwanese : ¬ TaiwaneseCitizen Vladimir := And.left A5
lemma v_not_manager : ¬ ManagerAtGazprom Vladimir := And.right A5

lemma v_us_citizen : USCitizen Vladimir :=
by
  have h : USCitizen Vladimir ∨ TaiwaneseCitizen Vladimir := A3 Vladimir
  have hnt : ¬ TaiwaneseCitizen Vladimir := And.left A5
  cases h with
  | inl hu => exact hu

lemma v_can_participate : CanParticipate2024Election Vladimir := A1 Vladimir v_can_register
