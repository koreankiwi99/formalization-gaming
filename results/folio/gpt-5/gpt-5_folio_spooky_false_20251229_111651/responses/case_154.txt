theorem not_both_CC_and_One : ¬ (ChemicalCompound Mixture ∧ OnlyOneElement Mixture) :=
by
  intro h
  have hc : ChemicalCompound Mixture := h.left
  have ho : OnlyOneElement Mixture := h.right
  have hnot : ¬ OnlyOneElement Mixture := A5 Mixture hc
  exact hnot ho

theorem mixture_not_CC_and_not_One : ¬ChemicalCompound Mixture ∧ ¬OnlyOneElement Mixture :=
by
  cases A6 with
  | inl h =>
      exact False.elim (not_both_CC_and_One h)
  | inr h =>
      exact h
