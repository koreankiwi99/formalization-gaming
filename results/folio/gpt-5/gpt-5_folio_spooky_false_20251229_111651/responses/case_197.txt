<lean>
axiom obj : Type
axiom Jane : obj
axiom KiKi : obj
axiom Animal : obj → Prop
axiom Dog : obj → Prop
axiom Bark : obj → Prop
axiom Jump : obj → Prop
axiom Legs : obj → Prop
axiom Terricolous : obj → Prop
axiom Likes : obj → obj → Prop

-- Jane does not like any animals that bark.
axiom A1 : ∀ x : obj, (Animal x ∧ Bark x) → ¬Likes Jane x
-- All dogs are animals that bark.
axiom A2 : ∀ x : obj, Dog x → (Animal x ∧ Bark x)
-- Jane likes all animals that jump.
axiom A3 : ∀ x : obj, (Animal x ∧ Jump x) → Likes Jane x
-- If an animal has legs, then it jumps.
axiom A4 : ∀ x : obj, (Animal x ∧ Legs x) → Jump x
-- If an animal is terricolous, then it has legs.
axiom A5 : ∀ x : obj, (Animal x ∧ Terricolous x) → Legs x
-- KiKi is an animal.
axiom A6 : Animal KiKi
-- If KiKi neither jumps nor has legs, then KiKi is terricolous or has legs.
axiom A7 : (¬Jump KiKi ∧ ¬Legs KiKi) → (Terricolous KiKi ∨ Legs KiKi)

-- A useful consequence: it's impossible that KiKi both doesn't jump and doesn't have legs.
theorem kiki_not_both_nojump_nolegs : ¬(¬Jump KiKi ∧ ¬Legs KiKi) :=
by
  intro h
  have hnotJ : ¬Jump KiKi := And.left h
  have hnotL : ¬Legs KiKi := And.right h
  have hT_or_L : Terricolous KiKi ∨ Legs KiKi := A7 h
  cases hT_or_L with
  | inl hT =>
      have hL : Legs KiKi := A5 KiKi (And.intro A6 hT)
      exact hnotL hL
  | inr hL =>
      exact hnotL hL

-- From the like/dislike rules, any jumping animal cannot be barking.
theorem animal_jump_implies_not_bark (x : obj) : Animal x → Jump x → ¬Bark x :=
by
  intro hA hJ
  intro hB
  have hLike : Likes Jane x := A3 x (And.intro hA hJ)
  have hNotLike : ¬Likes Jane x := A1 x (And.intro hA hB)
  exact hNotLike hLike

-- Dogs bark.
theorem dog_implies_bark (x : obj) : Dog x → Bark x :=
by
  intro hD
  have hAandB : Animal x ∧ Bark x := A2 x hD
  exact And.right hAandB
