<lean>
axiom Rest : Type
axiom HamdenPlaza : Rest

axiom Listed : Rest → Prop
axiom ManyNeg : Rest → Prop
axiom RatingGt4 : Rest → Prop
axiom NoTakeout : Rest → Prop
axiom Popular : Rest → Prop

-- If a restaurant is listed, it has not received many negative reviews.
axiom A1 : ∀ r : Rest, Listed r → ¬ ManyNeg r
-- All restaurants with rating > 4 are listed.
axiom A2 : ∀ r : Rest, RatingGt4 r → Listed r
-- Some restaurants without take-out receive many negative reviews.
axiom A3 : ∃ r : Rest, NoTakeout r ∧ ManyNeg r
-- All popular restaurants have rating > 4.
axiom A4 : ∀ r : Rest, Popular r → RatingGt4 r
-- The Hamden Plaza store has rating > 4 or is popular.
axiom A5 : RatingGt4 HamdenPlaza ∨ Popular HamdenPlaza

-- From the premises, we can deduce the Hamden Plaza store has not received many negative reviews.
theorem HamdenPlaza_not_many_neg : ¬ ManyNeg HamdenPlaza := by
  have hR : RatingGt4 HamdenPlaza := by
    cases A5 with
    | inl h => exact h
    | inr hp => exact A4 HamdenPlaza hp
  have hL : Listed HamdenPlaza := A2 HamdenPlaza hR
  exact A1 HamdenPlaza hL
