axiom LeapsStraightUp : obj → Prop

-- If someone can jump shots, they leap straight into the air.

-- If someone leaps straight into the air, they activate their leg muscles.
axiom Prem4 : ∀ x : obj, LeapsStraightUp x → ActivatesLegMuscles x
-- Yuri does not activate his leg muscles.
axiom Prem5 : ¬ActivatesLegMuscles Yuri

-- From Yuri not activating leg muscles and the universal implications,
-- we can conclude Yuri is not a professional basketball player.

  have hLS : LeapsStraightUp Yuri := Prem3 Yuri hCJ
