<lean>
axiom Person : Type
axiom Book : Type

axiom Reads : Person → Book → Prop
axiom GainsKnowledge : Person → Prop
axiom Smarter : Person → Prop

-- Premises
axiom read_implies_gains : ∀ p : Person, ∀ b : Book, Reads p b → GainsKnowledge p
axiom gains_implies_smarter : ∀ p : Person, GainsKnowledge p → Smarter p

-- Facts
axiom Harry : Person
axiom Walden : Book
axiom harry_read_walden : Reads Harry Walden

theorem harry_gains : GainsKnowledge Harry := by
  exact read_implies_gains Harry Walden harry_read_walden

theorem harry_smarter : Smarter Harry := by
  exact gains_implies_smarter Harry harry_gains

theorem exists_smarter_with_knowledge : ∃ p : Person, Smarter p ∧ GainsKnowledge p := by
  refine Exists.intro Harry ?_
  apply And.intro
  · exact harry_smarter
  · exact harry_gains
