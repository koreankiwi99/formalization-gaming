<lean>
axiom Event : Type
axiom Happy : Event → Prop
axiom Sad : Event → Prop

-- "Events are either happy or sad" interpreted as an exclusive, exhaustive dichotomy
axiom either_or : ∀ e : Event, Happy e ↔ ¬Sad e

-- "At least one event is happy"
axiom exists_happy : ∃ e : Event, Happy e

-- Negation of "All events are sad"
theorem not_all_events_are_sad : ¬ (∀ e : Event, Sad e) := by
  intro h_all_sad
  have ⟨e0, he0⟩ := exists_happy
  have h_not_sad : ¬Sad e0 := (Iff.mp (either_or e0)) he0
  exact h_not_sad (h_all_sad e0)
</lean>
