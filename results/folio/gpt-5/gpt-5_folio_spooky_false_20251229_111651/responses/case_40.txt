<lean>
axiom obj : Type

axiom ByMax : obj → Prop
axiom Adores : obj → Prop
axiom Zaha : obj → Prop
axiom Kelly : obj → Prop
axiom Brutalist : obj → Prop
axiom Interesting : obj → Prop
axiom Evocative : obj → Prop
axiom Dreamy : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, Zaha x ∧ Adores x → Interesting x
axiom P2 : ∀ x : obj, Brutalist x ∧ Adores x → ¬Interesting x
axiom P3 : ∀ x : obj, Adores x → Zaha x ∨ Kelly x
axiom P4 : ∀ x : obj, Kelly x ∧ Adores x → Evocative x
axiom P5 : ∀ x : obj, Kelly x ∧ Adores x → Dreamy x
axiom P6 : ∀ x : obj, ByMax x ∧ Adores x ∧ Interesting x → Brutalist x ∧ Evocative x

-- Useful derived consequences

-- No design by Max that he adores can be interesting.
theorem no_ByMax_adores_interesting : ∀ x : obj, ByMax x ∧ Adores x ∧ Interesting x → False := by
  intro x
  intro h
  cases h with
  | intro hbm h2 =>
    cases h2 with
    | intro had hi =>
      have hbe : Brutalist x ∧ Evocative x := P6 x (And.intro hbm (And.intro had hi))
      have hb : Brutalist x := hbe.left
      have hnot : ¬Interesting x := P2 x (And.intro hb had)
      exact hnot hi

-- No Zaha-designed style by Max that he adores exists.
theorem no_ByMax_adores_Zaha : ∀ x : obj, ByMax x ∧ Adores x ∧ Zaha x → False := by
  intro x
  intro h
  cases h with
  | intro hbm h2 =>
    cases h2 with
    | intro had hz =>
      have hi : Interesting x := P1 x (And.intro hz had)
      exact no_ByMax_adores_interesting x (And.intro hbm (And.intro had hi))

-- Any design by Max that he adores must be Kelly (not Zaha).
theorem ByMax_adores_implies_Kelly : ∀ x : obj, ByMax x ∧ Adores x → Kelly x := by
  intro x
  intro h
  cases h with
  | intro hbm had =>
    have hz_or_hk : Zaha x ∨ Kelly x := P3 x had
    cases hz_or_hk with
    | inl hz =>
      have f : False := no_ByMax_adores_Zaha x (And.intro hbm (And.intro had hz))
      exact False.elim f
    | inr hk =>
      exact hk

-- Therefore, any design by Max that he adores is evocative and dreamy.
theorem ByMax_adores_evocative_and_dreamy : ∀ x : obj, ByMax x ∧ Adores x → Evocative x ∧ Dreamy x := by
  intro x
  intro h
  cases h with
  | intro hbm had =>
    have hk : Kelly x := ByMax_adores_implies_Kelly x (And.intro hbm had)
    have hev : Evocative x := P4 x (And.intro hk had)
    have hdr : Dreamy x := P5 x (And.intro hk had)
    exact And.intro hev hdr
