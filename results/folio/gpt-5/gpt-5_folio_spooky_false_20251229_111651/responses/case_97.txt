axiom Company : obj → Prop
axiom PrivateCompany : obj → Prop
axiom LegalEntity : obj → Prop
axiom CreatedUnderLaw : obj → Prop
axiom HasLegalObligations : obj → Prop

axiom AllBusinessAreLegal : ∀ x : obj, BusinessOrganization x → LegalEntity x
axiom AllCompaniesAreBusiness : ∀ x : obj, Company x → BusinessOrganization x
axiom AllPrivateAreCompanies : ∀ x : obj, PrivateCompany x → Company x
axiom AllLegalCreated : ∀ x : obj, LegalEntity x → CreatedUnderLaw x
axiom AllLegalHaveObligations : ∀ x : obj, LegalEntity x → HasLegalObligations x
axiom ClubCreatedImpliesNotPrivate : CreatedUnderLaw H → ¬ PrivateCompany H

-- The statement to evaluate is provable from the premises.
theorem statement_true :
  PrivateCompany H → (HasLegalObligations H ∨ CreatedUnderLaw H) := by
  intro hpc
  have hc : Company H := AllPrivateAreCompanies H hpc
  have hbo : BusinessOrganization H := AllCompaniesAreBusiness H hc
  have hle : LegalEntity H := AllBusinessAreLegal H hbo
  have hobl : HasLegalObligations H := AllLegalHaveObligations H hle
  exact Or.inl hobl
