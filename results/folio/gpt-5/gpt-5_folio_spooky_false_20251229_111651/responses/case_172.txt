axiom obj : Type
axiom Taylor : obj

axiom EatsSalads : obj → Prop
axiom VeryConscious : obj → Prop
axiom GrewUpHCH : obj → Prop
axiom FulfillsDailyIntake : obj → Prop
axiom DisregardsWellBeing : obj → Prop
axiom VisitsGymDaily : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, EatsSalads x → VeryConscious x
axiom P2 : ∀ x : obj, GrewUpHCH x → EatsSalads x
axiom P3 : ∀ x : obj, FulfillsDailyIntake x → GrewUpHCH x
axiom P4 : ∀ x : obj, DisregardsWellBeing x → ¬VeryConscious x
axiom P5 : ∀ x : obj, VisitsGymDaily x → FulfillsDailyIntake x
axiom Taylor_either_or :
  (GrewUpHCH Taylor ∧ DisregardsWellBeing Taylor) ∨
  (¬GrewUpHCH Taylor ∧ ¬DisregardsWellBeing Taylor)

-- Useful derived lemma: Growing up in a health-conscious home implies being very conscious
theorem grew_up_implies_very : ∀ x : obj, GrewUpHCH x → VeryConscious x := by
  intro x
  intro hx
  have hs : EatsSalads x := P2 x hx
  have hv : VeryConscious x := P1 x hs
  exact hv

-- From the disjunction and the incompatibility of VeryConscious with DisregardsWellBeing,
-- we can derive that Taylor did neither.
theorem taylor_not_grewup_and_not_disregard : ¬GrewUpHCH Taylor ∧ ¬DisregardsWellBeing Taylor := by
  have h := Taylor_either_or
  cases h with
  | inl hgd =>
    cases hgd with
    | intro hg hd =>
      have hv : VeryConscious Taylor := grew_up_implies_very Taylor hg
      have hnv : ¬VeryConscious Taylor := P4 Taylor hd
      have f : False := hnv hv
      exact False.elim f
  | inr hneither =>
    exact hneither
