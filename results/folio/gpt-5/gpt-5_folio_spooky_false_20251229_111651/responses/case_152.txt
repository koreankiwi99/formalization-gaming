axiom Organic : obj → Prop

axiom ContainsCarbon : obj → Prop
axiom OnlyOneElement : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, Hydrocarbon x → Organic x
axiom A2 : ∀ x : obj, Alkane x → Hydrocarbon x
axiom A3 : ∀ x : obj, Organic x → Chemical x
axiom A4 : ∀ x : obj, Organic x → ContainsCarbon x
axiom A5 : ∀ x : obj, Chemical x → ¬OnlyOneElement x
axiom A6 : (Chemical Mixture ∧ OnlyOneElement Mixture) ∨ (¬Chemical Mixture ∧ ¬OnlyOneElement Mixture)

-- Consequences we can derive about the mixture
theorem mixture_not_chemical : ¬Chemical Mixture :=
by
  intro hc
  have hnoe : ¬OnlyOneElement Mixture := A5 Mixture hc
  have hdisj := A6
  cases hdisj with
  | inl h =>
      exact hnoe h.right
  | inr h =>
      exact h.left hc

theorem mixture_not_one_element : ¬OnlyOneElement Mixture :=
by
  have hdisj := A6
  cases hdisj with
  | inl h =>
      have hnoe : ¬OnlyOneElement Mixture := A5 Mixture h.left
      exact hnoe h.right
  | inr h =>
      exact h.right

theorem mixture_not_organic : ¬Organic Mixture :=
by
  intro ho
  apply mixture_not_chemical
  exact A3 Mixture ho
