<lean>
axiom obj : Type

-- Entities
axiom Hachi : obj

-- Predicates
axiom Film : obj → Prop
axiom AppropriateAllAges : obj → Prop
axiom CanWatchWithoutGuidance : obj → Prop
axiom ExcessiveEroticViolent : obj → Prop
axiom RatedGA : obj → Prop
axiom FamilyFriendlyAnimated : obj → Prop
axiom FrozenSeries : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, AppropriateAllAges x → CanWatchWithoutGuidance x
axiom P2 : ∀ x : obj, ExcessiveEroticViolent x → ¬ CanWatchWithoutGuidance x
axiom P3 : ∀ x : obj, RatedGA x → AppropriateAllAges x
axiom P4 : ∀ x : obj, FamilyFriendlyAnimated x → RatedGA x
axiom P5 : ∀ x : obj, FrozenSeries x → FamilyFriendlyAnimated x
axiom P6 : Film Hachi
axiom P7 : FamilyFriendlyAnimated Hachi ∨ AppropriateAllAges Hachi

-- Useful consequences from the premises
theorem hachi_can_watch : CanWatchWithoutGuidance Hachi := by
  cases P7 with
  | inl hF =>
      have hG : RatedGA Hachi := P4 Hachi hF
      have hA : AppropriateAllAges Hachi := P3 Hachi hG
      exact P1 Hachi hA
  | inr hA =>
      exact P1 Hachi hA

theorem hachi_not_excessive : ¬ExcessiveEroticViolent Hachi := by
  intro hE
  have hW : CanWatchWithoutGuidance Hachi := hachi_can_watch
  have hNotW : ¬CanWatchWithoutGuidance Hachi := P2 Hachi hE
  exact hNotW hW

theorem not_excessive_and_frozen_hachi : ¬(ExcessiveEroticViolent Hachi ∧ FrozenSeries Hachi) := by
  intro h
  have hE : ExcessiveEroticViolent Hachi := h.left
  have hZ : FrozenSeries Hachi := h.right
  have hF : FamilyFriendlyAnimated Hachi := P5 Hachi hZ
  have hG : RatedGA Hachi := P4 Hachi hF
  have hA : AppropriateAllAges Hachi := P3 Hachi hG
  have hW : CanWatchWithoutGuidance Hachi := P1 Hachi hA
  have hNotW : ¬CanWatchWithoutGuidance Hachi := P2 Hachi hE
  exact hNotW hW
