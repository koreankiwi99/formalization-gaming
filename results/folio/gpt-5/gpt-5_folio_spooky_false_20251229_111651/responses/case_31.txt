axiom obj : Type
axiom Djokovic : obj

axiom AtMixer : obj → Prop
axiom GrandSlamChampion : obj → Prop
axiom OscarNominatedActor : obj → Prop
axiom ProTennisPlayer : obj → Prop
axiom Celebrity : obj → Prop
axiom Athlete : obj → Prop
axiom Famous : obj → Prop
axiom WellPaid : obj → Prop
axiom LivesInTaxHaven : obj → Prop

axiom A1 : ∀ x : obj, AtMixer x → (GrandSlamChampion x ∨ OscarNominatedActor x)
axiom A2 : ∀ x : obj, AtMixer x → GrandSlamChampion x → ProTennisPlayer x
axiom A3 : ∀ x : obj, AtMixer x → OscarNominatedActor x → Celebrity x
axiom A4 : ∀ x : obj, AtMixer x → ProTennisPlayer x → Athlete x
axiom A5 : ∀ x : obj, AtMixer x → Celebrity x → WellPaid x
axiom A6 : ∀ x : obj, AtMixer x → Athlete x → Famous x
axiom A7 : ∀ x : obj, AtMixer x → WellPaid x → LivesInTaxHaven x
axiom A8 : AtMixer Djokovic
axiom A9 : Athlete Djokovic → Famous Djokovic → WellPaid Djokovic

theorem djokovic_tax_haven : LivesInTaxHaven Djokovic := by
  have dj_at : AtMixer Djokovic := A8
  have dj_cats : GrandSlamChampion Djokovic ∨ OscarNominatedActor Djokovic := A1 Djokovic dj_at
  cases dj_cats with
  | inl dj_gs =>
      have dj_pt : ProTennisPlayer Djokovic := A2 Djokovic dj_at dj_gs
      have dj_ath : Athlete Djokovic := A4 Djokovic dj_at dj_pt
      have dj_fam : Famous Djokovic := A6 Djokovic dj_at dj_ath
      have dj_wp : WellPaid Djokovic := A9 dj_ath dj_fam
      exact A7 Djokovic dj_at dj_wp
  | inr dj_ona =>
      have dj_celeb : Celebrity Djokovic := A3 Djokovic dj_at dj_ona
      have dj_wp : WellPaid Djokovic := A5 Djokovic dj_at dj_celeb
      exact A7 Djokovic dj_at dj_wp
