axiom Heck : obj → Prop
axiom Aurochs : obj → Prop
axiom Extinct : obj → Prop
axiom BredBack : obj → Prop
axiom ToBeBredBack : obj → Prop
axiom ArtificiallySelected : obj → Prop
axiom Resembles : obj → obj → Prop

-- Breeding back is a form of artificial selection
axiom AX1 : ∀ x : obj, BredBack x → ArtificiallySelected x

-- Heck cattle were bred back
axiom AX2 : ∃ x : obj, Heck x ∧ BredBack x

-- Heck cattle are animals
axiom AX3 : ∀ x : obj, Heck x → Animal x

-- Aurochs are animals
axiom AX4 : ∀ x : obj, Aurochs x → Animal x

-- Some animals to be bred back resemble extinct animals
axiom AX5 : ∃ x y : obj, Animal x ∧ ToBeBredBack x ∧ Animal y ∧ Extinct y ∧ Resembles x y

-- From the premises, we can derive the statement: Some Heck cattle are artificially selected.
theorem some_heck_artificial : ∃ x : obj, Heck x ∧ ArtificiallySelected x := by
  rcases AX2 with ⟨x, hx⟩
  rcases hx with ⟨hxHeck, hxBred⟩
  have hxArt : ArtificiallySelected x := AX1 x hxBred
  exact ⟨x, And.intro hxHeck hxArt⟩

-- Therefore, attempting to prove the negation leads to a contradiction if assumed.
theorem negation_impossible : (¬ ∃ x : obj, Heck x ∧ ArtificiallySelected x) → False := by
  intro hneg
  apply hneg
  exact some_heck_artificial
