<lean>
axiom obj : Type
axiom John : obj

axiom Engaged : obj → Prop
axiom MakesPlans : obj → Prop
axiom InvitesOthers : obj → Prop
axiom InvitesFriends : obj → Prop
axiom WellAttended : obj → Prop
axiom LargerFamily : obj → Prop

-- 1. All who make wedding plans are engaged.
axiom A1 : ∀ x : obj, MakesPlans x → Engaged x
-- 2. All who invite others to their ceremony make wedding plans.
axiom A2 : ∀ x : obj, InvitesOthers x → MakesPlans x
-- 3. Anyone with a well-attended wedding invites others to their ceremony.
axiom A3 : ∀ x : obj, WellAttended x → InvitesOthers x
-- 4. Some people with well-attended weddings have larger families.
axiom A4 : ∃ x : obj, WellAttended x ∧ LargerFamily x
-- 5. If John is engaged, then John does not invite friends to their ceremony and does not make wedding plans.
axiom A5 : Engaged John → (¬InvitesFriends John ∧ ¬MakesPlans John)
-- 6. If John has a larger family, then either John has a well-attended wedding or invites friends to their ceremony.
axiom A6 : LargerFamily John → (WellAttended John ∨ InvitesFriends John)
-- Subtle link: inviting friends implies inviting others.
axiom A7 : ∀ x : obj, InvitesFriends x → InvitesOthers x

theorem john_not_larger_family : ¬LargerFamily John := by
  intro hlf
  have hdisj := A6 hlf
  cases hdisj with
  | inl hwaw =>
    have hio : InvitesOthers John := A3 John hwaw
    have hmp : MakesPlans John := A2 John hio
    have heng : Engaged John := A1 John hmp
    have hpair := A5 heng
    have hnot_mp : ¬MakesPlans John := And.right hpair
    exact hnot_mp hmp
  | inr hif =>
    have hio : InvitesOthers John := A7 John hif
    have hmp : MakesPlans John := A2 John hio
    have heng : Engaged John := A1 John hmp
    have hnot_if : ¬InvitesFriends John := And.left (A5 heng)
    exact hnot_if hif
</lean>

ANSWER: False
