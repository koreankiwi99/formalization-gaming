<lean>
axiom obj : Type

-- Entities
axiom Tom : obj
axiom Fluffy : obj
axiom OliveGarden : obj

-- Predicates and relations
axiom Managed : obj → Prop
axiom Pet : obj → Prop
axiom Cat : obj → Prop
axiom AllowsPets : obj → Prop
axiom AllowedMoveInWith : obj → obj → obj → Prop  -- person, building, pet
axiom WillRent : obj → obj → Prop                 -- person, building

-- Numeric attributes
axiom Rent : obj → Nat
axiom SecurityDeposit : obj → Nat

-- Premises
-- Pets are allowed in some managed buildings.
axiom some_managed_allow_pets : ∃ b : obj, Managed b ∧ AllowsPets b

-- A deposit is required to rent an apartment in a managed building. (modeled implicitly by having SecurityDeposit)
-- The security deposit can be either equal to the monthly rent at a managed building or more.
axiom deposit_at_least_rent : ∀ b : obj, Managed b → Rent b ≤ SecurityDeposit b

-- Fluffy is Tom's cat.
axiom fluffy_is_cat : Cat Fluffy

-- Cats are pets.
axiom cats_are_pets : ∀ x : obj, Cat x → Pet x

-- The Olive Garden is a managed building.
axiom olive_is_managed : Managed OliveGarden

-- The monthly rent at the Olive Garden is $2000.
axiom rent_olive_is_2000 : Rent OliveGarden = 2000

-- $2000 is more than $1500.
axiom two_thousand_gt_one_thousand_five_hundred : 1500 < 2000

-- Tom will rent an apartment in a managed building if he is allowed to move in with Fluffy, and the security deposit is no more than $1500.
axiom tom_rents_if_allowed_and_deposit_le_1500 :
  ∀ b : obj, Managed b → AllowedMoveInWith Tom b Fluffy → SecurityDeposit b ≤ 1500 → WillRent Tom b

-- If a managed building allows pets, then people are allowed to move in with a pet.
axiom allowed_if_pets_allowed :
  ∀ (b per pet : obj), Managed b → AllowsPets b → Pet pet → AllowedMoveInWith per b pet

-- A small derived fact: Fluffy is a pet.
theorem fluffy_is_pet : Pet Fluffy := cats_are_pets Fluffy fluffy_is_cat
