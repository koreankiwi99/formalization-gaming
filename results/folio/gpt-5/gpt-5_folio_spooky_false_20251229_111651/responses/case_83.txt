<lean>
axiom obj : Type
axiom Hamden : obj

axiom Listed : obj → Prop
axiom NegReviews : obj → Prop
axiom RatingGt4 : obj → Prop
axiom TakeOut : obj → Prop
axiom Popular : obj → Prop

-- Premises
axiom A1 : ∀ x : obj, Listed x → ¬NegReviews x
axiom A2 : ∀ x : obj, RatingGt4 x → Listed x
axiom A3 : ∃ x : obj, ¬TakeOut x ∧ NegReviews x
axiom A4 : ∀ x : obj, Popular x → RatingGt4 x
axiom A5 : RatingGt4 Hamden ∨ Popular Hamden

theorem hamden_rating_gt4 : RatingGt4 Hamden := by
  cases A5 with
  | inl hr => exact hr
  | inr hp =>
    have hr := A4 Hamden hp
    exact hr

theorem hamden_not_neg : ¬NegReviews Hamden := by
  have l : Listed Hamden := A2 Hamden hamden_rating_gt4
  exact A1 Hamden l

-- Target statement:
-- If Hamden provides take-out and receives many negative reviews,
-- then its rating is greater than 4 and it does not provide take-out service.
theorem statement_holds :
  (TakeOut Hamden ∧ NegReviews Hamden) → (RatingGt4 Hamden ∧ ¬TakeOut Hamden) := by
  intro h
  have f : False := hamden_not_neg h.right
  have nTo : ¬TakeOut Hamden := False.elim f
  exact And.intro hamden_rating_gt4 nTo
</lean>
