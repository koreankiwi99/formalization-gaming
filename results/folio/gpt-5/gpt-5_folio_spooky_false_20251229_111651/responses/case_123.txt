axiom obj : Type
axiom JohnCar : obj

-- Predicates
axiom TransportMultiple : obj → Prop
axiom OneSeater : obj → Prop
axiom TeslaModel3 : obj → Prop
axiom SingleSeatElectric : obj → Prop
axiom SoloEV : obj → Prop

-- Premises
-- 1. If something can transport multiple passengers, then they are not one-seaters.
axiom A1 : ∀ x : obj, TransportMultiple x → ¬ OneSeater x
-- 2. All Tesla Model 3 can transport multiple passengers.
axiom A2 : ∀ x : obj, TeslaModel3 x → TransportMultiple x
-- 3. All single-seat electric cars are one-seaters.
axiom A3 : ∀ x : obj, SingleSeatElectric x → OneSeater x
-- 4. All Solo EV cars are single-seat electric cars.
axiom A4 : ∀ x : obj, SoloEV x → SingleSeatElectric x
-- 5. If John's car is not a Tesla Model 3, then John's car is not a Tesla Model 3 or a single-seat electric car.
axiom A5 : ¬ TeslaModel3 JohnCar → (¬ TeslaModel3 JohnCar ∨ SingleSeatElectric JohnCar)

-- Some derived lemmas
theorem tesla_not_one : ∀ x : obj, TeslaModel3 x → ¬ OneSeater x := by
  intro x hT3
  have hTM : TransportMultiple x := A2 x hT3
  exact A1 x hTM

theorem soloEV_implies_one : ∀ x : obj, SoloEV x → OneSeater x := by
  intro x hSEV
  have hSSE : SingleSeatElectric x := A4 x hSEV
  exact A3 x hSSE

theorem tesla_implies_not_solo : ∀ x : obj, TeslaModel3 x → ¬ SoloEV x := by
  intro x hT3
  intro hSEV
  have hOne : OneSeater x := soloEV_implies_one x hSEV
  have hNotOne : ¬ OneSeater x := tesla_not_one x hT3
  exact hNotOne hOne
