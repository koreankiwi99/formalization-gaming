axiom obj : Type

-- Entities
axiom Harry : obj

-- Predicates
axiom Animal : obj → Prop
axiom Invertebrate : obj → Prop
axiom Vertebrate : obj → Prop
axiom Backbone : obj → Prop
axiom ReproMF : obj → Prop
axiom Bee : obj → Prop
axiom QueenBee : obj → Prop

-- Premises
-- Animals are either invertebrates or vertebrates.
axiom A1 : ∀ x : obj, Animal x → (Invertebrate x ∨ Vertebrate x)
-- All animals with backbones reproduce by male-and-female mating.
axiom A2 : ∀ x : obj, Animal x → Backbone x → ReproMF x
-- All vertebrate animals have a backbone.
axiom A3 : ∀ x : obj, Animal x → Vertebrate x → Backbone x
-- All bees do not reproduce by male-and-female mating.
axiom A4 : ∀ x : obj, Bee x → ¬ ReproMF x
-- All queen bees are bees.
axiom A5 : ∀ x : obj, QueenBee x → Bee x
-- Harry is a bee.
axiom A6 : Bee Harry

-- A useful derived lemma: vertebrate animals reproduce by male-and-female mating.
theorem vertebrate_animals_repro : ∀ x : obj, Animal x → Vertebrate x → ReproMF x := by
  intro x hA hV
  have hB : Backbone x := A3 x hA hV
  exact A2 x hA hB

-- From Harry being a bee (hence not reproducing MF), we can show he cannot be both an animal and vertebrate.
theorem not_animal_and_vertebrate_Harry : ¬ (Animal Harry ∧ Vertebrate Harry) := by
  intro h
  have hA : Animal Harry := And.left h
  have hV : Vertebrate Harry := And.right h
  have hRepro : ReproMF Harry := vertebrate_animals_repro Harry hA hV
  have hN : ¬ ReproMF Harry := A4 Harry A6
  exact hN hRepro
