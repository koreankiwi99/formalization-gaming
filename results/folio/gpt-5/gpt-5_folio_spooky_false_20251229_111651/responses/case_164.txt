<lean>
axiom obj : Type
axiom Show : obj → Prop
axiom BreakingBad : obj
axiom RealityTV : obj → Prop
axiom PortraysRealPeople : obj → Prop
-- Predicate meaning: x is an episode of The Bachelor
axiom Bachelor : obj → Prop
axiom GivesRosesOnTV : obj → Prop
axiom AirsMon8pm : obj → Prop

-- Some show airing at 8 pm on Monday gives out roses on TV.
axiom A1 : ∃ s : obj, Show s ∧ AirsMon8pm s ∧ GivesRosesOnTV s
-- If a show gives out roses on TV, then the show is an episode of The Bachelor.
axiom A2 : ∀ x : obj, GivesRosesOnTV x → Bachelor x
-- The Bachelor portrays the lives of real people (for any episode of The Bachelor).
axiom A3 : ∀ x : obj, Bachelor x → PortraysRealPeople x
-- All shows portraying the lives of real people are reality TV shows.
axiom A4 : ∀ x : obj, PortraysRealPeople x → RealityTV x
-- Breaking Bad is a show.
axiom BB_show : Show BreakingBad
-- Breaking Bad is not a reality TV show.
axiom BB_not_reality : ¬RealityTV BreakingBad

theorem roses_implies_reality : ∀ x : obj, GivesRosesOnTV x → RealityTV x := by
  intro x
  intro hx
  have hb : Bachelor x := A2 x hx
  have hp : PortraysRealPeople x := A3 x hb
  exact A4 x hp

theorem not_roses_BB : ¬GivesRosesOnTV BreakingBad := by
  intro hgr
  have hrtv : RealityTV BreakingBad := roses_implies_reality BreakingBad hgr
  exact BB_not_reality hrtv

theorem negate_statement :
  ¬(Show BreakingBad ∧ AirsMon8pm BreakingBad ∧ GivesRosesOnTV BreakingBad) := by
  intro h
  have : GivesRosesOnTV BreakingBad := And.right (And.right h)
  exact not_roses_BB this
</lean>
