<lean>
axiom obj : Type
axiom Marvin : obj
axiom Alien : obj → Prop
axiom Extraterrestrial : obj → Prop
axiom Human : obj → Prop
axiom FromMars : obj → Prop
axiom FromEarth : obj → Prop
axiom Intelligent : obj → Prop

axiom A1 : ∀ x : obj, Alien x → Extraterrestrial x
axiom A2 : ∀ x : obj, FromMars x → Alien x
axiom A3 : ∀ x : obj, Extraterrestrial x → ¬Human x
axiom A4 : ∀ x : obj, (Intelligent x ∧ FromEarth x) → Human x
axiom A5 : Intelligent Marvin
axiom A6 : (FromEarth Marvin ∧ FromMars Marvin) ∨ (¬FromEarth Marvin ∧ ¬FromMars Marvin)
axiom A7 : ¬FromEarth Marvin → Extraterrestrial Marvin

theorem conclusion_false : ¬(¬FromMars Marvin → Human Marvin) := by
  have hdisj := A6
  have hNE_NM : ¬FromEarth Marvin ∧ ¬FromMars Marvin :=
    Or.elim hdisj
      (fun hEM => by
        have hE := And.left hEM
        have hM := And.right hEM
        have hAlien : Alien Marvin := A2 Marvin hM
        have hExtra : Extraterrestrial Marvin := A1 Marvin hAlien
        have hNotHuman : ¬Human Marvin := A3 Marvin hExtra
        have hHuman : Human Marvin := A4 Marvin ⟨A5, hE⟩
        exact False.elim (hNotHuman hHuman)
      )
      (fun h => h)
  have hNotMars : ¬FromMars Marvin := And.right hNE_NM
  have hNotEarth : ¬FromEarth Marvin := And.left hNE_NM
  have hExtra2 : Extraterrestrial Marvin := A7 hNotEarth
  have hNotHuman2 : ¬Human Marvin := A3 Marvin hExtra2
  intro hImp
  exact hNotHuman2 (hImp hNotMars)
</lean>

ANSWER: False
