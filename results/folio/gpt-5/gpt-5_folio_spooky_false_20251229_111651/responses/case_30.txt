<lean>
axiom Person : Type
axiom Djokovic : Person

axiom AtMixer : Person → Prop
axiom GrandSlamChampion : Person → Prop
axiom OscarNominatedActor : Person → Prop
axiom ProfessionalTennisPlayer : Person → Prop
axiom Celebrity : Person → Prop
axiom Athlete : Person → Prop
axiom Famous : Person → Prop
axiom WellPaid : Person → Prop
axiom LivesInTaxHaven : Person → Prop

-- Premises
axiom P1 : ∀ x : Person, AtMixer x → (GrandSlamChampion x ∨ OscarNominatedActor x)
axiom P2 : ∀ x : Person, AtMixer x → GrandSlamChampion x → ProfessionalTennisPlayer x
axiom P3 : ∀ x : Person, AtMixer x → OscarNominatedActor x → Celebrity x
axiom P4 : ∀ x : Person, AtMixer x → ProfessionalTennisPlayer x → Athlete x
axiom P5 : ∀ x : Person, AtMixer x → Celebrity x → WellPaid x
axiom P6 : ∀ x : Person, AtMixer x → Athlete x → Famous x
axiom P7 : ∀ x : Person, AtMixer x → WellPaid x → LivesInTaxHaven x
axiom P8 : AtMixer Djokovic
axiom P9 : (Famous Djokovic ∧ Athlete Djokovic) → WellPaid Djokovic

-- A derivable consequence: Djokovic is well-paid.
theorem Djokovic_well_paid : WellPaid Djokovic := by
  have hdisj : GrandSlamChampion Djokovic ∨ OscarNominatedActor Djokovic := P1 Djokovic P8
  cases hdisj with
  | inl hGS =>
      have hPT : ProfessionalTennisPlayer Djokovic := P2 Djokovic P8 hGS
      have hAth : Athlete Djokovic := P4 Djokovic P8 hPT
      have hFam : Famous Djokovic := P6 Djokovic P8 hAth
      exact P9 (And.intro hFam hAth)
  | inr hOsc =>
      have hCel : Celebrity Djokovic := P3 Djokovic P8 hOsc
      exact P5 Djokovic P8 hCel

-- Another consequence: Djokovic lives in a tax haven.
theorem Djokovic_tax_haven : LivesInTaxHaven Djokovic := by
  have hw : WellPaid Djokovic := Djokovic_well_paid
  exact P7 Djokovic P8 hw
