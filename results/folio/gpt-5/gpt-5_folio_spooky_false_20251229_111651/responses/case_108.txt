<lean>
axiom obj : Type

axiom Peter : obj

-- Predicates
axiom Spills : obj → Prop
axiom Tidy : obj → Prop
axiom CF : obj → Prop                     -- clumsy foodie who goes out frequently to find new food restaurants to try
axiom Cleanly : obj → Prop
axiom VOS : obj → Prop                    -- values order and spotlessness
axiom FPOS : obj → Prop                   -- family prioritizes order and spotlessness

-- Premises
axiom P1 : ∀ x : obj, Spills x → ¬Tidy x
axiom P2 : ∀ x : obj, CF x → Spills x
axiom P3 : ∀ x : obj, Cleanly x → Tidy x
axiom P4 : ∀ x : obj, VOS x → Cleanly x
axiom P5 : ∀ x : obj, FPOS x → VOS x
axiom P6 : (Spills Peter ∧ Cleanly Peter) ∨ (¬Spills Peter ∧ ¬Cleanly Peter)

-- Goal: ¬(CF Peter ∨ FPOS Peter)
theorem not_peter_CF_or_FPOS : ¬(CF Peter ∨ FPOS Peter) := by
  intro h
  have disj := P6
  cases disj with
  | inl hsc =>
      -- hsc : Spills Peter ∧ Cleanly Peter
      have ht : Tidy Peter := P3 Peter hsc.right
      have hnt : ¬Tidy Peter := P1 Peter hsc.left
      exact hnt ht
  | inr hs_nc =>
      -- hs_nc : ¬Spills Peter ∧ ¬Cleanly Peter
      have hNotCF : ¬CF Peter := by
        intro hcf
        have sp : Spills Peter := P2 Peter hcf
        exact hs_nc.left sp
      have fpos_to_clean : ∀ x : obj, FPOS x → Cleanly x :=
        fun x hf => P4 x (P5 x hf)
      have hNotFPOS : ¬FPOS Peter := by
        intro hf
        have cl : Cleanly Peter := fpos_to_clean Peter hf
        exact hs_nc.right cl
      cases h with
      | inl hcf => exact hNotCF hcf
      | inr hf => exact hNotFPOS hf
</lean>

ANSWER: False
