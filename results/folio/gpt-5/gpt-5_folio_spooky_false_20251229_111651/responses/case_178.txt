<lean>
axiom obj : Type
axiom Human : obj → Prop
axiom Horse : obj → Prop
axiom Hooves : obj → Prop

-- Premises
axiom AllHorsesHaveHooves : ∀ x : obj, Horse x → Hooves x
axiom NoHumansHaveHooves : ∀ x : obj, Human x → ¬Hooves x

-- Negation of the statement "Some humans are horses"
theorem no_humans_are_horses : ¬(∃ x : obj, Human x ∧ Horse x) := by
  intro h
  apply Exists.elim h
  intro x hx
  have hxHuman : Human x := And.left hx
  have hxHorse : Horse x := And.right hx
  have hxHooves : Hooves x := AllHorsesHaveHooves x hxHorse
  have hxNoHooves : ¬Hooves x := NoHumansHaveHooves x hxHuman
  exact hxNoHooves hxHooves
</lean>

ANSWER: False
