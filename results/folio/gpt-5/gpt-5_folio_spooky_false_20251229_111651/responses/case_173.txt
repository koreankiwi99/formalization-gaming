<lean>
axiom person : Type
axiom Taylor : person

axiom EatsSaladsRegularly : person → Prop
axiom VeryConscious : person → Prop
axiom GrewUpHealthConsciousHome : person → Prop
axiom FulfillDailyNutrition : person → Prop
axiom DisregardPhysicalWellbeing : person → Prop
axiom VisitGymDaily : person → Prop

axiom A1 : ∀ x : person, EatsSaladsRegularly x → VeryConscious x
axiom A2 : ∀ x : person, GrewUpHealthConsciousHome x → EatsSaladsRegularly x
axiom A3 : ∀ x : person, FulfillDailyNutrition x → GrewUpHealthConsciousHome x
axiom A4 : ∀ x : person, DisregardPhysicalWellbeing x → ¬VeryConscious x
axiom A5 : ∀ x : person, VisitGymDaily x → FulfillDailyNutrition x
axiom A6 : (GrewUpHealthConsciousHome Taylor ∧ DisregardPhysicalWellbeing Taylor) ∨ (¬GrewUpHealthConsciousHome Taylor ∧ ¬DisregardPhysicalWellbeing Taylor)

theorem taylor_not_visitGym : ¬VisitGymDaily Taylor := by
  intro hGym
  have hFul : FulfillDailyNutrition Taylor := A5 Taylor hGym
  have hHome : GrewUpHealthConsciousHome Taylor := A3 Taylor hFul
  have hSal : EatsSaladsRegularly Taylor := A2 Taylor hHome
  have hVery : VeryConscious Taylor := A1 Taylor hSal
  cases A6 with
  | inl h =>
      exact (A4 Taylor h.right) hVery
  | inr h =>
      exact (h.left) hHome
</lean>

ANSWER: False
