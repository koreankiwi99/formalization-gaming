axiom obj : Type

axiom Animal : obj → Prop
axiom Mammal : obj → Prop
axiom Human : obj → Prop
axiom Person : obj → Prop

axiom VirusOccurs : obj → Prop
axiom Monkeypox : obj → Prop

axiom Fever : obj → Prop
axiom Headache : obj → Prop
axiom MusclePains : obj → Prop
axiom Tired : obj → Prop

axiom Flu : obj → Prop

-- When the Monkeypox virus occurs in a being, it may get Monkeypox.
axiom A1 : ∀ x : obj, VirusOccurs x → Monkeypox x
-- Monkeypox virus can occur in certain animals. (existence)
axiom A2 : ∃ x : obj, Animal x ∧ VirusOccurs x
-- Humans are mammals.
axiom A3 : ∀ x : obj, Human x → Mammal x
-- Mammals are animals.
axiom A4 : ∀ x : obj, Mammal x → Animal x
-- Symptoms of Monkeypox include fever, headache, muscle pains, and tiredness.
axiom A5 : ∀ x : obj, Monkeypox x → Fever x ∧ Headache x ∧ MusclePains x ∧ Tired x
-- People feel tired when they get the flu.
axiom A6 : ∀ x : obj, Person x → Flu x → Tired x

-- From the given premises we can derive that there exists a tired animal (due to Monkeypox),
-- but we cannot derive that anyone gets the flu.
theorem exists_tired_animal : ∃ x : obj, Animal x ∧ Tired x := by
  rcases A2 with ⟨a, ha⟩
  have hA : Animal a := And.left ha
  have hV : VirusOccurs a := And.right ha
  have hM : Monkeypox a := A1 a hV
  have hSym : Fever a ∧ Headache a ∧ MusclePains a ∧ Tired a := A5 a hM
  have hT : Tired a := And.right (And.right (And.right hSym))
  exact ⟨a, And.intro hA hT⟩
