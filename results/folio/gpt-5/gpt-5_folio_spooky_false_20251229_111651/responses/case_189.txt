<lean>
axiom obj : Type
axiom Yuri : obj

axiom American : obj → Prop
axiom Pro : obj → Prop
axiom JumpShots : obj → Prop
axiom LeapStraight : obj → Prop
axiom ActivateLegs : obj → Prop

-- Some professional basketball players are not American nationals.
axiom S1 : ∃ x : obj, Pro x ∧ ¬American x
-- All professional basketball players can do jump shots.
axiom P1 : ∀ x : obj, Pro x → JumpShots x
-- If someone can jump shots, they leap straight into the air.
axiom P2 : ∀ x : obj, JumpShots x → LeapStraight x
-- If someone leaps straight into the air, they activate their leg muscles.
axiom P3 : ∀ x : obj, LeapStraight x → ActivateLegs x
-- Yuri does not activate his leg muscles.
axiom P4 : ¬ActivateLegs Yuri

-- Negation of the statement to evaluate: Yuri is an American professional basketball player.
theorem yuri_not_american_pro : ¬(American Yuri ∧ Pro Yuri) := by
  intro h
  have hyPro : Pro Yuri := h.right
  have hj : JumpShots Yuri := P1 Yuri hyPro
  have hl : LeapStraight Yuri := P2 Yuri hj
  have ha : ActivateLegs Yuri := P3 Yuri hl
  exact P4 ha
</lean>

ANSWER: False
