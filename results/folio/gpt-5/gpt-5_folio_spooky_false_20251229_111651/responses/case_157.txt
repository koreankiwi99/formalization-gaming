axiom obj : Type
axiom James : obj

axiom WorksInLibrary : obj → Prop
axiom FromCS : obj → Prop
axiom UnivPTJob : obj → Prop
axiom TakingDBCourse : obj → Prop
axiom TakingWithDavid : obj → Prop
axiom WorksInLab : obj → Prop

-- None of the students who currently work in the library are from the CS department.
axiom H1 : ∀ x : obj, WorksInLibrary x → ¬ FromCS x
-- All of the students with part-time jobs offered by the university currently work in the library.
axiom H2 : ∀ x : obj, UnivPTJob x → WorksInLibrary x
-- All of the students who are taking the database course are from the CS department.
axiom H3 : ∀ x : obj, TakingDBCourse x → FromCS x
-- All of the students taking a class with Professor David are taking the database course.
axiom H4 : ∀ x : obj, TakingWithDavid x → TakingDBCourse x
-- All students working in the lab are taking a class with Professor David.
axiom H5 : ∀ x : obj, WorksInLab x → TakingWithDavid x
-- James is a student working in the lab.
axiom H6 : WorksInLab James
-- James doesn't work in the lab or have a part-time job offered by the university.
axiom H7 : ¬WorksInLab James ∨ ¬UnivPTJob James

theorem james_from_cs : FromCS James := by
  have hwd : TakingWithDavid James := H5 James H6
  have hdb : TakingDBCourse James := H4 James hwd
  exact H3 James hdb

theorem james_not_univ_pt_job : ¬UnivPTJob James := by
  intro hpt
  have hwil : WorksInLibrary James := H2 James hpt
  have hnotcs : ¬FromCS James := H1 James hwil
  exact hnotcs james_from_cs
