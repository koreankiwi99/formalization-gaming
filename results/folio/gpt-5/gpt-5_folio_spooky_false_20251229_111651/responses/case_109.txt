axiom obj : Type
axiom Peter : obj

axiom Spills : obj → Prop
axiom Tidy : obj → Prop
axiom Cleanly : obj → Prop
axiom ValuesOrderSpotless : obj → Prop
axiom FamilyPrioritizesOrderSpotless : obj → Prop
axiom CFGo : obj → Prop  -- "clumsy foodie who goes out frequently to find new food restaurants to try"

-- Premises
axiom P1 : ∀ x : obj, Spills x → ¬ Tidy x
axiom P2 : ∀ x : obj, CFGo x → Spills x
axiom P3 : ∀ x : obj, Cleanly x → Tidy x
axiom P4 : ∀ x : obj, ValuesOrderSpotless x → Cleanly x
axiom P5 : ∀ x : obj, FamilyPrioritizesOrderSpotless x → ValuesOrderSpotless x
axiom P6 : (Spills Peter ∧ Cleanly Peter) ∨ (¬ Spills Peter ∧ ¬ Cleanly Peter)

-- No one can both spill and be cleanly
lemma no_spill_and_cleanly : ∀ x : obj, ¬ (Spills x ∧ Cleanly x) := by
  intro x
  intro hx
  have hs : Spills x := And.left hx
  have hc : Cleanly x := And.right hx
  have ht : Tidy x := P3 x hc
  have hnt : ¬ Tidy x := P1 x hs
  exact hnt ht

-- From P6 and the contradiction above, deduce Peter neither spills nor is cleanly
lemma peter_not_spill_and_not_cleanly : ¬ Spills Peter ∧ ¬ Cleanly Peter := by
  have h := P6
  cases h with
  | inl hsc =>
      exact False.elim (no_spill_and_cleanly Peter hsc)
  | inr hnc =>
      exact hnc

-- CFGo → Spills, so since Peter does not spill, Peter is not CFGo
lemma not_CFGo_Peter : ¬ CFGo Peter := by
  have hns : ¬ Spills Peter := And.left peter_not_spill_and_not_cleanly
  intro hcf
  exact hns (P2 Peter hcf)

-- Family implies values implies cleanly
lemma family_implies_cleanly : ∀ x : obj, FamilyPrioritizesOrderSpotless x → Cleanly x := by
  intro x
  intro hf
  exact P4 x (P5 x hf)

-- Since Peter is not cleanly, he is not from a family that prioritizes order and spotlessness
lemma not_family_Peter : ¬ FamilyPrioritizesOrderSpotless Peter := by
  have hnc : ¬ Cleanly Peter := And.right peter_not_spill_and_not_cleanly
  intro hf
  exact hnc (family_implies_cleanly Peter hf)

-- The evaluated statement actually holds: Peter neither is CFGo nor from such a family
theorem statement_true :
  (CFGo Peter ∧ FamilyPrioritizesOrderSpotless Peter) ∨
  (¬ CFGo Peter ∧ ¬ FamilyPrioritizesOrderSpotless Peter) := by
  apply Or.inr
  exact And.intro not_CFGo_Peter not_family_Peter
