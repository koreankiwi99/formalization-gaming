axiom obj : Type
axiom LastSummerOlympics : obj
axiom InTokyo : obj → Prop

-- The statement to evaluate is: ¬ InTokyo LastSummerOlympics
-- We prove its negation, i.e., that the statement is false.
theorem statement_false : ¬ (¬ InTokyo LastSummerOlympics) := by
  intro h
  exact h last_in_tokyo

-- Stronger: assuming the statement yields a contradiction.
theorem assuming_statement_leads_to_false (h : ¬ InTokyo LastSummerOlympics) : False := by
  exact h last_in_tokyo
