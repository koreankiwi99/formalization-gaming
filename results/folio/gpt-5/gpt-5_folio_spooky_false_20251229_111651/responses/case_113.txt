<lean>
axiom obj : Type

axiom Picuris : obj
axiom Harding : obj

axiom MountainRange : obj → Prop
axiom InNM : obj → Prop
axiom InTX : obj → Prop
axiom LocatedIn : obj → obj → Prop
axiom Donated : obj → Prop

-- Premises
axiom P1 : MountainRange Picuris
axiom P1b : InNM Picuris ∨ InTX Picuris
axiom A1 : LocatedIn Harding Picuris
axiom A2 : Donated Harding
axiom NoTXDonatedMineRange :
  ∀ r : obj, MountainRange r ∧ InTX r → ¬(∃ m : obj, LocatedIn m r ∧ Donated m)

-- The statement to evaluate is: ¬∃ r, MountainRange r ∧ InNM r ∧ LocatedIn Harding r
-- We prove its negation by exhibiting such a mountain range (Picuris).
theorem harding_in_nm_range :
  ∃ r : obj, MountainRange r ∧ InNM r ∧ LocatedIn Harding r := by
  -- There exists a donated mine (Harding) located in Picuris
  have exHarding : ∃ m : obj, LocatedIn m Picuris ∧ Donated m := by
    exact Exists.intro Harding (And.intro A1 A2)
  -- Picuris cannot be in Texas, otherwise it would contradict NoTXDonatedMineRange
  have notTX : ¬InTX Picuris := by
    intro htx
    have hno : ¬∃ m : obj, LocatedIn m Picuris ∧ Donated m := by
      have hpair : MountainRange Picuris ∧ InTX Picuris := And.intro P1 htx
      exact NoTXDonatedMineRange Picuris hpair
    exact hno exHarding
  -- Therefore, given it's in NM or TX, it must be in NM
  have inNM : InNM Picuris := by
    exact Or.resolve_right P1b notTX
  -- Conclude the witness r = Picuris
  exact Exists.intro Picuris (And.intro P1 (And.intro inNM A1))

theorem statement_is_false :
  ¬(¬∃ r : obj, MountainRange r ∧ InNM r ∧ LocatedIn Harding r) := by
  intro h
  exact h harding_in_nm_range
</lean>

ANSWER: False
