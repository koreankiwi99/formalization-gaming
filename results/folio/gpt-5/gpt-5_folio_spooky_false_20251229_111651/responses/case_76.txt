<lean>
axiom obj : Type

axiom AtMS : obj → Prop
axiom Work : obj → Prop
axiom Need : obj → Prop
axiom Takeout : obj → Prop
axiom Enjoy : obj → Prop
axiom Picky : obj → Prop
axiom DineALot : obj → Prop

axiom Hannah : obj

-- Premises
axiom R1 : ∀ x : obj, AtMS x → Work x → Need x
axiom R2 : ∀ x : obj, AtMS x → Takeout x → Work x
axiom R3 : ∀ x : obj, AtMS x → (Takeout x ∨ Enjoy x)
axiom R4 : ∀ x : obj, AtMS x → Enjoy x → ¬Picky x
axiom R5 : ∀ x : obj, AtMS x → Enjoy x → DineALot x

axiom H_at : AtMS Hannah
axiom H_work : Work Hannah
axiom H_imp : Need Hannah → (¬Picky Hannah ∧ ¬Need Hannah)

-- Derive a contradiction from the premises
theorem global_contradiction : False := by
  have hNeed : Need Hannah := R1 Hannah H_at H_work
  have hNoNeed : ¬Need Hannah := And.right (H_imp hNeed)
  exact hNoNeed hNeed

-- Negation of the statement to evaluate
theorem neg_statement :
  ¬(AtMS Hannah ∧ ¬Picky Hannah ∧ DineALot Hannah) := by
  intro _
  exact False.elim global_contradiction
</lean>

ANSWER: False
