axiom obj : Type
axiom Pet : obj → Prop
axiom Animal : obj → Prop
axiom Mammal : obj → Prop
axiom Monkey : obj → Prop
axiom Bird : obj → Prop
axiom Fly : obj → Prop
axiom Wings : obj → Prop
axiom Breathe : obj → Prop

axiom Rock : obj

-- All of Peter's pets that are mammals are also animals.
axiom A1 : ∀ x : obj, Pet x ∧ Mammal x → Animal x
-- All monkeys are mammals.
axiom A2 : ∀ x : obj, Monkey x → Mammal x
-- Peter's pets are all either monkeys or birds.
axiom A3 : ∀ x : obj, Pet x → Monkey x ∨ Bird x
-- Peter's birds can fly.
axiom A4 : ∀ x : obj, Pet x ∧ Bird x → Fly x
-- All animals that are Peter's pets can breathe.
axiom A5 : ∀ x : obj, Pet x ∧ Animal x → Breathe x
-- If Peter's pet can fly, then it has wings.
axiom A6 : ∀ x : obj, Pet x ∧ Fly x → Wings x
-- Rock is Peter's pet.
axiom A7 : Pet Rock
-- Rock can fly, or Rock is a bird, or Rock cannot breathe.
axiom A8 : Fly Rock ∨ Bird Rock ∨ ¬Breathe Rock

-- From the premises, Rock has wings.
theorem rock_has_wings : Wings Rock := by
  have hPet : Pet Rock := A7
  have h_mb := A3 Rock hPet
  cases h_mb with
  | inl hMonkey =>
    -- Use the disjunction about Rock (fly ∨ bird ∨ ¬breathe)
    cases A8 with
    | inl hFly =>
      exact A6 Rock ⟨hPet, hFly⟩
    | inr hRest =>
      cases hRest with
      | inl hBird =>
        have hFly : Fly Rock := A4 Rock ⟨hPet, hBird⟩
        exact A6 Rock ⟨hPet, hFly⟩
      | inr hNotBreathe =>
        have hMammal : Mammal Rock := A2 Rock hMonkey
        have hAnimal : Animal Rock := A1 Rock ⟨hPet, hMammal⟩
        have hBreathe : Breathe Rock := A5 Rock ⟨hPet, hAnimal⟩
        have : False := hNotBreathe hBreathe
        exact False.elim this
  | inr hBird =>
    have hFly : Fly Rock := A4 Rock ⟨hPet, hBird⟩
    exact A6 Rock ⟨hPet, hFly⟩
