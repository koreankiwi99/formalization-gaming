<lean>
axiom obj : Type

axiom StrangerThings : obj
axiom BlackMirror : obj

axiom Netflix : obj → Prop
axiom Popular : obj → Prop
axiom Binge : obj → Prop        -- Karen binge-watches it
axiom Download : obj → Prop     -- Karen downloads it
axiom ShareLisa : obj → Prop    -- Karen shares it with Lisa

-- "Stranger Things" is a popular Netflix show.
axiom ST_Netflix : Netflix StrangerThings
axiom ST_Popular : Popular StrangerThings

-- If a Netflix show is popular, Karen will binge-watch it.
axiom A1 : ∀ x : obj, Netflix x → Popular x → Binge x

-- If and only if Karen binge-watches a Netflix show, she will download it. (Binge ↔ Download for Netflix shows)
axiom A2 : ∀ x : obj, Netflix x → (Binge x ↔ Download x)

-- Karen does not download "Black Mirror."
axiom A3 : ¬ Download BlackMirror

-- "Black Mirror" is a Netflix show.
axiom A4 : Netflix BlackMirror

-- If Karen binge-watches a Netflix show, she will share it with Lisa.
axiom A5 : ∀ x : obj, Netflix x → Binge x → ShareLisa x

-- Prove the negation of the statement: "Black Mirror" is popular.
theorem blackmirror_not_popular : ¬ Popular BlackMirror := by
  intro hpop
  have hbinge : Binge BlackMirror := A1 BlackMirror A4 hpop
  have hiff : Binge BlackMirror ↔ Download BlackMirror := A2 BlackMirror A4
  have hdl : Download BlackMirror := (Iff.mp hiff) hbinge
  exact A3 hdl
</lean>

ANSWER: False
