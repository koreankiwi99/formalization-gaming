axiom Person : Type
axiom Luke : Person

axiom Baby : Person → Prop
axiom MultipleBirth : Person → Prop
axiom SiblingsBornTogether : Person → Prop
axiom ComplainOften : Person → Prop
axiom LiveAtHome : Person → Prop
axiom LiveWithStrangers : Person → Prop
axiom SpendTimeWithSiblings : Person → Prop

-- Premises
axiom P1 : ∀ x : Person, MultipleBirth x → SpendTimeWithSiblings x
axiom P2 : ∀ x : Person, SiblingsBornTogether x → MultipleBirth x
axiom P3 : ∀ x : Person, ComplainOften x → SiblingsBornTogether x
axiom P4 : ∀ x : Person, LiveAtHome x → ¬LiveWithStrangers x
axiom P5 : ∀ x : Person, SpendTimeWithSiblings x → LiveAtHome x

-- Luke either is a baby born in a multiple birth and live with strangers,
-- or is not a baby, is born in a multiple birth, and does not live with strangers.
axiom Luke_split :
  (Baby Luke ∧ MultipleBirth Luke ∧ LiveWithStrangers Luke) ∨
  ((¬Baby Luke) ∧ MultipleBirth Luke ∧ ¬LiveWithStrangers Luke)

lemma multiple_implies_not_strangers : ∀ x : Person, MultipleBirth x → ¬LiveWithStrangers x := by
  intro x hmult
  have hSpend : SpendTimeWithSiblings x := P1 x hmult
  have hHome : LiveAtHome x := P5 x hSpend
  exact P4 x hHome

theorem luke_multiple : MultipleBirth Luke := by
  have not_first :
      ¬(Baby Luke ∧ MultipleBirth Luke ∧ LiveWithStrangers Luke) := by
    intro h
    have hmult : MultipleBirth Luke := And.left (And.right h)
    have hnot : ¬LiveWithStrangers Luke := multiple_implies_not_strangers Luke hmult
    have hstr : LiveWithStrangers Luke := And.right (And.right h)
    exact hnot hstr
  cases Luke_split with
  | inl h1 =>
      exact False.elim (not_first h1)
  | inr h2 =>
      exact And.left (And.right h2)

-- Statement to evaluate: Luke is neither born in a multiple birth nor complains often about annoying siblings.
-- We prove its negation.
theorem not_statement : ¬(¬MultipleBirth Luke ∧ ¬ComplainOften Luke) := by
  intro h
  have hMB : MultipleBirth Luke := luke_multiple
  exact (And.left h) hMB
</lean>

ANSWER: False

axiom Person : Type
axiom Luke : Person

axiom SiblingsBornTogether : Person → Prop
axiom ComplainOften : Person → Prop

axiom LiveWithStrangers : Person → Prop
axiom SpendTimeWithSiblings : Person → Prop

-- Premises
axiom P1 : ∀ x : Person, MultipleBirth x → SpendTimeWithSiblings x
axiom P2 : ∀ x : Person, SiblingsBornTogether x → MultipleBirth x
axiom P3 : ∀ x : Person, ComplainOften x → SiblingsBornTogether x
axiom P4 : ∀ x : Person, LiveAtHome x → ¬LiveWithStrangers x
axiom P5 : ∀ x : Person, SpendTimeWithSiblings x → LiveAtHome x

-- Luke either is a baby born in a multiple birth and lives with strangers,
-- or is not a baby, is born in a multiple birth, and does not live with strangers.
axiom Luke_split :
  (Baby Luke ∧ MultipleBirth Luke ∧ LiveWithStrangers Luke) ∨
  ((¬Baby Luke) ∧ MultipleBirth Luke ∧ ¬LiveWithStrangers Luke)

theorem luke_multiple : MultipleBirth Luke := by
  cases Luke_split with
  | inl h1 =>
      -- h1 : Baby Luke ∧ MultipleBirth Luke ∧ LiveWithStrangers Luke
      have hMB : MultipleBirth Luke := And.left (And.right h1)
      have hSpend : SpendTimeWithSiblings Luke := P1 Luke hMB
      have hHome : LiveAtHome Luke := P5 Luke hSpend
      have hNotStr : ¬LiveWithStrangers Luke := P4 Luke hHome
      have hStr : LiveWithStrangers Luke := And.right (And.right h1)
      exact False.elim (hNotStr hStr)
  | inr h2 =>
      -- h2 : ¬Baby Luke ∧ MultipleBirth Luke ∧ ¬LiveWithStrangers Luke
      exact And.left (And.right h2)

-- Statement's negation: it's not the case that (¬MultipleBirth Luke ∧ ¬ComplainOften Luke)
theorem not_statement : ¬(¬MultipleBirth Luke ∧ ¬ComplainOften Luke) := by
  intro h
  have hMB : MultipleBirth Luke := luke_multiple
  exact (And.left h) hMB
</lean>

ANSWER: False
