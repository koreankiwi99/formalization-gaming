axiom Mean : obj → Prop
axiom Grumpy : obj → Prop
axiom LovesAnimals : obj → Prop
axiom PetOwner : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, Nice x → ¬Mean x
axiom P2 : ∃ x : obj, Grumpy x ∧ Mean x
axiom P3 : ∀ x : obj, LovesAnimals x → Nice x
axiom P4 : ∀ x : obj, PetOwner x → LovesAnimals x
axiom P5 : PetOwner Tom

-- Negation of the statement:
-- A := ¬(Grumpy Tom ∧ Mean Tom)
-- B := (¬Grumpy Tom ∧ ¬LovesAnimals Tom)
-- We prove A ∧ ¬B
theorem negate_statement :
  (¬(Grumpy Tom ∧ Mean Tom)) ∧ ¬(¬Grumpy Tom ∧ ¬LovesAnimals Tom) :=
by
  have hL : LovesAnimals Tom := P4 Tom P5
  have hN : Nice Tom := P3 Tom hL
  have hNotMean : ¬Mean Tom := P1 Tom hN
  have left : ¬(Grumpy Tom ∧ Mean Tom) := by
    intro hGM
    exact hNotMean hGM.right
  have right : ¬(¬Grumpy Tom ∧ ¬LovesAnimals Tom) := by
    intro h
    exact h.right hL
  exact And.intro left right
