axiom deficiency : ∀ a : Athlete, LacksSkill a ∨ LacksStrength a

-- Premise: Proper training enhances skills
axiom training_skill : ∀ a : Athlete, TrainProperly a → ¬ LacksSkill a

-- Premise: Dedication leads to proper training
axiom dedication_training : ∀ a : Athlete, Dedicated a → TrainProperly a

-- Given fact: Mark eats healthy
axiom mark_eats_healthy : HealthyNutrition Mark

-- Conclusion: Is Mark dedicated to his sport?
theorem goal : ¬ Dedicated Mark :=
  fun hDed : Dedicated Mark =>
    Or.elim (deficiency Mark)
      (fun hSkill : LacksSkill Mark => 
        training_skill Mark (dedication_training Mark hDed) hSkill)
      (fun hStrength : LacksStrength Mark => 
        nutrition_strength Mark mark_eats_healthy hStrength)
</lean>
