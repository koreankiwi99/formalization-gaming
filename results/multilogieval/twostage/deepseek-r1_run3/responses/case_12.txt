axiom FollowPrecisely : Chef → Prop
axiom DishesWell : Chef → Prop
axiom GainConfidence : Chef → Prop
axiom WatchShows : Chef → Prop

theorem conclusion : ¬ GainConfidence James → WatchShows James := 
  fun h : ¬ GainConfidence James =>
