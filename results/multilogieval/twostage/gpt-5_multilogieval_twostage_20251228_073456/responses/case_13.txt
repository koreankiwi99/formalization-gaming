--- Iteration 1 ---

<lean>
axiom Company : Type
axiom AC : Company

axiom WantsIncreaseSales : Company → Prop
axiom InvestedMarketingLQ : Company → Prop
axiom FocusedBudgetOnMarketingLQ : Company → Prop
axiom MoreCustomersLQ : Company → Prop
axiom SalesIncreasedLQ : Company → Prop
axiom ProductionCostsIncreasedLQ : Company → Prop

-- Premises
axiom P1 : ∀ c : Company, WantsIncreaseSales c
axiom P2 : ∀ c : Company, InvestedMarketingLQ c → MoreCustomersLQ c
axiom P3 : ∀ c : Company, MoreCustomersLQ c → SalesIncreasedLQ c
axiom P4 : FocusedBudgetOnMarketingLQ AC ∨ ProductionCostsIncreasedLQ AC ∨ (FocusedBudgetOnMarketingLQ AC ∧ ProductionCostsIncreasedLQ AC)
axiom P5 : ∀ c : Company, FocusedBudgetOnMarketingLQ c → InvestedMarketingLQ c

-- Conclusion to formalize
theorem goal : (¬ SalesIncreasedLQ AC) → ProductionCostsIncreasedLQ AC := by
  intro hnot
  cases P4 with
  | inl hF =>
      have hIM : InvestedMarketingLQ AC := P5 AC hF
      have hMC : MoreCustomersLQ AC := P2 AC hIM
      have hSI : SalesIncreasedLQ AC := P3 AC hMC
      exact False.elim (hnot hSI)
  | inr h =>
      cases h with
      | inl hPC => exact hPC
      | inr hBoth => exact hBoth.right
</lean>
ANSWER: Yes
