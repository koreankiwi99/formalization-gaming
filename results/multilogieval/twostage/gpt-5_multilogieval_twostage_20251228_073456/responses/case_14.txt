<lean>
axiom obj : Type
axiom Someone : obj

axiom HopedDreamHome : obj → Prop
axiom SavedAggressively : obj → Prop
axiom AffordDownPayment : obj → Prop
axiom ImprovedCredit : obj → Prop
axiom QualifyMortgage : obj → Prop
axiom BuyDreamHome : obj → Prop

-- Someone hoped to live in their dream home.
axiom H1 : HopedDreamHome Someone

-- If someone saved aggressively, they could afford a down payment.
axiom R1 : ∀ x : obj, SavedAggressively x → AffordDownPayment x

-- If someone improved their credit score, they could qualify for a mortgage.
axiom R2 : ∀ x : obj, ImprovedCredit x → QualifyMortgage x

-- Someone either saves aggressively or improves their credit.
axiom D1 : SavedAggressively Someone ∨ ImprovedCredit Someone

-- If someone qualifies for a mortgage, they can buy their dream home.
axiom R3 : ∀ x : obj, QualifyMortgage x → BuyDreamHome x

-- Conclusion to formalize: Given that they could not afford a down payment, did they buy their dream home?
theorem goal : ¬ AffordDownPayment Someone → BuyDreamHome Someone :=
fun h =>
  Or.elim D1
    (fun hS => False.elim (h (R1 Someone hS)))
    (fun hI => R3 Someone (R2 Someone hI))
</lean>
