<lean>
axiom Person : Type
axiom Priya : Person

axiom StudiesHard : Person → Prop
axiom GoodGrades : Person → Prop
axiom ParentsProud : Person → Prop
axiom IceCreamOuting : Person → Prop

axiom Snowstorm : Prop
axiom BigSnowstorm : Prop
axiom SchoolCanceled : Prop

axiom P1 : ∀ x : Person, StudiesHard x → GoodGrades x
axiom P2 : Snowstorm → SchoolCanceled
axiom P3 : StudiesHard Priya ∨ ¬SchoolCanceled
axiom P4 : ∀ x : Person, GoodGrades x → ParentsProud x
axiom P5 : ∀ x : Person, ParentsProud x → IceCreamOuting x
axiom P6 : BigSnowstorm → Snowstorm

theorem goal : BigSnowstorm → IceCreamOuting Priya :=
by
  intro hBig
  have hSnow : Snowstorm := P6 hBig
  have hCanceled : SchoolCanceled := P2 hSnow
  have hGG : GoodGrades Priya :=
    Or.elim P3
      (fun hStud => P1 Priya hStud)
      (fun hNotCancel => False.elim (hNotCancel hCanceled))
  have hProud : ParentsProud Priya := P4 Priya hGG
  exact P5 Priya hProud
</lean>
