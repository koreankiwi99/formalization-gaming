--- Iteration 1 ---

<lean>
axiom obj : Type
axiom Matt : obj

axiom WantedRedecorate : obj → Prop
axiom WatchedDIY : obj → Prop
axiom GainSkills : obj → Prop
axiom RedoLivingRoom : obj → Prop
axiom ReadBlogs : obj → Prop
axiom NewIdeas : obj → Prop

-- Premises
axiom P1 : ∃ x : obj, WantedRedecorate x
axiom P2 : ∀ x : obj, WatchedDIY x → GainSkills x
axiom P3 : ∀ x : obj, GainSkills x → RedoLivingRoom x
axiom P4 : ∃ x : obj, ¬ RedoLivingRoom x
axiom P5 : ∀ x : obj, ¬ RedoLivingRoom x → ¬ GainSkills x
axiom P6 : ∀ x : obj, WatchedDIY x ∨ ReadBlogs x
axiom P7 : ∀ x : obj, ReadBlogs x → NewIdeas x

-- Conclusion to formalize
theorem goal : (¬ RedoLivingRoom Matt) → NewIdeas Matt := by
  intro hnr
  have hdisj := P6 Matt
  cases hdisj with
  | inl hwat =>
      have hgs : GainSkills Matt := P2 Matt hwat
      have hrl : RedoLivingRoom Matt := P3 Matt hgs
      have hfalse : False := hnr hrl
      exact False.elim hfalse
  | inr hblog =>
      exact P7 Matt hblog
</lean>
