axiom company : Type

axiom DevelopInnovative : company → product → Prop
axiom MarketsEffectively : company → product → Prop
axiom CommerciallySuccessful : product → Prop

axiom Acme : company
axiom Gadget : product

-- Premise: If a company develops an innovative product, then if they market it effectively, the product will be commercially successful.
axiom P1 : ∀ c : company, ∀ p : product, DevelopInnovative c p → (MarketsEffectively c p → CommerciallySuccessful p)

-- Facts about Acme Inc. and their gadget
axiom F1 : MarketsEffectively Acme Gadget
axiom F2 : ¬ CommerciallySuccessful Gadget

-- Conclusion to formalize: Did Acme Inc. develop an innovative product?
-- From the given facts, we aim to conclude they did not.
theorem goal : ¬ DevelopInnovative Acme Gadget := by
  intro h
  have hcs : CommerciallySuccessful Gadget := (P1 Acme Gadget h) F1
  exact F2 hcs
