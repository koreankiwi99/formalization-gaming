--- Iteration 1 ---

<lean>
axiom obj : Type
axiom Mary : obj

axiom HomeOwner : obj → Prop
axiom LovesGrowingPlants : obj → Prop
axiom WantsBeautifulGarden : obj → Prop
axiom LearnedBotany : obj → Prop
axiom UnderstandPlantNeeds : obj → Prop
axiom GrowThrivingPlants : obj → Prop
axiom SoughtAdviceFromExperiencedGardeners : obj → Prop

-- Premises about home owners
axiom P1 : ∀ x : obj, HomeOwner x → LovesGrowingPlants x
axiom P2 : ∀ x : obj, HomeOwner x → WantsBeautifulGarden x
axiom P3 : ∀ x : obj, HomeOwner x → (LearnedBotany x → UnderstandPlantNeeds x)
axiom P4 : ∀ x : obj, HomeOwner x → (UnderstandPlantNeeds x → GrowThrivingPlants x)
axiom P5 : ∀ x : obj, HomeOwner x → (LearnedBotany x ∨ SoughtAdviceFromExperiencedGardeners x)

-- Conclusion premise: Mary's plants did not thrive
axiom Mary_Plants_Did_Not_Thrive : ¬ GrowThrivingPlants Mary

-- Assuming Mary is a homeowner
axiom Mary_is_HomeOwner : HomeOwner Mary

-- Question to formalize: Did Mary seek advice from experienced gardeners?
theorem goal : SoughtAdviceFromExperiencedGardeners Mary := by
  have hH : HomeOwner Mary := Mary_is_HomeOwner
  have hUtoG : UnderstandPlantNeeds Mary → GrowThrivingPlants Mary := P4 Mary hH
  have hNotU : ¬ UnderstandPlantNeeds Mary := fun hU => Mary_Plants_Did_Not_Thrive (hUtoG hU)
  have hLtoU : LearnedBotany Mary → UnderstandPlantNeeds Mary := P3 Mary hH
  have hNotL : ¬ LearnedBotany Mary := fun hL => hNotU (hLtoU hL)
  have hOr : LearnedBotany Mary ∨ SoughtAdviceFromExperiencedGardeners Mary := P5 Mary hH
  exact Or.elim hOr (fun hL => False.elim (hNotL hL)) (fun hS => hS)
</lean>
