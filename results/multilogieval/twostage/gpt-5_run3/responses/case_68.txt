--- Iteration 1 ---

<lean>
axiom Person : Type
axiom Exam : Type

axiom Emma : Person
axiom BioExam : Exam

axiom StudiesHard : Person → Prop
axiom EnoughSleepBefore : Person → Exam → Prop
axiom PerformWell : Person → Exam → Prop
axiom AceTest : Person → Exam → Prop

-- Belief: If someone studies hard, then if they get enough sleep before an exam,
-- they will perform well and ace the test.
axiom Belief :
  ∀ p : Person, ∀ e : Exam,
    StudiesHard p → (EnoughSleepBefore p e → PerformWell p e ∧ AceTest p e)

-- Facts about Emma and her biology exam
axiom SleepFact : EnoughSleepBefore Emma BioExam
axiom NotAceFact : ¬ AceTest Emma BioExam

-- Question to formalize: Did Emma study hard?
-- From the above, this is represented as the negation of StudiesHard Emma.
theorem goal : ¬ StudiesHard Emma := by
  intro hStudy
  have hImp :
      EnoughSleepBefore Emma BioExam →
      PerformWell Emma BioExam ∧ AceTest Emma BioExam :=
    Belief Emma BioExam hStudy
  have hAce : AceTest Emma BioExam := (hImp SleepFact).right
  exact NotAceFact hAce
</lean>
