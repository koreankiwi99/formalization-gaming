--- Iteration 1 ---

<lean>
axiom Person : Type
axiom Sam : Person

axiom BroughtDish : Person → Person → Prop  -- BroughtDish host guest
axiom EnoughFood : Person → Prop            -- EnoughFood at host's party
axiom BirthdayParty : Person → Prop
axiom Cake : Person → Prop                  -- Birthday cake at host's party
axiom IceCream : Person → Prop

-- Premises
axiom R1 : ∀ p : Person, ∀ g : Person, BroughtDish p g → EnoughFood p
axiom R2 : ∀ p : Person, BirthdayParty p → Cake p
axiom R3 : ∀ p : Person, BirthdayParty p → IceCream p
axiom R4 : ∀ p : Person, (∃ g : Person, BroughtDish p g) ∨ ¬ Cake p

-- Conclusion setup: It is Sam's birthday party
axiom T_SamBirthday : BirthdayParty Sam

-- Question to formalize: Does Sam have enough food for everyone to eat?
theorem goal : EnoughFood Sam := by
  have hCake : Cake Sam := R2 Sam T_SamBirthday
  have h := R4 Sam
  cases h with
  | inl hex =>
      cases hex with
      | intro g hBD =>
          exact R1 Sam g hBD
  | inr hNotCake =>
      exact False.elim (hNotCake hCake)
</lean>
ANSWER: Yes
