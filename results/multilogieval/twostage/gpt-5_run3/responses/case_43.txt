--- Iteration 1 ---

<lean>
axiom obj : Type
axiom James : obj

axiom SubwayOnSchedule : Prop
axiom BusHasRoom : Prop

axiom TakesSubwayToWork : obj → Prop
axiom TakesBusDowntown : obj → Prop
axiom SmoothCommute : obj → Prop
axiom OnTimeToWork : obj → Prop

-- If the subway is running on schedule, then someone will take it to work.
axiom R1 : SubwayOnSchedule → ∃ x : obj, TakesSubwayToWork x

-- If the bus has room, then someone else can catch a ride downtown.
axiom R2 : BusHasRoom → ∃ x : obj, TakesBusDowntown x

-- Whether it's the subway or the bus, someone's commute will be smooth sailing.
axiom R3 : ∀ x : obj, (TakesSubwayToWork x ∨ TakesBusDowntown x) → SmoothCommute x

-- If someone takes the subway, they'll get to work on time.
axiom R4 : ∀ x : obj, TakesSubwayToWork x → OnTimeToWork x

-- Conclusion to formalize: Given that James did not take the bus downtown, did he get to work on time?
theorem goal : ¬ TakesBusDowntown James → OnTimeToWork James := sorry
</lean>
