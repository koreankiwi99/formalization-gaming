STAGE 2: PROVING

--- Iteration 1 ---

<lean>
axiom MarketingFocus : Prop
axiom InvestedInMarketing : Prop
axiom MoreCustomers : Prop
axiom SalesIncreased : Prop
axiom ProductionCostsIncreased : Prop

axiom R_focus_invest : MarketingFocus → InvestedInMarketing
axiom R_invest_customers : InvestedInMarketing → MoreCustomers
axiom R_customers_sales : MoreCustomers → SalesIncreased

axiom disj : MarketingFocus ∨ ProductionCostsIncreased

theorem goal : ¬ SalesIncreased → ProductionCostsIncreased :=
by
  intro hns
  cases disj with
  | inl hf =>
      have hi : InvestedInMarketing := R_focus_invest hf
      have hc : MoreCustomers := R_invest_customers hi
      have hs : SalesIncreased := R_customers_sales hc
      exact False.elim (hns hs)
  | inr hpc =>
      exact hpc
</lean>
ANSWER: Yes
