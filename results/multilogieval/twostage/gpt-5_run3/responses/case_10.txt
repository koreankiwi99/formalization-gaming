--- Iteration 1 ---

<lean>
axiom obj : Type

-- Individuals
axiom Alex : obj
axiom Student0 : obj  -- the specific student mentioned in the premises

-- Predicates
axiom Student : obj → Prop
axiom StudiedHard : obj → Prop
axiom Participated : obj → Prop
axiom GoodGrades : obj → Prop
axiom MadeNewFriends : obj → Prop
axiom Invited : obj → Prop

-- Premises
axiom P1 : Student Student0
axiom P2 : StudiedHard Student0 → GoodGrades Student0
axiom P3 : StudiedHard Student0 ∨ Participated Student0
axiom P4 : ∀ x : obj, Participated x → MadeNewFriends x
axiom P5 : ∀ x : obj, MadeNewFriends x → Invited x

-- Additional identification to relate Alex to the mentioned student
axiom AlexIsStudent0 : Alex = Student0

-- Conclusion to formalize (as a question about Alex)
theorem goal : ¬ GoodGrades Alex → Invited Alex :=
by
  intro hneg
  have disj : StudiedHard Alex ∨ Participated Alex := by
    simpa [AlexIsStudent0] using P3
  cases disj with
  | inl hs =>
      have hs0 : StudiedHard Student0 := by simpa [AlexIsStudent0] using hs
      have gg0 : GoodGrades Student0 := P2 hs0
      have ggA : GoodGrades Alex := by simpa [AlexIsStudent0] using gg0
      have : False := hneg ggA
      exact False.elim this
  | inr hp =>
      have mnfA : MadeNewFriends Alex := P4 Alex hp
      exact P5 Alex mnfA
</lean>
