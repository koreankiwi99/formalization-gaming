axiom Bookshelf : obj

-- Predicates
axiom HighQualityWood : obj → Prop
axiom JointsCutPrecisely : obj → Prop
axiom Sturdy : obj → Prop
axiom LongLasting : obj → Prop

-- Premise: If the wood is high quality, then if the joints are cut precisely, the finished product will be sturdy and long-lasting.
axiom CarpentryRule :
  ∀ x : obj, HighQualityWood x → (JointsCutPrecisely x → (Sturdy x ∧ LongLasting x))

-- Given facts about the specific bookshelf
axiom PreciseJoints_Bookshelf : JointsCutPrecisely Bookshelf
axiom NotSturdy_Bookshelf : ¬ Sturdy Bookshelf
axiom NotLongLasting_Bookshelf : ¬ LongLasting Bookshelf

-- Question/Goal: Was the wood used of high quality?
theorem goal : ¬ HighQualityWood Bookshelf := by
  intro hq
  have h := CarpentryRule Bookshelf hq PreciseJoints_Bookshelf
  have hs : Sturdy Bookshelf := And.left h
  exact NotSturdy_Bookshelf hs
