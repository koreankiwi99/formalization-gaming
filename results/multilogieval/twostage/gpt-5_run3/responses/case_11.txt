--- Iteration 1 ---

<lean>
axiom obj : Type
axiom Project : obj

axiom NiceDisplay : obj → Prop
axiom JudgesImpressed : obj → Prop
axiom ThoroughResearch : obj → Prop
axiom TakenSeriouslyByExperts : obj → Prop
axiom Awarded : obj → Prop

-- If a display looks nice, judges will be impressed.
axiom R1 : ∀ x : obj, NiceDisplay x → JudgesImpressed x
-- If research is thorough, experts will take the project seriously.
axiom R2 : ∀ x : obj, ThoroughResearch x → TakenSeriouslyByExperts x
-- The project has either a good display or thorough research.
axiom R3 : NiceDisplay Project ∨ ThoroughResearch Project
-- If experts take a project seriously, it will receive an award.
axiom R4 : ∀ x : obj, TakenSeriouslyByExperts x → Awarded x
-- Given: the judges were not impressed with the display.
axiom NotImpressed : ¬ JudgesImpressed Project

theorem goal : Awarded Project := by
  cases R3 with
  | inl nd =>
      have ji : JudgesImpressed Project := R1 Project nd
      exact False.elim (NotImpressed ji)
  | inr tr =>
      exact R4 Project (R2 Project tr)
</lean>
ANSWER: Yes
