<lean>
axiom Practice : Prop
axiom Prepared : Prop
axiom Hot : Prop
axiom Postponed : Prop
axiom PlayWell : Prop
axiom Win : Prop

axiom A1 : Practice → Prepared
axiom A2 : Hot → Postponed
axiom A3 : Practice ∨ ¬ Postponed
axiom A4 : Prepared → PlayWell
axiom A5 : PlayWell → Win

theorem goal : Hot → Win :=
by
  intro hHot
  have hPost : Postponed := A2 hHot
  cases A3 with
  | inl hPractice =>
      have hPrepared : Prepared := A1 hPractice
      have hPlay : PlayWell := A4 hPrepared
      exact A5 hPlay
  | inr hNotPost =>
      exact False.elim (hNotPost hPost)
</lean>
