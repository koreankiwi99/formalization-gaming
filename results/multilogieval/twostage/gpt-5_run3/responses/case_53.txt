--- Iteration 1 ---

<lean>
axiom obj : Type
axiom Mary : obj

axiom WantedToWinArtCompetition : obj → Prop
axiom PracticedSkillsOften : obj → Prop
axiom TechniqueImproved : obj → Prop
axiom ArtworkStoodOutHighQuality : obj → Prop
axiom TookArtClass : obj → Prop
axiom LearnedNewMethodsAndMediums : obj → Prop

-- Premises
axiom P1 : WantedToWinArtCompetition Mary
axiom R1 : ∀ x : obj, PracticedSkillsOften x → TechniqueImproved x
axiom R2 : ∀ x : obj, TechniqueImproved x → ArtworkStoodOutHighQuality x
axiom D1 : PracticedSkillsOften Mary ∨ TookArtClass Mary
axiom R3 : ∀ x : obj, TookArtClass x → LearnedNewMethodsAndMediums x

-- Additional given in the conclusion
axiom NotHQ : ¬ ArtworkStoodOutHighQuality Mary

-- Question to resolve: Did Mary learn new methods and mediums for creating art?
theorem goal : LearnedNewMethodsAndMediums Mary :=
by
  have hnotPSO : ¬ PracticedSkillsOften Mary :=
    fun hps => NotHQ (R2 Mary (R1 Mary hps))
  have hTAC : TookArtClass Mary :=
    Or.elim D1
      (fun hps => False.elim (hnotPSO hps))
      (fun htac => htac)
  exact R3 Mary hTAC
</lean>
ANSWER: Yes
