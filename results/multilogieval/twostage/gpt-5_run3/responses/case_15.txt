--- Iteration 1 ---

<lean>
axiom obj : Type
axiom Student : obj

axiom StudyDiligently : obj → Prop
axiom ParticipateActively : obj → Prop
axiom ExcelsAcademically : obj → Prop
axiom Leader : obj → Prop
axiom SelectedAsClassPrefect : obj → Prop

-- If someone studies diligently, they will excel academically.
axiom R1 : ∀ x : obj, StudyDiligently x → ExcelsAcademically x

-- If someone participates actively, they will gain leadership skills (be a leader).
axiom R2 : ∀ x : obj, ParticipateActively x → Leader x

-- The student either studies diligently or participates actively.
axiom T1 : StudyDiligently Student ∨ ParticipateActively Student

-- If someone is a leader, they will be selected as class prefect.
axiom R3 : ∀ x : obj, Leader x → SelectedAsClassPrefect x

-- Conclusion to formalize: Given that the student did not excel academically, were they chosen as class prefect?
theorem goal : ¬ExcelsAcademically Student → SelectedAsClassPrefect Student :=
by
  intro hNot
  cases T1 with
  | inl hStudy =>
      have hExc : ExcelsAcademically Student := R1 Student hStudy
      have : False := hNot hExc
      exact False.elim this
  | inr hPart =>
      have hLeader : Leader Student := R2 Student hPart
      exact R3 Student hLeader
</lean>
ANSWER: Yes
