<lean>
axiom obj : Type
axiom Casey : obj

axiom VolunteersAtShelter : obj → Prop
axiom HelpsCommunity : obj → Prop
axiom AdoptsPetFromShelter : obj → Prop
axiom ProvidesHomeToPetInNeed : obj → Prop
axiom FeelsSenseOfFulfillment : obj → Prop

axiom R1 : ∀ x : obj, VolunteersAtShelter x → HelpsCommunity x
axiom R2 : ∀ x : obj, AdoptsPetFromShelter x → ProvidesHomeToPetInNeed x
axiom R3 : ∀ x : obj, VolunteersAtShelter x ∨ ¬ ProvidesHomeToPetInNeed x ∨ (VolunteersAtShelter x ∧ ¬ ProvidesHomeToPetInNeed x)
axiom R4 : ∀ x : obj, HelpsCommunity x → FeelsSenseOfFulfillment x

theorem goal : AdoptsPetFromShelter Casey → FeelsSenseOfFulfillment Casey := by
  intro hAdopt
  have hHome : ProvidesHomeToPetInNeed Casey := R2 Casey hAdopt
  have h := R3 Casey
  cases h with
  | inl hVol =>
      exact R4 Casey (R1 Casey hVol)
  | inr h2 =>
      cases h2 with
      | inl hNotHome =>
          have : False := hNotHome hHome
          exact False.elim this
      | inr hConj =>
          have hVol : VolunteersAtShelter Casey := And.left hConj
          exact R4 Casey (R1 Casey hVol)
</lean>
