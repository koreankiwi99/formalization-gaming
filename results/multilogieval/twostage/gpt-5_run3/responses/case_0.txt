<lean>
axiom obj : Type
axiom Someone : obj

axiom HopedToLiveInDreamHome : obj → Prop
axiom SavedAggressively : obj → Prop
axiom CouldAffordDownPayment : obj → Prop
axiom ImprovedCreditScore : obj → Prop
axiom QualifiedForMortgage : obj → Prop
axiom BoughtDreamHome : obj → Prop

-- Premises
axiom P1 : HopedToLiveInDreamHome Someone
axiom P2 : ∀ x : obj, SavedAggressively x → CouldAffordDownPayment x
axiom P3 : ∀ x : obj, ImprovedCreditScore x → QualifiedForMortgage x
axiom P4 : SavedAggressively Someone ∨ ImprovedCreditScore Someone
axiom P5 : ∀ x : obj, QualifiedForMortgage x → BoughtDreamHome x

-- Given condition
axiom G1 : ¬ CouldAffordDownPayment Someone

-- Conclusion to formalize
theorem goal : BoughtDreamHome Someone :=
by
  have h := P4
  apply Or.elim h
  · intro hsave
    have hcap : CouldAffordDownPayment Someone := P2 Someone hsave
    have hfalse : False := G1 hcap
    exact False.elim hfalse
  · intro himprove
    have hq : QualifiedForMortgage Someone := P3 Someone himprove
    exact P5 Someone hq
