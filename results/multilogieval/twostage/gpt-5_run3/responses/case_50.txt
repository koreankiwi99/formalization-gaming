--- Iteration 1 ---

<lean>
axiom Person : Type
axiom Jesse : Person

axiom SprintingFrequently : Person → Prop
axiom StretchingClasses : Person → Prop
axiom RunningImproved : Person → Prop
axiom GoalTimeAchieved : Person → Prop
axiom Flexible : Person → Prop
axiom LessProneToInjury : Person → Prop

-- If someone does sprint workouts several times a week, their running form and stamina improve.
axiom R1 : ∀ x : Person, SprintingFrequently x → RunningImproved x

-- If someone's running improved, they could achieve their goal time at the big race.
axiom R2 : ∀ x : Person, RunningImproved x → GoalTimeAchieved x

-- Either Jesse was sprinting frequently, or Jesse was taking stretching classes, or both.
axiom R3 : SprintingFrequently Jesse ∨ StretchingClasses Jesse

-- If someone takes stretching classes, they become more flexible and less prone to injury.
axiom R4 : ∀ x : Person, StretchingClasses x → (Flexible x ∧ LessProneToInjury x)

-- Conclusion part 1: Improving running form and stamina did not happen for Jesse.
axiom NotImprovedJesse : ¬ RunningImproved Jesse

-- Conclusion query: Did Jesse become more flexible and less prone to injury?
theorem goal : Flexible Jesse ∧ LessProneToInjury Jesse :=
by
  cases R3 with
  | inl sf =>
      have hri : RunningImproved Jesse := R1 Jesse sf
      exact False.elim (NotImprovedJesse hri)
  | inr sc =>
      exact R4 Jesse sc
</lean>
ANSWER: Yes
