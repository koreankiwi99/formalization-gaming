--- Iteration 1 ---

<lean>
axiom obj : Type

axiom Jake : obj

axiom WeatherNice : Prop
axiom RelaxOutside : obj → Prop
axiom ReadBooks : obj → Prop
axiom PackEnough : obj → Prop
axiom Prepared : obj → Prop

-- Premises
axiom P1 : ∃ x : obj, Prepared x
axiom P2 : ∀ x : obj, WeatherNice → RelaxOutside x
axiom P3 : ∀ x : obj, ReadBooks x → PackEnough x
axiom P4 : ∀ x : obj, WeatherNice ∨ ¬ PackEnough x ∨ (WeatherNice ∧ ¬ PackEnough x)
axiom P5 : ∀ x : obj, Prepared x → ReadBooks x

-- Conclusion to formalize
theorem goal : ¬ RelaxOutside Jake → Prepared Jake :=
by
  intro hnr
  -- Derive WeatherNice from the existence of a prepared person and P4
  have hW : WeatherNice :=
    by
      cases P1 with
      | intro x hxPrep =>
        have hxRB : ReadBooks x := P5 x hxPrep
        have hxPE : PackEnough x := P3 x hxRB
        have disj := P4 x
        cases disj with
        | inl w => exact w
        | inr rest =>
          cases rest with
          | inl notPE =>
            exact False.elim (notPE hxPE)
          | inr w_and_notPE =>
            exact w_and_notPE.left
  -- From ¬RelaxOutside Jake and P2, deduce ¬WeatherNice
  have notW : ¬ WeatherNice :=
    by
      intro w
      have ro : RelaxOutside Jake := P2 Jake w
      exact hnr ro
  -- Contradiction yields any proposition, in particular Prepared Jake
  exact False.elim (notW hW)
</lean>
