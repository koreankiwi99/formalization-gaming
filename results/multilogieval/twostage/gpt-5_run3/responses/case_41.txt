--- Iteration 1 ---

<lean>
axiom obj : Type

-- Individuals
axiom Priya : obj

-- Predicates
axiom WatersDaily : obj → Prop
axiom GrowWell : obj → Prop
axiom Drought : Prop
axiom WaterRestrictions : Prop
axiom LackedWater : obj → Prop
axiom WiltDie : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, WatersDaily x → GrowWell x
axiom P2 : Drought → WaterRestrictions
axiom P3 : WatersDaily Priya ∨ Drought
axiom P4 : WaterRestrictions → ∀ x : obj, LackedWater x
axiom P5 : ∀ x : obj, LackedWater x → WiltDie x

-- Conclusion to formalize: If Priya's plants did not grow well, did they wilt and die?
theorem goal : (¬ GrowWell Priya) → WiltDie Priya :=
by
  intro hngw
  cases P3 with
  | inl hWD =>
    have hg : GrowWell Priya := P1 Priya hWD
    exact False.elim (hngw hg)
  | inr hD =>
    have hWR : WaterRestrictions := P2 hD
    have hLW : LackedWater Priya := (P4 hWR) Priya
    exact P5 Priya hLW
</lean>
ANSWER: Yes
