--- Iteration 1 ---

<lean>
axiom obj : Type

-- Individuals
axiom James : obj

-- Predicates
axiom StartedNewWorkout : obj → Prop
axiom Motivated : obj → Prop
axiom GymClosedForRenovations : obj → Prop
axiom FollowRoutine : obj → Prop
axiom Frustrated : obj → Prop
axiom IndulgedUnhealthySnacks : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, StartedNewWorkout x → Motivated x
axiom P2 : ∀ x : obj, GymClosedForRenovations x → ¬ FollowRoutine x
axiom P3 : StartedNewWorkout James ∨ GymClosedForRenovations James
axiom P4 : ∀ x : obj, ¬ FollowRoutine x → Frustrated x
axiom P5 : ∀ x : obj, Frustrated x → IndulgedUnhealthySnacks x

-- Given conclusion statement (as an additional fact about James)
axiom C1 : ¬ Motivated James

-- Question to formalize as a theorem goal
theorem goal : IndulgedUnhealthySnacks James :=
by
  cases P3 with
  | inl hstart =>
      have hm : Motivated James := P1 James hstart
      exact False.elim (C1 hm)
  | inr hgym =>
      have hnf : ¬ FollowRoutine James := P2 James hgym
      have hf : Frustrated James := P4 James hnf
      exact P5 James hf
</lean>
ANSWER: Yes
