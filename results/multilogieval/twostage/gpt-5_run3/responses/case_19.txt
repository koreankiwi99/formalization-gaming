<lean>
axiom GreatService : Prop
axiom HostEvents : Prop
axiom ReputationGrows : Prop
axiom TrafficIncreases : Prop
axiom RevenueImproves : Prop
axiom ServiceDecreased : Prop

-- Premises
axiom P1 : GreatService → ReputationGrows
axiom P2 : HostEvents → TrafficIncreases
axiom P3 : GreatService ∨ HostEvents
axiom P4 : TrafficIncreases → RevenueImproves
axiom P5 : ServiceDecreased → ¬ GreatService

-- Conclusion to formalize: Given the service quality decreased, did the store's revenue improve?
theorem goal : ServiceDecreased → RevenueImproves := by
  intro sd
  have hng : ¬ GreatService := P5 sd
  have he : HostEvents := by
    cases P3 with
    | inl gs =>
      exact False.elim (hng gs)
    | inr he =>
      exact he
  have ti : TrafficIncreases := P2 he
  exact P4 ti
</lean>
