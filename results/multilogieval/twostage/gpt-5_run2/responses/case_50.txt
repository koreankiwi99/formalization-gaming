--- Iteration 1 ---

<lean>
axiom obj : Type
axiom Jesse : obj

axiom SprintingFrequently : obj → Prop
axiom ImprovedRunning : obj → Prop
axiom AchievedGoalTime : obj → Prop
axiom TakesStretchingClasses : obj → Prop
axiom Flexible : obj → Prop
axiom LessProneToInjury : obj → Prop

-- If someone does sprint workouts several times a week, their running form and stamina improve.
axiom R1 : ∀ x : obj, SprintingFrequently x → ImprovedRunning x
-- If someone's running improved, they could achieve their goal time at a big race.
axiom R2 : ∀ x : obj, ImprovedRunning x → AchievedGoalTime x
-- Either they were sprinting frequently, or they were taking stretching classes, or both.
axiom R3 : ∀ x : obj, SprintingFrequently x ∨ TakesStretchingClasses x
-- If someone takes stretching classes, they become more flexible and less prone to injury.
axiom R4 : ∀ x : obj, TakesStretchingClasses x → Flexible x ∧ LessProneToInjury x

-- Conclusion fact: Improving running form and stamina did not happen for Jesse.
axiom NotImproved_Jesse : ¬ ImprovedRunning Jesse

-- Question: Did Jesse become more flexible and less prone to injury?
theorem goal : Flexible Jesse ∧ LessProneToInjury Jesse :=
by
  have hTSC : TakesStretchingClasses Jesse :=
    Or.elim (R3 Jesse)
      (fun hSF => False.elim (NotImproved_Jesse (R1 Jesse hSF)))
      (fun hSC => hSC)
  exact R4 Jesse hTSC
</lean>
ANSWER: Yes
