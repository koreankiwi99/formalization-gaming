--- Iteration 1 ---

<lean>
axiom Person : Type
axiom Exam : Type

axiom Alex : Person
axiom BiologyExam : Exam
axiom HistoryExam : Exam

axiom StudiedHard : Person → Exam → Prop
axiom Confident : Person → Exam → Prop
axiom DidWell : Person → Exam → Prop
axiom Satisfied : Person → Exam → Prop

axiom SevereWeatherInForecast : Prop
axiom ClassCanceledDueToWeather : Prop

-- If someone studies hard for an exam, they will feel confident going into the test.
axiom P1 : ∀ p : Person, ∀ e : Exam, StudiedHard p e → Confident p e

-- If there is severe weather in the forecast, schools may cancel class that day.
axiom P2 : SevereWeatherInForecast → ClassCanceledDueToWeather

-- Either Alex studied diligently for his biology exam, or class was not canceled due to the weather.
axiom P3 : StudiedHard Alex BiologyExam ∨ ¬ ClassCanceledDueToWeather

-- If a student feels confident about the material, they will likely do well on the exam.
axiom P4 : ∀ p : Person, ∀ e : Exam, Confident p e → DidWell p e

-- If a student performs well on an exam, they will be satisfied with their hard work.
axiom P5 : ∀ p : Person, ∀ e : Exam, DidWell p e → Satisfied p e

-- Conclusion to formalize: If there was severe weather in the forecast, did Alex feel satisfied with his performance on the history exam?
theorem goal : SevereWeatherInForecast → Satisfied Alex HistoryExam := sorry
</lean>
