<lean>
axiom obj : Type
axiom City : obj

axiom Drought : obj → Prop
axiom WaterLimitsEnforced : obj → Prop
axiom WaterUsageRationed : obj → Prop

-- Premises:
-- If a city experiences a drought, authorities should implement water restrictions.
axiom P1 : ∀ x : obj, Drought x → WaterLimitsEnforced x
-- Drought conditions require rationing water usage.
axiom P2 : ∀ x : obj, Drought x → WaterUsageRationed x

-- Given conclusion part: Either the city did not enforce water limits, or it did not ration water usage.
axiom Given : ¬ WaterLimitsEnforced City ∨ ¬ WaterUsageRationed City

-- Question to formalize: Is the city undergoing a drought?
theorem goal : ¬ Drought City :=
by
  intro hD
  have hWLE : WaterLimitsEnforced City := P1 City hD
  have hWUR : WaterUsageRationed City := P2 City hD
  cases Given with
  | inl hNotWLE => exact hNotWLE hWLE
  | inr hNotWUR => exact hNotWUR hWUR
</lean>
