<lean>
axiom obj : Type
axiom Student : obj

axiom StudiesDiligently : obj → Prop
axiom ParticipatesActively : obj → Prop
axiom ExcelsAcademically : obj → Prop
axiom Leader : obj → Prop
axiom ChosenAsClassPrefect : obj → Prop

axiom R1 : ∀ x : obj, StudiesDiligently x → ExcelsAcademically x
axiom R2 : ∀ x : obj, ParticipatesActively x → Leader x
axiom R3 : StudiesDiligently Student ∨ ParticipatesActively Student
axiom R4 : ∀ x : obj, Leader x → ChosenAsClassPrefect x

theorem goal : ¬ ExcelsAcademically Student → ChosenAsClassPrefect Student :=
by
  intro hne
  cases R3 with
  | inl hS =>
      have hx : ExcelsAcademically Student := R1 Student hS
      exact False.elim (hne hx)
  | inr hP =>
      have hL : Leader Student := R2 Student hP
      exact R4 Student hL
</lean>
