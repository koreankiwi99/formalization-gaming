<lean>
axiom Oven350 : Prop
axiom BakesProperly : Prop
axiom BatterHasEggs : Prop
axiom FluffyWhenBaked : Prop
axiom TastesDelicious : Prop
axiom GuestsHappy : Prop
axiom SamSetCorrect : Prop

axiom R1 : Oven350 → BakesProperly
axiom R2 : BatterHasEggs → FluffyWhenBaked
axiom R3 : SamSetCorrect ∨ ¬ BatterHasEggs
axiom R4 : BakesProperly → TastesDelicious
axiom R5 : TastesDelicious → GuestsHappy
axiom R6 : SamSetCorrect → Oven350

theorem goal : BatterHasEggs → GuestsHappy := by
  intro hEggs
  cases R3 with
  | inl hSam =>
      have hOven : Oven350 := R6 hSam
      have hBake : BakesProperly := R1 hOven
      have hTaste : TastesDelicious := R4 hBake
      exact R5 hTaste
  | inr hNoEggs =>
      have hFalse : False := hNoEggs hEggs
      exact False.elim hFalse
</lean>
