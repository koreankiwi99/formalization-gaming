<lean>
axiom obj : Type
axiom Project1 : obj

axiom NiceDisplay : obj → Prop
axiom JudgesImpressed : obj → Prop
axiom ThoroughResearch : obj → Prop
axiom ExpertsTakeSeriously : obj → Prop
axiom Awarded : obj → Prop

-- If a project's display looks nice, judges will be impressed.
axiom R1 : ∀ x : obj, NiceDisplay x → JudgesImpressed x
-- If a project's research is thorough, experts will take it seriously.
axiom R2 : ∀ x : obj, ThoroughResearch x → ExpertsTakeSeriously x
-- The project either has a nice display or thorough research.
axiom D1 : NiceDisplay Project1 ∨ ThoroughResearch Project1
-- If experts take a project seriously, it will receive an award.
axiom R3 : ∀ x : obj, ExpertsTakeSeriously x → Awarded x
-- Given: judges were not impressed with the display of this project.
axiom NotImpressed : ¬ JudgesImpressed Project1

theorem goal : Awarded Project1 :=
by
  apply Or.elim D1
  · intro nd
    have ji : JudgesImpressed Project1 := R1 Project1 nd
    exact False.elim (NotImpressed ji)
  · intro tr
    have ets : ExpertsTakeSeriously Project1 := R2 Project1 tr
    exact R3 Project1 ets
</lean>

ANSWER: Yes
