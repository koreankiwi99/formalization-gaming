<lean>
axiom Rain : Prop
axiom WateredNaturally : Prop
axiom VeryHot : Prop
axiom PlayInSprinklers : Prop

-- Premises
axiom R1 : Rain → WateredNaturally
axiom R2 : VeryHot → PlayInSprinklers
axiom R3 : Rain ∨ ¬ PlayInSprinklers

-- Conclusion assertion: The weather was very hot.
axiom HotToday : VeryHot

-- Question: Did the grass and plants get watered naturally today?
theorem goal : WateredNaturally :=
by
  cases R3 with
  | inl r =>
      exact R1 r
  | inr hnps =>
      have pips : PlayInSprinklers := R2 HotToday
      exact False.elim (hnps pips)
</lean>
ANSWER: Yes
