--- Iteration 1 ---

<lean>
axiom Person : Type
axiom Priya : Person

axiom WatersDaily : Person → Prop
axiom GrowWell : Person → Prop
axiom Drought : Prop
axiom Restrictions : Prop
axiom NotEnoughWater : Person → Prop
axiom WiltAndDie : Person → Prop

-- Premises
axiom P1 : ∀ p : Person, WatersDaily p → GrowWell p
axiom P2 : Drought → Restrictions
axiom P3 : WatersDaily Priya ∨ Drought ∨ (WatersDaily Priya ∧ Drought)
axiom P4 : ∀ p : Person, Restrictions → NotEnoughWater p
axiom P5 : ∀ p : Person, NotEnoughWater p → WiltAndDie p

-- Conclusion to formalize: If Priya's plants did not grow well, then did they wilt and die?
theorem goal : (¬ GrowWell Priya) → WiltAndDie Priya :=
by
  intro hnot
  exact
    Or.elim P3
      (fun hWD => False.elim (hnot (P1 Priya hWD)))
      (fun h2 =>
        Or.elim h2
          (fun hD => P5 Priya (P4 Priya (P2 hD)))
          (fun hBoth => P5 Priya (P4 Priya (P2 hBoth.right))))
</lean>
ANSWER: Yes
