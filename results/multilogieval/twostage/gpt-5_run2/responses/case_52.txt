--- Iteration 1 ---

<lean>
axiom obj : Type
axiom James : obj

axiom ForecastRain : Prop
axiom BigGame : Prop
axiom FriendsGather : Prop
axiom HighViewership : Prop
axiom GoPlayOut : Prop

axiom BringUmbrella : obj → Prop
axiom GoOut : obj → Prop

-- If the forecast predicts rain, then people will bring an umbrella when they go out.
axiom P1 : ForecastRain → ∀ x : obj, GoOut x → BringUmbrella x

-- If there is a big game on TV, then friends will get together to watch it.
axiom P2 : BigGame → FriendsGather

-- Either rain is forecasted, or friends did not get together to watch the game, or both.
axiom P3 : ForecastRain ∨ ¬ FriendsGather ∨ (ForecastRain ∧ ¬ FriendsGather)

-- If the TV viewership is high, then there is a big game on TV.
axiom P4 : HighViewership → BigGame

-- Either the TV viewership is high, or they all decide to go play out, or both.
axiom P5 : HighViewership ∨ GoPlayOut ∨ (HighViewership ∧ GoPlayOut)

-- Conclusion to formalize: From "James did not bring an umbrella", does it entail that they go out to play?
theorem goal : (¬ BringUmbrella James) → GoPlayOut := sorry
</lean>
