--- Iteration 1 ---

<lean>
axiom obj : Type
axiom Party : obj → Prop
axiom Drums : obj → Prop
axiom NoisyHouse : obj → Prop
axiom NoiseUp : Prop
axiom NeighborsComplain : Prop
axiom PoliceCalled : Prop

-- If someone hosts a party, their house gets very noisy.
axiom A1 : ∀ x : obj, Party x → NoisyHouse x
-- If someone is practicing drums, the noise levels go up.
axiom A2 : ∀ x : obj, Drums x → NoiseUp
-- Either there is a party, or someone is playing drums (or both).
axiom A3 : (∃ x : obj, Party x) ∨ (∃ x : obj, Drums x)
-- If the house gets noisy from a party, neighbors start complaining.
axiom A4 : ∀ x : obj, (Party x ∧ NoisyHouse x) → NeighborsComplain
-- If neighbors complain, the police are called to intervene.
axiom A5 : NeighborsComplain → PoliceCalled

-- Conclusion: If there was no noise level up, were the police called to intervene?
theorem goal : (¬ NoiseUp) → PoliceCalled := by
  intro hNot
  cases A3 with
  | inl hPartyExists =>
    cases hPartyExists with
    | intro x hx =>
      have noisy : NoisyHouse x := A1 x hx
      have compl : NeighborsComplain := A4 x (And.intro hx noisy)
      exact A5 compl
  | inr hDrumsExists =>
    cases hDrumsExists with
    | intro x hx =>
      have up : NoiseUp := A2 x hx
      exact False.elim (hNot up)
</lean>
ANSWER: Yes
