<lean>
axiom Company : Type
axiom Firm : Company

-- Predicates about last quarter
axiom InvestedInMarketingLastQuarter : Company → Prop
axiom FocusedBudgetOnMarketingLastQuarter : Company → Prop
axiom GainedMoreCustomersLastQuarter : Company → Prop
axiom SalesIncreasedLastQuarter : Company → Prop
axiom ProductionCostsIncreasedLastQuarter : Company → Prop

-- Premises:
-- If a company invested in marketing (ads/promotions), it gained more customers.
axiom R1 : ∀ x : Company, InvestedInMarketingLastQuarter x → GainedMoreCustomersLastQuarter x
-- With more customers, sales numbers go up.
axiom R2 : ∀ x : Company, GainedMoreCustomersLastQuarter x → SalesIncreasedLastQuarter x
-- Focusing budget on marketing implies investing in marketing.
axiom R3 : ∀ x : Company, FocusedBudgetOnMarketingLastQuarter x → InvestedInMarketingLastQuarter x
-- Either they focused budget on marketing, or production costs increased, or both.
axiom R4 : ∀ x : Company, (FocusedBudgetOnMarketingLastQuarter x) ∨ (ProductionCostsIncreasedLastQuarter x)

-- Conclusion to formalize: If sales did not increase last quarter, then did production costs increase?
theorem goal : ¬ SalesIncreasedLastQuarter Firm → ProductionCostsIncreasedLastQuarter Firm := sorry
</lean>

<lean>
axiom Company : Type
axiom Firm : Company

-- Predicates about last quarter
axiom InvestedInMarketingLastQuarter : Company → Prop
axiom FocusedBudgetOnMarketingLastQuarter : Company → Prop
axiom GainedMoreCustomersLastQuarter : Company → Prop
axiom SalesIncreasedLastQuarter : Company → Prop
axiom ProductionCostsIncreasedLastQuarter : Company → Prop

-- Premises:
-- If a company invested in marketing (ads/promotions), it gained more customers.
axiom R1 : ∀ x : Company, InvestedInMarketingLastQuarter x → GainedMoreCustomersLastQuarter x
-- With more customers, sales numbers go up.
axiom R2 : ∀ x : Company, GainedMoreCustomersLastQuarter x → SalesIncreasedLastQuarter x
-- Focusing budget on marketing implies investing in marketing.
axiom R3 : ∀ x : Company, FocusedBudgetOnMarketingLastQuarter x → InvestedInMarketingLastQuarter x
-- Either they focused budget on marketing, or production costs increased, or both.
axiom R4 : ∀ x : Company, (FocusedBudgetOnMarketingLastQuarter x) ∨ (ProductionCostsIncreasedLastQuarter x)

-- Conclusion to formalize: If sales did not increase last quarter, then did production costs increase?
theorem goal : ¬ SalesIncreasedLastQuarter Firm → ProductionCostsIncreasedLastQuarter Firm :=
by
  intro h
  have hdisj := R4 Firm
  cases hdisj with
  | inl hf =>
      have hs : SalesIncreasedLastQuarter Firm := R2 Firm (R1 Firm (R3 Firm hf))
      exact False.elim (h hs)
  | inr hp =>
      exact hp
</lean>
ANSWER: Yes
