--- Iteration 1 ---

<lean>
axiom obj : Type
axiom James : obj

axiom SubwayRunningOnSchedule : Prop
axiom BusHasRoom : Prop

axiom TakesSubway : obj → Prop
axiom TakesBusDowntown : obj → Prop
axiom OnTime : obj → Prop
axiom SmoothCommute : obj → Prop

-- If the subway is running on schedule, then someone will take it to work.
axiom P1 : SubwayRunningOnSchedule → ∃ x : obj, TakesSubway x

-- If the bus has room, then someone can catch a ride downtown.
axiom P2 : BusHasRoom → ∃ x : obj, TakesBusDowntown x

-- Whether it's the subway or the bus, someone's commute will be smooth sailing.
axiom P3 : ∀ x : obj, (TakesSubway x ∨ TakesBusDowntown x) → SmoothCommute x

-- If someone takes the subway, they'll get to work on time.
axiom P4 : ∀ x : obj, TakesSubway x → OnTime x

-- Conclusion to formalize: Given James did not take the bus downtown, did he get to work on time?
theorem goal : ¬ TakesBusDowntown James → OnTime James := sorry
</lean>
