axiom P2 : ∀ x : obj, PracticedOften x → TechniqueImproved x
axiom P3 : ∀ x : obj, TechniqueImproved x → ArtworkHighQuality x

theorem goal : LearnedNewMethodsAndMediums Mary :=
by
  cases P4 with
  | inl hpo =>
      have hti : TechniqueImproved Mary := P2 Mary hpo
