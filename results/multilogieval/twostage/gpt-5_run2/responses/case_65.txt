axiom R1 : ∀ w p : obj, HighQuality w → UsedIn w p → PreciselyCutJoints p → Sturdy p ∧ LongLasting p

-- Interpreting "shaky" and "fell apart quickly"
axiom A1 : ∀ p : obj, Shaky p → ¬ Sturdy p
