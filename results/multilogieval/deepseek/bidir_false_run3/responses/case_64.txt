<lean>
axiom Student : Type
axiom Priya : Student
axiom AttendsRegularly : Student → Prop
axiom SufficientRevising : Student → Prop
axiom PerformsWell : Student → Prop
axiom Context : ∀ (s : Student), AttendsRegularly s → (SufficientRevising s → PerformsWell s)
axiom Fact1 : SufficientRevising Priya
axiom Fact2 : ¬ PerformsWell Priya

theorem Priya_not_attends : ¬ AttendsRegularly Priya :=
  fun (hAttend : AttendsRegularly Priya) =>
    have impl : SufficientRevising Priya → PerformsWell Priya := Context Priya hAttend
    have performsWell : PerformsWell Priya := impl Fact1
    Fact2 performsWell
</lean>

ANSWER: No
