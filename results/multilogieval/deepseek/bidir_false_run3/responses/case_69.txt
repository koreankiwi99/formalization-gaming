<lean>
axiom Athlete : Type
axiom Lebron : Athlete
axiom TrainHard : Athlete → Prop
axiom Focused : Athlete → Prop
axiom PerformBest : Athlete → Prop

axiom coach_preach : ∀ (a : Athlete), TrainHard a → (Focused a → PerformBest a)
axiom focused_Lebron : Focused Lebron
axiom not_perform_Lebron : ¬ PerformBest Lebron

theorem not_trained_hard : ¬ TrainHard Lebron :=
  fun (h_train : TrainHard Lebron) =>
    have h1 : Focused Lebron → PerformBest Lebron := coach_preach Lebron h_train
    have h2 : PerformBest Lebron := h1 focused_Lebron
    not_perform_Lebron h2
</lean>

ANSWER: No
