The Lean code defines a structure `Model` to represent the propositions in the reasoning problem. A specific model `M0` is created where:
- `peaches` (farmer's market has ripe peaches) is False
- `in_season` (peaches in season) is True
- `bake_pie` (someone bakes pie) is False
- `won_contest` (someone won contest) is False
- `smell_amazing` (house smells amazing) is False
- `cozy` (house feels cozy) is False
- `celebrate` (winner celebrates) is False
- `james_celebrate` (James celebrates) is False

Theorems `P1_M0` to `P5_M0` prove that in `M0`:
1. `peaches → bake_pie` holds (vacuously, as `peaches` is False)
2. `won_contest → celebrate` holds (vacuously)
3. `in_season ∨ won_contest` holds (since `in_season` is True)
4. `bake_pie → smell_amazing` holds (vacuously)
5. `smell_amazing → cozy` holds (vacuously)

The key theorem `counterexample_found` proves that the universal claim "∀ (M : Model), [premises] → (¬M.james_celebrate → M.cozy)" is false. It does this by:
1. Assuming the universal claim
2. Specializing it to `M0`
3. Showing `¬M0.james_celebrate` holds (since `james_celebrate` is False)
4. Deriving `M0.cozy` (which is False, leading to contradiction)

This exhibits a model satisfying all premises where James didn't celebrate (`¬james_celebrate` is True) but the house wasn't cozy (`cozy` is False), falsifying the implication.

structure Model where
  peaches : Prop
  in_season : Prop
  bake_pie : Prop
  won_contest : Prop
  smell_amazing : Prop
  cozy : Prop
  celebrate : Prop
  james_celebrate : Prop

def M0 : Model := {
  peaches := False,
  in_season := True,
  bake_pie := False,
  won_contest := False,
  smell_amazing := False,
  cozy := False,
  celebrate := False,
  james_celebrate := False
}

theorem P1_M0 : M0.peaches → M0.bake_pie := 
  fun h => False.elim h

theorem P2_M0 : M0.won_contest → M0.celebrate := 
  fun h => False.elim h

theorem P3_M0 : M0.in_season ∨ M0.won_contest := 
  Or.inl True.intro

theorem P4_M0 : M0.bake_pie → M0.smell_amazing := 
  fun h => False.elim h

theorem P5_M0 : M0.smell_amazing → M0.cozy := 
  fun h => False.elim h

theorem counterexample_found: 
  ¬ (∀ (M : Model), 
    (M.peaches → M.bake_pie) → 
    (M.won_contest → M.celebrate) → 
    (M.in_season ∨ M.won_contest) → 
    (M.bake_pie → M.smell_amazing) → 
    (M.smell_amazing → M.cozy) → 
    (¬M.james_celebrate → M.cozy)) :=
  by
  intro H
  have H0 := H M0 P1_M0 P2_M0 P3_M0 P4_M0 P5_M0
  have : ¬M0.james_celebrate := by 
    intro j
    exact j
  have : M0.cozy := H0 this
  exact this

end
