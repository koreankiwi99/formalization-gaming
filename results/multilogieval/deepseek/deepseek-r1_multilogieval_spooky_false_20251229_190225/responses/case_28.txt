<lean>
axiom Person : Type
axiom Jane : Person
axiom Hurt : Person → Prop
axiom SeeDoctor : Person → Prop
axiom RestInjury : Person → Prop

axiom Premise1 : ∀ (x : Person), Hurt x → SeeDoctor x
axiom Premise2 : ∀ (x : Person), Hurt x → RestInjury x
axiom Given : ¬ SeeDoctor Jane ∨ ¬ RestInjury Jane

theorem jane_not_hurt : ¬ Hurt Jane := by
  intro hHurt
  have seeDoc : SeeDoctor Jane := Premise1 Jane hHurt
  have restInj : RestInjury Jane := Premise2 Jane hHurt
  cases Given with
  | inl noSee => exact noSee seeDoc
  | inr noRest => exact noRest restInj
</lean>

ANSWER: No
