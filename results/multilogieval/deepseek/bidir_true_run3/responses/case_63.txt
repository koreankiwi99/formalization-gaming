<lean>
axiom Gardener : Type
axiom Linda : Gardener
axiom PlantsDiseaseResistant : Gardener → Prop
axiom WatersAndFeedsProperly : Gardener → Prop
axiom BountifulHarvest : Gardener → Prop

-- Context: If someone plants disease-resistant varieties, then proper watering/feeding leads to bountiful harvest
axiom gardener_rule : ∀ (x : Gardener), PlantsDiseaseResistant x → (WatersAndFeedsProperly x → BountifulHarvest x)

-- Given facts
axiom watered_fed : WatersAndFeedsProperly Linda
axiom no_harvest : ¬ BountifulHarvest Linda

-- Proving Linda did NOT plant disease-resistant varieties
theorem linda_no_resistant : ¬ PlantsDiseaseResistant Linda := by
  intro h_resistant  -- Assume Linda planted disease-resistant varieties
  have h_implication : WatersAndFeedsProperly Linda → BountifulHarvest Linda := gardener_rule Linda h_resistant
  have h_harvest : BountifulHarvest Linda := h_implication watered_fed
  exact no_harvest h_harvest  -- Contradiction
</lean>

ANSWER: Yes
