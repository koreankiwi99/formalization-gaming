<lean>
axiom Habit : Prop  -- Team practices every day (as a habit, leading to preparation)
axiom Sunny : Prop  -- It is sunny outside
axiom Park : Prop   -- Children play at the park
axiom Prep : Prop   -- Team is prepared
axiom Conf : Prop   -- Team has confidence
axiom High : Prop   -- Team holds head up high

axiom A1 : Habit → Prep
axiom A2 : Sunny → Park
axiom A3 : Habit ∨ ¬Park
axiom A4 : Prep → Conf
axiom A5 : Conf → High

theorem sunny_implies_high : Sunny → High := by
  intro hSunny
  have hPark : Park := A2 hSunny
  cases A3 with
  | inl hHabit =>
    have hPrep : Prep := A1 hHabit
    have hConf : Conf := A4 hPrep
    exact A5 hConf
  | inr noPark =>
    contradiction  -- hPark and noPark contradict
</lean>

ANSWER: Yes
