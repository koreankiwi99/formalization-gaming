<lean>
axiom obj : Type
axiom Mary : obj

axiom PracticeOften : obj → Prop
axiom ImproveTechnique : obj → Prop
axiom StandOutQuality : obj → Prop
axiom TakeArtClass : obj → Prop
axiom LearnNewMethods : obj → Prop

axiom practice_improve : ∀ x : obj, PracticeOften x → ImproveTechnique x
axiom improve_standout : ∀ x : obj, ImproveTechnique x → StandOutQuality x
axiom practice_or_class : ∀ x : obj, PracticeOften x ∨ TakeArtClass x
axiom class_learn : ∀ x : obj, TakeArtClass x → LearnNewMethods x

axiom mary_not_standout : ¬ StandOutQuality Mary

theorem mary_learned : LearnNewMethods Mary := by
  have h_mary_choice := practice_or_class Mary
  cases h_mary_choice with
  | inl h_practice =>
    have h_improve := practice_improve Mary h_practice
    have h_standout := improve_standout Mary h_improve
    exact absurd h_standout mary_not_standout
  | inr h_class =>
    exact class_learn Mary h_class
</lean>

ANSWER: Yes
