The reasoning can be translated to Lean as follows. The conclusion follows from the premises: if there was a snowstorm, Priya's parents took her out for ice cream.

<lean>
axiom obj : Type
axiom Priya : obj
axiom StudiesHard : obj → Prop
axiom GoodGrades : obj → Prop
axiom ProudParents : obj → Prop
axiom IceCream : obj → Prop
axiom Snowstorm : Prop
axiom SchoolCanceled : Prop

axiom A1 : ∀ x : obj, StudiesHard x → GoodGrades x
axiom A2 : Snowstorm → SchoolCanceled
axiom A3 : StudiesHard Priya ∨ ¬ SchoolCanceled
axiom A4 : ∀ x : obj, GoodGrades x → ProudParents x
axiom A5 : ∀ x : obj, ProudParents x → IceCream x

theorem ice_cream_conclusion : Snowstorm → IceCream Priya := by
  intro h_snow
  have h_sc : SchoolCanceled := A2 h_snow
  rcases A3 with h_studied | h_not_sc
  · have h_grades := A1 Priya h_studied
    have h_proud := A4 Priya h_grades
    exact A5 Priya h_proud
  · contradiction
</lean>
