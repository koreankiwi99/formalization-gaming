<lean>
axiom obj : Type
axiom Alex : obj
axiom AdvertisesCleverly : obj → Prop
axiom AttractMoreCustomers : obj → Prop
axiom OffersDiscounts : obj → Prop
axiom IncreaseSales : obj → Prop
axiom ExpandBusiness : obj → Prop

axiom A1 : ∀ x : obj, AdvertisesCleverly x → AttractMoreCustomers x
axiom A2 : ∀ x : obj, OffersDiscounts x → IncreaseSales x
axiom A3 : ∀ x : obj, AdvertisesCleverly x ∨ OffersDiscounts x
axiom A4 : ∀ x : obj, IncreaseSales x → ExpandBusiness x

axiom given : ¬ AttractMoreCustomers Alex

theorem alex_expanded : ExpandBusiness Alex := by
  have no_advert : ¬ AdvertisesCleverly Alex := 
    fun h => given (A1 Alex h)
  have discount : OffersDiscounts Alex := by
    cases (A3 Alex) with
    | inl adv => contradiction
    | inr dis => exact dis
  have sales : IncreaseSales Alex := A2 Alex discount
  exact A4 Alex sales
</lean>

ANSWER: Yes
