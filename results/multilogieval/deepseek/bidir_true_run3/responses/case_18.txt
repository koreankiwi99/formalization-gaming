<lean>
axiom obj : Type
axiom Sam : obj
axiom Impressed : obj → Prop
axiom Promotion : obj → Prop
axiom Extra : obj → Prop
axiom Experience : obj → Prop
axiom Lead : obj → Prop

axiom A1 : ∀ x : obj, Impressed x → Promotion x
axiom A2 : ∀ x : obj, Extra x → Experience x
axiom A3 : Impressed Sam ∨ Extra Sam
axiom A4 : ∀ x : obj, Experience x → Lead x
axiom given : ¬ Promotion Sam

theorem sam_leads : Lead Sam := by
  have not_impressed : ¬ Impressed Sam := by
    intro h
    apply given
    exact A1 Sam h
  cases A3 with
  | inl h_imp => contradiction
  | inr h_extra =>
    have h_exp : Experience Sam := A2 Sam h_extra
    exact A4 Sam h_exp
</lean>

ANSWER: Yes
