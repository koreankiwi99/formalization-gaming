-- Premise: If there was excessive rain, reservoirs would overflow
axiom R1 : ∀ c : City, excessive_rain c → overflow c

-- Premise: If reservoirs overflow, flooding occurs
axiom R2 : ∀ c : City, overflow c → flooding c

-- Given: Heavy rainfall in Dallas

-- Prove flooding occurs in Dallas
theorem dallas_flooding : flooding Dallas := by

  exact T1
