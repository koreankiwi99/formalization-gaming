<lean>
axiom Flip : Prop
axiom Gasp : Prop
axiom High : Prop
axiom Clowns : Prop
axiom Laugh : Prop

axiom T1 : Flip → Gasp
axiom T2 : Gasp → High
axiom T3 : Flip ∨ Clowns
axiom T4 : Clowns → Laugh
axiom notHigh : ¬High

theorem kids_laughed : Laugh := by
  have noGasp : ¬Gasp := fun hGasp => notHigh (T2 hGasp)
  have noFlip : ¬Flip := fun hFlip => noGasp (T1 hFlip)
  cases T3 with
  | inl flip => contradiction
  | inr c => exact T4 c
</lean>

ANSWER: Yes
