<lean>
axiom obj : Type
axiom Alex : obj
axiom StudyHard : obj → Prop
axiom GetGoodGrades : obj → Prop
axiom ParticipateInExtracurricular : obj → Prop
axiom MakeNewFriends : obj → Prop
axiom GetInvitedToSocialEvents : obj → Prop

-- If someone studies hard, they get good grades
axiom R1 : ∀ x : obj, StudyHard x → GetGoodGrades x
-- If someone participates in extracurriculars, they make new friends
axiom R2 : ∀ x : obj, ParticipateInExtracurricular x → MakeNewFriends x
-- If someone makes new friends, they get invited to social events
axiom R3 : ∀ x : obj, MakeNewFriends x → GetInvitedToSocialEvents x
-- Alex either studied hard or participated in extracurriculars
axiom A1 : StudyHard Alex ∨ ParticipateInExtracurricular Alex
-- Alex did not get good grades
axiom A2 : ¬ GetGoodGrades Alex

theorem alex_invited : GetInvitedToSocialEvents Alex := by
  -- Since Alex didn't get good grades, they didn't study hard (by contraposition of R1)
  have no_study : ¬ StudyHard Alex := by
    intro h
    have good_grades := R1 Alex h
    exact A2 good_grades
  -- From A1 (disjunction) and no_study, Alex must have participated in extracurriculars
  have participated : ParticipateInExtracurricular Alex := by
    cases A1 with
    | inl study => contradiction
    | inr part => exact part
  -- Apply R2 to get that Alex made new friends
  have new_friends : MakeNewFriends Alex := R2 Alex participated
  -- Apply R3 to conclude Alex gets invited
  exact R3 Alex new_friends
</lean>

ANSWER: Yes
