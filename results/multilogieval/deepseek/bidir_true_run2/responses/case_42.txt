variable Sunny : time → Prop
variable GetSick : time → Prop

axiom A1 : ∀ t : time, Sunny t → PlayOutside t
axiom A2 : ∀ t : time, FluSeason t → GetSick t
axiom A3 : ∀ t : time, Sunny t ∨ ¬ GetSick t ∨ (Sunny t ∧ ¬ GetSick t)
axiom A4 : ∀ t : time, PlayOutside t → SleepWell t
hypothesis flu : FluSeason t0
theorem sleep_well : SleepWell t0 := by
  have sick : GetSick t0 := A2 t0 flu
  have disj : Sunny t0 ∨ ¬ GetSick t0 ∨ (Sunny t0 ∧ ¬ GetSick t0) := A3 t0
  have sunny : Sunny t0 := by
    cases disj with
    | inl sun => exact sun
    | inr rest => 
        cases rest with
        | inl not_sick => exact False.elim (not_sick sick)
        | inr conj => exact False.elim (conj.2 sick)
  exact A4 t0 (A1 t0 sunny)
