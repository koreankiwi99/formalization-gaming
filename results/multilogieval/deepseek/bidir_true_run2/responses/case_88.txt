<lean>
axiom set350 : Prop
axiom bakeProperly : Prop
axiom containsEggs : Prop
axiom tasteDelicious : Prop
axiom guestsHappy : Prop

axiom A1 : set350 → bakeProperly
axiom A2 : bakeProperly → tasteDelicious
axiom A3 : tasteDelicious → guestsHappy
axiom A4 : set350 ∨ ¬containsEggs

theorem target : containsEggs → guestsHappy :=
  fun h_eggs : containsEggs =>
  have h_set : set350 := by
    cases A4 with
    | inl h => exact h
    | inr not_eggs => exact False.elim (not_eggs h_eggs)
  have h_bake : bakeProperly := A1 h_set
  have h_taste : tasteDelicious := A2 h_bake
  show guestsHappy from A3 h_taste
</lean>

ANSWER: Yes
