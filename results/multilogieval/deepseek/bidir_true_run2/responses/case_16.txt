<lean>
axiom obj : Type
axiom Sarah : obj
axiom PracticeDaily : obj → Prop
axiom TechniqueImprove : obj → Prop
axiom LearnFaster : obj → Prop
axiom ListenAttentively : obj → Prop

axiom practice_imp_tech : ∀ x : obj, PracticeDaily x → TechniqueImprove x
axiom tech_imp_learn : ∀ x : obj, TechniqueImprove x → LearnFaster x
axiom progress_method : PracticeDaily Sarah ∨ ListenAttentively Sarah

axiom not_learn_faster : ¬ LearnFaster Sarah

theorem sarah_listens : ListenAttentively Sarah := by
  have not_tech : ¬ TechniqueImprove Sarah := fun hTech => 
    not_learn_faster (tech_imp_learn Sarah hTech)
  have not_practice : ¬ PracticeDaily Sarah := fun hPrac => 
    not_tech (practice_imp_tech Sarah hPrac)
  cases progress_method with
  | inl practice => contradiction
  | inr listen => exact listen
</lean>

ANSWER: Yes
