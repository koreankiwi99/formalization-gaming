<lean>
axiom SalesIncreased : Prop
axiom InvestedInMarketing : Prop
axiom ProductionCostsIncreased : Prop
axiom GainedMoreCustomers : Prop

-- If invested in marketing, gained more customers
axiom A1 : InvestedInMarketing → GainedMoreCustomers
-- More customers lead to increased sales
axiom A2 : GainedMoreCustomers → SalesIncreased
-- Either focused on marketing or production costs increased (or both)
axiom A3 : InvestedInMarketing ∨ ProductionCostsIncreased

-- Sales did not increase last quarter
axiom Fact : ¬ SalesIncreased

theorem production_costs_increased : ProductionCostsIncreased := by
  -- Show marketing investment would have led to sales increase
  have contra : InvestedInMarketing → False := by
    intro hM
    have hC : GainedMoreCustomers := A1 hM
    have hS : SalesIncreased := A2 hC
    exact Fact hS
  -- If marketing didn't happen, production costs must have increased
  cases A3
  case inl hM => contradiction  -- Contradicts contra
  case inr hP => exact hP
</lean>

ANSWER: Yes
