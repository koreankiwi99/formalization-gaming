<lean>
axiom Person : Type
axiom Jesse : Person
axiom SprintWorkouts : Person → Prop
axiom RunningImprove : Person → Prop
axiom StretchingClasses : Person → Prop
axiom FlexibleAndInjuryFree : Person → Prop

-- If sprint workouts then running improves
axiom A1 : ∀ p : Person, SprintWorkouts p → RunningImprove p
-- Either sprinting or stretching classes (or both)
axiom A2 : ∀ p : Person, SprintWorkouts p ∨ StretchingClasses p
-- If stretching classes then flexible/injury-free
axiom A3 : ∀ p : Person, StretchingClasses p → FlexibleAndInjuryFree p

-- Given premise: Running did NOT improve for Jesse
axiom Premise : ¬ RunningImprove Jesse

theorem Jesse_flexible : FlexibleAndInjuryFree Jesse := by
  -- Show Jesse didn't do sprint workouts (from A1 and ¬RunningImprove)
  have no_sprint : ¬ SprintWorkouts Jesse := 
    fun hS => Premise (A1 Jesse hS)
  -- From A2: either sprinting or stretching classes
  cases (A2 Jesse) with
  | inl sprint => contradiction  -- contradicts no_sprint
  | inr stretch => 
      -- If stretching classes, then flexible by A3
      exact A3 Jesse stretch

</lean>

ANSWER: Yes
