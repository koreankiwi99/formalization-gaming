<lean>
axiom obj : Type
axiom Alex : obj

axiom StudiesHard : obj → Prop
axiom GetsGoodGrades : obj → Prop
axiom ParticipatesInExtra : obj → Prop
axiom MakesNewFriends : obj → Prop
axiom GetsInvited : obj → Prop

axiom R1 : ∀ x : obj, StudiesHard x → GetsGoodGrades x
axiom R2 : ∀ x : obj, ParticipatesInExtra x → MakesNewFriends x
axiom A1 : StudiesHard Alex ∨ ParticipatesInExtra Alex
axiom R3 : ∀ x : obj, MakesNewFriends x → GetsInvited x
axiom T1 : ¬ GetsGoodGrades Alex

theorem alex_invited : GetsInvited Alex := by
  have no_study : ¬ StudiesHard Alex := by
    intro h
    apply T1
    exact R1 Alex h
  cases A1 with
  | inl h => contradiction
  | inr h =>
      have friends : MakesNewFriends Alex := R2 Alex h
      exact R3 Alex friends

</lean>

ANSWER: Yes
