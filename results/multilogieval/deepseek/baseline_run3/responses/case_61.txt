Given that Priya studied diligently but did not excel, we can conclude that Priya did not attend lectures regularly. axiom Priya : obj
axiom AttendsRegularly : obj → Prop
axiom StudiesDiligently : obj → Prop
axiom ExcelsInCourse : obj → Prop
axiom context : ∀ x : obj, AttendsRegularly x → (StudiesDiligently x → ExcelsInCourse x)
axiom fact1 : StudiesDiligently Priya
axiom fact2 : ¬ ExcelsInCourse Priya

theorem not_attended : ¬ AttendsRegularly Priya :=
  fun h : AttendsRegularly Priya =>

  have H2 : ExcelsInCourse Priya := H1 fact1
  fact2 H2
