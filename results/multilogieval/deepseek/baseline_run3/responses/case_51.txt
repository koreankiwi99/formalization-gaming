axiom Person : Type
axiom mary : Person
axiom Sale : Prop
axiom PriceUp : Prop
axiom Payday : Prop

axiom A2 : ∀ p, MakeList p → Plan p
axiom A3 : Sale ∨ (∃ p, ¬ Plan p)

theorem prices_went_up (h_notStockUp_mary : ¬ StockUp mary) : PriceUp := by
  have notSale : ¬ Sale := fun hSale => h_notStockUp_mary (A1 hSale mary)
  cases A3 with
  | inl h_sale => contradiction
  | inr h_exists =>
      rcases h_exists with ⟨q, hq⟩
      have notMakeList_q : ¬ MakeList q := fun h_make_q => hq (A2 q h_make_q)
      have notPayday : ¬ Payday := fun h_payday => notMakeList_q (A4 h_payday q)
      cases A5 with
      | inl h_payday => contradiction
      | inr h_priceUp => exact h_priceUp
