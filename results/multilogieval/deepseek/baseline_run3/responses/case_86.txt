<lean>
axiom Person : Type
axiom Snowstorm : Prop
axiom SchoolCanceled : Prop
axiom StudiesHard : Person → Prop
axiom GoodGrades : Person → Prop
axiom ProudParents : Person → Prop
axiom IceCream : Person → Prop
axiom Priya : Person

axiom R1 : ∀ (p : Person), StudiesHard p → GoodGrades p
axiom R2 : Snowstorm → SchoolCanceled
axiom T1 : StudiesHard Priya ∨ ¬ SchoolCanceled
axiom R3 : ∀ (p : Person), GoodGrades p → ProudParents p
axiom R4 : ∀ (p : Person), ProudParents p → IceCream p

theorem answer : Snowstorm → IceCream Priya := by
  intro hSnow
  have hCancel : SchoolCanceled := R2 hSnow
  cases T1 with
  | inl hStudy => 
      apply R4
      apply R3
      apply R1
      exact hStudy
  | inr noCancel => 
      contradiction
</lean>

ANSWER: Yes
