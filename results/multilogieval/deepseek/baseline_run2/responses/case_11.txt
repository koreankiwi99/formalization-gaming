<lean>
axiom nice_display : Prop
axiom judges_impressed : Prop
axiom thorough_research : Prop
axiom experts_serious : Prop
axiom award : Prop

axiom A1 : nice_display → judges_impressed
axiom A2 : thorough_research → experts_serious
axiom A3 : nice_display ∨ thorough_research
axiom A4 : experts_serious → award

axiom given : ¬judges_impressed

theorem project_received_award : award :=
  have not_nice : ¬nice_display := fun h => given (A1 h)
  have has_thorough : thorough_research :=
    Or.elim A3
      (fun nice => absurd nice not_nice)
      (fun thorough => thorough)
  have serious : experts_serious := A2 has_thorough
  show award from A4 serious
</lean>

ANSWER: Yes
