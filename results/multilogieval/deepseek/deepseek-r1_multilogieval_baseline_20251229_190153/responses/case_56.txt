<lean>
axiom Person : Type
axiom Emma : Person
axiom SpeaksDaily : Person → Prop
axiom ListensPodcasts : Person → Prop
axiom ImprovesFluency : Person → Prop
axiom RealConversations : Person → Prop
axiom GainsListening : Person → Prop

axiom A1 : ∀ p : Person, SpeaksDaily p → ImprovesFluency p
axiom A2 : ∀ p : Person, ImprovesFluency p → RealConversations p
axiom A3 : ∀ p : Person, SpeaksDaily p ∨ ListensPodcasts p
axiom A4 : ∀ p : Person, ListensPodcasts p → GainsListening p

theorem answer (h : ¬ RealConversations Emma) : GainsListening Emma := by
  have not_improved : ¬ ImprovesFluency Emma := 
    fun h_imp => h (A2 Emma h_imp)
  have not_spoken : ¬ SpeaksDaily Emma := 
    fun h_speak => not_improved (A1 Emma h_speak)
  have podcast_case : ListensPodcasts Emma := by
    cases (A3 Emma) with
    | inl spoken => contradiction
    | inr podcast => exact podcast
  exact A4 Emma podcast_case
</lean>

ANSWER: Yes
