<lean>
axiom Event : Type
axiom Lost : Event → Prop
axiom Important : Event → Prop
axiom BigMatch : Event → Prop
axiom Significant : Event → Prop
axiom Disappointed : Prop
axiom Demoralized : Prop

-- Context: Losing important game → disappointed
axiom A1 : ∀ e : Event, Important e ∧ Lost e → Disappointed
-- Context: Losing big match → demoralized
axiom A2 : ∀ e : Event, BigMatch e ∧ Lost e → Demoralized

-- Given: Players are neither disappointed nor demoralized
axiom Given : ¬Disappointed ∧ ¬Demoralized

-- We can prove no important game was lost (from ¬Disappointed and A1)
theorem no_important_lost : ¬∃ e, Important e ∧ Lost e := by
  intro h
  have ⟨e, heImp, heLost⟩ := h
  have : Disappointed := A1 e ⟨heImp, heLost⟩
  exact Given.left this

-- We can prove no big match was lost (from ¬Demoralized and A2)
theorem no_bigmatch_lost : ¬∃ e, BigMatch e ∧ Lost e := by
  intro h
  have ⟨e, heBig, heLost⟩ := h
  have : Demoralized := A2 e ⟨heBig, heLost⟩
  exact Given.right this

-- But we cannot prove anything about significant games:
-- 1. Cannot prove ∃ e, Significant e ∧ Lost e (no positive evidence)
