-- Rotating crops maintained soil nutrients

-- Crops were either unhealthy or lacked nutrients (i.e., not both healthy and nutrient-maintained)
axiom A3 : ∀ f : Farmer, ¬(crops_healthy f ∧ soil_nutrients_maintained f)
-- An aware farmer knew that being one with nature led to using organic methods
axiom A4 : ∀ f : Farmer, aware f → (one_with_nature f → uses_organic f)

-- Hypothesis: Bob rotates his crops well
hypothesis H_rotate : rotates_crops Bob

-- We want to know: Is Bob one with nature?
-- We cannot prove one_with_nature Bob nor ¬one_with_nature Bob

-- Derive consequences from H_rotate and axioms

theorem Bob_not_both : ¬(crops_healthy Bob ∧ soil_nutrients_maintained Bob) := A3 Bob

-- Since soil_nutrients_maintained is true, crops_healthy must be false

  fun h_healthy : crops_healthy Bob =>
    Bob_not_both ⟨h_healthy, Bob_soil_nutrients⟩

  fun h_organic : uses_organic Bob =>
    Bob_not_healthy (A1 Bob h_organic)
