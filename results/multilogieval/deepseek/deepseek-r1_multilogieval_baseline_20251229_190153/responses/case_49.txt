<lean>
axiom W : Prop  -- Driveway is wet
axiom R : Prop  -- Rained recently
axiom CarWashed : Prop  -- Someone washed their car
axiom WaterUsed : Prop  -- Water was used on the driveway
axiom HoseUncoiled : Prop  -- Hose is uncoiled
axiom SpongeSoapy : Prop  -- Sponge is soapy

axiom A1 : W → R  -- If driveway wet, then rained recently
axiom A2 : CarWashed → WaterUsed  -- If car washed, then water used
axiom A3 : W ∨ ¬WaterUsed  -- Driveway wet or no water used (simplified disjunction)
axiom A4 : HoseUncoiled → CarWashed  -- If hose uncoiled, then car washed
axiom A5 : HoseUncoiled ∨ SpongeSoapy  -- Hose uncoiled or sponge soapy (or both)

theorem sponge_soapy (notR : ¬R) : SpongeSoapy :=
  by
    -- Prove driveway not wet (from ¬R and A1)
    have notW : ¬W := fun w => notR (A1 w)
    -- Case split on A3: W ∨ ¬WaterUsed
    cases A3 with
    | inl w => contradiction  -- Contradiction with notW
    | inr noWater =>
        -- Prove car not washed (from ¬WaterUsed and A2)
        have notCar : ¬CarWashed := fun car => noWater (A2 car)
        -- Prove hose not uncoiled (from ¬CarWashed and A4)
        have notHose : ¬HoseUncoiled := fun hose => notCar (A4 hose)
        -- Case split on A5: HoseUncoiled ∨ SpongeSoapy
        cases A5 with
        | inl hose => contradiction  -- Contradiction with notHose
        | inr soapy => exact soapy  -- SpongeSoapy holds
</lean>
