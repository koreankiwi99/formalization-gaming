axiom Sore : Prop
axiom Shower : Prop
axiom Relax : Prop
axiom Cold : Prop
axiom Rain : Prop

axiom A1 : Hard → Sore
axiom A2 : Shower → Relax
axiom A3 : Hard ∨ ¬ Relax
axiom A4 : Cold → Shower
axiom A5 : Rain → Cold

theorem no_rain_if_not_sore (h : ¬Sore) : ¬Rain := by
  intro hRain
  have hCold : Cold := A5 hRain
  have hShower : Shower := A4 hCold
  have hRelax : Relax := A2 hShower
  have hNotHard : ¬Hard := by
    intro hHard
    apply h
    exact A1 hHard
  cases A3 with
  | inl hHard => contradiction
  | inr hNotRelax => contradiction
