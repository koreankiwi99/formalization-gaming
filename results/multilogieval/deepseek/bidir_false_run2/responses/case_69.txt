<lean>
axiom obj : Type
axiom Lebron : obj
axiom TrainsHard : obj → Prop
axiom RemainsFocused : obj → Prop
axiom PerformsBest : obj → Prop
axiom R1 : ∀ x : obj, TrainsHard x → (RemainsFocused x → PerformsBest x)
axiom T1 : RemainsFocused Lebron
axiom T2 : ¬ PerformsBest Lebron

theorem not_trained_hard : ¬ TrainsHard Lebron := by
  intro h_train
  have h_imply : RemainsFocused Lebron → PerformsBest Lebron := R1 Lebron h_train
  have h_best : PerformsBest Lebron := h_imply T1
  exact T2 h_best
</lean>
