<lean>
axiom Athlete : Type
axiom simone : Athlete
axiom TrainsWithDedication : Athlete → Prop
axiom GetsAdequateRest : Athlete → Prop
axiom PerformsAtPeak : Athlete → Prop

-- Context: For athletes: if trains with dedication then (if adequate rest → peak performance)
axiom R1 : ∀ (a : Athlete), TrainsWithDedication a → (GetsAdequateRest a → PerformsAtPeak a)

-- Given: Simone got rest (T1) and didn't perform at peak (T2)
axiom T1 : GetsAdequateRest simone
axiom T2 : ¬ PerformsAtPeak simone

-- Prove: Simone did NOT train with dedication
theorem not_dedicated : ¬ TrainsWithDedication simone := 
  fun h : TrainsWithDedication simone =>
    -- R1 applied to simone (using h) gives: adequate rest → peak performance
    have rest_implies_peak : GetsAdequateRest simone → PerformsAtPeak simone := R1 simone h
    -- From T1 and rest_implies_peak, we get peak performance
    have peak : PerformsAtPeak simone := rest_implies_peak T1
    -- T2 says ¬peak, contradiction
    T2 peak
</lean>

ANSWER: No
