<lean>
axiom obj : Type
axiom Bookshelf : obj
axiom HighQuality : obj → Prop
axiom CutPrecisely : obj → Prop
axiom Sturdy : obj → Prop

-- Context: If wood is high quality, then precise joints imply sturdy product
axiom carpentry_rule : ∀ x : obj, HighQuality x → (CutPrecisely x → Sturdy x)

-- Given: Joints were cut precisely
axiom precise_joints : CutPrecisely Bookshelf

-- Given: Bookshelf was shaky (not sturdy)
axiom not_sturdy : ¬Sturdy Bookshelf

-- Prove wood was NOT high quality
theorem wood_not_high_quality : ¬HighQuality Bookshelf :=
  fun h : HighQuality Bookshelf =>
    have must_be_sturdy : Sturdy Bookshelf := carpentry_rule Bookshelf h precise_joints
    show False from not_sturdy must_be_sturdy
</lean>

ANSWER: No
