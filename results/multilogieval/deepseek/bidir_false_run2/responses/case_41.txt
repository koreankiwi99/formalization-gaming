axiom WatersDaily : Prop
axiom Drought : Prop
axiom Restrictions : Prop
axiom EnoughWater : Prop
axiom GrowsWell : Prop
axiom WiltAndDie : Prop

axiom A1 : WatersDaily → GrowsWell
axiom A2 : Drought → Restrictions
axiom A3 : WatersDaily ∨ Drought
axiom A4 : Restrictions → ¬EnoughWater
axiom A5 : ¬EnoughWater → WiltAndDie

-- Given premise: Priya's plants did not grow well
axiom not_grows_well : ¬GrowsWell

-- Attempt to prove ¬WiltAndDie fails because:
-- 1. ¬GrowsWell and A1 imply ¬WatersDaily (by contraposition)
-- 2. ¬WatersDaily and A3 imply Drought
-- 3. Drought and A2 imply Restrictions
-- 4. Restrictions and A4 imply ¬EnoughWater
-- 5. ¬EnoughWater and A5 imply WiltAndDie

-- Therefore, we derive WiltAndDie, not ¬WiltAndDie
theorem must_wilt : WiltAndDie := by
  have no_water : ¬WatersDaily := fun h => not_grows_well (A1 h)
  have drought : Drought := by
    cases A3 with
    | inl water => contradiction
    | inr d => exact d
  have restrictions : Restrictions := A2 drought
  have not_enough : ¬EnoughWater := A4 restrictions
  exact A5 not_enough

-- Since we proved WiltAndDie, we cannot prove ¬WiltAndDie
</lean>
