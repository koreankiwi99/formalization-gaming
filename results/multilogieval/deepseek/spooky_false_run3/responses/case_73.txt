<lean>
axiom athlete : Type
axiom Simone : athlete
axiom TrainsWithDedication : athlete → Prop
axiom AdequateRest : athlete → Prop
axiom PerformAtPeak : athlete → Prop

-- Context: If someone trains with dedication, then if they get adequate rest, they perform at peak
axiom coach_saying : ∀ a : athlete, TrainsWithDedication a → (AdequateRest a → PerformAtPeak a)

-- Premises: Simone got rest but didn't perform at peak
axiom simone_rested : AdequateRest Simone
axiom not_peak_performance : ¬PerformAtPeak Simone

-- Prove Simone did NOT train with dedication
theorem simone_not_dedicated : ¬TrainsWithDedication Simone := by
  intro h_dedicated  -- Assume Simone trained with dedication
  have rest_implies_peak : AdequateRest Simone → PerformAtPeak Simone := coach_saying Simone h_dedicated
  have peak_performance : PerformAtPeak Simone := rest_implies_peak simone_rested
  exact not_peak_performance peak_performance  -- Contradiction
</lean>

ANSWER: No
