<lean>
axiom obj : Type
axiom Priya : obj
axiom AttendsRegularly : obj → Prop
axiom StudiesDiligently : obj → Prop
axiom ExcelsInCourse : obj → Prop

-- Context: If a student attends lectures regularly, then if they study diligently, they will excel
axiom Context : ∀ (s : obj), AttendsRegularly s → (StudiesDiligently s → ExcelsInCourse s)

-- Given: Priya studied diligently and did not excel
axiom Given1 : StudiesDiligently Priya
axiom Given2 : ¬ ExcelsInCourse Priya

-- Prove Priya did NOT attend lectures regularly
theorem result : ¬ AttendsRegularly Priya :=
  fun (hAttend : AttendsRegularly Priya) =>
    have hImplication : StudiesDiligently Priya → ExcelsInCourse Priya := Context Priya hAttend
    have hExcel : ExcelsInCourse Priya := hImplication Given1
    Given2 hExcel
</lean>

ANSWER: No
