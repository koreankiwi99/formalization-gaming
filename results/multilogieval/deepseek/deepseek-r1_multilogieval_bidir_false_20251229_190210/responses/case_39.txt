axiom Well : Prop
axiom Hard : Prop
axiom Good : Prop

axiom A1 : Monday → Quiz

axiom A3 : Monday ∨ ¬Hard
axiom A4 : Good → Well

theorem no_quiz_implies_not_good : ¬Quiz → ¬Good :=
  fun hNoQuiz : ¬Quiz =>
  have hNotMonday : ¬Monday := fun hMon => hNoQuiz (A1 hMon)
  have hNotHard : ¬Hard := by
    cases A3 with
    | inl hMon => exact absurd hMon hNotMonday
    | inr hNotHard => exact hNotHard
  have hNotWell : ¬Well := fun hWell => hNotHard (A2 hWell)
  fun hGood : Good => hNotWell (A4 hGood)
