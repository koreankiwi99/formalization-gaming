<lean>
axiom obj : Type
axiom LizCar : obj
axiom PM : obj → Prop
axiom QP : obj → Prop
axiom LS : obj → Prop
axiom context : ∀ x : obj, PM x → (QP x → LS x)
axiom fact1 : QP LizCar
axiom fact2 : ¬ LS LizCar

theorem no_preventive_maintenance : ¬ PM LizCar := 
  fun h : PM LizCar ↦ 
    have qp_imp_ls : QP LizCar → LS LizCar := context LizCar h
    have ls : LS LizCar := qp_imp_ls fact1
    fact2 ls
</lean>

ANSWER: No
