4. Either the alarm went off, or John is not late for school.
5. If John gets up late, then he misses the bus.

axiom TimeToGetReady : Prop
axiom MissBus : Prop

axiom P1 : Alarm → TimeToGetReady
axiom P2 : ¬TimeToGetReady
axiom P3 : MissBus → Late

axiom P5 : GotUpLate → MissBus
axiom P6 : WentToBedLate → GotUpLate

-- Prove ¬Late using P4 and not_alarm
theorem not_late : ¬Late := by
  cases P4 with
  | inl a => exact (not_alarm a).elim

-- Prove ¬MissBus from P3 and not_late

-- Prove ¬GotUpLate from P5 and not_miss_bus
