<lean>
axiom obj : Type
axiom Priya : obj
axiom AttendsLectures : obj → Prop
axiom SpendsTimeRevising : obj → Prop
axiom PerformsWell : obj → Prop

-- Context: If attends lectures, then (if revises sufficiently then performs well)
axiom R1 : ∀ x : obj, AttendsLectures x → (SpendsTimeRevising x → PerformsWell x)

-- Given: Priya spent sufficient time revising and did not perform well
axiom T1 : SpendsTimeRevising Priya
axiom T2 : ¬ PerformsWell Priya

-- Prove Priya did NOT attend lectures regularly
theorem Priya_not_attends : ¬ AttendsLectures Priya :=
  fun hAttend : AttendsLectures Priya =>
    have h1 : SpendsTimeRevising Priya → PerformsWell Priya := R1 Priya hAttend
    have h2 : PerformsWell Priya := h1 T1
    show False from T2 h2
</lean>

ANSWER: No
