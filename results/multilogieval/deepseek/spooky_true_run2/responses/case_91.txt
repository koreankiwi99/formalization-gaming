The premises lead to a contradiction when assuming no noise, allowing any conclusion, including that the police were called.

<lean>
axiom P : Prop  -- There is a party
axiom D : Prop  -- Someone is practicing drums
axiom N : Prop  -- Noise levels go up
axiom C : Prop  -- Neighbors complain
axiom Q : Prop  -- Police are called

axiom host_party_noisy : P → N
axiom drums_noisy : D → N
axiom party_or_drums : P ∨ D
axiom noisy_party_complaint : P → C
axiom complaint_police : C → Q

theorem police_called_if_no_noise : ¬N → Q :=
fun h : ¬N =>
  by
    cases party_or_drums
    case inl p =>
      apply False.elim
      exact h (host_party_noisy p)
    case inr d =>
      apply False.elim
      exact h (drums_noisy d)
</lean>

ANSWER: Yes
