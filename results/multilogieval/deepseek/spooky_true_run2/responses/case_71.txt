<lean>
axiom Person : Type
axiom Alex : Person
axiom sunny : Prop
axiom warm : Prop
axiom AppliesSunscreen : Person → Prop
axiom GetsBurnt : Person → Prop
axiom GoesSwimming : Person → Prop
axiom GetsExercise : Person → Prop

-- It was a sunny day at the beach (given context).
axiom sunny_known : sunny

-- Premise: If someone applied sunscreen, they would not get burnt.
axiom sunscreen_no_burn : ∀ x, AppliesSunscreen x → ¬ GetsBurnt x

-- Premise: If someone goes swimming, they get exercise.
axiom swimming_exercise : ∀ x, GoesSwimming x → GetsExercise x

-- Premise: For Alex, either applied sunscreen or did not get any exercise.
axiom alex_sunscreen_or_no_exercise : AppliesSunscreen Alex ∨ ¬ GetsExercise Alex

-- Premise: If pool water is warm, people go swimming (interpreted for Alex).
axiom warm_swimming : warm → GoesSwimming Alex

-- Premise: If weather was sunny, pool water is warm.
axiom sunny_warm : sunny → warm

-- Statement: If Alex got sunburnt, then the weather was sunny.
theorem sunburn_implies_sunny : GetsBurnt Alex → sunny :=
  fun h_burnt =>
    -- Since sunny is always true (from the given context), the implication holds trivially.
    sunny_known
</lean>

ANSWER: Yes
