axiom TrainsHard : Prop
axiom Stretches : Prop
axiom RunRacesFaster : Prop
axiom TakesRecoverySeriously : Prop
axiom FocusesOnFlexibility : Prop

-- Premise 3: Trained hard weekly → run races faster
axiom trained_imp_race : TrainsHard → RunRacesFaster

-- Premise 4: Trained hard ∨ stretched (or both)
axiom train_or_stretch : TrainsHard ∨ Stretches

-- Premise 5: Stretching → takes recovery seriously
axiom stretch_imp_recovery : Stretches → TakesRecoverySeriously

-- Premise 6: Takes recovery seriously → focuses on flexibility
axiom recovery_imp_flex : TakesRecoverySeriously → FocusesOnFlexibility

-- Theorem: ¬RunRacesFaster → FocusesOnFlexibility
theorem no_faster_imp_flex : ¬RunRacesFaster → FocusesOnFlexibility := by
  intro no_faster
  have no_train : ¬TrainsHard := fun train => no_faster (trained_imp_race train)
  cases train_or_stretch with
  | inl train => contradiction
  | inr stretch =>
      have recovery : TakesRecoverySeriously := stretch_imp_recovery stretch
      exact recovery_imp_flex recovery
</lean>
