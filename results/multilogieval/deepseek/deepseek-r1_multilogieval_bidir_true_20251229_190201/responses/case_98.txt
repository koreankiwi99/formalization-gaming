axiom T1 : PracticesHard Team → Prepared Team

axiom T4 : Prepared Team → PlaysWell Team
axiom T5 : PlaysWell Team → Win Team

-- Proof that VeryHot implies Win Team
theorem team_wins_if_very_hot : VeryHot → Win Team := by
  intro hHot
  -- Show game is postponed due to heat
  have postponed : Postponed := T2 hHot
  -- Case analysis on the disjunction T3
  cases T3 with
  | inl practiced => 
    -- Team practiced hard
    have prepared : Prepared Team := T1 practiced
    have plays_well : PlaysWell Team := T4 prepared
    exact T5 plays_well
  | inr not_postponed => 
    -- Contradiction: we have Postponed and ¬Postponed
    contradiction

-- Since we constructed a proof, the statement is verified
