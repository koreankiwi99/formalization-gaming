<lean>
axiom Person : Type
axiom Alex : Person

axiom AdvertisesCleverly : Person → Prop
axiom OffersDiscounts : Person → Prop
axiom AttractMoreCustomers : Person → Prop
axiom IncreaseSales : Person → Prop
axiom ExpandBusiness : Person → Prop

axiom clever_ad_attracts : ∀ p, AdvertisesCleverly p → AttractMoreCustomers p
axiom discount_increases_sales : ∀ p, OffersDiscounts p → IncreaseSales p
axiom ad_or_discount : ∀ p, AdvertisesCleverly p ∨ OffersDiscounts p
axiom sales_expand : ∀ p, IncreaseSales p → ExpandBusiness p

-- Given premise: Alex did not attract more customers
axiom no_customers : ¬ AttractMoreCustomers Alex

theorem alex_expands : ExpandBusiness Alex := by
  have no_clever_ad : ¬ AdvertisesCleverly Alex := fun h => no_customers (clever_ad_attracts Alex h)
  have must_discount : OffersDiscounts Alex := by
    cases (ad_or_discount Alex) with
    | inl clever => contradiction
    | inr discount => exact discount
  have sales_increase : IncreaseSales Alex := discount_increases_sales Alex must_discount
  exact sales_expand Alex sales_increase
</lean>

ANSWER: Yes
