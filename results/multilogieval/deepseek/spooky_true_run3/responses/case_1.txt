The reasoning involves proving that if the Lakers did not limit inside scoring, then they forced many turnovers. The premises provide general rules for any team, and we instantiate them for the Lakers.

<lean>
axiom obj : Type
axiom Lakers : obj
axiom PerimeterDefenseStrong : obj → Prop
axiom ForceManyTurnovers : obj → Prop
axiom BigMenProtectPaint : obj → Prop
axiom LimitInsideScoring : obj → Prop

-- If strong perimeter defense, then force many turnovers
axiom R1 : ∀ t : obj, PerimeterDefenseStrong t → ForceManyTurnovers t
-- If big men protect the paint, then limit inside scoring
axiom R2 : ∀ t : obj, BigMenProtectPaint t → LimitInsideScoring t
-- Either perimeter defense is strong or paint is protected (or both)
axiom R3 : ∀ t : obj, PerimeterDefenseStrong t ∨ BigMenProtectPaint t

theorem lakers_turnovers : ¬ LimitInsideScoring Lakers → ForceManyTurnovers Lakers :=
  fun h : ¬ LimitInsideScoring Lakers =>
  have h_case : PerimeterDefenseStrong Lakers ∨ BigMenProtectPaint Lakers := R3 Lakers
  Or.elim h_case
    (fun p : PerimeterDefenseStrong Lakers => R1 Lakers p)
    (fun b : BigMenProtectPaint Lakers => 
      have L : LimitInsideScoring Lakers := R2 Lakers b
      absurd L h)
</lean>

ANSWER: Yes
