<lean>
axiom obj : Type
axiom Jesse : obj

-- Predicates:
axiom Sprint : obj → Prop      -- did sprint workouts several times a week
axiom Running : obj → Prop      -- running form and stamina improved
axiom Stretching : obj → Prop   -- taking stretching classes
axiom Flexible : obj → Prop     -- become more flexible and less prone to injury

-- Axioms from context:
axiom A1 : ∀ a : obj, Sprint a → Running a
axiom A2 : ∀ a : obj, Stretching a → Flexible a
axiom A3 : ∀ a : obj, Sprint a ∨ Stretching a

-- Given premise for Jesse:
axiom not_Running_Jesse : ¬ Running Jesse

-- We want to prove: Flexible Jesse
theorem Jesse_flexible : Flexible Jesse := by
  -- Show Jesse is not sprinting (contrapositive of A1)
  have not_Sprint_Jesse : ¬ Sprint Jesse := 
    fun hSprint : Sprint Jesse => not_Running_Jesse (A1 Jesse hSprint)
  -- From disjunction (A3): Sprint ∨ Stretching must hold for Jesse
  cases A3 Jesse with
  | inl hSprint => 
    -- If Sprinting, contradiction with not_Sprint_Jesse
    contradiction
  | inr hStretch =>
    -- If Stretching, then Flexible by A2
    exact A2 Jesse hStretch
</lean>

ANSWER: Yes
