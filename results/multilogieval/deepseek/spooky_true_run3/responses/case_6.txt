<lean>
axiom City : Type
axiom Dallas : City
axiom HeavyRain : City → Prop
axiom Overflow : City → Prop
axiom Flooding : City → Prop

-- Premise: If there was excessive rain (heavy rainfall), reservoirs and dams would overflow
axiom rain_overflow : ∀ c : City, HeavyRain c → Overflow c

-- Premise: If reservoirs overflow, flooding occurs
axiom overflow_flood : ∀ c : City, Overflow c → Flooding c

-- Given fact: There is heavy rainfall in Dallas
axiom dallas_rain : HeavyRain Dallas

-- Prove flooding occurs in Dallas
theorem dallas_flooding : Flooding Dallas := by
  apply overflow_flood Dallas
  apply rain_overflow Dallas
  exact dallas_rain
</lean>

ANSWER: Yes
