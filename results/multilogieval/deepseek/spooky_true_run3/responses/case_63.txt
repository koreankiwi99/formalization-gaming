<lean>
axiom obj : Type
axiom Linda : obj
axiom PlantsDiseaseResistant : obj → Prop
axiom WaterAndFeedProperly : obj → Prop
axiom BountifulHarvest : obj → Prop

-- Context axiom: for anyone, if they plant disease-resistant varieties and they water and feed properly, they will have a bountiful harvest
axiom common_saying : ∀ (x : obj), PlantsDiseaseResistant x → WaterAndFeedProperly x → BountifulHarvest x

-- Given facts about Linda
axiom linda_wf : WaterAndFeedProperly Linda
axiom linda_no_bh : ¬ BountifulHarvest Linda

-- Prove Linda did NOT plant disease-resistant varieties
theorem linda_no_dr : ¬ PlantsDiseaseResistant Linda := 
  fun (h_dr : PlantsDiseaseResistant Linda) => 
    have bh_linda : BountifulHarvest Linda := common_saying Linda h_dr linda_wf
    linda_no_bh bh_linda

</lean>

ANSWER: Yes
