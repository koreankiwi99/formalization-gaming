4. Premise 1 states practice implies preparedness, so the team is prepared.

<lean>
axiom sunny : Prop
axiom practices : Prop
axiom play : Prop
axiom prepared : Prop
axiom confident : Prop
axiom hold_high : Prop

axiom P1 : practices → prepared
axiom P2 : sunny → play
axiom P3 : practices ∨ ¬play
axiom P4 : prepared → confident
axiom P5 : confident → hold_high

theorem sunny_implies_hold_high : sunny → hold_high :=
  fun h_sunny : sunny =>
    have h_play : play := P2 h_sunny
    have h_practices : practices :=
      match P3 with
      | Or.inl p => p
      | Or.inr np => False.elim (np h_play)
    have h_prepared : prepared := P1 h_practices
    have h_confident : confident := P4 h_prepared
    P5 h_confident
