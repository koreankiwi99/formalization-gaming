<lean>
axiom waters_daily : Prop
axiom drought : Prop
axiom restrictions : Prop
axiom enough_water : Prop
axiom grow_well : Prop
axiom wilt_and_die : Prop

-- Context axioms
axiom A1 : waters_daily → grow_well
axiom A2 : drought → restrictions
axiom A3 : waters_daily ∨ drought
axiom A4 : restrictions → ¬ enough_water
axiom A5 : ¬ enough_water → wilt_and_die

-- Given premise
axiom not_grow_well : ¬ grow_well

-- We need to prove wilt_and_die
theorem plants_died : wilt_and_die := by
  have not_waters : ¬ waters_daily := fun h_waters => not_grow_well (A1 h_waters)
  have must_drought : drought := by
    cases A3 with
    | inl h_waters => contradiction
    | inr h_drought => exact h_drought
  have restr : restrictions := A2 must_drought
  have not_enough : ¬ enough_water := A4 restr
  exact A5 not_enough
</lean>

ANSWER: Yes
