axiom L : obj → Prop  -- gains valuable listening comprehension

axiom A2 : ∀ x, F x → C x

axiom A4 : ∀ x, P x → L x

theorem emma_conclusion : ¬ C Emma → L Emma := by
  intro h_notC      -- Assume Emma is unable to have real conversations
  have h_notF : ¬ F Emma := by 
    intro hF 
    exact h_notC (A2 Emma hF)  -- Contrapositive of A2: ¬C → ¬F
  have h_notS : ¬ S Emma := by 
    intro hS

  have h_P : P Emma := by 
    cases (A3 Emma) with

  exact A4 Emma h_P  -- P implies L
