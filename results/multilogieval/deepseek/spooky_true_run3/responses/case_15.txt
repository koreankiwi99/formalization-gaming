<lean>
axiom obj : Type
axiom S : obj  -- the student

-- Predicates
axiom StudyDiligently : obj → Prop
axiom ExcelAcademically : obj → Prop
axiom ParticipateActively : obj → Prop
axiom GainLeadershipSkills : obj → Prop
axiom SelectedAsPrefect : obj → Prop

-- Premises
axiom A1 : StudyDiligently S → ExcelAcademically S
axiom A2 : ParticipateActively S → GainLeadershipSkills S
axiom A3 : StudyDiligently S ∨ ParticipateActively S
axiom A4 : ∀ x, GainLeadershipSkills x → SelectedAsPrefect x

-- Given fact
axiom notExcel : ¬ ExcelAcademically S

-- We want to prove SelectedAsPrefect S
theorem chosen_as_prefect : SelectedAsPrefect S := by
  -- Since the student didn't excel, they didn't study diligently (contrapositive of A1)
  have notStudy : ¬ StudyDiligently S := by
    intro hStudy
    apply notExcel (A1 hStudy)
  -- From A3 (they either study or participate) and not studying, they must participate
  have participates : ParticipateActively S := by
    cases A3 with
    | inl h => contradiction
    | inr h => exact h
  -- From participation (A2) they gain leadership skills
  have leader : GainLeadershipSkills S := A2 participates
  -- From leadership skills (A4) they are selected as prefect
  exact A4 S leader

</lean>

ANSWER: Yes
