<lean>
axiom obj : Type
axiom Today : obj
axiom Rains : obj → Prop
axiom Watered : obj → Prop
axiom Hot : obj → Prop
axiom Sprinklers : obj → Prop

axiom rule1 : ∀ x : obj, Rains x → Watered x
axiom rule2 : ∀ x : obj, Hot x → Sprinklers x
axiom fact : Rains Today ∨ ¬ Sprinklers Today
axiom hot_today : Hot Today

theorem grass_watered : Watered Today :=
  have sprinklers_today : Sprinklers Today := rule2 Today hot_today
  have rains_today : Rains Today := by
    cases fact
    case inl r => exact r
    case inr ns => exact absurd sprinklers_today ns
  rule1 Today rains_today
</lean>

ANSWER: Yes
