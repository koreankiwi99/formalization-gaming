<lean>
axiom Person : Type
axiom Lucy : Person
axiom Hacked : Person → Prop
axiom ResetPwd : Person → Prop
axiom NotifyBank : Person → Prop

-- From the context: if hacked, then must reset password and notify bank
axiom reset_if_hacked : ∀ x : Person, Hacked x → ResetPwd x
axiom notify_if_hacked : ∀ x : Person, Hacked x → NotifyBank x

-- Statement to evaluate (interpreted as: at least one of the required actions was not done)
axiom lucy_either_failed_reset_or_notify : ¬ ResetPwd Lucy ∨ ¬ NotifyBank Lucy

-- From the context and the statement, we can derive that Lucy was not hacked
theorem lucy_not_hacked : ¬ Hacked Lucy :=
by
  intro h
  cases lucy_either_failed_reset_or_notify with
  | inl nReset =>
      have r : ResetPwd Lucy := reset_if_hacked Lucy h
      exact nReset r
  | inr nNotify =>
      have n : NotifyBank Lucy := notify_if_hacked Lucy h
      exact nNotify n
</lean>
