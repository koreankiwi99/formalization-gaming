<lean>
axiom GreatService : Prop
axiom HostEvents : Prop
axiom ReputationGrows : Prop
axiom TrafficIncreases : Prop
axiom RevenueImproves : Prop

-- If they offer great customer service, their reputation will grow.
axiom H1 : GreatService → ReputationGrows
-- If they host author events, more customers (traffic) will increase.
axiom H2 : HostEvents → TrafficIncreases
-- The owner either provides excellent service or hosts events.
axiom H3 : GreatService ∨ HostEvents
-- If customer traffic increases, revenue will improve.
axiom H4 : TrafficIncreases → RevenueImproves
-- Given: service quality decreased (not great service).
axiom NotGreat : ¬ GreatService

theorem revenue_improved : RevenueImproves :=
by
  have hevents : HostEvents :=
    Or.elim H3
      (fun gs => False.elim (NotGreat gs))
      (fun he => he)
  exact H4 (H2 hevents)
</lean>

ANSWER: Yes
