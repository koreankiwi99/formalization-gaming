<lean>
axiom obj : Type
axiom Sarah : obj

axiom PracticeDaily : obj → Prop
axiom TechniqueImproved : obj → Prop
axiom LearnFaster : obj → Prop
axiom ListenAttentively : obj → Prop

axiom R1 : ∀ x : obj, PracticeDaily x → TechniqueImproved x
axiom R2 : ∀ x : obj, TechniqueImproved x → LearnFaster x
axiom Disj : ∀ x : obj, PracticeDaily x ∨ ListenAttentively x

theorem sarah_listens : (¬ LearnFaster Sarah) → ListenAttentively Sarah := by
  intro hnot
  have hdisj := Disj Sarah
  cases hdisj with
  | inl hp =>
      have hi : TechniqueImproved Sarah := R1 Sarah hp
      have hl : LearnFaster Sarah := R2 Sarah hi
      exact False.elim (hnot hl)
  | inr hlis => exact hlis
</lean>

ANSWER: Yes
