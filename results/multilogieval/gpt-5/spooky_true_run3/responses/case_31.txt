-- Infidelity concerns warrant re-evaluating the relationship.
axiom R2 : ∀ x : Person, SuspectsUnfaithful x → ReassessedRelationship x

-- Either Priya did not talk to her partner, or she did not reassess the relationship.
axiom H : ¬ TalkedOpenly Priya ∨ ¬ ReassessedRelationship Priya

-- From the premises we can derive that Priya did not suspect infidelity.
theorem not_suspected_priya : ¬ SuspectsUnfaithful Priya :=
by
  intro hS
  cases H with
  | inl hNoTalk =>
      exact hNoTalk (R1 Priya hS)
  | inr hNoReassess =>
      exact hNoReassess (R2 Priya hS)
