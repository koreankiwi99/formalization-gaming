<lean>
axiom Person : Type
axiom Mary : Person

axiom LearnedBotany : Person → Prop
axiom UnderstoodNeeds : Person → Prop
axiom Thrived : Person → Prop
axiom SoughtAdvice : Person → Prop

axiom R1 : ∀ x : Person, LearnedBotany x → UnderstoodNeeds x
axiom R2 : ∀ x : Person, UnderstoodNeeds x → Thrived x
axiom R3 : ∀ x : Person, LearnedBotany x ∨ SoughtAdvice x

axiom not_thrived_Mary : ¬ Thrived Mary

theorem mary_sought_advice : SoughtAdvice Mary :=
by
  have hImp : LearnedBotany Mary → Thrived Mary := fun h => R2 Mary (R1 Mary h)
  have hOr : LearnedBotany Mary ∨ SoughtAdvice Mary := R3 Mary
  apply Or.elim hOr
  · intro hL
    have : Thrived Mary := hImp hL
    exact False.elim (not_thrived_Mary this)
  · intro hS
    exact hS
</lean>

ANSWER: Yes
