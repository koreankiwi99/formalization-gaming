<lean>
axiom Rain : Prop
axiom Watered : Prop
axiom Hot : Prop
axiom PlaySprinklers : Prop

axiom R1 : Rain → Watered
axiom R2 : Hot → PlaySprinklers
axiom R3 : Rain ∨ ¬ PlaySprinklers
axiom GivenHot : Hot

theorem watered_today : Watered :=
by
  have hplay : PlaySprinklers := R2 GivenHot
  have hrain : Rain :=
    Or.elim R3
      (fun r => r)
      (fun nplay => False.elim (nplay hplay))
  exact R1 hrain
</lean>

ANSWER: Yes
