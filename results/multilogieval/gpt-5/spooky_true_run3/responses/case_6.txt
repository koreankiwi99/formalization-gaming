<lean>
axiom Place : Type
axiom Dallas : Place

axiom HeavyRain : Place → Prop
axiom ExcessiveRain : Place → Prop
axiom ReservoirsOverflow : Place → Prop
axiom Flooding : Place → Prop

-- Heavy rainfall over several days counts as excessive rain
axiom heavy_to_excessive : ∀ x : Place, HeavyRain x → ExcessiveRain x

-- If there is excessive rain, reservoirs overflow
axiom excessive_to_overflow : ∀ x : Place, ExcessiveRain x → ReservoirsOverflow x

-- If reservoirs overflow, flooding occurs
axiom overflow_to_flooding : ∀ x : Place, ReservoirsOverflow x → Flooding x

-- Given: There is heavy rainfall in Dallas
axiom HR_Dallas : HeavyRain Dallas

theorem flooding_in_Dallas : Flooding Dallas :=
by
  have h_ex : ExcessiveRain Dallas := heavy_to_excessive Dallas HR_Dallas
  have h_ov : ReservoirsOverflow Dallas := excessive_to_overflow Dallas h_ex
  exact overflow_to_flooding Dallas h_ov
</lean>

ANSWER: Yes
