axiom Enjoy : Person → Prop
axiom JoinClub : Person → Prop
axiom Busy : Person → Prop
axiom Outgoing : Person → Prop
axiom Introduced : Person → Prop

-- Premises
axiom MF_implies_Enjoy : ∀ x : Person, MadeFriends x → Enjoy x
axiom JC_implies_Busy : ∀ x : Person, JoinClub x → Busy x
axiom MF_or_notBusy : ∀ x : Person, MadeFriends x ∨ ¬ Busy x
axiom Out_implies_JC : ∀ x : Person, Outgoing x → JoinClub x
axiom Intro_implies_Out : ∀ x : Person, Introduced x → Outgoing x

-- From the premises we can derive that if Mark did not enjoy, then he did not introduce himself.
theorem not_enjoy_implies_not_intro : ¬ Enjoy Mark → ¬ Introduced Mark := by
  intro hNotEnjoy
  have hNotMF : ¬ MadeFriends Mark := fun hMF => hNotEnjoy (MF_implies_Enjoy Mark hMF)
  have hNotBusy : ¬ Busy Mark := by
    cases MF_or_notBusy Mark with
    | inl hMF => exact fun _ => False.elim (hNotMF hMF)
    | inr hnb => exact hnb
  have hNotJC : ¬ JoinClub Mark := fun hJC => hNotBusy (JC_implies_Busy Mark hJC)
  have hNotOut : ¬ Outgoing Mark := fun hOut => hNotJC (Out_implies_JC Mark hOut)
  exact fun hIntro => hNotOut (Intro_implies_Out Mark hIntro)
