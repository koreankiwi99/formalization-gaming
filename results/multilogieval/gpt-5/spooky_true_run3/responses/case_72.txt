open Classical

axiom obj : Type
axiom Sam : obj
axiom Close : obj → Prop
axiom StartCooking : obj → Prop
axiom GoShopping : obj → Prop
axiom MakeDinner : obj → Prop
axiom FridgeEmpty : obj → Prop
axiom PantryEmpty : obj → Prop

axiom R1 : ∀ x : obj, Close x → StartCooking x
axiom R2 : ∀ x : obj, GoShopping x → MakeDinner x
axiom R3 : ∀ x : obj, Close x ∨ ¬ MakeDinner x
axiom R4 : ∀ x : obj, FridgeEmpty x → GoShopping x
axiom R5 : ∀ x : obj, PantryEmpty x → FridgeEmpty x

-- From the premises we can derive that an empty pantry suffices to start cooking.
theorem pantry_empty_implies_start : PantryEmpty Sam → StartCooking Sam := by
  intro hp
  have hfe : FridgeEmpty Sam := R5 Sam hp
  have hgs : GoShopping Sam := R4 Sam hfe
  have hmd : MakeDinner Sam := R2 Sam hgs
  cases R3 Sam with
  | inl hc =>
    exact R1 Sam hc
  | inr hnot =>
    exact False.elim (hnot hmd)
