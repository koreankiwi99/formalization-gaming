axiom R3 : (∃ x : obj, ¬ Promotion x) ∨ (∃ x : obj, ¬ Influence x)
-- If someone is motivated, they work hard.
axiom R4 : ∀ x : obj, Motivated x → WorkHard x

-- Fact: Mark impresses his bosses.

-- From impressing, Mark gains influence.

-- From the global constraints, we can derive that either someone is not motivated
-- or someone does not impress their bosses, but this does not force Mark to be motivated.
theorem someone_not_motivated_or_not_impress :

  cases R3 with
  | inl h =>

      have h_not_mot : ¬ Motivated x := fun hmot => h_not_wh (R4 x hmot)
      exact Or.inl ⟨x, h_not_mot⟩
  | inr h =>

    | intro x hx =>
