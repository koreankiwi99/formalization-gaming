<lean>
axiom person : Type
axiom exam : Type

axiom Emma : person
axiom Biology : exam

axiom StudyHard : person → Prop
axiom EnoughSleepBefore : person → exam → Prop
axiom PerformWellOn : person → exam → Prop
axiom AceOn : person → exam → Prop

-- Belief: if someone studies hard, then if they get enough sleep before an exam,
-- they will perform well and ace the test.
axiom belief :
  ∀ x : person, ∀ e : exam,
    StudyHard x → (EnoughSleepBefore x e → (PerformWellOn x e ∧ AceOn x e))

-- Facts about Emma for her biology exam
axiom sleep_Emma : EnoughSleepBefore Emma Biology
axiom not_ace_Emma : ¬ AceOn Emma Biology

-- From the belief and the facts, Emma did not study hard.
theorem emma_not_study_hard : ¬ StudyHard Emma :=
by
  intro hStudy
  have h := belief Emma Biology hStudy sleep_Emma
  -- h : PerformWellOn Emma Biology ∧ AceOn Emma Biology
  exact not_ace_Emma h.right
</lean>
