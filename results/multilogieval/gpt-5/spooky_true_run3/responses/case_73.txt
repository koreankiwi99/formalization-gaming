<lean>
axiom obj : Type
axiom Simone : obj

axiom TrainsWithDedication : obj → Prop
axiom AdequateRest : obj → Prop
axiom PerformsAtPeak : obj → Prop

-- Coaches' rule: if someone trains with dedication, then if they get adequate rest, they will perform at their peak.
axiom coaches_rule : ∀ x : obj, TrainsWithDedication x → AdequateRest x → PerformsAtPeak x

-- Facts about Simone
axiom rest_Simone : AdequateRest Simone
axiom not_peak_Simone : ¬ PerformsAtPeak Simone

-- From the rule and Simone's rest and not performing at peak, we can derive that Simone did not train with dedication.
theorem not_trained_Simone : ¬ TrainsWithDedication Simone := by
  intro hTrain
  have hImp : AdequateRest Simone → PerformsAtPeak Simone := coaches_rule Simone hTrain
  have hPeak : PerformsAtPeak Simone := hImp rest_Simone
  exact not_peak_Simone hPeak
