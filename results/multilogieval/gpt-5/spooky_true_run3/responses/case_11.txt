<lean>
axiom NiceDisplay : Prop
axiom JudgesImpressed : Prop
axiom ThoroughResearch : Prop
axiom ExpertsTakeSeriously : Prop
axiom Award : Prop

-- Premises
axiom P1 : NiceDisplay → JudgesImpressed
axiom P2 : ThoroughResearch → ExpertsTakeSeriously
axiom P3 : NiceDisplay ∨ ThoroughResearch
axiom P4 : ExpertsTakeSeriously → Award

-- Given: judges were not impressed
axiom NotImpressed : ¬ JudgesImpressed

theorem project_awarded : Award :=
by
  have notNice : ¬ NiceDisplay := fun h => NotImpressed (P1 h)
  have thorough : ThoroughResearch :=
    Or.elim P3
      (fun hNice => False.elim (notNice hNice))
      (fun hTh => hTh)
  exact P4 (P2 thorough)
</lean>

ANSWER: Yes
