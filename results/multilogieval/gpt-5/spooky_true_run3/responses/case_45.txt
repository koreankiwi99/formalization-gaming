<lean>
axiom obj : Type
axiom Matt : obj

axiom WatchDIY : obj → Prop
axiom GainSkills : obj → Prop
axiom RedoLR : obj → Prop
axiom ReadBlogs : obj → Prop
axiom Ideas : obj → Prop

axiom A1 : ∀ x : obj, WatchDIY x → GainSkills x
axiom A2 : ∀ x : obj, GainSkills x → RedoLR x
axiom A3 : ¬ RedoLR Matt
axiom A4 : WatchDIY Matt ∨ ReadBlogs Matt
axiom A5 : ∀ x : obj, ReadBlogs x → Ideas x

theorem matt_gets_ideas : Ideas Matt := by
  have h_not_skills : ¬ GainSkills Matt := by
    intro gs
    have : RedoLR Matt := A2 Matt gs
    exact A3 this
  have h_not_watch : ¬ WatchDIY Matt := by
    intro w
    exact h_not_skills (A1 Matt w)
  have h_rb : ReadBlogs Matt := by
    apply Or.elim A4
    · intro w
      exact False.elim (h_not_watch w)
    · intro rb
      exact rb
  exact A5 Matt h_rb
</lean>

ANSWER: Yes
