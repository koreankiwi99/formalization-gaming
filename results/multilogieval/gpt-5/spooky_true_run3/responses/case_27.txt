<lean>
axiom Person : Type
axiom John : Person

axiom CloseToLunch : Prop
axiom CoworkersHungry : Person → Prop
axiom PrepareWell : Person → Prop
axiom ReadyForMeeting : Person → Prop
axiom Diligent : Person → Prop

-- If it is close to lunchtime, then their coworkers are hungry.
axiom H1 : ∀ p : Person, CloseToLunch → CoworkersHungry p
-- If they prepared well for the presentation, then they are ready for the meeting.
axiom H2 : ∀ p : Person, PrepareWell p → ReadyForMeeting p
-- Either it is close to lunchtime, or they are not ready for the meeting,
-- or it is close to lunchtime and they are not ready for it (for John).
axiom H3 : CloseToLunch ∨ ¬ ReadyForMeeting John ∨ (CloseToLunch ∧ ¬ ReadyForMeeting John)
-- Everyone: if someone is diligent, they prepare well for presentations.
axiom H4 : ∀ p : Person, Diligent p → PrepareWell p

-- From the premises, we can derive that if John's coworkers are not hungry, then John is not diligent.
theorem notHungry_implies_notDiligent : ¬ CoworkersHungry John → ¬ Diligent John := by
  intro nh
  intro hd
  -- From nh and H1 we get ¬ CloseToLunch
  have nclose : ¬ CloseToLunch := by
    intro cl
    exact nh (H1 John cl)
  -- From H3 and ¬ CloseToLunch, deduce ¬ ReadyForMeeting John
  have nready : ¬ ReadyForMeeting John := by
    cases H3 with
    | inl cl =>
      exact False.elim (nclose cl)
    | inr h =>
      cases h with
      | inl nr => exact nr
      | inr cl_nr => exact cl_nr.right
  -- From H2 (contrapositive), deduce ¬ PrepareWell John
  have npw : ¬ PrepareWell John := by
    intro pw
    exact nready (H2 John pw)
  -- From H4 and ¬ PrepareWell John, conclude ¬ Diligent John
  exact npw (H4 John hd)
</lean>
