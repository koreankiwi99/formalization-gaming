<lean>
axiom obj : Type
axiom Priya : obj

axiom StudiedHard : obj → Prop
axiom GoodGrades : obj → Prop
axiom ParentsProud : obj → Prop
axiom IceCreamOuting : obj → Prop

axiom Snowstorm : Prop
axiom BigSnowstorm : Prop
axiom SchoolCanceled : Prop

axiom H1 : ∀ x : obj, StudiedHard x → GoodGrades x
axiom H2 : Snowstorm → SchoolCanceled
axiom H3 : StudiedHard Priya ∨ ¬ SchoolCanceled
axiom H4 : ∀ x : obj, GoodGrades x → ParentsProud x
axiom H5 : ∀ x : obj, ParentsProud x → IceCreamOuting x
axiom H6 : BigSnowstorm → Snowstorm

theorem big_snowstorm_implies_icecream : BigSnowstorm → IceCreamOuting Priya := by
  intro hb
  have hs : Snowstorm := H6 hb
  have hc : SchoolCanceled := H2 hs
  have hstudy : StudiedHard Priya := by
    cases H3 with
    | inl hs => exact hs
    | inr hnot =>
        have : False := hnot hc
        exact False.elim this
  have hg : GoodGrades Priya := H1 Priya hstudy
  have hp : ParentsProud Priya := H4 Priya hg
  exact H5 Priya hp
</lean>

ANSWER: Yes
