<lean>
axiom obj : Type
axiom Priya : obj

axiom WatersDaily : obj → Prop
axiom GrowWell : obj → Prop
axiom Drought : Prop
axiom Restrictions : Prop
axiom LackedWater : obj → Prop
axiom WiltDie : obj → Prop

axiom A1 : ∀ x : obj, WatersDaily x → GrowWell x
axiom A2 : Drought → Restrictions
axiom A3 : WatersDaily Priya ∨ Drought
axiom A4 : ∀ x : obj, Restrictions → LackedWater x
axiom A5 : ∀ x : obj, LackedWater x → WiltDie x

theorem priya_wilt_die_of_not_grow : ¬ GrowWell Priya → WiltDie Priya := by
  intro hNotGrow
  have hNotWater : ¬ WatersDaily Priya := by
    intro hWat
    exact hNotGrow (A1 Priya hWat)
  have hDrought : Drought := by
    cases A3 with
    | inl hWat => exact False.elim (hNotWater hWat)
    | inr hD => exact hD
  have hRestr : Restrictions := A2 hDrought
  have hLack : LackedWater Priya := (A4 Priya) hRestr
  exact A5 Priya hLack
</lean>

ANSWER: Yes
