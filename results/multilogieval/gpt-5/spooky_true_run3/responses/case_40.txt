<lean>
axiom obj : Type
axiom James : obj

axiom Started : obj → Prop
axiom Motivated : obj → Prop
axiom CanFollow : obj → Prop
axiom Frustrated : obj → Prop
axiom Indulged : obj → Prop
axiom GymClosed : Prop

-- If someone starts a new workout routine, they will be motivated to keep it up.
axiom P1 : ∀ x : obj, Started x → Motivated x
-- If the gym closes, then they can't follow their routine (for anyone).
axiom P2 : GymClosed → ∀ x : obj, ¬ CanFollow x
-- Either they (James) started a new workout, or the gym was closed (or both).
axiom P3 : Started James ∨ GymClosed
-- If they couldn't follow their routine, they would feel frustrated.
axiom P4 : ∀ x : obj, ¬ CanFollow x → Frustrated x
-- If they felt frustrated, they might indulge in some unhealthy snacks.
axiom P5 : ∀ x : obj, Frustrated x → Indulged x

theorem James_indulged (HnotMotivated : ¬ Motivated James) : Indulged James :=
by
  -- From not motivated and "Started -> Motivated", deduce not started
  have h_not_started : ¬ Started James :=
    fun hs => HnotMotivated (P1 James hs)
  -- From (Started ∨ GymClosed) and ¬Started, deduce GymClosed
  have h_closed : GymClosed :=
    Or.elim P3
      (fun hs => False.elim (h_not_started hs))
      (fun hg => hg)
  -- Gym closed implies James cannot follow his routine
  have h_cannot_follow : ¬ CanFollow James :=
    (P2 h_closed) James
  -- Then James is frustrated
  have h_frustrated : Frustrated James :=
    P4 James h_cannot_follow
  -- Hence James indulged in unhealthy snacks
  exact P5 James h_frustrated
</lean>

ANSWER: Yes
