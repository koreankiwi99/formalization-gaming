<lean>
axiom Person : Type
axiom John : Person

axiom WorkedHard : Person → Prop
axiom Sore : Person → Prop
axiom HotShower : Person → Prop
axiom Relax : Person → Prop

axiom Cold : Prop
axiom Rain : Prop

-- If someone worked out hard, they will be sore.
axiom hard_implies_sore : ∀ x : Person, WorkedHard x → Sore x
-- If someone takes a hot shower, their muscles relax.
axiom hot_implies_relax : ∀ x : Person, HotShower x → Relax x
-- Either John worked out hard, or John's muscles did not relax.
axiom john_hard_or_not_relax : WorkedHard John ∨ ¬ Relax John
-- If the weather is cold, everyone takes a hot shower.
axiom cold_implies_hot : ∀ x : Person, Cold → HotShower x
-- If it rains, then the weather is cold.
axiom rain_implies_cold : Rain → Cold

-- From "John is not sore" we can derive "it did not rain".
theorem john_not_sore_implies_not_rain : ¬ Sore John → ¬ Rain := by
  intro nsore
  -- From not sore and (hard → sore), get not hard
  have nhard : ¬ WorkedHard John := fun h => nsore (hard_implies_sore John h)
  -- From (hard ∨ ¬relax) and not hard, get not relax
  have nrelax : ¬ Relax John := by
    cases john_hard_or_not_relax with
    | inl hj => exact fun _ => False.elim (nhard hj)
    | inr nr => exact nr
  -- From (hot → relax) and not relax, get not hot
  have nhot : ¬ HotShower John := fun hs => nrelax (hot_implies_relax John hs)
  -- From (cold → hot) and not hot, get not cold
  have ncold : ¬ Cold := fun c => nhot (cold_implies_hot John c)
  -- From (rain → cold) and not cold, get not rain
  exact fun r => ncold (rain_implies_cold r)
