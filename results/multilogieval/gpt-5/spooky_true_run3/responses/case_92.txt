<lean>
axiom Water : Prop
axiom GrowWell : Prop
axiom PestsInvaded : Prop
axiom DamagedCrops : Prop
axiom HarvestBountiful : Prop
axiom GoodProfit : Prop

axiom H1 : Water → GrowWell
axiom H2 : PestsInvaded → DamagedCrops
axiom H3 : Or Water (Or (Not DamagedCrops) (And Water DamagedCrops))
axiom H4 : GrowWell → HarvestBountiful
axiom H5 : HarvestBountiful → GoodProfit

theorem pests_profit : PestsInvaded → GoodProfit := by
  intro hp
  have hd : DamagedCrops := H2 hp
  have hw : Water := by
    cases H3 with
    | inl hw => exact hw
    | inr h =>
      cases h with
      | inl hnotd =>
        have : False := hnotd hd
        exact False.elim this
      | inr hwd => exact hwd.left
  exact H5 (H4 (H1 hw))
</lean>

ANSWER: Yes
