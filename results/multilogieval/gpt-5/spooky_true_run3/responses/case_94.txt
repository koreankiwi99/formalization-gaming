<lean>
axiom obj : Type
axiom Lauren : obj

axiom TrainHard : obj → Prop
axiom Stretched : obj → Prop
axiom Recovery : obj → Prop
axiom Flex : obj → Prop
axiom RunFast : obj → Prop

-- If they trained hard weekly, they could improve their race times.
axiom A1 : ∀ x : obj, TrainHard x → RunFast x
-- Either they trained hard at sprinting, or they stretched regularly after runs, or both (for Lauren).
axiom A2 : TrainHard Lauren ∨ Stretched Lauren
-- If they stretched after runs, it means they took recovery seriously.
axiom A3 : ∀ x : obj, Stretched x → Recovery x
-- If they took recovery seriously, then they focused on muscle flexibility to prevent injury.
axiom A4 : ∀ x : obj, Recovery x → Flex x

theorem lauren_focuses_on_flex_if_not_fast : (¬ RunFast Lauren) → Flex Lauren := by
  intro hNotFast
  -- From ¬RunFast and (TrainHard → RunFast), get ¬TrainHard
  have hNotTrain : ¬ TrainHard Lauren := by
    intro hTrain
    exact hNotFast (A1 Lauren hTrain)
  -- From (TrainHard ∨ Stretched) and ¬TrainHard, derive Stretched
  have hStretch : Stretched Lauren := by
    apply Or.elim A2
    · intro hTrain
      exact False.elim (hNotTrain hTrain)
    · intro hS
      exact hS
  -- Chain implications to Flex
  have hRec : Recovery Lauren := A3 Lauren hStretch
  exact A4 Lauren hRec
</lean>

ANSWER: Yes
