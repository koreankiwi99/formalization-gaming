<lean>
axiom SetCorrect : Prop
axiom Eggs : Prop
axiom ProperBake : Prop
axiom Fluffy : Prop
axiom Delicious : Prop
axiom Happy : Prop

axiom A1 : SetCorrect → ProperBake
axiom A2 : Eggs → Fluffy
axiom A3 : SetCorrect ∨ ¬ Eggs
axiom A4 : ProperBake → Delicious
axiom A5 : Delicious → Happy

theorem eggs_imply_happy : Eggs → Happy := by
  intro hE
  have hS : SetCorrect :=
    Or.elim A3
      (fun hs => hs)
      (fun hne => False.elim (hne hE))
  have hB : ProperBake := A1 hS
  have hD : Delicious := A4 hB
  have hH : Happy := A5 hD
  exact hH
</lean>

ANSWER: Yes
