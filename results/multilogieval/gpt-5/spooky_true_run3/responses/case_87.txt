<lean>
axiom obj : Type
axiom John : obj

axiom Early : obj → Prop
axiom LongLunch : obj → Prop
axiom Birthday : Prop
axiom Bring : obj → Prop
axiom Recharged : obj → Prop
axiom Productive : obj → Prop

-- Premises
axiom R1 : ∀ x : obj, Early x → LongLunch x
axiom R2 : Birthday → ∀ x : obj, Bring x
axiom R3 : (∀ x : obj, Early x) ∨ Birthday
axiom R4 : ∀ x : obj, LongLunch x → Recharged x
axiom R5 : ∀ x : obj, Recharged x → Productive x

theorem john_productive_from_not_bring : (¬ Bring John) → Productive John := by
  intro hnot
  have hnotB : ¬ Birthday := fun hb => hnot ((R2 hb) John)
  have hallEarly : ∀ x : obj, Early x := by
    cases R3 with
    | inl hall => exact hall
    | inr hb => exact False.elim (hnotB hb)
  have hEarlyJ : Early John := hallEarly John
  have hlunch : LongLunch John := R1 John hEarlyJ
  have hrecharged : Recharged John := R4 John hlunch
  exact R5 John hrecharged
</lean>

ANSWER: Yes
