-- If someone studies hard for an exam, they will feel confident going into the test.
axiom R1 : ∀ x : obj, ∀ e : exam, StudyHard x e → Confident x e
-- If there is severe weather in the forecast, schools may cancel class that day.
axiom R2 : SevereWeather → ClassCanceled
-- Either Alex studied diligently for his biology exam, or class was not canceled due to the weather.
axiom R3 : StudyHard Alex Bio ∨ ¬ ClassCanceled
-- If a student feels confident about the material, they will likely do well on the exam.
axiom R4 : ∀ x : obj, ∀ e : exam, Confident x e → DoWell x e
-- If a student performs well on an exam, they will be satisfied with their hard work.
axiom R5 : ∀ x : obj, ∀ e : exam, DoWell x e → SatisfiedOn x e

-- From severe weather, deduce Alex studied for Biology (using the disjunction and cancellation).
lemma studied_bio_if_weather : SevereWeather → StudyHard Alex Bio := by
  intro sw
  have cc : ClassCanceled := R2 sw
  cases R3 with
  | inl h => exact h
  | inr hnot =>
      have f : False := hnot cc
      exact False.elim f

-- Consequently, from severe weather, Alex is satisfied on the Biology exam.
lemma satisfied_bio_if_weather : SevereWeather → SatisfiedOn Alex Bio := by
  intro sw
  have sh : StudyHard Alex Bio := studied_bio_if_weather sw
  have conf : Confident Alex Bio := R1 Alex Bio sh
  have good : DoWell Alex Bio := R4 Alex Bio conf
  exact R5 Alex Bio good
