<lean>
axiom obj : Type
axiom Emma : obj

axiom LearnBasic : obj → Prop
axiom PlaySimple : obj → Prop
axiom Entertain : obj → Prop
axiom Lessons : obj → Prop
axiom Knowledge : obj → Prop

axiom A1 : ∀ x : obj, LearnBasic x → PlaySimple x
axiom A2 : ∀ x : obj, PlaySimple x → Entertain x
axiom A3 : ∀ x : obj, LearnBasic x ∨ Lessons x
axiom A4 : ∀ x : obj, Lessons x → Knowledge x

-- Emma is unable to entertain friends with the guitar
axiom A5 : ¬ Entertain Emma

theorem emma_gains_knowledge : Knowledge Emma := by
  have hNotLB : ¬ LearnBasic Emma := by
    intro hLB
    have hPS : PlaySimple Emma := A1 Emma hLB
    have hEnt : Entertain Emma := A2 Emma hPS
    exact A5 hEnt
  have hLessons : Lessons Emma :=
    Or.elim (A3 Emma)
      (fun hLB => False.elim (hNotLB hLB))
      (fun hL => hL)
  exact A4 Emma hLessons
</lean>

ANSWER: Yes
