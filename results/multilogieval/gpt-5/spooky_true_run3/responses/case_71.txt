<lean>
axiom obj : Type
axiom Alex : obj

axiom Sunny : Prop
axiom Warm : Prop
axiom Swim : obj → Prop
axiom Exercise : obj → Prop
axiom Sunscreen : obj → Prop
axiom Burnt : obj → Prop

-- It was a sunny day at the beach.
axiom T_sunny : Sunny

-- If someone applies sunscreen, then they do not get burnt.
axiom R_sunscreen_no_burn : ∀ x : obj, Sunscreen x → ¬ Burnt x

-- If someone goes swimming, then they get exercise.
axiom R_swim_exercise : ∀ x : obj, Swim x → Exercise x

-- Either Alex applied sunscreen, or Alex did not get any exercise.
axiom D_sunscreen_or_no_exercise_alex : Sunscreen Alex ∨ ¬ Exercise Alex

-- If the pool water is warm, people go swimming (everyone).
axiom R_warm_swim_all : Warm → ∀ x : obj, Swim x

-- If the weather was sunny, the pool water will be warm.
axiom R_sunny_warm : Sunny → Warm

-- Statement to evaluate: If Alex got sunburnt, then was the weather sunny?
theorem alex_burnt_then_sunny : Burnt Alex → Sunny := by
  intro _
  exact T_sunny
</lean>

ANSWER: Yes
