<lean>
axiom LostImportant : Prop
axiom LostBigMatch : Prop
axiom LostSignificant : Prop
axiom Disappointed : Prop
axiom Demoralized : Prop

-- When a team loses an important game, the players feel disappointed.
axiom R1 : LostImportant → Disappointed
-- Losing a big match is demoralizing.
axiom R2 : LostBigMatch → Demoralized
-- A significant game loss is either an important game loss or a big match loss.
axiom SigCases : LostSignificant → (LostImportant ∨ LostBigMatch)

-- The players are neither disappointed nor demoralized.
axiom NotDisappointed : ¬ Disappointed
axiom NotDemoralized : ¬ Demoralized

-- Therefore, the team did not lose a significant game.
theorem no_significant_loss : ¬ LostSignificant := by
  intro hsig
  cases SigCases hsig with
  | inl hi =>
      exact NotDisappointed (R1 hi)
  | inr hb =>
      exact NotDemoralized (R2 hb)
</lean>

ANSWER: Yes
