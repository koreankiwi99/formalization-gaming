<lean>
axiom person : Type
axiom James : person

axiom Subway : person → Prop
axiom BusDowntown : person → Prop
axiom OnTime : person → Prop
axiom SmoothSailing : person → Prop

-- Transit conditions (not essential for the final conclusion)
axiom SubwayOnSchedule : Prop
axiom BusHasRoom : Prop
axiom subway_exists : SubwayOnSchedule → ∃ x : person, Subway x
axiom bus_exists : BusHasRoom → ∃ x : person, BusDowntown x

-- Dichotomy of commute modes: if not bus, then subway
axiom commute_choice : ∀ x : person, ¬ BusDowntown x → Subway x

-- Taking the subway ensures getting to work on time
axiom subway_on_time : ∀ x : person, Subway x → OnTime x

-- Given: James did not take the bus downtown
axiom James_not_bus : ¬ BusDowntown James

theorem james_on_time : OnTime James :=
  subway_on_time James (commute_choice James James_not_bus)
</lean>

ANSWER: Yes
