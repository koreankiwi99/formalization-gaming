<lean>
axiom obj : Type
axiom Alex : obj

axiom Adv : obj → Prop           -- advertises cleverly
axiom Offer : obj → Prop         -- offers discounts
axiom Attract : obj → Prop       -- attracts more customers
axiom Increase : obj → Prop      -- increases sales
axiom Expand : obj → Prop        -- expands business

axiom H1 : ∀ x : obj, Adv x → Attract x
axiom H2 : ∀ x : obj, Offer x → Increase x
axiom H3 : ∀ x : obj, Adv x ∨ Offer x
axiom H4 : ∀ x : obj, Increase x → Expand x

axiom HAlex : ¬ Attract Alex

theorem alex_expands : Expand Alex := by
  have hNotAdv : ¬ Adv Alex := by
    intro hAdvAlex
    exact HAlex (H1 Alex hAdvAlex)
  have hDisj : Adv Alex ∨ Offer Alex := H3 Alex
  have hOffer : Offer Alex := by
    cases hDisj with
    | inl hAdvAlex => exact False.elim (hNotAdv hAdvAlex)
    | inr hOfferAlex => exact hOfferAlex
  have hIncrease : Increase Alex := H2 Alex hOffer
  exact H4 Alex hIncrease
</lean>

ANSWER: Yes
