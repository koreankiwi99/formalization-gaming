<lean>
axiom obj : Type

axiom Mark : obj

axiom Athlete : obj → Prop
axiom ProperTraining : obj → Prop
axiom HealthyDiet : obj → Prop
axiom EnhancedSkills : obj → Prop
axiom IncreasedStrength : obj → Prop
axiom LackingSkill : obj → Prop
axiom DeficientStrength : obj → Prop
axiom Dedicated : obj → Prop

-- Contextual knowledge
axiom K1 : ∀ x : obj, ProperTraining x → EnhancedSkills x
axiom K2 : ∀ x : obj, HealthyDiet x → IncreasedStrength x
axiom K3 : ∀ x : obj, Athlete x → (LackingSkill x ∨ DeficientStrength x)
axiom K4 : ∀ x : obj, Dedicated x → ProperTraining x
axiom K5 : ∀ x : obj, IncreasedStrength x → ¬ DeficientStrength x

-- Assumptions about Mark for this scenario
axiom A_MarkAthlete : Athlete Mark
axiom A_MarkHealthy : HealthyDiet Mark

-- What we can derive: Mark is not deficient in strength, hence (from the disjunction) he lacks skill.
theorem mark_lacking_skill : LackingSkill Mark := by
  have h_inc : IncreasedStrength Mark := K2 Mark A_MarkHealthy
  have h_not_def : ¬ DeficientStrength Mark := K5 Mark h_inc
  have h_disj : LackingSkill Mark ∨ DeficientStrength Mark := K3 Mark A_MarkAthlete
  exact Or.resolve_right h_disj h_not_def
</lean>
