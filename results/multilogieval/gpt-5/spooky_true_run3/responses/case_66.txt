<lean>
axiom Company : Type
axiom Acme : Company
axiom Innovative : Company → Prop
axiom MarketedEff : Company → Prop
axiom Successful : Company → Prop

-- Context: If a company develops an innovative product, then if they market it effectively, the product will be commercially successful.
axiom rule : ∀ x : Company, Innovative x → (MarketedEff x → Successful x)

-- Facts: Acme marketed effectively; Acme's product was not commercially successful.
axiom marketed_Acme : MarketedEff Acme
axiom not_success_Acme : ¬ Successful Acme

-- From the premises we can derive that Acme did not develop an innovative product.
theorem not_innovative_Acme : ¬ Innovative Acme := by
  intro hInnov
  have hImp : MarketedEff Acme → Successful Acme := rule Acme hInnov
  have hSucc : Successful Acme := hImp marketed_Acme
  exact not_success_Acme hSucc
</lean>
