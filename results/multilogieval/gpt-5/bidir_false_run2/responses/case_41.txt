axiom WatersDaily : obj → Prop
axiom GrowWell : obj → Prop
axiom LackedWater : obj → Prop
axiom WiltDie : obj → Prop

axiom Drought : Prop
axiom Restrictions : Prop

-- If someone waters their plants daily, the plants will grow well.
axiom R1 : ∀ x : obj, WatersDaily x → GrowWell x
-- If there is a drought, then water restrictions will be imposed.
axiom R2 : Drought → Restrictions
-- Either they watered the plants daily, or there was a drought (or both).
axiom T1 : WatersDaily Priya ∨ Drought
-- If water restrictions were imposed, then the plants would not get enough water (lacked water).
axiom R3 : Restrictions → LackedWater Priya
-- If the plants lacked water, they would wilt and die.
axiom R4 : ∀ x : obj, LackedWater x → WiltDie x

-- From "Priya's plants did not grow well", we can in fact derive that they wilted and died.
theorem wilt_die_from_not_grow (H : ¬GrowWell Priya) : WiltDie Priya := by
  have hdisj : WatersDaily Priya ∨ Drought := T1
  cases hdisj with
  | inl hw =>
      have hg : GrowWell Priya := R1 Priya hw
      have : False := H hg
      exact False.elim this
  | inr hd =>
      have hr : Restrictions := R2 hd
      have hl : LackedWater Priya := R3 hr
      exact R4 Priya hl
