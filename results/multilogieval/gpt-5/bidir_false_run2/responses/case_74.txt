<lean>
axiom obj : Type
axiom James : obj

axiom Friday : Prop
axiom Party : obj → Prop
axiom WatchGame : obj → Prop
axiom FinishHW : obj → Prop
axiom StudyEveryday : obj → Prop
axiom PlanWell : obj → Prop
axiom Student : obj → Prop

-- Premises from the context
axiom studentJames : Student James
axiom P1 : ∀ x : obj, Student x → (Friday → Party x)                 -- If it's Friday, students go to the party
axiom P2 : ∀ x : obj, FinishHW x → WatchGame x                       -- If they finish homework, they can watch the game
axiom P3 : ∀ x : obj, Friday ∨ ¬WatchGame x                          -- Either it's Friday night, or they are not watching the game
axiom P4 : ∀ x : obj, StudyEveryday x → FinishHW x                   -- If they study everyday, they finish all their homework
axiom P5 : ∀ x : obj, PlanWell x → StudyEveryday x                   -- If someone plans well, they can study everyday

-- Given statement: James is not going to the party
axiom notPartyJames : ¬Party James

-- Goal: prove that James did not plan his schedule well
theorem james_did_not_plan_well : ¬PlanWell James :=
by
  intro hPlan
  have hStudy : StudyEveryday James := P5 James hPlan
  have hFinish : FinishHW James := P4 James hStudy
  have hWatch : WatchGame James := P2 James hFinish
  have hFriday : Friday :=
    Or.elim (P3 James)
      (fun hF => hF)
      (fun hNotWatch => False.elim (hNotWatch hWatch))
  have hParty : Party James := P1 James studentJames hFriday
  exact notPartyJames hParty
</lean>

ANSWER: No
