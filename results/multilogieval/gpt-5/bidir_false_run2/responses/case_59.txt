<lean>
axiom obj : Type
axiom Emma : obj

axiom LearnedBasicChords : obj → Prop
axiom LearnedStrumming : obj → Prop
axiom CanPlaySimpleSongs : obj → Prop
axiom EntertainFriendsWithGuitar : obj → Prop
axiom TookGuitarLessons : obj → Prop
axiom GainsMusicTheory : obj → Prop

-- If someone learned basic chords and strumming patterns, they could play simple songs.
axiom R1 : ∀ x : obj, LearnedBasicChords x ∧ LearnedStrumming x → CanPlaySimpleSongs x
-- If someone could play simple songs, they would be able to entertain friends with the guitar.
axiom R2 : ∀ x : obj, CanPlaySimpleSongs x → EntertainFriendsWithGuitar x
-- Either they learned basic chords, or they took guitar lessons, or both.
axiom R4 : ∀ x : obj, LearnedBasicChords x ∨ TookGuitarLessons x
-- If someone takes guitar lessons, they gain knowledge of music theory.
axiom R3 : ∀ x : obj, TookGuitarLessons x → GainsMusicTheory x

-- Given: Emma is unable to entertain friends with the guitar.
axiom Emma_not_entertains : ¬ EntertainFriendsWithGuitar Emma

-- From the above, we can show Emma did not both learn chords and strumming.
theorem Emma_not_both_chords_and_strum : ¬ (LearnedBasicChords Emma ∧ LearnedStrumming Emma) := by
  intro h
  have hplay : CanPlaySimpleSongs Emma := R1 Emma h
  have hent : EntertainFriendsWithGuitar Emma := R2 Emma hplay
  exact Emma_not_entertains hent

-- Cannot prove ¬GainsMusicTheory Emma from given axioms.
