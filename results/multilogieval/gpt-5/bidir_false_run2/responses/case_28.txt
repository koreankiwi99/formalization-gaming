<lean>
axiom obj : Type
axiom Jane : obj
axiom Hurt : obj → Prop
axiom SeeDoctor : obj → Prop
axiom RestInjury : obj → Prop

-- If someone gets injured/hurt, they should see a doctor.
axiom R1 : ∀ x : obj, Hurt x → SeeDoctor x
-- Getting hurt warrants resting the affected area.
axiom R2 : ∀ x : obj, Hurt x → RestInjury x

-- Statement to evaluate (used as a premise here):
axiom S : ¬SeeDoctor Jane ∨ ¬RestInjury Jane

theorem jane_not_hurt : ¬Hurt Jane :=
by
  intro hj
  have sd : SeeDoctor Jane := R1 Jane hj
  have rs : RestInjury Jane := R2 Jane hj
  apply Or.elim S
  · intro nsd
    exact nsd sd
  · intro nrs
    exact nrs rs
</lean>

ANSWER: No
