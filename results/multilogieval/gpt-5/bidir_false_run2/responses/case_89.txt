theorem monsoon_implies_profit : Monsoon → Profit := by
  intro hm
  have hhr : HeavyRains := A2 hm
  have hwd : WatersDaily := Or.elim A3
    (fun hwd => hwd)
    (fun nhr => False.elim (nhr hhr))
  have hw : Waters := A6 hwd
  have hhs : Healthy ∧ Strong := A1 hw
  have hs : Strong := And.right hhs
  have hb : Bountiful := A4 hs
  exact A5 hb
