<lean>
axiom obj : Type
axiom Priya : obj
axiom Attend : obj → Prop
axiom Study : obj → Prop
axiom Excel : obj → Prop
axiom Rule : ∀ x : obj, Attend x → (Study x → Excel x)
axiom Priya_studied : Study Priya
axiom Priya_not_excel : ¬Excel Priya

theorem Priya_not_attend : ¬Attend Priya :=
by
  intro hAttend
  have hImp : Study Priya → Excel Priya := Rule Priya hAttend
  have hExcel : Excel Priya := hImp Priya_studied
  exact Priya_not_excel hExcel
</lean>

ANSWER: No
