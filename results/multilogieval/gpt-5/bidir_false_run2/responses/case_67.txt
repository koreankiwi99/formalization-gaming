-- If the stove is on, the kitchen gets hot.
axiom R1 : StoveOn → KitchenHot
-- If someone makes pancakes, they use the stove.
axiom R2 : ∀ x : Person, MakesPancakes x → UsesStove x
-- Either the stove was on, or they did not use the stove (no one used it).
axiom R3 : StoveOn ∨ ¬(∃ x : Person, UsesStove x)
-- If someone is hungry, they make pancakes.
axiom R4 : ∀ x : Person, Hungry x → MakesPancakes x
-- If Sam did not eat dinner last night, he was hungry this morning.
axiom R5 : ¬AteDinnerLastNight Sam → Hungry Sam

-- From "Sam did not eat dinner last night" we can derive "the kitchen got hot".
theorem hot_from_not_eat : ¬AteDinnerLastNight Sam → KitchenHot := by
  intro hNotEat
  have hHungry : Hungry Sam := R5 hNotEat
  have hPanc : MakesPancakes Sam := R4 Sam hHungry
  have hUse : UsesStove Sam := R2 Sam hPanc
  have hExists : ∃ x : Person, UsesStove x := Exists.intro Sam hUse
  have hDisj : StoveOn ∨ ¬(∃ x : Person, UsesStove x) := R3
  have hOn : StoveOn :=
    Or.elim hDisj
      (fun h => h)
      (fun h => False.elim (h hExists))
  exact R1 hOn

open Classical
-- From "the kitchen did not get hot" we can conclude "Sam ate dinner last night".

  intro hNotHot
  by_contra hNotEat
  have hHot : KitchenHot := hot_from_not_eat hNotEat
  exact hNotHot hHot
