axiom Person : Type
axiom Exam : Type
axiom Alex : Person
axiom Biology : Exam
axiom History : Exam

axiom StudiesHard : Person → Exam → Prop
axiom Confident : Person → Exam → Prop
axiom DoWell : Person → Exam → Prop
axiom Satisfied : Person → Exam → Prop

axiom SevereWeather : Prop
axiom Canceled : Prop

-- If someone studies hard for an exam, they will feel confident going into the test.
axiom R1 : ∀ p e, StudiesHard p e → Confident p e
-- If there is severe weather in the forecast, schools may cancel class that day.
axiom R2 : SevereWeather → Canceled
-- Either Alex studied diligently for his biology exam, or class was not canceled due to the weather.
axiom D1 : StudiesHard Alex Biology ∨ ¬Canceled
-- If a student feels confident about the material, they will likely do well on the exam.
axiom R3 : ∀ p e, Confident p e → DoWell p e
-- If a student performs well on an exam, they will be satisfied with their hard work.
axiom R4 : ∀ p e, DoWell p e → Satisfied p e

theorem studied_bio_if_severe : SevereWeather → StudiesHard Alex Biology :=
  fun sw => by
    have c : Canceled := R2 sw
    have not_not_c : ¬(¬Canceled) := fun h => h c
    have hdisj : StudiesHard Alex Biology ∨ ¬Canceled := D1
    exact Or.elim hdisj (fun h => h) (fun hnc => False.elim (not_not_c hnc))

theorem satisfied_bio_if_severe : SevereWeather → Satisfied Alex Biology :=
  fun sw => by
    have sh : StudiesHard Alex Biology := studied_bio_if_severe sw
    have co : Confident Alex Biology := R1 Alex Biology sh
    have dw : DoWell Alex Biology := R3 Alex Biology co
    exact R4 Alex Biology dw
