-- If there is a big game on TV, then friends get together.

-- Either rain, or not friends gathering, or rain and not friends gathering.
axiom D1 : Rain ∨ ¬FriendsGather ∨ (Rain ∧ ¬FriendsGather)
-- If TV viewership is high, then there is a big game on TV.
axiom A3 : HighViewership → BigGame
-- Either high viewership, or they go play out, or both.
axiom D2 : HighViewership ∨ GoPlay ∨ (HighViewership ∧ GoPlay)

-- From "James did not bring an umbrella" we can derive "they go out to play".
theorem go_play_from_not_umbrella : (¬Umbrella James) → GoPlay := by

  have hNotRain : ¬Rain := by
    intro hRain
    have hUmb : Umbrella James := A1 James hRain
    exact hNotUmb hUmb
  have hNotGather : ¬FriendsGather := by
    cases D1 with
    | inl hRain =>
      exact False.elim (hNotRain hRain)
    | inr h =>
      cases h with
      | inl hNotFG => exact hNotFG
      | inr hRandNot => exact hRandNot.right
  have hvToFg : HighViewership → FriendsGather := by
    intro hv
    exact A2 (A3 hv)
  have hNotHV : ¬HighViewership := by
    intro hv
    exact hNotGather (hvToFg hv)
  cases D2 with
  | inl hv =>
    exact False.elim (hNotHV hv)
  | inr rest =>
    cases rest with
