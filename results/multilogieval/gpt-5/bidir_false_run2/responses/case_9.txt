axiom SamParty : Party

axiom BroughtDish : Party → Prop
axiom EnoughFood : Party → Prop
axiom Birthday : Party → Prop
axiom Cake : Party → Prop
axiom IceCream : Party → Prop

-- If someone brought a dish to share at a party, there will be enough food for everyone at that party.
axiom R1 : ∀ p : Party, BroughtDish p → EnoughFood p
-- If it is a birthday party, there will be cake and ice cream.
axiom R2 : ∀ p : Party, Birthday p → (Cake p ∧ IceCream p)
-- Either someone brought a dish to share, or there was no birthday cake (for the party in question).
axiom D : BroughtDish SamParty ∨ ¬Cake SamParty

-- It is Sam's birthday party.
axiom T : Birthday SamParty

-- From the given information, we can actually prove there is enough food at Sam's party.
theorem enough_food_Sam : EnoughFood SamParty :=
by
  have hCake : Cake SamParty := (R2 SamParty T).left
  cases D with
  | inl hDish =>
      exact R1 SamParty hDish
  | inr hNoCake =>
      have contra : False := hNoCake hCake
      exact False.elim contra
