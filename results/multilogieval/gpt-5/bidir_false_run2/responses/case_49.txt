axiom Wet : Prop
axiom Washed : Prop

axiom wet_implies_rain : Wet → Rain
axiom washed_implies_used : Washed → UsedWater
axiom wet_or_noWater_or_both : (Wet ∨ ¬UsedWater) ∨ (Wet ∧ ¬UsedWater)
axiom hose_implies_washed : Hose → Washed
axiom hose_or_soapy_or_both : (Hose ∨ Soapy) ∨ (Hose ∧ Soapy)

-- From "It didn't rain recently" (¬Rain), we can actually prove the sponge is soapy.
theorem soapy_from_not_rain : ¬Rain → Soapy := by
  intro hNotRain
  have hNotWet : ¬Wet := by
    intro hW
    exact hNotRain (wet_implies_rain hW)
  have hNotUsed : ¬UsedWater := by
    cases wet_or_noWater_or_both with
    | inl h1 =>
        cases h1 with
        | inl hW => exact (fun _ => hNotWet hW)
        | inr hNo => exact hNo
    | inr hBoth =>
        have hW : Wet := hBoth.left
        exact (fun _ => hNotWet hW)
  have hNotWashed : ¬Washed := by
    intro hWsh
    apply hNotUsed
    exact washed_implies_used hWsh
  have hNotHose : ¬Hose := by
    intro hH
    exact hNotWashed (hose_implies_washed hH)
  have : Soapy := by
    cases hose_or_soapy_or_both with
    | inl h1 =>
        cases h1 with
        | inl hH => exact False.elim (hNotHose hH)
        | inr hS => exact hS
    | inr hBoth =>
        exact hBoth.right
  exact this
