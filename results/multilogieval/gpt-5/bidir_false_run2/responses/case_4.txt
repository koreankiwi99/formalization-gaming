<lean>
axiom InvestMarketing : Prop
axiom FocusMarketing : Prop
axiom MoreCustomers : Prop
axiom SalesUp : Prop
axiom ProdCostsInc : Prop

-- If they invested in marketing, they would gain more customers.
axiom R1 : InvestMarketing → MoreCustomers
-- With more customers, sales numbers would go up.
axiom R2 : MoreCustomers → SalesUp
-- Focusing budget on marketing implies investing in marketing.
axiom R3 : FocusMarketing → InvestMarketing
-- Either they focused budget on marketing, or production costs increased, or both.
axiom R4 : FocusMarketing ∨ ProdCostsInc ∨ (FocusMarketing ∧ ProdCostsInc)

-- Sales did not increase last quarter.
axiom T1 : ¬SalesUp

theorem fm_implies_sales : FocusMarketing → SalesUp := by
  intro hFM
  have hIM : InvestMarketing := R3 hFM
  have hMC : MoreCustomers := R1 hIM
  exact R2 hMC

theorem not_fm : ¬FocusMarketing := by
  intro hFM
  exact T1 (fm_implies_sales hFM)

-- From the premises, production costs must have increased.
theorem pc_must_have_increased : ProdCostsInc := by
  have h := R4
  cases h with
  | inl hFM =>
      exact False.elim (not_fm hFM)
  | inr h' =>
      cases h' with
      | inl hPC => exact hPC
      | inr hBoth => exact hBoth.right
