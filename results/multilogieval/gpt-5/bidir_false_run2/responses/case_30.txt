<lean>
axiom obj : Type
axiom Mark : obj

axiom Healthy : obj → Prop
axiom Strong : obj → Prop
axiom ProperTraining : obj → Prop
axiom Skilled : obj → Prop
axiom Dedicated : obj → Prop

-- Proper training enhances skills
axiom R1 : ∀ x : obj, ProperTraining x → Skilled x
-- Healthy nutrition increases strength
axiom R2 : ∀ x : obj, Healthy x → Strong x
-- Athletes are lacking in skill or deficient in strength (domain = athletes)
axiom R3 : ∀ x : obj, (¬Skilled x ∨ ¬Strong x)
-- Being dedicated leads to proper training
axiom R4 : ∀ x : obj, Dedicated x → ProperTraining x

-- Mark eats a healthy diet
axiom T1 : Healthy Mark

theorem mark_not_dedicated : ¬Dedicated Mark := by
  intro hded
  have hpt : ProperTraining Mark := R4 Mark hded
  have hsk : Skilled Mark := R1 Mark hpt
  have hst : Strong Mark := R2 Mark T1
  have hdisj : ¬Skilled Mark ∨ ¬Strong Mark := R3 Mark
  cases hdisj with
  | inl hnSk => exact hnSk hsk
  | inr hnSt => exact hnSt hst
</lean>

ANSWER: No
