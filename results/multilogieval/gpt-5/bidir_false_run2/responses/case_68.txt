<lean>
axiom obj : Type
axiom exam : Type

axiom Emma : obj
axiom Bio : exam

axiom StudyHard : obj → Prop
axiom SleepEnough : obj → exam → Prop
axiom PerformWell : obj → exam → Prop
axiom Ace : obj → exam → Prop

-- If someone studies hard, then if they get enough sleep before an exam, they will perform well and ace the test.
axiom Rule : ∀ x : obj, ∀ e : exam, StudyHard x → SleepEnough x e → (PerformWell x e ∧ Ace x e)

-- Facts about Emma and her biology exam
axiom SleepEmma : SleepEnough Emma Bio
axiom NotAceEmma : ¬Ace Emma Bio

theorem emma_not_study_hard : ¬StudyHard Emma := by
  intro hStudy
  have h := Rule Emma Bio hStudy SleepEmma
  have hAce : Ace Emma Bio := And.right h
  exact NotAceEmma hAce
</lean>

ANSWER: No
