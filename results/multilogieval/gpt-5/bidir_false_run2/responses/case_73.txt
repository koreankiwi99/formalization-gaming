<lean>
axiom obj : Type
axiom Simone : obj
axiom TrainsWithDedication : obj → Prop
axiom AdequateRest : obj → Prop
axiom PeakPerformance : obj → Prop

-- Premise: If someone trains with dedication, then if they get adequate rest, they will perform at their peak.
axiom R1 : ∀ x : obj, TrainsWithDedication x → (AdequateRest x → PeakPerformance x)

-- Facts about Simone
axiom RestSimone : AdequateRest Simone
axiom NotPeakSimone : ¬PeakPerformance Simone

theorem simone_not_trained : ¬TrainsWithDedication Simone := by
  intro hTrain
  have hImp : AdequateRest Simone → PeakPerformance Simone := R1 Simone hTrain
  have hPeak : PeakPerformance Simone := hImp RestSimone
  exact NotPeakSimone hPeak
</lean>

ANSWER: No
