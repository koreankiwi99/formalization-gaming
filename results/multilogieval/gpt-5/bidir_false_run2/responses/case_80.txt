axiom Rain : Prop
axiom Game : Prop
axiom PoolOpen : Prop
axiom KidsSwim : Prop
axiom EnjoyPlaying : Prop
axiom EnjoyCatchingBatting : Prop
axiom Smiles : Prop

axiom A1 : ¬Rain → Game
axiom A2 : PoolOpen → KidsSwim
axiom A3 : ¬Rain ∨ PoolOpen
axiom A4 : Game → EnjoyCatchingBatting
axiom A5 : EnjoyPlaying → Smiles

-- From "no kids swam" and the disjunction (¬Rain ∨ PoolOpen) together with (PoolOpen → KidsSwim),
-- we can derive ¬Rain.
theorem not_rain_of_no_swim : ¬KidsSwim → ¬Rain := by
  intro hNoSwim
  have h := A3
  cases h with
  | inl hNotRain => exact hNotRain
  | inr hPoolOpen =>
    have hSwim : KidsSwim := A2 hPoolOpen
    have hFalse : False := hNoSwim hSwim
    exact False.elim hFalse

-- Hence, from "no kids swam" we can derive Game.
theorem game_of_no_swim : ¬KidsSwim → Game := by
  intro hNoSwim
  have hnRain : ¬Rain := not_rain_of_no_swim hNoSwim
  exact A1 hnRain

-- And thus, EnjoyCatchingBatting holds from "no kids swam".
theorem enjoy_cb_of_no_swim : ¬KidsSwim → EnjoyCatchingBatting := by
  intro hNoSwim
  have hGame : Game := game_of_no_swim hNoSwim
  exact A4 hGame
