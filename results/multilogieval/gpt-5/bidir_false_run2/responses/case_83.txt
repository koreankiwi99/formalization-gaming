-- If it is sunny outside, children will play at the park.
axiom R2 : Sunny → KidsAtPark
-- Either the team practiced daily, or kids did not go to the park today.
axiom R3 : PracticesDaily Team ∨ ¬KidsAtPark
-- If a team feels prepared, then they will have confidence.
axiom R4 : Prepared Team → Confident Team
-- When people feel confident, they hold their head up high.
axiom R5 : ∀ x : obj, Confident x → HeadUpHigh x

-- In fact, from the premises we can derive the statement:
theorem sunny_implies_team_headUpHigh : Sunny → HeadUpHigh Team := by
  intro hSunny
  have hKids : KidsAtPark := R2 hSunny
  have hDisj : PracticesDaily Team ∨ ¬KidsAtPark := R3
  cases hDisj with
  | inl hPractice =>
      have hPrepared : Prepared Team := R1 hPractice
      have hConf : Confident Team := R4 hPrepared
      exact R5 Team hConf
  | inr hNotKids =>
      have hFalse : False := hNotKids hKids
      exact False.elim hFalse
