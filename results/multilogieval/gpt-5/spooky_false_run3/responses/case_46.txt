axiom obj : Type
axiom Casey : obj

axiom Volunteers : obj → Prop
axiom Helps : obj → Prop
axiom Adopts : obj → Prop
axiom Home : obj → Prop
axiom Fulfill : obj → Prop

-- If someone volunteers at a local shelter, they will help the community.
axiom A1 : ∀ x : obj, Volunteers x → Helps x
-- If someone adopts a pet from the shelter, then they provide a home to a pet in need.
axiom A2 : ∀ x : obj, Adopts x → Home x
-- Either someone volunteers at the shelter, or they do not provide a home to a pet in need,
-- or someone volunteered at the shelter, and they did not provide a home to a pet in need.
axiom A3 : ∀ x : obj, Volunteers x ∨ ¬Home x ∨ (Volunteers x ∧ ¬Home x)
-- If someone helps the community, then they feel a sense of fulfillment.
axiom A4 : ∀ x : obj, Helps x → Fulfill x

theorem adopt_implies_fulfill_all : ∀ x : obj, Adopts x → Fulfill x := by
  intro x hAdopt
  have hHome : Home x := A2 x hAdopt
  have hVol : Volunteers x := by
    have h := A3 x
    cases h with
    | inl hV => exact hV
    | inr rest =>
      cases rest with
      | inl hNotHome =>
        exact False.elim (hNotHome hHome)
      | inr hBoth =>
        exact hBoth.left
  have hHelp : Helps x := A1 x hVol
  exact A4 x hHelp

theorem casey_adopt_implies_fulfill : Adopts Casey → Fulfill Casey :=
  adopt_implies_fulfill_all Casey
