<lean>
axiom obj : Type
axiom SamParty : obj

axiom BirthdayParty : obj → Prop
axiom Cake : obj → Prop
axiom IceCream : obj → Prop
axiom BroughtDish : obj → Prop
axiom EnoughFood : obj → Prop

-- Premises
axiom P1 : ∀ p : obj, BroughtDish p → EnoughFood p
axiom P2 : ∀ p : obj, BirthdayParty p → (Cake p ∧ IceCream p)
axiom P3 : ∀ p : obj, BroughtDish p ∨ ¬Cake p

-- Given: It is Sam's birthday party
axiom Sam_bday : BirthdayParty SamParty

-- From the premises, we can derive that there is enough food at Sam's party
theorem enough_food_for_everyone_at_Sam : EnoughFood SamParty := by
  have hCI : Cake SamParty ∧ IceCream SamParty := P2 SamParty Sam_bday
  have hCake : Cake SamParty := And.left hCI
  have hDisj : BroughtDish SamParty ∨ ¬Cake SamParty := P3 SamParty
  have hDish : BroughtDish SamParty :=
    Or.elim hDisj
      (fun hdish => hdish)
      (fun nCake => False.elim (nCake hCake))
  exact P1 SamParty hDish
</lean>
