-- Premises about the specific student in the town
axiom P1 : Studied Student → GoodGrades Student
axiom P2 : Activities Student → NewFriends Student
axiom P3 : Studied Student ∨ Activities Student
axiom P4 : ∀ x : obj, NewFriends x → Invited x

-- Given fact about Alex
axiom Alex_not_good : ¬GoodGrades Alex

-- An unrelated but derivable fact about the specific Student (not Alex):
-- If that Student did not get good grades, then they are invited.
theorem student_invited_if_not_good : (¬GoodGrades Student) → Invited Student := by
  intro hng
  have hns : ¬Studied Student := by
    intro hs
    have hg : GoodGrades Student := P1 hs
    exact hng hg
  have ha : Activities Student := Or.elim P3
    (fun hs => False.elim (hns hs))
    (fun ha => ha)
  have nf : NewFriends Student := P2 ha
  exact P4 Student nf
