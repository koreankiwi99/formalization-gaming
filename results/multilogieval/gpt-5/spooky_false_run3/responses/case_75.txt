<lean>
axiom obj : Type
axiom Liz : obj

axiom Preventive : obj → Prop
axiom QualityParts : obj → Prop
axiom LongLife : obj → Prop

-- If preventive maintenance is done, then if quality parts are used, the car will have a long service life.
axiom R : ∀ x : obj, Preventive x → (QualityParts x → LongLife x)

-- Facts about Liz's car
axiom QLiz : QualityParts Liz
axiom nLLiz : ¬LongLife Liz

theorem not_preventive_Liz : ¬Preventive Liz := by
  intro hP
  have hImp : QualityParts Liz → LongLife Liz := R Liz hP
  have hLL : LongLife Liz := hImp QLiz
  exact nLLiz hLL
</lean>

ANSWER: No
