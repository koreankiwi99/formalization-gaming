axiom obj : Type

-- Entities
axiom Emma : obj

-- Predicates
axiom SpeakDaily : obj → Prop
axiom FluencyImprove : obj → Prop
axiom VocabImprove : obj → Prop
axiom RealConvo : obj → Prop
axiom ListenPod : obj → Prop
axiom ListeningComp : obj → Prop

-- Premises from the context
-- If someone speaks daily with native speakers, their fluency (and vocabulary) improves.
axiom P1a : ∀ x : obj, SpeakDaily x → FluencyImprove x
axiom P1b : ∀ x : obj, SpeakDaily x → VocabImprove x

-- If someone's fluency improves, they can have real conversations.
axiom P2 : ∀ x : obj, FluencyImprove x → RealConvo x

-- Either someone speaks daily with native speakers or they listen to educational podcasts (or both).
axiom P3 : ∀ x : obj, SpeakDaily x ∨ ListenPod x

-- If someone listens to educational podcasts, they gain valuable listening comprehension.
axiom P4 : ∀ x : obj, ListenPod x → ListeningComp x

-- Derived helper: speaking daily implies being able to have real conversations.
theorem speak_daily_to_real : ∀ x : obj, SpeakDaily x → RealConvo x :=
  fun x => fun hsd => P2 x (P1a x hsd)

-- Target statement: If Emma is unable to have real conversations, then she gains valuable listening comprehension.
-- We prove the statement is TRUE from the premises.
theorem emma_listening_from_not_convo : (¬RealConvo Emma) → ListeningComp Emma := by
  intro hNotRC
  have hDisj : SpeakDaily Emma ∨ ListenPod Emma := P3 Emma
  cases hDisj with
  | inl hSD =>
      have hRC : RealConvo Emma := speak_daily_to_real Emma hSD
      exact False.elim (hNotRC hRC)
  | inr hLP =>
      exact P4 Emma hLP
