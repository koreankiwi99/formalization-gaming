<lean>
axiom Person : Type
axiom Priya : Person

axiom StudyHard : Person → Prop
axiom GoodGrades : Person → Prop
axiom Proud : Person → Prop
axiom IceCream : Person → Prop

axiom Snowstorm : Prop
axiom BigSnowstorm : Prop
axiom SchoolCanceled : Prop

-- Premises
axiom P1 : ∀ x : Person, StudyHard x → GoodGrades x
axiom P2 : Snowstorm → SchoolCanceled
axiom P3 : StudyHard Priya ∨ ¬SchoolCanceled
axiom P4 : ∀ x : Person, GoodGrades x → Proud x
axiom P5 : ∀ x : Person, Proud x → IceCream x
-- Lexical link: a big snowstorm is a snowstorm
axiom P6 : BigSnowstorm → Snowstorm

theorem icecream_if_snowstorm : Snowstorm → IceCream Priya :=
by
  intro hs
  have hsc : SchoolCanceled := P2 hs
  have h_study : StudyHard Priya :=
    Or.elim P3
      (fun h => h)
      (fun hnc => False.elim (hnc hsc))
  have hg : GoodGrades Priya := P1 Priya h_study
  have hp : Proud Priya := P4 Priya hg
  exact P5 Priya hp

theorem icecream_if_big_snowstorm : BigSnowstorm → IceCream Priya :=
by
  intro hbs
  have hs : Snowstorm := P6 hbs
  exact icecream_if_snowstorm hs
