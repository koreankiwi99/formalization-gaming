axiom Enjoy : obj → Prop

axiom A1 : ∀ x : obj, MadeFriends x → Enjoy x

axiom A2 : ∀ x : obj, JoinClub x → Busy x
-- If someone is outgoing, then they would join a club.
axiom A3 : ∀ x : obj, Outgoing x → JoinClub x
-- If someone introduced themselves to classmates, then they are outgoing.
axiom A4 : ∀ x : obj, Introduced x → Outgoing x
-- Either they made friends, or they are not busy. (interpreted universally)
axiom A5 : ∀ x : obj, MadeFriends x ∨ ¬Busy x

-- From the disjunction, derive: if Busy then MadeFriends.
lemma busy_implies_madefriends : ∀ x : obj, Busy x → MadeFriends x := by
  intro x hb
  have h := A5 x
  cases h with
  | inl hm => exact hm

-- Introduced ⇒ Outgoing ⇒ JoinClub ⇒ Busy ⇒ MadeFriends ⇒ Enjoy

  intro x hi
  have ho := A4 x hi
  have hj := A3 x ho
  have hb := A2 x hj
  have hm := busy_implies_madefriends x hb
  exact A1 x hm

-- Contrapositive: ¬Enjoy ⇒ ¬Introduced
lemma not_enjoy_implies_not_introduced : ∀ x : obj, ¬Enjoy x → ¬Introduced x := by
  intro x hne
