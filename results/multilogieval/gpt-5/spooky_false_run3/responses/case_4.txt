-- With more customers, sales numbers would go up.

-- Either they focused budget on marketing, or production costs increased, or both.

-- From the first two premises, marketing implies sales up.
theorem M_to_S : Marketing → SalesUp := by
  intro hm
  apply C_to_S
  exact M_to_C hm

-- Contraposition: if sales did not increase, then they did not focus on marketing.

  intro hns
  intro hm
  apply hns
  exact M_to_S hm
