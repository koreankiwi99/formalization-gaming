<lean>
axiom Person : Type
axiom Emma : Person

axiom LearnedBasicChords : Person → Prop
axiom LearnedStrummingPatterns : Person → Prop
axiom CanPlaySimpleSongs : Person → Prop
axiom CanEntertainFriendsWithGuitar : Person → Prop
axiom TookLessons : Person → Prop
axiom KnowsMusicTheory : Person → Prop

-- If someone learned basic chords and strumming patterns, they can play simple songs.
axiom R1 : ∀ x : Person, LearnedBasicChords x ∧ LearnedStrummingPatterns x → CanPlaySimpleSongs x
-- If someone can play simple songs, they can entertain friends with the guitar.
axiom R2 : ∀ x : Person, CanPlaySimpleSongs x → CanEntertainFriendsWithGuitar x
-- Either the specific person (not necessarily Emma) learned chords or took lessons (or both).
axiom Someone : Person
axiom R3_someone : LearnedBasicChords Someone ∨ TookLessons Someone
-- If someone takes guitar lessons, they gain knowledge of music theory.
axiom R4 : ∀ x : Person, TookLessons x → KnowsMusicTheory x

-- Given: Emma is unable to entertain friends with the guitar.
axiom Emma_not_entertain : ¬CanEntertainFriendsWithGuitar Emma

-- What we can derive from the given implications and Emma_not_entertain:
theorem emma_not_simple : ¬CanPlaySimpleSongs Emma := by
  intro h
  have h2 : CanEntertainFriendsWithGuitar Emma := R2 Emma h
  exact Emma_not_entertain h2

theorem emma_not_both_chords_and_strumming : ¬(LearnedBasicChords Emma ∧ LearnedStrummingPatterns Emma) := by
  intro h
  have hs : CanPlaySimpleSongs Emma := R1 Emma h
  exact emma_not_simple hs
