axiom Jesse : obj

axiom SprintsFrequently : obj → Prop
axiom RunningImproved : obj → Prop
axiom GoalTime : obj → Prop
axiom StretchingClasses : obj → Prop
axiom Flexible : obj → Prop
axiom LessProneInjury : obj → Prop

-- If someone sprints frequently, their running form and stamina improve.
axiom A1 : ∀ x : obj, SprintsFrequently x → RunningImproved x
-- If running improved, they could achieve their goal time.
axiom A2 : ∀ x : obj, RunningImproved x → GoalTime x
-- Either Jesse was sprinting frequently or taking stretching classes (or both).
axiom A3 : SprintsFrequently Jesse ∨ StretchingClasses Jesse
-- If someone takes stretching classes, they become more flexible and less prone to injury.
axiom A4 : ∀ x : obj, StretchingClasses x → Flexible x ∧ LessProneInjury x
-- Given: Improving running form and stamina did not happen for Jesse.
axiom A5 : ¬RunningImproved Jesse

theorem not_sprint_Jesse : ¬SprintsFrequently Jesse := by
  intro hSprint
  have hRun : RunningImproved Jesse := A1 Jesse hSprint
  exact A5 hRun

theorem stretching_Jesse : StretchingClasses Jesse := by
  have hns : ¬SprintsFrequently Jesse := not_sprint_Jesse
  cases A3 with
  | inl hs => exact False.elim (hns hs)
  | inr hst => exact hst

theorem jesse_flexible_lessInjury : Flexible Jesse ∧ LessProneInjury Jesse := by
  exact A4 Jesse stretching_Jesse
