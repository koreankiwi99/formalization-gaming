<lean>
axiom CloseLunch : Prop
axiom Hungry : Prop
axiom PrepareWell : Prop
axiom Ready : Prop
axiom Diligent : Prop

-- If it is close to lunchtime, then coworkers are hungry.
axiom R1 : CloseLunch → Hungry
-- If someone prepares well for the presentation, then they are ready.
axiom R2 : PrepareWell → Ready
-- If someone is diligent, they prepare well for presentations.
axiom R3 : Diligent → PrepareWell
-- Either it is close to lunchtime, or they are not ready, or both.
axiom Cert : CloseLunch ∨ (¬Ready ∨ (CloseLunch ∧ ¬Ready))

-- From not hungry, derive not ready.
theorem notHungry_implies_notReady : ¬Hungry → ¬Ready := by
  intro nh
  have ncl : ¬CloseLunch := by
    intro cl
    exact nh (R1 cl)
  cases Cert with
  | inl cl =>
      exact fun r => False.elim (ncl cl)
  | inr rest =>
      cases rest with
      | inl nr => exact nr
      | inr clnr =>
          exact fun r => False.elim (ncl clnr.left)

-- Diligent implies ready.
theorem diligent_implies_ready : Diligent → Ready := by
  intro d
  exact R2 (R3 d)

-- From not hungry, derive not diligent.
theorem notHungry_implies_notDiligent : ¬Hungry → ¬Diligent := by
  intro nh
  intro d
  have rdy : Ready := diligent_implies_ready d
  have nr : ¬Ready := notHungry_implies_notReady nh
  exact nr rdy

-- Hence, assuming coworkers are not hungry, the implication (¬Hungry → Diligent) is false.
theorem negate_claim_from_nh : (¬Hungry) → ¬(¬Hungry → Diligent) := by
  intro nh
  intro himp
  have nd : ¬Diligent := notHungry_implies_notDiligent nh
  exact nd (himp nh)
</lean>
