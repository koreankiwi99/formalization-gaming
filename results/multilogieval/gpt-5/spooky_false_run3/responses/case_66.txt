<lean>
axiom obj : Type
axiom Acme : obj
axiom DevelopedInnovative : obj → Prop
axiom MarketedEffectively : obj → Prop
axiom CommerciallySuccessful : obj → Prop

-- If a company develops an innovative product, then if they market it effectively, the product will be commercially successful.
axiom R1 : ∀ c : obj, DevelopedInnovative c → (MarketedEffectively c → CommerciallySuccessful c)

-- Acme marketed their product effectively.
axiom T1 : MarketedEffectively Acme

-- Acme's product was not commercially successful.
axiom T2 : ¬CommerciallySuccessful Acme

theorem acme_not_developed_innovative : ¬DevelopedInnovative Acme := by
  intro h
  have hImp : MarketedEffectively Acme → CommerciallySuccessful Acme := R1 Acme h
  have hSucc : CommerciallySuccessful Acme := hImp T1
  exact T2 hSucc
</lean>

ANSWER: No
