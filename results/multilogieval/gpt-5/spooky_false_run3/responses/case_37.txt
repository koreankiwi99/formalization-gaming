axiom Relax : obj → Prop
axiom Read : obj → Prop
axiom Packed : obj → Prop
axiom Prepared : obj → Prop

axiom R1 : ∀ x : obj, Nice x → Relax x
axiom R2 : ∀ x : obj, Read x → Packed x
axiom R3 : ∀ x : obj, Nice x ∨ ¬Packed x ∨ (Nice x ∧ ¬Packed x)
axiom R4 : ∀ x : obj, Prepared x → Read x

theorem not_relax_implies_not_prepared_forall : ∀ x : obj, ¬Relax x → ¬Prepared x := by
  intro x hnotrel
  have hnotnice : ¬Nice x := by
    intro hnice
    exact hnotrel (R1 x hnice)
  have hnotpacked : ¬Packed x := by
    have hdisj := R3 x
    cases hdisj with
    | inl hNice =>
        exact False.elim (hnotnice hNice)
    | inr h₂ =>
        cases h₂ with
        | inl hNotPacked => exact hNotPacked
        | inr hBoth => exact hBoth.right
  have hnotread : ¬Read x := by
    intro hread
    exact hnotpacked (R2 x hread)
  intro hprep
  exact hnotread (R4 x hprep)

theorem jake_not_prepared_if_cannot_relax : ¬Relax Jake → ¬Prepared Jake :=
  not_relax_implies_not_prepared_forall Jake
