<lean>
axiom LostImportant : Prop
axiom LostBig : Prop
axiom Disappointed : Prop
axiom Demoralized : Prop

-- Context rules
axiom R1 : LostImportant → Disappointed
axiom R2 : LostBig → Demoralized

-- Given statement: players are neither disappointed nor demoralized
axiom H : ¬Disappointed ∧ ¬Demoralized

-- "Significant game" modeled as either an important game or a big match
def SignificantLoss : Prop := LostImportant ∨ LostBig

theorem not_significant_loss : ¬SignificantLoss := by
  intro hSig
  have hNotD : ¬Disappointed := And.left H
  have hNotM : ¬Demoralized := And.right H
  cases hSig with
  | inl hImp =>
    exact hNotD (R1 hImp)
  | inr hBig =>
    exact hNotM (R2 hBig)
</lean>

ANSWER: No
