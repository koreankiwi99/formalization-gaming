<lean>
axiom person : Type
axiom Jane : person

axiom Exercises : person → Prop
axiom Fit : person → Prop
axiom EatsWell : person → Prop
axiom Energetic : person → Prop
axiom ValuesWellness : person → Prop

-- If people exercised regularly, they would be fit.
axiom H1 : ∀ x : person, Exercises x → Fit x
-- If they ate nutritious foods, they would have energy.
axiom H2 : ∀ x : person, EatsWell x → Energetic x
-- Not everyone in town was active or energetic.
-- We interpret "active" as "exercises regularly", giving a witness that is neither exercising nor energetic.
axiom H3 : ∃ x : person, ¬Exercises x ∧ ¬Energetic x
-- If someone valued wellness, they would exercise regularly.
axiom H4 : ∀ x : person, ValuesWellness x → Exercises x

-- Given statement: Jane eats very nutritious foods.
axiom Jane_eats : EatsWell Jane

-- From Jane_eats and H2, Jane has energy.
theorem jane_energetic : Energetic Jane := by
  have hw : EatsWell Jane := Jane_eats
  exact H2 Jane hw

-- From H3 and H2, some citizen must not eat well (mayor's conclusion, one part).
theorem some_not_eat : ∃ x : person, ¬EatsWell x := by
  rcases H3 with ⟨y, hy⟩
  have hne : ¬EatsWell y := by
    intro ew
    have eny : Energetic y := H2 y ew
    exact hy.right eny
  exact ⟨y, hne⟩

-- From H3, directly: some citizen must not exercise or must not eat well.
theorem some_not_exercise_or_not_eat : ∃ x : person, ¬Exercises x ∨ ¬EatsWell x := by
  rcases H3 with ⟨y, hy⟩
  exact ⟨y, Or.inl hy.left⟩
