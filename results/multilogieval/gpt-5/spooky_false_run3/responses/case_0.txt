axiom ImproveCredit : Prop
axiom AffordDownPayment : Prop
axiom QualifyMortgage : Prop
axiom BuyDreamHome : Prop

axiom IC_implies_QM : ImproveCredit → QualifyMortgage
axiom SA_or_IC : SaveAggressive ∨ ImproveCredit
axiom QM_implies_Buy : QualifyMortgage → BuyDreamHome
axiom not_ADP : ¬AffordDownPayment

theorem bought_dream_home : BuyDreamHome := by
  have not_SA : ¬SaveAggressive := by
    intro hSA
    have hADP : AffordDownPayment := SA_implies_ADP hSA
    exact not_ADP hADP
  have hIC : ImproveCredit := by
    cases SA_or_IC with
    | inl hSA =>
        exact False.elim (not_SA hSA)
    | inr hIC =>
        exact hIC
  have hQM : QualifyMortgage := IC_implies_QM hIC
  exact QM_implies_Buy hQM
