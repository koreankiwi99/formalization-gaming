<lean>
axiom person : Type
axiom James : person

axiom Started : person → Prop
axiom Motivated : person → Prop
axiom GymClosed : Prop
axiom CannotFollow : person → Prop
axiom Frustrated : person → Prop
axiom Indulged : person → Prop

-- If someone starts a new workout routine, they will be motivated to keep it up.
axiom A1 : ∀ x : person, Started x → Motivated x
-- If the gym closes for renovations, then they can't follow their routine.
axiom A2 : ∀ x : person, GymClosed → CannotFollow x
-- If they couldn't follow their routine, they would feel frustrated.
axiom A3 : ∀ x : person, CannotFollow x → Frustrated x
-- If they felt frustrated, they might indulge in some unhealthy snacks.
axiom A4 : ∀ x : person, Frustrated x → Indulged x

-- Either they started a new workout, or the gym was closed (or both).
axiom D1 : Started James ∨ GymClosed
-- James was not motivated to workout.
axiom NotMotivatedJames : ¬Motivated James

theorem james_indulged : Indulged James :=
by
  have h_not_started : ¬Started James :=
    fun hs => NotMotivatedJames (A1 James hs)
  have h_closed : GymClosed :=
    Or.elim D1
      (fun hs => False.elim (h_not_started hs))
      (fun hg => hg)
  have h_cannot : CannotFollow James := A2 James h_closed
  have h_frustrated : Frustrated James := A3 James h_cannot
  exact A4 James h_frustrated
</lean>
