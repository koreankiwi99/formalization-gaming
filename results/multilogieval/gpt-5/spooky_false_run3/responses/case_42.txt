<lean>
axiom Sunny : Prop
axiom WantPlayOutside : Prop
axiom PlayOutside : Prop
axiom FluSeasonStarted : Prop
axiom ManyChildrenSick : Prop
axiom SleepWellAtNight : Prop

-- If the weather is sunny, children will want to play outside.
axiom R1 : Sunny → WantPlayOutside

-- If flu season begins, then many children will get sick.
axiom R2 : FluSeasonStarted → ManyChildrenSick

-- Either it is sunny outside, or not many kids get the flu, or it was sunny and many kids did not get sick.
axiom R3 : Or (Or Sunny (¬ManyChildrenSick)) (Sunny ∧ ¬ManyChildrenSick)

-- If children play outside, they will sleep well at night.
axiom R4 : PlayOutside → SleepWellAtNight

-- Assumption from the statement: Flu season started this month.
axiom Hflu : FluSeasonStarted

-- Some derivable consequences (but they do not yield ¬SleepWellAtNight).
theorem many_sick : ManyChildrenSick := by
  exact R2 Hflu

theorem sunny : Sunny := by
  have ms : ManyChildrenSick := R2 Hflu
  cases R3 with
  | inl h =>
      cases h with
      | inl s => exact s
      | inr not_ms =>
          have f : False := not_ms ms
          exact False.elim f
  | inr both =>
      exact both.left

theorem want_to_play : WantPlayOutside := by
  apply R1
  exact sunny
</lean>
