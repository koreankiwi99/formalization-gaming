<lean>
axiom Person : Type
axiom Jake : Person

axiom WeatherNice : Prop
axiom RelaxOutside : Person → Prop
axiom Prepared : Person → Prop
axiom ReadsBooks : Person → Prop
axiom PackedEnough : Person → Prop

-- If the weather is nice, then anyone can relax outside.
axiom A1 : ∀ x : Person, WeatherNice → RelaxOutside x
-- If someone reads books, then they packed enough.
axiom A2 : ∀ x : Person, ReadsBooks x → PackedEnough x
-- If someone is prepared, they read books.
axiom A3 : ∀ x : Person, Prepared x → ReadsBooks x
-- Either the weather is nice, or a person did not pack enough, or both.
axiom A4 : ∀ x : Person, WeatherNice ∨ ¬PackedEnough x ∨ (WeatherNice ∧ ¬PackedEnough x)

theorem jake_not_prepared_if_cannot_relax : ¬RelaxOutside Jake → ¬Prepared Jake := by
  intro hnr
  intro hprep
  have hreads : ReadsBooks Jake := A3 Jake hprep
  have hpack : PackedEnough Jake := A2 Jake hreads
  have hnotW : ¬WeatherNice := mt (A1 Jake) hnr
  have hdisj : WeatherNice ∨ ¬PackedEnough Jake ∨ (WeatherNice ∧ ¬PackedEnough Jake) := A4 Jake
  have hnotpack : ¬PackedEnough Jake := by
    cases hdisj with
    | inl hW => exact False.elim (hnotW hW)
    | inr h2 =>
      cases h2 with
      | inl hnp => exact hnp
      | inr hWnp => exact False.elim (hnotW hWnp.left)
  exact hnotpack hpack
</lean>
