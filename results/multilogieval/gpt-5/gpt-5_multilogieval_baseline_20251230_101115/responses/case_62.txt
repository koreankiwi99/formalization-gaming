<lean>
axiom Person : Type
axiom Mark : Person

axiom MadeFriends : Person → Prop
axiom Enjoy : Person → Prop
axiom Joins : Person → Prop
axiom Busy : Person → Prop
axiom Outgoing : Person → Prop
axiom Introduced : Person → Prop

-- If someone made friends, then they would enjoy school.
axiom R1 : ∀ x : Person, MadeFriends x → Enjoy x
-- If someone joins a club, then they will be busy.
axiom R2 : ∀ x : Person, Joins x → Busy x
-- Either they made friends, or they are not busy. (modeled universally)
axiom R3 : ∀ x : Person, MadeFriends x ∨ ¬ Busy x
-- If someone is outgoing, then they would join a club.
axiom R4 : ∀ x : Person, Outgoing x → Joins x
-- If someone introduced themselves to classmates, then they are outgoing.
axiom R5 : ∀ x : Person, Introduced x → Outgoing x

-- From the given rules, if someone introduced themselves, then they enjoyed school.
theorem introduced_implies_enjoy : ∀ x : Person, Introduced x → Enjoy x := by
  intro x
  intro hIntro
  have hOut : Outgoing x := R5 x hIntro
  have hJoin : Joins x := R4 x hOut
  have hBusy : Busy x := R2 x hJoin
  have hMF : MadeFriends x := by
    have hdisj := R3 x
    cases hdisj with
    | inl hmf =>
      exact hmf
    | inr hNotBusy =>
      have hFalse : False := hNotBusy hBusy
      exact False.elim hFalse
  exact R1 x hMF

-- Contrapositive: If someone did not enjoy, then they did not introduce themselves.
theorem not_enjoy_implies_not_introduced : ∀ x : Person, ¬ Enjoy x → ¬ Introduced x := by
  intro x
  intro hNotEnjoy
  intro hIntro
  have hEnjoy : Enjoy x := introduced_implies_enjoy x hIntro
  exact hNotEnjoy hEnjoy

-- Specialize to Mark.
theorem mark_not_enjoy_implies_not_introduced : ¬ Enjoy Mark → ¬ Introduced Mark := by
  exact not_enjoy_implies_not_introduced Mark
