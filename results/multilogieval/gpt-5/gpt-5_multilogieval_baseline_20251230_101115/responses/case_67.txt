<lean>
open Classical

axiom obj : Type
axiom Sam : obj

axiom StoveOn : Prop
axiom KitchenHot : Prop
axiom UseStove : obj → Prop
axiom MakesPancakes : obj → Prop
axiom Hungry : obj → Prop
axiom AteDinner : obj → Prop

axiom R1 : StoveOn → KitchenHot
axiom R2 : ∀ x : obj, MakesPancakes x → UseStove x
axiom R3 : StoveOn ∨ ¬ UseStove Sam
axiom R4 : ∀ x : obj, Hungry x → MakesPancakes x
axiom R5 : ¬ AteDinner Sam → Hungry Sam

theorem sam_ate_from_not_hot : ¬ KitchenHot → AteDinner Sam :=
fun nh => by
  have hNotOn : ¬ StoveOn := fun hOn => nh (R1 hOn)
  have hNotUse : ¬ UseStove Sam := by
    cases R3 with
    | inl hOn => exact False.elim (hNotOn hOn)
    | inr hNUse => exact hNUse
  by_contra hNotEat
  have hHun : Hungry Sam := R5 hNotEat
  have hPan : MakesPancakes Sam := R4 Sam hHun
  have hUse : UseStove Sam := R2 Sam hPan
  exact hNotUse hUse
</lean>

ANSWER: Yes

fun nh => by

  have hNotOn : ¬ StoveOn := fun hOn => nh (R1 hOn)

    Or.elim R3
      (fun hOn => False.elim (hNotOn hOn))
      (fun hNUse => hNUse)

    have hUse : UseStove Sam := R2 Sam hPan
