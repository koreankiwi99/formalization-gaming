<lean>
axiom obj : Type
axiom Alex : obj

axiom WeekendComing : Prop
axiom Relax : obj → Prop
axiom FinishTasks : obj → Prop
axiom ClearMind : obj → Prop
axiom Organized : obj → Prop
axiom TodoList : obj → Prop

axiom A1 : ∀ x : obj, WeekendComing → Relax x
axiom A2 : ∀ x : obj, FinishTasks x → ClearMind x
axiom A3 : ∀ x : obj, WeekendComing ∨ ¬ ClearMind x
axiom A4 : ∀ x : obj, Organized x → FinishTasks x
axiom A5 : ∀ x : obj, TodoList x → Organized x

axiom not_relax_Alex : ¬ Relax Alex

theorem alex_not_todolist : ¬ TodoList Alex := by
  have h_not_clear : ¬ ClearMind Alex := by
    have disj := A3 Alex
    cases disj with
    | inl hW =>
      have hrel : Relax Alex := A1 Alex hW
      have contra : False := not_relax_Alex hrel
      exact False.elim contra
    | inr hnc => exact hnc
  have h_not_finish : ¬ FinishTasks Alex := by
    intro hfin
    exact h_not_clear (A2 Alex hfin)
  have h_not_org : ¬ Organized Alex := by
    intro horg
    exact h_not_finish (A4 Alex horg)
  intro htodo
  exact h_not_org (A5 Alex htodo)
</lean>

ANSWER: No
