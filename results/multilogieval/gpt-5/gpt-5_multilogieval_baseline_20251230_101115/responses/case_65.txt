<lean>
axiom obj : Type
axiom Bookshelf : obj

axiom HighQualityWood : obj → Prop
axiom PreciseJoints : obj → Prop
axiom Sturdy : obj → Prop
axiom LongLasting : obj → Prop

-- If the wood is high quality, then if the joints are precise, the product is sturdy and long-lasting
axiom R1 : ∀ x : obj, HighQualityWood x → (PreciseJoints x → (Sturdy x ∧ LongLasting x))

-- Facts about the bookshelf
axiom JointsPrecise : PreciseJoints Bookshelf
axiom NotSturdy : ¬ Sturdy Bookshelf
axiom NotLongLasting : ¬ LongLasting Bookshelf

theorem wood_not_high_quality : ¬ HighQualityWood Bookshelf := by
  intro hq
  have hgood : Sturdy Bookshelf ∧ LongLasting Bookshelf := R1 Bookshelf hq JointsPrecise
  exact NotSturdy hgood.left
</lean>
