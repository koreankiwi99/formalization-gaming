<lean>
axiom obj : Type
axiom Team : obj

axiom PracticesDaily : obj → Prop
axiom Prepared : obj → Prop
axiom Confident : obj → Prop
axiom HeadUpHigh : obj → Prop

axiom Sunny : Prop
axiom KidsAtPark : Prop

axiom R1 : ∀ x : obj, PracticesDaily x → Prepared x
axiom R2 : Sunny → KidsAtPark
axiom R3 : PracticesDaily Team ∨ ¬ KidsAtPark
axiom R4 : ∀ x : obj, Prepared x → Confident x
axiom R5 : ∀ x : obj, Confident x → HeadUpHigh x

theorem sunny_team_headsup : Sunny → HeadUpHigh Team := by
  intro hs
  have hk : KidsAtPark := R2 hs
  have hprac : PracticesDaily Team := by
    cases R3 with
    | inl h => exact h
    | inr hnot =>
        have : False := hnot hk
        exact False.elim this
  have hprep : Prepared Team := R1 Team hprac
  have hconf : Confident Team := R4 Team hprep
  exact R5 Team hconf
</lean>

ANSWER: Yes
