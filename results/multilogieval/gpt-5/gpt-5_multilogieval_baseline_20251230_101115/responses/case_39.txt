axiom R2 : ∀ x : obj, GoodAtMath x → DoWell x
-- If they do well on the quiz, then they studied hard.
axiom R3 : ∀ x : obj, DoWell x → StudiedHard x
-- Either it was Monday, or they did not study hard, or it was Monday and they did not study hard.
axiom R4 : Monday ∨ ¬StudiedHard Alex ∨ (Monday ∧ ¬StudiedHard Alex)

-- From the given information, we can derive: if there was no quiz, then Alex is not good at math.
theorem no_quiz_not_good : ¬Quiz → ¬GoodAtMath Alex := by

      exact hNoQ hQ

  have hDW : DoWell Alex := R2 Alex hGood
