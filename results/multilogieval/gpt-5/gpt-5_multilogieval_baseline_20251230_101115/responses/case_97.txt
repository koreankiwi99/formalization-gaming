<lean>
axiom Sub : Prop
axiom Pizza : Prop
axiom Hyper : Prop
axiom GoofOff : Prop
axiom LearnMuch : Prop
axiom LotsHomework : Prop

axiom R1 : Sub → GoofOff
axiom R2 : Pizza → Hyper
axiom R3 : Sub ∨ Pizza
axiom R4 : GoofOff → ¬ LearnMuch
axiom R5 : ¬ LearnMuch → LotsHomework

theorem homework_if_not_hyper : ¬ Hyper → LotsHomework := by
  intro nh
  have not_pizza : ¬ Pizza := fun hp => nh (R2 hp)
  have hs : Sub := Or.elim R3 (fun hs => hs) (fun hp => False.elim (not_pizza hp))
  have goof : GoofOff := R1 hs
  have nlearn : ¬ LearnMuch := R4 goof
  exact R5 nlearn
</lean>

ANSWER: Yes
