<lean>
axiom obj : Type
axiom Alex : obj

axiom AdvertisesCleverly : obj → Prop
axiom OffersDiscounts : obj → Prop
axiom AttractsMoreCustomers : obj → Prop
axiom IncreasesSales : obj → Prop
axiom ExpandsBusiness : obj → Prop

-- If someone advertises cleverly, they attract more customers.
axiom R1 : ∀ x : obj, AdvertisesCleverly x → AttractsMoreCustomers x
-- If someone offers discounts, they increase sales.
axiom R2 : ∀ x : obj, OffersDiscounts x → IncreasesSales x
-- Either they advertise cleverly or they offer discounts (for everyone).
axiom R3 : ∀ x : obj, AdvertisesCleverly x ∨ OffersDiscounts x
-- If someone increases sales, they expand their business.
axiom R4 : ∀ x : obj, IncreasesSales x → ExpandsBusiness x

-- Given: Alex did not attract more customers.
axiom HAlexNotAttract : ¬ AttractsMoreCustomers Alex

theorem alex_expands_business : ExpandsBusiness Alex :=
by
  -- From ¬(AttractsMoreCustomers Alex) and (AdvertisesCleverly Alex → AttractsMoreCustomers Alex),
  -- we get ¬(AdvertisesCleverly Alex).
  have hNotAC : ¬ AdvertisesCleverly Alex := fun hAC => HAlexNotAttract (R1 Alex hAC)
  -- From (AdvertisesCleverly Alex ∨ OffersDiscounts Alex) and ¬(AdvertisesCleverly Alex), deduce OffersDiscounts Alex.
  have hOD : OffersDiscounts Alex :=
    match R3 Alex with
    | Or.inl hAC => False.elim (hNotAC hAC)
    | Or.inr hOD => hOD
  -- Chain implications: OffersDiscounts → IncreasesSales → ExpandsBusiness.
  exact R4 Alex (R2 Alex hOD)
</lean>

ANSWER: Yes
