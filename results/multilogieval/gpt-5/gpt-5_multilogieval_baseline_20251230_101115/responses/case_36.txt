<lean>
axiom obj : Type
axiom Mary : obj

axiom Problems : obj → Prop
axiom Restart : obj → Prop
axiom Updated : obj → Prop

-- If someone is having computer problems, they should try restarting it.
axiom R1 : ∀ x : obj, Problems x → Restart x
-- If someone is having computer problems, they should install the latest updates.
axiom R2 : ∀ x : obj, Problems x → Updated x

-- Either Mary did not restart her computer or she did not install the latest updates.
axiom H : ¬Restart Mary ∨ ¬Updated Mary

theorem mary_not_having_problems : ¬Problems Mary :=
by
  intro hp
  apply Or.elim H
  · intro nres
    exact nres (R1 Mary hp)
  · intro nupd
    exact nupd (R2 Mary hp)
</lean>

ANSWER: No
