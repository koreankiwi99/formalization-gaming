axiom Person : Type
axiom Exam : Type
axiom Alex : Person
axiom Biology : Exam

axiom StudiedHard : Person → Exam → Prop
axiom Confident : Person → Exam → Prop
axiom SevereForecast : Prop
axiom ClassCanceled : Prop
axiom DidWell : Person → Exam → Prop
axiom Satisfied : Person → Exam → Prop

-- If someone studies hard for an exam, they will feel confident going into the test.
axiom R1 : ∀ x : Person, ∀ e : Exam, StudiedHard x e → Confident x e
-- If there is severe weather in the forecast, schools may cancel class that day.
axiom R2 : SevereForecast → ClassCanceled
-- Either Alex studied diligently for his biology exam, or class was not canceled due to the weather.
axiom R3 : StudiedHard Alex Biology ∨ ¬ClassCanceled
-- If a student feels confident about the material, they will likely do well on the exam.
axiom R4 : ∀ x : Person, ∀ e : Exam, Confident x e → DidWell x e
-- If a student performs well on an exam, they will be satisfied with their hard work.
axiom R5 : ∀ x : Person, ∀ e : Exam, DidWell x e → Satisfied x e

-- From severe weather forecast and the disjunction, Alex must have studied for biology.
theorem studied_bio_if_severe : SevereForecast → StudiedHard Alex Biology := by
  intro hsev
  have hc : ClassCanceled := R2 hsev
  have hdisj := R3
  cases hdisj with
  | inl hstudy =>
      exact hstudy
  | inr hnot =>
      have : False := hnot hc
      exact False.elim this

-- Hence, if severe weather was forecast, Alex is satisfied with his biology exam performance.
theorem satisfied_bio_if_severe : SevereForecast → Satisfied Alex Biology := by
  intro hsev
  have hb : StudiedHard Alex Biology := studied_bio_if_severe hsev
  have hc : Confident Alex Biology := R1 Alex Biology hb
  have hw : DidWell Alex Biology := R4 Alex Biology hc
  exact R5 Alex Biology hw
