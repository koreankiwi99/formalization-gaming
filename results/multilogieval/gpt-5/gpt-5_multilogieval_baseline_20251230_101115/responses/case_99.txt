<lean>
axiom W : Prop        -- garden is watered regularly
axiom B : Prop        -- flowers bloom brightly
axiom R : Prop        -- heavy rain
axiom S : Prop        -- soil saturated
axiom F : Prop        -- fragrance released
axiom A : Prop        -- attracts butterflies

axiom R1 : W → B
axiom R2 : R → S
axiom R3 : W ∨ ¬ S
axiom R4 : B → F
axiom R5 : F → A

theorem heavy_rain_attracts : R → A :=
by
  intro hR
  have hS : S := R2 hR
  have hnnS : ¬¬S := fun hns => hns hS
  have hW : W :=
    Or.elim R3
      (fun hW => hW)
      (fun hns => False.elim (hnnS hns))
  have hB : B := R1 hW
  have hF : F := R4 hB
  exact R5 hF
</lean>

ANSWER: Yes
