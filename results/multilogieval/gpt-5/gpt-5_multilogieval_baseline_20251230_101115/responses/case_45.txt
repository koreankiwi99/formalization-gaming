<lean>
axiom obj : Type
axiom Matt : obj

axiom Watch : obj → Prop
axiom ReadBlog : obj → Prop
axiom Skills : obj → Prop
axiom RedoLR : obj → Prop
axiom Ideas : obj → Prop

axiom A1 : ∀ x : obj, Watch x → Skills x
axiom A2 : ∀ x : obj, Skills x → RedoLR x
axiom A3 : ∀ x : obj, Watch x ∨ ReadBlog x ∨ (Watch x ∧ ReadBlog x)
axiom A4 : ∀ x : obj, ReadBlog x → Ideas x

theorem matt_gets_ideas : ¬RedoLR Matt → Ideas Matt := by
  intro hNotRedo
  have hNotSkills : ¬Skills Matt := by
    intro hskills
    have hredo : RedoLR Matt := A2 Matt hskills
    exact hNotRedo hredo
  have hNotWatch : ¬Watch Matt := by
    intro hwatch
    exact hNotSkills (A1 Matt hwatch)
  have hdisj := A3 Matt
  have hRead : ReadBlog Matt := by
    cases hdisj with
    | inl hw =>
        exact False.elim (hNotWatch hw)
    | inr h2 =>
        cases h2 with
        | inl hr => exact hr
        | inr hboth => exact hboth.right
  exact A4 Matt hRead
</lean>

ANSWER: Yes
