<lean>
axiom obj : Type
axiom Emma : obj
axiom StudyHard : obj → Prop
axiom EnoughSleep : obj → Prop
axiom PerformWell : obj → Prop
axiom Ace : obj → Prop

-- Belief: If someone studies hard, then if they get enough sleep before an exam,
-- they will perform well and ace the test.
axiom belief : ∀ x : obj, StudyHard x → (EnoughSleep x → (PerformWell x ∧ Ace x))

-- Facts about Emma
axiom sleep_Emma : EnoughSleep Emma
axiom not_ace_Emma : ¬ Ace Emma

-- From the belief, Emma's enough sleep, and the fact she did not ace,
-- we conclude Emma did not study hard.
theorem emma_not_studyhard : ¬ StudyHard Emma := by
  intro h
  have h_pair : PerformWell Emma ∧ Ace Emma := belief Emma h sleep_Emma
  have h_ace : Ace Emma := And.right h_pair
  exact not_ace_Emma h_ace
</lean>

ANSWER: No
