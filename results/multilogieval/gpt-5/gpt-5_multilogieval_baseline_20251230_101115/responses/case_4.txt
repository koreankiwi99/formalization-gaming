<lean>
axiom Marketing : Prop
axiom MoreCustomers : Prop
axiom SalesIncreased : Prop
axiom ProductionCostsIncreased : Prop

axiom invest_implies_customers : Marketing → MoreCustomers
axiom customers_implies_sales : MoreCustomers → SalesIncreased
axiom marketing_or_costs : Marketing ∨ ProductionCostsIncreased

axiom sales_not_increased : ¬ SalesIncreased

theorem costs_increased_last_quarter : ProductionCostsIncreased := by
  have m_implies_s : Marketing → SalesIncreased := fun m => customers_implies_sales (invest_implies_customers m)
  have not_m : ¬ Marketing := fun m => sales_not_increased (m_implies_s m)
  cases marketing_or_costs with
  | inl m => exact False.elim (not_m m)
  | inr p => exact p
</lean>

ANSWER: Yes
