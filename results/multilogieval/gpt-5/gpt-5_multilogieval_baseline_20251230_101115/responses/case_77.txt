axiom obj : Type
axiom James : obj

axiom PracticedHard : obj → Prop
axiom PlayWell : obj → Prop
axiom PickedStart : obj → Prop
axiom Talented : obj → Prop
axiom Natural : obj → Prop
axiom TopYouth : obj → Prop

-- If someone practiced hard, then they play well
axiom R1 : ∀ x : obj, PracticedHard x → PlayWell x
-- If someone is picked to start, then they are talented
axiom R2 : ∀ x : obj, PickedStart x → Talented x
-- Either they practiced hard, or they are not talented
axiom R3 : ∀ x : obj, PracticedHard x ∨ ¬Talented x
-- If someone has natural ability, then they get picked to start
axiom R4 : ∀ x : obj, Natural x → PickedStart x
-- If someone was a top player in youth leagues, then they have natural talent
axiom R5 : ∀ x : obj, TopYouth x → Natural x

-- From "did not play well" we can derive "not a top youth player"
theorem james_not_play_imp_not_top : ¬PlayWell James → ¬TopYouth James := by
  intro h_not_play
  -- First, derive ¬Talented James
  have h_not_talented : ¬Talented James := by
    -- From ¬PlayWell and (PracticedHard → PlayWell), get ¬PracticedHard
    have h_not_practiced : ¬PracticedHard James := by
      intro h_practiced
      exact h_not_play (R1 James h_practiced)
    -- Use the disjunction PracticedHard ∨ ¬Talented
    have disj := R3 James
    cases disj with
    | inl h_practiced =>
        intro h_talented
        exact False.elim (h_not_practiced h_practiced)
    | inr h_not_talented =>
        exact h_not_talented
  -- From ¬Talented and (PickedStart → Talented), get ¬PickedStart
  have h_not_picked : ¬PickedStart James := by
    intro h_picked
    exact h_not_talented (R2 James h_picked)
  -- From ¬PickedStart and (Natural → PickedStart), get ¬Natural
  have h_not_natural : ¬Natural James := by
    intro h_natural
    exact h_not_picked (R4 James h_natural)
  -- From ¬Natural and (TopYouth → Natural), get ¬TopYouth
  intro h_top
  exact h_not_natural (R5 James h_top)
