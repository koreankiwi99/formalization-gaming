<lean>
axiom obj : Type
axiom Max : obj

axiom PackageOnTime : Prop
axiom SpecialOrderReady : Prop

axiom CanPickUpMaterials : obj → Prop
axiom AbleToStartProject : obj → Prop
axiom StartsProject : obj → Prop
axiom LearnsNewSkills : obj → Prop

-- If the package arrives on time, people will be able to start a new project.
axiom R1 : ∀ x : obj, PackageOnTime → AbleToStartProject x
-- If the special order is in stock, they can pick up the materials they need.
axiom R2 : ∀ x : obj, SpecialOrderReady → CanPickUpMaterials x
-- Either the package will arrive on time or the special order will be ready.
axiom R3 : PackageOnTime ∨ SpecialOrderReady
-- If someone starts his new project, they will learn some new skills.
axiom R4 : ∀ x : obj, StartsProject x → LearnsNewSkills x

-- Max was unable to pick up the materials he needed.
axiom T1 : ¬ CanPickUpMaterials Max

-- From T1 and R2 we can deduce the special order is not ready.
theorem not_special_order_ready : ¬ SpecialOrderReady := fun h => T1 (R2 Max h)

-- Using the disjunction and ¬SpecialOrderReady, we can deduce the package is on time.
theorem package_on_time : PackageOnTime := by
  cases R3 with
  | inl pot => exact pot
  | inr sr => exact False.elim (not_special_order_ready sr)

-- Therefore Max is able to start a new project.
theorem max_able_to_start : AbleToStartProject Max := R1 Max package_on_time

-- However, we do not have any axiom connecting "able to start" to actually "starts",
-- so we cannot derive LearnsNewSkills Max or its negation from the given axioms.
-- In particular, we cannot prove StartsProject Max.
</lean>
