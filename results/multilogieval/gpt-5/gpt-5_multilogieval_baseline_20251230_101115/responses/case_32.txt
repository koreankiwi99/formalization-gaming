axiom Energetic : Person → Prop
axiom ValuesWellness : Person → Prop

-- If someone exercises regularly, they are fit.
axiom H_ex_fit : ∀ x : Person, Exercises x → Fit x
-- If someone eats nutritious foods, they have energy.
axiom H_eat_energy : ∀ x : Person, EatsNutritious x → Energetic x
-- Not everyone in town was active or energetic (interpret "active" as exercises regularly):
axiom H_not_all_active_or_energetic : ∃ x : Person, ¬(Exercises x ∨ Energetic x)
-- If someone values wellness, they exercise regularly.
axiom H_value_exercise : ∀ x : Person, ValuesWellness x → Exercises x

-- Jane eats very nutritious foods.
axiom H_jane_eats : EatsNutritious Jane

-- A consequence the mayor concluded: some citizen does not exercise or does not eat well.
theorem mayor_conclusion : ∃ x : Person, (¬Exercises x ∨ ¬EatsNutritious x) := by
  cases H_not_all_active_or_energetic with
  | intro x hx =>
    have hnx : ¬Exercises x := by
      intro hxex
      exact hx (Or.inl hxex)
    exact Exists.intro x (Or.inl hnx)

-- From Jane eating nutritious foods, we can conclude she is energetic.
theorem jane_energetic : Energetic Jane := H_eat_energy Jane H_jane_eats
