axiom PlaySimple : obj → Prop

-- If someone learned basic chords and strumming patterns, they can play simple songs
axiom R1 : ∀ x : obj, LearnedChords x ∧ LearnedStrumming x → PlaySimple x
-- If someone can play simple songs, they can entertain friends with the guitar

-- If someone takes guitar lessons, they gain knowledge of music theory
axiom R3 : ∀ x : obj, TookLessons x → KnowledgeTheory x

axiom D1 : LearnedChords Emma ∨ TookLessons Emma
-- Emma is unable to entertain friends with the guitar
axiom T_ent_not : ¬ Entertain Emma

  intro h
  have ps : PlaySimple Emma := R1 Emma h
  have ent : Entertain Emma := R2 Emma ps
  exact T_ent_not ent
