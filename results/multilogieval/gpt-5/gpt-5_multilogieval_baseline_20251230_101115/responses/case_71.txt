<lean>
axiom Person : Type
axiom Alex : Person
axiom Applied : Person → Prop
axiom Burnt : Person → Prop
axiom Swim : Person → Prop
axiom Exercise : Person → Prop
axiom Warm : Prop
axiom Sunny : Prop

axiom R1 : ∀ x : Person, Applied x → ¬ Burnt x
axiom R2 : ∀ x : Person, Swim x → Exercise x
axiom R3 : ∀ x : Person, Applied x ∨ ¬ Exercise x
axiom R4 : Warm → ∀ x : Person, Swim x
axiom R5 : Sunny → Warm

theorem burnt_implies_not_sunny : Burnt Alex → ¬ Sunny := by
  intro hb
  intro hsun
  have hw : Warm := R5 hsun
  have hswim_all : ∀ x : Person, Swim x := R4 hw
  have hex : Exercise Alex := R2 Alex (hswim_all Alex)
  have hdisj : Applied Alex ∨ ¬ Exercise Alex := R3 Alex
  have hnot_app : ¬ Applied Alex := by
    intro happ
    have hn : ¬ Burnt Alex := R1 Alex happ
    exact hn hb
  have : False := by
    cases hdisj with
    | inl happ => exact hnot_app happ
    | inr hnot_ex => exact hnot_ex hex
  exact False.elim this
</lean>
