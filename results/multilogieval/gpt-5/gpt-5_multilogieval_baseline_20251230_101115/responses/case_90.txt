<lean>
axiom HomeWinning : Prop
axiom RivalryGame : Prop
axiom CrowdLoudEnergized : Prop
axiom FansInvested : Prop
axiom AtmosphereElectric : Prop
axiom UnforgettableGame : Prop

axiom R1 : HomeWinning → CrowdLoudEnergized
axiom R2 : RivalryGame → FansInvested
axiom R3 : HomeWinning ∨ RivalryGame
axiom R4 : CrowdLoudEnergized → AtmosphereElectric
axiom R5 : AtmosphereElectric → UnforgettableGame

theorem atmosphere_electric_if_not_invested : ¬FansInvested → AtmosphereElectric := by
  intro hni
  have notR : ¬RivalryGame := by
    intro r
    exact hni (R2 r)
  have hw : HomeWinning := by
    cases R3 with
    | inl hw => exact hw
    | inr r =>
      have : False := notR r
      exact False.elim this
  have loud : CrowdLoudEnergized := R1 hw
  exact R4 loud
</lean>

ANSWER: Yes
