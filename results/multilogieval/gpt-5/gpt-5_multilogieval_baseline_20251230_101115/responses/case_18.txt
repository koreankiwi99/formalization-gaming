<lean>
axiom person : Type
axiom Sam : person

axiom Impressed : person → Prop
axiom TookExtra : person → Prop
axiom Experience : person → Prop
axiom Promotion : person → Prop
axiom LeadNewTeam : person → Prop

-- Rules from the context
axiom R_imp_to_prom : ∀ x : person, Impressed x → Promotion x
axiom R_extra_to_exp : ∀ x : person, TookExtra x → Experience x
axiom R_exp_to_lead : ∀ x : person, Experience x → LeadNewTeam x

-- Sam-specific facts
axiom D_Sam : Impressed Sam ∨ TookExtra Sam
axiom H_not_prom : ¬ Promotion Sam

theorem sam_assigned : LeadNewTeam Sam := by
  -- From ¬Promotion Sam and (Impressed → Promotion), derive ¬Impressed Sam
  have h_not_imp : ¬ Impressed Sam := by
    intro himp
    exact H_not_prom (R_imp_to_prom Sam himp)
  -- From (Impressed Sam ∨ TookExtra Sam) and ¬Impressed Sam, get TookExtra Sam
  have h_extra : TookExtra Sam := by
    cases D_Sam with
    | inl himp =>
      exact False.elim (H_not_prom (R_imp_to_prom Sam himp))
    | inr hextra =>
      exact hextra
  -- Then Experience Sam and hence LeadNewTeam Sam
  have h_exp : Experience Sam := R_extra_to_exp Sam h_extra
  exact R_exp_to_lead Sam h_exp
</lean>

ANSWER: Yes
