<lean>
axiom NiceDisplay : Prop
axiom JudgesImpressed : Prop
axiom ThoroughResearch : Prop
axiom ExpertsTakeSeriously : Prop
axiom ReceiveAward : Prop

axiom R1 : NiceDisplay → JudgesImpressed
axiom R2 : ThoroughResearch → ExpertsTakeSeriously
axiom R3 : NiceDisplay ∨ ThoroughResearch
axiom R4 : ExpertsTakeSeriously → ReceiveAward
axiom T_not_impressed : ¬ JudgesImpressed

theorem project_awarded : ReceiveAward :=
by
  have h_not_nice : ¬ NiceDisplay := fun h => T_not_impressed (R1 h)
  have h_thorough : ThoroughResearch :=
    Or.elim R3
      (fun hnice => False.elim (h_not_nice hnice))
      (fun hth => hth)
  exact R4 (R2 h_thorough)
</lean>

ANSWER: Yes
