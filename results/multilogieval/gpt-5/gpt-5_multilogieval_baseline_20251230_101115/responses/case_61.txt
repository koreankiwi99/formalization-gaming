<lean>
axiom obj : Type
axiom Priya : obj

axiom Attend : obj → Prop
axiom Study : obj → Prop
axiom Excel : obj → Prop

-- Context: If a student attends regularly, then if they study diligently, they will excel.
axiom R1 : ∀ x : obj, Attend x → (Study x → Excel x)

-- Given facts about Priya
axiom Priya_study : Study Priya
axiom Priya_not_excel : ¬ Excel Priya

-- Conclusion: Priya did not attend lectures regularly.
theorem Priya_not_attend : ¬ Attend Priya := by
  intro hAttend
  have hImpl : Study Priya → Excel Priya := R1 Priya hAttend
  have hExcel : Excel Priya := hImpl Priya_study
  exact Priya_not_excel hExcel
</lean>

ANSWER: No
