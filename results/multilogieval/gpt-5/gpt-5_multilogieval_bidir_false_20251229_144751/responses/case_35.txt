<lean>
axiom Person : Type
axiom Lucy : Person

axiom Hacked : Person → Prop
axiom ResetPassword : Person → Prop
axiom NotifyBank : Person → Prop

-- If a person's account is hacked, they need to reset their password
axiom R1 : ∀ x : Person, Hacked x → ResetPassword x
-- If a person detects unauthorized access (hacked), they should notify their bank
axiom R2 : ∀ x : Person, Hacked x → NotifyBank x

-- Either Lucy did not change her password or she did not inform her bank
axiom E1 : ¬ResetPassword Lucy ∨ ¬NotifyBank Lucy

theorem lucy_not_hacked : ¬Hacked Lucy := by
  intro h
  cases E1 with
  | inl notReset =>
      exact notReset (R1 Lucy h)
  | inr notNotify =>
      exact notNotify (R2 Lucy h)
</lean>

ANSWER: No
