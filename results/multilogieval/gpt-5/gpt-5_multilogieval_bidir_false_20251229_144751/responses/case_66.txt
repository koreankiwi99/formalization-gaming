<lean>
axiom obj : Type
axiom Acme : obj
axiom Innovative : obj → Prop
axiom Marketed : obj → Prop
axiom Successful : obj → Prop

-- If a company develops an innovative product, then if they market it effectively, the product will be commercially successful.
axiom R1 : ∀ x : obj, Innovative x → (Marketed x → Successful x)

-- Facts about Acme Inc.
axiom T_market : Marketed Acme
axiom T_not_success : ¬Successful Acme

theorem acme_not_innovative : ¬Innovative Acme :=
by
  intro hInnov
  have hImp : Marketed Acme → Successful Acme := R1 Acme hInnov
  have hSucc : Successful Acme := hImp T_market
  exact T_not_success hSucc
</lean>

ANSWER: No
