<lean>
axiom HighQuality : Prop
axiom PreciseJoints : Prop
axiom Sturdy : Prop
axiom LongLasting : Prop

axiom CarpentryRule : HighQuality → (PreciseJoints → (Sturdy ∧ LongLasting))
axiom JointsPrecise : PreciseJoints
axiom NotSturdy : ¬Sturdy
axiom NotLongLasting : ¬LongLasting

theorem wood_not_high_quality : ¬HighQuality :=
by
  intro hq
  have h := CarpentryRule hq JointsPrecise
  have hs : Sturdy := And.left h
  exact NotSturdy hs
</lean>
