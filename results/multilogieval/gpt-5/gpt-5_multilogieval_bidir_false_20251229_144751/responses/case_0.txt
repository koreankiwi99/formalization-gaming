<lean>
axiom obj : Type
axiom Person : obj

axiom SaveAgg : obj → Prop
axiom ImproveCredit : obj → Prop
axiom AffordDownPayment : obj → Prop
axiom QualifyMortgage : obj → Prop
axiom BuyDreamHome : obj → Prop

-- If someone saves aggressively, they can afford a down payment.
axiom S_to_A : ∀ x : obj, SaveAgg x → AffordDownPayment x
-- If someone improves their credit score, they can qualify for a mortgage.
axiom I_to_Q : ∀ x : obj, ImproveCredit x → QualifyMortgage x
-- The person either saves aggressively or improves their credit.
axiom Disj : SaveAgg Person ∨ ImproveCredit Person
-- If someone qualifies for a mortgage, they can buy their dream home.
axiom Q_to_B : ∀ x : obj, QualifyMortgage x → BuyDreamHome x

-- Given: they could not afford a down payment.
axiom NotAfford : ¬AffordDownPayment Person

-- From the premises and NotAfford, we can actually prove they buy their dream home.
theorem buy_dream_home : BuyDreamHome Person := by
  have notSave : ¬SaveAgg Person := by
    intro hs
    have ha : AffordDownPayment Person := S_to_A Person hs
    exact NotAfford ha
  have hi : ImproveCredit Person := by
    cases Disj with
    | inl hs => exact False.elim (notSave hs)
    | inr hi => exact hi
  have hq : QualifyMortgage Person := I_to_Q Person hi
  exact Q_to_B Person hq
