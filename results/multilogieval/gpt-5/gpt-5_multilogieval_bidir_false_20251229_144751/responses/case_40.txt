axiom Motivated : obj → Prop

axiom GymClosed : Prop

-- If someone starts a new workout routine, they will be motivated to keep it up.
axiom A1 : ∀ x : obj, StartNew x → Motivated x
-- If the gym closes for renovations, then they can't follow their routine.
axiom A2 : ∀ x : obj, GymClosed → ¬FollowRoutine x
-- Either they started a new workout, or the gym was closed (inclusive or).
axiom A3 : StartNew James ∨ GymClosed
-- If they couldn't follow their routine, they would feel frustrated.
axiom A4 : ∀ x : obj, ¬FollowRoutine x → Frustrated x
-- If they felt frustrated, they might indulge in some unhealthy snacks (modeled here as implication).
axiom A5 : ∀ x : obj, Frustrated x → Indulged x

-- Given: James was not motivated to workout.
axiom NM : ¬Motivated James

-- From the premises, we can actually derive that James indulged.
theorem indulged_james : Indulged James := by

    intro hStart
    have hMot := A1 James hStart
    exact NM hMot
  have hClosed : GymClosed := by
    cases A3 with
    | inl hStart => exact False.elim (hNotStart hStart)
    | inr hC => exact hC
  have hNotFollow : ¬FollowRoutine James := A2 James hClosed
  have hFrustrated : Frustrated James := A4 James hNotFollow
  exact A5 James hFrustrated
