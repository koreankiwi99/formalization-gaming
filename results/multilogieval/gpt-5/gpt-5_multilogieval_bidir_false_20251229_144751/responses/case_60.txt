<lean>
axiom obj : Type
axiom John : obj

-- Propositions/events
axiom Alarm : Prop
axiom TimeToGetReady : Prop

-- Person-specific predicates
axiom Late : obj → Prop
axiom MissedBus : obj → Prop
axiom GotUpLate : obj → Prop
axiom WentToBedLate : obj → Prop

-- Axioms from the context
-- If the alarm went off, then it was time to get ready.
axiom A1 : Alarm → TimeToGetReady
-- If they missed the bus, then they'd be late for school.
axiom A2 : ∀ x : obj, MissedBus x → Late x
-- Either the alarm went off, or they weren't late for school (for John).
axiom A3 : Alarm ∨ ¬Late John
-- If they got up late, then they miss the bus.
axiom A4 : ∀ x : obj, GotUpLate x → MissedBus x
-- If they went to bed late, then they get up late.
axiom A5 : ∀ x : obj, WentToBedLate x → GotUpLate x

-- John's knowledge: it is not the time to get ready (so, ¬TimeToGetReady).
axiom NotTime : ¬TimeToGetReady

theorem john_did_not_go_to_bed_late : ¬WentToBedLate John := by
  -- From ¬Time and (Alarm → Time), we get ¬Alarm.
  have hNotAlarm : ¬Alarm := by
    intro hA
    exact NotTime (A1 hA)
  -- From Alarm ∨ ¬Late John and ¬Alarm, deduce ¬Late John.
  have hNotLate : ¬Late John :=
    Or.elim A3
      (fun hA => False.elim (hNotAlarm hA))
      (fun hNL => hNL)
  -- From MissedBus → Late, get ¬Late → ¬MissedBus (for John).
  have hNotMiss : ¬MissedBus John := by
    intro hM
    exact hNotLate (A2 John hM)
  -- From GotUpLate → MissedBus, get ¬MissedBus → ¬GotUpLate (for John).
  have hNotUpLate : ¬GotUpLate John := by
    intro hU
    exact hNotMiss (A4 John hU)
  -- From WentToBedLate → GotUpLate, get ¬GotUpLate → ¬WentToBedLate (for John).
  intro hBed
  exact hNotUpLate (A5 John hBed)
</lean>

ANSWER: No
