axiom obj : Type
axiom Linda : obj

axiom Cold : obj → Prop
axiom Chill : obj → Prop
axiom AddLayers : obj → Prop
axiom AdjustThermostat : obj → Prop

-- If someone is cold, they put on more layers.
axiom R1 : ∀ x : obj, Cold x → AddLayers x
-- If someone feels a chill, they turn up the heat.
axiom R2 : ∀ x : obj, Chill x → AdjustThermostat x

-- Given: Linda is either not adding layers or not adjusting the thermostat.
axiom S1 : ¬AddLayers Linda ∨ ¬AdjustThermostat Linda

-- From not adding layers, we can derive Linda is not cold.
theorem not_cold_if_not_layers : ¬AddLayers Linda → ¬Cold Linda :=
  fun hnot hCold => hnot (R1 Linda hCold)

-- From not adjusting the thermostat, we can derive Linda is not feeling a chill.
theorem not_chill_if_not_adjust : ¬AdjustThermostat Linda → ¬Chill Linda :=
  fun hnot hChill => hnot (R2 Linda hChill)

-- From the given disjunction, we can only conclude a disjunction:
-- either Linda is not cold or she is not feeling a chill.
theorem not_cold_or_not_chill : ¬Cold Linda ∨ ¬Chill Linda :=
  Or.elim S1
    (fun h1 => Or.inl (not_cold_if_not_layers h1))
    (fun h2 => Or.inr (not_chill_if_not_adjust h2))
