axiom obj : Type
axiom Priya : obj

axiom Suspects : obj → Prop
axiom Talk : obj → Prop
axiom Reassess : obj → Prop

-- From the context:
-- If someone suspects their partner is cheating, they should talk.
axiom R1 : ∀ x : obj, Suspects x → Talk x
-- Infidelity concerns warrant re-evaluating the relationship.
axiom R2 : ∀ x : obj, Suspects x → Reassess x

-- Given statement to evaluate as a premise:
axiom H : ¬Talk Priya ∨ ¬Reassess Priya

-- Prove Priya did not suspect her partner was being unfaithful.
theorem priya_not_suspects : ¬Suspects Priya := by
  intro hS
  cases H with
  | inl hntalk =>
      exact hntalk (R1 Priya hS)
  | inr hnreassess =>
      exact hnreassess (R2 Priya hS)
</lean>
