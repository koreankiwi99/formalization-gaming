<lean>
axiom obj : Type
axiom Store : obj

-- Predicates
axiom Service : obj → Prop          -- offers great customer service
axiom Reputation : obj → Prop       -- reputation grows
axiom Events : obj → Prop           -- hosts author events
axiom MoreCustomers : obj → Prop    -- more customers will visit (traffic increases)
axiom RevenueImproves : obj → Prop  -- revenue will improve

-- Premises
axiom P1 : Service Store → Reputation Store
axiom P2 : Events Store → MoreCustomers Store
axiom P3 : Service Store ∨ Events Store
axiom P4 : MoreCustomers Store → RevenueImproves Store

-- Given: service quality decreased (i.e., not excellent service)
axiom DecreasedService : ¬Service Store

-- From the given premises, we can actually prove that revenue improves.
theorem revenue_improves : RevenueImproves Store := by
  -- From Service ∨ Events and ¬Service, deduce Events
  have hEvents : Events Store := by
    cases P3 with
    | inl hService => exact False.elim (DecreasedService hService)
    | inr hEvents  => exact hEvents
  -- Then MoreCustomers and hence RevenueImproves
  have hMore : MoreCustomers Store := P2 hEvents
  exact P4 hMore
