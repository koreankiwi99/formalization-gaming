axiom Person : Type
axiom Mary : Person

axiom WeatherClear : Prop
axiom GameOn : Prop
axiom FansCheer : Prop
axiom ExcitingNightAtBallpark : Prop

axiom FinishesErrands : Person → Prop
axiom MeetsFriendsForDinner : Person → Prop

-- If the weather clears up, the baseball game will go on.
axiom R1 : WeatherClear → GameOn
-- If someone finishes their errands, they'll meet friends for dinner.
axiom R2 : ∀ p : Person, FinishesErrands p → MeetsFriendsForDinner p
-- Either it will clear up, or the errands (Mary's) will get done (or both).
axiom R3 : WeatherClear ∨ FinishesErrands Mary
-- If the game is on, fans will get to cheer for their team.
axiom R4 : GameOn → FansCheer
-- If fans get to cheer, it will be an exciting night at the ballpark.
axiom R5 : FansCheer → ExcitingNightAtBallpark

-- In fact, from ¬MeetsFriendsForDinner Mary we can prove ExcitingNightAtBallpark.
-- So we cannot prove the negation ¬((¬MeetsFriendsForDinner Mary) → ExcitingNightAtBallpark).
theorem exciting_from_not_meet :
  (¬MeetsFriendsForDinner Mary) → ExcitingNightAtBallpark := by
  intro hNotMeet
  have hNotFinish : ¬FinishesErrands Mary := by
    intro hFin
    exact hNotMeet (R2 Mary hFin)
  have hClear : WeatherClear := by
    cases R3 with
    | inl hW => exact hW
    | inr hFin =>
      exact False.elim (hNotFinish hFin)
  have hOn : GameOn := R1 hClear
  have hCheer : FansCheer := R4 hOn
  exact R5 hCheer
