<lean>
axiom obj : Type
axiom John : obj

axiom FinishesEarly : obj → Prop
axiom LongLunch : obj → Prop
axiom Recharged : obj → Prop
axiom ProductiveAfternoon : obj → Prop
axiom BringsTreats : obj → Prop
axiom Birthday : Prop

-- If someone finishes early, they can take a long lunch.
axiom A1 : ∀ x : obj, FinishesEarly x → LongLunch x
-- If someone takes a long lunch, they come back recharged.
axiom A2 : ∀ x : obj, LongLunch x → Recharged x
-- If someone is recharged, they have a very productive afternoon.
axiom A3 : ∀ x : obj, Recharged x → ProductiveAfternoon x
-- If it's a coworker's birthday, people will bring in treats.
axiom A4 : ∀ x : obj, Birthday → BringsTreats x
-- Either someone finishes early, or it's a birthday (or both).
axiom A5 : (∃ x : obj, FinishesEarly x) ∨ Birthday

-- From "John did not bring treats", we can infer "not Birthday".
theorem notBrings_impl_notBirthday : ¬BringsTreats John → ¬Birthday := by
  intro hnot hB
  exact hnot (A4 John hB)

-- From "not Birthday" and the disjunction, we can infer someone finished early (but not necessarily John).
theorem existsEarly_of_notBirthday : ¬Birthday → ∃ x : obj, FinishesEarly x := by
  intro hnotB
  cases A5 with
  | inl hex => exact hex
  | inr hB => exact False.elim (hnotB hB)

-- If John finished early, then John is productive.
theorem productive_if_early_John : FinishesEarly John → ProductiveAfternoon John := by
  intro hE
  have hL : LongLunch John := A1 John hE
  have hR : Recharged John := A2 John hL
  exact A3 John hR
