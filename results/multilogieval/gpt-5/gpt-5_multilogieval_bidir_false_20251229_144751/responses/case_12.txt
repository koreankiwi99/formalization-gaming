axiom obj : Type
axiom James : obj

axiom Chef : obj → Prop
axiom FollowPrecisely : obj → Prop
axiom DishesTurnOutWell : obj → Prop
axiom GainConfidence : obj → Prop
axiom WatchCookingShows : obj → Prop

-- James is one of the chefs discussed
axiom ChefJames : Chef James

-- If a chef follows recipes precisely, their dishes turn out well
axiom R1 : ∀ x : obj, Chef x → (FollowPrecisely x → DishesTurnOutWell x)
-- If a chef's dishes turn out well, they gain confidence in the kitchen
axiom R2 : ∀ x : obj, Chef x → (DishesTurnOutWell x → GainConfidence x)
-- Either a chef followed recipes precisely, or they watched cooking shows (or both)
axiom R3 : ∀ x : obj, Chef x → (FollowPrecisely x ∨ WatchCookingShows x)

-- Given statement: James does not gain confidence in the kitchen
axiom HnotConf : ¬GainConfidence James

-- From the premises, we can actually derive that James watches cooking shows
theorem watch_from_not_conf : WatchCookingShows James := by
  have h_disj : FollowPrecisely James ∨ WatchCookingShows James := R3 James ChefJames
  -- Derive ¬FollowPrecisely James from ¬GainConfidence James using the two rules
  have h_not_well : ¬DishesTurnOutWell James := by
    intro hw
    have conf : GainConfidence James := R2 James ChefJames hw
    exact HnotConf conf
  have h_not_follow : ¬FollowPrecisely James := by
    intro hf
    have hw : DishesTurnOutWell James := R1 James ChefJames hf
    exact h_not_well hw
  -- Use the disjunction and ¬FollowPrecisely to conclude WatchCookingShows
  cases h_disj with
  | inl hf =>
      exact False.elim (h_not_follow hf)
  | inr hw =>
      exact hw
