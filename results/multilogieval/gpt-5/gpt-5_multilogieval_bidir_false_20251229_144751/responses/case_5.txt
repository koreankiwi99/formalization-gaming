axiom obj : Type
axiom Mary : obj

axiom LearnedBotany : obj → Prop
axiom UnderstoodNeeds : obj → Prop
axiom Thrived : obj → Prop
axiom SoughtAdvice : obj → Prop

-- If someone learned botany, they understand plant needs
axiom R1 : ∀ x : obj, LearnedBotany x → UnderstoodNeeds x
-- If someone understands plant needs, their plants thrive
axiom R2 : ∀ x : obj, UnderstoodNeeds x → Thrived x
-- Either they learned botany or they sought advice (for Mary)
axiom Disj : LearnedBotany Mary ∨ SoughtAdvice Mary
-- Mary's plants did not thrive
axiom NotThrivedMary : ¬Thrived Mary

theorem learned_implies_thrived : ∀ x : obj, LearnedBotany x → Thrived x := by
  intro x hLB
  exact R2 x (R1 x hLB)

theorem not_learned_Mary : ¬LearnedBotany Mary := by
  intro hLB
  have hT : Thrived Mary := learned_implies_thrived Mary hLB
  exact NotThrivedMary hT

-- From not thriving and the disjunction, we can deduce Mary sought advice
theorem soughtAdvice_Mary : SoughtAdvice Mary := by
  cases Disj with
  | inl hLB =>
    have : False := not_learned_Mary hLB
    exact False.elim this
  | inr hSA =>
    exact hSA
