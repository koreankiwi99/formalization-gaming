<lean>
axiom obj : Type
axiom Casey : obj

axiom Volunteers : obj → Prop
axiom HelpsCommunity : obj → Prop
axiom AdoptsFromShelter : obj → Prop
axiom ProvidesHome : obj → Prop
axiom FeelsFulfillment : obj → Prop

-- If someone volunteers at a local shelter, they will help the community.
axiom R1 : ∀ x : obj, Volunteers x → HelpsCommunity x
-- If someone adopts a pet from the shelter, then they provide a home to a pet in need.
axiom R2 : ∀ x : obj, AdoptsFromShelter x → ProvidesHome x
-- Either someone volunteers at the shelter, or they do not provide a home to a pet in need, 
-- or someone volunteered at the shelter, and they did not provide a home to a pet in need.
axiom R3 : ∀ x : obj, Volunteers x ∨ ¬ProvidesHome x ∨ (Volunteers x ∧ ¬ProvidesHome x)
-- If someone helps the community, then they feel a sense of fulfillment.
axiom R4 : ∀ x : obj, HelpsCommunity x → FeelsFulfillment x

-- From R3, if someone provides a home, then they must volunteer.
theorem providesHome_implies_volunteers (x : obj) : ProvidesHome x → Volunteers x :=
  fun PHx => by
    have h := R3 x
    cases h with
    | inl hv => exact hv
    | inr h2 =>
      cases h2 with
      | inl hnot =>
          have contra : False := hnot PHx
          exact False.elim contra
      | inr hvhnot =>
          exact hvhnot.left

-- Therefore, adoption entails fulfillment for Casey.
theorem adopts_entails_fulfillment_for_Casey : AdoptsFromShelter Casey → FeelsFulfillment Casey :=
  fun hAdopt => by
    have hPH : ProvidesHome Casey := R2 Casey hAdopt
    have hVol : Volunteers Casey := providesHome_implies_volunteers Casey hPH
    have hHelp : HelpsCommunity Casey := R1 Casey hVol
    exact R4 Casey hHelp
