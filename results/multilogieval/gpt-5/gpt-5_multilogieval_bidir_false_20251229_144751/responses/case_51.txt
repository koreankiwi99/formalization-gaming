axiom Person : Type
axiom Mary : Person

axiom Sale : Prop
axiom PricesUp : Prop
axiom Payday : Prop

axiom Stock : Person → Prop
axiom List : Person → Prop
axiom PlanBuy : Person → Prop
axiom PlanAhead : Person → Prop

-- If the supermarket has a sale, then people will stock up on groceries.
axiom R1 : ∀ x : Person, Sale → Stock x
-- If someone makes a shopping list, then they plan what to buy.
axiom R2 : ∀ x : Person, List x → PlanBuy x
-- Either the store has a sale, or people don't plan what to buy, or there's a sale but they don't plan ahead.
axiom R3 : Sale ∨ (∀ x : Person, ¬PlanBuy x) ∨ (Sale ∧ (∀ x : Person, ¬PlanAhead x))
-- If payday was this week, then people make a shopping list.
axiom R4 : Payday → ∀ x : Person, List x
-- Either it was payday this week, or prices went up recently, or both.
axiom R5 : Payday ∨ PricesUp ∨ (Payday ∧ PricesUp)

-- From "Mary did not stock up on groceries" we can derive that prices went up recently.
theorem prices_up_if_mary_not_stock : (¬Stock Mary) → PricesUp := by
  intro hns
  -- ¬Sale
  have hNotSale : ¬Sale := by
    intro hSale
    have hSm : Stock Mary := R1 Mary hSale
    exact hns hSm
  -- From R3 and ¬Sale, deduce ∀x, ¬PlanBuy x
  have hNoPlan : ∀ x : Person, ¬PlanBuy x := by
    have h := R3
    cases h with
    | inl hSale =>
      have hf : False := hNotSale hSale
      intro x
      exact False.elim hf
    | inr hRest =>
      cases hRest with
      | inl hNoPlan' =>
        exact hNoPlan'
      | inr hSaleAnd =>
        have hSale := hSaleAnd.left
        have hf : False := hNotSale hSale
        intro x
        exact False.elim hf
  -- From hNoPlan and R4, deduce ¬Payday
  have hNotPay : ¬Payday := by
    intro hPay
    have hListAll : ∀ x : Person, List x := R4 hPay
    have hPBMary : PlanBuy Mary := R2 Mary (hListAll Mary)
    have hNPBMary : ¬PlanBuy Mary := hNoPlan Mary
    exact hNPBMary hPBMary
  -- From R5 and ¬Payday, deduce PricesUp
  have hPU : PricesUp := by
    have h := R5
    cases h with
    | inl hPay =>
      exact False.elim (hNotPay hPay)
    | inr hRest =>
      cases hRest with
      | inl hP =>
        exact hP
      | inr hBoth =>
        exact hBoth.right
  exact hPU
