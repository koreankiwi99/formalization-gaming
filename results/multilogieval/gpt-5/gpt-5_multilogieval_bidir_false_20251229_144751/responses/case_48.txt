axiom Gasps : Prop

axiom R2 : Gasps → High
axiom R3 : Flip ∨ Clowns
axiom R4 : Clowns → Laugh

theorem flip_implies_high : Flip → High := by
  intro f
  have g : Gasps := R1 f
  exact R2 g

theorem notHigh_implies_notFlip : ¬High → ¬Flip := by
  intro nh
  intro f
  exact nh (flip_implies_high f)

theorem clowns_if_not_flip : ¬Flip → Clowns := by
  intro nf
  cases R3 with
  | inl f =>
      have h : False := nf f
      exact False.elim h
  | inr c => exact c

theorem laugh_if_notHigh : ¬High → Laugh := by
  intro nh
  have nf : ¬Flip := notHigh_implies_notFlip nh
  have c : Clowns := clowns_if_not_flip nf
  exact R4 c
