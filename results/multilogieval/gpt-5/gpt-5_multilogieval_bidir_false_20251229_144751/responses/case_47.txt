<lean>
axiom Person : Type
axiom Max : Person

axiom PackageOnTime : Prop
axiom SpecialOrderReady : Prop

axiom AbleStart : Person → Prop
axiom CanPickUp : Person → Prop
axiom StartsProject : Person → Prop
axiom LearnsSkills : Person → Prop

-- If the package arrives on time, people will be able to start a new project.
axiom A1 : PackageOnTime → ∀ p : Person, AbleStart p
-- If the office supply store has the special order in stock, they can pick up the materials they need.
axiom A2 : SpecialOrderReady → ∀ p : Person, CanPickUp p
-- Either the package will arrive on time or the special order will be ready.
axiom A3 : PackageOnTime ∨ SpecialOrderReady
-- If someone starts his new project, they will learn some new skills.
axiom A4 : ∀ p : Person, StartsProject p → LearnsSkills p

-- Given: Max was unable to pick up the materials he needed.
axiom HMax : ¬ CanPickUp Max

theorem not_ready : ¬ SpecialOrderReady := by
  intro hready
  have hAll : ∀ p : Person, CanPickUp p := A2 hready
  have hMax : CanPickUp Max := hAll Max
  exact HMax hMax

theorem package_on_time : PackageOnTime := by
  have hnr : ¬SpecialOrderReady := not_ready
  cases A3 with
  | inl h => exact h
  | inr h => exact False.elim (hnr h)

theorem able_start_Max : AbleStart Max := by
  have hp : PackageOnTime := package_on_time
  have hAll : ∀ p : Person, AbleStart p := A1 hp
  exact hAll Max

-- Cannot prove ¬LearnsSkills Max from given axioms
