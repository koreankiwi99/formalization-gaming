<lean>
axiom Person : Type
axiom Mary : Person

axiom BelowFreezing : Prop
axiom SnowOnGround : Prop
axiom EnoughSnow : Prop
axiom BuildSnowman : Person → Prop
axiom LovesWinter : Person → Prop

-- If the temperature dropped below freezing, then there would be snow on the ground.
axiom R1 : BelowFreezing → SnowOnGround
-- If someone built a snowman, then there was enough snow.
axiom R2 : ∀ x : Person, BuildSnowman x → EnoughSnow
-- If someone loves winter, then they build snowmen.
axiom R3 : ∀ x : Person, LovesWinter x → BuildSnowman x
-- Either below freezing, or not enough snow, or both.
axiom Cert : BelowFreezing ∨ ¬EnoughSnow ∨ (BelowFreezing ∧ ¬EnoughSnow)

-- From no snow on the ground, Mary does not love winter.
theorem no_snow_implies_mary_not_love : ¬SnowOnGround → ¬LovesWinter Mary := by
  intro hNoSnow
  intro hLove
  have hBuild : BuildSnowman Mary := R3 Mary hLove
  have hEnough : EnoughSnow := R2 Mary hBuild
  have hNotBF : ¬BelowFreezing := by
    intro hBF
    exact hNoSnow (R1 hBF)
  have hNotEnough : ¬EnoughSnow := by
    have d := Cert
    apply Or.elim d
    · intro hBF
      exact False.elim (hNotBF hBF)
    · intro hRest
      apply Or.elim hRest
      · intro hne
        exact hne
      · intro hBoth
        exact hBoth.right
  exact hNotEnough hEnough
</lean>

ANSWER: No
