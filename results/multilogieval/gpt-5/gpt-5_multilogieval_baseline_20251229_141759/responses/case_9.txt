<lean>
axiom Party : Type
axiom SamParty : Party
axiom Birthday : Party → Prop
axiom Cake : Party → Prop
axiom IceCream : Party → Prop
axiom BringDish : Party → Prop
axiom EnoughFood : Party → Prop

-- If someone brought a dish to share at the party, there will be enough food for everyone.
axiom R1 : ∀ p : Party, BringDish p → EnoughFood p
-- If it is a birthday party, there will be cake and ice cream.
axiom R2 : ∀ p : Party, Birthday p → Cake p ∧ IceCream p

-- Either someone brought a dish to share, or there was no birthday cake.
axiom T3 : BringDish SamParty ∨ ¬Cake SamParty
-- It is Sam's birthday party.
axiom T4 : Birthday SamParty

theorem sam_enough_food : EnoughFood SamParty := by
  have hCake : Cake SamParty := (R2 SamParty T4).left
  cases T3 with
  | inl hDish =>
      exact R1 SamParty hDish
  | inr hNoCake =>
      have hFalse : False := hNoCake hCake
      exact False.elim hFalse
</lean>

ANSWER: Yes
