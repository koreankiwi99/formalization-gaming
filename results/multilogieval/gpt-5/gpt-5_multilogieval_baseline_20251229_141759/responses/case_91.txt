<lean>
axiom Party : Prop
axiom Drums : Prop
axiom Noise : Prop
axiom Complain : Prop
axiom Police : Prop

-- If someone hosts a party, their house gets very noisy.
axiom party_noise : Party → Noise
-- If someone is practicing drums, the noise levels go up.
axiom drums_noise : Drums → Noise
-- Either there is a party, or someone is playing drums (or both).
axiom party_or_drums : Party ∨ Drums
-- If the house gets noisy from a party, neighbors start complaining.
axiom noisy_party_complain : Noise ∧ Party → Complain
-- If neighbors complain, the police are called to intervene.
axiom complain_police : Complain → Police

-- Question: If there was no noise level up, were the police called to intervene?
theorem no_noise_police : ¬Noise → Police := by
  -- From party_or_drums and the noise implications, we can derive Noise.
  have n : Noise := Or.elim party_or_drums (fun hp => party_noise hp) (fun hd => drums_noise hd)
  -- Assuming no noise leads to a contradiction, from which anything follows (including Police).
  intro hnn
  have f : False := hnn n
  exact False.elim f
</lean>

ANSWER: Yes
