<lean>
axiom obj : Type
axiom Jake : obj

axiom Nice : Prop
axiom RelaxOutside : obj → Prop
axiom ReadBooks : obj → Prop
axiom PackedEnough : obj → Prop
axiom Prepared : obj → Prop

-- If the weather is nice, then anyone can relax outside
axiom A1 : ∀ x : obj, Nice → RelaxOutside x
-- If someone reads books, then they packed enough
axiom A2 : ∀ x : obj, ReadBooks x → PackedEnough x
-- If someone is prepared, then they read books
axiom A3 : ∀ x : obj, Prepared x → ReadBooks x
-- Either the weather is nice, or a person did not pack enough, or both
axiom A4 : ∀ x : obj, Nice ∨ ¬PackedEnough x ∨ (Nice ∧ ¬PackedEnough x)

theorem jake_not_prepared_if_cannot_relax : ¬RelaxOutside Jake → ¬Prepared Jake := by
  intro hNotRelax
  -- From not relaxing and Nice → Relax, derive ¬Nice
  have hNotNice : ¬Nice := by
    intro hNice
    exact hNotRelax (A1 Jake hNice)
  -- From ¬Nice and the disjunction, derive ¬PackedEnough Jake
  have hNotPacked : ¬PackedEnough Jake := by
    have hdisj := A4 Jake
    cases hdisj with
    | inl hNice =>
        exact False.elim (hNotNice hNice)
    | inr hrest =>
      cases hrest with
      | inl hNotP => exact hNotP
      | inr hBoth => exact hBoth.right
  -- From Read → Packed, get ¬Packed → ¬Read
  have hNotRead : ¬ReadBooks Jake := by
    intro hRead
    exact hNotPacked (A2 Jake hRead)
  -- From Prepared → Read, get ¬Read → ¬Prepared
  intro hPrep
  exact hNotRead (A3 Jake hPrep)
</lean>
