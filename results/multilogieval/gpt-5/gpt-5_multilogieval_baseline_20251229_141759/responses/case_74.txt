<lean>
axiom obj : Type
axiom James : obj

axiom GoParty : obj → Prop
axiom WatchGame : obj → Prop
axiom FinishHW : obj → Prop
axiom StudyEveryday : obj → Prop
axiom PlanWell : obj → Prop
axiom Friday : Prop

-- If it is Friday night, then they are going to the party.
axiom R1 : ∀ x : obj, Friday → GoParty x
-- If they finish all their homework, then they can watch the game.
axiom R2 : ∀ x : obj, FinishHW x → WatchGame x
-- Either it is Friday night, or they are not watching the game (for each person).
axiom R3 : ∀ x : obj, Friday ∨ ¬ WatchGame x
-- If they study everyday, they finish all their homework.
axiom R4 : ∀ x : obj, StudyEveryday x → FinishHW x
-- If someone plans their schedule well, then they can study everyday.
axiom R5 : ∀ x : obj, PlanWell x → StudyEveryday x

-- Given: James is not going to the party.
axiom NotGoPartyJames : ¬ GoParty James

theorem james_not_plan_well : ¬ PlanWell James := by
  intro hplan
  have hstudy : StudyEveryday James := R5 James hplan
  have hfin : FinishHW James := R4 James hstudy
  have hwatch : WatchGame James := R2 James hfin
  have hdisj : Friday ∨ ¬ WatchGame James := R3 James
  have hfri : Friday :=
    Or.elim hdisj
      (fun hf => hf)
      (fun hnot => False.elim (hnot hwatch))
  have hgo : GoParty James := R1 James hfri
  exact NotGoPartyJames hgo
</lean>

ANSWER: No
