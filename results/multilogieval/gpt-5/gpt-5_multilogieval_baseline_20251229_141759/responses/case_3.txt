<lean>
axiom ScoreMoreA : Prop
axiom WinA : Prop
axiom FumbleB : Prop
axiom LoseB : Prop

axiom R1 : ScoreMoreA → WinA
axiom R2 : FumbleB → LoseB
axiom E1 : ScoreMoreA ∨ FumbleB

theorem teamA_scored_more_if_B_not_lose : ¬LoseB → ScoreMoreA :=
by
  intro hNotLoseB
  have hNotFumbleB : ¬FumbleB := fun hF => hNotLoseB (R2 hF)
  cases E1 with
  | inl hScore => exact hScore
  | inr hF => exact False.elim (hNotFumbleB hF)
</lean>

ANSWER: Yes
