<lean>
axiom obj : Type
axiom Matt : obj

axiom WatchedDIY : obj → Prop
axiom ReadBlogs : obj → Prop
axiom Skills : obj → Prop
axiom RedoLivingRoom : obj → Prop
axiom NewIdeas : obj → Prop

axiom watch_imp_skills : ∀ x : obj, WatchedDIY x → Skills x
axiom skills_imp_redo : ∀ x : obj, Skills x → RedoLivingRoom x
axiom blogs_imp_ideas : ∀ x : obj, ReadBlogs x → NewIdeas x
axiom watch_or_blog_matt : WatchedDIY Matt ∨ ReadBlogs Matt
axiom matt_not_redo : ¬RedoLivingRoom Matt

theorem matt_new_ideas : NewIdeas Matt := by
  have h_not_skills : ¬Skills Matt := by
    intro hskills
    have hredo : RedoLivingRoom Matt := skills_imp_redo Matt hskills
    exact matt_not_redo hredo
  have h_not_watch : ¬WatchedDIY Matt := by
    intro hwatch
    exact h_not_skills (watch_imp_skills Matt hwatch)
  have h_read : ReadBlogs Matt := Or.resolve_left watch_or_blog_matt h_not_watch
  exact blogs_imp_ideas Matt h_read
</lean>

ANSWER: Yes
