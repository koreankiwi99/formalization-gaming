<lean>
axiom obj : Type
axiom Alex : obj

axiom Applied : obj → Prop
axiom Burnt : obj → Prop
axiom Swim : obj → Prop
axiom Exercise : obj → Prop

axiom Warm : Prop
axiom Sunny : Prop

-- If someone goes swimming, then they get exercise
axiom R_swim_exercise : ∀ x : obj, Swim x → Exercise x
-- If the pool water is warm, people go swimming
axiom R_warm_swim : Warm → ∀ x : obj, Swim x
-- If the weather was sunny, the pool water will be warm
axiom R_sunny_warm : Sunny → Warm

-- It was a sunny day at the beach
axiom T_sunny : Sunny

-- Someone knew that if they applied sunscreen, then they would not get burnt,
-- and either they applied sunscreen, or they did not get any exercise.
axiom someone :
  ∃ x : obj, (Applied x → ¬ Burnt x) ∧ (Applied x ∨ ¬ Exercise x)

-- Goal: If Alex got sunburnt, then the weather was sunny.
theorem alex_burnt_then_sunny : Burnt Alex → Sunny := fun _ => T_sunny
</lean>

ANSWER: Yes
