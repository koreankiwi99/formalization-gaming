<lean>
axiom obj : Type
axiom Alex : obj

-- Propositions about the day and the quiz
axiom Monday : Prop
axiom Quiz : Prop

-- Predicates about a person
axiom Good : obj → Prop
axiom Well : obj → Prop
axiom Studied : obj → Prop

-- If it was Monday, then there would be a quiz.
axiom H1 : Monday → Quiz

-- If someone does well on the quiz, then they studied hard.
axiom H2 : ∀ x : obj, Well x → Studied x

-- Either it was Monday, or Alex did not study hard, or it was Monday and Alex did not study hard.
axiom H3 : Monday ∨ (¬ Studied Alex) ∨ (Monday ∧ ¬ Studied Alex)

-- Someone who is good at math does well on quizzes.
axiom H4 : ∀ x : obj, Good x → Well x

-- From "no quiz" we can derive "Alex is not good at math".
theorem no_quiz_not_good : ¬ Quiz → ¬ Good Alex := by
  intro hNoQuiz
  -- From no quiz and Monday → Quiz, deduce not Monday
  have hNotMonday : ¬ Monday := fun hMon => hNoQuiz (H1 hMon)
  -- From the disjunction H3 and not Monday, deduce Alex did not study
  have hNotStudied : ¬ Studied Alex := by
    apply Or.elim H3
    · intro hMon
      have : False := hNotMonday hMon
      exact this.elim
    · intro hRest
      apply Or.elim hRest
      · intro hNS; exact hNS
      · intro hAnd; exact hAnd.right
  -- Contrapose the implications to get not well and then not good
  have hNotWell : ¬ Well Alex := fun hWell => hNotStudied (H2 Alex hWell)
  have hNotGood : ¬ Good Alex := fun hGood => hNotWell (H4 Alex hGood)
  exact hNotGood
</lean>
