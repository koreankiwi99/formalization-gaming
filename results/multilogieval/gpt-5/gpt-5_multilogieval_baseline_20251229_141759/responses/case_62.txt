<lean>
axiom obj : Type
axiom Mark : obj

axiom MadeFriends : obj → Prop
axiom Enjoyed : obj → Prop
axiom JoinsClub : obj → Prop
axiom Busy : obj → Prop
axiom Outgoing : obj → Prop
axiom Introduced : obj → Prop

-- If someone made friends, then they would enjoy school.
axiom A1 : ∀ x : obj, MadeFriends x → Enjoyed x
-- If someone joins a club, then they will be busy.
axiom A2 : ∀ x : obj, JoinsClub x → Busy x
-- Either they made friends, or they are not busy. (taken universally)
axiom A3 : ∀ x : obj, MadeFriends x ∨ ¬ Busy x
-- If someone is outgoing, then they would join a club.
axiom A4 : ∀ x : obj, Outgoing x → JoinsClub x
-- If someone introduced themselves to classmates, then they are outgoing.
axiom A5 : ∀ x : obj, Introduced x → Outgoing x

-- Introduced implies Busy (via Outgoing and JoinsClub)
lemma introduced_implies_busy : ∀ x : obj, Introduced x → Busy x := by
  intro x
  intro hI
  have hOut : Outgoing x := A5 x hI
  have hJoin : JoinsClub x := A4 x hOut
  exact A2 x hJoin

-- From "made friends → enjoyed", we get "¬enjoyed → ¬made friends"
lemma not_enjoy_implies_not_madefriends : ∀ x : obj, ¬Enjoyed x → ¬MadeFriends x := by
  intro x
  intro hne
  intro hmf
  exact hne (A1 x hmf)

-- From ¬enjoyed, we can derive ¬introduced
lemma not_enjoy_implies_not_introduced : ∀ x : obj, ¬Enjoyed x → ¬Introduced x := by
  intro x
  intro hne
  have hnmf : ¬MadeFriends x := not_enjoy_implies_not_madefriends x hne
  have disj : MadeFriends x ∨ ¬Busy x := A3 x
  have hnbusy : ¬Busy x := Or.elim disj (fun hmf => False.elim (hnmf hmf)) (fun hnb => hnb)
  intro hintro
  have hbusy : Busy x := introduced_implies_busy x hintro
  exact hnbusy hbusy

theorem mark_not_intro_of_not_enjoy : ¬Enjoyed Mark → ¬Introduced Mark :=
  not_enjoy_implies_not_introduced Mark

-- We cannot derive either (¬Enjoyed Mark → Introduced Mark) or its negation from the given axioms:
-- - We proved ¬Enjoyed Mark → ¬Introduced Mark, but we do not have ¬Enjoyed Mark itself,
--   so we cannot conclude ¬(¬Enjoyed Mark → Introduced Mark).
-- - And we cannot prove (¬Enjoyed Mark → Introduced Mark) since it contradicts the derived lemma above unless ¬Enjoyed Mark is false.

<lean>
axiom obj : Type
axiom Mark : obj

axiom MadeFriends : obj → Prop
axiom Enjoyed : obj → Prop
axiom JoinsClub : obj → Prop
axiom Busy : obj → Prop
axiom Outgoing : obj → Prop
axiom Introduced : obj → Prop

-- If someone made friends, then they would enjoy school.
axiom A1 : ∀ x : obj, MadeFriends x → Enjoyed x
-- If someone joins a club, then they will be busy.
axiom A2 : ∀ x : obj, JoinsClub x → Busy x
-- Either they made friends, or they are not busy. (taken universally)
axiom A3 : ∀ x : obj, MadeFriends x ∨ ¬ Busy x
-- If someone is outgoing, then they would join a club.
axiom A4 : ∀ x : obj, Outgoing x → JoinsClub x
-- If someone introduced themselves to classmates, then they are outgoing.
axiom A5 : ∀ x : obj, Introduced x → Outgoing x

-- Introduced implies Busy (via Outgoing and JoinsClub)
lemma introduced_implies_busy : ∀ x : obj, Introduced x → Busy x := by
  intro x hx
  exact A2 x (A4 x (A5 x hx))

-- From the disjunction (MadeFriends ∨ ¬Busy), Busy rules out ¬Busy, so we get MadeFriends.
lemma introduced_implies_madefriends : ∀ x : obj, Introduced x → MadeFriends x := by
  intro x hx
  have hb : Busy x := introduced_implies_busy x hx
  have disj : MadeFriends x ∨ ¬ Busy x := A3 x
  exact Or.elim disj (fun hmf => hmf) (fun hnb => False.elim (hnb hb))

-- Thus Introduced implies Enjoyed.
lemma introduced_implies_enjoyed : ∀ x : obj, Introduced x → Enjoyed x := by
  intro x hx
  exact A1 x (introduced_implies_madefriends x hx)

-- Contraposition: ¬Enjoyed → ¬Introduced.
lemma not_enjoy_implies_not_introduced : ∀ x : obj, ¬Enjoyed x → ¬Introduced x := by
  intro x hne hx
  have henj : Enjoyed x := introduced_implies_enjoyed x hx
  exact hne henj

-- Specific to Mark: if Mark did not enjoy, then he did not introduce.
theorem mark_not_intro_of_not_enjoy : ¬Enjoyed Mark → ¬Introduced Mark :=
  not_enjoy_implies_not_introduced Mark
</lean>
