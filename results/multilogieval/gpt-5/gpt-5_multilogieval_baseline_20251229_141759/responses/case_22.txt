<lean>
axiom Drought : Prop
axiom WaterLimitsEnforced : Prop
axiom RationingInPlace : Prop

-- If there is a drought, authorities implement water restrictions (enforce limits)
axiom R1 : Drought → WaterLimitsEnforced
-- If there is a drought, rationing is required (in place)
axiom R2 : Drought → RationingInPlace

-- Given: Either the city did not enforce water limits, or it did not ration water usage
axiom H : ¬WaterLimitsEnforced ∨ ¬RationingInPlace

theorem no_drought : ¬Drought :=
by
  intro hD
  have hW : WaterLimitsEnforced := R1 hD
  have hR : RationingInPlace := R2 hD
  cases H with
  | inl hNotW => exact hNotW hW
  | inr hNotR => exact hNotR hR
</lean>

ANSWER: No
