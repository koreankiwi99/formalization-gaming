<lean>
axiom HR : Prop        -- It rained a lot / heavy rain
axiom WG : Prop        -- The garden is watered routinely
axiom SS : Prop        -- The soil is saturated
axiom FB : Prop        -- The flowers bloom (brightly)
axiom FR : Prop        -- There is fragrance
axiom AB : Prop        -- The garden attracts butterflies

axiom R1 : WG → FB         -- If the garden is watered routinely, flowers bloom brightly
axiom R2 : HR → SS         -- If it rains a lot, the soil becomes saturated
axiom R3 : WG ∨ ¬ SS       -- Either the garden is watered routinely, or the soil did not saturate
axiom R4 : FB → FR         -- If the flowers bloom, they release fragrance
axiom R5 : FR → AB         -- If there is fragrance, it attracts butterflies

theorem heavy_rain_attracts : HR → AB := by
  intro hr
  have ss : SS := R2 hr
  have wg : WG := Or.elim R3 (fun hwg => hwg) (fun nss => False.elim (nss ss))
  have fb : FB := R1 wg
  have fr : FR := R4 fb
  exact R5 fr
</lean>

ANSWER: Yes
