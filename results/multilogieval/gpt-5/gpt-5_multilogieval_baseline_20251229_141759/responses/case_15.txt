<lean>
axiom obj : Type
axiom Student : obj

axiom Studies : obj → Prop
axiom Participates : obj → Prop
axiom Excels : obj → Prop
axiom Leader : obj → Prop
axiom Prefect : obj → Prop

axiom R1 : ∀ x : obj, Studies x → Excels x
axiom R2 : ∀ x : obj, Participates x → Leader x
axiom R3 : ∀ x : obj, Leader x → Prefect x

axiom Disj : Studies Student ∨ Participates Student
axiom NotExcels : ¬ Excels Student

theorem student_prefect : Prefect Student := by
  have not_studies : ¬ Studies Student := by
    intro hs
    have hx : Excels Student := R1 Student hs
    exact NotExcels hx
  have participates : Participates Student := by
    cases Disj with
    | inl hs => exact False.elim (not_studies hs)
    | inr hp => exact hp
  have leader : Leader Student := R2 Student participates
  exact R3 Student leader
</lean>

ANSWER: Yes
