<lean>
axiom Company : Type
axiom Acme : Company

axiom Innovative : Company → Prop
axiom MarketedEffectively : Company → Prop
axiom CommerciallySuccessful : Company → Prop

-- If a company develops an innovative product, then if they market it effectively, the product will be commercially successful.
axiom R1 : ∀ c : Company, Innovative c → (MarketedEffectively c → CommerciallySuccessful c)

-- Facts about Acme Inc.
axiom T1 : MarketedEffectively Acme
axiom T2 : ¬CommerciallySuccessful Acme

-- From marketing effectively and not being successful, Acme did not develop an innovative product.
theorem acme_not_innovative : ¬Innovative Acme := by
  intro hInnov
  have himp : MarketedEffectively Acme → CommerciallySuccessful Acme := R1 Acme hInnov
  have hsucc : CommerciallySuccessful Acme := himp T1
  exact T2 hsucc
</lean>

ANSWER: No
