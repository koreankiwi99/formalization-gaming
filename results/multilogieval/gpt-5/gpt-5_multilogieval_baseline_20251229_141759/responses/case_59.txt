<lean>
axiom person : Type
axiom Emma : person

axiom LearnChords : person → Prop
axiom LearnStrum : person → Prop
axiom PlaySimple : person → Prop
axiom Entertain : person → Prop
axiom TakesLessons : person → Prop
axiom KnowTheory : person → Prop

-- If someone learned basic chords and strumming patterns, they could play simple songs.
axiom A1 : ∀ x : person, LearnChords x ∧ LearnStrum x → PlaySimple x
-- If someone can play simple songs, they can entertain friends with the guitar.
axiom A2 : ∀ x : person, PlaySimple x → Entertain x
-- If someone takes guitar lessons, they gain knowledge of music theory.
axiom A3 : ∀ x : person, TakesLessons x → KnowTheory x
-- Either they learned basic chords, or they took guitar lessons, or both.
axiom A4 : ∀ x : person, LearnChords x ∨ TakesLessons x

-- Given: Emma is unable to entertain friends with the guitar.
axiom Emma_not_entertain : ¬ Entertain Emma

-- From A2 and Emma_not_entertain, we can derive that Emma cannot play simple songs.
theorem not_play_simple_Emma : ¬ PlaySimple Emma :=
  fun h => Emma_not_entertain (A2 Emma h)

-- Cannot prove KnowTheory Emma or ¬ KnowTheory Emma from given axioms.
