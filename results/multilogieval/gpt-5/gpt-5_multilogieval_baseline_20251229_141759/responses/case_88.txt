<lean>
axiom Oven350 : Prop
axiom BakesProperly : Prop
axiom EggsInBatter : Prop
axiom FluffyWhenBaked : Prop
axiom DeliciousTaste : Prop
axiom GuestsHappy : Prop

-- If the oven temperature is set to 350 degrees, the cake will bake properly.
axiom A1 : Oven350 → BakesProperly
-- If the cake batter contains eggs, it will be fluffy when baked.
axiom A2 : EggsInBatter → FluffyWhenBaked
-- Either the oven was set correctly (to 350), or the batter did not contain eggs.
axiom A3 : Oven350 ∨ ¬ EggsInBatter
-- If a cake bakes properly, it will taste delicious.
axiom A4 : BakesProperly → DeliciousTaste
-- If a cake tastes delicious, guests at the party will be happy.
axiom A5 : DeliciousTaste → GuestsHappy

theorem eggs_implies_happy : EggsInBatter → GuestsHappy :=
fun hEggs =>
  Or.elim A3
    (fun h350 => A5 (A4 (A1 h350)))
    (fun hNotEggs => False.elim (hNotEggs hEggs))
</lean>

ANSWER: Yes
