<lean>
axiom BelowFreezing : Prop
axiom SnowOnGround : Prop
axiom EnoughSnow : Prop

axiom obj : Type
axiom Mary : obj
axiom LovesWinter : obj → Prop
axiom BuildsSnowman : obj → Prop

axiom R1 : BelowFreezing → SnowOnGround
axiom R2 : ∀ x : obj, BuildsSnowman x → EnoughSnow
axiom R3 : (BelowFreezing ∨ ¬EnoughSnow) ∨ (BelowFreezing ∧ ¬EnoughSnow)
axiom R4 : ∀ x : obj, LovesWinter x → BuildsSnowman x

-- From no snow on the ground, conclude Mary does not love winter
theorem mary_not_love_if_no_snow : ¬SnowOnGround → ¬LovesWinter Mary := by
  intro hNoSnow
  -- From BelowFreezing → SnowOnGround and ¬SnowOnGround, get ¬BelowFreezing
  have hNotBF : ¬BelowFreezing := by
    intro hBF
    exact hNoSnow (R1 hBF)
  -- From the certainty disjunction and ¬BelowFreezing, get ¬EnoughSnow
  have hNotEnough : ¬EnoughSnow := by
    cases R3 with
    | inl h =>
      cases h with
      | inl hBF =>
        exact False.elim (hNotBF hBF)
      | inr hNES =>
        exact hNES
    | inr hBFNES =>
      exact hBFNES.right
  -- If Mary loved winter, she'd build a snowman and there would be enough snow, contradiction
  intro hLove
  have hBuild : BuildsSnowman Mary := R4 Mary hLove
  have hEnough : EnoughSnow := R2 Mary hBuild
  exact hNotEnough hEnough
</lean>

ANSWER: No
