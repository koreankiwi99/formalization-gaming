<lean>
axiom obj : Type
axiom Max : obj

axiom PackageOnTime : Prop
axiom SpecialOrderReady : Prop

axiom CanPickUp : obj → Prop
axiom AbleStartNewProject : obj → Prop
axiom StartsNewProject : obj → Prop
axiom LearnNewSkills : obj → Prop

-- If the package arrives on time, people will be able to start a new project.
axiom R1 : PackageOnTime → ∀ x : obj, AbleStartNewProject x
-- If the special order is in stock, they can pick up the materials they need.
axiom R2 : SpecialOrderReady → ∀ x : obj, CanPickUp x
-- Either the package will arrive on time or the special order will be ready.
axiom R3 : PackageOnTime ∨ SpecialOrderReady
-- If someone starts his new project, they will learn some new skills.
axiom R4 : ∀ x : obj, StartsNewProject x → LearnNewSkills x

-- Max was unable to pick up the materials he needed.
axiom Max_not_pickup : ¬ CanPickUp Max

theorem not_special_order : ¬ SpecialOrderReady := by
  intro s
  have hall : ∀ x, CanPickUp x := R2 s
  have hmax : CanPickUp Max := hall Max
  exact Max_not_pickup hmax

theorem package_on_time : PackageOnTime := by
  cases R3 with
  | inl hp => exact hp
  | inr hs =>
      have : False := not_special_order hs
      exact False.elim this

theorem able_start_max : AbleStartNewProject Max := by
  have hp : PackageOnTime := package_on_time
  have hall : ∀ x, AbleStartNewProject x := R1 hp
  exact hall Max

-- We cannot derive that Max starts his project from merely being able to start it,
-- and thus cannot prove LearnNewSkills Max.
-- Likewise, we have no information to prove ¬LearnNewSkills Max.
-- Therefore, the queried statement is unprovable from the given axioms.
</lean>
