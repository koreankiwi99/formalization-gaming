<lean>
axiom obj : Type
axiom Alex : obj

axiom Relax : obj → Prop
axiom ClearMind : obj → Prop
axiom FinishTasks : obj → Prop
axiom Organized : obj → Prop
axiom TodoList : obj → Prop

axiom WeekendComing : Prop

-- If it was the weekend coming up, then Alex would relax.
axiom R1 : WeekendComing → Relax Alex
-- If someone finishes their tasks, then they have a clear mind.
axiom R2 : ∀ x : obj, FinishTasks x → ClearMind x
-- Either the weekend was coming, or Alex did not have a clear mind.
axiom R3 : WeekendComing ∨ ¬ ClearMind Alex
-- If someone is organized, then they finish their tasks.
axiom R4 : ∀ x : obj, Organized x → FinishTasks x
-- If they made a to-do list, then they are organized.
axiom R5 : ∀ x : obj, TodoList x → Organized x

-- Given: Alex does not relax this weekend.
axiom H : ¬ Relax Alex

theorem alex_not_todolist : ¬ TodoList Alex := by
  -- From ¬Relax and (WeekendComing → Relax), get ¬WeekendComing
  have hNotW : ¬ WeekendComing := by
    intro hW
    exact H (R1 hW)
  -- From WeekendComing ∨ ¬ClearMind and ¬WeekendComing, get ¬ClearMind
  have hNotCM : ¬ ClearMind Alex := by
    cases R3 with
    | inl hW =>
        exact fun _ => False.elim (hNotW hW)
    | inr hNcm =>
        exact hNcm
  -- From ¬ClearMind and (Finish → ClearMind), get ¬Finish
  have hNotF : ¬ FinishTasks Alex := by
    intro hF
    exact hNotCM (R2 Alex hF)
  -- From ¬Finish and (Organized → Finish), get ¬Organized
  have hNotOrg : ¬ Organized Alex := by
    intro hOrg
    exact hNotF (R4 Alex hOrg)
  -- From ¬Organized and (TodoList → Organized), get ¬TodoList
  intro hTodo
  exact hNotOrg (R5 Alex hTodo)
</lean>

ANSWER: No
