<lean>
axiom PracticesHard : Prop
axiom Prepared : Prop
axiom Injured : Prop
axiom CanPlay : Prop
axiom Wins : Prop
axiom Celebrate : Prop

axiom A1 : PracticesHard → Prepared
axiom A2 : Injured → ¬ CanPlay
axiom A3 : PracticesHard ∨ ¬ Injured
axiom A4 : Prepared → Wins
axiom A5 : Wins → Celebrate

theorem injured_implies_celebrate : Injured → Celebrate := by
  intro hI
  cases A3 with
  | inl hP =>
      have hPrep := A1 hP
      have hWin := A4 hPrep
      exact A5 hWin
  | inr hNotI =>
      have hFalse : False := hNotI hI
      exact False.elim hFalse
</lean>

ANSWER: Yes
