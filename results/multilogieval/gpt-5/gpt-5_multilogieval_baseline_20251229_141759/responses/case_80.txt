<lean>
axiom Rain : Prop
axiom Game : Prop
axiom PoolOpen : Prop
axiom Swim : Prop
axiom Enjoy : Prop
axiom Smiles : Prop

axiom A1 : ¬Rain → Game
axiom A2 : PoolOpen → Swim
axiom A3 : ¬Rain ∨ PoolOpen
axiom A4 : Game → Enjoy
axiom A5 : Enjoy → Smiles

theorem smiles_if_no_swim : (¬ Swim) → Smiles := by
  intro hns
  cases A3 with
  | inl hnr =>
      have hg : Game := A1 hnr
      have he : Enjoy := A4 hg
      exact A5 he
  | inr hpo =>
      have hs : Swim := A2 hpo
      exact False.elim (hns hs)
</lean>

ANSWER: Yes
