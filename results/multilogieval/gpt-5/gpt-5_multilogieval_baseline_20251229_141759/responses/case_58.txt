<lean>
axiom AddSalt : Prop
axiom Savory : Prop
axiom Stir : Prop
axiom Thicken : Prop
axiom Taste : Prop
axiom PrepareSalad : Prop

axiom A1 : AddSalt → Savory
axiom A2 : Stir → Thicken
axiom A3 : AddSalt ∨ ¬ Thicken
axiom A4 : Taste → Stir
axiom A5 : Taste ∨ PrepareSalad

theorem prepared_salad_of_not_savory (hns : ¬ Savory) : PrepareSalad :=
by
  -- Taste implies Savory
  have h_ts : Taste → Savory :=
    fun ht =>
      let hs : Stir := A4 ht
      let htkn : Thicken := A2 hs
      have hnn : ¬¬ Thicken := fun hnt => hnt htkn
      have has : AddSalt := by
        cases A3 with
        | inl ha => exact ha
        | inr hnt => exact False.elim (hnn hnt)
      show Savory from A1 has
  -- Thus not Taste from not Savory
  have hntaste : ¬ Taste := fun ht => hns (h_ts ht)
  -- From Taste ∨ PrepareSalad and ¬Taste, conclude PrepareSalad
  cases A5 with
  | inl ht => exact False.elim (hntaste ht)
  | inr hs => exact hs
</lean>

ANSWER: Yes
