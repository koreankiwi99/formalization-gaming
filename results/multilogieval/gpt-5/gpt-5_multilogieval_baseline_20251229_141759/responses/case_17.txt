<lean>
axiom obj : Type
axiom James : obj

axiom Practiced : obj → Prop
axiom Prepared : obj → Prop
axiom Confident : obj → Prop
axiom JoinedClub : obj → Prop

-- If someone practiced thoroughly, they feel prepared.
axiom A1 : ∀ x : obj, Practiced x → Prepared x
-- If someone feels prepared, they can speak confidently.
axiom A2 : ∀ x : obj, Prepared x → Confident x
-- Either they practiced or they joined a club (inclusive or).
axiom A3 : ∀ x : obj, Practiced x ∨ JoinedClub x

theorem james_join_if_not_confident : ¬Confident James → JoinedClub James := by
  intro hnc
  have hnp : ¬Prepared James := fun hp => hnc (A2 James hp)
  have hnpra : ¬Practiced James := fun hpr => hnp (A1 James hpr)
  have disj := A3 James
  cases disj with
  | inl hpr => exact False.elim (hnpra hpr)
  | inr hc => exact hc
</lean>

ANSWER: Yes
