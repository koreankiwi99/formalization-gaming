<lean>
axiom Practices : Prop
axiom Prepared : Prop
axiom Hot : Prop
axiom Postponed : Prop
axiom PlayWell : Prop
axiom Win : Prop

-- If the team practices hard, they will be prepared for the game.
axiom R1 : Practices → Prepared
-- If it is very hot outside, the game is postponed.
axiom R2 : Hot → Postponed
-- Either the team practiced hard, or the game was not postponed.
axiom R3 : Practices ∨ ¬ Postponed
-- If a team is prepared, they will play well.
axiom R4 : Prepared → PlayWell
-- When a team plays well, they will win.
axiom R5 : PlayWell → Win

theorem hot_implies_win : Hot → Win := by
  intro hHot
  have hPost : Postponed := R2 hHot
  have hDisj : Practices ∨ ¬ Postponed := R3
  cases hDisj with
  | inl hPract =>
    have hPrep : Prepared := R1 hPract
    have hPlay : PlayWell := R4 hPrep
    exact R5 hPlay
  | inr hNotPost =>
    have hFalse : False := hNotPost hPost
    exact False.elim hFalse
</lean>

ANSWER: Yes
