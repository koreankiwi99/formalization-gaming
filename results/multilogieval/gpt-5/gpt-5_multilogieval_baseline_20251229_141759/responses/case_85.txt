<lean>
axiom obj : Type
axiom John : obj

axiom Water : obj → Prop
axiom Bloom : obj → Prop
axiom Attract : obj → Prop
axiom Organic : obj → Prop
axiom SoilImproves : obj → Prop
axiom Healthier : obj → Prop

-- If someone waters adequately, their plants bloom brightly
axiom R1 : ∀ x : obj, Water x → Bloom x
-- If the plants bloom brightly, the garden attracts more bees and butterflies
axiom R2 : ∀ x : obj, Bloom x → Attract x
-- Either someone waters adequately or they use organic fertilizers (or both)
axiom R3 : ∀ x : obj, Water x ∨ Organic x
-- If someone uses organic fertilizers, the soil quality improves
axiom R4 : ∀ x : obj, Organic x → SoilImproves x
-- If the soil quality improves, the plants grow healthier
axiom R5 : ∀ x : obj, SoilImproves x → Healthier x

-- Question: If John's garden does not attract more bees and butterflies, is his soil high quality?
theorem john_soil_high_quality : ¬ Attract John → SoilImproves John := by
  intro hNotAttr
  -- From Water → Bloom → Attract, derive ¬Attract → ¬Water
  have hNotWater : ¬ Water John := by
    intro hWater
    have hBloom : Bloom John := R1 John hWater
    have hAttr : Attract John := R2 John hBloom
    exact hNotAttr hAttr
  -- From Water ∨ Organic and ¬Water, conclude Organic, hence SoilImproves
  have hDisj : Water John ∨ Organic John := R3 John
  exact Or.elim hDisj
    (fun hW => False.elim (hNotWater hW))
    (fun hOrg => R4 John hOrg)
</lean>

ANSWER: Yes
