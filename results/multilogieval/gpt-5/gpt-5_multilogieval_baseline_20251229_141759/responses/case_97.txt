<lean>
axiom Sub : Prop
axiom Pizza : Prop
axiom GoofOff : Prop
axiom Hyper : Prop
axiom LearnMuch : Prop
axiom Homework : Prop

axiom R1 : Sub → GoofOff
axiom R2 : Pizza → Hyper
axiom T1 : Sub ∨ Pizza
axiom R3 : GoofOff → ¬ LearnMuch
axiom R4 : ¬ LearnMuch → Homework

theorem notHyper_homework : ¬ Hyper → Homework := by
  intro nHyper
  have nPizza : ¬ Pizza := fun hPizza => nHyper (R2 hPizza)
  have hSub : Sub := by
    cases T1 with
    | inl h => exact h
    | inr hP => exact False.elim (nPizza hP)
  have hGoof : GoofOff := R1 hSub
  have nLearn : ¬ LearnMuch := R3 hGoof
  exact R4 nLearn
</lean>

ANSWER: Yes
