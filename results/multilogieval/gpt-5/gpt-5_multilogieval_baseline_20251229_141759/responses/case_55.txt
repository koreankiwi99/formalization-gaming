<lean>
axiom obj : Type
axiom m : obj

axiom Good : obj → Prop
axiom Many : obj → Prop
axiom BigStar : obj → Prop
axiom Sequel : obj → Prop

-- If the movie gets good reviews, then many people will go see it.
axiom A1 : ∀ x : obj, Good x → Many x

-- Either the movie got good reviews, or not many people went to see it,
-- or it got good reviews but attendance was still low. (not used in the proof)
axiom A3 : Good m ∨ ¬ Many m ∨ (Good m ∧ ¬ Many m)

-- If a big star is in the movie, then it got good reviews.
axiom A4 : ∀ x : obj, BigStar x → Good x

-- Either there is a big star in it, or it is a sequel, or both.
axiom A5 : BigStar m ∨ Sequel m ∨ (BigStar m ∧ Sequel m)

-- If attendance was low (not many people), then it is a sequel.
theorem low_attendance_implies_sequel : (¬ Many m) → Sequel m := by
  intro hLow
  have hBS_many : BigStar m → Many m := by
    intro hBS
    exact A1 m (A4 m hBS)
  have hNotBS : ¬ BigStar m := by
    intro hBS
    exact hLow (hBS_many hBS)
  have h := A5
  cases h with
  | inl hBS =>
      exact False.elim (hNotBS hBS)
  | inr h' =>
      cases h' with
      | inl hSeq => exact hSeq
      | inr hBoth => exact hBoth.right
</lean>

ANSWER: Yes
