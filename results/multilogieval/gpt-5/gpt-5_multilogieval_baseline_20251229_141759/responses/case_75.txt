<lean>
axiom obj : Type
axiom LizCar : obj

axiom PreventiveMaintenance : obj → Prop
axiom QualityParts : obj → Prop
axiom LongServiceLife : obj → Prop

-- If preventive maintenance is done on a car, then if quality parts are used, the car will have a long service life.
axiom R1 : ∀ x : obj, PreventiveMaintenance x → (QualityParts x → LongServiceLife x)

-- Facts about Liz's car
axiom Q_Liz : QualityParts LizCar
axiom not_Long_Liz : ¬ LongServiceLife LizCar

theorem not_preventive_Liz : ¬ PreventiveMaintenance LizCar := by
  intro hP
  have hImp : QualityParts LizCar → LongServiceLife LizCar := R1 LizCar hP
  have hL : LongServiceLife LizCar := hImp Q_Liz
  exact not_Long_Liz hL
</lean>

ANSWER: No
