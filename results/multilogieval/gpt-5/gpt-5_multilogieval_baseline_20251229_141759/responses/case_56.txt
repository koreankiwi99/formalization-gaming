<lean>
axiom obj : Type
axiom Emma : obj

-- Predicates
axiom SpeaksDaily : obj → Prop
axiom ImprovedFluency : obj → Prop
axiom ImprovedVocabulary : obj → Prop
axiom RealConversations : obj → Prop
axiom ListensPodcasts : obj → Prop
axiom GainsListening : obj → Prop

-- Axioms from the context
axiom R1 : ∀ x : obj, SpeaksDaily x → (ImprovedFluency x ∧ ImprovedVocabulary x)
axiom R2 : ∀ x : obj, ImprovedFluency x → RealConversations x
axiom R3 : ∀ x : obj, SpeaksDaily x ∨ ListensPodcasts x
axiom R4 : ∀ x : obj, ListensPodcasts x → GainsListening x

-- Goal: If Emma cannot have real conversations, then she gains valuable listening comprehension
theorem emma_gains_listening : ¬ RealConversations Emma → GainsListening Emma := by
  intro hNoConv
  -- Show Emma does not speak daily with native speakers
  have hNotSD : ¬ SpeaksDaily Emma := by
    intro hSD
    have hfluvoc := R1 Emma hSD
    have hFlu : ImprovedFluency Emma := And.left hfluvoc
    have hConv : RealConversations Emma := R2 Emma hFlu
    exact hNoConv hConv
  -- From the disjunction, deduce that Emma listens to podcasts
  have hDisj := R3 Emma
  have hListen : ListensPodcasts Emma := by
    cases hDisj with
    | inl hSD => exact False.elim (hNotSD hSD)
    | inr hLP => exact hLP
  -- Thus she gains valuable listening comprehension
  exact R4 Emma hListen
</lean>

ANSWER: Yes
