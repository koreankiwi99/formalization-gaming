<lean>
axiom obj : Type
axiom Alex : obj

axiom StudiedHard : obj → Prop
axiom GoodGrades : obj → Prop
axiom ParticipatedEA : obj → Prop
axiom NewFriends : obj → Prop
axiom InvitedToEvents : obj → Prop

-- Rules from the context
axiom R1 : ∀ x : obj, StudiedHard x → GoodGrades x
axiom R2 : ∀ x : obj, ParticipatedEA x → NewFriends x
axiom R3 : ∀ x : obj, NewFriends x → InvitedToEvents x

-- The student (Alex) either studied hard or participated in extracurricular activities
axiom disj : StudiedHard Alex ∨ ParticipatedEA Alex

-- Given: Alex did not get good grades
axiom notGG : ¬ GoodGrades Alex

theorem alex_invited : InvitedToEvents Alex :=
by
  -- From the disjunction and the fact that studying implies good grades (which Alex doesn't have),
  -- we deduce Alex participated in extracurricular activities.
  have hParticipated : ParticipatedEA Alex :=
    Or.elim disj
      (fun hS => False.elim (notGG (R1 Alex hS)))
      (fun hP => hP)
  -- Participation gives new friends
  have hNF : NewFriends Alex := R2 Alex hParticipated
  -- New friends imply invitation to social events
  exact R3 Alex hNF
</lean>

ANSWER: Yes
