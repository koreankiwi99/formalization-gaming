<lean>
axiom Person : Type
axiom James : Person

axiom StartNew : Person → Prop
axiom Motivated : Person → Prop
axiom FollowRoutine : Person → Prop
axiom Frustrated : Person → Prop
axiom IndulgedSnacks : Person → Prop
axiom GymClosed : Prop

-- If someone starts a new workout routine, they will be motivated.
axiom A1 : ∀ x : Person, StartNew x → Motivated x
-- If the gym closes, they can't follow their routine.
axiom A2 : ∀ x : Person, GymClosed → ¬ FollowRoutine x
-- Either James started a new workout, or the gym was closed (or both).
axiom A3 : StartNew James ∨ GymClosed
-- If they couldn't follow their routine, they would feel frustrated.
axiom A4 : ∀ x : Person, ¬ FollowRoutine x → Frustrated x
-- If they felt frustrated, they might indulge in some unhealthy snacks.
axiom A5 : ∀ x : Person, Frustrated x → IndulgedSnacks x

-- James was not motivated to workout.
axiom NotMotivatedJames : ¬ Motivated James

theorem james_indulged : IndulgedSnacks James := by
  -- From the disjunction and not-motivated fact, deduce the gym was closed.
  have hGym : GymClosed := by
    cases A3 with
    | inl hstart =>
      have hmot : Motivated James := A1 James hstart
      have : False := NotMotivatedJames hmot
      exact this.elim
    | inr hgc =>
      exact hgc
  -- Gym closed implies James couldn't follow his routine.
  have hnotfollow : ¬ FollowRoutine James := A2 James hGym
  -- Then he felt frustrated.
  have hfr : Frustrated James := A4 James hnotfollow
  -- And thus he indulged in unhealthy snacks.
  exact A5 James hfr
</lean>

ANSWER: Yes
