<lean>
axiom obj : Type
axiom Linda : obj

axiom DiseaseResistant : obj → Prop
axiom ProperCare : obj → Prop
axiom BountifulHarvest : obj → Prop

-- If someone plants disease-resistant varieties, then if they water and feed properly, they will have a bountiful harvest.
axiom Rule : ∀ x : obj, DiseaseResistant x → ProperCare x → BountifulHarvest x

-- Facts about Linda
axiom LindaProper : ProperCare Linda
axiom LindaNotBountiful : ¬ BountifulHarvest Linda

-- Conclusion: Linda did not plant disease-resistant varieties.
theorem linda_not_disease_resistant : ¬ DiseaseResistant Linda :=
by
  intro hDR
  have hB : BountifulHarvest Linda := Rule Linda hDR LindaProper
  exact LindaNotBountiful hB
</lean>

ANSWER: No
