<lean>
axiom obj : Type
axiom Sarah : obj

axiom Practice : obj → Prop            -- practiced guitar scales and chords daily
axiom Technique : obj → Prop           -- technique improved
axiom LearnFast : obj → Prop           -- can learn songs faster
axiom Listen : obj → Prop              -- listens attentively to music

axiom R1 : ∀ x : obj, Practice x → Technique x
axiom R2 : ∀ x : obj, Technique x → LearnFast x
axiom H3 : Practice Sarah ∨ Listen Sarah

theorem sarah_listens_of_not_learnfast : ¬ LearnFast Sarah → Listen Sarah := by
  intro hnot
  have hImp : Practice Sarah → LearnFast Sarah := by
    intro hp
    have ht : Technique Sarah := R1 Sarah hp
    exact R2 Sarah ht
  have hNotP : ¬ Practice Sarah := by
    intro hp
    exact hnot (hImp hp)
  have disj := H3
  cases disj with
  | inl hp => exact False.elim (hNotP hp)
  | inr hl => exact hl
</lean>

ANSWER: Yes
