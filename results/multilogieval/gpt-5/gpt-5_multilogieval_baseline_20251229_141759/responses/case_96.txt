axiom Confident : Person → Exam → Prop
axiom PerformsWell : Person → Exam → Prop

axiom SevereWeather : Prop
axiom ClassCanceledDueToWeather : Prop

-- If someone studies hard for an exam, they will feel confident for that exam.
axiom R1 : ∀ p e, StudiedHard p e → Confident p e
-- If there is severe weather in the forecast, class is canceled due to the weather (modeled as an implication).
axiom R2 : SevereWeather → ClassCanceledDueToWeather
-- Either Alex studied hard for his biology exam, or class was not canceled due to the weather.
axiom R3 : StudiedHard Alex Biology ∨ ¬ ClassCanceledDueToWeather
-- If a student feels confident about the material, they will do well on the exam.
axiom R4 : ∀ p e, Confident p e → PerformsWell p e
-- If a student performs well on an exam, they will be satisfied with their hard work.
axiom R5 : ∀ p e, PerformsWell p e → Satisfied p e

-- A derived fact: studying hard for an exam implies being satisfied with that exam's result.
theorem studied_implies_satisfied : ∀ p e, StudiedHard p e → Satisfied p e := by
  intro p e h
  have hc : Confident p e := R1 p e h
  have hw : PerformsWell p e := R4 p e hc
  exact R5 p e hw
