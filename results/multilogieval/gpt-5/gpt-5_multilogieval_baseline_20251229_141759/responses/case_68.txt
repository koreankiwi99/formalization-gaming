<lean>
axiom Person : Type
axiom Exam : Type

axiom Emma : Person
axiom BioExam : Exam

axiom StudyHard : Person → Prop
axiom EnoughSleepBeforeExam : Person → Exam → Prop
axiom PerformWell : Person → Exam → Prop
axiom AceTest : Person → Exam → Prop

-- Belief: If someone studies hard, then if they get enough sleep before an exam,
-- they will perform well and ace the test.
axiom Belief :
  ∀ x : Person, StudyHard x → ∀ e : Exam, EnoughSleepBeforeExam x e → PerformWell x e ∧ AceTest x e

-- Facts about Emma and her biology exam
axiom EmmaSlept : EnoughSleepBeforeExam Emma BioExam
axiom EmmaNotAce : ¬ AceTest Emma BioExam

theorem emma_not_study_hard : ¬ StudyHard Emma := by
  intro hStudy
  have hImp : EnoughSleepBeforeExam Emma BioExam → PerformWell Emma BioExam ∧ AceTest Emma BioExam :=
    Belief Emma hStudy BioExam
  have hConj : PerformWell Emma BioExam ∧ AceTest Emma BioExam := hImp EmmaSlept
  have hAce : AceTest Emma BioExam := And.right hConj
  exact EmmaNotAce hAce
</lean>

ANSWER: No
