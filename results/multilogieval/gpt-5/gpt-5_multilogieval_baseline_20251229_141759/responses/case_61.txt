<lean>
axiom obj : Type
axiom Priya : obj

axiom Attends : obj → Prop        -- attends lectures regularly
axiom Studies : obj → Prop        -- studies diligently
axiom Excels  : obj → Prop        -- excels in the course

-- Context rule: if a student attends regularly, then if they study diligently, they will excel
axiom R : ∀ x : obj, Attends x → (Studies x → Excels x)

-- Given facts about Priya
axiom Priya_studies : Studies Priya
axiom Priya_not_excels : ¬ Excels Priya

-- Conclusion: Priya did not attend lectures regularly
theorem Priya_not_attends : ¬ Attends Priya := by
  intro hAtt
  have hStudyImp : Studies Priya → Excels Priya := R Priya hAtt
  have hExcels : Excels Priya := hStudyImp Priya_studies
  exact Priya_not_excels hExcels
</lean>

ANSWER: No
