<lean>
axiom Winning : Prop
axiom Rivalry : Prop
axiom LoudEnergized : Prop
axiom Invested : Prop
axiom Electric : Prop
axiom Unforgettable : Prop

axiom R1 : Winning → LoudEnergized
axiom R2 : Rivalry → Invested
axiom R3 : Winning ∨ Rivalry
axiom R4 : LoudEnergized → Electric
axiom R5 : Electric → Unforgettable

theorem not_invested_implies_electric : ¬Invested → Electric := by
  intro hni
  cases R3 with
  | inl hw =>
      apply R4
      exact R1 hw
  | inr hr =>
      have hi : Invested := R2 hr
      exact False.elim (hni hi)
</lean>

ANSWER: Yes
