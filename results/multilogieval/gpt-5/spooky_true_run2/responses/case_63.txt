<lean>
axiom obj : Type
axiom Linda : obj

axiom PlantsDiseaseResistant : obj → Prop
axiom WateredProperly : obj → Prop
axiom FedProperly : obj → Prop
axiom BountifulHarvest : obj → Prop

-- Among gardeners it's said: if someone plants disease-resistant varieties,
-- then if they water and feed properly, they will have a bountiful harvest.
axiom Rule :
  ∀ x : obj, PlantsDiseaseResistant x → (WateredProperly x ∧ FedProperly x → BountifulHarvest x)

-- Facts about Linda
axiom Linda_watered : WateredProperly Linda
axiom Linda_fed : FedProperly Linda
axiom Linda_not_bountiful : ¬ BountifulHarvest Linda

-- From the rule and Linda's facts, we can derive that Linda did not plant disease-resistant varieties.
theorem linda_not_planted_DR : ¬ PlantsDiseaseResistant Linda :=
by
  intro hPlant
  have hWF : WateredProperly Linda ∧ FedProperly Linda := And.intro Linda_watered Linda_fed
  have hB : BountifulHarvest Linda := Rule Linda hPlant hWF
  exact Linda_not_bountiful hB
