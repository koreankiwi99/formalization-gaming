<lean>
axiom obj : Type
axiom Priya : obj

axiom StudiesHard : obj → Prop
axiom GoodGrades : obj → Prop
axiom ParentsProud : obj → Prop
axiom IceCream : obj → Prop

axiom Snowstorm : Prop
axiom SchoolCanceled : Prop

-- If someone studies hard for an exam, they will get good grades.
axiom H1 : ∀ x : obj, StudiesHard x → GoodGrades x
-- If there is a snowstorm, then school is canceled.
axiom H2 : Snowstorm → SchoolCanceled
-- Either Priya studied hard, or school was not called off.
axiom H3 : StudiesHard Priya ∨ ¬ SchoolCanceled
-- If a student gets good grades, their parents are proud.
axiom H4 : ∀ x : obj, GoodGrades x → ParentsProud x
-- When parents are proud, they take their kids out for ice cream.
axiom H5 : ∀ x : obj, ParentsProud x → IceCream x

theorem priya_icecream_if_snowstorm : Snowstorm → IceCream Priya := by
  intro hs
  have hc : SchoolCanceled := H2 hs
  have hsh : StudiesHard Priya := by
    cases H3 with
    | inl h => exact h
    | inr hnot => exact False.elim (hnot hc)
  have hgg : GoodGrades Priya := H1 Priya hsh
  have hpp : ParentsProud Priya := H4 Priya hgg
  exact H5 Priya hpp
</lean>

ANSWER: Yes
