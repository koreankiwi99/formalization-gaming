<lean>
axiom person : Type
axiom Sam : person

axiom Impressed : person → Prop
axiom Promoted : person → Prop
axiom Extra : person → Prop
axiom Experience : person → Prop
axiom Assigned : person → Prop

-- If someone impressed the manager, they get a promotion.
axiom A1 : ∀ x : person, Impressed x → Promoted x
-- If someone took on extra responsibilities, they gain experience.
axiom A2 : ∀ x : person, Extra x → Experience x
-- This employee (Sam) either impressed the manager or took on extra work.
axiom A3 : Impressed Sam ∨ Extra Sam
-- If someone gains experience, they will be assigned to lead a new team.
axiom A4 : ∀ x : person, Experience x → Assigned x
-- Given: Sam did not get a promotion.
axiom Hnp : ¬ Promoted Sam

theorem sam_assigned : Assigned Sam := by
  have h_not_impressed : ¬ Impressed Sam := by
    intro hI
    exact Hnp (A1 Sam hI)
  have h_extra : Extra Sam := by
    cases A3 with
    | inl hI =>
      exact False.elim (h_not_impressed hI)
    | inr hE =>
      exact hE
  have h_exp : Experience Sam := A2 Sam h_extra
  exact A4 Sam h_exp
</lean>

ANSWER: Yes
