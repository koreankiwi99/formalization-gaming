<lean>
axiom Monsoon : Prop
axiom HeavyRains : Prop
axiom Waters : Prop
axiom WatersDaily : Prop
axiom Strong : Prop
axiom Bountiful : Prop
axiom GoodProfit : Prop

axiom R1 : Waters → Strong
axiom R2 : Monsoon → HeavyRains
axiom R3 : WatersDaily ∨ ¬ HeavyRains
axiom R4 : Strong → Bountiful
axiom R5 : Bountiful → GoodProfit
axiom R6 : WatersDaily → Waters

theorem profit_if_monsoon : Monsoon → GoodProfit := by
  intro hM
  have hH : HeavyRains := R2 hM
  have nnHR : ¬¬HeavyRains := by
    intro nhr
    exact nhr hH
  have hD : WatersDaily := by
    apply Or.elim R3
    · intro wd
      exact wd
    · intro nhr
      exact False.elim (nnHR nhr)
  have hW : Waters := R6 hD
  have hS : Strong := R1 hW
  have hB : Bountiful := R4 hS
  exact R5 hB
</lean>

ANSWER: Yes
