<lean>
axiom Rain : Prop
axiom SoilSaturated : Prop
axiom Watered : Prop
axiom Bloom : Prop
axiom Fragrance : Prop
axiom Butterflies : Prop

axiom R1 : Watered → Bloom
axiom R2 : Rain → SoilSaturated
axiom R3 : Bloom → Fragrance
axiom R4 : Fragrance → Butterflies
axiom R5 : Watered ∨ ¬SoilSaturated

theorem heavy_rain_attracts_butterflies : Rain → Butterflies := by
  intro hr
  have s : SoilSaturated := R2 hr
  cases R5 with
  | inl hw =>
    exact R4 (R3 (R1 hw))
  | inr ns =>
    exact False.elim (ns s)
</lean>

ANSWER: Yes
