<lean>
axiom obj : Type
axiom John : obj

axiom FinishEarly : obj → Prop
axiom LongLunch : obj → Prop
axiom Recharged : obj → Prop
axiom ProductiveAfternoon : obj → Prop
axiom BringTreats : obj → Prop
axiom Birthday : Prop

-- If someone finishes early, they can take a long lunch.
axiom A1 : ∀ x : obj, FinishEarly x → LongLunch x
-- If it's a birthday, people will bring in treats.
axiom A2 : Birthday → ∀ x : obj, BringTreats x
-- Either John finishes early or it's a birthday (or both).
axiom A3 : FinishEarly John ∨ Birthday
-- If someone takes a long lunch, they'll come back recharged.
axiom A4 : ∀ x : obj, LongLunch x → Recharged x
-- If they come back recharged, they'll have a very productive afternoon.
axiom A5 : ∀ x : obj, Recharged x → ProductiveAfternoon x

theorem john_productive_from_not_bring : (¬ BringTreats John) → ProductiveAfternoon John := by
  intro hnb
  have hb2 : Birthday → BringTreats John := by
    intro hb
    have hall : ∀ x : obj, BringTreats x := A2 hb
    exact hall John
  have hnotB : ¬ Birthday := by
    intro hb
    exact hnb (hb2 hb)
  have hE : FinishEarly John := by
    cases A3 with
    | inl hE => exact hE
    | inr hB =>
      exact False.elim (hnotB hB)
  have hL : LongLunch John := A1 John hE
  have hR : Recharged John := A4 John hL
  exact A5 John hR
</lean>

ANSWER: Yes
