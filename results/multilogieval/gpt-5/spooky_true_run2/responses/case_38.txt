-- Premises
-- If the car had gas, then they could drive long distances.
axiom P1 : Gas → DriveLong Person
-- If they see the sights, then they are taking a scenic route.
axiom P2 : SeeSights Person → Scenic Person
-- Either the car had gas, or they were not taking the scenic route, or the car had gas and they were not taking the scenic route.
axiom P3 : Gas ∨ ¬Scenic Person ∨ (Gas ∧ ¬Scenic Person)
-- If someone loves road trips, they take the scenic route to see the sights.
axiom P4 : ∀ x : obj, LovesRoadTrips x → Scenic x ∧ SeeSights x

-- Consequences we can derive:
theorem not_gas_of_not_drive : ¬DriveLong Person → ¬Gas := by
  intro hnd
  intro hgas
  have : DriveLong Person := P1 hgas
  exact hnd this

  intro hnd
  have hng : ¬Gas := not_gas_of_not_drive hnd
  have hdisj := P3
  cases hdisj with
  | inl h1 =>
    cases h1 with
    | inl hgas =>
      exact False.elim (hng hgas)
    | inr hnotSc =>
      exact hnotSc
  | inr h2 =>
    exact h2.right
