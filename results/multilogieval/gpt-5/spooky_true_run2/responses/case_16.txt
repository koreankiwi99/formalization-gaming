<lean>
axiom obj : Type

-- Individuals
axiom Sarah : obj

-- Predicates
axiom Practice : obj → Prop           -- practiced guitar (regularly / scales & chords daily)
axiom Listen : obj → Prop             -- listens attentively to music
axiom Technique : obj → Prop          -- technique improved
axiom LearnFast : obj → Prop          -- able to learn songs faster

-- Rules learned
axiom R1 : ∀ x : obj, Practice x → Technique x
axiom R2 : ∀ x : obj, Technique x → LearnFast x

-- Either practiced or listened (or both)
axiom Disj : Practice Sarah ∨ Listen Sarah

-- Goal: From "Sarah is not able to learn songs faster", show she listens attentively to music.
theorem sarah_listens : (¬ LearnFast Sarah) → Listen Sarah := by
  intro h_not_fast
  -- From practicing, technique improves, hence learns faster
  have h_prac_to_fast : Practice Sarah → LearnFast Sarah := by
    intro h_prac
    have h_tech : Technique Sarah := R1 Sarah h_prac
    exact R2 Sarah h_tech
  -- Therefore, if she cannot learn faster, she did not practice
  have h_not_prac : ¬ Practice Sarah := by
    intro h_prac
    exact h_not_fast (h_prac_to_fast h_prac)
  -- Use the disjunction: Practice Sarah ∨ Listen Sarah
  cases Disj with
  | inl h_prac =>
      exact False.elim (h_not_prac h_prac)
  | inr h_listen =>
      exact h_listen
</lean>

ANSWER: Yes
