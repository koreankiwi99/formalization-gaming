axiom A3 : ∀ x : obj, ¬ Healthy x ∨ ¬ Nutrients x

  have hN : Nutrients Bob := A2 Bob hrot
  have hdisj : ¬ Healthy Bob ∨ ¬ Nutrients Bob := A3 Bob
  cases hdisj with
  | inl hnH =>
    intro hone
    have hor : Organic Bob := A4 Bob hone
    have hH : Healthy Bob := A1 Bob hor
    exact hnH hH
  | inr hnN =>
    have contra : False := hnN hN
    intro hone
