axiom LearnedStrumming : obj → Prop
axiom CanPlaySimple : obj → Prop
axiom CanEntertain : obj → Prop
axiom TookLessons : obj → Prop
axiom KnowsTheory : obj → Prop

  ∀ x : obj, CanPlaySimple x → CanEntertain x

  exact not_simple_Emma h_simple

  intro h
