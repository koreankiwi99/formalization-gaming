<lean>
axiom Sunny : Prop
axiom KidsAtPark : Prop
axiom TeamPracticedDaily : Prop
axiom TeamPrepared : Prop
axiom TeamConfident : Prop
axiom TeamHoldHeadUpHigh : Prop

axiom A1 : TeamPracticedDaily → TeamPrepared
axiom A2 : Sunny → KidsAtPark
axiom A3 : TeamPracticedDaily ∨ ¬ KidsAtPark
axiom A4 : TeamPrepared → TeamConfident
axiom A5 : TeamConfident → TeamHoldHeadUpHigh

theorem team_heads_up_if_sunny : Sunny → TeamHoldHeadUpHigh := by
  intro s
  have k : KidsAtPark := A2 s
  have th : TeamHoldHeadUpHigh := Or.elim A3
    (fun hp => A5 (A4 (A1 hp)))
    (fun hnk => False.elim (hnk k))
  exact th
</lean>

ANSWER: Yes
