axiom obj : Type

axiom Amy : obj

axiom PracticeDaily : obj → Prop
axiom MuscleMemory : obj → Prop
axiom PlaySimpleSongs : obj → Prop
axiom FocusedOnLearningTheory : obj → Prop
axiom LearnedMusicTheory : obj → Prop
axiom KnowledgeSongStructure : obj → Prop
axiom AnalyzeCompositions : obj → Prop

axiom A1 : ∀ x : obj, PracticeDaily x → MuscleMemory x
axiom A2 : ∀ x : obj, MuscleMemory x → PlaySimpleSongs x
axiom A3 : ∀ x : obj, LearnedMusicTheory x → KnowledgeSongStructure x
axiom A4 : ∀ x : obj, KnowledgeSongStructure x → AnalyzeCompositions x

-- Either practicing daily or focused on learning music theory (or both)
axiom D : ∀ x : obj, PracticeDaily x ∨ FocusedOnLearningTheory x

theorem practice_to_play : ∀ x : obj, PracticeDaily x → PlaySimpleSongs x := by
  intro x
  intro hpd
  have hmm : MuscleMemory x := A1 x hpd
  exact A2 x hmm

theorem not_play_implies_not_practice : ∀ x : obj, ¬ PlaySimpleSongs x → ¬ PracticeDaily x := by
  intro x
  intro hnp
  intro hpd
  apply hnp
  exact practice_to_play x hpd

theorem amy_focused_if_not_play : ¬ PlaySimpleSongs Amy → FocusedOnLearningTheory Amy := by
  intro hnp
  have hdisj : PracticeDaily Amy ∨ FocusedOnLearningTheory Amy := D Amy
  cases hdisj with
  | inl hpd =>
      have hps : PlaySimpleSongs Amy := practice_to_play Amy hpd
      exact False.elim (hnp hps)
  | inr hfoc =>
      exact hfoc
