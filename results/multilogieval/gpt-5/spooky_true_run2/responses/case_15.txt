<lean>
axiom obj : Type
axiom Student : obj

axiom Study : obj → Prop
axiom Excel : obj → Prop
axiom Participate : obj → Prop
axiom Leader : obj → Prop
axiom Prefect : obj → Prop

-- If someone studies diligently, they will excel academically.
axiom A1 : ∀ x : obj, Study x → Excel x
-- If someone participates actively, they will be a leader.
axiom A2 : ∀ x : obj, Participate x → Leader x
-- The student either studies hard or participates actively.
axiom A3 : Study Student ∨ Participate Student
-- If someone is a leader, they will be selected as class prefect.
axiom A4 : ∀ x : obj, Leader x → Prefect x
-- The student did not excel academically.
axiom A5 : ¬ Excel Student

theorem student_chosen_prefect : Prefect Student := by
  -- From ¬Excel and (Study → Excel), infer ¬Study (modus tollens).
  have h_not_study : ¬ Study Student := by
    exact mt (A1 Student) A5
  -- From (Study ∨ Participate) and ¬Study, infer Participate.
  have h_participate : Participate Student := by
    cases A3 with
    | inl hStudy =>
      exact False.elim (h_not_study hStudy)
    | inr hParticipate =>
      exact hParticipate
  -- From Participate, infer Leader, then Prefect.
  have h_leader : Leader Student := A2 Student h_participate
  exact A4 Student h_leader
</lean>

ANSWER: Yes
