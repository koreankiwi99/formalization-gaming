axiom Jane : obj
axiom Hurt : obj → Prop
axiom SeeDoc : obj → Prop
axiom Rest : obj → Prop

-- If someone gets injured/hurt, they should see a doctor.
axiom H_to_D : ∀ x : obj, Hurt x → SeeDoc x
-- If someone gets hurt, they should rest the affected area.
axiom H_to_R : ∀ x : obj, Hurt x → Rest x

-- Either Jane did not see a doctor or she did not rest her injury.
axiom disj : ¬ SeeDoc Jane ∨ ¬ Rest Jane

-- From the disjunction and the two rules, we can conclude Jane did not get hurt.
theorem jane_not_hurt : ¬ Hurt Jane := by
  apply Or.elim disj
  · intro hNotSee
    intro hHurt
    have hSee : SeeDoc Jane := H_to_D Jane hHurt
    exact hNotSee hSee
  · intro hNotRest
    intro hHurt
    have hR : Rest Jane := H_to_R Jane hHurt
    exact hNotRest hR
