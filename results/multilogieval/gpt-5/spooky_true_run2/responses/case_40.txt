<lean>
axiom obj : Type
axiom James : obj

axiom Start : obj → Prop
axiom Motivated : obj → Prop
axiom Follow : obj → Prop
axiom GymClosed : Prop
axiom Frustrated : obj → Prop
axiom Snacks : obj → Prop

-- If someone starts a new workout routine, they will be motivated.
axiom A1 : ∀ x : obj, Start x → Motivated x
-- If the gym closes, then they can't follow their routine.
axiom A2 : ∀ x : obj, GymClosed → ¬ Follow x
-- Either James started a new workout, or the gym was closed (or both).
axiom A3 : Start James ∨ GymClosed
-- If they couldn't follow their routine, they would feel frustrated.
axiom A4 : ∀ x : obj, ¬ Follow x → Frustrated x
-- If they felt frustrated, they might indulge in unhealthy snacks.
axiom A5 : ∀ x : obj, Frustrated x → Snacks x

-- Additional given: James was not motivated to workout.
axiom A6 : ¬ Motivated James

theorem james_indulged : Snacks James :=
by
  -- From not motivated and (Start → Motivated), derive ¬Start James
  have hnotstart : ¬ Start James := by
    intro hs
    have hm : Motivated James := A1 James hs
    exact A6 hm
  -- From (Start James ∨ GymClosed) and ¬Start James, conclude GymClosed
  have hg : GymClosed := by
    cases A3 with
    | inl hs => exact False.elim (hnotstart hs)
    | inr hgc => exact hgc
  -- Gym closed implies James couldn't follow his routine
  have hnf : ¬ Follow James := A2 James hg
  -- Couldn't follow implies frustrated
  have hf : Frustrated James := A4 James hnf
  -- Frustrated implies indulged in unhealthy snacks
  exact A5 James hf
</lean>

ANSWER: Yes
