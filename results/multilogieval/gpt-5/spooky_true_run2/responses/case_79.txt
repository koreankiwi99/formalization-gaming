axiom Person : Type

axiom WorkedOutHard : Person → Prop
axiom Sore : Person → Prop
axiom HotShower : Person → Prop
axiom Relaxed : Person → Prop
axiom Cold : Prop
axiom Rain : Prop

axiom hard_implies_sore : ∀ x : Person, WorkedOutHard x → Sore x
axiom hot_implies_relax : ∀ x : Person, HotShower x → Relaxed x
axiom hard_or_not_relax : ∀ x : Person, WorkedOutHard x ∨ ¬ Relaxed x
axiom cold_implies_hot : ∀ x : Person, Cold → HotShower x
axiom rain_implies_cold : Rain → Cold

theorem rain_implies_sore (x : Person) : Rain → Sore x := by
  intro hRain
  have hCold : Cold := rain_implies_cold hRain
  have hHot : HotShower x := (cold_implies_hot x) hCold
  have hRelax : Relaxed x := (hot_implies_relax x) hHot
  have hDisj : WorkedOutHard x ∨ ¬ Relaxed x := hard_or_not_relax x
  cases hDisj with
  | inl hHard =>
    exact (hard_implies_sore x) hHard
  | inr hNotRelax =>
    have hFalse : False := hNotRelax hRelax
    exact False.elim hFalse

theorem not_rain_of_not_sore : ¬ Sore John → ¬ Rain := by
  intro hNotSore
  intro hRain
  have hSoreJohn : Sore John := rain_implies_sore John hRain
  exact hNotSore hSoreJohn
