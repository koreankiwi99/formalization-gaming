axiom TastesAmazing : obj → Prop

-- We treat this as the rule available for reasoning.

-- Given: If the food tastes amazing, positive reviews spread on social media.
axiom R_amazing_to_reviews : ∀ x : obj, TastesAmazing x → ReviewsSpread x

-- From the given premises we can derive that reviews spread,
-- but there is no provided link from reviews to long lines.
theorem reviews_spread_new_restaurant : ReviewsSpread NewRestaurant := by
  have h_amazing : TastesAmazing NewRestaurant := R_fresh_to_amazing NewRestaurant H_uses
  exact R_amazing_to_reviews NewRestaurant h_amazing
