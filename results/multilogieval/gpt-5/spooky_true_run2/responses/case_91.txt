<lean>
axiom Party : Prop
axiom Drums : Prop
axiom Noise : Prop
axiom Complain : Prop
axiom Police : Prop

axiom R1 : Party → Noise
axiom R2 : Drums → Noise
axiom R3 : Party ∨ Drums
axiom R4 : (Party ∧ Noise) → Complain
axiom R5 : Complain → Police

theorem no_noise_police : (¬Noise) → Police :=
  fun hnn => by
    have n : Noise := Or.elim R3 (fun hp => R1 hp) (fun hd => R2 hd)
    exact False.elim (hnn n)
</lean>

ANSWER: Yes
