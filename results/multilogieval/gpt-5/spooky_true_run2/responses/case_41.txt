<lean>
axiom WaterDaily : Prop
axiom GrowWell : Prop
axiom Drought : Prop
axiom Restrictions : Prop
axiom NotEnoughWater : Prop
axiom WiltDie : Prop

axiom R1 : WaterDaily → GrowWell
axiom R2 : Drought → Restrictions
axiom R3 : WaterDaily ∨ Drought
axiom R4 : Restrictions → NotEnoughWater
axiom R5 : NotEnoughWater → WiltDie

theorem priya_plants_wilt : ¬ GrowWell → WiltDie := by
  intro hNotGrow
  cases R3 with
  | inl hWD =>
      have gw : GrowWell := R1 hWD
      have hFalse : False := hNotGrow gw
      exact False.elim hFalse
  | inr hD =>
      have res : Restrictions := R2 hD
      have few : NotEnoughWater := R4 res
      exact R5 few
</lean>

ANSWER: Yes
