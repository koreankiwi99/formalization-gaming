<lean>
axiom LostImportantGame : Prop
axiom LostBigMatch : Prop
axiom DisappointedPlayers : Prop
axiom DemoralizedPlayers : Prop

-- Premises from the context
axiom R1 : LostImportantGame → DisappointedPlayers        -- Losing an important game ⇒ players disappointed
axiom R2 : LostBigMatch → DemoralizedPlayers              -- Losing a big match ⇒ demoralizing
axiom NotDisappointed : ¬ DisappointedPlayers             -- Given: players are not disappointed
axiom NotDemoralized : ¬ DemoralizedPlayers               -- Given: players are not demoralized

def SignificantLoss : Prop := LostImportantGame ∨ LostBigMatch

-- From the given negations, we can refute any significant loss
theorem no_significant_loss : ¬ SignificantLoss := by
  intro h
  cases h with
  | inl hImp =>
      have hd : DisappointedPlayers := R1 hImp
      exact NotDisappointed hd
  | inr hBig =>
      have hm : DemoralizedPlayers := R2 hBig
      exact NotDemoralized hm
</lean>
