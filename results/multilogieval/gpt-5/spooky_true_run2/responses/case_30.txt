axiom R_diet_strength : ∀ x : obj, HealthyDiet x → IncreasedStrength x

axiom R_ath_disj : ∀ x : obj, Athlete x → (LackingSkill x ∨ DeficientStrength x)

  exact R_diet_strength Mark MarkHealthy

theorem mark_not_def_strength : ¬ DeficientStrength Mark := by
  have hinc : IncreasedStrength Mark := R_diet_strength Mark MarkHealthy
  exact (R_strength_not_def Mark) hinc

theorem mark_lacking_skill : LackingSkill Mark := by
  have hdisj : LackingSkill Mark ∨ DeficientStrength Mark := R_ath_disj Mark MarkAthlete
  have hnotdef : ¬ DeficientStrength Mark := mark_not_def_strength
  cases hdisj with
  | inl h => exact h
  | inr hdef => exact False.elim (hnotdef hdef)
