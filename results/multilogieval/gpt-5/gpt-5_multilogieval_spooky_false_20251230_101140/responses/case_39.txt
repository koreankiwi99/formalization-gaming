axiom Studied : obj → Prop

-- If it was Monday, then there would be a quiz.

-- If someone does well on the quiz, then they studied hard.
axiom DoWell_to_Studied : ∀ x : obj, DoWell x → Studied x

-- Either it was Monday, or they did not study hard, or it was Monday and they did not study hard.
-- Right-associative: Monday ∨ (¬Studied Alex ∨ (Monday ∧ ¬Studied Alex))
axiom sure : Monday ∨ ¬Studied Alex ∨ (Monday ∧ ¬Studied Alex)

-- Someone who is good at math does well on quizzes.
axiom Good_to_DoWell : ∀ x : obj, Good x → DoWell x

-- From the premises we can derive: If there was no quiz, then Alex is not good at math.
theorem no_quiz_implies_not_good : ¬Quiz → ¬Good Alex := by
  intro hNoQuiz
  -- From Monday → Quiz, get ¬Monday
  have hNotMon : ¬Monday := by
    intro hMon
    exact hNoQuiz (MtoQ hMon)
  -- From the disjunction and ¬Monday, deduce ¬Studied Alex
  have hNotStud : ¬Studied Alex := by
    cases sure with
    | inl hMon =>
        exact False.elim (hNotMon hMon)
    | inr hRight =>
        cases hRight with
        | inl hNS => exact hNS
        | inr hMonAndNS =>
            exact And.right hMonAndNS
  -- From DoWell → Studied and ¬Studied, derive ¬DoWell
  have hNotDoWell : ¬DoWell Alex := by
    intro hDoWell
    have hStud : Studied Alex := DoWell_to_Studied Alex hDoWell
    exact hNotStud hStud
  -- From Good → DoWell and ¬DoWell, derive ¬Good
  intro hGood
  exact hNotDoWell (Good_to_DoWell Alex hGood)
