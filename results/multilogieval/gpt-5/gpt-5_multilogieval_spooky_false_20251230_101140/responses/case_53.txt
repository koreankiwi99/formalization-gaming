-- If their technique improves, their artwork stands out as high quality.

axiom D : Practice Mary ∨ ArtClass Mary
-- If someone takes an art class, they learn new methods and mediums.
axiom R3 : ∀ x : obj, ArtClass x → LearnedMethods x

-- Statement to evaluate includes: Mary's artwork did not stand out as high quality.
axiom NotHigh : ¬HighQuality Mary

-- From the above, we can derive that Mary learned new methods and mediums.
theorem mary_learned : LearnedMethods Mary := by
  have not_practice : ¬Practice Mary := fun hp => NotHigh (R2 Mary (R1 Mary hp))
  apply Or.elim D
  · intro hp
    exact False.elim (not_practice hp)
  · intro hc
    exact R3 Mary hc
