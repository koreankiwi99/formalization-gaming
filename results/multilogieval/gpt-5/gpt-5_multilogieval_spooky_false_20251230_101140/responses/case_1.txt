axiom P2 : ∀ t : Team, ProtectPaint t → LimitInside t
axiom P3 : StrongPerimeter Lakers ∨ ProtectPaint Lakers

-- From the premises we can prove the statement:
-- If the Lakers did not limit inside scoring, then they forced many turnovers.
theorem lakers_turnovers_from_not_limit : ¬LimitInside Lakers → ForceTurnovers Lakers := by
  intro hNotLimit
  have hNotProtect : ¬ProtectPaint Lakers := by
    intro hProtect
    exact hNotLimit (P2 Lakers hProtect)
  have hStrong : StrongPerimeter Lakers := by
    cases P3 with
    | inl hS => exact hS
    | inr hP => exact False.elim (hNotProtect hP)
  exact P1 Lakers hStrong
