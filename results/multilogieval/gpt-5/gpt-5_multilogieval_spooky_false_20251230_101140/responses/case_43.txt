axiom James : obj

axiom TakesSubwayToWork : obj → Prop
axiom TakesBusDowntown : obj → Prop
axiom OnTime : obj → Prop
axiom SmoothCommute : obj → Prop

-- Global conditions about the transit system
axiom SubwayOnSchedule : Prop
axiom BusHasRoom : Prop

-- Premises from the text
-- If the subway is running on schedule, then someone will take it to work.
axiom P1 : SubwayOnSchedule → ∃ x : obj, TakesSubwayToWork x

-- If the bus has room, then someone can catch a ride downtown.
axiom P2 : BusHasRoom → ∃ x : obj, TakesBusDowntown x

-- Whether it's the subway or the bus, someone's commute will be smooth sailing.
-- Interpreted as: if there exists someone who takes the subway or someone who takes the bus,
-- then there exists someone whose commute is smooth.
axiom P3 : (∃ x : obj, TakesSubwayToWork x) ∨ (∃ x : obj, TakesBusDowntown x) → ∃ x : obj, SmoothCommute x

-- If someone takes the subway, they'll get to work on time.
axiom P4 : ∀ x : obj, TakesSubwayToWork x → OnTime x

-- Assumption for the query: James did not take the bus downtown.
axiom HJamesNotBus : ¬TakesBusDowntown James

-- A useful derived fact (doesn't settle the query): if James took the subway, he'd be on time.
lemma james_on_time_if_subway : TakesSubwayToWork James → OnTime James := by
  intro h
  exact P4 James h
