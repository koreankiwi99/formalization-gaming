<lean>
axiom Rain : Prop
axiom Saturated : Prop
axiom Watered : Prop
axiom Bloom : Prop
axiom Fragrance : Prop
axiom AttractButterflies : Prop

-- Premises
axiom P1 : Watered → Bloom
axiom P2 : Rain → Saturated
axiom P3 : Watered ∨ ¬Saturated
axiom P4 : Bloom → Fragrance
axiom P5 : Fragrance → AttractButterflies

-- From P3 (Watered ∨ ¬Saturated) we can derive Saturated → Watered
theorem saturated_implies_watered : Saturated → Watered := by
  intro s
  -- Use Or.resolve_right to eliminate the right disjunct ¬Saturated via ¬¬Saturated from s
  exact Or.resolve_right P3 (by intro ns; exact ns s)

-- Therefore, heavy rain implies attraction of butterflies
theorem rain_implies_attract : Rain → AttractButterflies := by
  intro r
  have s : Saturated := P2 r
  have w : Watered := saturated_implies_watered s
  have b : Bloom := P1 w
  have f : Fragrance := P4 b
  exact P5 f
