<lean>
axiom Person : Type
axiom John : Person

axiom Sore : Person → Prop
axiom Relax : Person → Prop
axiom HotShower : Person → Prop
axiom Hard : Person → Prop

axiom Cold : Prop
axiom Rain : Prop

-- If someone worked out hard, they will be sore.
axiom A1 : ∀ x : Person, Hard x → Sore x
-- If someone takes a hot shower, their muscles will relax.
axiom A2 : ∀ x : Person, HotShower x → Relax x
-- Either they worked out hard, or their muscles did not relax. (interpreted universally)
axiom A3 : ∀ x : Person, Hard x ∨ ¬Relax x
-- If the weather is cold, they take a hot shower. (everyone)
axiom A4 : ∀ x : Person, Cold → HotShower x
-- If it rains, then the weather is cold.
axiom A5 : Rain → Cold

theorem rain_relax_john : Rain → Relax John := by
  intro r
  have c : Cold := A5 r
  have hs : HotShower John := A4 John c
  exact A2 John hs

theorem rain_sore_john : Rain → Sore John := by
  intro r
  have rj : Relax John := rain_relax_john r
  have d : Hard John ∨ ¬Relax John := A3 John
  cases d with
  | inl hj =>
      exact A1 John hj
  | inr nrel =>
      have f : False := nrel rj
      exact False.elim f
