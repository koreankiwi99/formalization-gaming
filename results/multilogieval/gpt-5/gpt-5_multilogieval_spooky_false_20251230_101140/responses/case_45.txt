axiom Skill : obj → Prop
axiom Redo : obj → Prop
axiom Ideas : obj → Prop

axiom P1 : ∀ x : obj, Watch x → Skill x
axiom P2 : ∀ x : obj, Skill x → Redo x
axiom P3 : ∀ x : obj, Watch x ∨ ReadBlog x
axiom P4 : ∀ x : obj, ReadBlog x → Ideas x

theorem matt_gets_ideas_from_not_redo : ¬Redo Matt → Ideas Matt := by
  intro notRedo
  have w_to_r : Watch Matt → Redo Matt := by
    intro w
    have s : Skill Matt := P1 Matt w
    exact P2 Matt s
  have notWatch : ¬Watch Matt := by
    intro w
    have r : Redo Matt := w_to_r w
    exact notRedo r
  have disj : Watch Matt ∨ ReadBlog Matt := P3 Matt
  have read : ReadBlog Matt := by
    cases disj with
    | inl w => exact False.elim (notWatch w)
    | inr rb => exact rb
  exact P4 Matt read
