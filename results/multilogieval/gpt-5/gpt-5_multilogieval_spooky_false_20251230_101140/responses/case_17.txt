<lean>
axiom obj : Type
axiom James : obj

axiom Practice : obj → Prop
axiom Prepared : obj → Prop
axiom Confident : obj → Prop
axiom JoinedClub : obj → Prop

-- If someone practices thoroughly, they feel prepared.
axiom P1 : ∀ x : obj, Practice x → Prepared x
-- If someone feels prepared, they can speak confidently.
axiom P2 : ∀ x : obj, Prepared x → Confident x
-- Either they practiced thoroughly, or they joined a public speaking club (or both).
axiom P3 : ∀ x : obj, Practice x ∨ JoinedClub x

-- James cannot speak confidently in public.
axiom H_not_conf : ¬ Confident James

-- From the premises and H_not_conf, we can derive that James joined a public speaking club.
theorem james_joined_club : JoinedClub James := by
  have h_disj : Practice James ∨ JoinedClub James := P3 James
  have h_not_practice : ¬ Practice James := by
    intro hP
    have hPrep : Prepared James := P1 James hP
    have hConf : Confident James := P2 James hPrep
    exact H_not_conf hConf
  cases h_disj with
  | inl hP =>
    have contra : False := h_not_practice hP
    exact False.elim contra
  | inr hJ =>
    exact hJ
