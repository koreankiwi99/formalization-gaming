<lean>
axiom Person : Type
axiom Emma : Person

axiom StudyHard : Person → Prop
axiom EnoughSleepBeforeExam : Person → Prop
axiom PerformWellOnExam : Person → Prop
axiom AceExam : Person → Prop

axiom belief :
  ∀ x : Person, StudyHard x → (EnoughSleepBeforeExam x → (PerformWellOnExam x ∧ AceExam x))

axiom sleep_Emma : EnoughSleepBeforeExam Emma
axiom not_ace_Emma : ¬ AceExam Emma

theorem emma_not_studyhard : ¬ StudyHard Emma :=
by
  intro hStudy
  have hPair : PerformWellOnExam Emma ∧ AceExam Emma := (belief Emma hStudy) sleep_Emma
  have hAce : AceExam Emma := And.right hPair
  exact not_ace_Emma hAce
</lean>

ANSWER: No
