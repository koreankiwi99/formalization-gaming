axiom obj : Type
axiom Max : obj

axiom OnTimePackage : Prop
axiom SpecialOrderReady : Prop

axiom CanStart : obj → Prop
axiom Starts : obj → Prop
axiom CanPickUp : obj → Prop
axiom LearnsSkills : obj → Prop

-- If the package arrives on time, people will be able to start a new project.
axiom A1 : OnTimePackage → ∀ x : obj, CanStart x
-- If the special order is in stock, they can pick up the materials they need.
axiom A2 : SpecialOrderReady → ∀ x : obj, CanPickUp x
-- Either the package will arrive on time or the special order will be ready.
axiom A3 : OnTimePackage ∨ SpecialOrderReady
-- If someone starts his new project, they will learn some new skills.
axiom A4 : ∀ x : obj, Starts x → LearnsSkills x

-- Max was unable to pick up the materials he needed.
axiom Max_cannot_pickup : ¬ CanPickUp Max

-- From Max's inability to pick up materials, the special order is not ready.
theorem not_special_order_ready : ¬ SpecialOrderReady :=
by
  intro hSR
  have hPickup : CanPickUp Max := (A2 hSR) Max
  exact Max_cannot_pickup hPickup

-- Therefore, the package arrived on time.
theorem on_time_package : OnTimePackage :=
by
  cases A3 with
  | inl hP => exact hP
  | inr hSR => exact False.elim (not_special_order_ready hSR)

-- Hence, Max can start a new project (ability), but this does not imply he actually starts.
theorem Max_can_start : CanStart Max := (A1 on_time_package) Max
