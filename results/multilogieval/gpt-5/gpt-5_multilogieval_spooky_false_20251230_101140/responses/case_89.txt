<lean>
axiom Monsoon : Prop
axiom HeavyRains : Prop
axiom Waters : Prop
axiom GrowStrong : Prop
axiom Bountiful : Prop
axiom Profit : Prop

-- If the farmer waters his crops, they will grow healthy and strong.
axiom H1 : Waters → GrowStrong
-- If it is monsoon season, there will be heavy rains.
axiom H2 : Monsoon → HeavyRains
-- Either the farmer waters his crops (daily), or there are no heavy monsoon rains this season.
axiom H3 : Waters ∨ ¬HeavyRains
-- When crops grow strong, the harvest will be bountiful.
axiom H4 : GrowStrong → Bountiful
-- If the harvest is bountiful, the farmer will make a good profit.
axiom H5 : Bountiful → Profit

-- From the premises we can derive: If it is monsoon season, the farmer will make a good profit.
theorem profit_if_monsoon : Monsoon → Profit :=
by
  intro mon
  have hr : HeavyRains := H2 mon
  have w : Waters :=
    Or.elim H3
      (fun hw => hw)
      (fun nhr => False.elim (nhr hr))
  have gs : GrowStrong := H1 w
  have b : Bountiful := H4 gs
  exact H5 b
