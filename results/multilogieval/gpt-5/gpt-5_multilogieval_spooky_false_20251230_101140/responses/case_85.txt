axiom obj : Type
axiom John : obj

axiom Bloom : obj → Prop
axiom Attract : obj → Prop
axiom UsesOrganic : obj → Prop
axiom SoilImproves : obj → Prop
axiom HealthyPlants : obj → Prop
axiom SoilHighQuality : obj → Prop

-- If someone waters adequately, plants bloom.
axiom R1 : ∀ x : obj, Watered x → Bloom x
-- If plants bloom brightly, it attracts more bees and butterflies.
axiom R2 : ∀ x : obj, Bloom x → Attract x
-- Either someone waters adequately or uses organic fertilizers (or both) — for John.
axiom R3 : Watered John ∨ UsesOrganic John
-- If someone uses organic fertilizers, the soil quality improves.
axiom R4 : ∀ x : obj, UsesOrganic x → SoilImproves x
-- If the soil quality improves, the plants grow healthier.
axiom R5 : ∀ x : obj, SoilImproves x → HealthyPlants x

-- Derived: Watered implies Attract.
theorem watered_implies_attract : ∀ x : obj, Watered x → Attract x :=
by
  intro x
  intro hw
  have hb : Bloom x := R1 x hw
  exact R2 x hb

-- From not Attract, deduce UsesOrganic for John (since Watered would imply Attract).
theorem not_attract_implies_usesorganic_John : ¬Attract John → UsesOrganic John :=
by
  intro hna
  cases R3 with
  | inl hw =>
      have ha : Attract John := watered_implies_attract John hw
      exact False.elim (hna ha)
  | inr ho =>
      exact ho

-- Hence from not Attract, soil quality improves for John.
theorem not_attract_implies_soil_improves_John : ¬Attract John → SoilImproves John :=
by
  intro hna
  have ho : UsesOrganic John := not_attract_implies_usesorganic_John hna
  exact R4 John ho
