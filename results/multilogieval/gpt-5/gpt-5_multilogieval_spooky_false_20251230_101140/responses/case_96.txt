axiom Exam : Type

axiom StudiedHard : obj → Exam → Prop
axiom Confident : obj → Exam → Prop
axiom DidWell : obj → Exam → Prop
axiom Satisfied : obj → Exam → Prop

axiom SevereWeather : Prop
axiom CancelClass : Prop

-- If someone studies hard for an exam, they will feel confident
axiom A1 : ∀ p : obj, ∀ e : Exam, StudiedHard p e → Confident p e

-- If there is severe weather, schools cancel class (interpreting "may cancel" as a simple implication)
axiom A2 : SevereWeather → CancelClass

-- Either Alex studied for biology, or class was not canceled
axiom A3 : StudiedHard Alex Biology ∨ ¬CancelClass

-- If confident, then do well
axiom A4 : ∀ p : obj, ∀ e : Exam, Confident p e → DidWell p e

-- If do well, then satisfied
axiom A5 : ∀ p : obj, ∀ e : Exam, DidWell p e → Satisfied p e

-- A derivable consequence (about biology), not about history
theorem satisfied_biology_if_severe : SevereWeather → Satisfied Alex Biology := by
  intro sw
  have cc : CancelClass := A2 sw
  have h := A3
  cases h with
  | inl sh =>
    have conf : Confident Alex Biology := A1 Alex Biology sh
    have well : DidWell Alex Biology := A4 Alex Biology conf
    exact A5 Alex Biology well
  | inr ncc =>
    have f : False := ncc cc
    exact False.elim f
