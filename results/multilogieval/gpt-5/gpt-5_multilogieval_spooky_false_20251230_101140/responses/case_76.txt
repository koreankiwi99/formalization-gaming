<lean>
axiom obj : Type
axiom Alex : obj

axiom Relax : obj → Prop
axiom ClearMind : obj → Prop
axiom Finish : obj → Prop
axiom Organized : obj → Prop
axiom ToDoList : obj → Prop

axiom WeekendComing : Prop

-- If it was the weekend coming up, then they (Alex) would relax.
axiom W_relax : WeekendComing → Relax Alex
-- If someone finishes their tasks, then they have a clear mind.
axiom finish_clear : ∀ x : obj, Finish x → ClearMind x
-- Either the weekend was coming, or they (Alex) did not have a clear mind.
axiom weekend_or_not_clear : WeekendComing ∨ ¬ClearMind Alex
-- If someone is organized, then they finish their tasks.
axiom org_finish : ∀ x : obj, Organized x → Finish x
-- If they made a to-do list, then they are organized.
axiom todo_org : ∀ x : obj, ToDoList x → Organized x

-- Assumption from the statement: Alex does not relax this weekend.
axiom not_relax : ¬Relax Alex

theorem alex_not_todo : ¬ToDoList Alex := by
  intro htodo
  have horg : Organized Alex := todo_org Alex htodo
  have hfin : Finish Alex := org_finish Alex horg
  have hclr : ClearMind Alex := finish_clear Alex hfin
  have hwk : WeekendComing :=
    Or.elim weekend_or_not_clear
      (fun hw => hw)
      (fun hnc => False.elim (hnc hclr))
  have hrelax : Relax Alex := W_relax hwk
  exact not_relax hrelax
</lean>
