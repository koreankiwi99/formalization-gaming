axiom Lauren : obj

-- Predicates
axiom SprintWorkouts : obj → Prop
axiom SpeedImproved : obj → Prop
axiom RunFaster : obj → Prop
axiom TrainHardWeekly : obj → Prop
axiom TrainHardSprint : obj → Prop
axiom StretchedAfterRuns : obj → Prop
axiom RecoverySerious : obj → Prop
axiom FocusFlex : obj → Prop

-- Premises from the text
axiom P1 : ∀ x : obj, SprintWorkouts x → SpeedImproved x
axiom P2 : ∀ x : obj, SpeedImproved x → RunFaster x
-- The "Therefore" in the text: if they trained hard weekly, they improve race times
axiom P3 : ∀ x : obj, TrainHardWeekly x → RunFaster x
-- Either they trained hard at sprinting or they stretched after runs (or both)
axiom P4 : ∀ x : obj, TrainHardSprint x ∨ StretchedAfterRuns x
-- Stretching after runs implies taking recovery seriously
axiom P5 : ∀ x : obj, StretchedAfterRuns x → RecoverySerious x
-- Taking recovery seriously implies focusing on flexibility to prevent injury
axiom P6 : ∀ x : obj, RecoverySerious x → FocusFlex x
-- Bridge: training hard at sprinting implies training hard weekly
axiom P7 : ∀ x : obj, TrainHardSprint x → TrainHardWeekly x

-- From the premises we can derive the queried implication:
theorem lauren_focus_if_not_fast : (¬RunFaster Lauren) → FocusFlex Lauren := by
  intro hNotFast
  have hNotWeekly : ¬TrainHardWeekly Lauren := by
    intro hWeekly
    have hFast : RunFaster Lauren := by
      exact P3 Lauren hWeekly
    exact hNotFast hFast
  have hNotSprint : ¬TrainHardSprint Lauren := by
    intro hSprint
    have hWeekly : TrainHardWeekly Lauren := by
      exact P7 Lauren hSprint
    exact hNotWeekly hWeekly
  have hStretch : StretchedAfterRuns Lauren := by
    have hDisj := P4 Lauren
    cases hDisj with
    | inl hSprint =>
        have : False := hNotSprint hSprint
        exact False.elim this
    | inr hStretch => exact hStretch
  have hRec : RecoverySerious Lauren := by
    exact P5 Lauren hStretch
  have hFlex : FocusFlex Lauren := by
    exact P6 Lauren hRec
  exact hFlex
