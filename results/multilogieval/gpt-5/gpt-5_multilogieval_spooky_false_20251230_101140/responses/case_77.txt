<lean>
axiom Person : Type
axiom James : Person

axiom PracticedHard : Person → Prop
axiom PlayWell : Person → Prop
axiom PickedStart : Person → Prop
axiom Talented : Person → Prop
axiom NaturalAbility : Person → Prop
axiom TopYouth : Person → Prop

-- Premises
axiom P1 : ∀ x : Person, PracticedHard x → PlayWell x
axiom P2 : ∀ x : Person, PickedStart x → Talented x
axiom P3 : ∀ x : Person, PracticedHard x ∨ ¬Talented x
axiom P4 : ∀ x : Person, NaturalAbility x → PickedStart x
axiom P5 : ∀ x : Person, TopYouth x → NaturalAbility x

-- From "not play well" we can derive "not top youth"
theorem not_playwell_implies_not_top (j : Person) : ¬PlayWell j → ¬TopYouth j := by
  intro hNotPW
  intro hTop
  have hNat : NaturalAbility j := P5 j hTop
  have hPick : PickedStart j := P4 j hNat
  have hTal : Talented j := P2 j hPick
  have hNotPrac : ¬PracticedHard j := by
    intro hPrac
    exact hNotPW (P1 j hPrac)
  have hNotTal : ¬Talented j := by
    have disj := P3 j
    cases disj with
    | inl hPrac =>
        have f : False := hNotPrac hPrac
        exact False.elim f
    | inr hNt => exact hNt
  exact hNotTal hTal

-- This shows that if James did not play well, then the queried implication is false.
theorem conditional_refutation :
  ¬PlayWell James → ¬(¬PlayWell James → TopYouth James) := by
  intro hNotPW
  intro hImp
  have hTop : TopYouth James := hImp hNotPW
  have hNotTop : ¬TopYouth James := not_playwell_implies_not_top James hNotPW
  exact hNotTop hTop
