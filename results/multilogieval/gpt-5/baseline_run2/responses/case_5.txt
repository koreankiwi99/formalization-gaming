axiom obj : Type
axiom Mary : obj

axiom HomeOwner : obj → Prop
axiom LearnedBotany : obj → Prop
axiom UnderstoodNeeds : obj → Prop
axiom ThrivingPlants : obj → Prop
axiom SoughtAdvice : obj → Prop

-- For homeowners:
axiom R1 : ∀ x : obj, HomeOwner x → (LearnedBotany x → UnderstoodNeeds x)
axiom R2 : ∀ x : obj, HomeOwner x → (UnderstoodNeeds x → ThrivingPlants x)
axiom R3 : ∀ x : obj, HomeOwner x → (LearnedBotany x ∨ SoughtAdvice x)

-- Mary's plants did not thrive
axiom NT_Mary : ¬ ThrivingPlants Mary

-- If a homeowner's plants did not thrive, they must have sought advice
theorem sought_if_not_thriving_of_homeowner (x : obj) (hHO : HomeOwner x) (hNT : ¬ ThrivingPlants x) : SoughtAdvice x := by
  have hLtoU : LearnedBotany x → UnderstoodNeeds x := R1 x hHO
  have hUtoT : UnderstoodNeeds x → ThrivingPlants x := R2 x hHO
  have hLtoT : LearnedBotany x → ThrivingPlants x := fun hL => hUtoT (hLtoU hL)
  have hNotL : ¬ LearnedBotany x := fun hL => hNT (hLtoT hL)
  have hDisj : LearnedBotany x ∨ SoughtAdvice x := R3 x hHO
  exact Or.elim hDisj (fun hL => False.elim (hNotL hL)) (fun hS => hS)

-- Specialization to Mary, assuming she is a homeowner
theorem mary_sought_of_homeowner : HomeOwner Mary → SoughtAdvice Mary := by
  intro hHO
  exact sought_if_not_thriving_of_homeowner Mary hHO NT_Mary
