<lean>
axiom obj : Type
axiom Priya : obj

axiom Student : obj → Prop
axiom AttendsRegularly : obj → Prop
axiom RevisesSufficiently : obj → Prop
axiom PerformsWellOnExams : obj → Prop

-- General rule: for any student, if they attend regularly, then if they revise sufficiently, they perform well.
axiom Rule :
  ∀ x : obj, Student x → (AttendsRegularly x → (RevisesSufficiently x → PerformsWellOnExams x))

-- Facts about Priya
axiom PriyaStudent : Student Priya
axiom PriyaRevised : RevisesSufficiently Priya
axiom PriyaDidNotPerformWell : ¬ PerformsWellOnExams Priya

-- Conclusion: Priya did not attend lectures regularly
theorem priya_did_not_attend : ¬ AttendsRegularly Priya := by
  intro hAttend
  have hImp : RevisesSufficiently Priya → PerformsWellOnExams Priya :=
    (Rule Priya PriyaStudent) hAttend
  have hPerf : PerformsWellOnExams Priya := hImp PriyaRevised
  exact PriyaDidNotPerformWell hPerf
</lean>

ANSWER: No
