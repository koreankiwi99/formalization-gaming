<lean>
axiom Monsoon : Prop
axiom HeavyRains : Prop
axiom WaterDaily : Prop
axiom GrowStrong : Prop
axiom Bountiful : Prop
axiom Profit : Prop

axiom R1 : WaterDaily → GrowStrong
axiom R2 : Monsoon → HeavyRains
axiom R3 : WaterDaily ∨ ¬HeavyRains
axiom R4 : GrowStrong → Bountiful
axiom R5 : Bountiful → Profit

theorem monsoon_profit : Monsoon → Profit := by
  intro m
  have h : HeavyRains := R2 m
  have notNotH : ¬¬HeavyRains := by
    intro nh
    exact nh h
  have w : WaterDaily := Or.resolve_right R3 notNotH
  have g : GrowStrong := R1 w
  have b : Bountiful := R4 g
  exact R5 b
</lean>

ANSWER: Yes
