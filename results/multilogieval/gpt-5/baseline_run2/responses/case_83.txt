<lean>
axiom obj : Type
axiom Team : obj
axiom Kids : obj

axiom Sunny : Prop
axiom PracticeDaily : obj → Prop
axiom Prepared : obj → Prop
axiom Confident : obj → Prop
axiom HoldHeadHigh : obj → Prop
axiom PlayAtParkToday : obj → Prop

-- If someone practices every day, they will be prepared for the big game.
axiom R1 : ∀ x : obj, PracticeDaily x → Prepared x
-- If it is sunny outside, children will play at the park.
axiom R2 : Sunny → PlayAtParkToday Kids
-- Either the team practiced daily, or kids did not go to the park today.
axiom R3 : PracticeDaily Team ∨ ¬ PlayAtParkToday Kids
-- If a team feels prepared, then they will have confidence.
axiom R4 : ∀ x : obj, Prepared x → Confident x
-- When people feel confident, they hold their head up high.
axiom R5 : ∀ x : obj, Confident x → HoldHeadHigh x

theorem sunny_team_hold_head_high : Sunny → HoldHeadHigh Team := by
  intro hs
  have pk : PlayAtParkToday Kids := R2 hs
  exact Or.elim R3
    (fun hpd =>
      let hprep : Prepared Team := R1 Team hpd
      let hconf : Confident Team := R4 Team hprep
      R5 Team hconf)
    (fun hnpk =>
      False.elim (hnpk pk))
</lean>

ANSWER: Yes
