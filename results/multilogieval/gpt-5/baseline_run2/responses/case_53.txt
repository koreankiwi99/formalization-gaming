<lean>
axiom obj : Type
axiom Mary : obj

axiom PracticedOften : obj → Prop
axiom ImprovedTechnique : obj → Prop
axiom StoodOutHighQuality : obj → Prop
axiom TookArtClass : obj → Prop
axiom LearnedNewMethodsMediums : obj → Prop

-- If someone practiced often, their technique improved.
axiom R1 : ∀ x : obj, PracticedOften x → ImprovedTechnique x
-- If someone's technique improved, their artwork stood out as high quality.
axiom R2 : ∀ x : obj, ImprovedTechnique x → StoodOutHighQuality x
-- Either Mary practiced often or took an art class (or both).
axiom E1 : PracticedOften Mary ∨ TookArtClass Mary
-- If someone takes an art class, they learn new methods and mediums.
axiom R3 : ∀ x : obj, TookArtClass x → LearnedNewMethodsMediums x
-- Mary's artwork did not stand out as high quality.
axiom Hnot : ¬ StoodOutHighQuality Mary

theorem mary_learned : LearnedNewMethodsMediums Mary :=
by
  -- From ¬(stood out) and (practiced → improved → stood out), deduce ¬practiced
  have h_not_practice : ¬ PracticedOften Mary :=
    fun hp => Hnot (R2 Mary (R1 Mary hp))
  -- From (practiced ∨ took class) and ¬practiced, deduce took class
  have h_class : TookArtClass Mary :=
    Or.elim E1
      (fun hp => False.elim (h_not_practice hp))
      (fun hc => hc)
  -- Therefore, she learned new methods and mediums
  exact R3 Mary h_class
</lean>

ANSWER: Yes
