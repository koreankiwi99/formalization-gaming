<lean>
axiom obj : Type
axiom John : obj

axiom Alarm : obj → Prop
axiom Ready : obj → Prop
axiom MissBus : obj → Prop
axiom Late : obj → Prop
axiom UpLate : obj → Prop
axiom BedLate : obj → Prop

-- If the alarm went off, then it is time to get ready.
axiom R1 : ∀ x : obj, Alarm x → Ready x
-- If they missed the bus, then they'd be late for school.
axiom R2 : ∀ x : obj, MissBus x → Late x
-- Either the alarm went off, or they weren't late for school.
axiom R3 : ∀ x : obj, Alarm x ∨ ¬Late x
-- If they got up late, then they miss the bus.
axiom R4 : ∀ x : obj, UpLate x → MissBus x
-- If they went to bed late, then they get up late.
axiom R5 : ∀ x : obj, BedLate x → UpLate x

-- John knows it is not the time to get ready (we take this as ¬ Ready John).
axiom John_not_ready : ¬ Ready John

theorem john_not_bedlate : ¬ BedLate John := by
  intro hBed
  have hUp : UpLate John := R5 John hBed
  have hMiss : MissBus John := R4 John hUp
  have hLate : Late John := R2 John hMiss
  have hNotAlarm : ¬ Alarm John := by
    intro hAl
    apply John_not_ready
    exact R1 John hAl
  have hOr : Alarm John ∨ ¬ Late John := R3 John
  have hNotLate : ¬ Late John := by
    cases hOr with
    | inl hAl => exact False.elim (hNotAlarm hAl)
    | inr hNL => exact hNL
  exact hNotLate hLate
