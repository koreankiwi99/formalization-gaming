<lean>
axiom Person : Type
axiom Mary : Person

axiom ClearUp : Prop
axiom GameOn : Prop
axiom FansCheer : Prop
axiom ExcitingNight : Prop

axiom FinishErrands : Person → Prop
axiom MeetDinner : Person → Prop

axiom R1 : ClearUp → GameOn
axiom R2 : ∀ x : Person, FinishErrands x → MeetDinner x
axiom R3 : ClearUp ∨ FinishErrands Mary
axiom R4 : GameOn → FansCheer
axiom R5 : FansCheer → ExcitingNight

theorem exciting_if_not_meet : (¬MeetDinner Mary) → ExcitingNight := by
  intro hNotMeet
  have hNotFinish : ¬FinishErrands Mary := by
    intro hFin
    have hMeet : MeetDinner Mary := R2 Mary hFin
    exact hNotMeet hMeet
  have hClear : ClearUp := Or.elim R3 (fun hC => hC) (fun hFin => False.elim (hNotFinish hFin))
  have hGame : GameOn := R1 hClear
  have hCheer : FansCheer := R4 hGame
  exact R5 hCheer
</lean>

ANSWER: Yes
