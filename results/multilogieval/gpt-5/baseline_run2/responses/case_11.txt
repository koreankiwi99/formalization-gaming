<lean>
axiom obj : Type
axiom Proj : obj

axiom Nice : obj → Prop
axiom Impressed : obj → Prop
axiom Thorough : obj → Prop
axiom Serious : obj → Prop
axiom Award : obj → Prop

-- If the display looks nice, judges will be impressed.
axiom H1 : Nice Proj → Impressed Proj
-- If the research is thorough, experts will take the project seriously.
axiom H2 : Thorough Proj → Serious Proj
-- Either a good display or thorough research.
axiom H3 : Nice Proj ∨ Thorough Proj
-- If experts take a project seriously, it will receive an award.
axiom H4 : Serious Proj → Award Proj
-- Judges were not impressed.
axiom H5 : ¬ Impressed Proj

theorem project_award : Award Proj := by
  have hNotNice : ¬ Nice Proj := by
    intro hN
    apply H5
    apply H1
    exact hN
  have hTh : Thorough Proj := by
    cases H3 with
    | inl hN =>
        have : False := by
          apply H5
          apply H1
          exact hN
        exact False.elim this
    | inr hT => exact hT
  have hSer : Serious Proj := by
    apply H2
    exact hTh
  apply H4
  exact hSer
</lean>

ANSWER: Yes
