<lean>
axiom obj : Type
axiom James : obj

axiom Follow : obj → Prop
axiom Well : obj → Prop
axiom Confidence : obj → Prop
axiom Watch : obj → Prop

axiom R1 : ∀ x : obj, Follow x → Well x
axiom R2 : ∀ x : obj, Well x → Confidence x
axiom D : Follow James ∨ Watch James

theorem james_watches : ¬Confidence James → Watch James :=
by
  intro hnc
  have hnf : ¬ Follow James := by
    intro hF
    have hw : Well James := R1 James hF
    have hc : Confidence James := R2 James hw
    exact hnc hc
  exact Or.elim D (fun hF => False.elim (hnf hF)) (fun hW => hW)
</lean>

ANSWER: Yes
