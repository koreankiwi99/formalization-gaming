<lean>
axiom person : Type
axiom Lucy : person
axiom Hacked : person → Prop
axiom Reset : person → Prop
axiom Notify : person → Prop

-- If hacked, then the person resets their password
axiom R1 : ∀ x : person, Hacked x → Reset x
-- If hacked, then the person notifies their bank
axiom R2 : ∀ x : person, Hacked x → Notify x

-- Either Lucy did not reset her password or she did not notify her bank
axiom E : ¬ Reset Lucy ∨ ¬ Notify Lucy

theorem lucy_not_hacked : ¬ Hacked Lucy :=
by
  intro h
  have r : Reset Lucy := R1 Lucy h
  have n : Notify Lucy := R2 Lucy h
  cases E with
  | inl nr => exact nr r
  | inr nn => exact nn n
</lean>

ANSWER: No
