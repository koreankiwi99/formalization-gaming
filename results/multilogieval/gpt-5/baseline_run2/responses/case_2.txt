<lean>
axiom Rained : Prop
axiom Hot : Prop
axiom Sprinklers : Prop
axiom WateredNaturally : Prop

axiom R1 : Rained → WateredNaturally
axiom R2 : Hot → Sprinklers
axiom R3 : Rained ∨ ¬ Sprinklers

axiom T_hot : Hot

theorem watered_today : WateredNaturally := by
  have hs : Sprinklers := R2 T_hot
  have hr : Rained := by
    cases R3 with
    | inl h => exact h
    | inr hns =>
        have f : False := hns hs
        exact False.elim f
  exact R1 hr
</lean>

ANSWER: Yes
