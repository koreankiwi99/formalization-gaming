<lean>
axiom obj : Type
axiom Sam : obj

axiom Impressed : obj → Prop
axiom Extra : obj → Prop
axiom Experience : obj → Prop
axiom Promotion : obj → Prop
axiom Lead : obj → Prop

-- Rules from the context
axiom R1 : ∀ x : obj, Impressed x → Promotion x
axiom R2 : ∀ x : obj, Extra x → Experience x
axiom R3 : Impressed Sam ∨ Extra Sam
axiom R4 : ∀ x : obj, Experience x → Lead x

-- Given: Sam did not get a promotion
axiom Hnp : ¬ Promotion Sam

theorem sam_assigned_to_lead_new_team : Lead Sam := by
  have hdisj : Impressed Sam ∨ Extra Sam := R3
  have hnp : ¬ Promotion Sam := Hnp
  cases hdisj with
  | inl hI =>
      have hp : Promotion Sam := R1 Sam hI
      exact False.elim (hnp hp)
  | inr hE =>
      have hx : Experience Sam := R2 Sam hE
      exact R4 Sam hx
</lean>

ANSWER: Yes
