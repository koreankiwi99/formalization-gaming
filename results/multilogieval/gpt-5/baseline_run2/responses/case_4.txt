<lean>
axiom Marketing : Prop
axiom MoreCustomers : Prop
axiom SalesIncrease : Prop
axiom ProdCostsIncreased : Prop

axiom R1 : Marketing → MoreCustomers
axiom R2 : MoreCustomers → SalesIncrease
axiom Ax : Marketing ∨ ProdCostsIncreased

theorem production_costs_increased (Hns : ¬ SalesIncrease) : ProdCostsIncreased := by
  have hnotM : ¬ Marketing := by
    intro hM
    apply Hns
    exact R2 (R1 hM)
  cases Ax with
  | inl hM =>
      apply False.elim
      exact hnotM hM
  | inr hP =>
      exact hP
</lean>

ANSWER: Yes
