<lean>
axiom obj : Type
axiom Priya : obj

axiom StudiesHard : obj → Prop
axiom GoodGrades : obj → Prop
axiom Proud : obj → Prop
axiom IceCream : obj → Prop

axiom Snowstorm : Prop
axiom SchoolCanceled : Prop

-- If someone studies hard for an exam, they will get good grades.
axiom R1 : ∀ x : obj, StudiesHard x → GoodGrades x
-- If there is a snowstorm, then school is canceled.
axiom R2 : Snowstorm → SchoolCanceled
-- Either Priya studied hard, or school was not called off.
axiom R3 : StudiesHard Priya ∨ ¬ SchoolCanceled
-- If a student gets good grades, their parents are proud.
axiom R4 : ∀ x : obj, GoodGrades x → Proud x
-- When parents are proud, they take their kids out for ice cream.
axiom R5 : ∀ x : obj, Proud x → IceCream x

theorem priya_icecream_if_snowstorm : Snowstorm → IceCream Priya := by
  intro hs
  have hc : SchoolCanceled := R2 hs
  have hstudies : StudiesHard Priya := by
    cases R3 with
    | inl h => exact h
    | inr hnot =>
        have contra : False := hnot hc
        exact False.elim contra
  have hg : GoodGrades Priya := R1 Priya hstudies
  have hp : Proud Priya := R4 Priya hg
  exact R5 Priya hp
</lean>

ANSWER: Yes
