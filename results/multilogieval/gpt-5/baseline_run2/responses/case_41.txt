<lean>
axiom WatDaily : Prop
axiom GrowWell : Prop
axiom Drought : Prop
axiom Restrictions : Prop
axiom NotEnoughWater : Prop
axiom WiltAndDie : Prop

axiom A1 : WatDaily → GrowWell
axiom A2 : Drought → Restrictions
axiom A3 : WatDaily ∨ Drought
axiom A4 : Restrictions → NotEnoughWater
axiom A5 : NotEnoughWater → WiltAndDie

theorem wilt_and_die_of_not_growwell : ¬GrowWell → WiltAndDie := by
  intro hngw
  have hnotwd : ¬WatDaily := by
    intro hwd
    exact hngw (A1 hwd)
  have hd : Drought := by
    cases A3 with
    | inl hwd =>
      exact False.elim (hnotwd hwd)
    | inr hd =>
      exact hd
  have hrs : Restrictions := A2 hd
  have hne : NotEnoughWater := A4 hrs
  exact A5 hne
</lean>

ANSWER: Yes
