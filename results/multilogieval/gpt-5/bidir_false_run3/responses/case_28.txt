<lean>
axiom obj : Type
axiom Jane : obj

axiom Hurt : obj → Prop
axiom SeeDoc : obj → Prop
axiom Rest : obj → Prop

-- Context rules
axiom R1 : ∀ x : obj, Hurt x → SeeDoc x
axiom R2 : ∀ x : obj, Hurt x → Rest x

-- Statement to evaluate (assumed as given)
axiom S : ¬SeeDoc Jane ∨ ¬Rest Jane

-- Prove that Jane did not get hurt
theorem jane_not_hurt : ¬Hurt Jane := by
  intro h
  have sdoc : SeeDoc Jane := R1 Jane h
  have r : Rest Jane := R2 Jane h
  have not_disj : ¬(¬SeeDoc Jane ∨ ¬Rest Jane) := by
    intro disj
    cases disj with
    | inl nsee => exact nsee sdoc
    | inr nrest => exact nrest r
  exact not_disj S
</lean>

ANSWER: No
