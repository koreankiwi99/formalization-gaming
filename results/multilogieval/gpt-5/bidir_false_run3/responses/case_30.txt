<lean>
axiom obj : Type
axiom Mark : obj

axiom HealthyDiet : obj → Prop
axiom Strong : obj → Prop
axiom Skilled : obj → Prop
axiom TrainProperly : obj → Prop
axiom Dedicated : obj → Prop

-- Context axioms
axiom A1 : ∀ x : obj, TrainProperly x → Skilled x
axiom A2 : ∀ x : obj, HealthyDiet x → Strong x
axiom A3 : ∀ x : obj, ¬Skilled x ∨ ¬Strong x
axiom A4 : ∀ x : obj, Dedicated x → TrainProperly x

-- Given: Mark eats a healthy diet
axiom HM : HealthyDiet Mark

theorem mark_not_dedicated : ¬Dedicated Mark := by
  have hStrong : Strong Mark := A2 Mark HM
  have hdisj : ¬Skilled Mark ∨ ¬Strong Mark := A3 Mark
  have hNotSkilled : ¬Skilled Mark := by
    cases hdisj with
    | inl h => exact h
    | inr h =>
      have contra : False := h hStrong
      exact False.elim contra
  have hNotTrain : ¬TrainProperly Mark := by
    intro htp
    exact hNotSkilled (A1 Mark htp)
  intro hded
  exact hNotTrain (A4 Mark hded)
</lean>

ANSWER: No
