<lean>
axiom obj : Type
axiom Simone : obj
axiom TrainsWithDedication : obj → Prop
axiom AdequateRest : obj → Prop
axiom PeakPerformance : obj → Prop
axiom CoachRule : ∀ x : obj, TrainsWithDedication x → AdequateRest x → PeakPerformance x
axiom Rest_Simone : AdequateRest Simone
axiom NotPeak_Simone : ¬PeakPerformance Simone

theorem simone_not_trained : ¬TrainsWithDedication Simone := by
  intro hTrain
  have hPeak : PeakPerformance Simone := CoachRule Simone hTrain Rest_Simone
  exact NotPeak_Simone hPeak
</lean>
