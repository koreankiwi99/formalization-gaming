<lean>
axiom obj : Type
axiom someone : obj

axiom SavedAggressively : obj → Prop
axiom ImprovedCredit : obj → Prop
axiom AffordDownPayment : obj → Prop
axiom QualifyMortgage : obj → Prop
axiom BuyDreamHome : obj → Prop

-- If they saved aggressively, they could afford a down payment.
axiom R1 : ∀ x : obj, SavedAggressively x → AffordDownPayment x
-- If they improved their credit score, they could qualify for a mortgage.
axiom R2 : ∀ x : obj, ImprovedCredit x → QualifyMortgage x
-- If someone qualifies for a mortgage, they can buy their dream home.
axiom R3 : ∀ x : obj, QualifyMortgage x → BuyDreamHome x
-- Someone either saves aggressively or improves their credit.
axiom R4 : SavedAggressively someone ∨ ImprovedCredit someone
-- Given that they could not afford a down payment.
axiom Hnot : ¬AffordDownPayment someone

-- Cannot prove ¬BuyDreamHome someone from given axioms; in fact, we can derive BuyDreamHome someone:
theorem can_buy : BuyDreamHome someone := by
  -- From ¬AffordDownPayment and (Saved → AffordDownPayment), deduce ¬Saved
  have h_not_saved : ¬SavedAggressively someone := by
    intro hs
    exact Hnot (R1 someone hs)
  -- From Saved ∨ Improved and ¬Saved, deduce Improved
  have h_improved : ImprovedCredit someone := by
    cases R4 with
    | inl hs =>
        exact False.elim (Hnot (R1 someone hs))
    | inr hi =>
        exact hi
  -- Then qualify and buy
  have h_qual : QualifyMortgage someone := R2 someone h_improved
  exact R3 someone h_qual
</lean>
