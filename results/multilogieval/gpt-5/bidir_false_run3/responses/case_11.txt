<lean>
axiom obj : Type
axiom Project : obj

axiom LooksNice : obj → Prop
axiom Impressed : obj → Prop
axiom Thorough : obj → Prop
axiom Serious : obj → Prop
axiom Award : obj → Prop

-- If a display looks nice, judges will be impressed
axiom R1 : ∀ x : obj, LooksNice x → Impressed x
-- If research is thorough, experts will take it seriously
axiom R2 : ∀ x : obj, Thorough x → Serious x
-- The project has either a good display or thorough research
axiom R3 : LooksNice Project ∨ Thorough Project
-- If experts take a project seriously, it will receive an award
axiom R4 : ∀ x : obj, Serious x → Award x

-- Given: judges were not impressed
axiom H1 : ¬Impressed Project

-- From not impressed and (LooksNice → Impressed), derive not LooksNice
theorem not_looks_nice : ¬LooksNice Project := by
  intro h
  have : Impressed Project := R1 Project h
  exact H1 this

-- From (LooksNice ∨ Thorough) and ¬LooksNice, derive Thorough
theorem thorough_project : Thorough Project := by
  cases R3 with
  | inl hNice =>
    have : False := H1 (R1 Project hNice)
    exact False.elim this
  | inr hTh =>
    exact hTh

-- Therefore the project receives an award
theorem award_project : Award Project := by
  have hSerious : Serious Project := R2 Project thorough_project
  exact R4 Project hSerious
