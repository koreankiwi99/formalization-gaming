<lean>
axiom obj : Type
axiom Alex : obj

axiom Sunny : Prop
axiom Warm : Prop

axiom Applied : obj → Prop
axiom Burnt : obj → Prop
axiom Swim : obj → Prop
axiom Exercise : obj → Prop

axiom T1 : Sunny
axiom R1 : ∀ x : obj, Applied x → ¬Burnt x
axiom R2 : ∀ x : obj, Swim x → Exercise x
axiom R3 : ∀ x : obj, Applied x ∨ ¬Exercise x
axiom R4 : Warm → ∀ x : obj, Swim x
axiom R5 : Sunny → Warm

theorem alex_not_burnt : ¬Burnt Alex := by
  have hWarm : Warm := R5 T1
  have hAllSwim : ∀ x : obj, Swim x := R4 hWarm
  have hSwimAlex : Swim Alex := hAllSwim Alex
  have hExAlex : Exercise Alex := R2 Alex hSwimAlex
  have hDisj : Applied Alex ∨ ¬Exercise Alex := R3 Alex
  have hAppAlex : Applied Alex := by
    cases hDisj with
    | inl hA => exact hA
    | inr hNotEx =>
        exact False.elim (hNotEx hExAlex)
  exact R1 Alex hAppAlex
</lean>

ANSWER: No
