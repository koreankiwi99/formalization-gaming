<lean>
axiom obj : Type
axiom Max : obj

axiom PackageOnTime : Prop
axiom SpecialOrderReady : Prop

axiom AbleStartNewProject : obj → Prop
axiom CanPickUpMaterials : obj → Prop
axiom StartNewProject : obj → Prop
axiom LearnSkills : obj → Prop

-- If the package arrives on time, people will be able to start a new project.
axiom R1 : PackageOnTime → ∀ x : obj, AbleStartNewProject x
-- If the office supply store has the special order in stock, they can pick up the materials they need.
axiom R2 : SpecialOrderReady → ∀ x : obj, CanPickUpMaterials x
-- Either the package will arrive on time or the special order will be ready.
axiom R3 : PackageOnTime ∨ SpecialOrderReady
-- If someone starts his new project, they will learn some new skills.
axiom R4 : ∀ x : obj, StartNewProject x → LearnSkills x

-- Max was unable to pick up the materials he needed.
axiom TMaxUnable : ¬CanPickUpMaterials Max

theorem special_order_not_ready : ¬SpecialOrderReady := by
  intro hSOR
  have hCanAll : ∀ x : obj, CanPickUpMaterials x := R2 hSOR
  have hCanMax : CanPickUpMaterials Max := hCanAll Max
  exact TMaxUnable hCanMax

theorem package_on_time : PackageOnTime := by
  have hNotSOR : ¬SpecialOrderReady := special_order_not_ready
  cases R3 with
  | inl hP => exact hP
  | inr hS =>
      have : False := hNotSOR hS
      exact False.elim this

theorem max_able_to_start : AbleStartNewProject Max := by
  have hP : PackageOnTime := package_on_time
  have hAll : ∀ x : obj, AbleStartNewProject x := R1 hP
  exact hAll Max

-- Cannot prove ¬LearnSkills Max from the given axioms.
