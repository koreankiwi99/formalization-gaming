<lean>
axiom obj : Type
axiom Alex : obj

axiom WeekendComing : Prop
axiom Relax : obj → Prop
axiom Finishes : obj → Prop
axiom ClearMind : obj → Prop
axiom Organized : obj → Prop
axiom ToDoList : obj → Prop

-- Rules from the context
axiom R1 : ∀ x : obj, WeekendComing → Relax x
axiom R2 : ∀ x : obj, Finishes x → ClearMind x
axiom R3 : WeekendComing ∨ ¬ClearMind Alex
axiom R4 : ∀ x : obj, Organized x → Finishes x
axiom R5 : ∀ x : obj, ToDoList x → Organized x

-- Given in the statement to evaluate
axiom NotRelaxAlex : ¬Relax Alex

theorem alex_not_todolist : ¬ToDoList Alex := by
  intro hTodo
  have hOrg : Organized Alex := R5 Alex hTodo
  have hFin : Finishes Alex := R4 Alex hOrg
  have hCM  : ClearMind Alex := R2 Alex hFin
  have hNotWC : ¬WeekendComing := by
    intro hWC
    have hRelax : Relax Alex := R1 Alex hWC
    exact NotRelaxAlex hRelax
  have hNotCM : ¬ClearMind Alex := by
    apply Or.elim R3
    · intro hWC
      exact False.elim (hNotWC hWC)
    · intro hNCM
      exact hNCM
  exact hNotCM hCM
</lean>

ANSWER: No
