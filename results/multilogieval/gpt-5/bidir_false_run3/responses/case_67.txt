axiom Person : Type
axiom Sam : Person

axiom KitchenHot : Prop
axiom StoveOn : Prop

axiom UseStove : Person → Prop
axiom MakePancakes : Person → Prop
axiom HungryMorning : Person → Prop
axiom AteDinnerLastNight : Person → Prop

-- If the stove is on, the kitchen gets hot.
axiom R1 : StoveOn → KitchenHot
-- If someone makes pancakes, they use the stove.
axiom R2 : ∀ x : Person, MakePancakes x → UseStove x
-- If someone is hungry, they make pancakes.
axiom R3 : ∀ x : Person, HungryMorning x → MakePancakes x
-- If Sam did not eat dinner last night, he was hungry this morning.
axiom R4 : ¬AteDinnerLastNight Sam → HungryMorning Sam
-- Either the stove was on, or they (whoever x is) did not use the stove.
axiom R5 : ∀ x : Person, StoveOn ∨ ¬UseStove x

-- From "Sam did not eat dinner", the kitchen gets hot.
theorem hot_if_no_dinner : ¬AteDinnerLastNight Sam → KitchenHot := by
  intro hNoEat
  have hHun : HungryMorning Sam := R4 hNoEat
  have hPanc : MakePancakes Sam := R3 Sam hHun
  have hUse : UseStove Sam := R2 Sam hPanc
  have hOn : StoveOn :=
    Or.elim (R5 Sam)
      (fun hOn => hOn)
      (fun hNotUse => False.elim (hNotUse hUse))
  exact R1 hOn

-- From "the kitchen did not get hot", we get ¬¬(Sam ate dinner last night).
theorem not_not_ate_if_not_hot : ¬KitchenHot → ¬¬AteDinnerLastNight Sam := by
  intro hNotHot
  intro hNotAte
  apply hNotHot
  exact hot_if_no_dinner hNotAte
