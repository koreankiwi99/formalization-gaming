<lean>
axiom obj : Type
axiom Jake : obj

axiom WeatherNice : Prop
axiom Relax : obj → Prop
axiom Reads : obj → Prop
axiom PackedEnough : obj → Prop
axiom Prepared : obj → Prop

-- If the weather is nice, then one can relax outside
axiom R1 : ∀ x : obj, WeatherNice → Relax x
-- If someone reads books, then they packed enough
axiom R2 : ∀ x : obj, Reads x → PackedEnough x
-- For any person, either the weather is nice, or they did not pack enough, or both
axiom R3 : ∀ x : obj, WeatherNice ∨ ¬PackedEnough x ∨ (WeatherNice ∧ ¬PackedEnough x)
-- If someone is prepared, they read books
axiom R4 : ∀ x : obj, Prepared x → Reads x

-- From "Jake cannot relax outside" we derive "Jake is not prepared"
theorem not_prepared_if_not_relax : ¬Relax Jake → ¬Prepared Jake := by
  intro hnr
  -- ¬WeatherNice
  have hnw : ¬WeatherNice := by
    intro hw
    exact hnr (R1 Jake hw)
  -- ¬PackedEnough Jake
  have hnpe : ¬PackedEnough Jake := by
    have d := R3 Jake
    cases d with
    | inl hw =>
        intro _; exact hnw hw
    | inr rest =>
        cases rest with
        | inl hnp =>
            exact hnp
        | inr hw_and_hnp =>
            exact hw_and_hnp.right
  -- Prepared Jake → PackedEnough Jake
  have prep_to_pe : Prepared Jake → PackedEnough Jake := by
    intro hp
    exact R2 Jake (R4 Jake hp)
  -- Conclude ¬Prepared Jake
  intro hp
  exact hnpe (prep_to_pe hp)
</lean>
