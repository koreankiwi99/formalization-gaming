axiom Emma : Person

axiom Fluency : Person → Prop
axiom Vocabulary : Person → Prop
axiom RealConversations : Person → Prop
axiom ListenPodcasts : Person → Prop
axiom ListeningComp : Person → Prop

-- If someone speaks with native speakers daily, their fluency and vocabulary improve
axiom R1f : ∀ x : Person, SpeaksDaily x → Fluency x
axiom R1v : ∀ x : Person, SpeaksDaily x → Vocabulary x

-- If fluency improves, they can have real conversations
axiom R2 : ∀ x : Person, Fluency x → RealConversations x

-- Everyone either speaks daily with natives or listens to educational podcasts (or both)
axiom R3 : ∀ x : Person, SpeaksDaily x ∨ ListenPodcasts x

-- If someone listens to educational podcasts, they gain listening comprehension
axiom R4 : ∀ x : Person, ListenPodcasts x → ListeningComp x

-- In fact, the statement in question is derivable:
theorem emma_listening_if_unable_converse : (¬RealConversations Emma) → ListeningComp Emma :=
by
  intro hNotRC
  have hdisj : SpeaksDaily Emma ∨ ListenPodcasts Emma := R3 Emma
  cases hdisj with
  | inl hSpeak =>
      have hFl : Fluency Emma := R1f Emma hSpeak
      have hRC : RealConversations Emma := R2 Emma hFl
      exact False.elim (hNotRC hRC)
  | inr hPod =>
      exact R4 Emma hPod
