<lean>
axiom obj : Type
axiom Sam : obj

axiom Impressed : obj → Prop
axiom Promotion : obj → Prop
axiom ExtraWork : obj → Prop
axiom Experience : obj → Prop
axiom LeadNewTeam : obj → Prop

-- If someone impressed the manager, they get a promotion
axiom R1 : ∀ x : obj, Impressed x → Promotion x
-- If someone took on extra work, they gain experience
axiom R2 : ∀ x : obj, ExtraWork x → Experience x
-- If someone gains experience, they will be assigned to lead a new team
axiom R3 : ∀ x : obj, Experience x → LeadNewTeam x

-- For Sam specifically: either impressed the manager or took on extra work
axiom Dsam : Impressed Sam ∨ ExtraWork Sam
-- Given: Sam did not get a promotion
axiom NP : ¬ Promotion Sam

theorem sam_leads_new_team : LeadNewTeam Sam := by
  have h_not_impressed : ¬ Impressed Sam := fun hI => NP (R1 Sam hI)
  have h_extra : ExtraWork Sam :=
    Or.elim Dsam
      (fun hI => False.elim (h_not_impressed hI))
      (fun hE => hE)
  have h_exp : Experience Sam := R2 Sam h_extra
  exact R3 Sam h_exp
</lean>

ANSWER: Yes
