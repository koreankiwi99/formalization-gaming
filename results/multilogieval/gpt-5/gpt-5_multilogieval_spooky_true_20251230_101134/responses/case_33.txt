-- Entities
axiom Bob : obj

-- Predicates
axiom Rotates : obj → Prop            -- rotates crops well
axiom NutrientRich : obj → Prop       -- crops maintain/have nutrients
axiom Healthy : obj → Prop            -- crops are healthy
axiom Organic : obj → Prop            -- uses organic methods
axiom OneWithNature : obj → Prop      -- is one with nature

-- Contextual rules from the text
axiom R_organic_healthy : ∀ x : obj, Organic x → Healthy x
axiom R_rotate_nutrients : ∀ x : obj, Rotates x → NutrientRich x
axiom R_bad_crops : ∀ x : obj, ¬ Healthy x ∨ ¬ NutrientRich x
axiom R_one_with_nature_implies_organic : ∀ x : obj, OneWithNature x → Organic x

-- From the context, rotating combined with the global disjunction forces unhealthiness,
-- which then via contrapositive of R_organic_healthy gives not organic,
-- and thus by modus tollens on R_one_with_nature_implies_organic gives not one with nature.
theorem not_one_with_nature_of_rotates : Rotates Bob → ¬ OneWithNature Bob := by
  intro hRot
  have hNutr : NutrientRich Bob := R_rotate_nutrients Bob hRot
  have hDisj : ¬ Healthy Bob ∨ ¬ NutrientRich Bob := R_bad_crops Bob
  have hNotHealthy : ¬ Healthy Bob := by
    cases hDisj with
    | inl nh => exact nh
    | inr nNutr =>
      exact False.elim (nNutr hNutr)
  have hNotOrganic : ¬ Organic Bob := by
    intro hOrg
    have hHealthy : Healthy Bob := R_organic_healthy Bob hOrg
    exact hNotHealthy hHealthy
  intro hOne
  have hOrg : Organic Bob := R_one_with_nature_implies_organic Bob hOne
  exact hNotOrganic hOrg
