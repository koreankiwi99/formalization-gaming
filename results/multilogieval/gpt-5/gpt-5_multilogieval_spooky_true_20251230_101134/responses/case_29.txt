-- Predicates
axiom SuspectsID : obj → Prop
axiom SuspectsFraud : obj → Prop
axiom ChecksCreditReport : obj → Prop
axiom ContactsBank : obj → Prop

-- Premises:
-- If someone suspects identity theft, they should check their credit report.
axiom R1 : ∀ x : obj, SuspectsID x → ChecksCreditReport x
-- If fraud is suspected, they should contact their bank.
axiom R2 : ∀ x : obj, SuspectsFraud x → ContactsBank x

-- Given statement to evaluate (as an assumption for Ty):
axiom Hdisj : ¬ ChecksCreditReport Ty ∨ ¬ ContactsBank Ty

-- From the given disjunction and the rules, we can only derive that
-- either Ty did not suspect identity theft or Ty did not suspect fraud.
theorem not_suspicions_from_neg_actions :
  (¬ ChecksCreditReport Ty ∨ ¬ ContactsBank Ty) →
  (¬ SuspectsID Ty ∨ ¬ SuspectsFraud Ty) := by
  intro h
  cases h with
  | inl hnc =>
      apply Or.inl
      intro hsid
      have hcheck : ChecksCreditReport Ty := R1 Ty hsid
      exact hnc hcheck
  | inr hnb =>
      apply Or.inr
      intro hsf
      have hcb : ContactsBank Ty := R2 Ty hsf
      exact hnb hcb

theorem derived_consequence : ¬ SuspectsID Ty ∨ ¬ SuspectsFraud Ty := by
  exact not_suspicions_from_neg_actions Hdisj
