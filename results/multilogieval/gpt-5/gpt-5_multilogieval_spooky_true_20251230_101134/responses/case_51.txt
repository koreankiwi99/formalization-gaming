<lean>
axiom Person : Type
axiom Mary : Person

axiom Sale : Prop
axiom StockUp : Person → Prop
axiom MakeList : Person → Prop
axiom Plan : Person → Prop
axiom Payday : Prop
axiom PricesUp : Prop

-- Premises
axiom P1 : Sale → StockUp Mary
axiom P2 : MakeList Mary → Plan Mary
axiom P3 : Sale ∨ (¬ Plan Mary) ∨ (Sale ∧ ¬ Plan Mary)
axiom P4 : Payday → MakeList Mary
axiom P5 : Payday ∨ PricesUp

theorem mary_prices_up : (¬ StockUp Mary) → PricesUp := by
  intro hNotStock
  -- Analyze the trilemma about sale/planning
  cases P3 with
  | inl hSale =>
      have hSU : StockUp Mary := P1 hSale
      exact False.elim (hNotStock hSU)
  | inr hRest =>
      cases hRest with
      | inl hNotPlan =>
          -- From Payday → MakeList and MakeList → Plan, get Payday → Plan
          -- Then with ¬Plan, we get ¬Payday
          have hNotPayday : ¬ Payday := by
            intro hPay
            have hList : MakeList Mary := P4 hPay
            have hPlan : Plan Mary := P2 hList
            exact hNotPlan hPlan
          -- From Payday ∨ PricesUp and ¬Payday, deduce PricesUp
          cases P5 with
          | inl hPay => exact False.elim (hNotPayday hPay)
          | inr hUp  => exact hUp
      | inr hSaleAndNotPlan =>
          have hSale : Sale := hSaleAndNotPlan.left
          have hSU : StockUp Mary := P1 hSale
          exact False.elim (hNotStock hSU)
</lean>

ANSWER: Yes
