<lean>
axiom Sunny : Prop
axiom PlayOutside : Prop
axiom SleepWell : Prop
axiom FluSeasonStarted : Prop
axiom ManyKidsSick : Prop

-- If it is sunny, children play outside.
axiom R1 : Sunny → PlayOutside
-- If flu season begins, many children get sick.
axiom R2 : FluSeasonStarted → ManyKidsSick
-- Either sunny, or not many kids get the flu, or (sunny and not many kids get sick).
axiom R3 : Sunny ∨ ¬ManyKidsSick ∨ (Sunny ∧ ¬ManyKidsSick)
-- If children play outside, they sleep well at night.
axiom R4 : PlayOutside → SleepWell
-- Flu season started this month.
axiom H : FluSeasonStarted

theorem children_sleep_well : SleepWell := by
  have hMany : ManyKidsSick := R2 H
  cases R3 with
  | inl s =>
      exact R4 (R1 s)
  | inr rest =>
      cases rest with
      | inl notMany =>
          have : False := notMany hMany
          exact False.elim this
      | inr both =>
          exact R4 (R1 both.left)
</lean>

ANSWER: Yes
