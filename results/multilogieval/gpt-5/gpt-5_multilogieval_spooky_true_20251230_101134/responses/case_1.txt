<lean>
axiom StrongPerimeterDefense : Prop
axiom ForceManyTurnovers : Prop
axiom ProtectPaint : Prop
axiom LimitInsideScoring : Prop

axiom R1 : StrongPerimeterDefense → ForceManyTurnovers
axiom R2 : ProtectPaint → LimitInsideScoring
axiom R3 : StrongPerimeterDefense ∨ ProtectPaint

theorem lakers_force_turnovers : (¬ LimitInsideScoring) → ForceManyTurnovers :=
by
  intro hnotL
  apply Or.elim R3
  · intro hP
    exact R1 hP
  · intro hB
    have hL : LimitInsideScoring := R2 hB
    have : False := hnotL hL
    exact False.elim this
</lean>

ANSWER: Yes
