<lean>
axiom Party : Prop
axiom Drums : Prop
axiom NoiseUp : Prop
axiom Noisy : Prop
axiom Complaint : Prop
axiom Police : Prop

axiom party_noisy : Party → Noisy
axiom drums_noiseUp : Drums → NoiseUp
axiom party_or_drums : Party ∨ Drums
axiom partyNoisy_complaint : Party ∧ Noisy → Complaint
axiom complaint_police : Complaint → Police

theorem no_noiseUp_police : (¬ NoiseUp) → Police := by
  intro hNoU
  have h := party_or_drums
  cases h with
  | inl hP =>
    have hN : Noisy := party_noisy hP
    have hC : Complaint := partyNoisy_complaint ⟨hP, hN⟩
    exact complaint_police hC
  | inr hD =>
    have hU : NoiseUp := drums_noiseUp hD
    have hFalse : False := hNoU hU
    exact False.elim hFalse
</lean>

ANSWER: Yes
