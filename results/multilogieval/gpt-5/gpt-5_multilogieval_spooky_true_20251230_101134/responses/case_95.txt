<lean>
axiom ClearsUp : Prop
axiom GameOn : Prop
axiom ErrandsDoneMary : Prop
axiom DinnerMary : Prop
axiom FansCheer : Prop
axiom ExcitingNight : Prop

axiom H1 : ClearsUp → GameOn
axiom H2 : ErrandsDoneMary → DinnerMary
axiom H3 : ClearsUp ∨ ErrandsDoneMary
axiom H4 : GameOn → FansCheer
axiom H5 : FansCheer → ExcitingNight

theorem exciting_if_mary_no_dinner : (¬ DinnerMary) → ExcitingNight := by
  intro hNotDinner
  cases H3 with
  | inl hClear =>
      apply H5
      apply H4
      exact H1 hClear
  | inr hErrands =>
      have hFalse : False := hNotDinner (H2 hErrands)
      exact False.elim hFalse
</lean>

ANSWER: Yes
