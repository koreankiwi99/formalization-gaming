<lean>
axiom ScoreMoreA : Prop
axiom WinA : Prop
axiom FumbleManyB : Prop
axiom LoseB : Prop

axiom A_win_if_scores : ScoreMoreA → WinA
axiom B_lose_if_fumbles : FumbleManyB → LoseB
axiom either_A_scores_or_B_fumbles : ScoreMoreA ∨ FumbleManyB

theorem A_scored_if_B_not_lose : (¬ LoseB) → ScoreMoreA := by
  intro hNotLose
  have hNotFumble : ¬ FumbleManyB := by
    intro hF
    exact hNotLose (B_lose_if_fumbles hF)
  apply Or.elim either_A_scores_or_B_fumbles
  · intro hScore
    exact hScore
  · intro hFumble
    exfalso
    exact hNotFumble hFumble
</lean>

ANSWER: Yes
