<lean>
axiom GoodReviews : Prop
axiom ManyPeople : Prop
axiom LowAttendance : Prop
axiom BigStar : Prop
axiom Sequel : Prop

-- If the movie gets good reviews, then a lot of people will go see it.
axiom good_implies_many : GoodReviews → ManyPeople
-- If a big star is in the movie, then it got good reviews.
axiom star_implies_good : BigStar → GoodReviews
-- Either there is a big star in it, or it is a sequel.
axiom star_or_sequel : BigStar ∨ Sequel
-- If attendance is low, then not many people went to see it.
axiom low_implies_not_many : LowAttendance → ¬ ManyPeople

-- Goal: If attendance at the movie was low, then it is a sequel.
theorem low_attendance_implies_sequel : LowAttendance → Sequel := by
  intro hLow
  cases star_or_sequel with
  | inl hStar =>
    have hGood : GoodReviews := star_implies_good hStar
    have hMany : ManyPeople := good_implies_many hGood
    have hNotMany : ¬ ManyPeople := low_implies_not_many hLow
    have hFalse : False := hNotMany hMany
    exact False.elim hFalse
  | inr hSeq =>
    exact hSeq
</lean>

ANSWER: Yes
