axiom ProperCare : person → Prop

-- Facts about Linda
axiom LindaProper : ProperCare Linda
axiom LindaNotBountiful : ¬ BountifulHarvest Linda

-- From the rule and Linda's facts, we can derive that Linda did NOT plant disease-resistant varieties.
theorem linda_not_planted : ¬ PlantsResistant Linda :=
by
  intro hPR
  have hB : BountifulHarvest Linda := Rule Linda hPR LindaProper
  exact LindaNotBountiful hB
