<lean>
axiom obj : Type
axiom James : obj

axiom FridayNight : Prop
axiom GoingParty : obj → Prop
axiom Watch : obj → Prop
axiom CanWatch : obj → Prop
axiom FinishHW : obj → Prop
axiom StudyDaily : obj → Prop
axiom PlanWell : obj → Prop

-- If it is Friday night, then (each student) is going to the party.
axiom R1 : ∀ x : obj, FridayNight → GoingParty x
-- If they finish all their homework, then they can watch the game.
axiom R2 : ∀ x : obj, FinishHW x → CanWatch x
-- Either it is Friday night, or (each student) is not watching the game.
axiom R3 : ∀ x : obj, FridayNight ∨ ¬Watch x
-- If they study everyday, they finish all their homework.
axiom R4 : ∀ x : obj, StudyDaily x → FinishHW x
-- If someone plans their schedule well, then they can study everyday.
axiom R5 : ∀ x : obj, PlanWell x → StudyDaily x

-- From "James is not going to the party" we can derive "It is not Friday night".
theorem not_friday_of_not_party (hNP : ¬GoingParty James) : ¬FridayNight := by
  intro f
  have : GoingParty James := R1 James f
  exact hNP this

-- From "James is not going to the party" we can derive "James is not watching the game".
theorem not_watch_of_not_party (hNP : ¬GoingParty James) : ¬Watch James := by
  have nf : ¬FridayNight := not_friday_of_not_party hNP
  have disj : FridayNight ∨ ¬Watch James := R3 James
  cases disj with
  | inl f => exact (nf f).elim
  | inr nw => exact nw

-- Note: There is no way to derive PlanWell James from the given premises and "¬GoingParty James".
-- The premises only yield ¬FridayNight and ¬Watch James; they do not connect to PlanWell James.
</lean>
