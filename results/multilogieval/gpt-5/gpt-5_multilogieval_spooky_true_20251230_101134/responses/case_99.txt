<lean>
axiom HeavyRain : Prop
axiom RainsALot : Prop
axiom Saturated : Prop
axiom Watered : Prop
axiom BloomBrightly : Prop
axiom Bloom : Prop
axiom Fragrance : Prop
axiom AttractButterflies : Prop

-- Contextual rules
axiom rain_lot_saturated : RainsALot → Saturated
axiom heavy_rain_implies_rain_lot : HeavyRain → RainsALot
axiom watered_or_not_saturated : Watered ∨ ¬Saturated
axiom watering_bright_bloom : Watered → BloomBrightly
axiom bright_implies_bloom : BloomBrightly → Bloom
axiom bloom_implies_fragrance : Bloom → Fragrance
axiom fragrance_attracts_butterflies : Fragrance → AttractButterflies

theorem heavy_rain_attracts : HeavyRain → AttractButterflies := by
  intro hHeavy
  have hLot : RainsALot := heavy_rain_implies_rain_lot hHeavy
  have hSat : Saturated := rain_lot_saturated hLot
  have hWatered : Watered := by
    cases watered_or_not_saturated with
    | inl w => exact w
    | inr notSat =>
      have : False := notSat hSat
      exact False.elim this
  have hBright : BloomBrightly := watering_bright_bloom hWatered
  have hBloom : Bloom := bright_implies_bloom hBright
  have hFrag : Fragrance := bloom_implies_fragrance hBloom
  exact fragrance_attracts_butterflies hFrag
</lean>

ANSWER: Yes
