<lean>
axiom Person : Type
axiom John : Person

axiom Waters : Person → Prop
axiom Blooms : Person → Prop
axiom Attracts : Person → Prop
axiom Fertilizes : Person → Prop
axiom SoilHigh : Person → Prop
axiom Healthy : Person → Prop

axiom R1 : ∀ x : Person, Waters x → Blooms x
axiom R2 : ∀ x : Person, Blooms x → Attracts x
axiom R3 : ∀ x : Person, Waters x ∨ Fertilizes x
axiom R4 : ∀ x : Person, Fertilizes x → SoilHigh x
axiom R5 : ∀ x : Person, SoilHigh x → Healthy x

theorem john_soil_high : (¬ Attracts John) → SoilHigh John := by
  intro hNotA
  have hWtoA : Waters John → Attracts John := by
    intro hW
    exact R2 John (R1 John hW)
  have hNotW : ¬ Waters John := by
    intro hW
    apply hNotA
    exact hWtoA hW
  exact Or.elim (R3 John)
    (fun hW => False.elim (hNotW hW))
    (fun hFert => R4 John hFert)
</lean>

ANSWER: Yes
