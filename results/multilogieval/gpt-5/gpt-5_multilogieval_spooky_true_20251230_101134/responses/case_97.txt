<lean>
axiom S : Prop            -- there is a substitute teacher
axiom P : Prop            -- cafeteria serves pizza
axiom G : Prop            -- students goof off in class
axiom H : Prop            -- kids are hyper all afternoon
axiom L : Prop            -- students learn much that day
axiom W : Prop            -- they have lots of homework

axiom A1 : S → G
axiom A2 : P → H
axiom A3 : S ∨ P
axiom A4 : G → ¬ L
axiom A5 : ¬ L → W

theorem not_hyper_homework : ¬ H → W := by
  intro notH
  have notP : ¬ P := fun p => notH (A2 p)
  have s : S := Or.elim A3 (fun s => s) (fun p => False.elim (notP p))
  have g : G := A1 s
  have notL : ¬ L := A4 g
  exact A5 notL
</lean>

ANSWER: Yes
