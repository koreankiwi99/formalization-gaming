<lean>
axiom PracticeHardTeam : Prop
axiom PreparedTeam : Prop
axiom InjuredStar : Prop
axiom CanPlayStar : Prop
axiom WinImportantGame : Prop
axiom TeamDinner : Prop

axiom A1 : PracticeHardTeam → PreparedTeam
axiom A2 : InjuredStar → ¬ CanPlayStar
axiom A3 : PracticeHardTeam ∨ ¬ InjuredStar
axiom A4 : PreparedTeam → WinImportantGame
axiom A5 : WinImportantGame → TeamDinner

theorem dinner_if_injured : InjuredStar → TeamDinner := by
  intro inj
  have not_not_inj : ¬¬InjuredStar := by
    intro hni
    exact hni inj
  have hPractice : PracticeHardTeam := by
    cases A3 with
    | inl hp => exact hp
    | inr hNotInj => exact False.elim (not_not_inj hNotInj)
  have hPrepared : PreparedTeam := A1 hPractice
  have hWin : WinImportantGame := A4 hPrepared
  exact A5 hWin
</lean>

ANSWER: Yes
