<lean>
axiom obj : Type
axiom James : obj
axiom Practice : obj → Prop
axiom Prepared : obj → Prop
axiom Confident : obj → Prop
axiom Club : obj → Prop

axiom R1 : ∀ x : obj, Practice x → Prepared x
axiom R2 : ∀ x : obj, Prepared x → Confident x
axiom R3 : ∀ x : obj, Practice x ∨ Club x

theorem james_joins (h : ¬ Confident James) : Club James := by
  have hpc : Practice James → Confident James :=
    fun hp => R2 James (R1 James hp)
  have hnpr : ¬ Practice James := by
    intro hp
    exact h (hpc hp)
  cases R3 James with
  | inl hp => exact False.elim (hnpr hp)
  | inr hc => exact hc
</lean>

ANSWER: Yes
