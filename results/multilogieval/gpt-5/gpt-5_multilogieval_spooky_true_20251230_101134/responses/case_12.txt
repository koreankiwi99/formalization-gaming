<lean>
axiom Person : Type
axiom James : Person

axiom FollowRecipesPrecisely : Person → Prop
axiom DishesTurnOutWell : Person → Prop
axiom GainConfidence : Person → Prop
axiom WatchCookingShows : Person → Prop

axiom R1 : ∀ x : Person, FollowRecipesPrecisely x → DishesTurnOutWell x
axiom R2 : ∀ x : Person, DishesTurnOutWell x → GainConfidence x
axiom R3 : ∀ x : Person, FollowRecipesPrecisely x ∨ WatchCookingShows x

theorem james_watches_of_not_conf :
  ¬ GainConfidence James → WatchCookingShows James := by
  intro hnc
  have not_fr : ¬ FollowRecipesPrecisely James := by
    intro hfr
    apply hnc
    apply R2 James
    apply R1 James
    exact hfr
  have disj : FollowRecipesPrecisely James ∨ WatchCookingShows James := by
    exact R3 James
  cases disj with
  | inl hfr =>
      have : False := not_fr hfr
      exact False.elim this
  | inr hw =>
      exact hw
</lean>

ANSWER: Yes
