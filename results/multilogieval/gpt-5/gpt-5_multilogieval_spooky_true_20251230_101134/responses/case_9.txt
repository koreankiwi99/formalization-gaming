<lean>
axiom Party : Type
axiom SamParty : Party

axiom Birthday : Party → Prop
axiom Cake : Party → Prop
axiom IceCream : Party → Prop
axiom BringDish : Party → Prop
axiom EnoughFood : Party → Prop

-- If someone brings a dish to share at a party, there will be enough food for everyone.
axiom A1 : ∀ p : Party, BringDish p → EnoughFood p
-- If it is a birthday party, there will be cake and ice cream.
axiom A2 : ∀ p : Party, Birthday p → (Cake p ∧ IceCream p)
-- Either someone brought a dish to share, or there was no birthday cake.
axiom A3 : ∀ p : Party, BringDish p ∨ ¬ Cake p

-- It is Sam's birthday party.
axiom SamIsBirthday : Birthday SamParty

theorem sam_enough_food : EnoughFood SamParty :=
by
  have hc : Cake SamParty := (A2 SamParty SamIsBirthday).left
  have hdisj : BringDish SamParty ∨ ¬ Cake SamParty := A3 SamParty
  cases hdisj with
  | inl hb =>
      exact A1 SamParty hb
  | inr hnc =>
      exact False.elim (hnc hc)
</lean>

ANSWER: Yes
