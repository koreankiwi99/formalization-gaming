      cases hOr with
      | inl hNice => exact False.elim (nNice hNice)
      | inr hNPk => exact hNPk

      exact hAnd.right
  have pr_imp_pk : Prepared Jake → PackedEnough Jake := by
    intro pr
    have rb : ReadBooks Jake := H4 Jake pr
    exact H2 Jake rb
  intro pr
  exact nPk (pr_imp_pk pr)
