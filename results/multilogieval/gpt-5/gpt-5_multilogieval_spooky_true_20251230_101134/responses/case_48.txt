<lean>
axiom Flip : Prop
axiom Gasp : Prop
axiom High : Prop
axiom Clowns : Prop
axiom Laugh : Prop

axiom A1 : Flip → Gasp
axiom A2 : Gasp → High
axiom A3 : Flip ∨ Clowns
axiom A4 : Clowns → Laugh
axiom NotHigh : ¬ High

theorem youngest_kids_laughed : Laugh := by
  have notGasp : ¬ Gasp := by
    intro g
    exact NotHigh (A2 g)
  have notFlip : ¬ Flip := by
    intro f
    exact notGasp (A1 f)
  have cl : Clowns := by
    cases A3 with
    | inl f => exact False.elim (notFlip f)
    | inr c => exact c
  exact A4 cl
</lean>

ANSWER: Yes
