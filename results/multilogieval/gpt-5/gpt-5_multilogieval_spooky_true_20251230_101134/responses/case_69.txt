axiom NotBest_Lebron : ¬ PerformsBest Lebron

-- From the rule and the facts, we can derive that Lebron did NOT train hard.

  intro hTrain
  have hImp : Focused Lebron → PerformsBest Lebron := coaches_rule Lebron hTrain
  have hBest : PerformsBest Lebron := hImp Focused_Lebron
  exact NotBest_Lebron hBest
