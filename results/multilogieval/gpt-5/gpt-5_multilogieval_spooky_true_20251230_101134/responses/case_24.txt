axiom Person : Type
axiom Mary : Person

axiom BelowFreezing : Prop
axiom SnowOnGround : Prop
axiom EnoughSnow : Prop

axiom BuildsSnowman : Person → Prop
axiom LovesWinter : Person → Prop

-- If the temperature dropped below freezing, then there would be snow on the ground.
axiom A1 : BelowFreezing → SnowOnGround
-- If someone builds a snowman, then there is enough snow.
axiom A2 : ∀ x : Person, BuildsSnowman x → EnoughSnow
-- Either it was below freezing, or there was not enough snow, or both.
axiom A3 : BelowFreezing ∨ ¬EnoughSnow ∨ (BelowFreezing ∧ ¬EnoughSnow)
-- If someone loves winter, then they build snowmen.
axiom A4 : ∀ x : Person, LovesWinter x → BuildsSnowman x

-- From no snow on the ground, we can derive that Mary does not love winter.
theorem no_snow_implies_not_love_mary : ¬SnowOnGround → ¬LovesWinter Mary := by
  intro hNoSnow
  -- From ¬SnowOnGround and BelowFreezing → SnowOnGround, get ¬BelowFreezing
  have hNotBF : ¬BelowFreezing := fun hBF => hNoSnow (A1 hBF)
  -- From the disjunction and ¬BelowFreezing, derive ¬EnoughSnow
  have hNotES : ¬EnoughSnow := by
    cases A3 with
    | inl hBF => exact False.elim (hNotBF hBF)
    | inr hrest =>
      cases hrest with
      | inl hNES => exact hNES
      | inr hBoth => exact hBoth.right
  -- Use LovesWinter → BuildsSnowman → EnoughSnow, then modus tollens
  intro hLove
  have hBuild : BuildsSnowman Mary := A4 Mary hLove
  have hES : EnoughSnow := A2 Mary hBuild
  exact hNotES hES
