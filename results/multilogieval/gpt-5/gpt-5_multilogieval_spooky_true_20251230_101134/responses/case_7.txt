<lean>
axiom SoilEnriched : Prop
axiom HealthyPlants : Prop
axiom AbundantHarvest : Prop

-- From the statement: The gardener enriched the soil.
axiom enriched_soil_fact : SoilEnriched

-- From the context: The plants grew strong (so they were healthy).
axiom plants_healthy_fact : HealthyPlants

-- From the context: If plants are healthy, they yield bigger harvests.
axiom healthy_implies_abundant : HealthyPlants → AbundantHarvest

theorem gardener_enriched_soil : SoilEnriched := by
  exact enriched_soil_fact

theorem harvest_was_abundant : AbundantHarvest := by
  apply healthy_implies_abundant
  exact plants_healthy_fact
</lean>

ANSWER: Yes
