<lean>
axiom Person : Type
axiom Alex : Person

axiom StudiedHard : Person → Prop
axiom Participated : Person → Prop
axiom GoodGrades : Person → Prop
axiom MadeFriends : Person → Prop
axiom Invited : Person → Prop

-- Premises for the student (Alex)
axiom S_to_G : StudiedHard Alex → GoodGrades Alex
axiom P_to_F : Participated Alex → MadeFriends Alex
axiom disj_Alex : StudiedHard Alex ∨ Participated Alex
axiom F_to_I : ∀ x : Person, MadeFriends x → Invited x

-- Given: Alex did not get good grades
axiom not_G_Alex : ¬ GoodGrades Alex

theorem alex_invited : Invited Alex := by
  have notS : ¬ StudiedHard Alex := by
    intro hS
    exact not_G_Alex (S_to_G hS)
  have hP : Participated Alex :=
    Or.elim disj_Alex
      (fun hS => False.elim (notS hS))
      (fun hP => hP)
  have hF : MadeFriends Alex := P_to_F hP
  exact F_to_I Alex hF
</lean>

ANSWER: Yes
