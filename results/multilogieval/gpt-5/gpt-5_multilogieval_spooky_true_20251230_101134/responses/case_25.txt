<lean>
inductive Person
| Emma
| Alex
open Person

def Effective : Person → Prop
| Emma => True
| Alex => False

def Converts : Person → Prop
| Emma => True
| Alex => False

def BuildsRapport : Person → Prop
| _ => False

def BuildsTrust : Person → Prop
| _ => False

def Perceptive : Person → Prop
| _ => False

-- Premise: Effective presentations convert more sales
theorem P1 : ∀ x : Person, Effective x → Converts x :=
by
  intro x hx
  cases x
  · exact True.intro
  · cases hx

-- Premise: If a salesperson is perceptive, they build rapport
theorem P2 : ∀ x : Person, Perceptive x → BuildsRapport x :=
by
  intro x h
  cases h

-- Premise: Building rapport leads to trusting relationships
theorem P3 : ∀ x : Person, BuildsRapport x → BuildsTrust x :=
by
  intro x h
  cases h

-- Premise: Not every salesperson builds trust or converts sales
theorem P4 : ∃ x : Person, ¬ (BuildsTrust x ∨ Converts x) :=
by
  refine ⟨Alex, ?_⟩
  intro h
  cases h with
  | inl ht => cases ht
  | inr hc => cases hc

-- Fact from the statement: Emma gives effective sales presentations
theorem Emma_effective : Effective Emma := True.intro

-- Demonstrate that the conclusion "Emma is perceptive" does not follow
theorem not_Emma_perceptive : ¬ Perceptive Emma :=
by
  intro h
  cases h
