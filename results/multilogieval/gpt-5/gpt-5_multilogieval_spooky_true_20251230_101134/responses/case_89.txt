<lean>
axiom Monsoon : Prop
axiom HeavyRains : Prop
axiom WatersDaily : Prop
axiom Waters : Prop
axiom GrowStrong : Prop
axiom Bountiful : Prop
axiom Profit : Prop

axiom R_waters_growth : Waters → GrowStrong
axiom R_monsoon_heavy : Monsoon → HeavyRains
axiom Either_daily_or_not_heavy : WatersDaily ∨ ¬HeavyRains
axiom R_strong_bountiful : GrowStrong → Bountiful
axiom R_bountiful_profit : Bountiful → Profit
axiom Daily_implies_waters : WatersDaily → Waters

theorem monsoon_profit : Monsoon → Profit := by
  intro hMonsoon
  have hHeavy : HeavyRains := R_monsoon_heavy hMonsoon
  have hDaily : WatersDaily :=
    Or.elim Either_daily_or_not_heavy
      (fun w => w)
      (fun nhr => False.elim (nhr hHeavy))
  have hWater : Waters := Daily_implies_waters hDaily
  have hStrong : GrowStrong := R_waters_growth hWater
  have hBountiful : Bountiful := R_strong_bountiful hStrong
  exact R_bountiful_profit hBountiful
</lean>

ANSWER: Yes
