<lean>
axiom obj : Type
axiom Team : obj

axiom PracticesDaily : obj → Prop
axiom Prepared : obj → Prop
axiom Confident : obj → Prop
axiom HeadUpHigh : obj → Prop

axiom Sunny : Prop
axiom KidsAtPark : Prop

axiom A1 : PracticesDaily Team → Prepared Team
axiom A2 : Sunny → KidsAtPark
axiom A3 : PracticesDaily Team ∨ ¬ KidsAtPark
axiom A4 : Prepared Team → Confident Team
axiom A5 : ∀ x : obj, Confident x → HeadUpHigh x

theorem team_head_high_if_sunny : Sunny → HeadUpHigh Team := by
  intro s
  have kp : KidsAtPark := A2 s
  have pd : PracticesDaily Team := by
    cases A3 with
    | inl hp => exact hp
    | inr hnk => exact False.elim (hnk kp)
  have prep : Prepared Team := A1 pd
  have conf : Confident Team := A4 prep
  exact A5 Team conf
</lean>

ANSWER: Yes
