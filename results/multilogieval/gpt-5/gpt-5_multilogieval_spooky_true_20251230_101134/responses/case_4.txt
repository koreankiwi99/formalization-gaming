<lean>
axiom M : Prop          -- invested/focused budget on marketing
axiom C : Prop          -- production costs increased
axiom G : Prop          -- gained more customers
axiom S : Prop          -- sales increased

axiom invest_marketing_implies_more_customers : M → G
axiom more_customers_implies_sales_increase : G → S
axiom marketing_or_costs_increased : M ∨ C

theorem costs_increased_if_sales_not_increase : ¬ S → C :=
by
  intro hNotS
  have hMtoS : M → S := fun hm => more_customers_implies_sales_increase (invest_marketing_implies_more_customers hm)
  have hNotM : ¬ M := fun hm => hNotS (hMtoS hm)
  apply Or.elim marketing_or_costs_increased
  · intro hm
    have : False := hNotM hm
    exact False.elim this
  · intro hc
    exact hc
</lean>

ANSWER: Yes
