<lean>
axiom PracticeOften : Prop
axiom TechniqueImproved : Prop
axiom ArtworkStoodOut : Prop
axiom TookArtClass : Prop
axiom LearnedNewMethods : Prop

axiom R1 : PracticeOften → TechniqueImproved
axiom R2 : TechniqueImproved → ArtworkStoodOut
axiom R3 : PracticeOften ∨ TookArtClass
axiom R4 : TookArtClass → LearnedNewMethods
axiom NotStoodOut : ¬ ArtworkStoodOut

theorem mary_learned : LearnedNewMethods := by
  have hPS : PracticeOften → ArtworkStoodOut := fun hp => R2 (R1 hp)
  have hNotP : ¬ PracticeOften := fun hp => NotStoodOut (hPS hp)
  have hClass : TookArtClass :=
    Or.elim R3
      (fun hp => False.elim (hNotP hp))
      (fun hc => hc)
  exact R4 hClass
</lean>

ANSWER: Yes
