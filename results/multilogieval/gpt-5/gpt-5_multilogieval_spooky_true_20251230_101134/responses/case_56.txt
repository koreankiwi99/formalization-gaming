<lean>
axiom obj : Type
axiom Emma : obj

axiom SpeakDaily : obj → Prop
axiom ListenPod : obj → Prop
axiom Fluency : obj → Prop
axiom Vocab : obj → Prop
axiom Converse : obj → Prop
axiom ListeningComp : obj → Prop

axiom R1a : ∀ x : obj, SpeakDaily x → Fluency x
axiom R1b : ∀ x : obj, SpeakDaily x → Vocab x
axiom R2  : ∀ x : obj, Fluency x → Converse x
axiom R3  : ∀ x : obj, SpeakDaily x ∨ ListenPod x
axiom R4  : ∀ x : obj, ListenPod x → ListeningComp x

theorem emma_listening_if_not_converse : (¬ Converse Emma) → ListeningComp Emma := by
  intro hNotConv
  have hDisj : SpeakDaily Emma ∨ ListenPod Emma := R3 Emma
  have hNotFlu : ¬ Fluency Emma := by
    intro hFlu
    have hConv : Converse Emma := R2 Emma hFlu
    exact hNotConv hConv
  have hNotSpeak : ¬ SpeakDaily Emma := by
    intro hSpeak
    have hFlu : Fluency Emma := R1a Emma hSpeak
    exact hNotFlu hFlu
  cases hDisj with
  | inl hSpeak =>
      exact False.elim (hNotSpeak hSpeak)
  | inr hListen =>
      exact R4 Emma hListen
</lean>

ANSWER: Yes
