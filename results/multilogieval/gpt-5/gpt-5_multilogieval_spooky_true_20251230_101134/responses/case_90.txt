<lean>
axiom Winning : Prop
axiom Rivalry : Prop
axiom LoudEnergized : Prop
axiom EmotionallyInvested : Prop
axiom ElectricAtmosphere : Prop
axiom UnforgettableGame : Prop

axiom H1 : Winning → LoudEnergized
axiom H2 : Rivalry → EmotionallyInvested
axiom H3 : Winning ∨ Rivalry
axiom H4 : LoudEnergized → ElectricAtmosphere
axiom H5 : ElectricAtmosphere → UnforgettableGame

theorem atmosphere_electric_if_not_invested : (¬EmotionallyInvested) → ElectricAtmosphere := by
  intro hNotE
  have hNotR : ¬Rivalry := by
    intro r
    exact hNotE (H2 r)
  have hW : Winning := by
    cases H3 with
    | inl w => exact w
    | inr r => exact False.elim (hNotR r)
  have hL : LoudEnergized := H1 hW
  exact H4 hL
</lean>

ANSWER: Yes
