<lean>
axiom Rain : Prop
axiom Wet : Prop
axiom Washed : Prop
axiom Water : Prop
axiom Uncoiled : Prop
axiom Soapy : Prop

axiom R1 : Wet → Rain
axiom R2 : Washed → Water
axiom R3 : Wet ∨ ¬Water ∨ (Wet ∧ ¬Water)
axiom R4 : Uncoiled → Washed
axiom R5 : Uncoiled ∨ Soapy

theorem sponge_soapy_of_no_rain (hnr : ¬Rain) : Soapy := by
  have hNotWet : ¬Wet := by
    intro hW
    exact hnr (R1 hW)
  have hNotWater : ¬Water := by
    cases R3 with
    | inl hWet =>
        exact False.elim (hNotWet hWet)
    | inr h =>
        cases h with
        | inl hNW => exact hNW
        | inr hpair => exact hpair.right
  have hNotWashed : ¬Washed := by
    intro hWsh
    exact hNotWater (R2 hWsh)
  have hNotUncoiled : ¬Uncoiled := by
    intro hU
    exact hNotWashed (R4 hU)
  cases R5 with
  | inl hU => exact False.elim (hNotUncoiled hU)
  | inr hS => exact hS
</lean>

ANSWER: Yes
