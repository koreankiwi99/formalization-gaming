<lean>
axiom Monsoon : Prop
axiom HeavyRains : Prop
axiom WatersDaily : Prop
axiom Waters : Prop
axiom GrowStrong : Prop
axiom Bountiful : Prop
axiom Profit : Prop

axiom R1 : Waters → GrowStrong
axiom R2 : Monsoon → HeavyRains
axiom R3 : WatersDaily ∨ ¬HeavyRains
axiom R4 : GrowStrong → Bountiful
axiom R5 : Bountiful → Profit
axiom R6 : WatersDaily → Waters

-- Cannot prove ¬(Monsoon → Profit) from the given axioms.
-- In fact, we can derive Monsoon → Profit.
theorem profit_if_monsoon : Monsoon → Profit :=
by
  intro m
  have hr : HeavyRains := R2 m
  have wd : WatersDaily :=
    Or.elim R3
      (fun wd => wd)
      (fun nhr => False.elim (nhr hr))
  have w : Waters := R6 wd
  have gs : GrowStrong := R1 w
  have b : Bountiful := R4 gs
  exact R5 b
