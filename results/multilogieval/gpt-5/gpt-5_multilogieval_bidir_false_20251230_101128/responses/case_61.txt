<lean>
axiom obj : Type
axiom Priya : obj
axiom Attend : obj → Prop
axiom Study : obj → Prop
axiom Excel : obj → Prop

-- If a student attends lectures regularly, then if they study diligently, they will excel in a course.
axiom R1 : ∀ x : obj, Attend x → (Study x → Excel x)

-- Priya studied diligently.
axiom TStudy : Study Priya

-- Priya did not excel in the course.
axiom TNotExcel : ¬Excel Priya

theorem Priya_not_attend : ¬Attend Priya := by
  intro hAttend
  have hImp : Study Priya → Excel Priya := R1 Priya hAttend
  have hExcel : Excel Priya := hImp TStudy
  exact TNotExcel hExcel
</lean>

ANSWER: No
