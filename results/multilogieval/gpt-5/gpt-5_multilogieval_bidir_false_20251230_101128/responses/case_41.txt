<lean>
axiom obj : Type
axiom Plants : obj

axiom WaterDaily : obj → Prop
axiom GrowWell : obj → Prop
axiom NotEnoughWater : obj → Prop
axiom WiltDie : obj → Prop

axiom Drought : Prop
axiom Restrictions : Prop

-- If someone waters their plants daily, the plants will grow well.
axiom R1 : WaterDaily Plants → GrowWell Plants
-- If there is a drought, then water restrictions will be imposed.
axiom R2 : Drought → Restrictions
-- Either they watered the plants daily, or there was a drought.
axiom R3 : WaterDaily Plants ∨ Drought
-- If water restrictions were imposed, then the plants would not get enough water.
axiom R4 : Restrictions → NotEnoughWater Plants
-- If the plants lacked water, they would wilt and die.
axiom R5 : NotEnoughWater Plants → WiltDie Plants

-- From "Priya's plants did not grow well" we can derive they wilted and died.
theorem wilt_die_from_not_grow : ¬GrowWell Plants → WiltDie Plants := by
  intro notGrow
  have notWater : ¬WaterDaily Plants := fun h => notGrow (R1 h)
  have drought : Drought := Or.resolve_left R3 notWater
  have restr : Restrictions := R2 drought
  have notEnough : NotEnoughWater Plants := R4 restr
  exact R5 notEnough
