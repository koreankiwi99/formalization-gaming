<lean>
axiom person : Type
axiom Mike : person

axiom Gas : Prop
axiom DriveLong : Prop
axiom Scenic : person → Prop
axiom See : person → Prop
axiom Love : person → Prop

-- If the car had gas, then they could drive long distances.
axiom R1 : Gas → DriveLong
-- If they see the sights, then they are taking a scenic route.
axiom R2 : ∀ x : person, See x → Scenic x
-- Either the car had gas, or they were not taking the scenic route, or both.
axiom Or1 : Gas ∨ ¬Scenic Mike ∨ (Gas ∧ ¬Scenic Mike)
-- If someone loves road trips, they take the scenic route to see the sights.
axiom Common : ∀ x : person, Love x → (Scenic x ∧ See x)

-- From the premises, if they cannot drive long distances, then Mike does NOT love road trips.
theorem not_drive_implies_not_love : ¬DriveLong → ¬Love Mike := by
  intro hNotD
  intro hLove
  -- From loving road trips, Mike takes the scenic route (and sees the sights).
  have hScenic : Scenic Mike := (Common Mike hLove).left
  -- Use the disjunction to derive Gas, given that Mike is on the scenic route.
  have hGas : Gas := by
    cases Or1 with
    | inl hG => exact hG
    | inr rest =>
      cases rest with
      | inl hNotScenic =>
        exact False.elim (hNotScenic hScenic)
      | inr hBoth =>
        exact hBoth.left
  -- Gas implies ability to drive long distances, contradicting ¬DriveLong.
  have hD : DriveLong := R1 hGas
  exact hNotD hD
</lean>
