axiom MadeFriends : person → Prop
axiom Enjoy : person → Prop
axiom JoinClub : person → Prop
axiom Busy : person → Prop
axiom Outgoing : person → Prop
axiom Introduced : person → Prop

axiom A1 : ∀ x : person, MadeFriends x → Enjoy x
axiom A2 : ∀ x : person, JoinClub x → Busy x
axiom A3 : ∀ x : person, MadeFriends x ∨ ¬Busy x
axiom A4 : ∀ x : person, Outgoing x → JoinClub x
axiom A5 : ∀ x : person, Introduced x → Outgoing x

theorem not_enjoy_implies_not_introduced : ¬Enjoy Mark → ¬Introduced Mark := by
  intro hNotEnjoy
  -- from ¬Enjoy Mark and MadeFriends → Enjoy, get ¬MadeFriends Mark
  have hNotMF : ¬MadeFriends Mark := by
    intro hMF
    have hEn : Enjoy Mark := A1 Mark hMF
    exact hNotEnjoy hEn
  -- From MadeFriends Mark ∨ ¬Busy Mark, and ¬MadeFriends Mark, deduce ¬Busy Mark
  have hNotBusy : ¬Busy Mark := by
    have hDisj : MadeFriends Mark ∨ ¬Busy Mark := A3 Mark
    cases hDisj with
    | inl hMF =>
      have contra : False := hNotMF hMF
      exact False.elim contra
    | inr hNB => exact hNB
  -- From Introduced → Outgoing → JoinClub → Busy, contradict ¬Busy to get ¬Introduced
  intro hIntro
  have hOut : Outgoing Mark := A5 Mark hIntro
  have hJoin : JoinClub Mark := A4 Mark hOut
  have hBusy : Busy Mark := A2 Mark hJoin
  exact hNotBusy hBusy
