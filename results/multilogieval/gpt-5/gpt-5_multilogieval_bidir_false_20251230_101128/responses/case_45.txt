axiom obj : Type
axiom Matt : obj

axiom Watched : obj → Prop
axiom Skills : obj → Prop
axiom Redo : obj → Prop
axiom ReadBlogs : obj → Prop
axiom Ideas : obj → Prop

-- If someone watched DIY videos, they gain decorating skills
axiom R1 : ∀ x : obj, Watched x → Skills x
-- With these skills, they can redo their living room
axiom R2 : ∀ x : obj, Skills x → Redo x
-- Matt did not redo his living room
axiom T1 : ¬Redo Matt
-- Either Matt watched DIY videos or he read interior design blogs (or both)
axiom E1 : Watched Matt ∨ ReadBlogs Matt
-- If someone reads interior design blogs, then they get new decorating ideas
axiom R3 : ∀ x : obj, ReadBlogs x → Ideas x

theorem watched_implies_redo : ∀ x : obj, Watched x → Redo x := by
  intro x
  intro hW
  have hS : Skills x := R1 x hW
  exact R2 x hS

theorem not_watched_matt : ¬Watched Matt := by
  intro hW
  have hR : Redo Matt := by
    have hS : Skills Matt := R1 Matt hW
    exact R2 Matt hS
  exact T1 hR

theorem read_matt : ReadBlogs Matt := by
  have h := E1
  cases h with
  | inl hW =>
    exact False.elim (not_watched_matt hW)
  | inr hR =>
    exact hR

theorem ideas_matt : Ideas Matt := by
  apply R3 Matt
  exact read_matt
