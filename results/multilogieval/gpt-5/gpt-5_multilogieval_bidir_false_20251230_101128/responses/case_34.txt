axiom Collaborates : obj → Prop
axiom Published : obj → Prop

axiom Recognized : obj → Prop

axiom R_collab_rec : ∀ x : obj, Collaborates x → Recognized x

axiom R_either : ∀ x : obj, Professor x → (¬Published x ∨ ¬Collaborates x)
axiom R_cur_ask_pub : ∀ x : obj, (Curious x ∧ Asks x) → Published x

axiom T_collab : Collaborates DrLee

theorem drlee_not_curious_and_asks : ¬(Curious DrLee ∧ Asks DrLee) := by
  have hdisj : ¬Published DrLee ∨ ¬Collaborates DrLee := R_either DrLee T_prof
  have hnpub : ¬Published DrLee := by
    cases hdisj with
    | inl hnp => exact hnp
    | inr hnc =>
        intro hp
        have contra : False := hnc T_collab
        exact False.elim contra
  intro hCA
  have hpub : Published DrLee := R_cur_ask_pub DrLee hCA
