<lean>
axiom obj : Type
axiom James : obj

axiom Practice : obj → Prop
axiom Prepared : obj → Prop
axiom Confident : obj → Prop
axiom Club : obj → Prop

-- If someone practices thoroughly, they feel prepared.
axiom R1 : ∀ x : obj, Practice x → Prepared x
-- If someone feels prepared, they can speak confidently.
axiom R2 : ∀ x : obj, Prepared x → Confident x
-- Either they practiced, or they joined a club, or both.
axiom R3 : ∀ x : obj, Practice x ∨ Club x ∨ (Practice x ∧ Club x)

-- Given: James cannot speak confidently in public.
axiom Hnc : ¬Confident James

-- From ¬Confident and Prepared → Confident, we get ¬Prepared for James.
theorem not_prepared_James : ¬Prepared James :=
  fun hprep => Hnc (R2 James hprep)

-- From ¬Prepared and Practice → Prepared, we get ¬Practice for James.
theorem not_practice_James : ¬Practice James :=
  fun hprac => not_prepared_James (R1 James hprac)

-- From the disjunction and ¬Practice, James must have joined a club.
theorem club_James : Club James := by
  have d := R3 James
  cases d with
  | inl hP =>
      exact False.elim (not_practice_James hP)
  | inr rest =>
      cases rest with
      | inl hC => exact hC
      | inr hPC =>
          exact False.elim (not_practice_James hPC.left)
