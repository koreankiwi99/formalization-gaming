axiom Focused : obj → Prop

axiom Rule : ∀ x : obj, TrainsHard x → Focused x → PerformsBest x

axiom NotBest : ¬PerformsBest Lebron

theorem lebron_not_trainshard : ¬TrainsHard Lebron := by
  intro hTH
  have hBest : PerformsBest Lebron := Rule Lebron hTH T_focus
  exact NotBest hBest
