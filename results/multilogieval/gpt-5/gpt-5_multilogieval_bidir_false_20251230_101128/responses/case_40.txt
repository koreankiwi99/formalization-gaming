axiom Motivated : obj → Prop
axiom Closed : obj → Prop
axiom CannotFollow : obj → Prop
axiom Frustrated : obj → Prop
axiom Indulged : obj → Prop

-- If someone starts a new workout routine, they will be motivated to keep it up.
axiom R1 : ∀ x : obj, Started x → Motivated x
-- If the gym closes for renovations, then they can't follow their routine.
axiom R2 : ∀ x : obj, Closed x → CannotFollow x
-- Either they started a new workout, or the gym was closed (or both).
axiom R3 : ∀ x : obj, Started x ∨ Closed x
-- If they couldn't follow their routine, they would feel frustrated.
axiom R4 : ∀ x : obj, CannotFollow x → Frustrated x
-- If they felt frustrated, they might indulge in some unhealthy snacks.
axiom R5 : ∀ x : obj, Frustrated x → Indulged x

-- Statement to evaluate: James was not motivated to workout.
axiom NotMotivatedJames : ¬Motivated James

-- From the premises we can actually derive that James indulged in unhealthy snacks.
theorem indulged_james : Indulged James := by
  have hdisj : Started James ∨ Closed James := R3 James
  have hns : ¬Started James := by
    intro hs
    exact NotMotivatedJames (R1 James hs)
  have hclosed : Closed James :=
    Or.elim hdisj
      (fun hs => False.elim (hns hs))
      (fun hc => hc)
  have hcant : CannotFollow James := R2 James hclosed
  have hfr : Frustrated James := R4 James hcant
  exact R5 James hfr
