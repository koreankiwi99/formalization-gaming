<lean>
axiom obj : Type
axiom Shelf : obj

axiom HighQuality : obj → Prop
axiom PreciseJoints : obj → Prop
axiom Sturdy : obj → Prop
axiom LongLasting : obj → Prop

-- If the wood is high quality, then if the joints are precise, the product is sturdy and long-lasting
axiom R1 : ∀ x : obj, HighQuality x → (PreciseJoints x → (Sturdy x ∧ LongLasting x))

-- The joints for the shelf were cut precisely
axiom T1 : PreciseJoints Shelf
-- The shelf was shaky (not sturdy)
axiom T2 : ¬Sturdy Shelf
-- The shelf fell apart quickly (not long-lasting)
axiom T3 : ¬LongLasting Shelf

theorem shelf_wood_not_high_quality : ¬HighQuality Shelf := by
  intro hq
  have hconj : Sturdy Shelf ∧ LongLasting Shelf := R1 Shelf hq T1
  have hs : Sturdy Shelf := And.left hconj
  exact T2 hs
</lean>

ANSWER: No
