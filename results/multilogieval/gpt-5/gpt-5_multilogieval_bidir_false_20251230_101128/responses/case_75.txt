<lean>
axiom obj : Type
axiom Liz : obj
axiom PM : obj → Prop        -- Preventive maintenance done
axiom QP : obj → Prop        -- Quality parts used
axiom LL : obj → Prop        -- Long service life

-- If preventive maintenance is done, then if quality parts are used, the car will have a long service life.
axiom R1 : ∀ x : obj, PM x → (QP x → LL x)

-- Facts about Liz's car
axiom T1 : QP Liz
axiom T2 : ¬LL Liz

theorem no_PM_Liz : ¬PM Liz := by
  intro hPM
  have hImp : QP Liz → LL Liz := R1 Liz hPM
  have hLL : LL Liz := hImp T1
  exact T2 hLL
</lean>

ANSWER: No
