<lean>
axiom person : Type
axiom exam : Type

axiom Emma : person
axiom BioExam : exam

axiom StudyHard : person → exam → Prop
axiom EnoughSleep : person → exam → Prop
axiom PerformWell : person → exam → Prop
axiom AceTest : person → exam → Prop

-- Belief: if someone studies hard, then if they get enough sleep before an exam,
-- they will perform well and ace the test.
axiom Belief :
  ∀ x : person, ∀ e : exam,
    StudyHard x e → (EnoughSleep x e → (PerformWell x e ∧ AceTest x e))

-- Facts about Emma and her biology exam
axiom SleepEmmaBio : EnoughSleep Emma BioExam
axiom NotAceEmmaBio : ¬AceTest Emma BioExam

-- From the belief, Emma's sleep, and the fact she did not ace the exam,
-- we can conclude that Emma did not study hard for the biology exam.
theorem emma_not_study_hard : ¬StudyHard Emma BioExam := by
  intro hStudy
  have hImp := Belief Emma BioExam hStudy
  have hConj := hImp SleepEmmaBio
  have hAce : AceTest Emma BioExam := And.right hConj
  exact NotAceEmmaBio hAce
</lean>

ANSWER: No
