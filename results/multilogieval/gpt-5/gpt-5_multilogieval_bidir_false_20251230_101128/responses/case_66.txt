<lean>
axiom obj : Type
axiom Acme : obj
axiom Innovative : obj → Prop
axiom MarketedEffectively : obj → Prop
axiom CommerciallySuccessful : obj → Prop
-- If a company develops an innovative product, then if they market it effectively, the product will be commercially successful.
axiom Rule : ∀ x : obj, Innovative x → (MarketedEffectively x → CommerciallySuccessful x)
-- Given facts about Acme
axiom MAcme : MarketedEffectively Acme
axiom NSAcme : ¬CommerciallySuccessful Acme

theorem acme_not_innovative : ¬Innovative Acme := by
  intro hInnov
  have hImp : MarketedEffectively Acme → CommerciallySuccessful Acme := Rule Acme hInnov
  have hSucc : CommerciallySuccessful Acme := hImp MAcme
  exact NSAcme hSucc
</lean>

ANSWER: No
