<lean>
axiom Alarm : Prop
axiom TimeToGetReady : Prop
axiom MissBus : Prop
axiom LateForSchool : Prop
axiom GetUpLate : Prop
axiom WentToBedLate : Prop

-- If the alarm went off, then it was time to get ready.
axiom R1 : Alarm → TimeToGetReady
-- If they missed the bus, then they'd be late for school.
axiom R2 : MissBus → LateForSchool
-- If they got up late, then they miss the bus.
axiom R3 : GetUpLate → MissBus
-- If they went to bed late, then they get up late.
axiom R4 : WentToBedLate → GetUpLate
-- Either the alarm went off, or they weren't late for school.
axiom R5 : Alarm ∨ ¬LateForSchool

-- John knows it is not the time to get ready (treated as ¬TimeToGetReady).
axiom NotTime : ¬TimeToGetReady

theorem no_bed_late : ¬WentToBedLate := by
  intro hBed
  have hGet : GetUpLate := R4 hBed
  have hMiss : MissBus := R3 hGet
  have hLate : LateForSchool := R2 hMiss
  have hAlarm : Alarm :=
    Or.elim R5
      (fun hA => hA)
      (fun hNotLate => False.elim (hNotLate hLate))
  have hTime : TimeToGetReady := R1 hAlarm
  exact NotTime hTime
</lean>

ANSWER: No
