-- The given premises actually prove the statement:
-- If there was no noise level up, then the police were called.
theorem from_no_up_police : (¬NoiseUp) → Police := by
  intro hNoUp
  have hNotDrums : ¬Drums := by
    intro hD
    have hUp : NoiseUp := Drums_NoiseUp hD
    exact hNoUp hUp
  have hParty : Party := by
    cases Party_or_Drums with
    | inl hP => exact hP
    | inr hD => exact False.elim (hNotDrums hD)
  have hC : Complain := Party_Complain hParty
  exact Complain_Police hC
