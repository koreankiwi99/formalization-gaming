<lean>
axiom Monsoon : Prop
axiom HeavyRains : Prop
axiom Waters : Prop
axiom Strong : Prop
axiom Bountiful : Prop
axiom GoodProfit : Prop

axiom A1 : Waters → Strong
axiom A2 : Monsoon → HeavyRains
axiom A3 : Waters ∨ ¬ HeavyRains
axiom A4 : Strong → Bountiful
axiom A5 : Bountiful → GoodProfit

theorem monsoon_profit : Monsoon → GoodProfit := by
  intro m
  have hr : HeavyRains := A2 m
  have hdisj : Waters ∨ ¬ HeavyRains := A3
  cases hdisj with
  | inl w =>
      have s : Strong := A1 w
      have b : Bountiful := A4 s
      exact A5 b
  | inr nh =>
      have contra : False := nh hr
      exact False.elim contra
</lean>

ANSWER: Yes
