<lean>
axiom Watered : Prop
axiom Saturated : Prop
axiom Rain : Prop
axiom Bloom : Prop
axiom Fragrance : Prop
axiom Attracts : Prop

-- If the garden is watered regularly, the flowers will bloom.
axiom A1 : Watered → Bloom
-- If it rains a lot, the soil becomes saturated.
axiom A2 : Rain → Saturated
-- Either the garden is watered routinely, or the soil did not saturate.
axiom A3 : Watered ∨ ¬ Saturated
-- If the flowers bloom, they will release fragrance.
axiom A4 : Bloom → Fragrance
-- If there is fragrance, it will attract butterflies.
axiom A5 : Fragrance → Attracts

theorem heavy_rain_attracts : Rain → Attracts := by
  intro hr
  have hs : Saturated := A2 hr
  have hw : Watered := Or.elim A3 (fun hw => hw) (fun hns => False.elim (hns hs))
  have hb : Bloom := A1 hw
  have hf : Fragrance := A4 hb
  exact A5 hf
</lean>

ANSWER: Yes
