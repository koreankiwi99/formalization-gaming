<lean>
axiom Person : Type
axiom John : Person

axiom Rain : Prop
axiom Cold : Prop

axiom HotShower : Person → Prop
axiom Relaxed : Person → Prop
axiom Sore : Person → Prop
axiom Hard : Person → Prop

-- If someone worked out hard, they would be sore the next day.
axiom R1 : ∀ x : Person, Hard x → Sore x
-- If someone takes a hot shower, their muscles will relax.
axiom R2 : ∀ x : Person, HotShower x → Relaxed x
-- Either they worked out hard, or their muscles did not relax (for any person).
axiom R3 : ∀ x : Person, Hard x ∨ ¬ Relaxed x
-- If the weather is cold, everyone takes a hot shower.
axiom R4 : ∀ x : Person, Cold → HotShower x
-- If it rains, then the weather is cold.
axiom R5 : Rain → Cold

-- From the premises we can derive: if it rains, John's muscles are sore.
theorem rain_implies_sore_john : Rain → Sore John := by
  intro hRain
  have hCold : Cold := R5 hRain
  have hHot : HotShower John := (R4 John) hCold
  have hRelax : Relaxed John := R2 John hHot
  have hHard : Hard John := by
    have disj : Hard John ∨ ¬ Relaxed John := R3 John
    cases disj with
    | inl hH => exact hH
    | inr hNotRelax =>
        have : False := hNotRelax hRelax
        exact False.elim this
  exact R1 John hHard

-- Hence, if John's muscles are not sore today, then it did not rain.
theorem not_sore_implies_not_rain : ¬ Sore John → ¬ Rain := by
  intro hNotSore
  intro hRain
  have hSore : Sore John := rain_implies_sore_john hRain
  exact hNotSore hSore
