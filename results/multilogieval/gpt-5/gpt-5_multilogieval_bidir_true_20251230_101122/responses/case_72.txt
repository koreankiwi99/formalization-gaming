axiom obj : Type
axiom Sam : obj

-- Propositions and predicates
axiom CloseDinner : Prop
axiom StartCooking : obj → Prop
axiom GoShopping : obj → Prop
axiom MakeDinner : obj → Prop
axiom FridgeEmpty : Prop
axiom PantryEmpty : Prop

-- Axioms from the context
-- If it is close to dinnertime, then they should start cooking.
axiom R1 : CloseDinner → StartCooking Sam
-- They also needed to go grocery shopping. (Model as: they go shopping.)
axiom TShop : GoShopping Sam
-- If they go shopping, then they can make dinner.
axiom R2 : GoShopping Sam → MakeDinner Sam
-- Either it is close to dinnertime, or they won't make dinner.
axiom R3 : CloseDinner ∨ ¬ MakeDinner Sam
-- If the fridge is empty, then they must go shopping.
axiom R4 : FridgeEmpty → GoShopping Sam
-- If there are no ingredients in the pantry, then the fridge is empty.
axiom R5 : PantryEmpty → FridgeEmpty

-- We can derive that Sam does start cooking (but this does not imply the pantry is empty).
theorem sam_starts_cooking : StartCooking Sam := by
  have md : MakeDinner Sam := R2 TShop
  have cd : CloseDinner := by
    cases R3 with
    | inl h => exact h
    | inr h => exact False.elim (h md)
  exact R1 cd

-- There is no rule that connects starting to cook (or even being close to dinnertime or making dinner)
-- to the pantry being empty; the only direction provided is PantryEmpty → FridgeEmpty → GoShopping,
-- not the converse.
