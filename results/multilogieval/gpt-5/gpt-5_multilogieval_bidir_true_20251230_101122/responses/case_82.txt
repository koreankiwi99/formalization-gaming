<lean>
axiom Person : Type
axiom James : Person

axiom RipePeaches : Prop
axiom BakesPie : Person → Prop
axiom Won : Person → Prop
axiom Celebrate : Person → Prop
axiom HouseSmellsAmazing : Prop
axiom HomeyCozy : Prop

-- If the farmer's market has ripe, juicy peaches, someone will bake a peach pie.
axiom A1 : RipePeaches → ∃ p : Person, BakesPie p
-- If someone won a baking contest, they'll celebrate with friends and family.
axiom A2 : ∀ p : Person, Won p → Celebrate p
-- Either peaches are in season, or someone won a contest (or both).
axiom A3 : RipePeaches ∨ ∃ p : Person, Won p
-- If someone bakes a pie, the house will smell amazing.
axiom A4 : ∀ p : Person, BakesPie p → HouseSmellsAmazing
-- If the house smells amazing, it feels like a homey, cozy place to relax.
axiom A5 : HouseSmellsAmazing → HomeyCozy

-- From ripe peaches, we can derive the house feels homey and cozy.
theorem ripe_implies_homey : RipePeaches → HomeyCozy :=
  fun rp =>
    match A1 rp with
    | Exists.intro p bp =>
        A5 (A4 p bp)

-- Cannot prove (¬Celebrate James → HomeyCozy) from the given axioms:
-- the disjunction A3 allows the "won contest" branch without implying ripe peaches,
-- and James not celebrating does not eliminate the existence of some other winner.
</lean>
