<lean>
axiom obj : Type
axiom John : obj

axiom Waters : obj → Prop
axiom Blooms : obj → Prop
axiom Attracts : obj → Prop
axiom UsesOrganic : obj → Prop
axiom SoilHigh : obj → Prop
axiom Healthy : obj → Prop

-- If someone waters adequately, plants bloom
axiom R1 : ∀ x : obj, Waters x → Blooms x
-- If plants bloom, they attract more bees and butterflies
axiom R2 : ∀ x : obj, Blooms x → Attracts x
-- Either someone waters adequately or uses organic fertilizers (or both)
axiom R3 : ∀ x : obj, Waters x ∨ UsesOrganic x
-- If someone uses organic fertilizers, the soil quality improves (is high)
axiom R4 : ∀ x : obj, UsesOrganic x → SoilHigh x
-- If the soil quality improves, the plants grow healthier
axiom R5 : ∀ x : obj, SoilHigh x → Healthy x

theorem john_soil_high_if_not_attracts : (¬ Attracts John) → SoilHigh John := by
  intro hNotAttract
  have hOr : Waters John ∨ UsesOrganic John := R3 John
  have hNotWater : ¬ Waters John := by
    intro hWater
    have hBloom : Blooms John := R1 John hWater
    have hAttr : Attracts John := R2 John hBloom
    exact hNotAttract hAttr
  have hUses : UsesOrganic John := by
    cases hOr with
    | inl hW => exact False.elim (hNotWater hW)
    | inr hF => exact hF
  exact R4 John hUses
</lean>

ANSWER: Yes
