<lean>
axiom Sub : Prop
axiom Goof : Prop
axiom Pizza : Prop
axiom Hyper : Prop
axiom Learn : Prop
axiom Homework : Prop

-- If there's a substitute teacher, the students will goof off.
axiom R1 : Sub → Goof
-- If the cafeteria serves pizza for lunch, the kids will be hyper all afternoon.
axiom R2 : Pizza → Hyper
-- Either there will be a sub, or they'll have pizza (or both).
axiom T1 : Sub ∨ Pizza
-- If the students goof off in class, they won't learn much that day.
axiom R3 : Goof → ¬Learn
-- If they don't learn much, they'll have lots of homework.
axiom R4 : ¬Learn → Homework

-- Statement: If kids will not be hyper all afternoon, then the students have lots of homework.
theorem not_hyper_homework : ¬Hyper → Homework :=
by
  intro nh
  -- From ¬Hyper and (Pizza → Hyper), deduce ¬Pizza
  have np : ¬Pizza := by
    intro p
    have h : Hyper := R2 p
    exact nh h
  -- From Sub ∨ Pizza and ¬Pizza, deduce Sub
  have hs : Sub := by
    cases T1 with
    | inl hs => exact hs
    | inr hp => exact False.elim (np hp)
  -- Chain the implications to get Homework
  have g : Goof := R1 hs
  have nl : ¬Learn := R3 g
  exact R4 nl
</lean>

ANSWER: Yes
