<lean>
axiom obj : Type
axiom Sarah : obj

axiom Practice : obj → Prop
axiom Jam : obj → Prop
axiom Inspired : obj → Prop
axiom Flourish : obj → Prop
axiom Album : obj → Prop

-- If someone practiced new melodies daily, they could compose an album.
axiom R1 : ∀ x : obj, Practice x → Album x
-- Either Sarah practiced new melodies or she jammed with friends (or both).
axiom D1 : Practice Sarah ∨ Jam Sarah
-- If someone jammed with friends, they felt inspired.
axiom R2 : ∀ x : obj, Jam x → Inspired x
-- If someone felt inspired, their creativity flourished.
axiom R3 : ∀ x : obj, Inspired x → Flourish x

theorem sarah_flourish_if_no_album : ¬ Album Sarah → Flourish Sarah := by
  intro hNotAlbum
  have hNotPractice : ¬ Practice Sarah := by
    intro hPractice
    exact hNotAlbum (R1 Sarah hPractice)
  refine Or.elim D1 (fun hPractice => ?contra) (fun hJam => ?pos)
  · have : False := hNotPractice hPractice
    exact False.elim this
  · exact R3 Sarah (R2 Sarah hJam)
</lean>

ANSWER: Yes
