<lean>
axiom obj : Type
axiom Sam : obj

axiom StoveOn : Prop
axiom KitchenHot : Prop
axiom UsesStove : obj → Prop
axiom MakesPancakes : obj → Prop
axiom Hungry : obj → Prop
axiom AteDinner : obj → Prop

-- If the stove is on, the kitchen gets hot.
axiom R_on_hot : StoveOn → KitchenHot
-- If someone makes pancakes, they use the stove.
axiom R_panc_use : ∀ x : obj, MakesPancakes x → UsesStove x
-- If someone is hungry, they make pancakes.
axiom R_hungry_panc : ∀ x : obj, Hungry x → MakesPancakes x
-- Either the stove was on, or they (that person) did not use the stove.
axiom disj : ∀ x : obj, StoveOn ∨ ¬UsesStove x
-- If Sam did not eat dinner last night, he was hungry this morning.
axiom R_noDinner_hungrySam : ¬AteDinner Sam → Hungry Sam

theorem sam_ate_dinner_of_not_hot : ¬KitchenHot → AteDinner Sam := fun notHot => by
  classical
  by_contra hNoDinner
  have hun : Hungry Sam := R_noDinner_hungrySam hNoDinner
  have panc : MakesPancakes Sam := R_hungry_panc Sam hun
  have use : UsesStove Sam := R_panc_use Sam panc
  have d := disj Sam
  have on : StoveOn := by
    cases d with
    | inl hOn => exact hOn
    | inr hNotUse => exact False.elim (hNotUse use)
  have hot : KitchenHot := R_on_hot on
  exact notHot hot
</lean>

ANSWER: Yes
