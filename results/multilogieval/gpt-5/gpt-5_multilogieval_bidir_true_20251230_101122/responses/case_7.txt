axiom GardenerEnrichedSoil : Prop
axiom ProducedLotsOfVegetables : Prop
axiom AbundantHarvest : Prop

-- From the text:
-- The gardener enriched the soil.
axiom T1 : GardenerEnrichedSoil
-- The plants produced lots of vegetables.
axiom T2 : ProducedLotsOfVegetables
-- If plants produce lots of vegetables, the harvest is abundant.
axiom R1 : ProducedLotsOfVegetables → AbundantHarvest

theorem gardener_enriched : GardenerEnrichedSoil := by
  exact T1

theorem harvest_abundant : AbundantHarvest := by
  apply R1
  exact T2
