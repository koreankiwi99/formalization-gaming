<lean>
axiom obj : Type
axiom Sam : obj

axiom Impress : obj → Prop
axiom Promotion : obj → Prop
axiom ExtraWork : obj → Prop
axiom Experience : obj → Prop
axiom LeadTeam : obj → Prop

axiom R1 : ∀ x : obj, Impress x → Promotion x
axiom R2 : ∀ x : obj, ExtraWork x → Experience x
axiom R3 : ∀ x : obj, Experience x → LeadTeam x

axiom T1 : Impress Sam ∨ ExtraWork Sam
axiom T4 : ¬ Promotion Sam

theorem sam_assigned_team : LeadTeam Sam := by
  have notImpress : ¬ Impress Sam := fun hi => T4 (R1 Sam hi)
  have extra : ExtraWork Sam := by
    cases T1 with
    | inl hi =>
      have hFalse : False := T4 (R1 Sam hi)
      exact False.elim hFalse
    | inr he =>
      exact he
  have exp : Experience Sam := R2 Sam extra
  exact R3 Sam exp
</lean>

ANSWER: Yes
