<lean>
axiom Party : Prop
axiom Drums : Prop
axiom NoiseUp : Prop
axiom NoisyFromParty : Prop
axiom NeighborsComplain : Prop
axiom PoliceCalled : Prop

axiom R1 : Party → NoisyFromParty
axiom R2 : Drums → NoiseUp
axiom T1 : Party ∨ Drums
axiom R3 : NoisyFromParty → NeighborsComplain
axiom R4 : NeighborsComplain → PoliceCalled

theorem police_called_if_no_noiseup : (¬ NoiseUp) → PoliceCalled :=
fun hNoNoise =>
  have hNoDrums : ¬ Drums := fun hd => hNoNoise (R2 hd)
  have hParty : Party := Or.elim T1 (fun hp => hp) (fun hd => False.elim (hNoDrums hd))
  have hNoisyFromParty : NoisyFromParty := R1 hParty
  have hComplaints : NeighborsComplain := R3 hNoisyFromParty
  show PoliceCalled from R4 hComplaints
</lean>

ANSWER: Yes
