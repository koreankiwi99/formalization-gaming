<lean>
axiom obj : Type
axiom Sarah : obj

axiom Practice : obj → Prop
axiom ImproveTechnique : obj → Prop
axiom LearnFaster : obj → Prop
axiom Listen : obj → Prop

axiom A1 : ∀ x : obj, Practice x → ImproveTechnique x
axiom A2 : ∀ x : obj, ImproveTechnique x → LearnFaster x
axiom A3 : ∀ x : obj, Practice x ∨ Listen x

axiom Hnot : ¬ LearnFaster Sarah

theorem sarah_listens : Listen Sarah := by
  have hnotP : ¬ Practice Sarah := by
    intro hP
    have hI : ImproveTechnique Sarah := A1 Sarah hP
    have hL : LearnFaster Sarah := A2 Sarah hI
    exact Hnot hL
  have hdisj : Practice Sarah ∨ Listen Sarah := A3 Sarah
  exact hdisj.elim (fun hP => False.elim (hnotP hP)) (fun hL => hL)
</lean>

ANSWER: Yes
