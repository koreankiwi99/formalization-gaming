axiom James : obj

axiom Friday : Prop
axiom GoingParty : obj → Prop
axiom WatchGame : obj → Prop
axiom FinishHW : obj → Prop
axiom StudyEveryday : obj → Prop
axiom PlanWell : obj → Prop

-- If it is Friday night, then they are going to the party.
axiom R1 : ∀ x : obj, Friday → GoingParty x
-- If they finish all their homework, then they can watch the game.
axiom R2 : ∀ x : obj, FinishHW x → WatchGame x
-- Either it is Friday night, or they are not watching the game.
axiom R3 : ∀ x : obj, Friday ∨ ¬WatchGame x
-- If they study everyday, they finish all their homework.
axiom R4 : ∀ x : obj, StudyEveryday x → FinishHW x
-- If someone plans their schedule well, then they can study everyday.
axiom R5 : ∀ x : obj, PlanWell x → StudyEveryday x

-- Given: James is not going to the party.
axiom TJamesNotParty : ¬GoingParty James

-- From the premises, we can derive that James did NOT plan his schedule well.
theorem james_not_plan : ¬PlanWell James := by
  intro hPlan
  have hStudy : StudyEveryday James := R5 James hPlan
  have hFinish : FinishHW James := R4 James hStudy
  have hWatch : WatchGame James := R2 James hFinish
  have hNotFriday : ¬Friday := by
    intro hFri
    have hGo : GoingParty James := R1 James hFri
    exact TJamesNotParty hGo
  have hFri : Friday := by
    have disj := R3 James
    cases disj with
    | inl hf => exact hf
    | inr hNotWatch => exact False.elim (hNotWatch hWatch)
  exact hNotFriday hFri
