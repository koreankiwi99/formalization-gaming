<lean>
axiom obj : Type
axiom Mary : obj

axiom Practice : obj → Prop
axiom Improve : obj → Prop
axiom HighQuality : obj → Prop
axiom TookClass : obj → Prop
axiom LearnedMethods : obj → Prop

axiom A1 : ∀ x : obj, Practice x → Improve x
axiom A2 : ∀ x : obj, Improve x → HighQuality x
axiom A3 : ∀ x : obj, Practice x ∨ TookClass x
axiom A4 : ∀ x : obj, TookClass x → LearnedMethods x
axiom A5 : ¬ HighQuality Mary

theorem mary_learned : LearnedMethods Mary := by
  have hp_to_hq : Practice Mary → HighQuality Mary := fun hp => A2 Mary (A1 Mary hp)
  have not_p : ¬ Practice Mary := fun hp => A5 (hp_to_hq hp)
  have disj : Practice Mary ∨ TookClass Mary := A3 Mary
  cases disj with
  | inl hp =>
    exact False.elim (not_p hp)
  | inr hc =>
    exact A4 Mary hc
</lean>

ANSWER: Yes
