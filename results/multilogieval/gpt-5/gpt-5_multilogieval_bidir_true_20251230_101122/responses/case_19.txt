<lean>
axiom obj : Type
axiom Store : obj

axiom GreatService : obj → Prop
axiom HostsEvents : obj → Prop
axiom MoreCustomers : obj → Prop
axiom ReputationGrows : obj → Prop
axiom RevenueImproves : obj → Prop

axiom A1 : ∀ x : obj, GreatService x → ReputationGrows x
axiom A2 : ∀ x : obj, HostsEvents x → MoreCustomers x
axiom A3 : ∀ x : obj, MoreCustomers x → RevenueImproves x

axiom disj : GreatService Store ∨ HostsEvents Store
axiom not_service : ¬ GreatService Store

theorem revenue_improves : RevenueImproves Store := by
  have hEvents : HostsEvents Store := by
    cases disj
    case inl hG =>
      exact False.elim (not_service hG)
    case inr hE =>
      exact hE
  have hMore : MoreCustomers Store := A2 Store hEvents
  exact A3 Store hMore
</lean>

ANSWER: Yes
