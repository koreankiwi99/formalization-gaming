<lean>
axiom obj : Type
axiom Priya : obj

axiom StudiesHard : obj → Prop
axiom GoodGrades : obj → Prop
axiom ParentsProud : obj → Prop
axiom IceCream : obj → Prop

axiom Snowstorm : Prop
axiom BigSnowstorm : Prop
axiom SchoolCanceled : Prop

-- If someone studies hard for an exam, they will get good grades.
axiom R1 : ∀ x : obj, StudiesHard x → GoodGrades x
-- If there is a snowstorm, then school is canceled.
axiom R2 : Snowstorm → SchoolCanceled
-- Either Priya studied hard, or school was not called off.
axiom R3 : StudiesHard Priya ∨ ¬SchoolCanceled
-- If a student gets good grades, their parents are proud.
axiom R4 : ∀ x : obj, GoodGrades x → ParentsProud x
-- When parents are proud, they take their kids out for ice cream.
axiom R5 : ∀ x : obj, ParentsProud x → IceCream x
-- A big snowstorm is a snowstorm.
axiom R6 : BigSnowstorm → Snowstorm

theorem priya_icecream_if_big_snowstorm : BigSnowstorm → IceCream Priya := by
  intro hbig
  have hsnow : Snowstorm := R6 hbig
  have hcancel : SchoolCanceled := R2 hsnow
  have hnnc : ¬¬SchoolCanceled := by
    intro hnc
    exact hnc hcancel
  have hstudy : StudiesHard Priya := by
    cases R3 with
    | inl hs => exact hs
    | inr hnotc =>
        have contra : False := hnnc hnotc
        exact False.elim contra
  have hgrades : GoodGrades Priya := R1 Priya hstudy
  have hproud : ParentsProud Priya := R4 Priya hgrades
  exact R5 Priya hproud
</lean>

ANSWER: Yes
