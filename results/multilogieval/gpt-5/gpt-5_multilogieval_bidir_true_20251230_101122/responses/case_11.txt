<lean>
axiom obj : Type
axiom Project : obj

axiom NiceDisplay : obj → Prop
axiom JudgesImpressed : obj → Prop
axiom ThoroughResearch : obj → Prop
axiom ExpertsSerious : obj → Prop
axiom Award : obj → Prop

-- If the display looks nice, judges are impressed
axiom R1 : ∀ x : obj, NiceDisplay x → JudgesImpressed x
-- If research is thorough, experts take it seriously
axiom R2 : ∀ x : obj, ThoroughResearch x → ExpertsSerious x
-- If experts take it seriously, it receives an award
axiom R3 : ∀ x : obj, ExpertsSerious x → Award x

-- The project has either a nice display or thorough research
axiom F1 : NiceDisplay Project ∨ ThoroughResearch Project
-- The judges were not impressed
axiom F2 : ¬ JudgesImpressed Project

theorem project_award : Award Project := by
  cases F1 with
  | inl hD =>
      have hJ : JudgesImpressed Project := R1 Project hD
      exact False.elim (F2 hJ)
  | inr hT =>
      have hS : ExpertsSerious Project := R2 Project hT
      exact R3 Project hS
</lean>

ANSWER: Yes
