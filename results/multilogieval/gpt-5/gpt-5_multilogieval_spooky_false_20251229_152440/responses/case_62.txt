axiom obj : Type
axiom Mark : obj

axiom MadeFriends : obj → Prop
axiom Enjoy : obj → Prop        -- interpret as "enjoyed the (first day of) school" under the given context
axiom JoinClub : obj → Prop
axiom Busy : obj → Prop
axiom Outgoing : obj → Prop
axiom Introduced : obj → Prop

-- If someone made friends, then they would enjoy school.
axiom A1 : ∀ x : obj, MadeFriends x → Enjoy x
-- If someone joins a club, then they will be busy.
axiom A2 : ∀ x : obj, JoinClub x → Busy x
-- Either they made friends, or they are not busy.
axiom A3 : ∀ x : obj, MadeFriends x ∨ ¬Busy x
-- If someone is outgoing, then they would join a club.
axiom A4 : ∀ x : obj, Outgoing x → JoinClub x
-- If someone introduced themselves to classmates, then they are outgoing.
axiom A5 : ∀ x : obj, Introduced x → Outgoing x

-- From the premises we can derive: if Mark did not enjoy, then Mark did not introduce himself.
theorem not_enjoy_implies_not_introduced : (¬Enjoy Mark) → ¬Introduced Mark := by
  intro hne
  -- From ¬Enjoy Mark and (MadeFriends → Enjoy), we get ¬MadeFriends Mark
  have h_not_mf : ¬MadeFriends Mark := by
    intro mf
    have enj : Enjoy Mark := A1 Mark mf
    exact hne enj
  -- From (MadeFriends ∨ ¬Busy) and ¬MadeFriends, we get ¬Busy
  have disj : MadeFriends Mark ∨ ¬Busy Mark := A3 Mark
  have h_not_busy : ¬Busy Mark := by
    cases disj with
    | inl mf => exact False.elim (h_not_mf mf)
    | inr nb => exact nb
  -- From (JoinClub → Busy) and ¬Busy, we get ¬JoinClub
  have h_not_join : ¬JoinClub Mark := by
    intro hj
    have b : Busy Mark := A2 Mark hj
    exact h_not_busy b
  -- From (Outgoing → JoinClub) and ¬JoinClub, we get ¬Outgoing
  have h_not_out : ¬Outgoing Mark := by
    intro ho
    have hj : JoinClub Mark := A4 Mark ho
    exact h_not_join hj
  -- From (Introduced → Outgoing) and ¬Outgoing, we get ¬Introduced
  intro hi
  have ho : Outgoing Mark := A5 Mark hi
  exact h_not_out ho
