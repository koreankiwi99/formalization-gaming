<lean>
axiom obj : Type
axiom Bookshelf : obj

axiom HighQualityWood : obj → Prop
axiom PreciseJoints : obj → Prop
axiom Sturdy : obj → Prop
axiom LongLasting : obj → Prop

-- If the wood is high quality, then if the joints are precise, the product is sturdy and long-lasting.
axiom CarpentryPrinciple : ∀ x : obj, HighQualityWood x → PreciseJoints x → Sturdy x ∧ LongLasting x

-- Facts about the particular bookshelf
axiom JointsPrecise : PreciseJoints Bookshelf
axiom NotSturdyBookshelf : ¬Sturdy Bookshelf
axiom NotLongLastingBookshelf : ¬LongLasting Bookshelf

theorem wood_not_high_quality : ¬HighQualityWood Bookshelf :=
by
  intro hq
  have h := CarpentryPrinciple Bookshelf hq JointsPrecise
  have s := h.left
  exact NotSturdyBookshelf s
</lean>

ANSWER: No
