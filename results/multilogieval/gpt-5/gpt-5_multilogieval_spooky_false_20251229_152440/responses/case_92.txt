<lean>
axiom Watered : Prop
axiom GrowWell : Prop
axiom Pests : Prop
axiom Damaged : Prop
axiom Bountiful : Prop
axiom Profit : Prop

axiom A1 : Watered → GrowWell
axiom A2 : Pests → Damaged
axiom A3 : Watered ∨ ¬Damaged ∨ (Watered ∧ Damaged)
axiom A4 : GrowWell → Bountiful
axiom A5 : Bountiful → Profit

theorem pests_implies_profit : Pests → Profit := by
  intro hp
  have hd : Damaged := A2 hp
  have hdisj : Watered ∨ ¬Damaged ∨ (Watered ∧ Damaged) := A3
  cases hdisj with
  | inl hw =>
      have hg : GrowWell := A1 hw
      have hb : Bountiful := A4 hg
      exact A5 hb
  | inr h =>
      cases h with
      | inl hnd =>
          have contra : False := hnd hd
          exact False.elim contra
      | inr hwd =>
          have hw : Watered := hwd.left
          have hg : GrowWell := A1 hw
          have hb : Bountiful := A4 hg
          exact A5 hb

theorem not_not_pests_implies_profit : ¬¬(Pests → Profit) := by
  intro hneg
  exact hneg pests_implies_profit
