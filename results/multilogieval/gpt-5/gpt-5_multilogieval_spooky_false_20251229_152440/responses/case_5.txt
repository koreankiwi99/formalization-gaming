axiom Learned : person → Prop
axiom Understood : person → Prop
axiom Thrived : person → Prop
axiom SoughtAdvice : person → Prop

axiom LtoU : ∀ x : person, Learned x → Understood x
axiom UtoT : ∀ x : person, Understood x → Thrived x

-- Mary's plants did not thrive
axiom MaryNotThrived : ¬Thrived Mary

-- Either Mary learned about botany or she sought advice (or both)
axiom MaryLearnedOrAdvice : Learned Mary ∨ SoughtAdvice Mary

-- From the premises we can derive that Mary sought advice
theorem mary_sought_advice : SoughtAdvice Mary :=
by
  have h_not_L : ¬Learned Mary :=
    fun hL => MaryNotThrived (UtoT Mary (LtoU Mary hL))
  cases MaryLearnedOrAdvice with
  | inl hL =>
      exact False.elim (h_not_L hL)
  | inr hA =>
      exact hA
