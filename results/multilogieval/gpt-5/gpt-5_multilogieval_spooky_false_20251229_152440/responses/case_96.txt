axiom Exam : Type

axiom Biology : Exam

axiom FeelsConfident : Person → Exam → Prop
axiom DoesWell : Person → Exam → Prop
axiom FeelsSatisfied : Person → Exam → Prop

axiom SevereWeather : Prop
axiom ClassCancelledDueToWeather : Prop

-- If someone studies hard for an exam, they will feel confident going into the test.
axiom A1 : ∀ x : Person, ∀ e : Exam, StudiesHard x e → FeelsConfident x e
-- If there is severe weather in the forecast, schools may cancel class that day.
-- We model "may cancel" as cancellation to keep the relation usable.
axiom A2 : SevereWeather → ClassCancelledDueToWeather
-- Either Alex studied diligently for his biology exam, or class was not canceled due to the weather.
axiom A3 : StudiesHard Alex Biology ∨ ¬ClassCancelledDueToWeather
-- If a student feels confident about the material, they will likely do well on the exam.
axiom A4 : ∀ x : Person, ∀ e : Exam, FeelsConfident x e → DoesWell x e
-- If a student performs well on an exam, they will be satisfied with their hard work.
axiom A5 : ∀ x : Person, ∀ e : Exam, DoesWell x e → FeelsSatisfied x e

-- From the premises we can derive: if there is severe weather, Alex is satisfied with his biology exam.
-- This does not address the history exam in question.
theorem severe_implies_satisfied_bio : SevereWeather → FeelsSatisfied Alex Biology := by
  intro hSW
  have hCancel : ClassCancelledDueToWeather := A2 hSW
  have hStudyOr := A3
  have hStudyBio : StudiesHard Alex Biology := by
    cases hStudyOr with
    | inl h => exact h
    | inr hnc =>
      have : False := hnc hCancel
      contradiction
  have hConf : FeelsConfident Alex Biology := A1 Alex Biology hStudyBio
  have hWell : DoesWell Alex Biology := A4 Alex Biology hConf
  exact A5 Alex Biology hWell
