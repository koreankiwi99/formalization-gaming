<lean>
axiom obj : Type
axiom Store : obj

axiom GreatService : obj → Prop
axiom ReputationGrows : obj → Prop
axiom HostsEvents : obj → Prop
axiom TrafficIncreases : obj → Prop
axiom RevenueImproves : obj → Prop

-- If they offer great customer service, their reputation will grow.
axiom H1 : ∀ x : obj, GreatService x → ReputationGrows x
-- If they host author events, more customers will visit.
axiom H2 : ∀ x : obj, HostsEvents x → TrafficIncreases x
-- The owner either provides excellent service or hosts events.
axiom H3 : GreatService Store ∨ HostsEvents Store
-- If customer traffic increases, revenue will improve.
axiom H4 : ∀ x : obj, TrafficIncreases x → RevenueImproves x
-- Given: service quality decreased (not great service).
axiom Given : ¬GreatService Store

-- From the premises, we can derive that revenue improves.
theorem revenue_improves : RevenueImproves Store := by
  have hEvents : HostsEvents Store := Or.resolve_left H3 Given
  have hTraffic : TrafficIncreases Store := H2 Store hEvents
  exact H4 Store hTraffic
