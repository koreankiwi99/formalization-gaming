<lean>
axiom obj : Type
axiom Project : obj

axiom NiceDisplay : obj → Prop
axiom JudgesImpressed : obj → Prop
axiom ThoroughResearch : obj → Prop
axiom ExpertsTakeSeriously : obj → Prop
axiom Awarded : obj → Prop

axiom A1 : ∀ x : obj, NiceDisplay x → JudgesImpressed x
axiom A2 : ∀ x : obj, ThoroughResearch x → ExpertsTakeSeriously x
axiom A3 : NiceDisplay Project ∨ ThoroughResearch Project
axiom A4 : ∀ x : obj, ExpertsTakeSeriously x → Awarded x

axiom NotImpressed : ¬JudgesImpressed Project

theorem project_awarded : Awarded Project := by
  have notNice : ¬NiceDisplay Project := by
    intro hNice
    have : JudgesImpressed Project := A1 Project hNice
    exact NotImpressed this
  have thorough : ThoroughResearch Project := by
    cases A3 with
    | inl hNice =>
      exact False.elim (NotImpressed (A1 Project hNice))
    | inr hThorough =>
      exact hThorough
  have serious : ExpertsTakeSeriously Project := A2 Project thorough
  exact A4 Project serious
</lean>
