axiom obj : Type
axiom James : obj

axiom Chef : obj → Prop
axiom FollowRecipes : obj → Prop
axiom DishesTurnOutWell : obj → Prop
axiom GainConfidence : obj → Prop
axiom WatchShows : obj → Prop

-- For chefs: following recipes leads to good dishes.
axiom A1 : ∀ x : obj, Chef x → (FollowRecipes x → DishesTurnOutWell x)
-- For chefs: good dishes lead to gaining confidence.
axiom A2 : ∀ x : obj, Chef x → (DishesTurnOutWell x → GainConfidence x)
-- For chefs: they either follow recipes or watch shows (or both).
axiom A3 : ∀ x : obj, Chef x → (FollowRecipes x ∨ WatchShows x)

-- If James were a chef and did not gain confidence, then he watched cooking shows.
theorem watches_if_not_confident_and_chef :
  Chef James → ¬GainConfidence James → WatchShows James := by
  intro hChef hNotConf
  have hFRimpDW : FollowRecipes James → DishesTurnOutWell James := A1 James hChef
  have hDWimpGC : DishesTurnOutWell James → GainConfidence James := A2 James hChef
  have hFRimpGC : FollowRecipes James → GainConfidence James := by
    intro hfr
    exact hDWimpGC (hFRimpDW hfr)
  have hNotFR : ¬FollowRecipes James := by
    intro hfr
    exact hNotConf (hFRimpGC hfr)
  have hDisj : FollowRecipes James ∨ WatchShows James := A3 James hChef
  exact Or.elim hDisj (fun hfr => False.elim (hNotFR hfr)) (fun hws => hws)
