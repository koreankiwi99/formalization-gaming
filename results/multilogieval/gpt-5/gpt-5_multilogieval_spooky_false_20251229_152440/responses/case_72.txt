<lean>
axiom obj : Type
axiom Sam : obj
axiom Close : obj → Prop
axiom StartCooking : obj → Prop
axiom GoShopping : obj → Prop
axiom MakeDinner : obj → Prop
axiom FridgeEmpty : obj → Prop
axiom PantryEmpty : obj → Prop

-- If it's close to dinnertime, then they should start cooking.
axiom H1 : ∀ x : obj, Close x → StartCooking x
-- If they go shopping, then they can make dinner.
axiom H2 : ∀ x : obj, GoShopping x → MakeDinner x
-- Either it is close to dinnertime, or they won't make dinner.
axiom H3 : ∀ x : obj, Close x ∨ ¬MakeDinner x
-- If the fridge is empty, then they must go shopping.
axiom H4 : ∀ x : obj, FridgeEmpty x → GoShopping x
-- If there are no ingredients in the pantry, then the fridge is empty.
axiom H5 : ∀ x : obj, PantryEmpty x → FridgeEmpty x
-- They also needed to go grocery shopping. (take as: they go shopping)
axiom H6 : GoShopping Sam

theorem sam_makes_dinner : MakeDinner Sam := H2 Sam H6

theorem sam_close_to_dinnertime : Close Sam := by
  have md : MakeDinner Sam := H2 Sam H6
  have disj : Close Sam ∨ ¬MakeDinner Sam := H3 Sam
  apply Or.elim disj
  · intro hc; exact hc
  · intro hnot; exact False.elim (hnot md)

theorem sam_starts_cooking : StartCooking Sam := H1 Sam sam_close_to_dinnertime
