<lean>
axiom obj : Type
axiom James : obj

axiom FriNight : Prop
axiom GoParty : obj → Prop
axiom WatchGame : obj → Prop
axiom FinishHW : obj → Prop
axiom StudyEveryday : obj → Prop
axiom PlansWell : obj → Prop

-- If it is Friday night, then they are going to the party.
axiom P1 : ∀ x : obj, FriNight → GoParty x
-- If they finish all their homework, then they can watch the game.
axiom P2 : ∀ x : obj, FinishHW x → WatchGame x
-- Either it is Friday night, or they are not watching the game.
axiom P3 : ∀ x : obj, FriNight ∨ ¬WatchGame x
-- If they study everyday, they finish all their homework.
axiom P4 : ∀ x : obj, StudyEveryday x → FinishHW x
-- If someone plans their schedule well, then they can study everyday.
axiom P5 : ∀ x : obj, PlansWell x → StudyEveryday x

-- James is not going to the party.
axiom James_not_party : ¬GoParty James

theorem james_not_planned : ¬PlansWell James := by
  intro hplan
  have hstudy : StudyEveryday James := P5 James hplan
  have hfinish : FinishHW James := P4 James hstudy
  have hwatch : WatchGame James := P2 James hfinish
  have hfri : FriNight := by
    have disj := P3 James
    cases disj with
    | inl hfri => exact hfri
    | inr hnotwatch =>
        have contra : False := hnotwatch hwatch
        exact False.elim contra
  have hparty : GoParty James := P1 James hfri
  exact James_not_party hparty
</lean>

ANSWER: No
