<lean>
axiom StrongPerimeter : Prop
axiom ManyTurnovers : Prop
axiom PaintProtected : Prop
axiom LimitInside : Prop

-- If the team plays strong perimeter defense, then they will force many turnovers.
axiom R1 : StrongPerimeter → ManyTurnovers
-- If the big men protect the paint, then they will limit inside scoring.
axiom R2 : PaintProtected → LimitInside
-- Either the perimeter defense will be strong or the paint will be protected (or both).
axiom R3 : StrongPerimeter ∨ PaintProtected

-- From the premises, we can derive: If they did not limit inside scoring, then they forced many turnovers.
theorem turnovers_if_not_limited : ¬LimitInside → ManyTurnovers := by
  intro hNotL
  -- From ¬LimitInside and (PaintProtected → LimitInside), deduce ¬PaintProtected
  have hNotB : ¬PaintProtected := by
    intro hB
    exact hNotL (R2 hB)
  -- From StrongPerimeter ∨ PaintProtected and ¬PaintProtected, deduce StrongPerimeter
  have hP : StrongPerimeter := by
    cases R3 with
    | inl hP => exact hP
    | inr hB =>
      have : False := hNotB hB
      exact False.elim this
  -- Then force many turnovers
  exact R1 hP
</lean>
