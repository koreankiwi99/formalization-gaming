axiom obj : Type
axiom Acme : obj

axiom Innovative : obj → Prop
axiom Marketed : obj → Prop
axiom Successful : obj → Prop

-- Context: If a company develops an innovative product, then if they market it effectively, the product will be commercially successful.
axiom Ctx : ∀ x : obj, Innovative x → (Marketed x → Successful x)

-- Premises: Acme marketed effectively. Acme's product was not commercially successful.
axiom P1 : Marketed Acme
axiom P2 : ¬Successful Acme

-- Conclusion to refute: Acme developed an innovative product.
-- We prove its negation.
theorem acme_not_innovative : ¬Innovative Acme := by
  intro hInnov
  have hS : Successful Acme := (Ctx Acme hInnov) P1
  exact P2 hS
