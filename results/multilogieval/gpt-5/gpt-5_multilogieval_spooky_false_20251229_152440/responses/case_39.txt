<lean>
axiom obj : Type
axiom Alex : obj

axiom Monday : Prop
axiom Quiz : Prop
axiom Good : obj → Prop
axiom DoWell : obj → Prop
axiom StudiedHard : obj → Prop

-- Premises
axiom P1 : Monday → Quiz
axiom P2 : ∀ x : obj, DoWell x → StudiedHard x
axiom P3 : (Monday ∨ ¬StudiedHard Alex) ∨ (Monday ∧ ¬StudiedHard Alex)
axiom P4 : ∀ x : obj, Good x → DoWell x

-- From the premises we can show: Good Alex → Quiz
theorem good_Alex_implies_quiz : Good Alex → Quiz := by
  intro g
  have dw : DoWell Alex := P4 Alex g
  have sh : StudiedHard Alex := P2 Alex dw
  have mon : Monday := by
    have d := P3
    cases d with
    | inl h =>
      cases h with
      | inl m => exact m
      | inr nsh =>
        have f : False := nsh sh
        exact False.elim f
    | inr h =>
      exact h.left
  exact P1 mon

-- Hence, if there was no quiz, Alex is not good at math.
theorem no_quiz_implies_not_good_Alex : ¬Quiz → ¬Good Alex := by
  intro nquiz
  intro g
  have q : Quiz := good_Alex_implies_quiz g
  exact nquiz q
