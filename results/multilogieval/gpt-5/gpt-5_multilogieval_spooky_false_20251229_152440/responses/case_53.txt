axiom Mary : obj

axiom Practice : obj → Prop
axiom TechniqueImproved : obj → Prop
axiom ArtworkHighQuality : obj → Prop
axiom TookClass : obj → Prop
axiom LearnMethods : obj → Prop

axiom R1 : ∀ x : obj, Practice x → TechniqueImproved x
axiom R2 : ∀ x : obj, TechniqueImproved x → ArtworkHighQuality x
axiom D1 : Practice Mary ∨ TookClass Mary
axiom R3 : ∀ x : obj, TookClass x → LearnMethods x
axiom NotHQ : ¬ArtworkHighQuality Mary

theorem not_practice_mary : ¬Practice Mary := by
  intro hP
  have hTI : TechniqueImproved Mary := R1 Mary hP
  have hHQ : ArtworkHighQuality Mary := R2 Mary hTI
  exact NotHQ hHQ

theorem took_class_mary : TookClass Mary := by
  cases D1 with
  | inl hP =>
      exact False.elim (not_practice_mary hP)
  | inr hC =>
      exact hC

theorem mary_learned : LearnMethods Mary := by
  exact R3 Mary took_class_mary
