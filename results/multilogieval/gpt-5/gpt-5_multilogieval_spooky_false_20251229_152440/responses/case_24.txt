<lean>
axiom obj : Type

-- Individuals
axiom Mary : obj

-- Propositions about the world
axiom BelowFreezing : Prop
axiom SnowOnGround : Prop
axiom EnoughSnow : Prop

-- Predicates about people
axiom BuildSnowman : obj → Prop
axiom LovesWinter : obj → Prop

-- Premises extracted from the text
axiom H1 : BelowFreezing → SnowOnGround
axiom H2 : ∀ x : obj, BuildSnowman x → EnoughSnow
axiom H3 : (BelowFreezing ∨ ¬EnoughSnow) ∨ (BelowFreezing ∧ ¬EnoughSnow)
axiom H4 : ∀ x : obj, LovesWinter x → BuildSnowman x

-- From "no snow on the ground" we derive Mary does not love winter.
theorem mary_not_love_if_no_snow : ¬SnowOnGround → ¬LovesWinter Mary := by
  intro hNoSnow
  -- From BelowFreezing → SnowOnGround, get ¬SnowOnGround → ¬BelowFreezing
  have hNotBF : ¬BelowFreezing := fun hbf => hNoSnow (H1 hbf)
  -- Use the trichotomy H3 and ¬BelowFreezing to obtain ¬EnoughSnow
  have hNotEnough : ¬EnoughSnow := by
    cases H3 with
    | inl h =>
      cases h with
      | inl hbf =>
        exact False.elim (hNotBF hbf)
      | inr hne =>
        exact hne
    | inr hboth =>
      exact hboth.right
  -- From LovesWinter → BuildSnowman and BuildSnowman → EnoughSnow, get LovesWinter → EnoughSnow
  have love_implies_enough : LovesWinter Mary → EnoughSnow :=
    fun hLove => H2 Mary (H4 Mary hLove)
  -- Contraposition gives ¬EnoughSnow → ¬LovesWinter
  exact fun hLove => hNotEnough (love_implies_enough hLove)
</lean>
