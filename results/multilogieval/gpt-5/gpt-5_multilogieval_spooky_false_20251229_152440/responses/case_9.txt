<lean>
axiom Party : Type
axiom SamParty : Party

axiom BringDish : Party → Prop
axiom EnoughFood : Party → Prop
axiom Birthday : Party → Prop
axiom Cake : Party → Prop
axiom IceCream : Party → Prop

-- If someone brings a dish to share at a party, there will be enough food for everyone.
axiom A1 : ∀ p : Party, BringDish p → EnoughFood p
-- If it is a birthday party, there will be cake and ice cream.
axiom A2 : ∀ p : Party, Birthday p → Cake p ∧ IceCream p
-- Either someone brought a dish to share, or there was no birthday cake (for Sam's party).
axiom A3 : BringDish SamParty ∨ ¬Cake SamParty
-- It is Sam's birthday party.
axiom A4 : Birthday SamParty

-- From the premises, Sam's party has enough food for everyone.
theorem sam_enough : EnoughFood SamParty :=
by
  have cake_sam : Cake SamParty := And.left (A2 SamParty A4)
  cases A3 with
  | inl bd =>
    exact A1 SamParty bd
  | inr nCake =>
    have f : False := nCake cake_sam
    exact False.elim f
</lean>

ANSWER: Failure
