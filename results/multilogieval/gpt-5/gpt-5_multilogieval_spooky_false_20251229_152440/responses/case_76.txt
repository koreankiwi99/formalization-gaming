<lean>
axiom Person : Type
axiom Alex : Person

axiom WeekendComing : Prop
axiom Relax : Person → Prop
axiom FinishTasks : Person → Prop
axiom ClearMind : Person → Prop
axiom Organized : Person → Prop
axiom TodoList : Person → Prop

-- If it is the weekend coming up, then they (Alex) would relax.
axiom R1 : WeekendComing → Relax Alex
-- If someone finishes their tasks, then they have a clear mind.
axiom R2 : ∀ x : Person, FinishTasks x → ClearMind x
-- Either the weekend was coming, or they (Alex) did not have a clear mind.
axiom R3 : WeekendComing ∨ ¬ClearMind Alex
-- If someone is organized, then they finish their tasks.
axiom R4 : ∀ x : Person, Organized x → FinishTasks x
-- If they made a to-do list, then they are organized.
axiom R5 : ∀ x : Person, TodoList x → Organized x

-- From "Alex does not relax this weekend", conclude "Alex did not make a to-do list".
theorem alex_not_todolist_of_not_relax (hnr : ¬Relax Alex) : ¬TodoList Alex := by
  -- ¬WeekendComing from R1 and hnr
  have hnotWeek : ¬WeekendComing := by
    intro hw
    have : Relax Alex := R1 hw
    exact hnr this
  -- From WeekendComing ∨ ¬ClearMind Alex and ¬WeekendComing, deduce ¬ClearMind Alex
  have hNotClear : ¬ClearMind Alex := by
    cases R3 with
    | inr hnc => exact hnc
    | inl hw => exact False.elim (hnotWeek hw)
  -- From FinishTasks → ClearMind and ¬ClearMind, deduce ¬FinishTasks (for Alex)
  have hNotFinish : ¬FinishTasks Alex := by
    intro hf
    have hc : ClearMind Alex := R2 Alex hf
    exact hNotClear hc
  -- From Organized → FinishTasks and ¬FinishTasks, deduce ¬Organized (for Alex)
  have hNotOrg : ¬Organized Alex := by
    intro ho
    have hf : FinishTasks Alex := R4 Alex ho
    exact hNotFinish hf
  -- From TodoList → Organized and ¬Organized, deduce ¬TodoList (for Alex)
  intro ht
  have ho : Organized Alex := R5 Alex ht
  exact hNotOrg ho
</lean>

ANSWER: No
