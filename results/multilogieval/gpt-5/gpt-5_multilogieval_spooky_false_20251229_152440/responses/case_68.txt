<lean>
axiom Person : Type
axiom Exam : Type

axiom Emma : Person
axiom Bio : Exam

axiom StudyHard : Person → Prop
axiom SleepBefore : Person → Exam → Prop
axiom PerformWell : Person → Exam → Prop
axiom Ace : Person → Exam → Prop

-- Belief: if someone studies hard, then if they get enough sleep before an exam,
-- they will perform well and ace the test.
axiom Belief : ∀ p e, StudyHard p → SleepBefore p e → (PerformWell p e ∧ Ace p e)

-- Facts about Emma and her biology exam
axiom Sleep_Emma_Bio : SleepBefore Emma Bio
axiom NotAce_Emma_Bio : ¬ Ace Emma Bio

theorem emma_not_studyhard : ¬ StudyHard Emma := by
  intro hStudy
  have hAnd := Belief Emma Bio hStudy Sleep_Emma_Bio
  have hAce : Ace Emma Bio := And.right hAnd
  exact NotAce_Emma_Bio hAce
</lean>

ANSWER: No
