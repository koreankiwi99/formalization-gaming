<lean>
axiom obj : Type
axiom Alex : obj
axiom Student : obj

axiom StudiedHard : obj → Prop
axiom GoodGrades : obj → Prop
axiom ExtraCurric : obj → Prop
axiom MakesFriends : obj → Prop
axiom Invited : obj → Prop

-- The student believed/stated implications and had one of two activities
axiom R1 : StudiedHard Student → GoodGrades Student
axiom R2 : ExtraCurric Student → MakesFriends Student
axiom R3 : StudiedHard Student ∨ ExtraCurric Student

-- General rule: making friends leads to invitations
axiom R4 : ∀ x : obj, MakesFriends x → Invited x

-- Given: Alex did not get good grades
axiom Alex_not_good : ¬GoodGrades Alex

-- If Alex were the student in the context, then Alex would be invited
theorem invited_if_student_is_alex : Student = Alex → Invited Alex := by
  intro h
  have h_or : StudiedHard Alex ∨ ExtraCurric Alex := by
    have := R3
    simpa [h] using this
  have h_impS : StudiedHard Alex → GoodGrades Alex := by
    have := R1
    simpa [h] using this
  have h_impE : ExtraCurric Alex → MakesFriends Alex := by
    have := R2
    simpa [h] using this
  have h_notS : ¬StudiedHard Alex := by
    intro s
    have g : GoodGrades Alex := h_impS s
    exact Alex_not_good g
  have e : ExtraCurric Alex := by
    cases h_or with
    | inl s =>
      exact False.elim (h_notS s)
    | inr ex =>
      exact ex
  have f : MakesFriends Alex := h_impE e
  exact R4 Alex f
