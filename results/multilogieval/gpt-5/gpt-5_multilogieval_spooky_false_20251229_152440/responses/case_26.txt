<lean>
axiom LostImportantGame : Prop
axiom LostBigMatch : Prop
axiom LostSignificantGame : Prop
axiom PlayersDisappointed : Prop
axiom PlayersDemoralized : Prop

-- Premises from the text
axiom P1 : LostImportantGame → PlayersDisappointed
axiom P2 : LostBigMatch → PlayersDemoralized

-- Bridge: a "significant game" is either an important game or a big match
axiom SigDef : LostSignificantGame → (LostImportantGame ∨ LostBigMatch)

-- Statement to evaluate: players are neither disappointed nor demoralized
axiom Hdis : ¬PlayersDisappointed
axiom Hdem : ¬PlayersDemoralized

theorem no_significant_loss : ¬LostSignificantGame := by
  intro hsig
  have h := SigDef hsig
  cases h with
  | inl hImp =>
      exact Hdis (P1 hImp)
  | inr hBig =>
      exact Hdem (P2 hBig)
</lean>

ANSWER: No
