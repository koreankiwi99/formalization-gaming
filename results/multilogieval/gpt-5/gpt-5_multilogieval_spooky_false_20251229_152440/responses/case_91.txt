<lean>
axiom Party : Prop
axiom Drums : Prop
axiom Noisy : Prop
axiom NoiseUp : Prop
axiom Complaints : Prop
axiom Police : Prop

-- If someone hosts a party, their house gets very noisy.
axiom R1 : Party → Noisy
-- If someone is practicing drums, the noise levels go up.
axiom R2 : Drums → NoiseUp
-- Either there is a party or someone is playing drums (or both).
axiom R3 : Party ∨ Drums
-- If the house gets noisy from a party, neighbors start complaining.
axiom R4 : Noisy ∧ Party → Complaints
-- If neighbors complain, the police are called to intervene.
axiom R5 : Complaints → Police

theorem party_to_complaints : Party → Complaints := by
  intro hp
  have hnoisy : Noisy := R1 hp
  have hpair : Noisy ∧ Party := And.intro hnoisy hp
  exact R4 hpair

theorem party_to_police : Party → Police := by
  intro hp
  have hc : Complaints := party_to_complaints hp
  exact R5 hc

-- Main: If there was no noise level up, were the police called to intervene?
-- We prove (¬NoiseUp) → Police
theorem noise_not_up_implies_police : (¬NoiseUp) → Police := by
  intro hn
  have hdisj := R3
  cases hdisj with
  | inl hp =>
      exact party_to_police hp
  | inr hd =>
      have hnu : NoiseUp := R2 hd
      have contra : False := hn hnu
      exact False.elim contra
</lean>
