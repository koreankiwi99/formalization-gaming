axiom HeavyRain : Prop
axiom SoilSaturated : Prop
axiom WateredRoutinely : Prop
axiom FlowersBloom : Prop
axiom Fragrance : Prop
axiom Butterflies : Prop

-- Premises
axiom P1 : WateredRoutinely → FlowersBloom
axiom P2 : HeavyRain → SoilSaturated
axiom P3 : WateredRoutinely ∨ ¬SoilSaturated
axiom P4 : FlowersBloom → Fragrance
axiom P5 : Fragrance → Butterflies

-- From the premises, heavy rain implies the garden attracts butterflies.
theorem rain_implies_butterflies : HeavyRain → Butterflies := by
  intro hr
  have s : SoilSaturated := P2 hr
  have wr : WateredRoutinely := by
    cases P3 with
    | inl w => exact w
    | inr ns =>
      have f : False := ns s
      exact False.elim f
  have b : FlowersBloom := P1 wr
  have fr : Fragrance := P4 b
  exact P5 fr

-- Consequently, the negation of the statement is not provable from these premises.
theorem not_neg_of_statement : ¬(¬(HeavyRain → Butterflies)) := by
  intro h
  exact h rain_implies_butterflies
