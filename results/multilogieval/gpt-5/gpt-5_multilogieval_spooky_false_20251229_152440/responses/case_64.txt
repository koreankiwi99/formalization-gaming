<lean>
axiom Student : Type
axiom Priya : Student

axiom AttendsRegularly : Student → Prop
axiom RevisesSufficiently : Student → Prop
axiom PerformsWell : Student → Prop

-- If a student attends lectures regularly, then if they revise sufficiently, they will perform well.
axiom academic_rule : ∀ s : Student, AttendsRegularly s → (RevisesSufficiently s → PerformsWell s)

-- Facts about Priya
axiom priya_revised : RevisesSufficiently Priya
axiom priya_not_well : ¬ PerformsWell Priya

theorem priya_did_not_attend : ¬ AttendsRegularly Priya := by
  intro hAttend
  have hImp : RevisesSufficiently Priya → PerformsWell Priya := academic_rule Priya hAttend
  have hWell : PerformsWell Priya := hImp priya_revised
  exact priya_not_well hWell
</lean>

ANSWER: No
