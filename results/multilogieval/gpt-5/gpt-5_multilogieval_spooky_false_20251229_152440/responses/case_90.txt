-- If the fans will not be more emotionally invested, then the atmosphere was electric.
theorem electric_if_not_invested : (¬Invested) → Electric := by

    have l : LoudAndEnergized := HtoL h
    exact LtoE l
  | inr r =>
    have i : Invested := RtoI r
    have f : False := nI i
