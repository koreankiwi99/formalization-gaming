<lean>
axiom obj : Type
axiom Lebron : obj
axiom TrainHard : obj → Prop
axiom Focused : obj → Prop
axiom PerformBest : obj → Prop

-- If someone trains hard, then if they are focused, they will perform to the best of their ability.
axiom Preach : ∀ x : obj, TrainHard x → (Focused x → PerformBest x)

-- Facts about Lebron
axiom Focused_Lebron : Focused Lebron
axiom NotBest_Lebron : ¬PerformBest Lebron

theorem Lebron_not_trainhard : ¬TrainHard Lebron := by
  intro hTH
  have hImp : Focused Lebron → PerformBest Lebron := Preach Lebron hTH
  have hBest : PerformBest Lebron := hImp Focused_Lebron
  exact NotBest_Lebron hBest
</lean>

ANSWER: No
