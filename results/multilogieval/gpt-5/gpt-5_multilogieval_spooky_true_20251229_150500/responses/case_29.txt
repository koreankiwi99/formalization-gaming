axiom Person : Type
axiom Ty : Person

axiom SuspectsIdentityTheft : Person → Prop
axiom SuspectsFraud : Person → Prop
axiom CheckedCreditReport : Person → Prop
axiom ContactedBank : Person → Prop

-- If someone suspects identity theft, they should check their credit report.
axiom R1 : ∀ x : Person, SuspectsIdentityTheft x → CheckedCreditReport x
-- If fraud is suspected, they should contact their bank.
axiom R2 : ∀ x : Person, SuspectsFraud x → ContactedBank x
-- Identity theft is a kind of fraud: suspecting identity theft implies suspecting fraud.
axiom ID_implies_Fraud : ∀ x : Person, SuspectsIdentityTheft x → SuspectsFraud x

-- Either Ty did not verify his credit report or did not get in touch with his bank.
axiom Hdisj : ¬ CheckedCreditReport Ty ∨ ¬ ContactedBank Ty

-- From the premises and the disjunction, we can derive that Ty did not suspect identity theft.
theorem no_ID_suspicion_for_Ty : ¬ SuspectsIdentityTheft Ty := by
  intro hID
  have hC : CheckedCreditReport Ty := R1 Ty hID
  have hF : SuspectsFraud Ty := ID_implies_Fraud Ty hID
  have hB : ContactedBank Ty := R2 Ty hF
  cases Hdisj with
  | inl hNotC => exact hNotC hC
  | inr hNotB => exact hNotB hB
