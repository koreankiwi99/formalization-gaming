axiom James : obj

axiom FridayNight : Prop

-- Assumptions derived from the textual context
axiom James_student : Student James
axiom party_rule : ∀ x : obj, Student x → (FridayNight → GoingParty x)
axiom finish_watch_rule : ∀ x : obj, FinishHW x → WatchGame x

axiom plan_study_rule : ∀ x : obj, PlanWell x → StudyDaily x

axiom not_party_james : ¬ GoingParty James

-- From "if Friday then going to party" and "James is not going", deduce it's not Friday
theorem not_friday : ¬ FridayNight := by
  intro fn
  have himpl := party_rule James James_student
  have gp : GoingParty James := himpl fn
  exact not_party_james gp

-- From "Friday or not watching" and not Friday, deduce James is not watching
theorem not_watch_james : ¬ WatchGame James := by
  have h := friday_or_notwatch_rule James James_student
  exact Or.elim h
    (fun fn => False.elim (not_friday fn))
    (fun nhw => nhw)

  intro fhw
  have w := finish_watch_rule James fhw
  exact not_watch_james w

-- From "study → finish" and "not finish", deduce "not study"
theorem not_study_james : ¬ StudyDaily James := by
  intro sd
  have fhw := study_finish_rule James sd
  exact not_finish_james fhw

-- From "plan well → study" and "not study", deduce "not plan well"
