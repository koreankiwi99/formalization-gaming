<lean>
axiom obj : Type
axiom John : obj

axiom Water : obj → Prop
axiom Bloom : obj → Prop
axiom Attract : obj → Prop
axiom Organic : obj → Prop
axiom HighSoil : obj → Prop
axiom Healthy : obj → Prop

-- If someone waters adequately, plants bloom.
axiom R1 : ∀ x : obj, Water x → Bloom x
-- If plants bloom, the garden attracts more bees and butterflies.
axiom R2 : ∀ x : obj, Bloom x → Attract x
-- Either someone waters, or uses organic fertilizers, or both.
axiom R3 : ∀ x : obj, Water x ∨ Organic x ∨ (Water x ∧ Organic x)
-- If someone uses organic fertilizers, the soil quality is high.
axiom R4 : ∀ x : obj, Organic x → HighSoil x
-- If the soil quality improves, the plants grow healthier. (Not used in the proof.)
axiom R5 : ∀ x : obj, HighSoil x → Healthy x

theorem john_high_soil_of_not_attract : ¬ Attract John → HighSoil John := by
  intro hNotAttr
  have hWimpA : Water John → Attract John := fun w => R2 John (R1 John w)
  have hNotW : ¬ Water John := fun w => hNotAttr (hWimpA w)
  have hDisj := R3 John
  have hOrg : Organic John := by
    cases hDisj with
    | inl hW => exact False.elim (hNotW hW)
    | inr h2 =>
      cases h2 with
      | inl hF => exact hF
      | inr hWF => exact False.elim (hNotW hWF.left)
  exact R4 John hOrg
</lean>

ANSWER: Yes
