<lean>
axiom obj : Type
axiom Dallas : obj

axiom Heavy : obj → Prop
axiom ExcessiveRain : obj → Prop
axiom ReservoirsOverflow : obj → Prop
axiom Flooding : obj → Prop

-- Heavy rainfall over several days counts as excessive rain
axiom H_excessive : ∀ x : obj, Heavy x → ExcessiveRain x
-- If there is excessive rain, reservoirs overflow
axiom H_overflow : ∀ x : obj, ExcessiveRain x → ReservoirsOverflow x
-- If reservoirs overflow, flooding occurs
axiom H_flooding : ∀ x : obj, ReservoirsOverflow x → Flooding x

-- Given: There is heavy rainfall in Dallas
axiom Heavy_Dallas : Heavy Dallas

theorem flooding_in_dallas : Flooding Dallas := by
  have hx : ExcessiveRain Dallas := H_excessive Dallas Heavy_Dallas
  have ho : ReservoirsOverflow Dallas := H_overflow Dallas hx
  exact H_flooding Dallas ho
</lean>

ANSWER: Yes
