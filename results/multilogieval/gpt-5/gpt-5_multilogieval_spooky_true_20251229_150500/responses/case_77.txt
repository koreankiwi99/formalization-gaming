axiom James : obj

axiom PracticedHard : obj → Prop
axiom PlayWell : obj → Prop
axiom PickedStart : obj → Prop
axiom Talented : obj → Prop
axiom NaturalAbility : obj → Prop
axiom TopYouth : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, PracticedHard x → PlayWell x
axiom P2 : ∀ x : obj, PickedStart x → Talented x
axiom P3 : ∀ x : obj, PracticedHard x ∨ ¬ Talented x
axiom P4 : ∀ x : obj, NaturalAbility x → PickedStart x
axiom P5 : ∀ x : obj, TopYouth x → NaturalAbility x

-- From the premises, we can actually derive the opposite direction:
-- If James did not play well, then James was not a top youth player.
theorem not_playwell_implies_not_topyouth :
  ¬ PlayWell James → ¬ TopYouth James := by
  intro hnpw
  -- From P1, get ¬PlayWell → ¬PracticedHard
  have hnp : ¬ PracticedHard James := by
    apply mt (P1 James)
    exact hnpw
  -- From P3 and ¬PracticedHard, deduce ¬Talented
  have hdisj : PracticedHard James ∨ ¬ Talented James := P3 James
  have hntal : ¬ Talented James := by
    cases hdisj with
    | inl hpr => exact False.elim (hnp hpr)
    | inr hnt => exact hnt
  -- From P2 (PickedStart → Talented), contrapose to get ¬Talented → ¬PickedStart
  have hnpicked : ¬ PickedStart James := by
    apply mt (P2 James)
    exact hntal
  -- From P4 (NaturalAbility → PickedStart), contrapose to get ¬PickedStart → ¬NaturalAbility
  have hnnat : ¬ NaturalAbility James := by
    apply mt (P4 James)
    exact hnpicked
  -- From P5 (TopYouth → NaturalAbility), contrapose to get ¬NaturalAbility → ¬TopYouth
  have hnotTop : ¬ TopYouth James := by
    apply mt (P5 James)
    exact hnnat
  exact hnotTop
