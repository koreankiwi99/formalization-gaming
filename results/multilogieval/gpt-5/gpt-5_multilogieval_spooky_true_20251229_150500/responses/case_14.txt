<lean>
axiom obj : Type
axiom Alex : obj

axiom AdvertisesCleverly : obj → Prop
axiom OffersDiscounts : obj → Prop
axiom AttractMoreCustomers : obj → Prop
axiom IncreaseSales : obj → Prop
axiom ExpandBusiness : obj → Prop

-- If Alex advertised cleverly, then Alex would attract more customers.
axiom A1 : AdvertisesCleverly Alex → AttractMoreCustomers Alex
-- If Alex offered discounts, then Alex would increase sales.
axiom A2 : OffersDiscounts Alex → IncreaseSales Alex
-- Alex either advertises cleverly or offers discounts.
axiom A3 : AdvertisesCleverly Alex ∨ OffersDiscounts Alex
-- If someone increases sales, they will expand their business.
axiom A4 : ∀ x : obj, IncreaseSales x → ExpandBusiness x

theorem alex_expands : ¬ AttractMoreCustomers Alex → ExpandBusiness Alex := by
  intro hNotAttract
  have hNotAdvertise : ¬ AdvertisesCleverly Alex := fun h => hNotAttract (A1 h)
  cases A3 with
  | inl hAd =>
      exact False.elim (hNotAdvertise hAd)
  | inr hDisc =>
      have hInc : IncreaseSales Alex := A2 hDisc
      exact A4 Alex hInc
</lean>

ANSWER: Yes
