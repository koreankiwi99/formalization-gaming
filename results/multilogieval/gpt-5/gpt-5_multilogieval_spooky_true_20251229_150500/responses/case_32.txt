axiom obj : Type

-- Predicates
axiom Exercises : obj → Prop
axiom EatsNutritious : obj → Prop
axiom Fit : obj → Prop
axiom Energetic : obj → Prop
axiom ValuesWellness : obj → Prop

-- General rules from the context
axiom R_exercise_fit : ∀ x : obj, Exercises x → Fit x
axiom R_eat_energy : ∀ x : obj, EatsNutritious x → Energetic x
axiom R_values_exercise : ∀ x : obj, ValuesWellness x → Exercises x

-- There exists someone who is neither fit nor energetic
axiom exists_not_fit_and_not_energetic : ∃ x : obj, ¬ Fit x ∧ ¬ Energetic x

-- The specific individual Jane and the fact that she eats nutritious foods
axiom Jane : obj
axiom Jane_eats : EatsNutritious Jane

-- From the existential premise, we can derive the mayor's (weaker) conclusion:
-- some citizen does not exercise or does not eat nutritious foods.
theorem some_not_exercise_or_not_eat : ∃ x : obj, ¬ Exercises x ∨ ¬ EatsNutritious x := by
  rcases exists_not_fit_and_not_energetic with ⟨w, hNotFit, hNotEnergetic⟩
  have hNotExercises : ¬ Exercises w := by
    intro hEx
    exact hNotFit (R_exercise_fit w hEx)
  have hNotEats : ¬ EatsNutritious w := by
    intro hEat
    exact hNotEnergetic (R_eat_energy w hEat)
  exact ⟨w, Or.inl hNotExercises⟩

-- What we can conclude about Jane from the given premises:
theorem jane_energetic : Energetic Jane := by
  exact R_eat_energy Jane Jane_eats
