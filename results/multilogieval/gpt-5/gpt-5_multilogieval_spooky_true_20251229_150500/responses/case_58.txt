<lean>
axiom AddSalt : Prop
axiom Savory : Prop
axiom Stir : Prop
axiom Thickens : Prop
axiom Taste : Prop
axiom Salad : Prop

axiom R1 : AddSalt → Savory
axiom R2 : Stir → Thickens
axiom R3 : AddSalt ∨ ¬ Thickens
axiom R4 : Taste → Stir
axiom R5 : Taste ∨ Salad

theorem salad_of_not_savory (Hns : ¬ Savory) : Salad := by
  have nAdd : ¬ AddSalt := by
    intro hAdd
    exact Hns (R1 hAdd)
  have nThick : ¬ Thickens := by
    cases R3 with
    | inl hAdd => exact False.elim (nAdd hAdd)
    | inr hNThick => exact hNThick
  have nStir : ¬ Stir := by
    intro hStir
    have hTh : Thickens := R2 hStir
    exact nThick hTh
  have nTaste : ¬ Taste := by
    intro hTaste
    have hSt : Stir := R4 hTaste
    exact nStir hSt
  cases R5 with
  | inl hTaste => exact False.elim (nTaste hTaste)
  | inr hSalad => exact hSalad
</lean>

ANSWER: Yes
