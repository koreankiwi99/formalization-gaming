axiom MarketedEffectively : obj → Prop

  ∀ x : obj, DevelopedInnovative x → (MarketedEffectively x → CommerciallySuccessful x)

-- Given: Acme marketed their new gadget effectively.
axiom Marketed_Acme : MarketedEffectively Acme

-- Given: Acme's gadget was not commercially successful.
axiom Not_Success_Acme : ¬ CommerciallySuccessful Acme

-- From the context and givens, we can derive that Acme did not develop an innovative product.
theorem not_developed_innovative_Acme : ¬ DevelopedInnovative Acme := by
  intro hDev
  have hImp : MarketedEffectively Acme → CommerciallySuccessful Acme := Context Acme hDev
  have hSucc : CommerciallySuccessful Acme := hImp Marketed_Acme
  exact Not_Success_Acme hSucc
