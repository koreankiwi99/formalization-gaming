<lean>
axiom obj : Type
axiom Jesse : obj

axiom Sprinting : obj → Prop
axiom Stretching : obj → Prop
axiom RunningImproved : obj → Prop
axiom Flexible : obj → Prop
axiom LessInjury : obj → Prop

-- If someone did sprint workouts frequently, their running form and stamina improved.
axiom Sprint_improves_run : ∀ x : obj, Sprinting x → RunningImproved x
-- Either they were sprinting frequently, or they were taking stretching classes (or both).
axiom Sprint_or_Stretch : ∀ x : obj, Sprinting x ∨ Stretching x
-- If someone takes stretching classes, they become more flexible and less prone to injury.
axiom Stretch_gives_benefits : ∀ x : obj, Stretching x → (Flexible x ∧ LessInjury x)

-- Given: Improving running form and stamina did not happen for Jesse.
axiom not_run_improved_Jesse : ¬ RunningImproved Jesse

theorem jesse_flexible_and_less_injury : Flexible Jesse ∧ LessInjury Jesse := by
  -- First, Jesse was not sprinting frequently (by contrapositive reasoning)
  have not_sprinting : ¬ Sprinting Jesse := by
    intro hs
    have ri : RunningImproved Jesse := Sprint_improves_run Jesse hs
    exact not_run_improved_Jesse ri
  -- From the disjunction and not_sprinting, conclude Jesse took stretching classes
  have stretch_Jesse : Stretching Jesse := by
    have h := Sprint_or_Stretch Jesse
    cases h with
    | inl hs =>
        have f : False := not_sprinting hs
        exact False.elim f
    | inr ht => exact ht
  -- Therefore, Jesse became more flexible and less prone to injury
  exact Stretch_gives_benefits Jesse stretch_Jesse
</lean>

ANSWER: Yes
