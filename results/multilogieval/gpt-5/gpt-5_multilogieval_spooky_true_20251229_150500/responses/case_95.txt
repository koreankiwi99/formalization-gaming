<lean>
axiom obj : Type
axiom Mary : obj

-- Propositions
axiom Clear : Prop
axiom Game : Prop
axiom Cheer : Prop
axiom Exciting : Prop

-- Predicates about people
axiom Finish : obj → Prop
axiom Meet : obj → Prop

-- Premises
axiom P1 : Clear → Game
axiom P2 : ∀ x : obj, Finish x → Meet x
axiom P3 : Clear ∨ Finish Mary
axiom P4 : Game → Cheer
axiom P5 : Cheer → Exciting

-- Conclusion: If Mary did not meet her friends for dinner, it was an exciting night at the ballpark.
theorem exciting_if_mary_did_not_meet : (¬ Meet Mary) → Exciting := by
  intro hNotMeet
  cases P3 with
  | inl hClear =>
    have hGame : Game := P1 hClear
    have hCheer : Cheer := P4 hGame
    exact P5 hCheer
  | inr hFinishMary =>
    have hMeetMary : Meet Mary := P2 Mary hFinishMary
    have hFalse : False := hNotMeet hMeetMary
    exact False.elim hFalse
</lean>

ANSWER: Yes
