<lean>
axiom obj : Type
axiom Mary : obj

axiom Practices : obj → Prop
axiom TechImproves : obj → Prop
axiom StandsOutHQ : obj → Prop
axiom TakesArtClass : obj → Prop
axiom LearnsNewMethods : obj → Prop

axiom Practice_improves : ∀ x : obj, Practices x → TechImproves x
axiom Improve_standsout : ∀ x : obj, TechImproves x → StandsOutHQ x
axiom Class_learns : ∀ x : obj, TakesArtClass x → LearnsNewMethods x

axiom Mary_practice_or_class : Practices Mary ∨ TakesArtClass Mary
axiom Mary_not_standout : ¬ StandsOutHQ Mary

theorem mary_learned_new_methods : LearnsNewMethods Mary := by
  have hPS : ∀ x : obj, Practices x → StandsOutHQ x := by
    intro x
    intro hp
    have ht : TechImproves x := Practice_improves x hp
    exact Improve_standsout x ht
  have hNotP : ¬ Practices Mary := by
    intro hp
    have hs : StandsOutHQ Mary := hPS Mary hp
    exact Mary_not_standout hs
  have hClass : TakesArtClass Mary := by
    cases Mary_practice_or_class with
    | inl hp => exact False.elim (hNotP hp)
    | inr ha => exact ha
  exact Class_learns Mary hClass
</lean>

ANSWER: Yes
