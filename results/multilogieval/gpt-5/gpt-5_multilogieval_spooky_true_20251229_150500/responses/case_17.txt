<lean>
axiom obj : Type
axiom James : obj

axiom Practice : obj → Prop
axiom Prepared : obj → Prop
axiom Confident : obj → Prop
axiom JoinClub : obj → Prop

-- If they practice thoroughly, they feel prepared.
axiom R1 : ∀ x : obj, Practice x → Prepared x
-- If they feel prepared, they can speak confidently.
axiom R2 : ∀ x : obj, Prepared x → Confident x
-- Either they practiced thoroughly or they joined a public speaking club (or both).
axiom R3 : ∀ x : obj, Practice x ∨ JoinClub x

theorem james_joins_from_not_confident : ¬ Confident James → JoinClub James := by
  intro hNotConf
  have hNotPrepared : ¬ Prepared James := fun hPrep => hNotConf (R2 James hPrep)
  have hNotPractice : ¬ Practice James := fun hPrac => hNotPrepared (R1 James hPrac)
  have hOr := R3 James
  cases hOr with
  | inl hPrac => exact False.elim (hNotPractice hPrac)
  | inr hJoin => exact hJoin
</lean>

ANSWER: Yes
