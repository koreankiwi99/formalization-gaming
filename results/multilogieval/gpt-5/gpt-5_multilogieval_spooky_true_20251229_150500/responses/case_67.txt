<lean>
axiom Person : Type
axiom Sam : Person

axiom Hungry : Person → Prop
axiom MakesPancakes : Person → Prop
axiom UsesStove : Person → Prop
axiom AteDinnerLastNight : Person → Prop

axiom StoveOn : Prop
axiom KitchenHot : Prop

-- If the stove is on, the kitchen gets hot.
axiom H1 : StoveOn → KitchenHot
-- If someone makes pancakes, they use the stove.
axiom H2 : ∀ x : Person, MakesPancakes x → UsesStove x
-- Either the stove was on, or no one used the stove.
axiom H3 : StoveOn ∨ ∀ x : Person, ¬UsesStove x
-- If someone is hungry, they make pancakes.
axiom H4 : ∀ x : Person, Hungry x → MakesPancakes x
-- If Sam did not eat dinner last night, he was hungry this morning.
axiom H5 : ¬AteDinnerLastNight Sam → Hungry Sam
-- The kitchen did not get hot.
axiom H6 : ¬KitchenHot

theorem Sam_ate_dinner_last_night : AteDinnerLastNight Sam := by
  classical
  by_contra hNot
  have hHungry : Hungry Sam := H5 hNot
  have hPanc : MakesPancakes Sam := H4 Sam hHungry
  have hUse : UsesStove Sam := H2 Sam hPanc
  have hNoUseAll : ∀ x : Person, ¬UsesStove x := by
    cases H3 with
    | inl hOn =>
      -- From StoveOn we get KitchenHot, contradicting H6; from False we can derive any ¬UsesStove x.
      exact fun x => fun _ => False.elim (H6 (H1 hOn))
    | inr h =>
      exact h
  have hNoUseSam : ¬UsesStove Sam := hNoUseAll Sam
  exact hNoUseSam hUse
</lean>

ANSWER: Yes

<lean>
axiom Person : Type
axiom Sam : Person

axiom Hungry : Person → Prop
axiom MakesPancakes : Person → Prop
axiom UsesStove : Person → Prop
axiom AteDinnerLastNight : Person → Prop

axiom StoveOn : Prop
axiom KitchenHot : Prop

-- Someone made pancakes.
axiom H0 : ∃ x : Person, MakesPancakes x
-- If the stove is on, the kitchen gets hot.
axiom H1 : StoveOn → KitchenHot
-- If someone makes pancakes, they use the stove.
axiom H2 : ∀ x : Person, MakesPancakes x → UsesStove x
-- Either the stove was on, or no one used the stove.
axiom H3 : StoveOn ∨ ∀ x : Person, ¬UsesStove x
-- If someone is hungry, they make pancakes.
axiom H4 : ∀ x : Person, Hungry x → MakesPancakes x
-- If Sam did not eat dinner last night, he was hungry this morning.
axiom H5 : ¬AteDinnerLastNight Sam → Hungry Sam
-- The kitchen did not get hot.
axiom H6 : ¬KitchenHot

theorem Sam_ate_dinner_last_night : AteDinnerLastNight Sam := by
  have hFalse : False :=
    Or.elim H3
      (fun hOn => H6 (H1 hOn))
      (fun hNoUse =>
        Exists.elim H0 (fun w mw =>
          have uw : UsesStove w := H2 w mw
          show False from hNoUse w uw))
  exact False.elim hFalse
</lean>

ANSWER: Yes
