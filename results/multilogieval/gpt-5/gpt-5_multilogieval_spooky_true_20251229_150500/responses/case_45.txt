<lean>
axiom obj : Type
axiom Matt : obj

axiom Watch : obj → Prop
axiom Skills : obj → Prop
axiom Redo : obj → Prop
axiom Read : obj → Prop
axiom Ideas : obj → Prop

-- If someone watches DIY videos, they gain decorating skills.
axiom A1 : ∀ x : obj, Watch x → Skills x
-- With these skills, they can redo their living room.
axiom A2 : ∀ x : obj, Skills x → Redo x
-- Either they watched DIY videos or read interior design blogs (or both).
axiom A3 : Watch Matt ∨ Read Matt
-- If someone reads interior design blogs, they get new decorating ideas.
axiom A4 : ∀ x : obj, Read x → Ideas x

theorem matt_gets_ideas : (¬Redo Matt) → Ideas Matt := by
  intro hNotRedo
  have hNotSkills : ¬Skills Matt := by
    intro hSkills
    exact hNotRedo (A2 Matt hSkills)
  have hNotWatch : ¬Watch Matt := by
    intro hWatch
    exact hNotSkills (A1 Matt hWatch)
  have hRead : Read Matt := by
    exact Or.elim A3
      (fun hWatch => False.elim (hNotWatch hWatch))
      (fun hRead => hRead)
  exact A4 Matt hRead
</lean>

ANSWER: Yes
