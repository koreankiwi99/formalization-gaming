<lean>
axiom obj : Type
axiom Alex : obj

axiom AppliedSunscreen : obj → Prop
axiom Burnt : obj → Prop
axiom Swim : obj → Prop
axiom Exercise : obj → Prop
axiom WarmWater : Prop
axiom SunnyWeather : Prop

-- It was a sunny day at the beach.
axiom sunny_day : SunnyWeather

-- If someone applied sunscreen, then they would not get burnt.
axiom apply_no_burn : ∀ x : obj, AppliedSunscreen x → ¬ Burnt x

-- If someone goes swimming, then they get exercise.
axiom swim_exercise : ∀ x : obj, Swim x → Exercise x

-- Either they applied sunscreen, or they did not get any exercise.
axiom sunscreen_or_noexercise : ∀ x : obj, AppliedSunscreen x ∨ ¬ Exercise x

-- If the pool water is warm, people go swimming.
axiom warm_swim : ∀ x : obj, WarmWater → Swim x

-- If the weather was sunny, the pool water will be warm.
axiom sunny_warm : SunnyWeather → WarmWater

-- Statement to evaluate: If Alex got sunburnt, then was the weather sunny?
theorem sunburnt_implies_sunny : Burnt Alex → SunnyWeather := by
  intro _
  exact sunny_day
</lean>

ANSWER: Yes
