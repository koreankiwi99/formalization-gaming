<lean>
axiom obj : Type
axiom Alex : obj

axiom Weekend : obj → Prop
axiom Relax : obj → Prop
axiom Finished : obj → Prop
axiom ClearMind : obj → Prop
axiom Organized : obj → Prop
axiom MadeTodoList : obj → Prop

-- If it is the weekend coming up for someone, then they will relax.
axiom R1 : ∀ x : obj, Weekend x → Relax x
-- If someone finishes their tasks, then they have a clear mind.
axiom R2 : ∀ x : obj, Finished x → ClearMind x
-- Either the weekend was coming, or they did not have a clear mind.
axiom R3 : ∀ x : obj, Weekend x ∨ ¬ ClearMind x
-- If someone is organized, then they finish their tasks.
axiom R4 : ∀ x : obj, Organized x → Finished x
-- If they made a to-do list, then they are organized.
axiom R5 : ∀ x : obj, MadeTodoList x → Organized x

-- From "Alex does not relax", we can derive "Alex did not make a to-do list".
theorem not_todo_of_not_relax : ¬ Relax Alex → ¬ MadeTodoList Alex :=
by
  intro hNotRelax
  -- From ¬Relax and (Weekend → Relax), we get ¬Weekend.
  have hNotWeekend : ¬ Weekend Alex :=
  by
    intro hW
    exact hNotRelax (R1 Alex hW)
  -- From (Weekend ∨ ¬ClearMind) and ¬Weekend, we get ¬ClearMind.
  have hNotClear : ¬ ClearMind Alex :=
  by
    have hDisj := R3 Alex
    cases hDisj with
    | inl hW =>
        intro _
        exact hNotWeekend hW
    | inr hNC =>
        exact hNC
  -- From MadeTodoList → Organized → Finished → ClearMind, contrapose with ¬ClearMind to get ¬MadeTodoList.
  intro hTodo
  have hOrg : Organized Alex := R5 Alex hTodo
  have hFin : Finished Alex := R4 Alex hOrg
  have hClr : ClearMind Alex := R2 Alex hFin
  exact hNotClear hClr
</lean>
