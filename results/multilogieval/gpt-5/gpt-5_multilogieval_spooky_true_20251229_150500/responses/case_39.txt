axiom A3 : GoodAtMath → DoWell
-- Either it was Monday, or they did not study hard, or it was Monday and they did not study hard.
axiom A4 : Monday ∨ ¬StudiedHard ∨ (Monday ∧ ¬StudiedHard)

-- From "no quiz" we can derive "not good at math" (for Alex).
theorem no_quiz_implies_not_good : ¬Quiz → ¬GoodAtMath := by
  intro hNoQuiz
  intro hGood
  -- From ¬Quiz and (Monday → Quiz), deduce ¬Monday.
  have hNotMonday : ¬Monday := by
    intro hMon
    exact hNoQuiz (A1 hMon)
  -- From the trilemma and ¬Monday, deduce ¬StudiedHard.
  have hNotSH : ¬StudiedHard := by
    have h := A4
    cases h with
    | inl hMon =>
        -- From Monday contradicting ¬Monday, derive ¬StudiedHard by ex falso.
        exact (by intro _; exact False.elim (hNotMonday hMon))
    | inr h2 =>
      cases h2 with
      | inl hNotSH => exact hNotSH
      | inr hMonAndNotSH => exact hMonAndNotSH.right
  -- From GoodAtMath, get DoWell, then StudiedHard, contradicting ¬StudiedHard.
  have hDW : DoWell := A3 hGood
  have hSH : StudiedHard := A2 hDW
  exact hNotSH hSH
