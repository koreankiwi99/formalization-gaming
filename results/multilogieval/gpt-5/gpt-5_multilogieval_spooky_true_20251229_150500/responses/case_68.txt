axiom Person : Type
axiom Emma : Person

axiom StudyHard : Person → Prop
axiom SleepEnoughBeforeExam : Person → Prop
axiom PerformWell : Person → Prop
axiom AceTest : Person → Prop

-- Belief/premise: If someone studies hard, then if they get enough sleep before an exam, they will perform well and ace the test.
axiom belief : ∀ x : Person, StudyHard x → SleepEnoughBeforeExam x → (PerformWell x ∧ AceTest x)

-- Facts about Emma
axiom sleep_Emma : SleepEnoughBeforeExam Emma
axiom not_ace_Emma : ¬ AceTest Emma

-- From the premises, Emma did not study hard.
theorem emma_not_study_hard : ¬ StudyHard Emma :=
by
  intro h
  have hpair := belief Emma h sleep_Emma
  have hAce : AceTest Emma := And.right hpair
  exact not_ace_Emma hAce
