<lean>
axiom Team : Type
axiom A : Team
axiom B : Team

axiom ScoreMore : Team → Prop
axiom Win : Team → Prop
axiom FumbleMany : Team → Prop
axiom Lose : Team → Prop

axiom H1 : ScoreMore A → Win A
axiom H2 : FumbleMany B → Lose B
axiom H3 : ScoreMore A ∨ FumbleMany B

theorem conclude_score_more_if_not_lose : (¬ Lose B) → ScoreMore A := by
  intro hNotLose
  cases H3 with
  | inl hScore => exact hScore
  | inr hFumble =>
      have hLose : Lose B := H2 hFumble
      have hFalse : False := hNotLose hLose
      exact False.elim hFalse
</lean>

ANSWER: Yes
