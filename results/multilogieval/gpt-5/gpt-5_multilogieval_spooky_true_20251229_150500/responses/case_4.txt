<lean>
axiom M : Prop -- Focused budget on marketing (invested in marketing)
axiom C : Prop -- Production costs increased
axiom G : Prop -- Gained more customers
axiom S : Prop -- Sales increased

axiom marketing_to_customers : M → G
axiom customers_to_sales : G → S
axiom marketing_or_costs_or_both : M ∨ C ∨ (M ∧ C)

theorem costs_increased_if_sales_not_increase : ¬ S → C := by
  intro hNotS
  have hMtoS : M → S := fun hM => customers_to_sales (marketing_to_customers hM)
  have hNotM : ¬ M := fun hM => hNotS (hMtoS hM)
  cases marketing_or_costs_or_both with
  | inl hM =>
    exact False.elim (hNotM hM)
  | inr h =>
    cases h with
    | inl hC => exact hC
    | inr hBoth => exact hBoth.right
</lean>

ANSWER: Yes
