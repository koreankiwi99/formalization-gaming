<lean>
axiom Person : Type
axiom Lucy : Person

axiom Hacked : Person → Prop
axiom ChangedPwd : Person → Prop
axiom NotifiedBank : Person → Prop

-- If a person's account is hacked, they need to reset their password.
axiom R1 : ∀ x : Person, Hacked x → ChangedPwd x
-- If a person detects unauthorized access (hacked), they should notify their bank.
axiom R2 : ∀ x : Person, Hacked x → NotifiedBank x

-- Given statement about Lucy's actions: either she did not change her password or she did not inform her bank.
axiom Lucy_action : ¬ ChangedPwd Lucy ∨ ¬ NotifiedBank Lucy

-- From the above, we can derive that Lucy's account was not hacked.
theorem lucy_not_hacked : ¬ Hacked Lucy := by
  intro h
  have hc : ChangedPwd Lucy := R1 Lucy h
  have hn : NotifiedBank Lucy := R2 Lucy h
  cases Lucy_action with
  | inl hnc => exact hnc hc
  | inr hnn => exact hnn hn
</lean>
