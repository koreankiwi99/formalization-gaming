<lean>
axiom Sub : Prop
axiom Pizza : Prop
axiom GoofOff : Prop
axiom Hyper : Prop
axiom LearnMuch : Prop
axiom Homework : Prop

axiom A1 : Sub → GoofOff
axiom A2 : Pizza → Hyper
axiom A3 : Sub ∨ Pizza
axiom A4 : GoofOff → ¬ LearnMuch
axiom A5 : ¬ LearnMuch → Homework

theorem not_hyper_homework : (¬ Hyper) → Homework := by
  intro hNotHyper
  have hNotPizza : ¬ Pizza := by
    intro hPizza
    exact hNotHyper (A2 hPizza)
  have hSub : Sub := by
    cases A3 with
    | inl hS => exact hS
    | inr hP => exact False.elim (hNotPizza hP)
  have hGoof : GoofOff := A1 hSub
  have hNotLearn : ¬ LearnMuch := A4 hGoof
  exact A5 hNotLearn
</lean>

ANSWER: Yes
