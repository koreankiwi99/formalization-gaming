axiom obj : Type
axiom John : obj

axiom AlarmWentOff : obj → Prop
axiom TimeToGetReady : obj → Prop
axiom MissBus : obj → Prop
axiom LateForSchool : obj → Prop
axiom GetUpLate : obj → Prop
axiom WentToBedLate : obj → Prop

-- Premises
axiom P1 : AlarmWentOff John → TimeToGetReady John
axiom P2 : MissBus John → LateForSchool John
axiom P3 : AlarmWentOff John ∨ ¬ LateForSchool John
axiom P4 : GetUpLate John → MissBus John
axiom P5 : WentToBedLate John → GetUpLate John
axiom H : ¬ TimeToGetReady John  -- "John knows it is not time to get ready" treated as ¬TimeToGetReady

-- From the premises we can actually prove that John did NOT go to bed late.
theorem not_went_to_bed_late : ¬ WentToBedLate John := by
  intro w
  have u : GetUpLate John := P5 w
  have m : MissBus John := P4 u
  have l : LateForSchool John := P2 m
  have notA : ¬ AlarmWentOff John := by
    intro a
    have g : TimeToGetReady John := P1 a
    exact H g
  have notL : ¬ LateForSchool John := by
    apply Or.elim P3
    · intro a
      exact False.elim (notA a)
    · intro nl
      exact nl
  exact notL l
