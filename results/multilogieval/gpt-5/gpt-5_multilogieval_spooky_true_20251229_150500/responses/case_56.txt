<lean>
axiom Person : Type
axiom Emma : Person

axiom SpeaksDaily : Person → Prop
axiom ListensPod : Person → Prop
axiom FluencyImproved : Person → Prop
axiom VocabImproved : Person → Prop
axiom RealConversations : Person → Prop
axiom GainListeningComp : Person → Prop

-- If someone speaks daily with native speakers, their fluency and vocabulary improve.
axiom A1 : ∀ x : Person, SpeaksDaily x → (FluencyImproved x ∧ VocabImproved x)
-- If someone's fluency improves, they can have real conversations.
axiom A2 : ∀ x : Person, FluencyImproved x → RealConversations x
-- Either someone speaks daily with native speakers or they listen to educational podcasts (or both).
axiom A3 : ∀ x : Person, SpeaksDaily x ∨ ListensPod x
-- If someone listens to educational podcasts, they gain valuable listening comprehension.
axiom A4 : ∀ x : Person, ListensPod x → GainListeningComp x

theorem emma_listening_from_no_convo : (¬ RealConversations Emma) → GainListeningComp Emma := by
  intro hNotR
  have s_or_l : SpeaksDaily Emma ∨ ListensPod Emma := A3 Emma
  have hNotS : ¬ SpeaksDaily Emma := by
    intro hS
    have hpair := A1 Emma hS
    have hf : FluencyImproved Emma := And.left hpair
    have r : RealConversations Emma := A2 Emma hf
    exact hNotR r
  cases s_or_l with
  | inl hS =>
      have contra : False := hNotS hS
      exact False.elim contra
  | inr hL =>
      exact A4 Emma hL
</lean>

ANSWER: Yes
