<lean>
axiom Person : Type
axiom BuysAdvanceTickets : Person → Prop
axiom PlansToSee : Person → Prop

axiom GoodReviews : Prop
axiom ManyPeople : Prop
axiom BigStar : Prop
axiom Sequel : Prop

-- If the movie gets good reviews, then a lot of people will go see it.
axiom R1 : GoodReviews → ManyPeople

-- If someone buys advance tickets, then they plan to see that movie.
axiom R2 : ∀ p : Person, BuysAdvanceTickets p → PlansToSee p

-- Either the movie got good reviews, or not many people went to see it, or it got good reviews but attendance was still low.
axiom R3 : GoodReviews ∨ ¬ManyPeople ∨ (GoodReviews ∧ ¬ManyPeople)

-- If a big star is in the movie, then it likely got good reviews.
axiom R4 : BigStar → GoodReviews

-- Either there is a big star in it, or it is a sequel, or both.
axiom R5 : BigStar ∨ Sequel

-- Goal: If attendance at the movie was low (¬ManyPeople), then it is a sequel.
theorem low_implies_sequel : ¬ManyPeople → Sequel := by
  intro hLow
  -- From ¬ManyPeople and GoodReviews → ManyPeople, get ¬GoodReviews
  have hNotGood : ¬GoodReviews := by
    intro hG
    exact hLow (R1 hG)
  -- From ¬GoodReviews and BigStar → GoodReviews, get ¬BigStar
  have hNotBig : ¬BigStar := by
    intro hB
    exact hNotGood (R4 hB)
  -- From BigStar ∨ Sequel and ¬BigStar, deduce Sequel
  cases R5 with
  | inl hB => exact (False.elim (hNotBig hB))
  | inr hS => exact hS
</lean>

ANSWER: Yes
