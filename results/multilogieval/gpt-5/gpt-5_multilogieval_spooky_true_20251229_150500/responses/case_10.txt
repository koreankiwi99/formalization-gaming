<lean>
axiom obj : Type
axiom Alex : obj

axiom StudiedHard : obj → Prop
axiom Participated : obj → Prop
axiom GoodGrades : obj → Prop
axiom NewFriends : obj → Prop
axiom Invited : obj → Prop

axiom R1 : ∀ x : obj, StudiedHard x → GoodGrades x
axiom R2 : ∀ x : obj, Participated x → NewFriends x
axiom R3 : ∀ x : obj, NewFriends x → Invited x

axiom Alex_disj : StudiedHard Alex ∨ Participated Alex
axiom Alex_not_good : ¬ GoodGrades Alex

theorem Alex_invited : Invited Alex := by
  have notStudied : ¬ StudiedHard Alex := fun h => Alex_not_good (R1 Alex h)
  cases Alex_disj with
  | inl hS =>
    exact False.elim (notStudied hS)
  | inr hP =>
    have nf : NewFriends Alex := R2 Alex hP
    exact R3 Alex nf
</lean>

ANSWER: Yes
