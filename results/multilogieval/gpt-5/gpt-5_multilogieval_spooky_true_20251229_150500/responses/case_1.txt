<lean>
axiom obj : Type
axiom Lakers : obj

axiom StrongPerimeter : obj → Prop
axiom ForceTurnovers : obj → Prop
axiom ProtectPaint : obj → Prop
axiom LimitInsideScoring : obj → Prop

axiom A1 : ∀ x : obj, StrongPerimeter x → ForceTurnovers x
axiom A2 : ∀ x : obj, ProtectPaint x → LimitInsideScoring x
axiom A3 : StrongPerimeter Lakers ∨ ProtectPaint Lakers

theorem lakers_force_turnovers_of_not_limit :
  ¬ LimitInsideScoring Lakers → ForceTurnovers Lakers :=
by
  intro hNotLIS
  have hNotPP : ¬ ProtectPaint Lakers :=
    fun hPP => hNotLIS (A2 Lakers hPP)
  have hSP : StrongPerimeter Lakers :=
    Or.elim A3
      (fun hSP => hSP)
      (fun hPP => False.elim (hNotPP hPP))
  exact A1 Lakers hSP
</lean>

ANSWER: Yes
