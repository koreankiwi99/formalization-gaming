<lean>
axiom Person : Type
axiom James : Person
axiom Subway : Person → Prop
axiom Bus : Person → Prop
axiom OnTime : Person → Prop

-- Whether it's the subway or the bus, everyone's commute uses one of them
axiom commute_modes : ∀ x : Person, Subway x ∨ Bus x
-- If someone takes the subway, they'll get to work on time.
axiom subway_on_time : ∀ x : Person, Subway x → OnTime x

-- Given: James did not take the bus downtown.
axiom not_bus_james : ¬ Bus James

theorem james_on_time : OnTime James := by
  have h := commute_modes James
  have hSub : Subway James := Or.resolve_right h not_bus_james
  exact subway_on_time James hSub
</lean>

ANSWER: Yes
