<lean>
axiom Oven350 : Prop
axiom BakeProperly : Prop
axiom Eggs : Prop
axiom Fluffy : Prop
axiom Delicious : Prop
axiom GuestsHappy : Prop

axiom R1 : Oven350 → BakeProperly
axiom R2 : Eggs → Fluffy
axiom R3 : Or Oven350 (Not Eggs)
axiom R4 : BakeProperly → Delicious
axiom R5 : Delicious → GuestsHappy

theorem eggs_implies_guests_happy : Eggs → GuestsHappy := by
  intro hEggs
  have h := R3
  cases h with
  | inl h350 =>
      have hBake : BakeProperly := R1 h350
      have hDel : Delicious := R4 hBake
      exact R5 hDel
  | inr hNotEggs =>
      have : False := hNotEggs hEggs
      exact False.elim this
</lean>

ANSWER: Yes
