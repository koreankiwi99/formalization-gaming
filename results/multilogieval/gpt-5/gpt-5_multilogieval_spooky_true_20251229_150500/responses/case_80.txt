<lean>
axiom Rain : Prop
axiom PoolOpen : Prop
axiom KidsSwim : Prop
axiom Game : Prop
axiom Enjoy : Prop
axiom Smiles : Prop

-- Premises
axiom H1 : ¬Rain → Game
axiom H2 : PoolOpen → KidsSwim
axiom H3 : (¬Rain) ∨ PoolOpen
axiom H4 : Game → Enjoy
axiom H5 : Enjoy → Smiles

-- Conclusion: If the kids did not go for a swim, then the players had smiles at the end.
theorem smiles_if_no_swim : ¬KidsSwim → Smiles := by
  intro hNoSwim
  have hNotOpen : ¬PoolOpen := by
    intro hOpen
    exact hNoSwim (H2 hOpen)
  have hNoRain : ¬Rain := by
    intro hRain
    cases H3 with
    | inl hNotRain => exact hNotRain hRain
    | inr hOpen => exact hNotOpen hOpen
  have hGame : Game := H1 hNoRain
  have hEnjoy : Enjoy := H4 hGame
  exact H5 hEnjoy
</lean>

ANSWER: Yes
