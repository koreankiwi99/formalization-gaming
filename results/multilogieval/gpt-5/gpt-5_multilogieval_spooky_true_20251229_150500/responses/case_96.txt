<lean>
axiom Person : Type
axiom Exam : Type

axiom Alex : Person
axiom Biology : Exam
axiom History : Exam

axiom StudiedHard : Person → Exam → Prop
axiom Confident : Person → Exam → Prop
axiom DoWell : Person → Exam → Prop
axiom SatisfiedWithPerformance : Person → Exam → Prop

axiom SevereWeatherInForecast : Prop
axiom ClassCanceled : Prop

-- If someone studies hard for an exam, they will feel confident going into the test.
axiom A1 : ∀ p e, StudiedHard p e → Confident p e
-- If a student feels confident about the material, they will likely do well on the exam.
axiom A2 : ∀ p e, Confident p e → DoWell p e
-- If a student performs well on an exam, they will be satisfied with their hard work/performance.
axiom A3 : ∀ p e, DoWell p e → SatisfiedWithPerformance p e

-- If there is severe weather in the forecast, schools may cancel class that day.
-- We encode "may" weakly; it gives no determinate outcome about cancellation.
axiom A4 : SevereWeatherInForecast → (ClassCanceled ∨ ¬ ClassCanceled)

-- Either Alex studied diligently for his biology exam, or class was not canceled due to the weather.
-- We encode "class was not canceled due to the weather" as simply "class was not canceled".
axiom A5 : StudiedHard Alex Biology ∨ ¬ ClassCanceled

-- A derivable helper: studying hard for a specific exam yields satisfaction on that exam.
theorem study_to_satisfied : ∀ p e, StudiedHard p e → SatisfiedWithPerformance p e := by
  intros p e h
  apply A3
  apply A2
  apply A1
  exact h

-- Note: From the given premises, there is no connection from severe weather or biology study
-- to satisfaction on Alex's history exam, so we cannot prove:
-- SevereWeatherInForecast → SatisfiedWithPerformance Alex History
</lean>
