<lean>
axiom Rain : Prop
axiom Game : Prop
axiom PoolOpen : Prop
axiom KidsSwim : Prop
axiom PlayersEnjoy : Prop
axiom PlayersSmiles : Prop

axiom A1 : ¬Rain → Game
axiom A2 : PoolOpen → KidsSwim
axiom A3 : ¬Rain ∨ PoolOpen
axiom A4 : Game → PlayersEnjoy
axiom A5 : PlayersEnjoy → PlayersSmiles

theorem smiles_if_no_swim : (¬KidsSwim) → PlayersSmiles := by
  intro hNoSwim
  have hdisj := A3
  cases hdisj with
  | inl hNoRain =>
    have hGame : Game := A1 hNoRain
    have hEnjoy : PlayersEnjoy := A4 hGame
    exact A5 hEnjoy
  | inr hPoolOpen =>
    have hSwim : KidsSwim := A2 hPoolOpen
    have hFalse : False := hNoSwim hSwim
    exact False.elim hFalse
</lean>

ANSWER: Yes
