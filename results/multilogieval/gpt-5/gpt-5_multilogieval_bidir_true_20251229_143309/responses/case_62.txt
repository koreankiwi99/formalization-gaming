<lean>
axiom obj : Type
axiom Mark : obj

axiom MadeFriends : obj → Prop
axiom Enjoy : obj → Prop
axiom JoinClub : obj → Prop
axiom Busy : obj → Prop
axiom Outgoing : obj → Prop
axiom Introduced : obj → Prop

axiom R1 : ∀ x : obj, MadeFriends x → Enjoy x
axiom R2 : ∀ x : obj, JoinClub x → Busy x
axiom R3 : ∀ x : obj, MadeFriends x ∨ ¬ Busy x
axiom R4 : ∀ x : obj, Outgoing x → JoinClub x
axiom R5 : ∀ x : obj, Introduced x → Outgoing x

-- From the premises we can derive that if someone introduced themselves, then they enjoyed school.
theorem introduced_implies_enjoy : ∀ x : obj, Introduced x → Enjoy x :=
by
  intro x hin
  have hout : Outgoing x := R5 x hin
  have hj : JoinClub x := R4 x hout
  have hb : Busy x := R2 x hj
  have hmf : MadeFriends x :=
    Or.elim (R3 x)
      (fun h => h)
      (fun hnb => False.elim (hnb hb))
  exact R1 x hmf

-- Therefore, if Mark did not enjoy school, he did not introduce himself.
theorem not_enjoy_implies_not_introduced_mark : ¬ Enjoy Mark → ¬ Introduced Mark :=
by
  intro hne hin
  have h := introduced_implies_enjoy Mark hin
  exact hne h
