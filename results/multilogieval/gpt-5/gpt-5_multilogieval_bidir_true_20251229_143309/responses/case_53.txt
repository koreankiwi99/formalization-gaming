<lean>
axiom obj : Type
axiom Mary : obj

axiom PracticedOften : obj → Prop
axiom TechniqueImproved : obj → Prop
axiom ArtworkHighQuality : obj → Prop
axiom TookArtClass : obj → Prop
axiom LearnedNewMethods : obj → Prop

axiom A1 : ∀ x : obj, PracticedOften x → TechniqueImproved x
axiom A2 : ∀ x : obj, TechniqueImproved x → ArtworkHighQuality x
axiom A3 : PracticedOften Mary ∨ TookArtClass Mary
axiom A4 : ∀ x : obj, TookArtClass x → LearnedNewMethods x
axiom A5 : ¬ ArtworkHighQuality Mary

theorem mary_learned_new_methods : LearnedNewMethods Mary := by
  have h_q_from_p : PracticedOften Mary → ArtworkHighQuality Mary := by
    intro hp
    apply A2 Mary
    apply A1 Mary
    exact hp
  have h_class : TookArtClass Mary := by
    have hdisj := A3
    cases hdisj with
    | inl hp =>
      have hq : ArtworkHighQuality Mary := h_q_from_p hp
      exact False.elim (A5 hq)
    | inr ha =>
      exact ha
  exact A4 Mary h_class
</lean>

ANSWER: Yes
