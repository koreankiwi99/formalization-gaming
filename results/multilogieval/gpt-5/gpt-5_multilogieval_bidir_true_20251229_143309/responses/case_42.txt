axiom Sunny : Prop
axiom WantPlayOutside : Prop
axiom PlayOutside : Prop
axiom SleepWellAtNight : Prop
axiom FluSeasonStarted : Prop
axiom ManyKidsSick : Prop

-- If the weather is sunny, children will want to play outside.
axiom R1 : Sunny → WantPlayOutside
-- If flu season begins, then many children will get sick.
axiom R2 : FluSeasonStarted → ManyKidsSick
-- Either it is sunny outside, or not many kids get the flu, or it was sunny and many kids did not get sick.
axiom R3 : Sunny ∨ ¬ManyKidsSick ∨ (Sunny ∧ ¬ManyKidsSick)
-- If children play outside, they will sleep well at night.
axiom R4 : PlayOutside → SleepWellAtNight

-- Given: Flu season started this month.
axiom T1 : FluSeasonStarted

-- From flu season starting and the disjunction, we can derive that it is sunny.
theorem sunny_from_flu_and_disj : Sunny := by
  have ms : ManyKidsSick := R2 T1
  cases R3 with
  | inl hs => exact hs
  | inr h2 =>
    cases h2 with
    | inl hnot => exact False.elim (hnot ms)
    | inr hpair =>
      exact False.elim (hpair.right ms)

-- From sunny, children will want to play outside.
theorem want_play_outside : WantPlayOutside :=
  R1 sunny_from_flu_and_disj
