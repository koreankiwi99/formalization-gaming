<lean>
axiom obj : Type
axiom John : obj

axiom Early : obj → Prop
axiom LongLunch : obj → Prop
axiom Recharged : obj → Prop
axiom Productive : obj → Prop
axiom BringTreats : obj → Prop
axiom Birthday : Prop

-- If someone finishes early, they can take a long lunch
axiom R1 : ∀ x : obj, Early x → LongLunch x
-- If it's a birthday, people bring treats
axiom R2 : ∀ x : obj, Birthday → BringTreats x
-- Either John's project gets done early, or it's a birthday (or both)
axiom R3 : Early John ∨ Birthday
-- Long lunch implies recharged
axiom R4 : ∀ x : obj, LongLunch x → Recharged x
-- Recharged implies very productive afternoon
axiom R5 : ∀ x : obj, Recharged x → Productive x

theorem john_productive_if_no_treats : (¬ BringTreats John) → Productive John := by
  intro hnb
  have hb : Birthday → BringTreats John := R2 John
  have hnotB : ¬ Birthday := fun b => hnb (hb b)
  have hDisj : Early John ∨ Birthday := R3
  have hE : Early John := by
    cases hDisj with
    | inl hE => exact hE
    | inr hB => exact False.elim (hnotB hB)
  have hLL : LongLunch John := R1 John hE
  have hR : Recharged John := R4 John hLL
  exact R5 John hR
</lean>

ANSWER: Yes
