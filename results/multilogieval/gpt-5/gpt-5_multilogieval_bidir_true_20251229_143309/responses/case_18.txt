<lean>
axiom obj : Type
axiom Sam : obj

axiom Impress : obj → Prop
axiom Promotion : obj → Prop
axiom TookExtra : obj → Prop
axiom Experience : obj → Prop
axiom Assigned : obj → Prop

axiom R1 : ∀ x : obj, Impress x → Promotion x
axiom R2 : ∀ x : obj, TookExtra x → Experience x
axiom R3 : ∀ x : obj, Experience x → Assigned x

-- The specific employee (Sam) either impressed the manager or took on extra work
axiom D : Impress Sam ∨ TookExtra Sam

theorem sam_assigned (Hnp : ¬ Promotion Sam) : Assigned Sam := by
  cases D with
  | inl hImpress =>
      have hProm : Promotion Sam := R1 Sam hImpress
      exact False.elim (Hnp hProm)
  | inr hExtra =>
      have hExp : Experience Sam := R2 Sam hExtra
      exact R3 Sam hExp
</lean>

ANSWER: Yes
