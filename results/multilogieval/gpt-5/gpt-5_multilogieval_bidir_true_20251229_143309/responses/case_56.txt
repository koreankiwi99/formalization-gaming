<lean>
axiom person : Type
axiom Emma : person

axiom SpeaksDaily : person → Prop
axiom ListensPods : person → Prop
axiom FluencyImproves : person → Prop
axiom VocabularyImproves : person → Prop
axiom AbleConvo : person → Prop
axiom GainsListening : person → Prop

-- If someone speaks daily with native speakers, their fluency and vocabulary improve.
axiom A1 : ∀ x : person, SpeaksDaily x → (FluencyImproves x ∧ VocabularyImproves x)
-- If someone's fluency improves, they can have real conversations.
axiom A2 : ∀ x : person, FluencyImproves x → AbleConvo x
-- Either someone speaks daily with native speakers or they listen to educational podcasts (or both).
axiom A3 : ∀ x : person, SpeaksDaily x ∨ ListensPods x
-- If someone listens to educational podcasts, they gain valuable listening comprehension.
axiom A4 : ∀ x : person, ListensPods x → GainsListening x

theorem emma_gains : (¬ AbleConvo Emma) → GainsListening Emma := by
  intro hnot
  have hdisj := A3 Emma
  cases hdisj with
  | inl hsd =>
    have hf_and := A1 Emma hsd
    have hf : FluencyImproves Emma := And.left hf_and
    have hable : AbleConvo Emma := A2 Emma hf
    have hfalse : False := hnot hable
    exact False.elim hfalse
  | inr hlp =>
    exact A4 Emma hlp
</lean>

ANSWER: Yes
