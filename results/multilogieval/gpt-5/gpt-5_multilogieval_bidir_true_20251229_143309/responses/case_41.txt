<lean>
axiom Person : Type
axiom Priya : Person

axiom WaterDaily : Person → Prop
axiom GrowWell : Person → Prop
axiom LackedWater : Person → Prop
axiom WiltDie : Person → Prop

axiom Drought : Prop
axiom Restrictions : Prop

-- If someone waters their plants daily, the plants will grow well.
axiom R1 : ∀ x : Person, WaterDaily x → GrowWell x
-- If there is a drought, then water restrictions will be imposed.
axiom R2 : Drought → Restrictions
-- Either Priya watered daily or there was a drought (or both).
axiom E1 : WaterDaily Priya ∨ Drought
-- If restrictions were imposed, then all plants would not get enough water.
axiom R3 : Restrictions → ∀ x : Person, LackedWater x
-- If a person's plants lacked water, they would wilt and die.
axiom R4 : ∀ x : Person, LackedWater x → WiltDie x

theorem priya_wilt_if_not_grow : ¬ GrowWell Priya → WiltDie Priya := by
  intro hnot
  exact Or.elim E1
    (fun hwd => False.elim (hnot (R1 Priya hwd)))
    (fun hd => R4 Priya ((R3 (R2 hd)) Priya))
</lean>

ANSWER: Yes
