axiom EnoughSleepBeforeExam : obj → Prop
axiom PerformWell : obj → Prop
axiom AceTest : obj → Prop

-- Belief rule: If someone studies hard, then if they get enough sleep before an exam,
-- they will perform well and ace the test.
axiom BeliefRule :
  ∀ x : obj, StudiesHard x → (EnoughSleepBeforeExam x → (PerformWell x ∧ AceTest x))

-- Facts about Emma
axiom Sleep_Emma : EnoughSleepBeforeExam Emma
axiom NotAce_Emma : ¬ AceTest Emma

-- From the rule and facts, we can derive that Emma did not study hard.
theorem not_study_hard_Emma : ¬ StudiesHard Emma :=
by
  intro hStudy
  have hCond : EnoughSleepBeforeExam Emma → (PerformWell Emma ∧ AceTest Emma) :=
    BeliefRule Emma hStudy
  have hPair : PerformWell Emma ∧ AceTest Emma := hCond Sleep_Emma
  have hAce : AceTest Emma := hPair.right
  exact NotAce_Emma hAce
