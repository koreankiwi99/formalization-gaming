<lean>
axiom obj : Type
axiom Mary : obj

axiom Learned : obj → Prop
axiom UnderstoodNeeds : obj → Prop
axiom ThrivedPlants : obj → Prop
axiom SoughtAdvice : obj → Prop

-- If someone learned about botany, they understand plant needs
axiom R1 : ∀ x : obj, Learned x → UnderstoodNeeds x
-- If someone understands plant needs, their plants thrive
axiom R2 : ∀ x : obj, UnderstoodNeeds x → ThrivedPlants x
-- Either they learned about botany or they sought advice (or both)
axiom R3 : Learned Mary ∨ SoughtAdvice Mary
-- Mary's plants did not thrive
axiom T : ¬ ThrivedPlants Mary

theorem mary_sought_advice : SoughtAdvice Mary := by
  have hLT : Learned Mary → ThrivedPlants Mary :=
    fun hL => R2 Mary (R1 Mary hL)
  have hNotL : ¬ Learned Mary :=
    fun hL => T (hLT hL)
  cases R3 with
  | inl hL => exact False.elim (hNotL hL)
  | inr hS => exact hS
</lean>

ANSWER: Yes
