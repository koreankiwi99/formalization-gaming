<lean>
axiom Party : Prop
axiom Drums : Prop
axiom Noise : Prop
axiom Complaints : Prop
axiom Police : Prop

axiom R1 : Party → Noise
axiom R2 : Drums → Noise
axiom R3 : Party ∨ Drums
axiom R4 : Party → Complaints
axiom R5 : Complaints → Police

theorem no_noise_implies_police : ¬Noise → Police :=
by
  intro hno
  cases R3 with
  | inl hParty =>
      have hC : Complaints := R4 hParty
      exact R5 hC
  | inr hDrums =>
      have hN : Noise := R2 hDrums
      have hFalse : False := hno hN
      exact False.elim hFalse
</lean>

ANSWER: Yes
