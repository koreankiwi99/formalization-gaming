<lean>
axiom A_more : Prop
axiom A_win : Prop
axiom B_fumbles : Prop
axiom B_lose : Prop

axiom R1 : A_more → A_win
axiom R2 : B_fumbles → B_lose
axiom E : A_more ∨ B_fumbles

theorem conclusion : (¬ B_lose) → A_more := by
  intro hNotLose
  cases E with
  | inl hMore =>
      exact hMore
  | inr hFumbles =>
      have hFalse : False := hNotLose (R2 hFumbles)
      exact False.elim hFalse
</lean>

ANSWER: Yes
