<lean>
axiom Person : Type
axiom James : Person

axiom Started : Person → Prop
axiom Motivated : Person → Prop
axiom FollowRoutine : Person → Prop
axiom Frustrated : Person → Prop
axiom Indulged : Person → Prop

axiom GymClosed : Prop

axiom R_Start_Motivated : ∀ x : Person, Started x → Motivated x
axiom R_GymClosed_NoFollow : ∀ x : Person, GymClosed → ¬ FollowRoutine x
axiom E_Started_or_GymClosed : Started James ∨ GymClosed
axiom R_NoFollow_Frustrated : ∀ x : Person, ¬ FollowRoutine x → Frustrated x
axiom R_Frustrated_Indulged : ∀ x : Person, Frustrated x → Indulged x

axiom NotMotivatedJames : ¬ Motivated James

theorem james_indulged : Indulged James :=
by
  have notStarted : ¬ Started James :=
    fun h => NotMotivatedJames (R_Start_Motivated James h)
  have gymClosed : GymClosed :=
    Or.elim E_Started_or_GymClosed
      (fun h => False.elim (notStarted h))
      (fun h => h)
  have noFollow : ¬ FollowRoutine James := R_GymClosed_NoFollow James gymClosed
  have fr : Frustrated James := R_NoFollow_Frustrated James noFollow
  exact R_Frustrated_Indulged James fr
</lean>

ANSWER: Yes
