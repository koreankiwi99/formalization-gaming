<lean>
axiom obj : Type
axiom Alex : obj

axiom Studied : obj → Prop
axiom GoodGrades : obj → Prop
axiom Extracurricular : obj → Prop
axiom Friends : obj → Prop
axiom Invited : obj → Prop

-- Context-specific axioms for the student (Alex)
axiom H1 : Studied Alex → GoodGrades Alex
axiom H2 : Extracurricular Alex → Friends Alex
axiom H3 : Studied Alex ∨ Extracurricular Alex
axiom H4 : ∀ x : obj, Friends x → Invited x

-- Given: Alex did not get good grades
axiom NG : ¬ GoodGrades Alex

theorem alex_invited : Invited Alex := by
  -- From H1 and NG, deduce Alex did not study
  have hns : ¬ Studied Alex := by
    intro hS
    exact NG (H1 hS)
  -- From the disjunction H3 and ¬Studied, deduce Extracurricular Alex
  have hP : Extracurricular Alex := by
    cases H3 with
    | inl hS => exact False.elim (hns hS)
    | inr hP => exact hP
  -- Get Friends Alex, then Invited Alex
  have hF : Friends Alex := H2 hP
  exact H4 Alex hF
</lean>

ANSWER: Yes
