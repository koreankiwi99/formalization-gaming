axiom Winning : Prop
axiom Rivalry : Prop
axiom MoreInvested : Prop
axiom LoudEnergized : Prop
axiom Electric : Prop
axiom Unforgettable : Prop

axiom H1 : Winning → LoudEnergized
axiom H2 : Rivalry → MoreInvested
axiom H3 : Winning ∨ Rivalry
axiom H4 : LoudEnergized → Electric
axiom H5 : Electric → Unforgettable

theorem atmosphere_electric : (¬ MoreInvested) → Electric := by
  intro hNotM
  have hNotR : ¬ Rivalry := by
    intro hR
    exact hNotM (H2 hR)
  have hW : Winning := Or.elim H3 (fun hw => hw) (fun hr => False.elim (hNotR hr))
  exact H4 (H1 hW)
