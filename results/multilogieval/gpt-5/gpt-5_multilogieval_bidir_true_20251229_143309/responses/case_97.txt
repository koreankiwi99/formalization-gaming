<lean>
axiom S : Prop        -- there is a substitute teacher
axiom P : Prop        -- cafeteria serves pizza for lunch
axiom G : Prop        -- students goof off in class
axiom H : Prop        -- kids are hyper all afternoon
axiom M : Prop        -- students learn much that day
axiom W : Prop        -- students have lots of homework

axiom R1 : S → G      -- if there's a substitute teacher, students goof off
axiom R2 : P → H      -- if pizza is served, kids are hyper
axiom R3 : S ∨ P      -- either there is a sub or there is pizza
axiom R4 : G → ¬ M    -- if students goof off, they won't learn much
axiom R5 : ¬ M → W    -- if they don't learn much, they have lots of homework

theorem notH_implies_homework : (¬ H) → W := by
  intro notH
  have notP : ¬ P := by
    intro p
    exact notH (R2 p)
  have s : S := by
    cases R3 with
    | inl s => exact s
    | inr p => exact False.elim (notP p)
  have g : G := R1 s
  have notM : ¬ M := R4 g
  exact R5 notM
</lean>

ANSWER: Yes
