<lean>
axiom obj : Type
axiom Matt : obj

axiom Watched : obj → Prop
axiom Skills : obj → Prop
axiom Redo : obj → Prop
axiom ReadBlogs : obj → Prop
axiom NewIdeas : obj → Prop

-- If someone watches DIY videos, they gain decorating skills
axiom R_watch_skills : ∀ x : obj, Watched x → Skills x
-- With these skills, they redo their living room
axiom R_skills_redo : ∀ x : obj, Skills x → Redo x
-- Either they watched DIY videos or they read interior design blogs (or both)
axiom R_watch_or_blogs : ∀ x : obj, Watched x ∨ ReadBlogs x
-- If someone reads interior design blogs, then they get new decorating ideas
axiom R_blogs_ideas : ∀ x : obj, ReadBlogs x → NewIdeas x

theorem matt_gets_new_ideas (hNotRedo : ¬Redo Matt) : NewIdeas Matt := by
  -- From ¬Redo and (Skills → Redo), we get ¬Skills
  have hNotSkills : ¬Skills Matt := by
    intro hS
    exact hNotRedo (R_skills_redo Matt hS)
  -- From ¬Skills and (Watched → Skills), we get ¬Watched
  have hNotWatched : ¬Watched Matt := by
    intro hW
    exact hNotSkills (R_watch_skills Matt hW)
  -- From Watched ∨ ReadBlogs and ¬Watched, we get ReadBlogs
  have hReadBlogs : ReadBlogs Matt := by
    have hDisj := R_watch_or_blogs Matt
    cases hDisj with
    | inl hW => exact False.elim (hNotWatched hW)
    | inr hB => exact hB
  -- From ReadBlogs, we get NewIdeas
  exact R_blogs_ideas Matt hReadBlogs
</lean>

ANSWER: Yes
