axiom Product : Type
axiom Acme : Company
axiom Gadget : Product

axiom Develops : Company → Product → Prop
axiom MarketsEffectively : Company → Product → Prop
axiom CommerciallySuccessful : Product → Prop
axiom Innovative : Product → Prop

-- If a company develops an innovative product, then if they market it effectively, the product will be commercially successful.
axiom R1 : ∀ c p, (Develops c p ∧ Innovative p) → (MarketsEffectively c p → CommerciallySuccessful p)

-- Given facts:
axiom T1 : MarketsEffectively Acme Gadget
axiom T2 : ¬ CommerciallySuccessful Gadget

-- From the rule and the facts, we can derive that Acme did not both develop the gadget and have it be innovative.
theorem not_dev_and_innovative : ¬ (Develops Acme Gadget ∧ Innovative Gadget) := by
  intro h
  have hImp : MarketsEffectively Acme Gadget → CommerciallySuccessful Gadget := R1 Acme Gadget h
  have hSucc : CommerciallySuccessful Gadget := hImp T1
  exact T2 hSucc

-- We cannot prove Develops Acme Gadget ∧ Innovative Gadget from the given premises.
