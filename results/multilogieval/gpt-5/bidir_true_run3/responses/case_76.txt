<lean>
axiom obj : Type
axiom Alex : obj

axiom WeekendComing : Prop
axiom Relax : obj → Prop
axiom FinishTasks : obj → Prop
axiom ClearMind : obj → Prop
axiom Organized : obj → Prop
axiom ToDoList : obj → Prop

-- If it is the weekend coming up, then they (anyone) relax.
axiom R1 : ∀ x : obj, WeekendComing → Relax x
-- If someone finishes their tasks, then they have a clear mind.
axiom R2 : ∀ x : obj, FinishTasks x → ClearMind x
-- Either the weekend is coming, or they (anyone) do not have a clear mind.
axiom R3 : ∀ x : obj, WeekendComing ∨ ¬ ClearMind x
-- If someone is organized, then they finish their tasks.
axiom R4 : ∀ x : obj, Organized x → FinishTasks x
-- If they made a to-do list, then they are organized.
axiom R5 : ∀ x : obj, ToDoList x → Organized x

-- From the rules, making a to-do list implies having a clear mind.
theorem todo_implies_clearmind : ∀ x : obj, ToDoList x → ClearMind x :=
  fun x => fun tl => R2 x (R4 x (R5 x tl))

-- Given that Alex does not relax this weekend, we can show Alex did not make a to-do list.
theorem no_todo_from_not_relax : (¬ Relax Alex) → ¬ ToDoList Alex :=
  fun HnotRelax =>
    -- From HnotRelax and R1, infer that the weekend is not coming.
    let notW : ¬ WeekendComing :=
      fun w =>
        let relaxA : Relax Alex := R1 Alex w
        HnotRelax relaxA
    -- From R3 and notW, infer Alex does not have a clear mind.
    let notCM : ¬ ClearMind Alex :=
      match R3 Alex with
      | Or.inl w =>
          by
            intro cm
            exact False.elim (notW w)
      | Or.inr ncm => ncm
    -- Conclude Alex did not make a to-do list.
    by
      intro tl
      exact notCM (todo_implies_clearmind Alex tl)
