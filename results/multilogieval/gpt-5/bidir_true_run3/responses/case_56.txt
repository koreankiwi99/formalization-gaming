<lean>
axiom obj : Type
axiom Emma : obj

axiom SpeaksDailyWithNatives : obj → Prop
axiom FluencyImproves : obj → Prop
axiom VocabularyImproves : obj → Prop
axiom HasRealConversations : obj → Prop
axiom ListensEducationalPodcasts : obj → Prop
axiom GainsListeningComprehension : obj → Prop

-- If someone speaks daily with native speakers, their fluency and vocabulary improve.
axiom R1 : ∀ x : obj, SpeaksDailyWithNatives x → (FluencyImproves x ∧ VocabularyImproves x)
-- If someone's fluency improves, they can have real conversations.
axiom R2 : ∀ x : obj, FluencyImproves x → HasRealConversations x
-- Either someone speaks daily with natives or listens to educational podcasts (or both).
axiom R3 : ∀ x : obj, SpeaksDailyWithNatives x ∨ ListensEducationalPodcasts x
-- If someone listens to educational podcasts, they gain valuable listening comprehension.
axiom R4 : ∀ x : obj, ListensEducationalPodcasts x → GainsListeningComprehension x

theorem emma_listening_if_not_convo :
  (¬ HasRealConversations Emma) → GainsListeningComprehension Emma :=
by
  intro Hnr
  -- From ¬real, derive ¬speaks (using speaks → fluency → real)
  have ns : ¬ SpeaksDailyWithNatives Emma :=
    by
      intro Hs
      have hflu_and : FluencyImproves Emma ∧ VocabularyImproves Emma := R1 Emma Hs
      have hflu : FluencyImproves Emma := And.left hflu_and
      have hreal : HasRealConversations Emma := R2 Emma hflu
      exact Hnr hreal
  -- From (speaks ∨ listens) and ¬speaks, deduce listens
  have disj : SpeaksDailyWithNatives Emma ∨ ListensEducationalPodcasts Emma := R3 Emma
  have Hl : ListensEducationalPodcasts Emma :=
    by
      cases disj with
      | inl Hs =>
          have : False := ns Hs
          exact False.elim this
      | inr Hl =>
          exact Hl
  -- Listening implies gaining valuable listening comprehension
  exact R4 Emma Hl
</lean>

ANSWER: Yes
