<lean>
axiom WaterEnough : Prop
axiom GrowWell : Prop
axiom PestsInvaded : Prop
axiom CropsDamaged : Prop
axiom HarvestBountiful : Prop
axiom GoodProfit : Prop
axiom FarmerWateredAdequately : Prop

axiom R1 : WaterEnough → GrowWell
axiom R2 : PestsInvaded → CropsDamaged
axiom R3 : FarmerWateredAdequately ∨ (¬CropsDamaged ∨ (WaterEnough ∧ CropsDamaged))
axiom R4 : GrowWell → HarvestBountiful
axiom R5 : HarvestBountiful → GoodProfit
axiom R6 : FarmerWateredAdequately → WaterEnough

theorem pests_profit : PestsInvaded → GoodProfit := by
  intro hP
  have hD : CropsDamaged := R2 hP
  have hW : WaterEnough := by
    cases R3 with
    | inl hA =>
      exact R6 hA
    | inr hrest =>
      cases hrest with
      | inl hNotD =>
        have : False := hNotD hD
        exact False.elim this
      | inr hWD =>
        exact hWD.left
  have hG : GrowWell := R1 hW
  have hB : HarvestBountiful := R4 hG
  exact R5 hB
</lean>

ANSWER: Yes
