<lean>
axiom obj : Type
axiom Lauren : obj

-- Predicates
axiom TrainHardWeekly : obj → Prop
axiom StretchedAfterRuns : obj → Prop
axiom RecoverySerious : obj → Prop
axiom FocusFlexibility : obj → Prop
axiom FasterRaces : obj → Prop

-- Rules from the context
axiom R1 : ∀ x : obj, TrainHardWeekly x → FasterRaces x
axiom R2 : ∀ x : obj, TrainHardWeekly x ∨ StretchedAfterRuns x
axiom R3 : ∀ x : obj, StretchedAfterRuns x → RecoverySerious x
axiom R4 : ∀ x : obj, RecoverySerious x → FocusFlexibility x

theorem lauren_focus_from_not_faster :
  (¬ FasterRaces Lauren) → FocusFlexibility Lauren := by
  intro h
  have disj := R2 Lauren
  cases disj with
  | inl thw =>
      have fr := R1 Lauren thw
      exact False.elim (h fr)
  | inr sar =>
      have rs := R3 Lauren sar
      exact R4 Lauren rs
</lean>

ANSWER: Yes
