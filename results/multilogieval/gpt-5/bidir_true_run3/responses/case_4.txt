<lean>
axiom Marketing : Prop
axiom MoreCustomers : Prop
axiom SalesIncrease : Prop
axiom ProdCostsIncreased : Prop

axiom R1 : Marketing → MoreCustomers
axiom R2 : MoreCustomers → SalesIncrease
axiom R3 : Marketing ∨ ProdCostsIncreased
axiom NotSales : ¬ SalesIncrease

theorem production_costs_increased : ProdCostsIncreased := by
  have hMS : Marketing → SalesIncrease := fun hm => R2 (R1 hm)
  have hNotM : ¬ Marketing := fun hm => NotSales (hMS hm)
  exact Or.elim R3 (fun hm => False.elim (hNotM hm)) (fun hp => hp)
</lean>

ANSWER: Yes
