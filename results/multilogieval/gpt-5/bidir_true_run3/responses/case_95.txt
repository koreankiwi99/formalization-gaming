<lean>
axiom Person : Type
axiom Mary : Person

axiom ClearUp : Prop
axiom GameOn : Prop
axiom FansCheer : Prop
axiom ExcitingNight : Prop

axiom Finished : Person → Prop
axiom MetFriends : Person → Prop

axiom A1 : ClearUp → GameOn
axiom A2 : ∀ p : Person, Finished p → MetFriends p
axiom A3 : ClearUp ∨ Finished Mary
axiom A4 : GameOn → FansCheer
axiom A5 : FansCheer → ExcitingNight

theorem exciting_if_mary_did_not_meet : (MetFriends Mary → False) → ExcitingNight := by
  intro hNotMeet
  have hNotFin : Finished Mary → False := by
    intro hfin
    exact hNotMeet (A2 Mary hfin)
  have hClear : ClearUp := by
    cases A3 with
    | inl hcu => exact hcu
    | inr hfin => exact False.elim (hNotFin hfin)
  exact A5 (A4 (A1 hClear))
</lean>

ANSWER: Yes
