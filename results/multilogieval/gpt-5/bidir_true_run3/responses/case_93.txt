<lean>
axiom obj : Type
axiom Team : obj
axiom Star : obj

axiom Practiced : obj → Prop
axiom Prepared : obj → Prop
axiom Wins : obj → Prop
axiom Celebrate : obj → Prop
axiom Injured : obj → Prop
axiom CannotPlay : obj → Prop

-- If the team practices hard, they will be prepared for the big game.
axiom A1 : Practiced Team → Prepared Team
-- If the star player is injured, he cannot play in the game.
axiom A2 : Injured Star → CannotPlay Star
-- Either the team practiced hard, or the star player is not injured.
axiom A3 : Practiced Team ∨ ¬Injured Star
-- When the team is well-prepared, the team wins the important game.
axiom A4 : Prepared Team → Wins Team
-- If the team wins the important game, they will celebrate with a team dinner afterwards.
axiom A5 : Wins Team → Celebrate Team

-- Statement: If the star player is injured, the team will celebrate with a team dinner after the big game.
theorem injured_implies_celebrate : Injured Star → Celebrate Team :=
by
  intro hI
  have hP : Practiced Team := Or.resolve_right A3 (fun hNotInj => hNotInj hI)
  have hPrep : Prepared Team := A1 hP
  have hWin : Wins Team := A4 hPrep
  exact A5 hWin
</lean>

ANSWER: Yes
