<lean>
axiom Person : Type
axiom Exam : Type

axiom Emma : Person
axiom BiologyExam : Exam

axiom StudiesHard : Person → Prop
axiom EnoughSleepBefore : Person → Exam → Prop
axiom PerformWell : Person → Exam → Prop
axiom AceTest : Person → Exam → Prop

-- If someone studies hard, then if they get enough sleep before an exam, they will perform well and ace the test.
axiom Rule : ∀ x : Person, ∀ e : Exam, StudiesHard x → EnoughSleepBefore x e → PerformWell x e ∧ AceTest x e

-- Facts about Emma and her biology exam
axiom SleepEmma : EnoughSleepBefore Emma BiologyExam
axiom NotAceEmma : ¬ AceTest Emma BiologyExam

theorem not_study_Emma : ¬ StudiesHard Emma := by
  intro hStudy
  have h := Rule Emma BiologyExam hStudy SleepEmma
  have hAce : AceTest Emma BiologyExam := And.right h
  exact NotAceEmma hAce
</lean>
