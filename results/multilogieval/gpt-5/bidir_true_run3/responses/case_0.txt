<lean>
axiom obj : Type
axiom Person : obj

axiom SavedAgg : obj → Prop
axiom ImprovedCredit : obj → Prop
axiom AffordDownPayment : obj → Prop
axiom QualifyMortgage : obj → Prop
axiom BuyDreamHome : obj → Prop

axiom R1 : ∀ x : obj, SavedAgg x → AffordDownPayment x
axiom R2 : ∀ x : obj, ImprovedCredit x → QualifyMortgage x
axiom R3 : SavedAgg Person ∨ ImprovedCredit Person
axiom R4 : ∀ x : obj, QualifyMortgage x → BuyDreamHome x

axiom NotAfford : ¬ AffordDownPayment Person

theorem bought_dream_home : BuyDreamHome Person := by
  have hNotSaved : ¬ SavedAgg Person := by
    intro hs
    have hAff : AffordDownPayment Person := R1 Person hs
    exact NotAfford hAff
  have hImproved : ImprovedCredit Person := by
    cases R3 with
    | inl hs =>
      have : False := hNotSaved hs
      exact False.elim this
    | inr hi =>
      exact hi
  have hQual : QualifyMortgage Person := R2 Person hImproved
  exact R4 Person hQual
</lean>

ANSWER: Yes
