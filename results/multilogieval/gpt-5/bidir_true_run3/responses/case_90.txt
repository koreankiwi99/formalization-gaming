<lean>
axiom W : Prop          -- home team is winning
axiom R : Prop          -- it's a rivalry game
axiom L : Prop          -- crowd is loud and energized
axiom E : Prop          -- atmosphere is electric
axiom U : Prop          -- unforgettable game
axiom M : Prop          -- fans more emotionally invested

axiom A1 : W → L        -- if home team is winning then crowd is loud and energized
axiom A2 : R → M        -- if rivalry game then fans more emotionally invested
axiom A3 : W ∨ R        -- either home team is winning or it's a rivalry game (or both)
axiom A4 : L → E        -- if crowd is loud and energized, atmosphere is electric
axiom A5 : E → U        -- if atmosphere is electric, it will be an unforgettable game

theorem atmosphere_electric_of_not_emotionally_invested : (¬ M) → E := by
  intro notM
  have notR : ¬ R := by
    intro hR
    exact notM (A2 hR)
  have hW : W := by
    cases A3 with
    | inl hw => exact hw
    | inr hr => exact False.elim (notR hr)
  have hL : L := A1 hW
  exact A4 hL
</lean>

ANSWER: Yes
