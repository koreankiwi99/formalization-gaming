<lean>
axiom Flip : Prop
axiom Gasps : Prop
axiom High : Prop
axiom Clowns : Prop
axiom KidsLaugh : Prop

axiom R1 : Flip → Gasps
axiom R2 : Gasps → High
axiom R3 : Flip ∨ Clowns
axiom R4 : Clowns → KidsLaugh

axiom NotHigh : ¬ High

theorem youngest_kids_laughed : KidsLaugh :=
by
  have ng : ¬ Gasps := fun g => NotHigh (R2 g)
  have nf : ¬ Flip := fun f => ng (R1 f)
  have c : Clowns := Or.elim R3 (fun f => False.elim (nf f)) (fun c => c)
  exact R4 c
</lean>

ANSWER: Yes
