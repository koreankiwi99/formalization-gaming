<lean>
axiom Rain : Prop
axiom Watered : Prop
axiom Hot : Prop
axiom Sprinklers : Prop

axiom R1 : Rain → Watered
axiom R2 : Hot → Sprinklers
axiom R3 : Rain ∨ ¬ Sprinklers
axiom T1 : Hot

theorem watered_today : Watered := by
  have s : Sprinklers := R2 T1
  exact Or.elim R3 (fun hrain => R1 hrain) (fun hnot => False.elim (hnot s))
</lean>

ANSWER: Yes
