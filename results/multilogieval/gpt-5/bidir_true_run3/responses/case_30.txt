axiom obj : Type
axiom Mark : obj

axiom Athlete : obj → Prop
axiom Dedicated : obj → Prop
axiom ProperTraining : obj → Prop
axiom HealthyDiet : obj → Prop
axiom Skilled : obj → Prop
axiom Strong : obj → Prop

-- From context:
-- Proper training enhances skills; healthy nutrition increases strength.
axiom R_train_skill : ∀ x : obj, ProperTraining x → Skilled x
axiom R_diet_strength : ∀ x : obj, HealthyDiet x → Strong x
-- Athletes lack skill or lack strength.
axiom R_lack_disj : ∀ x : obj, Athlete x → (¬ Skilled x ∨ ¬ Strong x)
-- Being dedicated leads an athlete to train properly.
axiom R_ded_train : ∀ x : obj, Athlete x → Dedicated x → ProperTraining x

-- Facts about Mark
axiom Mark_is_athlete : Athlete Mark
axiom Mark_healthy_diet : HealthyDiet Mark

-- We can derive that Mark is not dedicated (hence cannot prove he is dedicated).
theorem not_dedicated_Mark : ¬ Dedicated Mark := by
  have hAth : Athlete Mark := Mark_is_athlete
  have hStrong : Strong Mark := R_diet_strength Mark Mark_healthy_diet
  have hDisj : ¬ Skilled Mark ∨ ¬ Strong Mark := R_lack_disj Mark hAth
  have hNotSkilled : ¬ Skilled Mark := by
    cases hDisj with
    | inl hns => exact hns
    | inr hnotStrong =>
        have contra : False := hnotStrong hStrong
        exact False.elim contra
  have hNotTrain : ¬ ProperTraining Mark := by
    intro hpt
    have hSk : Skilled Mark := R_train_skill Mark hpt
    exact hNotSkilled hSk
  intro hded
  have hpt : ProperTraining Mark := R_ded_train Mark hAth hded
  exact hNotTrain hpt
