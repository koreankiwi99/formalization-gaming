<lean>
axiom Party : Type
axiom SamParty : Party
axiom Birthday : Party → Prop
axiom Cake : Party → Prop
axiom IceCream : Party → Prop
axiom BringDish : Party → Prop
axiom EnoughFood : Party → Prop

axiom R1 : ∀ p : Party, BringDish p → EnoughFood p
axiom R2 : ∀ p : Party, Birthday p → Cake p ∧ IceCream p
axiom R3 : ∀ p : Party, BringDish p ∨ ¬Cake p

axiom T1 : Birthday SamParty

theorem sam_enough : EnoughFood SamParty := by
  have cakeSam : Cake SamParty := by
    have h := R2 SamParty T1
    exact And.left h
  have hdisj := R3 SamParty
  cases hdisj with
  | inl bd =>
      exact R1 SamParty bd
  | inr nCake =>
      have contra : False := nCake cakeSam
      exact False.elim contra
</lean>

ANSWER: Yes
