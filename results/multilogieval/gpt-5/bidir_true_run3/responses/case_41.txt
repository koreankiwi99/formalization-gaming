<lean>
axiom Daily : Prop
axiom Drought : Prop
axiom Restrictions : Prop
axiom GrowWell : Prop
axiom NotEnoughWater : Prop
axiom WiltDie : Prop

axiom R1 : Daily → GrowWell
axiom R2 : Drought → Restrictions
axiom R3 : Daily ∨ Drought
axiom R4 : Restrictions → NotEnoughWater
axiom R5 : NotEnoughWater → WiltDie

theorem wilt_from_not_grow : ¬ GrowWell → WiltDie := by
  intro hNotGrow
  cases R3 with
  | inl hDaily =>
      have hGrow : GrowWell := R1 hDaily
      have hFalse : False := hNotGrow hGrow
      exact False.elim hFalse
  | inr hDrought =>
      have hRes : Restrictions := R2 hDrought
      have hNotWater : NotEnoughWater := R4 hRes
      exact R5 hNotWater
</lean>

ANSWER: Yes
