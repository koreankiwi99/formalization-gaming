<lean>
axiom obj : Type
axiom John : obj

axiom Early : obj → Prop
axiom LongLunch : obj → Prop
axiom Recharged : obj → Prop
axiom Productive : obj → Prop
axiom BringsTreats : obj → Prop
axiom Birthday : Prop

-- If someone finishes early, they can take a long lunch
axiom R1 : ∀ x : obj, Early x → LongLunch x
-- If it's a birthday, people will bring in treats
axiom R2 : Birthday → ∀ x : obj, BringsTreats x
-- Either John's project gets done early, or it's a birthday (or both)
axiom R3 : Early John ∨ Birthday
-- If someone takes a long lunch, they'll come back recharged
axiom R4 : ∀ x : obj, LongLunch x → Recharged x
-- If someone comes back recharged, they'll have a very productive afternoon
axiom R5 : ∀ x : obj, Recharged x → Productive x

theorem john_productive_of_not_bring_treats : (¬ BringsTreats John) → Productive John :=
by
  intro hnb
  cases R3 with
  | inl he =>
      have hl : LongLunch John := R1 John he
      have hr : Recharged John := R4 John hl
      exact R5 John hr
  | inr hb =>
      have hbt : BringsTreats John := (R2 hb) John
      have : False := hnb hbt
      exact this.elim
</lean>

ANSWER: Yes
