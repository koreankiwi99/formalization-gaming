<lean>
axiom Rain : Prop
axiom PoolOpen : Prop
axiom KidsSwim : Prop
axiom Game : Prop
axiom Enjoy : Prop
axiom Smiles : Prop

axiom R1 : ¬ Rain → Game
axiom R2 : PoolOpen → KidsSwim
axiom T1 : ¬ Rain ∨ PoolOpen
axiom R3 : Game → Enjoy
axiom R4 : Enjoy → Smiles

theorem smiles_if_no_swim : (¬ KidsSwim) → Smiles := by
  intro hNoSwim
  have h := T1
  cases h with
  | inl hNoRain =>
      have hGame : Game := R1 hNoRain
      have hEnjoy : Enjoy := R3 hGame
      exact R4 hEnjoy
  | inr hPoolOpen =>
      have hSwim : KidsSwim := R2 hPoolOpen
      have hFalse : False := hNoSwim hSwim
      exact False.elim hFalse
</lean>

ANSWER: Yes
