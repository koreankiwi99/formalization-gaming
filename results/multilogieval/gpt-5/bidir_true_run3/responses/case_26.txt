<lean>
axiom LossImportant : Prop
axiom LossBig : Prop
axiom Disappointed : Prop
axiom Demoralized : Prop

-- Context rules
axiom R1 : LossImportant → Disappointed
axiom R2 : LossBig → Demoralized

-- Given statement: players are neither disappointed nor demoralized
axiom H1 : ¬ Disappointed
axiom H2 : ¬ Demoralized

-- Conclusion: the team did not lose a significant (important or big) game
theorem no_significant_loss : ¬ (LossImportant ∨ LossBig) := by
  intro h
  apply Or.elim h
  · intro hImp
    exact H1 (R1 hImp)
  · intro hBig
    exact H2 (R2 hBig)
</lean>

ANSWER: Yes
