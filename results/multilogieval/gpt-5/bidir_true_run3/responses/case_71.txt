<lean>
axiom obj : Type
axiom Alex : obj

axiom Sunny : Prop
axiom Warm : Prop
axiom Swim : obj → Prop
axiom Exercise : obj → Prop
axiom Sunscreen : obj → Prop
axiom Burnt : obj → Prop

-- It was a sunny day at the beach.
axiom T_sunny : Sunny

-- If someone applies sunscreen, they do not get burnt.
axiom R_sunscreen_no_burn : ∀ x : obj, Sunscreen x → ¬ Burnt x

-- If someone goes swimming, then they get exercise.
axiom R_swim_exercise : ∀ x : obj, Swim x → Exercise x

-- If the pool water is warm, people go swimming.
axiom R_warm_swim : Warm → ∀ x : obj, Swim x

-- If the weather was sunny, the pool water will be warm.
axiom R_sunny_warm : Sunny → Warm

-- Either they applied sunscreen, or they did not get any exercise.
axiom R_sunscreen_or_no_exercise : ∀ x : obj, Sunscreen x ∨ ¬ Exercise x

-- From the context, weather is sunny regardless; thus if Alex got sunburnt, then the weather was sunny.
theorem alex_burnt_implies_sunny : Burnt Alex → Sunny :=
  fun _ => T_sunny
</lean>

ANSWER: Yes
