<lean>
axiom A_scores_more : Prop
axiom B_fumbles : Prop
axiom A_wins : Prop
axiom B_loses : Prop

axiom R1 : A_scores_more → A_wins
axiom R2 : B_fumbles → B_loses
axiom T  : A_scores_more ∨ B_fumbles

theorem conclusion : (¬ B_loses) → A_scores_more := by
  intro hNotLose
  cases T with
  | inl hA =>
      exact hA
  | inr hBf =>
      have hLose : B_loses := R2 hBf
      have : False := hNotLose hLose
      exact False.elim this
</lean>

ANSWER: Yes
