<lean>
axiom Person : Type
axiom Exam : Type

axiom Alex : Person
axiom Biology : Exam
axiom History : Exam

axiom StudiedHard : Person → Exam → Prop
axiom Confident : Person → Exam → Prop
axiom DoWell : Person → Exam → Prop
axiom Satisfied : Person → Exam → Prop

axiom SevereWeather : Prop
axiom CancelledDueToWeather : Prop

-- If someone studies hard for an exam, they will feel confident going into the test.
axiom R1 : ∀ x e, StudiedHard x e → Confident x e
-- If there is severe weather in the forecast, schools cancel class that day (formalized deterministically).
axiom R2 : SevereWeather → CancelledDueToWeather
-- Either Alex studied diligently for his biology exam, or class was not canceled due to the weather.
axiom R3 : StudiedHard Alex Biology ∨ ¬ CancelledDueToWeather
-- If a student feels confident about the material, they will likely do well on the exam.
axiom R4 : ∀ x e, Confident x e → DoWell x e
-- If a student performs well on an exam, they will be satisfied with their hard work.
axiom R5 : ∀ x e, DoWell x e → Satisfied x e

-- From severe weather we can derive Alex is satisfied with his Biology exam (but not necessarily History).
theorem satisfied_bio_if_weather : SevereWeather → Satisfied Alex Biology :=
by
  intro sw
  have c : CancelledDueToWeather := R2 sw
  have sh : StudiedHard Alex Biology :=
    Or.elim R3
      (fun h => h)
      (fun notC => False.elim (notC c))
  have conf : Confident Alex Biology := R1 Alex Biology sh
  have dw : DoWell Alex Biology := R4 Alex Biology conf
  exact R5 Alex Biology dw

-- since all derivations pertain (at best) to the Biology exam, with no link to History.
