<lean>
axiom Person : Type
axiom James : Person
axiom StartNew : Person → Prop
axiom Motivated : Person → Prop
axiom GymClosed : Prop
axiom CantFollow : Person → Prop
axiom Frustrated : Person → Prop
axiom Indulged : Person → Prop
axiom MightIndulge : Person → Prop

-- If someone starts a new workout routine, they will be motivated to keep it up.
axiom A1 : ∀ x : Person, StartNew x → Motivated x
-- If the gym closes for renovations, then they can't follow their routine.
axiom A2 : ∀ x : Person, GymClosed → CantFollow x
-- Either they (James) started a new workout, or the gym was closed (or both).
axiom A3 : StartNew James ∨ GymClosed
-- If they couldn't follow their routine, they would feel frustrated.
axiom A4 : ∀ x : Person, CantFollow x → Frustrated x
-- If they felt frustrated, they might indulge in some unhealthy snacks.
axiom A5 : ∀ x : Person, Frustrated x → MightIndulge x

-- Given: James was not motivated to workout.
axiom A6 : ¬ Motivated James

-- We can derive that James might indulge (but not that he did).
theorem james_might_indulge : MightIndulge James := by
  exact Or.elim A3
    (fun hstart => False.elim (A6 (A1 James hstart)))
    (fun hclosed =>
      let hcant : CantFollow James := (A2 James) hclosed
      let hfr : Frustrated James := A4 James hcant
      A5 James hfr)

-- Cannot prove Indulged James from the given axioms (only "might indulge" is given).
