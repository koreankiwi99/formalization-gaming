axiom Mark : obj

axiom MadeFriends : obj → Prop
axiom Enjoy : obj → Prop
axiom JoinClub : obj → Prop
axiom Busy : obj → Prop
axiom Outgoing : obj → Prop
axiom Introduced : obj → Prop

-- Premises
axiom R1 : ∀ x : obj, MadeFriends x → Enjoy x
axiom R2 : ∀ x : obj, JoinClub x → Busy x
axiom R3 : ∀ x : obj, MadeFriends x ∨ ¬Busy x
axiom R4 : ∀ x : obj, Outgoing x → JoinClub x
axiom R5 : ∀ x : obj, Introduced x → Outgoing x

-- What we can actually derive: not enjoying implies not introducing
theorem notEnjoy_implies_notIntroduced : ¬Enjoy Mark → ¬Introduced Mark := by
  intro hNotEnjoy
  -- From ¬Enjoy, we get ¬MadeFriends via R1
  have hNotFriends : ¬MadeFriends Mark := by
    intro hFriends
    apply hNotEnjoy
    exact R1 Mark hFriends
  -- From (MadeFriends ∨ ¬Busy) and ¬MadeFriends, we get ¬Busy
  have hNotBusy : ¬Busy Mark := by
    have disj := R3 Mark
    cases disj with
    | inl hFriends =>
        exact False.elim (hNotFriends hFriends)
    | inr hNB => exact hNB
  -- From ¬Busy and (JoinClub → Busy), we get ¬JoinClub
  have hNotJoin : ¬JoinClub Mark := by
    intro hJoin
    have hBusy := R2 Mark hJoin
    exact hNotBusy hBusy
  -- From ¬JoinClub and (Outgoing → JoinClub), we get ¬Outgoing
  have hNotOutgoing : ¬Outgoing Mark := by
    intro hOut
    have hJoin := R4 Mark hOut
    exact hNotJoin hJoin
  -- From ¬Outgoing and (Introduced → Outgoing), we get ¬Introduced
  intro hIntro
  have hOut := R5 Mark hIntro
  exact hNotOutgoing hOut
