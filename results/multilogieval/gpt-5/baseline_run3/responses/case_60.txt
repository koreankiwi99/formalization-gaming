<lean>
axiom Alarm : Prop
axiom Ready : Prop
axiom MissBus : Prop
axiom Late : Prop
axiom UpLate : Prop
axiom BedLate : Prop

-- If the alarm went off, then it was time to get ready.
axiom A1 : Alarm → Ready
-- Either the alarm went off, or they weren't late for school.
axiom A2 : Alarm ∨ ¬Late
-- If they missed the bus, then they'd be late for school.
axiom A3 : MissBus → Late
-- If they got up late, then they miss the bus.
axiom A4 : UpLate → MissBus
-- If they went to bed late, then they get up late.
axiom A5 : BedLate → UpLate
-- John knows it is not the time to get ready (we take ¬Ready as a fact).
axiom A6 : ¬Ready

theorem john_not_bedlate : ¬BedLate := by
  have hNotAlarm : ¬Alarm := by
    intro hA
    have hR : Ready := A1 hA
    exact A6 hR
  have hNotLate : ¬Late := by
    apply Or.elim A2
    · intro hA
      exact False.elim (hNotAlarm hA)
    · intro hNotL
      exact hNotL
  intro hB
  have hU : UpLate := A5 hB
  have hM : MissBus := A4 hU
  have hL : Late := A3 hM
  exact hNotLate hL
</lean>

ANSWER: No
