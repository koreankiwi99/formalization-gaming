<lean>
axiom Person : Type
axiom James : Person

axiom Practice : Person → Prop
axiom Prepared : Person → Prop
axiom Confident : Person → Prop
axiom JoinedClub : Person → Prop

-- If someone practices thoroughly, they feel more prepared.
axiom R1 : ∀ x : Person, Practice x → Prepared x
-- If someone feels prepared, they can speak confidently.
axiom R2 : ∀ x : Person, Prepared x → Confident x
-- Either they practiced thoroughly or they joined a public speaking club (or both).
axiom R3 : ∀ x : Person, Practice x ∨ JoinedClub x

theorem james_joined_of_not_confident : ¬Confident James → JoinedClub James := by
  intro hnc
  have hnp : ¬Prepared James := by
    intro hp
    exact hnc (R2 James hp)
  have hnpra : ¬Practice James := by
    intro hpr
    exact hnp (R1 James hpr)
  have hdisj : Practice James ∨ JoinedClub James := R3 James
  exact Or.elim hdisj (fun hpra => False.elim (hnpra hpra)) (fun hj => hj)
</lean>

ANSWER: Yes
