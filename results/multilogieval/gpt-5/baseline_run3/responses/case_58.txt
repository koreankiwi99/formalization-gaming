axiom Salad : Prop

-- Rules and facts from the context

axiom R2 : Stir → Thick
axiom Law : AddSalt ∨ ¬Thick ∨ (AddSalt ∧ ¬Thick)
axiom R3 : Taste → Stir
axiom Secret : Taste ∨ Salad ∨ (Taste ∧ Salad)

-- Given: the stew is not savory
axiom NotSavory : ¬Savory

theorem salad_prepared : Salad := by
  -- Derive ¬AddSalt from ¬Savory and AddSalt → Savory
  have h_not_add : ¬AddSalt := by
    intro hA
    exact NotSavory (R1 hA)

  -- From the law and ¬Savory, derive ¬Thick
  have h_not_thick : ¬Thick := by
    cases Law with
    | inl h1 =>
      cases h1 with
      | inl hA =>
        exact (fun ht => NotSavory (R1 hA))
      | inr hNTh =>
        exact hNTh
    | inr hBoth =>
      exact hBoth.right

  -- From Stir → Thick and ¬Thick, get ¬Stir
  have h_not_stir : ¬Stir := by
    intro hS
    exact h_not_thick (R2 hS)

  -- From Taste → Stir and ¬Stir, get ¬Taste
  have h_not_taste : ¬Taste := by
    intro hT
    exact h_not_stir (R3 hT)

  -- From Secret (Taste ∨ Salad ∨ (Taste ∧ Salad)) and ¬Taste, deduce Salad
  cases Secret with
  | inl h_or =>
    cases h_or with
    | inl hT =>
      have f : False := h_not_taste hT
      exact False.elim f
    | inr hS =>
      exact hS
  | inr hBoth =>
    exact hBoth.right
