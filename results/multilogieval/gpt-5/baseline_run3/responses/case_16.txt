axiom Person : Type
axiom Sarah : Person

axiom PracticedDaily : Person → Prop          -- practiced guitar scales and chords daily
axiom TechniqueImproved : Person → Prop
axiom LearnFaster : Person → Prop
axiom PracticedRegularly : Person → Prop       -- practiced guitar regularly
axiom ListenedAttentively : Person → Prop      -- listened attentively to music

-- If they practiced scales and chords daily, their technique improved.
axiom A1 : ∀ x : Person, PracticedDaily x → TechniqueImproved x
-- If their technique improved, they could learn songs faster.
axiom A2 : ∀ x : Person, TechniqueImproved x → LearnFaster x
-- Either they practiced regularly or they listened attentively (or both).
axiom A3 : PracticedRegularly Sarah ∨ ListenedAttentively Sarah

-- From daily practice, they learn faster.
lemma daily_implies_fast (x : Person) : PracticedDaily x → LearnFaster x :=
  fun hp => A2 x (A1 x hp)

-- If they do not learn faster, then they did not practice daily.
lemma not_fast_implies_not_daily (x : Person) : ¬LearnFaster x → ¬PracticedDaily x :=
  fun hnf hp => hnf (daily_implies_fast x hp)
