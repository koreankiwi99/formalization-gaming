<lean>
axiom obj : Type
axiom James : obj

axiom Follow : obj → Prop            -- follows recipes precisely
axiom Well : obj → Prop              -- dishes turn out well
axiom Confidence : obj → Prop        -- gains confidence in the kitchen
axiom Watch : obj → Prop             -- watches cooking shows for inspiration

axiom Follow_implies_Well : ∀ x : obj, Follow x → Well x
axiom Well_implies_Confidence : ∀ x : obj, Well x → Confidence x
axiom Follow_or_Watch : ∀ x : obj, Follow x ∨ Watch x  -- either follow, or watch, or both

theorem james_watches (h : ¬ Confidence James) : Watch James := by
  have hnotFollow : ¬ Follow James := by
    intro hf
    have hw : Well James := Follow_implies_Well James hf
    have hc : Confidence James := Well_implies_Confidence James hw
    exact h hc
  have disj : Follow James ∨ Watch James := Follow_or_Watch James
  cases disj with
  | inl hf => exact False.elim (hnotFollow hf)
  | inr hw => exact hw
</lean>

ANSWER: Yes
