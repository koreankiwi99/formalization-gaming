-- Useful derived fact: if the market has ripe peaches, the house feels homey and cozy

  have hpieExists : ∃ x : Person, BakesPie x := H1 hRipe
  cases hpieExists with
  | intro x hx =>
    have hs : HouseSmellsAmazing := H4 x hx
    exact H5 hs
