<lean>
axiom obj : Type
axiom John : obj
axiom Person : obj

axiom WorkedHard : obj → Prop
axiom Sore : obj → Prop
axiom HotShower : obj → Prop
axiom Relaxed : obj → Prop

axiom Cold : Prop
axiom Rain : Prop

-- If someone takes a hot shower, their muscles relax.
axiom R_shower_relax : ∀ x : obj, HotShower x → Relaxed x
-- If the weather is cold, everyone takes a hot shower.
axiom R_cold_shower : ∀ x : obj, Cold → HotShower x
-- If it rains, then the weather is cold.
axiom R_rain_cold : Rain → Cold

-- For the particular person who went to the gym:
-- If they worked out hard, they are sore the next day.
axiom A_work_sore : WorkedHard Person → Sore Person
-- Either they worked out hard, or their muscles did not relax.
axiom A_disj : WorkedHard Person ∨ ¬ Relaxed Person

-- A derived fact from the above context (about that specific person):
theorem rain_implies_sore_person : Rain → Sore Person := by
  intro hRain
  have hCold : Cold := R_rain_cold hRain
  have hHot : HotShower Person := R_cold_shower Person hCold
  have hRelax : Relaxed Person := R_shower_relax Person hHot
  have hWork : WorkedHard Person := by
    cases A_disj with
    | inl hw => exact hw
    | inr hnr =>
        have : False := hnr hRelax
        exact False.elim this
  exact A_work_sore hWork
