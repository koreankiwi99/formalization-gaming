<lean>
axiom person : Type
axiom Mary : person

axiom HomeOwner : person → Prop
axiom LearnedBotany : person → Prop
axiom UnderstoodNeeds : person → Prop
axiom Thriving : person → Prop
axiom SoughtAdvice : person → Prop

-- If someone learned botany, they understand plant needs
axiom A1 : ∀ x : person, LearnedBotany x → UnderstoodNeeds x
-- If someone understands plant needs, their plants thrive
axiom A2 : ∀ x : person, UnderstoodNeeds x → Thriving x
-- Every homeowner either learned botany or sought advice (or both)
axiom A3 : ∀ x : person, HomeOwner x → (LearnedBotany x ∨ SoughtAdvice x)

-- Mary is a homeowner
axiom Mary_homeowner : HomeOwner Mary
-- Mary's plants did not thrive
axiom Mary_not_thriving : ¬ Thriving Mary

theorem mary_sought_advice : SoughtAdvice Mary := by
  have hdisj : LearnedBotany Mary ∨ SoughtAdvice Mary := A3 Mary Mary_homeowner
  cases hdisj with
  | inl hL =>
      have u : UnderstoodNeeds Mary := A1 Mary hL
      have t : Thriving Mary := A2 Mary u
      exact False.elim (Mary_not_thriving t)
  | inr hS =>
      exact hS
</lean>

ANSWER: Yes
