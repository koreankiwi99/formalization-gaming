<lean>
axiom Monsoon : Prop
axiom HeavyRains : Prop
axiom WatersDaily : Prop
axiom WatersCrops : Prop
axiom GrowStrong : Prop
axiom Bountiful : Prop
axiom GoodProfit : Prop

axiom R1 : WatersCrops → GrowStrong
axiom R2 : Monsoon → HeavyRains
axiom R3 : WatersDaily ∨ ¬ HeavyRains
axiom R4 : GrowStrong → Bountiful
axiom R5 : Bountiful → GoodProfit
axiom R6 : WatersDaily → WatersCrops

theorem monsoon_profit : Monsoon → GoodProfit := by
  intro m
  have hr : HeavyRains := R2 m
  have wd : WatersDaily := Or.elim R3 (fun wd => wd) (fun nh => False.elim (nh hr))
  have wc : WatersCrops := R6 wd
  have gs : GrowStrong := R1 wc
  have hb : Bountiful := R4 gs
  exact R5 hb
</lean>

ANSWER: Yes
