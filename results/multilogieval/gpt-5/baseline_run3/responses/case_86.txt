<lean>
axiom obj : Type
axiom Priya : obj

axiom StudiesHard : obj → Prop
axiom GoodGrades  : obj → Prop
axiom Proud       : obj → Prop
axiom IceCream    : obj → Prop

axiom Snowstorm      : Prop
axiom BigSnowstorm   : Prop
axiom SchoolCanceled : Prop

-- If someone studies hard for an exam, they will get good grades.
axiom A1 : ∀ x : obj, StudiesHard x → GoodGrades x
-- If there is a snowstorm, then school is canceled.
axiom A2 : Snowstorm → SchoolCanceled
-- Either Priya studied hard, or school was not called off.
axiom A3 : StudiesHard Priya ∨ ¬SchoolCanceled
-- If a student gets good grades, their parents are proud.
axiom A4 : ∀ x : obj, GoodGrades x → Proud x
-- When parents are proud, they take their kids out for ice cream.
axiom A5 : ∀ x : obj, Proud x → IceCream x
-- A big snowstorm is a snowstorm.
axiom A6 : BigSnowstorm → Snowstorm

theorem priya_icecream_if_big_snowstorm : BigSnowstorm → IceCream Priya := by
  intro HB
  have HS : Snowstorm := A6 HB
  have HC : SchoolCanceled := A2 HS
  have not_not_c : ¬(¬SchoolCanceled) := by
    intro h
    exact h HC
  have hStudy : StudiesHard Priya := by
    apply Or.elim A3
    · intro h; exact h
    · intro hnotc
      have f : False := not_not_c hnotc
      exact False.elim f
  have hGrades : GoodGrades Priya := A1 Priya hStudy
  have hProud  : Proud Priya := A4 Priya hGrades
  exact A5 Priya hProud
</lean>

ANSWER: Yes
