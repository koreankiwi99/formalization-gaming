<lean>
axiom obj : Type
axiom James : obj

axiom FridayNight : Prop
axiom GoingToParty : obj → Prop
axiom FinishAllHomework : obj → Prop
axiom WatchGame : obj → Prop
axiom StudyEveryday : obj → Prop
axiom PlanScheduleWell : obj → Prop

-- If it is Friday night, then (any) student is going to the party
axiom R1 : ∀ s : obj, FridayNight → GoingToParty s
-- If they finish all their homework, then they can watch the game
axiom R2 : ∀ s : obj, FinishAllHomework s → WatchGame s
-- Either it is Friday night, or they are not watching the game (for each student)
axiom R3 : ∀ s : obj, FridayNight ∨ ¬ WatchGame s
-- If they study everyday, they finish all their homework
axiom R4 : ∀ s : obj, StudyEveryday s → FinishAllHomework s
-- If someone plans their schedule well, then they can study everyday
axiom R5 : ∀ s : obj, PlanScheduleWell s → StudyEveryday s

-- Given: James is not going to the party
axiom HJamesNotGoing : ¬ GoingToParty James

theorem james_did_not_plan_well : ¬ PlanScheduleWell James := by
  intro hplan
  -- From planning well to studying everyday
  have hstudy : StudyEveryday James := R5 James hplan
  -- From studying everyday to finishing all homework
  have hfinish : FinishAllHomework James := R4 James hstudy
  -- From finishing all homework to watching the game
  have hwatch : WatchGame James := R2 James hfinish
  -- The disjunction: Friday night or not watching the game
  have hdisj : FridayNight ∨ ¬ WatchGame James := R3 James
  -- From "if Friday then going" and "not going", get "not Friday"
  have hnotfri : ¬ FridayNight := by
    intro hfri
    have hgo : GoingToParty James := R1 James hfri
    exact HJamesNotGoing hgo
  -- Using the disjunction and not Friday, deduce not watching the game
  have hnotwatch : ¬ WatchGame James := by
    cases hdisj with
    | inl hfri =>
        intro _
        exact False.elim (hnotfri hfri)
    | inr hnw =>
        exact hnw
  -- Contradiction: watching and not watching
  exact hnotwatch hwatch
</lean>

ANSWER: No
