<lean>
axiom obj : Type
axiom Priya : obj

axiom Student : obj → Prop
axiom AttendsRegularly : obj → Prop
axiom RevisesSufficiently : obj → Prop
axiom PerformsWellOnExams : obj → Prop

-- If a student attends lectures regularly, then if they revise sufficiently, they perform well on exams.
axiom Rule : ∀ x : obj, Student x → (AttendsRegularly x → (RevisesSufficiently x → PerformsWellOnExams x))

-- Facts about Priya
axiom Priya_is_student : Student Priya
axiom Priya_revised : RevisesSufficiently Priya
axiom Priya_not_well : ¬ PerformsWellOnExams Priya

theorem priya_did_not_attend : ¬ AttendsRegularly Priya := by
  intro hAttend
  have hImp : RevisesSufficiently Priya → PerformsWellOnExams Priya := Rule Priya Priya_is_student hAttend
  have hWell : PerformsWellOnExams Priya := hImp Priya_revised
  exact Priya_not_well hWell
</lean>

ANSWER: No
