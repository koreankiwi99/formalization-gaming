<lean>
axiom obj : Type
axiom Emma : obj

axiom SpeaksDaily : obj → Prop
axiom ListensPods : obj → Prop
axiom FluencyImproves : obj → Prop
axiom VocabularyImproves : obj → Prop
axiom RealConversations : obj → Prop
axiom ListeningComp : obj → Prop

-- If someone speaks daily with native speakers, their fluency and vocabulary improve
axiom R1 : ∀ x : obj, SpeaksDaily x → FluencyImproves x
axiom R2 : ∀ x : obj, SpeaksDaily x → VocabularyImproves x

-- If fluency improves, then they can have real conversations
axiom R3 : ∀ x : obj, FluencyImproves x → RealConversations x

-- Either someone speaks daily with natives or they listen to educational podcasts (or both)
axiom R4 : ∀ x : obj, SpeaksDaily x ∨ ListensPods x

-- If someone listens to educational podcasts, they gain valuable listening comprehension
axiom R5 : ∀ x : obj, ListensPods x → ListeningComp x

theorem emma_listening_comp :
  ¬ RealConversations Emma → ListeningComp Emma := by
  intro hNotRC
  have h := R4 Emma
  cases h with
  | inl hSpeak =>
      have hFlu := R1 Emma hSpeak
      have hRC := R3 Emma hFlu
      exact False.elim (hNotRC hRC)
  | inr hListen =>
      exact R5 Emma hListen
</lean>

ANSWER: Yes
