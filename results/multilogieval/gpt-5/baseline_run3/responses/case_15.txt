<lean>
axiom obj : Type
axiom Student : obj

axiom Study : obj → Prop
axiom Participate : obj → Prop
axiom Excel : obj → Prop
axiom Leader : obj → Prop
axiom Prefect : obj → Prop

axiom R1 : ∀ x : obj, Study x → Excel x
axiom R2 : ∀ x : obj, Participate x → Leader x
axiom R3 : ∀ x : obj, Leader x → Prefect x

axiom E1 : Study Student ∨ Participate Student
axiom NotExcel : ¬ Excel Student

theorem student_prefect : Prefect Student := by
  have hNotStudy : ¬ Study Student := fun hS => NotExcel ((R1 Student) hS)
  have hParticipate : Participate Student :=
    Or.elim E1 (fun hS => False.elim (hNotStudy hS)) (fun hP => hP)
  have hLeader : Leader Student := (R2 Student) hParticipate
  exact (R3 Student) hLeader
</lean>

ANSWER: Yes
