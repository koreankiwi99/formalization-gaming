<lean>
axiom GoodReviews : Prop
axiom ManyPeople : Prop
axiom BigStar : Prop
axiom Sequel : Prop

-- If the movie gets good reviews, then many people will go see it
axiom R1 : GoodReviews → ManyPeople

-- If a big star is in the movie, then it likely got good reviews
axiom R4 : BigStar → GoodReviews

-- Either there is a big star in it, or it is a sequel (or both)
axiom R5 : BigStar ∨ Sequel

-- Extra context (not needed for the proof)
axiom person : Type
axiom BuysAdvanceTickets : person → Prop
axiom PlansToSee : person → Prop
axiom R2 : ∀ x : person, BuysAdvanceTickets x → PlansToSee x
axiom R3 : GoodReviews ∨ ¬ManyPeople ∨ (GoodReviews ∧ ¬ManyPeople)

-- Question: If attendance at the movie was low, then is it a sequel?
theorem low_attendance_implies_sequel : ¬ManyPeople → Sequel := by
  intro hlow
  have hx := R5
  cases hx with
  | inl bs =>
    have gr : GoodReviews := R4 bs
    have mp : ManyPeople := R1 gr
    exact False.elim (hlow mp)
  | inr sq =>
    exact sq
</lean>

ANSWER: Yes
