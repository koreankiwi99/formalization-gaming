<lean>
axiom obj : Type
axiom Alex : obj

axiom StudiedHard : obj → Prop
axiom GoodGrades : obj → Prop
axiom Extracurricular : obj → Prop
axiom NewFriends : obj → Prop
axiom Invited : obj → Prop

-- If someone studies hard, they get good grades.
axiom A1 : ∀ x : obj, StudiedHard x → GoodGrades x
-- If someone participates in extracurriculars, they make new friends.
axiom A2 : ∀ x : obj, Extracurricular x → NewFriends x
-- The student (Alex) either studied hard or participated in extracurricular activities.
axiom A3 : StudiedHard Alex ∨ Extracurricular Alex
-- If someone makes new friends, they get invited to social events.
axiom A4 : ∀ x : obj, NewFriends x → Invited x
-- Given: Alex did not get good grades.
axiom A5 : ¬ GoodGrades Alex

theorem alex_invited : Invited Alex := by
  have h_not_studied : ¬ StudiedHard Alex := by
    intro hstud
    have hg : GoodGrades Alex := A1 Alex hstud
    exact A5 hg
  have h_extr : Extracurricular Alex := by
    have : Extracurricular Alex :=
      Or.elim A3
        (fun hstud => False.elim (h_not_studied hstud))
        (fun hextr => hextr)
    exact this
  have h_nf : NewFriends Alex := A2 Alex h_extr
  exact A4 Alex h_nf
</lean>

ANSWER: Yes
