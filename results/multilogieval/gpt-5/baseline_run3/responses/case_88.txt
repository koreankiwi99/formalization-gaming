<lean>
axiom Oven350 : Prop
axiom ProperBake : Prop
axiom Eggs : Prop
axiom Fluffy : Prop
axiom Delicious : Prop
axiom GuestsHappy : Prop

axiom A1 : Oven350 → ProperBake
axiom A2 : Eggs → Fluffy
axiom A3 : Oven350 ∨ ¬Eggs
axiom A4 : ProperBake → Delicious
axiom A5 : Delicious → GuestsHappy

theorem guests_happy_if_eggs : Eggs → GuestsHappy := by
  intro hEggs
  have nnEggs : ¬¬Eggs := by
    intro nEggs
    exact nEggs hEggs
  have hOven : Oven350 := by
    cases A3 with
    | inl h => exact h
    | inr nE => exact False.elim (nnEggs nE)
  have hProper : ProperBake := A1 hOven
  have hDelicious : Delicious := A4 hProper
  exact A5 hDelicious
</lean>

ANSWER: Yes
