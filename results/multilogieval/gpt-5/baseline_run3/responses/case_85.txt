<lean>
axiom Person : Type
axiom John : Person

axiom Waters : Person → Prop
axiom BloomsBrightly : Person → Prop
axiom AttractsPollinators : Person → Prop
axiom UsesOrganicFertilizers : Person → Prop
axiom HighQualitySoil : Person → Prop
axiom PlantsGrowHealthier : Person → Prop

-- If someone waters adequately, the plants bloom brightly
axiom R1 : ∀ x : Person, Waters x → BloomsBrightly x
-- If the plants bloom brightly, it attracts more bees and butterflies
axiom R2 : ∀ x : Person, BloomsBrightly x → AttractsPollinators x
-- Either someone waters adequately, or uses organic fertilizers, or both
axiom R3 : ∀ x : Person, Waters x ∨ UsesOrganicFertilizers x ∨ (Waters x ∧ UsesOrganicFertilizers x)
-- If someone uses organic fertilizers, the soil quality is high (improves)
axiom R4 : ∀ x : Person, UsesOrganicFertilizers x → HighQualitySoil x
-- If the soil quality is high (improves), the plants grow healthier
axiom R5 : ∀ x : Person, HighQualitySoil x → PlantsGrowHealthier x

theorem john_soil_high_quality : ¬ AttractsPollinators John → HighQualitySoil John := by
  intro hNotA
  have hWtoA : Waters John → AttractsPollinators John := by
    intro hW
    have hB : BloomsBrightly John := R1 John hW
    exact R2 John hB
  have hNotW : ¬ Waters John := by
    intro hW
    apply hNotA
    exact hWtoA hW
  have hOrg : UsesOrganicFertilizers John := by
    have disj := R3 John
    cases disj with
    | inl hW =>
      exact False.elim (hNotW hW)
    | inr rest =>
      cases rest with
      | inl hO => exact hO
      | inr hWandO =>
        exact False.elim (hNotW hWandO.left)
  exact R4 John hOrg
</lean>

ANSWER: Yes
