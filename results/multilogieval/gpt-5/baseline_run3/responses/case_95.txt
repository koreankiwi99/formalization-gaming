<lean>
axiom Person : Type
axiom Mary : Person

axiom ClearsUp : Prop
axiom GameOn : Prop
axiom FansCheer : Prop
axiom ExcitingNight : Prop

axiom Finished : Person → Prop
axiom Meet : Person → Prop

axiom R1 : ClearsUp → GameOn
axiom R2 : ∀ x : Person, Finished x → Meet x
axiom R3 : ClearsUp ∨ Finished Mary
axiom R4 : GameOn → FansCheer
axiom R5 : FansCheer → ExcitingNight

theorem exciting_if_no_dinner : (¬ Meet Mary) → ExcitingNight := by
  intro hNotMeet
  have hNotFin : ¬ Finished Mary := by
    intro hFin
    have hMeet : Meet Mary := R2 Mary hFin
    exact hNotMeet hMeet
  have hClears : ClearsUp := by
    cases R3 with
    | inl hcu => exact hcu
    | inr hfin => exact False.elim (hNotFin hfin)
  have hGame : GameOn := R1 hClears
  have hCheer : FansCheer := R4 hGame
  exact R5 hCheer
</lean>

ANSWER: Yes
