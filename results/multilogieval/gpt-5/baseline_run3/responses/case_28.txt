<lean>
axiom obj : Type
axiom Jane : obj
axiom Hurt : obj → Prop
axiom SeeDoctor : obj → Prop
axiom Rest : obj → Prop

-- If someone gets injured, they should see a doctor.
axiom R1 : ∀ x : obj, Hurt x → SeeDoctor x
-- Getting hurt warrants resting the affected area.
axiom R2 : ∀ x : obj, Hurt x → Rest x

-- Either Jane did not see a doctor or she did not rest her injury.
axiom E : ¬ SeeDoctor Jane ∨ ¬ Rest Jane

theorem jane_not_hurt : ¬ Hurt Jane := by
  intro hj
  cases E with
  | inl hnsd =>
      have sd : SeeDoctor Jane := R1 Jane hj
      exact hnsd sd
  | inr hnr =>
      have rj : Rest Jane := R2 Jane hj
      exact hnr rj
</lean>

ANSWER: No
