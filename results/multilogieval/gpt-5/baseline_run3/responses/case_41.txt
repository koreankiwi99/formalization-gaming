<lean>
axiom WaterDaily : Prop
axiom Drought : Prop
axiom Restrictions : Prop
axiom GrowWell : Prop
axiom NotEnoughWater : Prop
axiom WiltDie : Prop

axiom R1 : WaterDaily → GrowWell
axiom R2 : Drought → Restrictions
axiom R3 : WaterDaily ∨ Drought
axiom R4 : Restrictions → NotEnoughWater
axiom R5 : NotEnoughWater → WiltDie

theorem priya_plants_wilt_die : ¬GrowWell → WiltDie := by
  intro hNotGW
  have hNotWD : ¬WaterDaily := by
    intro hWD
    have hg : GrowWell := R1 hWD
    exact hNotGW hg
  have hD : Drought := by
    have h := R3
    exact Or.elim h (fun hWD => False.elim (hNotWD hWD)) (fun hD => hD)
  have hRes : Restrictions := R2 hD
  have hNotEnough : NotEnoughWater := R4 hRes
  exact R5 hNotEnough
</lean>

ANSWER: Yes
