<lean>
axiom Party : Prop
axiom Drums : Prop
axiom Noise : Prop
axiom Complain : Prop
axiom Police : Prop

axiom Party_to_Noise : Party → Noise
axiom Drums_to_Noise : Drums → Noise
axiom Party_or_Drums : Party ∨ Drums
axiom Party_to_Complain : Party → Complain
axiom Complain_to_Police : Complain → Police

theorem no_noise_then_police : ¬Noise → Police := by
  intro hNo
  have hN : Noise := by
    cases Party_or_Drums with
    | inl hP => exact Party_to_Noise hP
    | inr hD => exact Drums_to_Noise hD
  exact False.elim (hNo hN)
</lean>

ANSWER: Yes
