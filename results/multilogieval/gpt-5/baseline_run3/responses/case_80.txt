<lean>
axiom Rain : Prop
axiom PoolOpen : Prop
axiom Swim : Prop
axiom Game : Prop
axiom Enjoy : Prop
axiom Smiles : Prop

axiom A1 : (¬Rain) → Game
axiom A2 : PoolOpen → Swim
axiom A3 : (¬Rain) ∨ PoolOpen
axiom A4 : Game → Enjoy
axiom A5 : Enjoy → Smiles

theorem no_swim_smiles : (¬Swim) → Smiles :=
  fun ns =>
    -- From ¬Swim and (¬Rain ∨ PoolOpen) with PoolOpen → Swim, deduce ¬Rain
    have nr : ¬Rain :=
      Or.elim A3
        (fun hnr => hnr)
        (fun hpo => False.elim (ns (A2 hpo)))
    -- Chain implications to get Smiles
    A5 (A4 (A1 nr))
</lean>

ANSWER: Yes
