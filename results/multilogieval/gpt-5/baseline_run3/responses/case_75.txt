<lean>
axiom Car : Type
axiom Liz : Car

axiom PreventiveMaintenance : Car → Prop
axiom QualityPartsUsed : Car → Prop
axiom LongServiceLife : Car → Prop

-- If preventive maintenance is done, then if quality parts are used, the car will have a long service life
axiom R : ∀ c : Car, PreventiveMaintenance c → (QualityPartsUsed c → LongServiceLife c)

-- Facts about Liz's car
axiom Liz_quality : QualityPartsUsed Liz
axiom Liz_not_long : ¬ LongServiceLife Liz

-- Conclusion: Preventive maintenance was not done on Liz's car
theorem no_preventive_Liz : ¬ PreventiveMaintenance Liz :=
  fun hpm => Liz_not_long (R Liz hpm Liz_quality)
</lean>

ANSWER: No
