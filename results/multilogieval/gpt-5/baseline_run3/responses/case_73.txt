<lean>
axiom obj : Type
axiom Simone : obj

axiom Train : obj → Prop       -- trains with dedication
axiom Rest : obj → Prop        -- gets adequate rest
axiom Peak : obj → Prop        -- performs at their peak

-- Coaches' rule: if someone trains with dedication, then if they get adequate rest, they will perform at their peak.
axiom CoachesRule : ∀ x : obj, Train x → Rest x → Peak x

-- Facts about Simone
axiom Rest_Simone : Rest Simone
axiom NotPeak_Simone : ¬Peak Simone

theorem simone_not_trained : ¬Train Simone := by
  intro hTrain
  have hRestImp : Rest Simone → Peak Simone := CoachesRule Simone hTrain
  have hPeak : Peak Simone := hRestImp Rest_Simone
  exact NotPeak_Simone hPeak
</lean>

ANSWER: No
