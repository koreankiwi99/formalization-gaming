<lean>
axiom obj : Type
axiom John : obj

axiom FinishesEarly : obj → Prop
axiom LongLunch : obj → Prop
axiom Recharged : obj → Prop
axiom Productive : obj → Prop
axiom BringTreats : obj → Prop
axiom Birthday : Prop

-- If someone finishes early, they can take a long lunch
axiom R1 : ∀ x : obj, FinishesEarly x → LongLunch x
-- If it's a birthday, everyone brings treats
axiom R2 : Birthday → ∀ x : obj, BringTreats x
-- Either it's a birthday, or all projects get done early (or both)
axiom R3 : Birthday ∨ (∀ x : obj, FinishesEarly x)
-- Long lunch implies recharged
axiom R4 : ∀ x : obj, LongLunch x → Recharged x
-- Recharged implies very productive afternoon
axiom R5 : ∀ x : obj, Recharged x → Productive x

-- Given: John did not bring in treats
axiom T_not : ¬ BringTreats John

theorem john_productive : Productive John := by
  -- From John not bringing treats and "birthday → everyone brings treats", conclude not birthday
  have hbday_false : ¬ Birthday := by
    intro hb
    have : BringTreats John := (R2 hb) John
    exact T_not this
  -- From "birthday ∨ all finish early" and not birthday, conclude all finish early
  have hall : ∀ x : obj, FinishesEarly x :=
    Or.elim R3 (fun hb => False.elim (hbday_false hb)) (fun hall => hall)
  -- Specialize to John and chain implications to productivity
  have hfe : FinishesEarly John := hall John
  have hll : LongLunch John := R1 John hfe
  have hr : Recharged John := R4 John hll
  exact R5 John hr
</lean>

ANSWER: Yes
