<lean>
axiom Person : Type
axiom Mary : Person

axiom Practiced : Person → Prop
axiom Improved : Person → Prop
axiom StoodOutHQ : Person → Prop
axiom TookArtClass : Person → Prop
axiom LearnedNewMethods : Person → Prop

-- If someone practices often, their technique improves
axiom R1 : ∀ x : Person, Practiced x → Improved x
-- If someone's technique improves, their artwork stands out as high quality
axiom R2 : ∀ x : Person, Improved x → StoodOutHQ x
-- If someone takes an art class, they learn new methods and mediums
axiom R3 : ∀ x : Person, TookArtClass x → LearnedNewMethods x

-- Either Mary practiced often or she took an art class (or both)
axiom Mary_disj : Practiced Mary ∨ TookArtClass Mary
-- Mary's artwork did not stand out as high quality
axiom Mary_not_standout : ¬ StoodOutHQ Mary

theorem mary_learned : LearnedNewMethods Mary := by
  have not_improved : ¬ Improved Mary := mt (R2 Mary) Mary_not_standout
  have not_practiced : ¬ Practiced Mary := mt (R1 Mary) not_improved
  have took_class : TookArtClass Mary := Or.resolve_left Mary_disj not_practiced
  exact R3 Mary took_class
</lean>

ANSWER: Yes
