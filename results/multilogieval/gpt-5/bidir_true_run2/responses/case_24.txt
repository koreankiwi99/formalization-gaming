<lean>
axiom Person : Type
axiom Mary : Person

axiom BelowFreezing : Prop
axiom SnowOnGround : Prop
axiom EnoughSnow : Prop

axiom BuildsSnowman : Person → Prop
axiom LovesWinter : Person → Prop

-- If temperature dropped below freezing, there would be snow on the ground
axiom R1 : BelowFreezing → SnowOnGround

-- If someone builds a snowman, there is enough snow
axiom R2 : ∀ x : Person, BuildsSnowman x → EnoughSnow

-- If someone loves winter, they build snowmen
axiom R3 : ∀ x : Person, LovesWinter x → BuildsSnowman x

-- They were certain: either it was below freezing or there was not enough snow
axiom R4 : BelowFreezing ∨ ¬EnoughSnow

-- From no snow on the ground, we can derive that Mary does not love winter
theorem no_snowground_implies_not_loves_winter_Mary : ¬SnowOnGround → ¬LovesWinter Mary := by
  intro hNoGround
  -- Not below freezing, otherwise we'd have snow on the ground
  have hNotBF : ¬BelowFreezing := by
    intro hBF
    have hSOG : SnowOnGround := R1 hBF
    exact hNoGround hSOG
  -- From the certainty disjunction and ¬BelowFreezing, conclude ¬EnoughSnow
  have hNotEnough : ¬EnoughSnow := by
    cases R4 with
    | inl hBF =>
        have : False := hNotBF hBF
        exact False.elim this
    | inr hNE =>
        exact hNE
  -- If Mary loved winter, there would be enough snow; contradiction
  intro hLove
  have hBuild : BuildsSnowman Mary := R3 Mary hLove
  have hEnough : EnoughSnow := R2 Mary hBuild
  exact hNotEnough hEnough

-- This does not follow from the premises above.
</lean>
