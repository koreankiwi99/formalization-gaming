<lean>
axiom obj : Type
axiom Alex : obj

axiom Weekend : Prop
axiom Relax : obj → Prop
axiom Finish : obj → Prop
axiom ClearMind : obj → Prop
axiom Organized : obj → Prop
axiom Todo : obj → Prop

-- If it is the weekend coming up, then they relax
axiom Rw : ∀ x : obj, Weekend → Relax x
-- If someone finishes their tasks, then they have a clear mind
axiom R1 : ∀ x : obj, Finish x → ClearMind x
-- Either the weekend was coming, or they did not have a clear mind (for any person)
axiom D  : ∀ x : obj, Weekend ∨ ¬ ClearMind x
-- If someone is organized, then they finish their tasks
axiom R2 : ∀ x : obj, Organized x → Finish x
-- If they made a to-do list, then they are organized
axiom R3 : ∀ x : obj, Todo x → Organized x

-- Assumption from the query: Alex does not relax this weekend
axiom H : ¬ Relax Alex

-- From the premises and H, we can derive that Alex did not make a to-do list
theorem alex_not_todo : ¬ Todo Alex := by
  -- ¬Weekend
  have hNW : ¬ Weekend := by
    intro hW
    have hR : Relax Alex := Rw Alex hW
    exact H hR
  -- ¬ClearMind Alex (from Weekend ∨ ¬ClearMind Alex and ¬Weekend)
  have hNC : ¬ ClearMind Alex := by
    have disj := D Alex
    cases disj with
    | inl hW =>
        have : False := hNW hW
        exact False.elim this
    | inr hnotC => exact hnotC
  -- ¬Finish Alex (contrapose Finish → ClearMind)
  have hNF : ¬ Finish Alex := by
    intro hF
    have hC : ClearMind Alex := R1 Alex hF
    exact hNC hC
  -- ¬Organized Alex (contrapose Organized → Finish)
  have hNO : ¬ Organized Alex := by
    intro hO
    have hF : Finish Alex := R2 Alex hO
    exact hNF hF
  -- ¬Todo Alex (contrapose Todo → Organized)
  intro hT
  have hO : Organized Alex := R3 Alex hT
  exact hNO hO
</lean>
