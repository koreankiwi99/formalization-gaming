<lean>
axiom Person : Type
axiom Mary : Person

axiom Clear : Prop
axiom GameOn : Prop
axiom FansCheer : Prop
axiom Exciting : Prop

axiom Finish : Person → Prop
axiom Meet : Person → Prop

-- If the weather clears up, the baseball game will go on.
axiom A1 : Clear → GameOn
-- If someone finishes their errands, they'll meet friends for dinner.
axiom A2 : ∀ x : Person, Finish x → Meet x
-- Either it will clear up, or Mary's errands will get done (or both).
axiom A3 : Clear ∨ Finish Mary
-- If the game is on, fans will get to cheer for their team.
axiom A4 : GameOn → FansCheer
-- If fans get to cheer, it will be an exciting night at the ballpark.
axiom A5 : FansCheer → Exciting

theorem exciting_if_not_meet : (¬ Meet Mary) → Exciting := by
  intro hNotMeet
  have hNotFinish : ¬ Finish Mary := by
    intro hFinish
    have hMeet : Meet Mary := A2 Mary hFinish
    exact hNotMeet hMeet
  have hClear : Clear :=
    Or.elim A3
      (fun hC => hC)
      (fun hF => False.elim (hNotFinish hF))
  have hGame : GameOn := A1 hClear
  have hCheer : FansCheer := A4 hGame
  exact A5 hCheer
</lean>

ANSWER: Yes
