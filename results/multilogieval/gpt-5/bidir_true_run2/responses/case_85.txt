<lean>
axiom obj : Type
axiom John : obj

axiom Watered : obj → Prop
axiom Bloom : obj → Prop
axiom Attracts : obj → Prop
axiom UsesFert : obj → Prop
axiom SoilHigh : obj → Prop
axiom Healthy : obj → Prop

axiom A1 : ∀ x : obj, Watered x → Bloom x
axiom A2 : ∀ x : obj, Bloom x → Attracts x
axiom A3 : ∀ x : obj, Watered x ∨ UsesFert x
axiom A4 : ∀ x : obj, UsesFert x → SoilHigh x
axiom A5 : ∀ x : obj, SoilHigh x → Healthy x

theorem john_soil_high : (¬ Attracts John) → SoilHigh John := by
  intro hNotAttr
  have hNotWater : ¬ Watered John := fun hW => hNotAttr (A2 John (A1 John hW))
  have hdisj : Watered John ∨ UsesFert John := A3 John
  exact Or.elim hdisj
    (fun hW => False.elim (hNotWater hW))
    (fun hF => A4 John hF)
</lean>

ANSWER: Yes
