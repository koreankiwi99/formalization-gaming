<lean>
axiom Rain : Prop
axiom Hot : Prop
axiom PlaySprinklers : Prop
axiom WateredNaturally : Prop

axiom R1 : Rain → WateredNaturally
axiom R2 : Hot → PlaySprinklers
axiom R3 : Rain ∨ ¬ PlaySprinklers
axiom T_Hot : Hot

theorem watered_today : WateredNaturally := by
  have hplay : PlaySprinklers := R2 T_Hot
  have hrain : Rain := by
    cases R3 with
    | inl hr => exact hr
    | inr hnp => exact False.elim (hnp hplay)
  exact R1 hrain
</lean>

ANSWER: Yes
