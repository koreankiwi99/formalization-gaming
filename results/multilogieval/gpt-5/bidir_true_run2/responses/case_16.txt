<lean>
axiom obj : Type
axiom Sarah : obj

axiom Practice : obj → Prop
axiom ImproveTechnique : obj → Prop
axiom LearnFaster : obj → Prop
axiom ListenAttentively : obj → Prop

axiom R1 : ∀ x : obj, Practice x → ImproveTechnique x
axiom R2 : ∀ x : obj, ImproveTechnique x → LearnFaster x
axiom R3 : ∀ x : obj, Practice x ∨ ListenAttentively x

theorem sarah_listens : ¬ LearnFaster Sarah → ListenAttentively Sarah := by
  intro hnot
  have hdisj : Practice Sarah ∨ ListenAttentively Sarah := R3 Sarah
  apply Or.elim hdisj
  · intro hp
    have himp : ImproveTechnique Sarah := R1 Sarah hp
    have hlf : LearnFaster Sarah := R2 Sarah himp
    exact False.elim (hnot hlf)
  · intro hl
    exact hl
</lean>

ANSWER: Yes
