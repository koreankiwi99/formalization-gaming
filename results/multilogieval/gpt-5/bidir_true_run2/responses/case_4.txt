<lean>
axiom Marketing : Prop
axiom MoreCustomers : Prop
axiom SalesUp : Prop
axiom CostsUp : Prop

-- If they invested in marketing, they gained more customers.
axiom invest_imp_customers : Marketing → MoreCustomers
-- With more customers, sales go up.
axiom customers_imp_sales : MoreCustomers → SalesUp
-- Either they focused on marketing, or production costs increased, or both.
axiom disj : Marketing ∨ CostsUp ∨ (Marketing ∧ CostsUp)

theorem costs_increased_of_no_sales : ¬SalesUp → CostsUp := by
  intro hNotS
  have m_imp_s : Marketing → SalesUp := fun hm => customers_imp_sales (invest_imp_customers hm)
  have not_m : ¬Marketing := fun hm => hNotS (m_imp_s hm)
  cases disj with
  | inl hM =>
      have : False := not_m hM
      exact False.elim this
  | inr h =>
      cases h with
      | inl hC => exact hC
      | inr hMC =>
          exact And.right hMC
</lean>

ANSWER: Yes
