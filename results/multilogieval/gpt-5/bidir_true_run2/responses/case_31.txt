axiom obj : Type
axiom Priya : obj

axiom Suspects : obj → Prop
axiom Talked : obj → Prop
axiom Reassessed : obj → Prop

-- If someone suspects their partner is cheating, they talk openly.
axiom R1 : ∀ x : obj, Suspects x → Talked x
-- If someone suspects their partner is cheating, they reassess the relationship.
axiom R2 : ∀ x : obj, Suspects x → Reassessed x

-- Given statement to evaluate: Either Priya did not talk or she did not reassess.
axiom H : ¬ Talked Priya ∨ ¬ Reassessed Priya

-- From H and the rules, we can derive that Priya did not suspect her partner.
theorem not_suspects_Priya : ¬ Suspects Priya := by
  intro hs
  cases H with
  | inl h_not_talk =>
      exact h_not_talk (R1 Priya hs)
  | inr h_not_reassess =>
      exact h_not_reassess (R2 Priya hs)
