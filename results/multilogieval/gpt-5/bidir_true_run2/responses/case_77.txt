axiom A3 : ∀ x : obj, Start x → Talented x

axiom A5 : ∀ x : obj, Natural x → Start x
-- If someone was a top player in youth leagues, then they have natural talent.
axiom A6 : ∀ x : obj, TopYouth x → Natural x

-- From the premises we can derive the opposite: if James did not play well, then he was not a top youth player.
theorem james_not_playwell_implies_not_top : (¬ PlayWell James) → ¬ TopYouth James := by

  intro hTop
  have hNat : Natural James := A6 James hTop
  have hStart : Start James := A5 James hNat
  have hTal : Talented James := A3 James hStart
  have hNotPH : ¬ PracticeHard James := by
    intro hPH
    have hPW : PlayWell James := A1 James hPH
    exact hNotPlay hPW
  have hNotTal : ¬ Talented James :=
    Or.elim (A4 James)
      (fun hPH => False.elim (hNotPH hPH))
      (fun hNTal => hNTal)
