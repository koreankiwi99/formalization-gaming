<lean>
axiom obj : Type
axiom Chef : obj

axiom Fresh : obj → Prop
axiom Delicious : obj → Prop
axiom Complicated : obj → Prop
axiom MoreTime : obj → Prop

-- If someone uses fresh ingredients, the food will taste delicious.
axiom R1 : ∀ x : obj, Fresh x → Delicious x
-- If someone cooks a complicated recipe, it takes more time to prepare.
axiom R2 : ∀ x : obj, Complicated x → MoreTime x
-- The chef used fresh ingredients.
axiom Fresh_Chef : Fresh Chef

-- If the chef cooked a complicated recipe, the food tasted delicious.
theorem complicated_implies_delicious_for_chef : Complicated Chef → Delicious Chef := by
  intro _hComp
  exact R1 Chef Fresh_Chef
</lean>

ANSWER: Yes
