<lean>
axiom Waters : Prop
axiom GrowStrong : Prop
axiom Monsoon : Prop
axiom HeavyRains : Prop
axiom BountifulHarvest : Prop
axiom GoodProfit : Prop

axiom A1 : Waters → GrowStrong
axiom A2 : Monsoon → HeavyRains
axiom A3 : Waters ∨ ¬ HeavyRains
axiom A4 : GrowStrong → BountifulHarvest
axiom A5 : BountifulHarvest → GoodProfit

theorem monsoon_profit : Monsoon → GoodProfit := by
  intro hm
  have hr : HeavyRains := A2 hm
  have w : Waters := Or.elim A3 (fun hw => hw) (fun nh => False.elim (nh hr))
  have gs : GrowStrong := A1 w
  have bh : BountifulHarvest := A4 gs
  exact A5 bh
</lean>

ANSWER: Yes
