<lean>
axiom Good : Prop
axiom Many : Prop
axiom Low : Prop
axiom BigStar : Prop
axiom Sequel : Prop

-- If the movie gets good reviews, then many people will go see it.
axiom R1 : Good → Many
-- If a big star is in the movie, then it got good reviews.
axiom R4 : BigStar → Good
-- Either there is a big star, or it is a sequel, or both.
axiom D5 : BigStar ∨ Sequel ∨ (BigStar ∧ Sequel)
-- "Not many people went to see it" corresponds to "attendance was low."
axiom Low_iff_notMany : Low ↔ ¬ Many

theorem low_implies_sequel : Low → Sequel := by
  intro hLow
  have hNotMany : ¬ Many := Low_iff_notMany.mp hLow
  have h := D5
  cases h with
  | inl hBS =>
      have hMany : Many := R1 (R4 hBS)
      have contra : False := hNotMany hMany
      exact False.elim contra
  | inr h' =>
      cases h' with
      | inl hSeq => exact hSeq
      | inr hBoth => exact hBoth.right
</lean>

ANSWER: Yes
