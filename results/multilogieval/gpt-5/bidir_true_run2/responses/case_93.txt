<lean>
axiom PracticesHard : Prop
axiom Prepared : Prop
axiom Injured : Prop
axiom CannotPlay : Prop
axiom Wins : Prop
axiom Celebrate : Prop

axiom R1 : PracticesHard → Prepared
axiom R2 : Injured → CannotPlay
axiom R3 : PracticesHard ∨ ¬ Injured
axiom R4 : Prepared → Wins
axiom R5 : Wins → Celebrate

theorem injured_celebrate : Injured → Celebrate := by
  intro hI
  have hp : PracticesHard :=
    Or.elim R3
      (fun hp => hp)
      (fun hnotI => False.elim (hnotI hI))
  have hprep : Prepared := R1 hp
  have hwin : Wins := R4 hprep
  exact R5 hwin
</lean>

ANSWER: Yes
