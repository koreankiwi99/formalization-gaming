axiom person : Type
axiom Mary : person

axiom Problems : person → Prop
axiom Restarted : person → Prop
axiom CheckedUpdates : person → Prop
axiom InstalledLatest : person → Prop

-- Context rules:
-- If someone is having computer problems, they should try restarting it.
axiom R1 : ∀ x : person, Problems x → Restarted x
-- If there are issues with a computer, its user should check for any software updates.
axiom R2 : ∀ x : person, Problems x → CheckedUpdates x

-- Given statement to evaluate:
-- Either Mary did not restart her computer or she did not install the latest updates.
axiom S1 : ¬ Restarted Mary ∨ ¬ InstalledLatest Mary

-- From R1 we can derive the contrapositive: if Mary did not restart, then she did not have problems.
theorem not_problems_of_not_restarted (x : person) (hnr : ¬ Restarted x) : ¬ Problems x :=
by
  intro hp
  exact hnr (R1 x hp)

-- What S1 actually gives us in combination with R1 is only this disjunction:
theorem consequence_of_S1 : ¬ Problems Mary ∨ ¬ InstalledLatest Mary :=
by
  cases S1 with
  | inl hnr =>
      apply Or.inl
      exact not_problems_of_not_restarted Mary hnr
  | inr hni =>
      apply Or.inr
      exact hni
