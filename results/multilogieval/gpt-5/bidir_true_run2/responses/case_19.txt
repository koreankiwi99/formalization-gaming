<lean>
axiom GreatService : Prop
axiom HostEvents : Prop
axiom ReputationGrows : Prop
axiom TrafficIncreases : Prop
axiom RevenueImproves : Prop

axiom A1 : GreatService → ReputationGrows
axiom A2 : HostEvents → TrafficIncreases
axiom A3 : GreatService ∨ HostEvents
axiom A4 : TrafficIncreases → RevenueImproves

-- Given: service quality decreased, so not great service
axiom NotGreat : ¬ GreatService

theorem revenue_improved : RevenueImproves :=
by
  have hEvents : HostEvents := Or.resolve_left A3 NotGreat
  have hTraffic : TrafficIncreases := A2 hEvents
  exact A4 hTraffic
</lean>

ANSWER: Yes
