<lean>
axiom obj : Type
axiom Person : obj

axiom SavedAggressively : obj → Prop
axiom ImprovedCredit : obj → Prop
axiom AffordDownPayment : obj → Prop
axiom QualifyMortgage : obj → Prop
axiom BuyDreamHome : obj → Prop

axiom R1 : ∀ x : obj, SavedAggressively x → AffordDownPayment x
axiom R2 : ∀ x : obj, ImprovedCredit x → QualifyMortgage x
axiom R3 : SavedAggressively Person ∨ ImprovedCredit Person
axiom R4 : ∀ x : obj, QualifyMortgage x → BuyDreamHome x

axiom ND : ¬ AffordDownPayment Person

theorem bought_dream_home : BuyDreamHome Person :=
by
  have ic : ImprovedCredit Person :=
    Or.elim R3
      (fun s => False.elim (ND (R1 Person s)))
      (fun c => c)
  have qm : QualifyMortgage Person := R2 Person ic
  exact R4 Person qm
</lean>

ANSWER: Yes
