-- If someone studies hard for an exam, they will get good grades.
axiom R1 : ∀ x : obj, StudiedHard x → GoodGrades x
-- If there is a snowstorm, then school is canceled.
axiom R2 : Snowstorm → SchoolCanceled
-- Either Priya studied hard, or school was not called off.
axiom T1 : StudiedHard Priya ∨ ¬ SchoolCanceled
-- If a student gets good grades, their parents are proud.
axiom R3 : ∀ x : obj, GoodGrades x → ParentsProud x
-- When parents are proud, they take their kids out for ice cream.
axiom R4 : ∀ x : obj, ParentsProud x → IceCream x

-- From a (plain) snowstorm, we can derive ice cream for Priya.
theorem snowstorm_implies_icecream : Snowstorm → IceCream Priya := by
  intro hs
  have hc : SchoolCanceled := R2 hs
  have hstud : StudiedHard Priya := by
    cases T1 with
    | inl h => exact h
    | inr hnot =>
      have contra : False := hnot hc
      exact False.elim contra
  have hg : GoodGrades Priya := R1 Priya hstud
  have hp : ParentsProud Priya := R3 Priya hg
  exact R4 Priya hp
