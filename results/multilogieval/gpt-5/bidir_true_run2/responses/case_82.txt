axiom WonContest : obj → Prop
axiom Celebrates : obj → Prop
axiom BakesPie : obj → Prop
axiom HouseSmells : Prop
axiom HouseCozy : Prop

-- If the farmer's market has ripe, juicy peaches, someone will bake a peach pie.
axiom R1 : PeachesInSeason → ∃ x : obj, BakesPie x
-- If someone won a baking contest, they'll celebrate with friends and family.
axiom R2 : ∀ x : obj, WonContest x → Celebrates x
-- Either peaches are in season, or someone won a contest (or both).
axiom R3 : PeachesInSeason ∨ ∃ x : obj, WonContest x
-- If someone bakes a pie, the house will smell amazing.
axiom R4 : (∃ x : obj, BakesPie x) → HouseSmells
-- If the house smells amazing, it feels like a homey, cozy place to relax.
axiom R5 : HouseSmells → HouseCozy

-- From peaches in season, we can derive the house is cozy.
theorem cozy_if_peaches : PeachesInSeason → HouseCozy :=
fun hp => R5 (R4 (R1 hp))
