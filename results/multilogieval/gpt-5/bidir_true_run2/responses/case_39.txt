<lean>
axiom Person : Type
axiom Alex : Person

axiom Monday : Prop
axiom Quiz : Prop

axiom StudiedHard : Person → Prop
axiom DoWell : Person → Prop
axiom GoodAtMath : Person → Prop

-- If it was Monday, then there would be a quiz.
axiom A1 : Monday → Quiz
-- If they do well on the quiz, then they studied hard.
axiom A2 : ∀ x : Person, DoWell x → StudiedHard x
-- Either it was Monday, or they did not study hard, or it was Monday and they did not study hard.
axiom A3 : Monday ∨ ¬StudiedHard Alex ∨ (Monday ∧ ¬StudiedHard Alex)
-- Someone who is good at math does well on quizzes.
axiom A4 : ∀ x : Person, GoodAtMath x → DoWell x

-- From no quiz, we can derive Alex is not good at math.
theorem not_good_if_no_quiz : ¬Quiz → ¬GoodAtMath Alex :=
by
  intro hNoQuiz
  -- From ¬Quiz and (Monday → Quiz), deduce ¬Monday
  have hNotMon : ¬Monday := fun hMon => hNoQuiz (A1 hMon)
  -- From the given disjunction and ¬Monday, deduce ¬StudiedHard Alex
  have hNotStud : ¬StudiedHard Alex :=
    Or.elim A3
      (fun hMon => False.elim (hNotMon hMon))
      (fun hOr =>
        Or.elim hOr
          (fun hNot => hNot)
          (fun hMonAnd => hMonAnd.right)
      )
  -- From DoWell → StudiedHard and ¬StudiedHard, deduce ¬DoWell Alex
  have hNotDoWell : ¬DoWell Alex := fun hDo => hNotStud (A2 Alex hDo)
  -- From GoodAtMath → DoWell and ¬DoWell, deduce ¬GoodAtMath Alex
  exact fun hGood => hNotDoWell (A4 Alex hGood)
