<lean>
axiom Party : Type
axiom Person : Type

axiom BroughtDish : Person → Party → Prop
axiom EnoughFood : Party → Prop
axiom Birthday : Party → Prop
axiom Cake : Party → Prop
axiom IceCream : Party → Prop

-- If someone brings a dish to share at a party, there will be enough food for everyone.
axiom R1 : ∀ p : Party, ∀ g : Person, BroughtDish g p → EnoughFood p
-- If it is a birthday party, there will be cake and ice cream.
axiom R2 : ∀ p : Party, Birthday p → Cake p ∧ IceCream p
-- Either someone brought a dish to share, or there was no birthday cake.
axiom R3 : ∀ p : Party, (∃ g : Person, BroughtDish g p) ∨ ¬ Cake p

axiom SamParty : Party
axiom SamBirthday : Birthday SamParty

theorem sam_enough_food : EnoughFood SamParty := by
  have hcake : Cake SamParty := (R2 SamParty SamBirthday).left
  have h := R3 SamParty
  cases h with
  | inl hex =>
      cases hex with
      | intro g hg =>
          exact R1 SamParty g hg
  | inr hncake =>
      have : False := hncake hcake
      exact False.elim this
</lean>

ANSWER: Yes
