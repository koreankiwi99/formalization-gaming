<lean>
axiom D : Prop          -- display looks nice
axiom JI : Prop         -- judges are impressed
axiom T : Prop          -- research is thorough
axiom E : Prop          -- experts take it seriously
axiom A : Prop          -- receives an award

axiom r1 : D → JI       -- if display looks nice then judges are impressed
axiom r2 : T → E        -- if research is thorough then experts take it seriously
axiom r3 : D ∨ T        -- either good display or thorough research
axiom r4 : E → A        -- if experts take it seriously then it receives an award
axiom notJI : ¬ JI      -- judges were not impressed

theorem project_award : A := by
  have notD : ¬ D := by
    intro d
    exact notJI (r1 d)
  have t : T := by
    cases r3 with
    | inl d =>
      exact False.elim (notD d)
    | inr t =>
      exact t
  have e : E := r2 t
  exact r4 e
</lean>

ANSWER: Yes
