<lean>
axiom WR : Prop                  -- the garden is watered routinely
axiom Bloom : Prop               -- the flowers bloom brightly
axiom HeavyRain : Prop           -- there was heavy rain
axiom SoilSaturated : Prop       -- the soil is saturated
axiom Fragrance : Prop           -- there is fragrance
axiom Butterflies : Prop         -- the garden attracts butterflies

axiom R1 : WR → Bloom
axiom R2 : HeavyRain → SoilSaturated
axiom R3 : WR ∨ ¬SoilSaturated
axiom R4 : Bloom → Fragrance
axiom R5 : Fragrance → Butterflies

theorem attract_from_heavy_rain : HeavyRain → Butterflies :=
by
  intro hr
  have hs : SoilSaturated := R2 hr
  have hwr : WR := by
    cases R3 with
    | inl hw => exact hw
    | inr hnotsat =>
      have : False := hnotsat hs
      exact this.elim
  have hb : Bloom := R1 hwr
  have hf : Fragrance := R4 hb
  exact R5 hf
</lean>

ANSWER: Yes
