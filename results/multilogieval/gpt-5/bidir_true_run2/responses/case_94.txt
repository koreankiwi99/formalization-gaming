<lean>
axiom obj : Type
axiom Lauren : obj

axiom TrainedHardWeekly : obj → Prop
axiom StretchedAfterRuns : obj → Prop
axiom RecoverySerious : obj → Prop
axiom FlexibilityFocus : obj → Prop
axiom RunsRacesFaster : obj → Prop

-- If they trained hard weekly, they could improve their race times (run races faster).
axiom R_train_fast : ∀ x : obj, TrainedHardWeekly x → RunsRacesFaster x
-- If they stretched after runs, they took recovery seriously.
axiom R_stretch_recovery : ∀ x : obj, StretchedAfterRuns x → RecoverySerious x
-- If they took recovery seriously, they focused on flexibility to prevent injury.
axiom R_recovery_flex : ∀ x : obj, RecoverySerious x → FlexibilityFocus x
-- Either they trained hard at sprinting, or they stretched regularly after runs, or both.
axiom E_hard_or_stretch : ∀ x : obj, TrainedHardWeekly x ∨ StretchedAfterRuns x

theorem lauren_flex_if_not_faster : (¬ RunsRacesFaster Lauren) → FlexibilityFocus Lauren := by
  intro hnot
  have hdisj := E_hard_or_stretch Lauren
  cases hdisj with
  | inl hhard =>
    have hfaster := R_train_fast Lauren hhard
    exact False.elim (hnot hfaster)
  | inr hstretch =>
    have hrec := R_stretch_recovery Lauren hstretch
    exact R_recovery_flex Lauren hrec
</lean>

ANSWER: Yes
