<lean>
axiom obj : Type
axiom Jesse : obj

axiom Sprint : obj → Prop
axiom Stretch : obj → Prop
axiom RunningImproved : obj → Prop
axiom GoalTime : obj → Prop
axiom Flexible : obj → Prop
axiom LessInjuryProne : obj → Prop

axiom A1 : ∀ x : obj, Sprint x → RunningImproved x
axiom A2 : ∀ x : obj, RunningImproved x → GoalTime x
axiom A3 : Sprint Jesse ∨ Stretch Jesse
axiom A4 : ∀ x : obj, Stretch x → (Flexible x ∧ LessInjuryProne x)
axiom NotImprovedJ : ¬ RunningImproved Jesse

theorem jesse_flexible_and_less_injury_prone : Flexible Jesse ∧ LessInjuryProne Jesse :=
by
  have notSprint : ¬ Sprint Jesse := fun hs => NotImprovedJ (A1 Jesse hs)
  have stretchJ : Stretch Jesse :=
    Or.elim A3 (fun hs => False.elim (notSprint hs)) (fun ht => ht)
  exact A4 Jesse stretchJ
</lean>

ANSWER: Yes
