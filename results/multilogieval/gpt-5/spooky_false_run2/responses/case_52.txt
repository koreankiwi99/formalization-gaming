axiom RainF : Prop
axiom BringUmbrellaJames : Prop
axiom GoPlay : Prop
axiom BigGame : Prop
axiom Together : Prop
axiom HighView : Prop

-- Premises
axiom R1 : RainF → BringUmbrellaJames
axiom R2 : BigGame → Together
axiom R3 : RainF ∨ ¬Together ∨ (RainF ∧ ¬Together)
axiom R4 : HighView → BigGame
axiom R5 : HighView ∨ GoPlay ∨ (HighView ∧ GoPlay)

-- Given in the question: James did not bring an umbrella
axiom HnotUmb : ¬BringUmbrellaJames

theorem no_rain : ¬RainF := by
  intro r
  exact HnotUmb (R1 r)

theorem not_together : ¬Together := by
  have h := R3
  cases h with
  | inl hR =>
      exact False.elim (no_rain hR)
  | inr h' =>
      cases h' with
      | inl hNotT =>
          exact hNotT
      | inr hBoth =>
          exact hBoth.right

theorem not_high : ¬HighView := by
  intro hv
  have bg : BigGame := R4 hv
  have t : Together := R2 bg
  exact not_together t

theorem go_play_true : GoPlay := by
  cases R5 with
  | inl hv =>
      exact False.elim (not_high hv)
  | inr h' =>
      cases h' with
      | inl gp =>
          exact gp
      | inr both =>
          exact both.right

-- From the premises (including HnotUmb), we can in fact derive GoPlay.
-- Hence we cannot prove the negation of the queried implication.
theorem entailment_yes : ¬BringUmbrellaJames → GoPlay := by
  intro _
  exact go_play_true
