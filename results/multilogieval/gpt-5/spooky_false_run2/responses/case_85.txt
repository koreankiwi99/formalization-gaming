theorem john_soil_high_if_not_attract : (¬Attract John) → SoilHigh John :=
by
  intro hNotAttr
  have hNotWater : ¬Water John := by
    intro hW
    have hBloom : Bloom John := H1 John hW
    have hAttr : Attract John := H2 John hBloom
    exact hNotAttr hAttr
  have hWO : Water John ∨ Organic John := H3 John
  have hOrg : Organic John := by
    cases hWO with
    | inl hW => exact False.elim (hNotWater hW)
    | inr hO => exact hO
  exact H4 John hOrg
