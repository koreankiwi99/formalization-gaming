axiom obj : Type
axiom Priya : obj

axiom WaterDaily : obj → Prop
axiom GrowWell : obj → Prop
axiom NotEnoughWater : obj → Prop
axiom WiltDie : obj → Prop

axiom Drought : Prop
axiom Restrictions : Prop

-- If someone waters their plants daily, the plants will grow well.
axiom R1 : ∀ x : obj, WaterDaily x → GrowWell x
-- If there is a drought, then water restrictions will be imposed.
axiom R2 : Drought → Restrictions
-- Either they (Priya) watered the plants daily, or there was a drought (or both).
axiom R3 : WaterDaily Priya ∨ Drought
-- If water restrictions were imposed, then Priya's plants would not get enough water.
axiom R4 : Restrictions → NotEnoughWater Priya
-- If the plants lacked water, they would wilt and die.
axiom R5 : ∀ x : obj, NotEnoughWater x → WiltDie x

-- From "Priya's plants did not grow well", we can derive "they wilted and died".
theorem wilt_if_not_grow : ¬GrowWell Priya → WiltDie Priya :=
by
  intro hNotGrow
  -- Modus tollens on R1
  have hNotWater : ¬WaterDaily Priya := by
    intro hW
    have hG : GrowWell Priya := R1 Priya hW
    exact hNotGrow hG
  -- From (WaterDaily Priya ∨ Drought) and ¬WaterDaily Priya, infer Drought
  have hD : Drought := by
    cases R3 with
    | inl hW => exact False.elim (hNotWater hW)
    | inr hD => exact hD
  -- Chain the implications to WiltDie Priya
  have hRes : Restrictions := R2 hD
  have hNotEnough : NotEnoughWater Priya := R4 hRes
  exact R5 Priya hNotEnough
