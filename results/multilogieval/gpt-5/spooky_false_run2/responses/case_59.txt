<lean>
axiom obj : Type
axiom Emma : obj

-- Predicates
axiom LearnedBasicChords : obj → Prop
axiom StrummingPatterns : obj → Prop
axiom PlaySimpleSongs : obj → Prop
axiom EntertainFriends : obj → Prop
axiom TookLessons : obj → Prop
axiom KnowsMusicTheory : obj → Prop

-- Premises
axiom P1 : ∀ x : obj, (LearnedBasicChords x ∧ StrummingPatterns x) → PlaySimpleSongs x
axiom P2 : ∀ x : obj, PlaySimpleSongs x → EntertainFriends x
axiom P3 : ∀ x : obj, LearnedBasicChords x ∨ TookLessons x
axiom P4 : ∀ x : obj, TookLessons x → KnowsMusicTheory x

-- Given: Emma is unable to entertain friends with the guitar.
axiom Emma_unable_to_entertain : ¬EntertainFriends Emma

-- Consequences we can derive (but none imply ¬KnowsMusicTheory Emma)

-- From inability to entertain and P2, Emma cannot play simple songs.
theorem Emma_not_play_simple : ¬PlaySimpleSongs Emma := by
  intro hps
  apply Emma_unable_to_entertain
  apply P2 Emma
  exact hps

-- Therefore Emma did not both learn basic chords and strumming patterns.
theorem Emma_not_both_chords_and_strum : ¬(LearnedBasicChords Emma ∧ StrummingPatterns Emma) := by
  intro h
  have hps : PlaySimpleSongs Emma := P1 Emma h
  apply Emma_unable_to_entertain
  apply P2 Emma
  exact hps

-- If Emma learned basic chords, then she did not learn strumming patterns.
theorem Emma_chords_implies_not_strum : LearnedBasicChords Emma → ¬StrummingPatterns Emma := by
  intro hbc
  intro hsp
  have hboth : LearnedBasicChords Emma ∧ StrummingPatterns Emma := And.intro hbc hsp
  have hps : PlaySimpleSongs Emma := P1 Emma hboth
  apply Emma_unable_to_entertain
  apply P2 Emma
  exact hps

-- If Emma took lessons, then she knows music theory (does not help to prove the negation).
theorem Emma_lessons_imply_theory : TookLessons Emma → KnowsMusicTheory Emma := by
  intro h
  exact P4 Emma h
