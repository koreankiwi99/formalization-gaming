<lean>
axiom obj : Type
axiom John : obj

axiom Alarm : obj → Prop
axiom TimeToGetReady : obj → Prop
axiom MissedBus : obj → Prop
axiom LateForSchool : obj → Prop
axiom GotUpLate : obj → Prop
axiom WentToBedLate : obj → Prop

axiom T1 : ∀ x : obj, Alarm x → TimeToGetReady x
axiom T2 : ∀ x : obj, MissedBus x → LateForSchool x
axiom T3 : ∀ x : obj, Alarm x ∨ ¬LateForSchool x
axiom T4 : ∀ x : obj, GotUpLate x → MissedBus x
axiom T5 : ∀ x : obj, WentToBedLate x → GotUpLate x

axiom KJohn : ¬TimeToGetReady John

theorem john_not_went_to_bed_late : ¬WentToBedLate John := by
  intro hW
  have hG : GotUpLate John := T5 John hW
  have hM : MissedBus John := T4 John hG
  have hL : LateForSchool John := T2 John hM
  have hOr : Alarm John ∨ ¬LateForSchool John := T3 John
  have hNotAlarm : ¬Alarm John := by
    intro hA
    have hT : TimeToGetReady John := T1 John hA
    exact KJohn hT
  have hNotLate : ¬LateForSchool John := by
    cases hOr with
    | inl hA => exact False.elim (hNotAlarm hA)
    | inr hNL => exact hNL
  exact hNotLate hL
</lean>

ANSWER: No
