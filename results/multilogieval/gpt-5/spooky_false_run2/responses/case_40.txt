axiom obj : Type

axiom James : obj
axiom Started : obj → Prop
axiom Motivated : obj → Prop
axiom GymClosed : Prop
axiom CantFollow : obj → Prop
axiom Frustrated : obj → Prop
axiom Snacks : obj → Prop
axiom MightSnack : obj → Prop

-- Premises
axiom R1 : ∀ x : obj, Started x → Motivated x
axiom R2 : GymClosed → ∀ x : obj, CantFollow x
axiom R3 : Started James ∨ GymClosed
axiom R4 : ∀ x : obj, CantFollow x → Frustrated x
-- "might indulge" is captured as a weaker predicate than actually indulging
axiom R5 : ∀ x : obj, Frustrated x → MightSnack x

-- Given: James was not motivated to workout.
axiom NotMotivatedJames : ¬Motivated James

-- Derived consequences
theorem not_started : ¬Started James := by
  intro h
  apply NotMotivatedJames
  exact R1 James h

theorem gym_closed : GymClosed := by
  cases R3 with
  | inl hstart =>
      have : False := by
        apply NotMotivatedJames
        exact R1 James hstart
      exact False.elim this
  | inr hgc =>
      exact hgc

theorem cant_follow : CantFollow James := by
  have hgc := gym_closed
  have hAll : ∀ x : obj, CantFollow x := R2 hgc
  exact hAll James

theorem frustrated : Frustrated James := by
  apply R4 James
  exact cant_follow

theorem might_indulge : MightSnack James := by
  apply R5 James
  exact frustrated
