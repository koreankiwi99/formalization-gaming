axiom Casey : obj

axiom Adopts : obj → Prop
axiom ProvidesHome : obj → Prop
axiom HelpsCommunity : obj → Prop
axiom FeelsFulfillment : obj → Prop

-- Premises
axiom V_to_H : ∀ x : obj, Volunteers x → HelpsCommunity x
axiom A_to_PH : ∀ x : obj, Adopts x → ProvidesHome x
axiom H_to_F : ∀ x : obj, HelpsCommunity x → FeelsFulfillment x
axiom disj : ∀ x : obj, Volunteers x ∨ ¬ProvidesHome x ∨ (Volunteers x ∧ ¬ProvidesHome x)

-- From the premises, adopting implies feeling fulfillment (in particular for Casey)
theorem adopts_implies_fulfillment_Casey : Adopts Casey → FeelsFulfillment Casey := by
  intro hAdopt
  have hPH : ProvidesHome Casey := A_to_PH Casey hAdopt
  have hD : Volunteers Casey ∨ ¬ProvidesHome Casey ∨ (Volunteers Casey ∧ ¬ProvidesHome Casey) := disj Casey
  have hVol : Volunteers Casey := by
    cases hD
    case inl hv => exact hv
    case inr hrest =>
      cases hrest
      case inl hNotPH =>
        exact False.elim (hNotPH hPH)
      case inr hv_and_not =>
        have hNotPH2 : ¬ProvidesHome Casey := hv_and_not.right
        exact False.elim (hNotPH2 hPH)
  have hHelp : HelpsCommunity Casey := V_to_H Casey hVol
  exact H_to_F Casey hHelp
