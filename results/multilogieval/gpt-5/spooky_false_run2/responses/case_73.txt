axiom obj : Type
axiom Simone : obj

axiom Train : obj → Prop
axiom Rest : obj → Prop
axiom Peak : obj → Prop

-- Coaches' principle: if someone trains with dedication, then if they get adequate rest, they will perform at their peak.
axiom R1 : ∀ x : obj, Train x → Rest x → Peak x

-- Facts about Simone
axiom Rest_Simone : Rest Simone
axiom NotPeak_Simone : ¬Peak Simone

-- Conclusion: Simone did not train with dedication.
theorem simone_not_train : ¬Train Simone := by
  intro hTrain
  have hPeak : Peak Simone := R1 Simone hTrain Rest_Simone
  exact NotPeak_Simone hPeak
