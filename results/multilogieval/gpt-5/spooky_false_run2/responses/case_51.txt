axiom obj : Type
axiom Mary : obj

axiom Sale : Prop
axiom StockUp : obj → Prop
axiom ShoppingList : obj → Prop
axiom Plan : obj → Prop
axiom Payday : Prop
axiom PricesUp : Prop

axiom R1 : Sale → ∀ x : obj, StockUp x
axiom R2 : ∀ x : obj, ShoppingList x → Plan x
axiom R3 : Sale ∨ ¬Plan Mary ∨ (Sale ∧ ¬Plan Mary)
axiom R4 : Payday → ∀ x : obj, ShoppingList x
axiom R5 : Payday ∨ PricesUp

theorem prices_up_from_mary_not_stock : (¬StockUp Mary) → PricesUp := by
  intro h_not_stock
  have h_not_sale : ¬Sale := by
    intro hsale
    have h_su_mary : StockUp Mary := (R1 hsale) Mary
    exact h_not_stock h_su_mary
  have h_not_plan : ¬Plan Mary := by
    refine Or.elim R3 (fun hs => ?np1) (fun hrest => ?np2)
    case np1 =>
      intro _
      exact False.elim (h_not_sale hs)
    case np2 =>
      refine Or.elim hrest (fun hnp => hnp) (fun h_and => ?np3)
      case np3 =>
        intro _
        exact False.elim (h_not_sale h_and.left)
  have h_payday_implies_plan : Payday → Plan Mary := by
    intro hpay
    have h_list_mary : ShoppingList Mary := (R4 hpay) Mary
    exact (R2 Mary) h_list_mary
  have h_not_payday : ¬Payday := by
    intro hpay
    exact h_not_plan (h_payday_implies_plan hpay)
  have h_prices : PricesUp := by
    refine Or.elim R5 (fun hpay => ?contra) (fun hp => hp)
    case contra =>
      exact False.elim (h_not_payday hpay)
  exact h_prices
