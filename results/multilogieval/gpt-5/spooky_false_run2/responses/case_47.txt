axiom SpecialOrderReady : Prop

axiom LearnNewSkills : obj → Prop

-- If the package arrives on time, people will be able to start a new project.
axiom A1 : ∀ x : obj, PackageOnTime → StartProject x
-- If the special order is ready, they can pick up the materials they need.

axiom A3 : PackageOnTime ∨ SpecialOrderReady
-- If someone starts his new project, they will learn some new skills.
axiom A4 : ∀ x : obj, StartProject x → LearnNewSkills x

-- Max was unable to pick up the materials he needed.
axiom A5 : ¬PickUpMaterials Max

-- From the premises, Max actually learns some new skills.
theorem max_learned_new_skills : LearnNewSkills Max := by
  have hNotS : ¬SpecialOrderReady := fun s => A5 (A2 Max s)
  have hP : PackageOnTime :=
    Or.elim A3
      (fun hp => hp)
      (fun hs => False.elim (hNotS hs))
  have hStart : StartProject Max := A1 Max hP
  exact A4 Max hStart
