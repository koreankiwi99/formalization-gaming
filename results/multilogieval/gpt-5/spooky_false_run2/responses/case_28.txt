<lean>
axiom obj : Type
axiom Jane : obj
axiom Injured : obj → Prop
axiom SeeDoctor : obj → Prop
axiom Rest : obj → Prop

-- If someone gets injured, they should see a doctor.
axiom H1 : ∀ x : obj, Injured x → SeeDoctor x
-- Getting hurt warrants resting the affected area.
axiom H2 : ∀ x : obj, Injured x → Rest x
-- Either Jane did not see a doctor or she did not rest her injury.
axiom H3 : ¬SeeDoctor Jane ∨ ¬Rest Jane

theorem jane_not_injured : ¬Injured Jane := by
  intro hInj
  cases H3 with
  | inl hNoDoc =>
      exact hNoDoc (H1 Jane hInj)
  | inr hNoRest =>
      exact hNoRest (H2 Jane hInj)
</lean>

ANSWER: No
