<lean>
axiom obj : Type
axiom Priya : obj
axiom Attend : obj → Prop
axiom Study : obj → Prop
axiom Excel : obj → Prop

-- Context: If someone attends lectures regularly, then if they study diligently, they will excel.
axiom R1 : ∀ x : obj, Attend x → (Study x → Excel x)

-- Given facts about Priya
axiom PriyaStudied : Study Priya
axiom PriyaNotExcel : ¬ Excel Priya

-- Goal: Priya did not attend lectures regularly
theorem Priya_not_attend : ¬ Attend Priya := by
  intro hAttend
  have hImp : Study Priya → Excel Priya := R1 Priya hAttend
  have hExcel : Excel Priya := hImp PriyaStudied
  exact PriyaNotExcel hExcel
</lean>

ANSWER: No
