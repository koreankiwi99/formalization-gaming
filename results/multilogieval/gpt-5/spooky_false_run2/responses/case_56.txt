<lean>
axiom obj : Type
axiom Emma : obj

axiom SpeakDaily : obj → Prop
axiom ListenPodcasts : obj → Prop
axiom FluencyImproved : obj → Prop
axiom VocabImproved : obj → Prop
axiom RealConversations : obj → Prop
axiom ListeningComprehension : obj → Prop

-- If someone speaks daily with native speakers, their fluency and vocabulary improve.
axiom H1 : ∀ x : obj, SpeakDaily x → (FluencyImproved x ∧ VocabImproved x)
-- If fluency improves, they can have real conversations.
axiom H2 : ∀ x : obj, FluencyImproved x → RealConversations x
-- Either someone speaks daily with native speakers, or they listen to educational podcasts (or both).
axiom H3 : ∀ x : obj, SpeakDaily x ∨ ListenPodcasts x
-- If someone listens to educational podcasts, they gain valuable listening comprehension.
axiom H4 : ∀ x : obj, ListenPodcasts x → ListeningComprehension x

-- From the premises, the statement itself is derivable:
theorem emma_listening_from_no_convo : (¬RealConversations Emma → ListeningComprehension Emma) := by
  intro h_no_rc
  have h := H3 Emma
  cases h with
  | inl h_speak =>
      have h_fl := (H1 Emma) h_speak
      have h_rc := (H2 Emma) h_fl.left
      exact False.elim (h_no_rc h_rc)
  | inr h_listen =>
      exact H4 Emma h_listen
</lean>
