<lean>
axiom Marketing : Prop
axiom ProdCostUp : Prop
axiom MoreCustomers : Prop
axiom SalesUp : Prop

axiom invest_marketing_gives_customers : Marketing → MoreCustomers
axiom customers_increase_sales : MoreCustomers → SalesUp
axiom marketing_or_costs : Marketing ∨ ProdCostUp

-- From the context and the claim that sales did not increase, we can derive that production costs increased.
theorem costs_increased_if_sales_not_up : ¬SalesUp → ProdCostUp := by
  intro hNoSales
  -- Prove that Marketing did not happen
  have hNotMarketing : ¬Marketing := by
    intro m
    have mc : MoreCustomers := invest_marketing_gives_customers m
    have su : SalesUp := customers_increase_sales mc
    exact hNoSales su
  -- Use the disjunction to conclude ProdCostUp
  cases marketing_or_costs with
  | inl m =>
      exact False.elim (hNotMarketing m)
  | inr pc =>
      exact pc
</lean>
