<lean>
axiom obj : Type
axiom Liz : obj

axiom PreventiveMaintenance : obj → Prop
axiom QualityPartsUsed : obj → Prop
axiom LongServiceLife : obj → Prop

axiom MechanicsClaim : ∀ x : obj, PreventiveMaintenance x → (QualityPartsUsed x → LongServiceLife x)
axiom QLiz : QualityPartsUsed Liz
axiom NotLongLiz : ¬LongServiceLife Liz

theorem liz_no_preventive : ¬PreventiveMaintenance Liz := by
  intro hP
  have hImp : QualityPartsUsed Liz → LongServiceLife Liz := MechanicsClaim Liz hP
  have hL : LongServiceLife Liz := hImp QLiz
  exact NotLongLiz hL
</lean>

ANSWER: No
