axiom obj : Type

axiom Jake : obj
axiom Someone : obj

axiom WeatherNice : Prop
axiom RelaxOutside : obj → Prop
axiom PackedEnough : obj → Prop
axiom ReadsBooks : obj → Prop
axiom Prepared : obj → Prop

-- If the weather was nice, then they could relax outside.
axiom H1 : ∀ x : obj, WeatherNice → RelaxOutside x
-- If they read books, that means they packed enough.
axiom H2 : ∀ x : obj, ReadsBooks x → PackedEnough x
-- If someone is prepared, they read books.
axiom H3 : ∀ x : obj, Prepared x → ReadsBooks x
-- Either the weather was nice, or they did not pack enough, or both.
axiom H4 : ∀ x : obj, WeatherNice ∨ ¬PackedEnough x ∨ (WeatherNice ∧ ¬PackedEnough x)
-- Someone was getting ready (prepared).
axiom H5 : Prepared Someone

-- From the disjunction and the existence of a prepared person, we can deduce the weather was nice.
theorem weather_true : WeatherNice :=
by
  have hp : PackedEnough Someone := H2 Someone (H3 Someone H5)
  have hdisj : WeatherNice ∨ ¬PackedEnough Someone ∨ (WeatherNice ∧ ¬PackedEnough Someone) := H4 Someone
  cases hdisj with
  | inl hW =>
    exact hW
  | inr hrest =>
    cases hrest with
    | inl hNotPacked =>
      exact False.elim (hNotPacked hp)
    | inr hBoth =>
      exact hBoth.left

-- Therefore, Jake can relax outside.
theorem relax_jake : RelaxOutside Jake := H1 Jake weather_true
