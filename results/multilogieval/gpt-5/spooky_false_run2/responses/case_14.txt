<lean>
axiom obj : Type
axiom Alex : obj

axiom AdvertisesCleverly : obj → Prop
axiom OffersDiscounts : obj → Prop
axiom AttractsMoreCustomers : obj → Prop
axiom IncreasesSales : obj → Prop
axiom ExpandsBusiness : obj → Prop

-- If someone advertises cleverly, they attract more customers.
axiom A1 : ∀ x : obj, AdvertisesCleverly x → AttractsMoreCustomers x
-- If someone offers discounts, they increase sales.
axiom A2 : ∀ x : obj, OffersDiscounts x → IncreasesSales x
-- There exists someone who either advertises cleverly or offers discounts.
axiom A3 : ∃ x : obj, AdvertisesCleverly x ∨ OffersDiscounts x
-- If someone increases sales, they expand their business.
axiom A4 : ∀ x : obj, IncreasesSales x → ExpandsBusiness x

-- Given: Alex did not attract more customers.
axiom Alex_not_attracts : ¬AttractsMoreCustomers Alex

-- From not attracting more customers and A1, Alex did not advertise cleverly.
theorem Alex_not_advertises : ¬AdvertisesCleverly Alex :=
fun h => Alex_not_attracts (A1 Alex h)

-- If Alex offers discounts, then Alex increases sales and thus expands business.
theorem Alex_expands_if_offers : OffersDiscounts Alex → ExpandsBusiness Alex :=
fun h => A4 Alex (A2 Alex h)
