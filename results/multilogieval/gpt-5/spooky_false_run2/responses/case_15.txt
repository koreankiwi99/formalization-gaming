<lean>
axiom obj : Type
axiom Student : obj

axiom Studies : obj → Prop
axiom Participates : obj → Prop
axiom Excels : obj → Prop
axiom Leader : obj → Prop
axiom Prefect : obj → Prop

-- If they study diligently, they will excel in academics.
axiom R1 : ∀ x : obj, Studies x → Excels x
-- If they participate actively, they will gain leadership skills.
axiom R2 : ∀ x : obj, Participates x → Leader x
-- The student either studies hard or participates actively.
axiom R3 : Studies Student ∨ Participates Student
-- If someone is a leader, they will be selected as class prefect.
axiom R4 : ∀ x : obj, Leader x → Prefect x

-- From the given that the student did not excel academically, we can still conclude they were chosen as class prefect.
theorem chosen_prefect_from_not_excels : ¬Excels Student → Prefect Student := by
  intro hNotExcels
  -- From Studies → Excels and ¬Excels, derive ¬Studies (modus tollens)
  have hNotStudies : ¬Studies Student := by
    intro hStudies
    apply hNotExcels
    exact R1 Student hStudies
  -- From (Studies ∨ Participates) and ¬Studies, derive Participates
  have hParticipates : Participates Student := by
    cases R3 with
    | inl hStudies =>
        exact False.elim (hNotStudies hStudies)
    | inr hPart =>
        exact hPart
  -- Then Leader and hence Prefect
  have hLeader : Leader Student := R2 Student hParticipates
  exact R4 Student hLeader
</lean>
