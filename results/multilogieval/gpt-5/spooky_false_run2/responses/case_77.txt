axiom PlayWell : obj → Prop
axiom Start : obj → Prop

axiom Natural : obj → Prop
axiom TopYouth : obj → Prop

-- Premises from the text
axiom P1 : ∀ x : obj, PracticeHard x → PlayWell x
axiom P2 : ∀ x : obj, Start x → Talented x
axiom P3 : ∀ x : obj, PracticeHard x ∨ ¬Talented x
axiom P4 : ∀ x : obj, Natural x → Start x
axiom P5 : ∀ x : obj, TopYouth x → Natural x

-- From the premises, TopYouth implies PlayWell
theorem topYouth_implies_playWell : ∀ x : obj, TopYouth x → PlayWell x :=
  fun x hTop =>
    let hNat : Natural x := P5 x hTop
    let hStart : Start x := P4 x hNat
    let hTal : Talented x := P2 x hStart
    let disj : PracticeHard x ∨ ¬Talented x := P3 x
    have hPH : PracticeHard x :=
      Or.elim disj
        (fun h => h)
        (fun hnot => False.elim (hnot hTal))
    show PlayWell x from P1 x hPH

-- Hence, if James did not play well, then he was not a top youth player
theorem james_not_play_implies_not_topYouth : ¬PlayWell James → ¬TopYouth James :=
  fun hNotPW hTop => hNotPW (topYouth_implies_playWell James hTop)
