-- Premises
axiom P1 : MarketPeaches → ∃ p : Person, BakesPeachPie p
axiom P1b : ∀ p : Person, BakesPeachPie p → BakesPie p
axiom P2 : ∀ p : Person, WonContest p → Celebrate p
axiom P3 : InSeason ∨ ∃ p : Person, WonContest p
axiom P4 : ∀ p : Person, BakesPie p → HouseSmells
axiom P5 : HouseSmells → HomeyCozy
axiom P6 : InSeason → MarketPeaches

-- Derivations that follow from the premises
theorem homey_if_market_peaches : MarketPeaches → HomeyCozy := by
  intro mp
  have ex_pp : ∃ p : Person, BakesPeachPie p := P1 mp
  have ex_bp : ∃ p : Person, BakesPie p := by
    apply Exists.elim ex_pp
    intro p hpp
    exact Exists.intro p (P1b p hpp)
  have hs : HouseSmells := by
    apply Exists.elim ex_bp
    intro p hbp
    exact P4 p hbp
  exact P5 hs

theorem some_celebrates_if_someone_won : (∃ p : Person, WonContest p) → (∃ p : Person, Celebrate p) := by
  intro exw
  apply Exists.elim exw
  intro p hw
  exact Exists.intro p (P2 p hw)
