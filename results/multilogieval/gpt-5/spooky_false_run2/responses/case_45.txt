axiom not_redo_Matt : ¬Redo Matt
-- Either they watched videos or read blogs (inclusive).
axiom disj_watch_or_blogs_Matt : Watched Matt ∨ Blogs Matt
-- If someone reads interior design blogs, they get new decorating ideas.
axiom R_blogs_ideas : ∀ x : obj, Blogs x → Ideas x

theorem ideas_Matt : Ideas Matt := by
  -- From not Redo and Watched → Skills → Redo, derive ¬Watched
  have h_not_watch : ¬Watched Matt := by
    intro hW
    have hS : Skills Matt := R_watch_skills Matt hW
    have hR : Redo Matt := R_skills_redo Matt hS
    exact not_redo_Matt hR
  -- From (Watched ∨ Blogs) and ¬Watched, derive Blogs
  have hBlogs : Blogs Matt := by
    apply Or.elim disj_watch_or_blogs_Matt
    · intro hW
      exact False.elim (h_not_watch hW)
    · intro hB
      exact hB
  -- Blogs → Ideas
  exact R_blogs_ideas Matt hBlogs
